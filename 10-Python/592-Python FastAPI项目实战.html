// ========== FastAPI йЎ№зӣ®з»“жһ„ ========== # fastapi-blog/ # в”ңв”Җв”Җ app/ # в”Ӯ в”ңв”Җв”Җ __init__.py # еә”з”Ёе·ҘеҺӮ # в”Ӯ в”ңв”Җв”Җ main.py # е…ҘеҸЈ (uvicorn) # в”Ӯ в”ңв”Җв”Җ config.py # й…ҚзҪ® (зҺҜеўғеҸҳйҮҸ) # в”Ӯ в”ңв”Җв”Җ database.py # ж•°жҚ®еә“иҝһжҺҘ # в”Ӯ в”ңв”Җв”Җ models/ # в”Ӯ в”Ӯ в”ңв”Җв”Җ __init__.py # в”Ӯ в”Ӯ в”ңв”Җв”Җ user.py # SQLAlchemy жЁЎеһӢ # в”Ӯ в”Ӯ в””в”Җв”Җ post.py # в”Ӯ в”ңв”Җв”Җ schemas/ # в”Ӯ в”Ӯ в”ңв”Җв”Җ __init__.py # в”Ӯ в”Ӯ в”ңв”Җв”Җ user.py # Pydantic жЁЎејҸ # в”Ӯ в”Ӯ в””в”Җв”Җ post.py # в”Ӯ в”ңв”Җв”Җ api/ # в”Ӯ в”Ӯ в”ңв”Җв”Җ __init__.py # в”Ӯ в”Ӯ в”ңв”Җв”Җ users.py # з”ЁжҲ·и·Ҝз”ұ # в”Ӯ в”Ӯ в””в”Җв”Җ posts.py # ж–Үз« и·Ҝз”ұ # в”Ӯ в”ңв”Җв”Җ core/ # в”Ӯ в”Ӯ в”ңв”Җв”Җ __init__.py # в”Ӯ в”Ӯ в”ңв”Җв”Җ security.py # и®ӨиҜҒ/еҜҶз Ғ # в”Ӯ в”Ӯ в””в”Җв”Җ dependencies.py # е…ұдә«дҫқиө– # в”Ӯ в””в”Җв”Җ utils/ # в”Ӯ в””в”Җв”Җ __init__.py # в”ңв”Җв”Җ tests/ # в”Ӯ в”ңв”Җв”Җ __init__.py # в”Ӯ в”ңв”Җв”Җ conftest.py # в”Ӯ в”ңв”Җв”Җ test_users.py # в”Ӯ в””в”Җв”Җ test_posts.py # в”ңв”Җв”Җ .env # в”ңв”Җв”Җ .gitignore # в”ңв”Җв”Җ Dockerfile # в”ңв”Җв”Җ docker-compose.yml # в”ңв”Җв”Җ requirements.txt # в””в”Җв”Җ pyproject.toml

еә”з”Ёе·ҘеҺӮдёҺй…ҚзҪ®

// ========== config.py ========== import os from dotenv import load_dotenv load_dotenv() class Settings: PROJECT_NAME: str = "FastAPI Blog" VERSION: str = "1.0.0" API_V1_PREFIX: str = "/api/v1" DATABASE_URL: str = os.getenv( "DATABASE_URL", "postgresql://user:pass@localhost:5432/dbname" ) SECRET_KEY: str = os.getenv("SECRET_KEY", "change-this-key") ALGORITHM: str = "HS256" ACCESS_TOKEN_EXPIRE_MINUTES: int = 30 settings = Settings() # ========== database.py ========== from sqlalchemy import create_engine from sqlalchemy.orm import sessionmaker, declarative_base engine = create_engine(settings.DATABASE_URL) SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine) Base = declarative_base() def get_db(): db = SessionLocal() try: yield db finally: db.close()

еә”з”Ёе…ҘеҸЈдёҺи·Ҝз”ұ

// ========== main.py ========== from fastapi import FastAPI from app.core.config import settings from app.api import users, posts from app.database import engine, Base def create_app() -> FastAPI: app = FastAPI( title=settings.PROJECT_NAME, version=settings.VERSION, openapi_url=f"{settings.API_V1_PREFIX}/openapi.json" ) # еҲӣе»әиЎЁ: @app.on_event("startup") def on_startup(): Base.metadata.create_all(bind=engine) # жіЁеҶҢи·Ҝз”ұ: app.include_router(users.router, prefix=f"{settings.API_V1_PREFIX}/users", tags=["users"]) app.include_router(posts.router, prefix=f"{settings.API_V1_PREFIX}/posts", tags=["posts"]) # еҒҘеә·жЈҖжҹҘ: @app.get("/health") def health(): return {"status": "ok"} return app app = create_app() # ========== api/users.py ========== from fastapi import APIRouter, Depends, HTTPException router = APIRouter() @router.get("/", response_model=list[schemas.User]) def list_users(db: Session = Depends(get_db)): return crud.get_users(db) @router.post("/", response_model=schemas.User, status_code=201) def create_user(user: schemas.UserCreate, db: Session = Depends(get_db)): db_user = crud.get_user_by_email(db, email=user.email) if db_user: raise HTTPException(400, "йӮ®з®ұе·ІеӯҳеңЁ") return crud.create_user(db, user) @router.get("/{user_id}", response_model=schemas.User) def get_user(user_id: int, db: Session = Depends(get_db)): user = crud.get_user(db, user_id) if not user: raise HTTPException(404, "з”ЁжҲ·дёҚеӯҳеңЁ") return user

FastAPI зҹҘиҜҶе…ЁжҷҜ

// ========== FastAPI зҹҘиҜҶдҪ“зі» ========== // еҹәзЎҖ (581-585): // 581 - д»Ӣз»ҚдёҺе®үиЈ… (uvicorn/и·Ҝеҫ„ж“ҚдҪң/иҮӘеҠЁж–ҮжЎЈ) // 582 - и·Ҝеҫ„дёҺжҹҘиҜўеҸӮж•° (Path/Query/йӘҢиҜҒ) // 583 - иҜ·жұӮдҪ“дёҺ Pydantic (жЁЎеһӢ/Field/еөҢеҘ—) // 584 - е“Қеә”жЁЎеһӢ (response_model/status_code) // 585 - иҮӘеҠЁж–ҮжЎЈ (Tags/OpenAPI/Swaggerе®ҡеҲ¶) // иҝӣйҳ¶ (586-592): // 586 - дҫқиө–жіЁе…Ҙ (Depends/еөҢеҘ—/yield) // 587 - ж•°жҚ®еә“йӣҶжҲҗ (SQLAlchemy/Session/CRUD) // 588 - и®ӨиҜҒдёҺе®үе…Ё (OAuth2/JWT/и§’иүІжҺ§еҲ¶) // 589 - дёӯй—ҙд»¶ (иҮӘе®ҡд№ү/CORS/дәӢд»¶еӨ„зҗҶеҷЁ) // 590 - еҗҺеҸ°д»»еҠЎдёҺж–Үд»¶ (BackgroundTasks/Upload) // 591 - жөӢиҜ• (TestClient/дҫқиө–иҰҶзӣ–/ејӮжӯҘжөӢиҜ•) // 592 - йЎ№зӣ®е®һжҲҳ (з»“жһ„/йғЁзҪІ/е…ЁжҷҜ) // ========== FastAPI vs Flask йҖүжӢ© ========== // FastAPI вҶ’ JSON API / еҫ®жңҚеҠЎ / й«ҳжҖ§иғҪ // Flask вҶ’ жЁЎжқҝйЎөйқў / з®ҖеҚ•еә”з”Ё / жҲҗзҶҹз”ҹжҖҒ // ========== еӯҰд№ и·Ҝзәҝ ========== // 1. еҹәзЎҖи·Ҝз”ұе’ҢеҸӮж•° (581-583) // 2. е“Қеә”е’Ңж–ҮжЎЈ (584-585) // 3. дҫқиө–жіЁе…ҘжЁЎејҸ (586) // 4. ж•°жҚ®еә“йӣҶжҲҗ (587) // 5. и®ӨиҜҒе®үе…Ё (588) // 6. дёӯй—ҙд»¶/ж–Үд»¶/жөӢиҜ• (589-591) // 7. йЎ№зӣ®е®һжҲҳ (592)
рҹ’Ў FastAPI з»јеҗҲиҰҒзӮ№: (1) жЁЎеқ—еҢ–: APIRouter + еә”з”Ёе·ҘеҺӮ + Pydantic Schema; (2) дҫқиө–жіЁе…Ҙ: Depends з®ЎзҗҶи®ӨиҜҒ/ж•°жҚ®еә“/е…ұдә«йҖ»иҫ‘; (3) иҮӘеҠЁж–ҮжЎЈ: Swagger/ReDoc йӣ¶й…ҚзҪ®з”ҹжҲҗ; (4) ејӮжӯҘж”ҜжҢҒ: async/await й«ҳжҖ§иғҪ; (5) йҖӮеҗҲ: JSON API / еҫ®жңҚеҠЎ / еүҚеҗҺз«ҜеҲҶзҰ»гҖӮ
