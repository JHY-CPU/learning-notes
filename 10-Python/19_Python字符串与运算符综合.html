字符串与运算符结合

// ========== 字符串连接 ========== // + 运算符连接字符串 first = "Hello" last = "World" full = first + ", " + last + "!" # "Hello, World!" // += 追加 (注意: 创建新字符串!) text = "Hello" text += " World" # "Hello World" (新建字符串) // 重复: "ha" * 3 # "hahaha" "-" * 20 # "--------------------" // ========== 字符串与比较运算符 ========== "apple" < "banana" # True (字典序) "hello" == "Hello" # False (区分大小写) "abc" <= "abcd" # True (短的是前缀,认为更小) // 链式比较: "a" < "b" < "c" # True // ========== 字符串与成员运算符 ========== "he" in "hello" # True "xyz" not in "hello" # True // ========== 字符串与身份运算符 ========== a = "hello" b = "hello" a is b # True (短字符串驻留) // 但不要依赖! 始终用 == 比较值 // ========== 字符串与逻辑运算符 ========== name = "" or "default" # "default" (短路) text = "hello" and "world" # "world" // 判断非空: if name and name.strip(): # 既有值又不全是空白 print(f"用户: {name}")

实用字符串处理案例

// ========== 案例 1: 清理用户输入 ========== def clean_input(text): """清理用户输入: 去空白、压缩多余空格、转小写""" return " ".join(text.strip().lower().split()) clean_input(" Hello World! ") # "hello world!" // ========== 案例 2: 驼峰转蛇形 ========== import re def camel_to_snake(name): """将 camelCase 转 snake_case""" s1 = re.sub(r"([A-Z]+)([A-Z][a-z])", r"\1_\2", name) return re.sub(r"([a-z0-9])([A-Z])", r"\1_\2", s1).lower() camel_to_snake("getUserName") # "get_user_name" camel_to_snake("HTTPResponse") # "http_response" // ========== 案例 3: 敏感信息脱敏 ========== def mask_email(email): """隐藏邮箱中间部分: a***@example.com""" name, domain = email.split("@") return name[0] + "***@" + domain mask_email("alice@example.com") # "a***@example.com" // ========== 案例 4: URL 路径拼接 ========== def join_url_parts(*parts): """拼接 URL 路径""" return "/".join(p.strip("/") for p in parts) join_url_parts("https://example.com", "/api/", "/v1/users/") # "https://example.com/api/v1/users"

常见字符串处理技巧

// ========== 技巧 1: 反转字符串 ========== s[::-1] # 最简反转 // ========== 技巧 2: 判断回文 ========== def is_palindrome(s): s = "".join(s.lower().split()) return s == s[::-1] // ========== 技巧 3: 统计单词频率 ========== from collections import Counter text = "apple banana apple orange banana apple" words = text.split() freq = Counter(words) # Counter({"apple": 3, "banana": 2, "orange": 1}) // ========== 技巧 4: 去除标点 ========== import string text = "Hello, World! How's it going?" clean = text.translate(str.maketrans("", "", string.punctuation)) # "Hello World Hows it going" // ========== 技巧 5: 检查字符串是否数字 ========== "123".isdigit() # True "12.3".isdigit() # False (小数点不是数字) def is_number(s): try: float(s) return True except ValueError: return False is_number("12.3") # True is_number("abc") # False // ========== 技巧 6: 生成器表达式节省内存 ========== // 大文件处理: 逐行读取 with open("large_file.txt") as f: count = sum(1 for line in f if line.strip()) # 不创建列表,节省内存

运算符综合案例

// ========== 案例: 简易计算器 ========== def simple_calc(expression): """处理简单表达式: '2 + 3' → 5""" parts = expression.strip().split() if len(parts) != 3: return "格式: 数字 运算符 数字" a, op, b = parts try: a, b = float(a), float(b) except ValueError: return "无效数字" ops = { "+": a + b, "-": a - b, "*": a * b, "/": a / b if b != 0 else "除数不能为零", "//": a // b if b != 0 else "除数不能为零", "%": a % b if b != 0 else "除数不能为零", "**": a ** b, } return ops.get(op, "无效运算符") print(simple_calc("10 + 3")) # 13.0 print(simple_calc("10 / 3")) # 3.333... print(simple_calc("10 % 3")) # 1.0 // ========== 案例: 密码强度检查 ========== def check_password_strength(pwd): """检查密码强度""" has_upper = any(c.isupper() for c in pwd) has_lower = any(c.islower() for c in pwd) has_digit = any(c.isdigit() for c in pwd) has_special = any(not c.isalnum() for c in pwd) length_ok = len(pwd) >= 8 score = sum([has_upper, has_lower, has_digit, has_special, length_ok]) return score check_password_strength("Abc123!@") # 5 (强) check_password_strength("abc123") # 2 (弱)
💡 综合要点: (1) 字符串 + 和 * 是运算符重载的体现; (2) 短路求值与字符串结合可写简洁的默认值; (3) 链式比较也适用于字符串; (4) 实际开发中字符串处理占很大比重,掌握方法库很重要。
