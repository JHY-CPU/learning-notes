f-string 高级用法

// ========== f-string 基本回顾 ========== name = "Alice" age = 25 print(f"{name} is {age} years old") // ========== 表达式与函数调用 ========== print(f"{2 + 3 * 4}") # 14 (任意表达式) print(f"{name.upper()}") # "ALICE" (方法调用) print(f"{name[::-1]}") # "ecilA" (切片) // ========== 调试功能 (Python 3.8+) ========== x = 42 y = 7 print(f"{x = }") # "x = 42" print(f"{x + y = }") # "x + y = 49" print(f"{x = :#010b}") # "x = 0b00101010" (调试+格式) // ========== 多行 f-string ========== query = f""" SELECT {', '.join(fields)} FROM {table} WHERE {condition} """ // 注意: 多行 f-string 保留缩进和换行 // ========== 嵌套花括号 ========== // 输出花括号本身: 用 {{ 和 }} print(f"{{Hello}}") # "{Hello}" // ========== 条件表达式 ========== score = 85 grade = f"{'及格' if score >= 60 else '不及格'}" print(grade) # "及格" // ========== 字典与属性访问 ========== user = {"name": "Alice", "age": 25} print(f"{user['name']}") # "Alice" class Person: def __init__(self, name): self.name = name p = Person("Bob") print(f"{p.name}") # "Bob"

格式规格迷你语言

// ========== 完整语法 ========== // { [值]:[[填充]对齐][符号][#][0][宽][分组][.精度][类型] } // ========== 对齐与填充 ========== // < 左对齐 // > 右对齐 (默认) // ^ 居中对齐 // = 填充在符号后数字前 (仅数字) n = 42 print(f"|{n:<10}|") # |42 | (左对齐) print(f"|{n:>10}|") # | 42| (右对齐) print(f"|{n:^10}|") # | 42 | (居中对齐) print(f"|{n:*<10}|") # |42********| (用 * 填充) print(f"|{n:0>10}|") # |0000000042| (前补零) print(f"|{n:=+10}|") # |+ 42| (+ 在填充前) // ========== 数字符号 ========== x = 42 y = -7 print(f"{x:+} {y:+}") # "+42 -7" (强制显示符号) print(f"{x: } {y: }") # " 42 -7" (正数留空) print(f"{x:-} {y:-}") # "42 -7" (默认，负数才显示) // ========== 数字类型 ========== n = 255 print(f"{n:d}") # "255" (十进制) print(f"{n:b}") # "11111111" (二进制) print(f"{n:o}") # "377" (八进制) print(f"{n:x}") # "ff" (十六进制小写) print(f"{n:X}") # "FF" (十六进制大写) print(f"{n:#x}") # "0xff" (带 0x 前缀) // ========== 分组与精度 ========== big = 1234567890 print(f"{big:,}") # "1,234,567,890" (千位逗号) print(f"{big:_}") # "1_234_567_890" (下划线分组) pi = 3.1415926535 print(f"{pi:.2f}") # "3.14" (2位小数) print(f"{pi:.10f}") # "3.1415926535" print(f"{pi:.0f}") # "3" (整数) // ========== 百分比 ========== rate = 0.856 print(f"{rate:.0%}") # "86%" print(f"{rate:.1%}") # "85.6%" // ========== 指数 ========== print(f"{pi:e}") # "3.141593e+00" print(f"{pi:.2E}") # "3.14E+00"
💡 格式规格技巧: (1) {n:010d} 前补零到10位; (2) {n:,} 自动千位分隔; (3) {n:.2f} 保留2位小数; (4) {n:#x} 显示 0x 十六进制前缀; (5) {name!r} 等价于 repr(name)。

str.format() 详解

// ========== 位置参数 ========== "{} {} {}".format("a", "b", "c") # "a b c" "{1} {0} {2}".format("a", "b", "c") # "b a c" (索引) // ========== 关键字参数 ========== "{name} is {age}".format(name="Alice", age=25) // ========== 混合使用 ========== "{0} {lang} is {ver}".format("Python", lang="version", ver="3.12") // ========== 访问属性与项 ========== data = {"name": "Alice", "age": 25} "{p[name]} is {p[age]}".format(p=data) # "Alice is 25" class Point: def __init__(self, x, y): self.x, self.y = x, y p = Point(3, 4) "({p.x}, {p.y})".format(p=p) # "(3, 4)" // ========== 与 f-string 对比 ========== // f-string: 简洁直观，优先使用 name, age = "Alice", 25 f"{name} is {age}" # 推荐 ✅ // .format(): 动态格式，需要模板字符串时使用 template = "{name} is {age}" template.format(name="Bob", age=30) # 模板可复用 ✅ // ========== 对齐与填充 ========== "{:<20}".format("left") # 左对齐 "{:>20}".format("right") # 右对齐 "{:^20}".format("center") # 居中 "{:*^20}".format("center") # 居中用 * 填充

模板字符串与性能

// ========== string.Template ========== // 标准库模板，安全性高（适合用户输入） from string import Template t = Template("$name is $age years old") print(t.substitute(name="Alice", age=25)) // "Alice is 25 years old" // safe_substitute: 键不存在时不报错 t = Template("Hello $name $missing") print(t.safe_substitute(name="Bob")) // "Hello Bob $missing" // ========== 性能对比 ========== // 速度: f-string ≈ %格式化 > str.format() > Template import timeit n = 1000000 name, age = "Alice", 25 // f-string (最快): timeit.timeit('f"{name} is {age}"', globals=globals(), number=n) // ~0.08s // % 格式化: timeit.timeit('"%s is %d" % (name, age)', globals=globals(), number=n) // ~0.09s // str.format(): timeit.timeit('"{0} is {1}".format(name, age)', globals=globals(), number=n) // ~0.30s // Template (最慢): timeit.timeit('t.substitute(name=name, age=age)', setup='from string import Template; t=Template("$name is $age")', globals=globals(), number=n) // ~0.70s // ========== 选择指南 ========== // f-string: 95% 的情况，常规格式化 // .format(): 需要模板字符串动态填充 // %: 日志模块、旧代码兼容 // Template: 用户输入模板、安全优先场景
💡 格式化最佳实践: (1) 默认用 f-string，简洁且最快; (2) 需要复用的模板用 str.format(); (3) logging 中用 % 格式化 (延迟求值优化); (4) 用户提供的模板用 string.Template (安全); (5) Python 3.8+ 的 f"{var=}" 调试非常实用。
