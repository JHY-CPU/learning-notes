Logger 命名与继承

// ========== Logger 层级 ========== import logging # Logger 按名称层级组织,点号分隔 # root → myapp → myapp.api → myapp.api.users # 创建层级 logger: logging.getLogger() # root logger logging.getLogger("myapp") # myapp logging.getLogger("myapp.api") # myapp 的子 logger # 日志传播: 子 logger 默认将日志传给父 logger logger = logging.getLogger("myapp.api.users") logger.info("消息") # → myapp.api.users → myapp.api → myapp → root # 阻止传播: logger.propagate = False # 日志不传给父 logger # ========== 推荐模式 ========== # 每个模块使用 __name__ 作为 logger 名称 # __name__ 自动包含包路径 # myapp/api/users.py: logger = logging.getLogger(__name__) # logger name = "myapp.api.users" logger.info("用户模块启动") logger.warning("配置缺失") # main.py: logging.basicConfig(level=logging.INFO) # 所有子 logger 自动继承 root 的配置

Filter 过滤器

// ========== Filter ========== import logging # Filter: 精细化控制哪些日志被记录 # 1. 级别过滤: class LevelFilter(logging.Filter): """只允许特定级别的日志""" def __init__(self, min_level, max_level=None): super().__init__() self.min_level = min_level self.max_level = max_level or float("inf") def filter(self, record): return self.min_level <= record.levelno <= self.max_level handler = logging.StreamHandler() handler.addFilter(LevelFilter(logging.WARNING, logging.ERROR)) # 只输出 WARNING 和 ERROR 级别的日志 # 2. 内容过滤: class SensitiveFilter(logging.Filter): """过滤敏感信息""" def filter(self, record): # 替换信用卡号 if hasattr(record, "msg"): record.msg = record.msg.replace( r"\d{16}", "****-****-****-****" ) return True # 3. 上下文过滤: class RequestFilter(logging.Filter): """自动添加请求上下文""" def filter(self, record): record.ip = getattr(thread_local, "ip", "unknown") record.user = getattr(thread_local, "user", "anonymous") return True # 使用: handler.addFilter(SensitiveFilter())

结构化日志

// ========== 结构化日志 ========== import logging import json # 结构化日志: JSON 格式,便于日志分析系统 class JSONFormatter(logging.Formatter): """JSON 日志格式化器""" def format(self, record): log_data = { "timestamp": self.formatTime(record), "level": record.levelname, "logger": record.name, "message": record.getMessage(), } # 添加异常信息 if record.exc_info: log_data["exception"] = self.formatException(record.exc_info) # 添加自定义字段 for key, value in record.__dict__.items(): if key not in ("args", "asctime", "created", "exc_info", "exc_text", "filename", "funcName", "levelname", "levelno", "lineno", "module", "msecs", "message", "msg", "name", "pathname", "process", "processName", "relativeCreated", "stack_info", "thread", "threadName"): log_data[key] = value return json.dumps(log_data, ensure_ascii=False) # 配置: handler = logging.StreamHandler() handler.setFormatter(JSONFormatter()) logger = logging.getLogger("myapp") logger.addHandler(handler) logger.setLevel(logging.INFO) # 使用 extra 添加自定义字段: logger.info("用户登录", extra={ "user_id": 123, "ip": "192.168.1.1", "action": "login", }) # {"timestamp": "...", "level": "INFO", "user_id": 123, "ip": "192.168.1.1", ...} # ========== structlog 库 ========== # pip install structlog import structlog logger = structlog.get_logger() logger.info("user_login", user_id=123, ip="192.168.1.1")

日志轮转

// ========== 日志轮转 ========== import logging from logging.handlers import ( RotatingFileHandler, TimedRotatingFileHandler, WatchedFileHandler, ) # 1. 按大小轮转 handler = RotatingFileHandler( "app.log", maxBytes=10 * 1024 * 1024, # 10 MB backupCount=5, # 保留 5 个备份 encoding="utf-8", ) # 文件: app.log → app.log.1 → app.log.2 → ... → app.log.5 # 2. 按时间轮转 handler = TimedRotatingFileHandler( "app.log", when="midnight", # 每天午夜轮转 # when="H", # 每小时 # when="W0", # 每周一 interval=1, backupCount=30, # 保留 30 天 encoding="utf-8", ) # 文件: app.log → app.log.2024-01-01 → app.log.2024-01-02 → ... # 3. 生产配置示例 LOG_FORMAT = "%(asctime)s [%(levelname)s] %(name)s: %(message)s" DATE_FORMAT = "%Y-%m-%d %H:%M:%S" formatter = logging.Formatter(LOG_FORMAT, DATE_FORMAT) # 文件 handler (INFO+, 按天轮转) file_handler = TimedRotatingFileHandler( "logs/app.log", when="midnight", backupCount=90, encoding="utf-8", ) file_handler.setLevel(logging.INFO) file_handler.setFormatter(formatter) # 错误 handler (WARNING+, 单独文件) error_handler = TimedRotatingFileHandler( "logs/error.log", when="midnight", backupCount=90, ) error_handler.setLevel(logging.WARNING) error_handler.setFormatter(formatter) # 控制台 handler (开发用) console_handler = logging.StreamHandler() console_handler.setLevel(logging.DEBUG) console_handler.setFormatter(formatter) # 配置 root logger logging.basicConfig( level=logging.INFO, handlers=[file_handler, error_handler, console_handler], )

项目日志配置

// ========== 项目级日志配置 ========== import logging import logging.config import json # ========== dictConfig (推荐) ========== LOGGING_CONFIG = { "version": 1, "disable_existing_loggers": False, "formatters": { "standard": { "format": "%(asctime)s [%(levelname)s] %(name)s: %(message)s", "datefmt": "%Y-%m-%d %H:%M:%S", }, "json": { "()": "your_module.JSONFormatter", # 自定义格式化器 }, }, "filters": { "sensitive": { "()": "your_module.SensitiveFilter", }, }, "handlers": { "console": { "class": "logging.StreamHandler", "level": "DEBUG", "formatter": "standard", }, "file": { "class": "logging.handlers.TimedRotatingFileHandler", "level": "INFO", "formatter": "json", "filename": "logs/app.log", "when": "midnight", "backupCount": 30, "encoding": "utf-8", }, "error_file": { "class": "logging.handlers.TimedRotatingFileHandler", "level": "WARNING", "formatter": "json", "filename": "logs/error.log", "when": "midnight", "backupCount": 90, }, }, "loggers": { "myapp": { "handlers": ["console", "file"], "level": "INFO", "propagate": False, }, "myapp.api": { "handlers": ["console", "file", "error_file"], "level": "DEBUG", "propagate": False, }, "uvicorn": { # FastAPI 的日志 "handlers": ["console"], "level": "INFO", }, }, "root": { "handlers": ["console", "file"], "level": "WARNING", }, } logging.config.dictConfig(LOGGING_CONFIG)
