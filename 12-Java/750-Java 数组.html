数组基础

public class ArrayBasics { public static void main(String[] args) { // ========== 声明 ========== int[] nums1; // 推荐 int nums2[]; // C 风格, 不推荐 // ========== 创建 ========== nums1 = new int[5]; // 长度 5, 默认值 0 // 声明 + 创建 int[] arr = new int[10]; // ========== 初始化 ========== int[] a = new int[]{1, 2, 3, 4, 5}; // 简化 (必须在声明时) int[] b = {1, 2, 3, 4, 5}; // ========== 访问与修改 ========== int[] scores = {85, 92, 78, 90, 88}; System.out.println(scores[0]); // 85 (索引0开始) System.out.println(scores[4]); // 88 (最后一个) scores[2] = 95; // 修改: {85,92,95,90,88} // System.out.println(scores[5]); // ArrayIndexOutOfBoundsException! // ========== 长度 ========== int len = scores.length; // 5 (注意: 不是方法!) // length 是属性, 不是方法! // ========== 默认值 ========== int[] intArr = new int[3]; // [0, 0, 0] double[] dblArr = new double[3]; // [0.0, 0.0, 0.0] boolean[] boolArr = new boolean[3]; // [false, false, false] String[] strArr = new String[3]; // [null, null, null] } } // ========== 遍历 ========== public class ArrayTraverse { public static void main(String[] args) { int[] arr = {10, 20, 30, 40, 50}; // 1. for 循环 (有索引) for (int i = 0; i < arr.length; i++) { System.out.println("arr[" + i + "] = " + arr[i]); } // 2. 增强 for (foreach, 无索引) for (int num : arr) { System.out.println(num); } // 3. 倒序遍历 for (int i = arr.length - 1; i >= 0; i--) { System.out.println(arr[i]); } // 4. Java 8+ Stream // Arrays.stream(arr).forEach(System.out::println); } }

多维数组

// ========== 多维数组 ========== public class MultiArray { public static void main(String[] args) { // ========== 二维数组 ========== // 声明 + 创建 int[][] matrix = new int[3][4]; // 3行4列 // 初始化 int[][] grid = { {1, 2, 3}, {4, 5, 6}, {7, 8, 9}, }; // 访问 System.out.println(grid[0][0]); // 1 System.out.println(grid[1][2]); // 6 // 遍历 for (int i = 0; i < grid.length; i++) { for (int j = 0; j < grid[i].length; j++) { System.out.print(grid[i][j] + " "); } System.out.println(); } // ========== 不规则数组 ========== // Java 支持每行不同长度 int[][] triangle = new int[5][]; for (int i = 0; i < triangle.length; i++) { triangle[i] = new int[i + 1]; for (int j = 0; j < triangle[i].length; j++) { triangle[i][j] = j + 1; } } // ========== 三维数组 ========== int[][][] cube = new int[3][3][3]; cube[1][2][0] = 42; } } // ========== Arrays 工具类 ========== import java.util.Arrays; public class ArraysUtility { public static void main(String[] args) { int[] arr = {5, 2, 9, 1, 7, 3}; // 排序 Arrays.sort(arr); // [1, 2, 3, 5, 7, 9] // 部分排序 Arrays.sort(arr, 0, 3); // 只排序 [0,3) // 二分搜索 (必须先排序) int index = Arrays.binarySearch(arr, 5); // 3 // 填充 int[] filled = new int[5]; Arrays.fill(filled, 42); // [42, 42, 42, 42, 42] // 复制 int[] copy = Arrays.copyOf(arr, arr.length); // 完整复制 int[] copyRange = Arrays.copyOfRange(arr, 1, 4); // [2,3,5] // 比较 int[] a = {1, 2, 3}; int[] b = {1, 2, 3}; System.out.println(Arrays.equals(a, b)); // true // 转字符串 (打印) System.out.println(Arrays.toString(arr)); // "[1, 2, 3, 5, 7, 9]" // 多维打印 int[][] grid = {{1,2},{3,4}}; System.out.println(Arrays.deepToString(grid)); // "[[1,2],[3,4]]" // 转 Stream // Arrays.stream(arr).filter(n -> n > 3).toArray(); // 并行排序 (大数组) Arrays.parallelSort(arr); } }

数组与 List 互转

// ========== 数组 ↔ List ========== import java.util.Arrays; import java.util.List; import java.util.ArrayList; import java.util.stream.Collectors; public class ArrayToList { public static void main(String[] args) { // ========== 数组 → List ========== String[] arr = {"a", "b", "c"}; // 方法1: Arrays.asList() (固定大小) List list = Arrays.asList(arr); // list.add("d"); // ❌ UnsupportedOperationException! list.set(0, "z"); // ✅ 可以修改 // 方法2: new ArrayList<>() (完全可变的) List mutableList = new ArrayList<>(Arrays.asList(arr)); mutableList.add("d"); // ✅ // 方法3: Java 8+ Stream List streamList = Arrays.stream(arr) .collect(Collectors.toList()); // 方法4: List.of (Java 9+, 不可变) List immutable = List.of(arr); // immutable.set(0, "z"); // ❌ // ========== List → 数组 ========== List list2 = new ArrayList<>(); list2.add("a"); list2.add("b"); // 方法1: toArray(T[]) String[] array1 = list2.toArray(new String[0]); // 传入 0 长度, 自动分配合适大小 // 方法2: toArray() → Object[] Object[] array2 = list2.toArray(); // ========== 可变参数 (Varargs) ========== // 方法定义 printNumbers(1, 2, 3, 4, 5); printNumbers(); // 也可以不传 // int... 相当于 int[] } // 可变参数 (必须是最后一个参数) static void printNumbers(int... numbers) { for (int n : numbers) { System.out.print(n + " "); } System.out.println(); } // 多个参数 static void show(String prefix, double... values) { System.out.print(prefix + ": "); for (double v : values) { System.out.print(v + " "); } } }
💡 数组要点: 固定长度 (创建后不可变); length 是属性; 默认值 (0/false/null); 多维数组; Arrays.sort/binarySearch/copyOf/fill/equals/toString; 增强 for 遍历; 数组→List (Arrays.asList / new ArrayList<>); List→数组 (toArray); 可变参数 int...; ArrayIndexOutOfBoundsException 常见异常。
