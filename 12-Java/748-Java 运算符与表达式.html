运算符分类

public class Operators { public static void main(String[] args) { // ========== 1. 算术运算符 ========== int a = 10, b = 3; int sum = a + b; // 13 int diff = a - b; // 7 int prod = a * b; // 30 int quot = a / b; // 3 (整数除法, 截断!) int rem = a % b; // 1 (取模/余数) // 浮点除法: double div = 10.0 / 3; // 3.333... // ========== 2. 递增/递减 ========== int x = 5; int y = x++; // y=5, x=6 (先赋值后加) int z = ++x; // z=7, x=7 (先加后赋值) // ========== 3. 关系运算符 ========== // == != > < >= <= boolean eq = (a == b); // false boolean ne = (a != b); // true boolean gt = (a > b); // true boolean lt = (a < b); // false // 字符串比较用 equals()! String s1 = "hello"; String s2 = "hello"; System.out.println(s1 == s2); // true (常量池) System.out.println(s1.equals(s2)); // true ✅ String s3 = new String("hello"); System.out.println(s1 == s3); // false! ❌ System.out.println(s1.equals(s3)); // true ✅ // ========== 4. 逻辑运算符 ========== boolean t = true, f = false; boolean and = t && f; // false (短路: 左false不评估) boolean or = t || f; // true (短路: 左true不评估) boolean not = !t; // false boolean xor = t ^ f; // true (异或: 不同为true) // & | 不短路 (会评估两侧) boolean bitAnd = t & f; // false } }

位运算与三元

// ========== 位运算符 ========== public class BitwiseDemo { public static void main(String[] args) { int a = 0b1100; // 12 (二进制) int b = 0b1010; // 10 // & 与: 1100 & 1010 = 1000 (8) // | 或: 1100 | 1010 = 1110 (14) // ^ 异或: 1100 ^ 1010 = 0110 (6) // ~ 取反: ~1100 = ...0011 (-13) // << 左移: 1100 << 1 = 11000 (24) (×2) // >> 右移: 1100 >> 1 = 0110 (6) (÷2) // >>> 无符号右移: 负数高位补0 int and = a & b; // 8 int or = a | b; // 14 int xor = a ^ b; // 6 int not = ~a; // -13 int left = a << 2; // 48 (12 × 4) int right = a >> 1; // 6 (12 ÷ 2) // 位运算技巧: // 判断奇偶: (n & 1) == 1 → 奇数 // 2 的幂: (n & (n-1)) == 0 // 交换: a ^= b; b ^= a; a ^= b; // 取第 k 位: (n >> k) & 1 boolean isOdd = (5 & 1) == 1; // true // ========== 三元运算符 ========== int score = 85; String grade = score >= 60 ? "Pass" : "Fail"; // "Pass" // 嵌套三元 (不推荐): String level = score >= 90 ? "A" : score >= 80 ? "B" : score >= 70 ? "C" : "D"; // ========== 赋值运算符 ========== int n = 10; n += 5; // n = 15 (n = n + 5) n -= 3; // n = 12 n *= 2; // n = 24 n /= 4; // n = 6 n %= 2; // n = 0 n <<= 1; // n = 0 (左移赋值) } } // ========== 运算符优先级 (高→低) ========== // 1. () [] . 括号/数组/成员 // 2. ++ -- ~ ! 一元 // 3. * / % 乘除取模 // 4. + - 加减 // 5. << >> >>> 移位 // 6. < > <= >= instanceof 关系 // 7. == != 相等 // 8. & 位与 // 9. ^ 位异或 // 10. | 位或 // 11. && 逻辑与 // 12. || 逻辑或 // 13. ? : 三元 // 14. = += -= 赋值 // 规则: 不确定就加括号!

Math 工具与字符串拼接

// ========== Math 类 ========== public class MathDemo { public static void main(String[] args) { // 数学常量 double pi = Math.PI; // 3.1415926... double e = Math.E; // 2.71828... // 常用方法 double abs = Math.abs(-5); // 5 double max = Math.max(10, 20); // 20 double min = Math.min(10, 20); // 10 double ceil = Math.ceil(3.2); // 4.0 向上取整 double floor = Math.floor(3.8);// 3.0 向下取整 double round = Math.round(3.5);// 4 四舍五入 double pow = Math.pow(2, 10); // 1024.0 幂 double sqrt = Math.sqrt(144); // 12.0 平方根 double cbrt = Math.cbrt(27); // 3.0 立方根 double sin = Math.sin(Math.toRadians(30)); // 0.5 // 随机数 double random = Math.random(); // [0.0, 1.0) int dice = (int)(Math.random() * 6) + 1; // 1~6 // ========== 字符串拼接 ========== String firstName = "John"; String lastName = "Doe"; // + 运算符 String full = firstName + " " + lastName; // "John Doe" // 不同类型拼接 String msg = "Age: " + 25; // "Age: 25" String desc = "Value: " + 3.14; // "Value: 3.14" // String.format() (推荐) String formatted = String.format( "Name: %s, Age: %d, Score: %.1f", "Alice", 25, 92.5 ); // "Name: Alice, Age: 25, Score: 92.5" // StringBuilder (大量拼接) StringBuilder sb = new StringBuilder(); sb.append("Hello"); sb.append(", "); sb.append("World"); sb.append("!"); String result = sb.toString(); // "Hello, World!" } } // ========== instanceof 运算符 ========== // 检查对象是否是某个类型 // if (obj instanceof String) { // String str = (String) obj; // } // // // Java 16+: 模式匹配 (自动转换) // if (obj instanceof String str) { // System.out.println(str.length()); // }
