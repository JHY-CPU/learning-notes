🧪 Spring Boot 测试

@SpringBootTest 集成测试、@WebMvcTest 切片测试、@MockBean/@SpyBean、MockMvc 执行 HTTP 请求、TestRestTemplate、测试数据库。

MockMvc 详解

// ========== MockMvc ========== // 模拟 HTTP 请求, 不启动服务器 // 快速测试 Controller // ========== 请求构建 ========== mockMvc.perform(get("/api/users")) // GET mockMvc.perform(post("/api/users")) // POST mockMvc.perform(put("/api/users/{id}", 1L)) // PUT mockMvc.perform(delete("/api/users/{id}", 1L)) // DELETE // ========== 请求配置 ========== mockMvc.perform(get("/api/users") .param("page", "0") // 查询参数 .param("size", "20") .header("Authorization", "Bearer token") // 请求头 .contentType(MediaType.APPLICATION_JSON) // 内容类型 .content("{\"username\":\"Alice\"}") // 请求体 .accept(MediaType.APPLICATION_JSON) // 接受类型 .sessionAttr("userId", 1L) // Session 属性 .cookie(new Cookie("token", "abc")) // Cookie ); // ========== 响应断言 ========== mockMvc.perform(get("/api/users/1")) // 状态码 .andExpect(status().isOk()) // 200 .andExpect(status().isCreated()) // 201 .andExpect(status().isNoContent()) // 204 .andExpect(status().isBadRequest()) // 400 .andExpect(status().isNotFound()) // 404 .andExpect(status().isForbidden()) // 403 // 响应头 .andExpect(header().string("Content-Type", "application/json")) .andExpect(header().exists("Location")) // JSON 路径 (JSON Path) .andExpect(jsonPath("$.code").value(200)) .andExpect(jsonPath("$.data.id").value(1)) .andExpect(jsonPath("$.data.username").value("Alice")) .andExpect(jsonPath("$.data.email").value("alice@test.com")) // JSON Path 高级匹配 .andExpect(jsonPath("$.data").isNotEmpty()) .andExpect(jsonPath("$.data.tags").isArray()) .andExpect(jsonPath("$.data.tags.length()").value(3)) .andExpect(jsonPath("$.data.roles[*]", containsInAnyOrder("admin", "user"))); // ========== 异步结果 ========== mockMvc.perform(get("/api/users/async")) .andExpect(request().asyncStarted()) .andDo(MockMvcRequestBuilders.asyncDispatch()) .andExpect(status().isOk());

// ========== @MockBean ========== // 向 Spring 容器中注入 Mock 对象 // 替换已有的 Bean @WebMvcTest(UserController.class) class UserControllerTest { @Autowired private MockMvc mockMvc; @MockBean // 替换 UserService Bean private UserService userService; // 验证调用 verify(userService).findById(1L); } @Test void testUserNotFound() throws Exception { when(userService.findById(999L)).thenReturn(Optional.empty()); mockMvc.perform(get("/api/users/999")) .andExpect(status().isOk()) .andExpect(jsonPath("$.code").value(404)); } @Test void testCreateUser() throws Exception { when(userService.create(any())).thenReturn( new User(1L, "Alice", "alice@test.com") ); String body = """ {"username":"Alice","email":"alice@test.com","password":"123456"} """; mockMvc.perform(post("/api/users") .contentType(MediaType.APPLICATION_JSON) .content(body)) .andExpect(status().isOk()) .andExpect(jsonPath("$.data.id").value(1)); } } // ========== @SpyBean ========== // 部分模拟: 真实调用 + 可覆写特定方法 @SpringBootTest class OrderServiceTest { @SpyBean // Spry: 真实 Bean + 可覆写 private OrderService orderService; @MockBean private PaymentClient paymentClient; // 外部服务 Mock @Test void testCreateOrder() { // 覆写真实的 sendNotification doNothing().when(orderService).sendNotification(any()); // 其他方法真实调用, 只有 sendNotification 被 Mock orderService.createOrder(new Order()); verify(orderService).sendNotification(any()); } } // ========== Mockito 常用 ========== when(obj.method()).thenReturn(value); // 返回指定值 when(obj.method()).thenThrow(ex); // 抛异常 when(obj.method()).thenAnswer(invocation); // 动态返回 doNothing().when(obj).method(); // void 方法无操作 doThrow(ex).when(obj).method(); // void 方法抛异常 verify(obj).method(); // 验证调用 verify(obj, times(3)).method(); // 调用次数 verify(obj, never()).method(); // 从未调用 verify(obj, timeout(100)).method(); // 超时内调用 any() / anyString() / anyLong() / anyList() // 任意匹配器 eq(value) // 特定值匹配

@DataJpaTest 与数据库测试

// ========== @DataJpaTest ========== // JPA 切片: 只加载 Repository + 实体 // 默认使用嵌入式数据库 (H2) // 每个测试自动回滚 @DataJpaTest class UserRepositoryTest { @Autowired private TestEntityManager entityManager; // 测试专用 EntityManager @Autowired private UserRepository userRepository; @Test void shouldSaveAndFindUser() { // 使用 EntityManager 保存 User user = new User("Alice", "alice@test.com"); Long id = entityManager.persistAndGetId(user, Long.class); // 验证查询 Optional<User> found = userRepository.findById(id); assertThat(found).isPresent(); assertThat(found.get().getUsername()).isEqualTo("Alice"); } @Test void shouldFindByEmail() { // 准备数据 entityManager.persist(new User("Alice", "alice@test.com")); entityManager.persist(new User("Bob", "bob@test.com")); // 测试查询方法 Optional<User> user = userRepository.findByEmail("alice@test.com"); assertThat(user).isPresent(); assertThat(user.get().getUsername()).isEqualTo("Alice"); } @Test void shouldReturnEmptyWhenNotFound() { Optional<User> user = userRepository.findById(999L); assertThat(user).isEmpty(); } } // ========== 使用真实数据库 ========== // 自动回滚, 不影响数据库 @DataJpaTest @AutoConfigureTestDatabase(replace = AutoConfigureTestDatabase.Replace.NONE) // 不替换数据库, 使用配置的真实数据库 class UserRepositoryRealDbTest { ... } // ========== Testcontainers ========== @SpringBootTest @Testcontainers class UserRepositoryContainerTest { @Container static PostgreSQLContainer<?> postgres = new PostgreSQLContainer<>("postgres:16") .withDatabaseName("testdb"); @DynamicPropertySource static void configureProperties(DynamicPropertyRegistry registry) { registry.add("spring.datasource.url", postgres::getJdbcUrl); registry.add("spring.datasource.username", postgres::getUsername); registry.add("spring.datasource.password", postgres::getPassword); } @Autowired private UserRepository userRepository; @Test void shouldWorkWithPostgres() { userRepository.save(new User("Alice", "alice@test.com")); assertThat(userRepository.count()).isEqualTo(1); } }

最佳实践

// ========== 测试最佳实践 ========== // ========== 1. 测试分层 ========== // Controller: @WebMvcTest + MockMvc + @MockBean // Service: @SpringBootTest + @MockBean (外部依赖) // Repository: @DataJpaTest + TestEntityManager // 集成: @SpringBootTest + TestRestTemplate // ========== 2. 使用 AssertJ ========== // assertThat 比 JUnit assertEquals 更可读 assertThat(result).isNotNull() .extracting(User::getUsername).isEqualTo("Alice"); // ========== 3. 测试命名 ========== // should[预期行为]_when[条件] @Test void shouldReturnUser_whenIdExists() { ... } @Test void shouldThrowException_whenUserNotFound() { ... } // ========== 4. 独立测试 ========== // 不依赖其他测试的执行顺序 // @BeforeEach 初始化数据 // ========== 5. JSON 响应对比 ========== // 使用文件中的预期 JSON @Test void shouldMatchExpectedJson() throws Exception { mockMvc.perform(get("/api/users/1")) .andExpect(content().json( Files.readString(Path.of("src/test/resources/expected/user.json")) )); } // ========== 6. 测试 Profile ========== // application-test.yml 中配置测试专用参数 // @ActiveProfiles("test") // ========== 7. @TestPropertySource ========== // 覆盖特定属性 @SpringBootTest @TestPropertySource(properties = { "myapp.feature.enabled=false" }) class FeatureTest { ... } // ========== 8. 覆盖报告 ========== // JaCoCo 插件: mvn verify 时生成报告 // build/reports/jacoco/index.html
💡 测试要点: @SpringBootTest 全量集成; @WebMvcTest Controller 切片; @DataJpaTest Repository 切片; @MockBean 替换 Bean; MockMvc 模拟 HTTP; TestRestTemplate 真实 HTTP; TestEntityManager 数据准备; AssertJ 链式断言; @Testcontainers 真实数据库; 测试命名 shouldXxx_whenYyy。
