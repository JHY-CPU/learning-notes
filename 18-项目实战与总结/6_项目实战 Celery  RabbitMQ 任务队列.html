任务定义

# ========== 邮件任务 ========== # app/tasks/email.py from app.celery_app import celery_app import smtplib from time import sleep @celery_app.task( bind=True, max_retries=3, default_retry_delay=60, queue="email", route_name="email.send_welcome", ) def send_welcome_email(self, user_email: str, username: str): """发送欢迎邮件""" try: # 模拟发送 sleep(2) logger.info(f"欢迎邮件已发送至 {user_email}") return {"status": "sent", "email": user_email} except smtplib.SMTPException as exc: raise self.retry(exc=exc) # ========== 视频处理任务 ========== # app/tasks/video.py @celery_app.task(bind=True, queue="video", track_started=True) def process_video(self, video_id: str, input_path: str): """视频转码 + 缩略图生成""" self.update_state( state="PROGRESS", meta={"step": "validating", "percent": 10} ) # 步骤 1: 验证 validate(video_id) self.update_state(state="PROGRESS", meta={"step": "transcoding", "percent": 30}) # 步骤 2: 转码 transcode(input_path, "h264", "output.mp4") self.update_state(state="PROGRESS", meta={"step": "thumbnail", "percent": 70}) # 步骤 3: 生成缩略图 generate_thumbnail("output.mp4") self.update_state(state="PROGRESS", meta={"step": "done", "percent": 100}) return {"video_id": video_id, "status": "completed", "output": "output.mp4"} # ========== 任务链 (Workflow) ========== @celery_app.task(queue="default") def validate(data): if not data.get("email"): raise ValueError("email required") return data @celery_app.task(queue="default") def process(data): logger.info(f"Processing: {data}") return {"result": "success", "data": data} @celery_app.task(queue="default") def notify(result): logger.info(f"Notification: {result}") # 任务链: validate → process → notify # workflow = chain( # validate.s(data), # process.s(), # notify.s() # ) # workflow() # 或者使用 Canvas: # from celery import group, chord # # # 并行任务 # parallel = group([task_a.s(), task_b.s(), task_c.s()]) # # # 并行完成后汇总 # workflow = chord(parallel)(summarize.s())

Flask API 与 Flower 监控

# ========== Flask API ========== # app/web/app.py from flask import Flask, request, jsonify from app.celery_app import celery_app from app.tasks.email import send_welcome_email from app.tasks.video import process_video app = Flask(__name__) # 提交异步任务 @app.post("/api/tasks/email") def send_email(): data = request.get_json() task = send_welcome_email.delay( data["email"], data["username"] ) return jsonify({"task_id": task.id, "status": "queued"}), 202 # 提交视频处理 @app.post("/api/tasks/video") def process_video_task(): data = request.get_json() task = process_video.delay( data["video_id"], data["input_path"] ) return jsonify({"task_id": task.id, "status": "queued"}), 202 # 查询任务状态 @app.get("/api/tasks/") def get_task_status(task_id): task = celery_app.AsyncResult(task_id) result = { "task_id": task_id, "status": task.status, "state": task.state, } if task.state == "SUCCESS": result["result"] = task.result elif task.state == "FAILURE": result["error"] = str(task.info) elif task.state == "PROGRESS": result["meta"] = task.info return jsonify(result) # ========== 启动 Worker ========== # 终端命令: # # 启动 Celery Worker # celery -A app.celery_app worker \ # --loglevel=info \ # --concurrency=4 \ # --queues=default,email,video,high # # # 启动 Beat (定时任务) # celery -A app.celery_app beat --loglevel=info # # # 启动 Flower (监控面板) # celery -A app.celery_app flower \ # --port=5555 \ # --basic-auth=admin:secret # ========== Docker Compose ========== # docker-compose.yml: # version: '3.8' # services: # rabbitmq: # image: rabbitmq:3-management-alpine # ports: # - "5672:5672" # AMQP # - "15672:15672" # 管理界面 # # redis: # image: redis:7-alpine # # worker: # build: . # command: celery -A app.celery_app worker --loglevel=info # depends_on: [rabbitmq, redis] # volumes: # - ./app:/app # # beat: # build: . # command: celery -A app.celery_app beat --loglevel=info # depends_on: [rabbitmq, redis] # # flower: # build: . # command: celery -A app.celery_app flower --port=5555 # ports: ["5555:5555"] # depends_on: [rabbitmq, redis] # # web: # build: . # command: gunicorn -w 4 app.web.app:app -b 0.0.0.0:5000 # ports: ["5000:5000"] # depends_on: [rabbitmq, redis]
💡 Celery + RabbitMQ 任务队列要点: Celery Worker 异步消费任务; RabbitMQ 作为 Broker 消息中间件; Redis 作为结果后端; 任务路由到不同队列 (email/video/high); 任务链 Chain 和 Canvas (group/chord) 编排; Flower 监控; Beat 定时任务 cron; 重试机制 + 超时; 任务状态追踪 AsyncResult。
