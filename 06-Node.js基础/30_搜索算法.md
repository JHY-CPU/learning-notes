```
// ========== 顺序搜索 O(n) ==========
function linearSearch(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === target) return i;
    }
    return -1;
}

// ========== 二分搜索 O(log n) (已排序) ==========
function binarySearch(arr, target) {
    let left = 0, right = arr.length - 1;
    while (left <= right) {
        const mid = left + Math.floor((right - left) / 2);
        if (arr[mid] === target) return mid;
        if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

// ========== 二分搜索 (递归) ==========
function binarySearchRec(arr, target, left = 0, right = arr.length - 1) {
    if (left > right) return -1;
    const mid = left + Math.floor((right - left) / 2);
    if (arr[mid] === target) return mid;
    if (arr[mid] < target) return binarySearchRec(arr, target, mid + 1, right);
    return binarySearchRec(arr, target, left, mid - 1);
}

// ========== 搜索插入位置 ==========
function searchInsert(arr, target) {
    let left = 0, right = arr.length;
    while (left < right) {
        const mid = left + Math.floor((right - left) / 2);
        if (arr[mid] < target) left = mid + 1;
        else right = mid;
    }
    return left;
}
```

## 高级搜索算法

```javascript
// ========== 第一个/最后一个出现位置 ==========
function firstOccurrence(arr, target) {
    let left = 0, right = arr.length - 1, result = -1;
    while (left <= right) {
        const mid = left + Math.floor((right - left) / 2);
        if (arr[mid] === target) {
            result = mid;
            right = mid - 1; // 继续向左找
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return result;
}

function lastOccurrence(arr, target) {
    let left = 0, right = arr.length - 1, result = -1;
    while (left <= right) {
        const mid = left + Math.floor((right - left) / 2);
        if (arr[mid] === target) {
            result = mid;
            left = mid + 1; // 继续向右找
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return result;
}

// ========== 旋转数组搜索 ==========
// [4,5,6,7,0,1,2] 中搜索 target
function searchRotated(nums, target) {
    let left = 0, right = nums.length - 1;
    while (left <= right) {
        const mid = (left + right) >> 1;
        if (nums[mid] === target) return mid;
        if (nums[left] <= nums[mid]) {
            // 左半部分有序
            if (target >= nums[left] && target < nums[mid]) right = mid - 1;
            else left = mid + 1;
        } else {
            // 右半部分有序
            if (target > nums[mid] && target <= nums[right]) left = mid + 1;
            else right = mid - 1;
        }
    }
    return -1;
}

// ========== 峰值元素搜索 ==========
function findPeakElement(nums) {
    let left = 0, right = nums.length - 1;
    while (left < right) {
        const mid = (left + right) >> 1;
        if (nums[mid] < nums[mid + 1]) left = mid + 1;
        else right = mid;
    }
    return left;
}
```

## 实际应用场景

```javascript
// ========== API 限流中的二分搜索 ==========
// 在有序时间戳数组中找到指定时间窗口内的请求数
function countInWindow(timestamps, startTime, endTime) {
    const start = searchInsert(timestamps, startTime);
    const end = lastOccurrence(timestamps, endTime);
    return end === -1 ? 0 : end - start + 1;
}

// ========== 搜索建议中的模糊搜索 ==========
function fuzzySearch(items, query) {
    const lowerQuery = query.toLowerCase();
    return items.filter(item =>
        item.toLowerCase().includes(lowerQuery)
    ).sort((a, b) => {
        // 精确匹配优先
        const aExact = a.toLowerCase().startsWith(lowerQuery) ? 0 : 1;
        const bExact = b.toLowerCase().startsWith(lowerQuery) ? 0 : 1;
        return aExact - bExact;
    });
}
```
