日期计算方法

// ========== 日期差计算 ========== function daysDiff(d1, d2) { const msDiff = Math.abs(d2 - d1); return Math.floor(msDiff / (1000 * 60 * 60 * 24)); } // ========== 工作日判断 ========== function isWeekday(date) { const day = date.getDay(); return day !== 0 && day !== 6; } // ========== 添加天数 ========== function addDays(date, days) { const result = new Date(date); result.setDate(result.getDate() + days); return result; } // ========== 相对时间 ========== function timeAgo(date) { const diff = Date.now() - date; const units = [ { name: '年', ms: 365 * 24 * 60 * 60 * 1000 }, { name: '月', ms: 30 * 24 * 60 * 60 * 1000 }, { name: '天', ms: 24 * 60 * 60 * 1000 }, { name: '小时', ms: 60 * 60 * 1000 }, { name: '分钟', ms: 60 * 1000 }, ]; for (const unit of units) { if (diff >= unit.ms) { const count = Math.floor(diff / unit.ms); return `${count} ${unit.name}前`; } } return '刚刚'; }
