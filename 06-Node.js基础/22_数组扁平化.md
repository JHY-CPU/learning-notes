## 数组扁平化方法


```
// ========== flat (ES2019) ==========
[1, [2, [3, [4]]]].flat(2);      // [1, 2, 3, [4]]
[1, [2, [3, [4]]]].flat(Infinity); // [1, 2, 3, 4]

// ========== flatMap ==========
['a b', 'c d'].flatMap(s => s.split(' '));
// ['a', 'b', 'c', 'd']

// ========== 递归 reduce ==========
function flatten(arr, depth = Infinity) {
    return depth > 0
        ? arr.reduce((acc, v) => acc.concat(
            Array.isArray(v) ? flatten(v, depth - 1) : v
        ), [])
        : arr.slice();
}

// ========== 栈实现 (非递归) ==========
function flattenStack(arr) {
    const stack = [...arr];
    const result = [];
    while (stack.length) {
        const v = stack.shift();
        Array.isArray(v) ? stack.unshift(...v) : result.push(v);
    }
    return result;
}

// ========== toString (仅数字) ==========
[1, [2, [3, [4]]]].toString().split(',').map(Number);
// [1, 2, 3, 4] (只适用于数字)
```

## 深度扁平化与类型保留

```javascript
// ========== 带类型的扁平化 ==========
function flattenWithType(arr, path = '') {
    const result = [];
    for (let i = 0; i < arr.length; i++) {
        const currentPath = path ? `${path}[${i}]` : `[${i}]`;
        if (Array.isArray(arr[i])) {
            result.push(...flattenWithType(arr[i], currentPath));
        } else {
            result.push({ value: arr[i], path: currentPath, type: typeof arr[i] });
        }
    }
    return result;
}

// flattenWithType([1, [2, [3]]])
// → [
//   { value: 1, path: '[0]', type: 'number' },
//   { value: 2, path: '[1][0]', type: 'number' },
//   { value: 3, path: '[1][1][0]', type: 'number' }
// ]

// ========== 过滤空数组 ==========
function flattenClean(arr) {
    return arr.flat(Infinity).filter(v => v !== undefined && v !== null);
}
flattenClean([1, null, [2, undefined, [3]]]); // [1, 2, 3]

// ========== 按条件扁平化 (只展开特定类型) ==========
function flattenConditionally(arr, depth = Infinity, shouldFlatten = Array.isArray) {
    if (depth <= 0) return arr.slice();
    return arr.reduce((acc, item) => {
        if (shouldFlatten(item)) {
            acc.push(...flattenConditionally(item, depth - 1, shouldFlatten));
        } else {
            acc.push(item);
        }
        return acc;
    }, []);
}

// 只展开数字数组，保留字符串数组
flattenConditionally([1, [2, ['hello', [3]]]], Infinity, (v) => Array.isArray(v) && typeof v[0] === 'number');
// [1, 2, ['hello', [3]]]
```

## Generator 实现

```javascript
// ========== Generator 扁平化 (惰性求值) ==========
function* flattenGenerator(arr) {
    for (const item of arr) {
        if (Array.isArray(item)) {
            yield* flattenGenerator(item);
        } else {
            yield item;
        }
    }
}

// 惰性遍历 — 不一次性生成所有结果
const nested = [1, [2, [3, [4, [5]]]]];
for (const value of flattenGenerator(nested)) {
    console.log(value);
    if (value === 3) break; // 可以提前终止！
}

// 转为数组
const flat = [...flattenGenerator(nested)]; // [1, 2, 3, 4, 5]

// ========== 分块扁平化 (处理超大数组) ==========
function flattenInChunks(arr, chunkSize = 1000) {
    const flat = arr.flat(Infinity);
    const chunks = [];
    for (let i = 0; i < flat.length; i += chunkSize) {
        chunks.push(flat.slice(i, i + chunkSize));
    }
    return chunks;
}
// 适合百万级数据处理
```

## 实战：嵌套数据处理

```javascript
// ========== 组织架构扁平化 ==========
const org = {
    name: 'CEO',
    children: [
        { name: 'CTO', children: [
            { name: '前端主管', children: [{ name: '前端开发' }] },
            { name: '后端主管' }
        ]},
        { name: 'CFO' }
    ]
};

function flattenTree(node, parent = null) {
    const result = [{ name: node.name, parent: parent?.name || null }];
    if (node.children) {
        for (const child of node.children) {
            result.push(...flattenTree(child, node));
        }
    }
    return result;
}

// flattenTree(org)
// → [{name:'CEO',parent:null},{name:'CTO',parent:'CEO'},...]
```
