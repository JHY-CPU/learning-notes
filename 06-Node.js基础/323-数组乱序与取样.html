数组乱序与取样

Fisher-Yates 洗牌、随机取样、加权随机、抽奖算法。

乱序与取样方法

// ========== Fisher-Yates 洗牌 (最优) ========== function shuffle(arr) { const a = [...arr]; for (let i = a.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [a[i], a[j]] = [a[j], a[i]]; } return a; } // ========== sort 乱序 (有偏) ========== [1, 2, 3, 4, 5].sort(() => Math.random() - 0.5); // ⚠️ 不均匀, 不推荐 // ========== 随机取样 ========== function sample(arr, n = 1) { const shuffled = shuffle(arr); return shuffled.slice(0, n); } // ========== 加权随机 ========== function weightedRandom(items, weights) { const total = weights.reduce((a, b) => a + b, 0); let r = Math.random() * total; for (let i = 0; i < items.length; i++) { r -= weights[i]; if (r <= 0) return items[i]; } return items[items.length - 1]; }
