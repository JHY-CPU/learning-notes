# 分片上传


## 分片上传


文件分片(slice)、并发上传、进度合并、秒传(hash预检)。


## 分片上传 API


```
// ========== 文件分片 ==========
const CHUNK_SIZE = 5 * 1024 * 1024;  // 5MB
const chunks = [];
let start = 0;
while (start < file.size) {
    const end = Math.min(start + CHUNK_SIZE, file.size);
    chunks.push(file.slice(start, end));
    start = end;
}

// ========== 并发上传 ==========
const tasks = chunks.map((chunk, index) => {
    const formData = new FormData();
    formData.append('chunk', chunk);
    formData.append('index', index);
    formData.append('total', chunks.length);
    formData.append('hash', fileHash);
    return fetch('/upload/chunk', { method: 'POST', body: formData });
});
await Promise.all(tasks);

// ========== 合并请求 ==========
fetch('/upload/merge', {
    method: 'POST',
    body: JSON.stringify({ hash: fileHash, name: file.name, chunks: chunks.length })
});

// ========== 秒传 ==========
// 先计算文件 hash，查询服务器是否已存在
const hash = await calculateHash(file);
const res = await fetch('/upload/check', { body: JSON.stringify({ hash }) });
if (res.exist) { /* 秒传 */ }
```

## 完整分片上传实现

```javascript
// ========== 文件 Hash 计算 (Web Worker) ==========
// hash-worker.js
self.onmessage = async (e) => {
    const { chunks, chunkSize } = e.data;
    // 使用 SparkMD5 库计算增量 hash
    importScripts('https://cdn.jsdelivr.net/npm/spark-md5@3.0.2/spark-md5.min.js');
    const spark = new SparkMD5.ArrayBuffer();
    let count = 0;

    for (const chunk of chunks) {
        const reader = new FileReader();
        reader.onload = (event) => {
            spark.append(event.target.result);
            count++;
            self.postMessage({ type: 'progress', value: count / chunks.length });
            if (count === chunks.length) {
                self.postMessage({ type: 'hash', value: spark.end() });
            }
        };
        reader.readAsArrayBuffer(chunk);
    }
};

// ========== 分片上传管理器 ==========
class ChunkUploader {
    constructor(options = {}) {
        this.options = {
            chunkSize: 5 * 1024 * 1024,  // 5MB
            concurrency: 3,
            retries: 3,
            url: '/upload/chunk',
            ...options,
        };
    }

    async upload(file) {
        // 1. 计算文件 hash
        const hash = await this.calculateHash(file);

        // 2. 检查是否已存在 (秒传)
        const checkRes = await fetch('/upload/check', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ hash, size: file.size }),
        });
        const { exist, uploadedChunks } = await checkRes.json();
        if (exist) return { status: 'instant', url: exist };

        // 3. 分片
        const chunks = this.createChunks(file);

        // 4. 上传分片 (跳过已上传的)
        const pending = chunks
            .map((chunk, i) => ({ chunk, index: i }))
            .filter(({ index }) => !uploadedChunks?.includes(index));

        await this.uploadWithConcurrency(pending, hash);

        // 5. 合并
        const mergeRes = await fetch('/upload/merge', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ hash, name: file.name, totalChunks: chunks.length }),
        });
        return mergeRes.json();
    }

    createChunks(file) {
        const chunks = [];
        let start = 0;
        while (start < file.size) {
            chunks.push(file.slice(start, start + this.options.chunkSize));
            start += this.options.chunkSize;
        }
        return chunks;
    }

    async calculateHash(file) {
        return new Promise((resolve) => {
            const worker = new Worker('hash-worker.js');
            const chunks = this.createChunks(file);
            worker.postMessage({ chunks });
            worker.onmessage = (e) => {
                if (e.data.type === 'hash') {
                    resolve(e.data.value);
                    worker.terminate();
                } else if (e.data.type === 'progress') {
                    this.options.onHashProgress?.(e.data.value);
                }
            };
        });
    }

    async uploadWithConcurrency(tasks, hash) {
        const executing = new Set();

        for (const { chunk, index } of tasks) {
            const promise = this.uploadChunk(chunk, index, hash).then(() => {
                executing.delete(promise);
            });
            executing.add(promise);

            if (executing.size >= this.options.concurrency) {
                await Promise.race(executing);
            }
        }
        await Promise.all(executing);
    }

    async uploadChunk(chunk, index, hash) {
        const formData = new FormData();
        formData.append('chunk', chunk);
        formData.append('index', index);
        formData.append('hash', hash);

        for (let retry = 0; retry < this.options.retries; retry++) {
            try {
                const res = await fetch(this.options.url, { method: 'POST', body: formData });
                if (res.ok) return;
            } catch (err) {
                if (retry === this.options.retries - 1) throw err;
            }
        }
    }
}

const uploader = new ChunkUploader({
    concurrency: 4,
    onHashProgress: (pct) => console.log(`计算Hash: ${(pct * 100).toFixed(0)}%`),
});
```
