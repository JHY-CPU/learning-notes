// ========== 获取媒体流 ========== const stream = await navigator.mediaDevices.getUserMedia({ video: { width: 1920, height: 1080, frameRate: 30 }, audio: { echoCancellation: true, noiseSuppression: true } }); // ========== 屏幕共享 ========== const displayStream = await navigator.mediaDevices.getDisplayMedia({ video: { cursor: 'always' }, audio: true }); // ========== 拍照 ========== canvas.width = video.videoWidth; canvas.height = video.videoHeight; ctx.drawImage(video, 0, 0); const dataUrl = canvas.toDataURL('image/png'); // ========== 录制 ========== const recorder = new MediaRecorder(stream); recorder.ondataavailable = (e) => chunks.push(e.data); recorder.start(); recorder.stop();
