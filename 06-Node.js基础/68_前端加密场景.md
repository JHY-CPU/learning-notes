```
// 3. 数据完整性 (防篡改)
// 所有 Web Crypto 的前提

// ========== 前端加密误区 ==========
// ❌ "前端哈希密码更安全" — 仍然信任服务器
// ❌ "前端加密 = 数据安全" — 密钥在前端可知
// ❌ "自己实现加密算法" — 使用标准 API
```

## E2EE 聊天加密

```javascript
// ========== 端到端加密消息 ==========
class E2EEChat {
    constructor() {
        this.keyPair = null;
        this.peerPublicKey = null;
    }

    async init() {
        // 生成 RSA 密钥对
        this.keyPair = await crypto.subtle.generateKey(
            { name: 'RSA-OAEP', modulusLength: 2048, publicExponent: new Uint8Array([1, 0, 1]), hash: 'SHA-256' },
            true, ['encrypt', 'decrypt']
        );
    }

    async exportPublicKey() {
        const exported = await crypto.subtle.exportKey('jwk', this.keyPair.publicKey);
        return JSON.stringify(exported);
    }

    async importPeerPublicKey(jwk) {
        this.peerPublicKey = await crypto.subtle.importKey(
            'jwk', JSON.parse(jwk),
            { name: 'RSA-OAEP', hash: 'SHA-256' },
            true, ['encrypt']
        );
    }

    async encryptMessage(plaintext) {
        // 生成临时 AES 密钥
        const aesKey = await crypto.subtle.generateKey(
            { name: 'AES-GCM', length: 256 },
            true, ['encrypt', 'decrypt']
        );

        // 用 AES 加密消息
        const iv = crypto.getRandomValues(new Uint8Array(12));
        const encoded = new TextEncoder().encode(plaintext);
        const ciphertext = await crypto.subtle.encrypt(
            { name: 'AES-GCM', iv }, aesKey, encoded
        );

        // 用 RSA 加密 AES 密钥
        const exportedAesKey = await crypto.subtle.exportKey('raw', aesKey);
        const encryptedKey = await crypto.subtle.encrypt(
            { name: 'RSA-OAEP' }, this.peerPublicKey, exportedAesKey
        );

        return {
            encryptedKey: btoa(String.fromCharCode(...new Uint8Array(encryptedKey))),
            iv: btoa(String.fromCharCode(...iv)),
            ciphertext: btoa(String.fromCharCode(...new Uint8Array(ciphertext))),
        };
    }

    async decryptMessage(encrypted) {
        // 用 RSA 解密 AES 密钥
        const encryptedKey = Uint8Array.from(atob(encrypted.encryptedKey), c => c.charCodeAt(0));
        const aesKeyRaw = await crypto.subtle.decrypt(
            { name: 'RSA-OAEP' }, this.keyPair.privateKey, encryptedKey
        );

        const aesKey = await crypto.subtle.importKey(
            'raw', aesKeyRaw, { name: 'AES-GCM' }, false, ['decrypt']
        );

        // 用 AES 解密消息
        const iv = Uint8Array.from(atob(encrypted.iv), c => c.charCodeAt(0));
        const ciphertext = Uint8Array.from(atob(encrypted.ciphertext), c => c.charCodeAt(0));
        const decrypted = await crypto.subtle.decrypt(
            { name: 'AES-GCM', iv }, aesKey, ciphertext
        );

        return new TextDecoder().decode(decrypted);
    }
}
```

## 文件加密

```javascript
// ========== 文件加密下载 ==========
async function encryptFile(file, password) {
    // 从密码派生密钥
    const salt = crypto.getRandomValues(new Uint8Array(16));
    const key = await deriveKey(password, salt);

    const iv = crypto.getRandomValues(new Uint8Array(12));
    const fileBuffer = await file.arrayBuffer();
    const encrypted = await crypto.subtle.encrypt(
        { name: 'AES-GCM', iv }, key, fileBuffer
    );

    // 组合: salt(16) + iv(12) + encrypted
    const result = new Uint8Array(16 + 12 + encrypted.byteLength);
    result.set(salt, 0);
    result.set(iv, 16);
    result.set(new Uint8Array(encrypted), 28);

    return new Blob([result], { type: 'application/octet-stream' });
}

async function decryptFile(blob, password) {
    const buffer = await blob.arrayBuffer();
    const data = new Uint8Array(buffer);

    const salt = data.slice(0, 16);
    const iv = data.slice(16, 28);
    const encrypted = data.slice(28);

    const key = await deriveKey(password, salt);
    const decrypted = await crypto.subtle.decrypt(
        { name: 'AES-GCM', iv }, key, encrypted
    );

    return new Blob([decrypted]);
}
```

## 前端加密注意事项

- **HTTPS 是基础**：前端加密不能替代 HTTPS，传输层加密防止中间人攻击
- **密钥安全**：前端的密钥在用户设备上可知，无法防止用户自己解密
- **不要自己实现算法**：使用 Web Crypto API 或成熟库（tweetnacl-js, libsodium-wrappers）
- **性能考虑**：大文件加密使用流式处理，避免一次性加载到内存
- **密钥交换**：使用 ECDH 密钥交换比 RSA 更高效
- **密码学安全随机数**：必须使用 `crypto.getRandomValues()`，不要用 `Math.random()`
