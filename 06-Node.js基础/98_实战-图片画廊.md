# 实战 - 图片画廊 (Image Gallery)

## 项目需求与功能分析

图片画廊是展示图片集合的常见组件。本项目实现网格布局、Lightbox 放大查看、键盘导航和懒加载功能。

### 核心功能

- 响应式网格布局
- 点击放大查看（Lightbox）
- 左右箭头和键盘导航
- 缩略图导航条
- 图片懒加载
- 全屏模式

## 完整代码实现

```html
<!DOCTYPE html>

<meta charset="UTF-8">

<style>

  body { font-family: 'Segoe UI', sans-serif; background: #1a1a2e; padding: 30px; }
  h2 { text-align: center; color: #eee; margin-bottom: 8px; }
  .subtitle { text-align: center; color: #888; font-size: 13px; margin-bottom: 25px; }
  .gallery { display: grid; grid-template-columns: repeat(auto-fill, minmax(220px, 1fr)); gap: 12px; max-width: 1000px; margin: 0 auto; }

  .gallery-item img { width: 100%; height: 100%; object-fit: cover; transition: opacity 0.3s; }
  .gallery-item img.loading { opacity: 0; }
  .gallery-item .placeholder { position: absolute; inset: 0; display: flex; align-items: center; justify-content: center; color: #444; font-size: 14px; }

  /* Lightbox */

  .lightbox-body { flex: 1; display: flex; align-items: center; justify-content: center; position: relative; overflow: hidden; }

<script>
// 使用 picsum.photos 生成随机图片
const images = Array.from({ length: 12 }, (_, i) => ({
  id: i + 1,
  thumb: `https://picsum.photos/seed/${i+10}/400/300`,
  full: `https://picsum.photos/seed/${i+10}/1200/900`,
  caption: `风景照片 #${i + 1}`,
}));

class Gallery {
  constructor(container, images) {
    this.container = container;
    this.images = images;
    this.currentIndex = 0;
    this.lightbox = document.getElementById('lightbox');
    this.lbImage = document.getElementById('lbImage');
    this.lbInfo = document.getElementById('lbInfo');
    this.lbThumbs = document.getElementById('lbThumbs');

    this.renderGallery();
    this.bindEvents();
  }

  renderGallery() {
    this.container.innerHTML = this.images.map((img, i) => `
      <div class="gallery-item" data-index="${i}">
        <div class="placeholder">加载中...</div>
        <img data-src="${img.thumb}" class="loading" alt="${img.caption}">
        <div class="overlay"><span class="caption">${img.caption}</span></div>
      </div>`).join('');

    // 懒加载
    const observer = new IntersectionObserver(entries => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target.querySelector('img');
          if (img.dataset.src) {
            img.src = img.dataset.src;
            img.onload = () => { img.classList.remove('loading'); img.parentElement.querySelector('.placeholder')?.remove(); };
            delete img.dataset.src;
          }
          observer.unobserve(entry.target);
        }
      });
    });
    this.container.querySelectorAll('.gallery-item').forEach(el => observer.observe(el));
  }

  bindEvents() {
    // 点击打开
    this.container.addEventListener('click', e => {
      const item = e.target.closest('.gallery-item');
      if (item) this.open(parseInt(item.dataset.index));
    });

    // Lightbox 事件
    document.getElementById('lbClose').addEventListener('click', () => this.close());
    document.getElementById('lbPrev').addEventListener('click', () => this.prev());
    document.getElementById('lbNext').addEventListener('click', () => this.next());
    this.lightbox.addEventListener('click', e => { if (e.target === this.lightbox || e.target.classList.contains('lightbox-body')) this.close(); });

    // 键盘导航
    document.addEventListener('keydown', e => {
      if (!this.lightbox.classList.contains('active')) return;
      if (e.key === 'Escape') this.close();
      if (e.key === 'ArrowLeft') this.prev();
      if (e.key === 'ArrowRight') this.next();
    });
  }

  open(index) {
    this.currentIndex = index;
    this.lightbox.classList.add('active');
    document.body.style.overflow = 'hidden';
    this.updateLightbox();
    this.renderThumbnails();
  }

  close() {
    this.lightbox.classList.remove('active');
    document.body.style.overflow = '';
  }

  prev() { this.currentIndex = (this.currentIndex - 1 + this.images.length) % this.images.length; this.updateLightbox(); }
  next() { this.currentIndex = (this.currentIndex + 1) % this.images.length; this.updateLightbox(); }

  updateLightbox() {
    const img = this.images[this.currentIndex];
    this.lbImage.src = img.full;
    this.lbInfo.textContent = `${img.caption} (${this.currentIndex + 1}/${this.images.length})`;
    this.lbThumbs.querySelectorAll('.lightbox-thumb').forEach((t, i) => {
      t.classList.toggle('active', i === this.currentIndex);
    });
  }

  renderThumbnails() {
    this.lbThumbs.innerHTML = this.images.map((img, i) =>
      `<img class="lightbox-thumb ${i===this.currentIndex?'active':''}" src="${img.thumb}" alt="${img.caption}">`
    ).join('');
    this.lbThumbs.querySelectorAll('.lightbox-thumb').forEach((t, i) => {
      t.addEventListener('click', () => { this.currentIndex = i; this.updateLightbox(); });
    });
  }
}

new Gallery(document.getElementById('gallery'), images);
</script>
</body>
</html>
```

## 核心技术详解

### 懒加载

使用 `IntersectionObserver` 检测图片进入视口时才加载 `data-src`，减少初始加载流量。

### Lightbox 导航

支持左右箭头按钮和键盘方向键，循环导航。

## 扩展方向

1. **瀑布流布局**：Masonry 风格的不等高网格
2. **缩放拖拽**：放大后可拖拽查看细节
3. **图片上传**：支持拖拽上传新图片
4. **幻灯片模式**：自动轮播展示
5. **分享功能**：生成分享链接
6. **图片编辑**：裁剪、旋转、滤镜
7. **EXIF 信息**：显示拍摄参数
