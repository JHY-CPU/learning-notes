class DoublyListNode {
    constructor(val) {
        this.val = val;
        this.prev = null;
        this.next = null;
    }
}

// ========== 环检测 (快慢指针) ==========
function hasCycle(head) {
    let slow = head, fast = head;
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow === fast) return true;
    }
    return false;
}

## 完整单向链表实现

```javascript
// ========== 完整单向链表 ==========
class ListNode {
    constructor(val, next = null) {
        this.val = val;
        this.next = next;
    }
}

class LinkedList {
    constructor() {
        this.head = null;
        this.tail = null;
        this.length = 0;
    }

    append(val) {
        const node = new ListNode(val);
        if (!this.head) {
            this.head = this.tail = node;
        } else {
            this.tail.next = node;
            this.tail = node;
        }
        this.length++;
        return this;
    }

    prepend(val) {
        this.head = new ListNode(val, this.head);
        if (!this.tail) this.tail = this.head;
        this.length++;
        return this;
    }

    insertAt(index, val) {
        if (index < 0 || index > this.length) return false;
        if (index === 0) return !!this.prepend(val);
        if (index === this.length) return !!this.append(val);

        let prev = this.head;
        for (let i = 0; i < index - 1; i++) prev = prev.next;
        prev.next = new ListNode(val, prev.next);
        this.length++;
        return true;
    }

    remove(val) {
        if (!this.head) return null;
        if (this.head.val === val) {
            const removed = this.head;
            this.head = this.head.next;
            if (!this.head) this.tail = null;
            this.length--;
            return removed.val;
        }
        let curr = this.head;
        while (curr.next) {
            if (curr.next.val === val) {
                const removed = curr.next;
                curr.next = curr.next.next;
                if (!curr.next) this.tail = curr;
                this.length--;
                return removed.val;
            }
            curr = curr.next;
        }
        return null;
    }

    find(val) {
        let curr = this.head;
        let index = 0;
        while (curr) {
            if (curr.val === val) return { node: curr, index };
            curr = curr.next;
            index++;
        }
        return null;
    }

    reverse() {
        let prev = null, curr = this.head;
        this.tail = this.head;
        while (curr) {
            const next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        this.head = prev;
        return this;
    }

    toArray() {
        const result = [];
        let curr = this.head;
        while (curr) {
            result.push(curr.val);
            curr = curr.next;
        }
        return result;
    }

    // 中间节点 (快慢指针)
    getMiddle() {
        let slow = this.head, fast = this.head;
        while (fast?.next?.next) {
            slow = slow.next;
            fast = fast.next.next;
        }
        return slow;
    }

    // 检测环
    hasCycle() {
        let slow = this.head, fast = this.head;
        while (fast?.next) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow === fast) return true;
        }
        return false;
    }

    // 合并两个有序链表
    static mergeSorted(l1, l2) {
        const dummy = new ListNode(0);
        let curr = dummy;
        while (l1 && l2) {
            if (l1.val <= l2.val) {
                curr.next = l1;
                l1 = l1.next;
            } else {
                curr.next = l2;
                l2 = l2.next;
            }
            curr = curr.next;
        }
        curr.next = l1 || l2;
        const list = new LinkedList();
        list.head = dummy.next;
        return list;
    }
}
```

## 链表 vs 数组对比

| 操作 | 链表 | 数组 |
|------|------|------|
| 头部插入 | O(1) | O(n) |
| 尾部插入 | O(1) | O(1) 均摊 |
| 按索引访问 | O(n) | O(1) |
| 按值查找 | O(n) | O(n) |
| 中间插入/删除 | O(1) 已知位置 | O(n) |
| 内存 | 非连续 | 连续 |
