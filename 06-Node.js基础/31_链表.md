## 链表实现


```
// ========== 链表节点 ==========
class ListNode {
    constructor(val) {
        this.val = val;
        this.next = null;
    }
}

// ========== 单向链表 ==========
class LinkedList {
    constructor() {
        this.head = null;
        this.tail = null;
        this.length = 0;
    }

    append(val) { /* 尾部添加 */ }
    prepend(val) { /* 头部插入 */ }
    insertAt(index, val) { /* 指定位置 */ }
    remove(val) { /* 删除 */ }
    find(val) { /* 查找 */ }
    reverse() { /* 反转 */ }
    toArray() { /* 转数组 */ }
}

// ========== 双向链表节点 ==========
class DoublyListNode {
    constructor(val) {
        this.val = val;
        this.prev = null;
        this.next = null;
    }
}

// ========== 环检测 (快慢指针) ==========
function hasCycle(head) {
    let slow = head, fast = head;
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow === fast) return true;
    }
    return false;
}
```
