# 实战 - 待办清单 (Todo List)

## 项目需求与功能分析

待办清单是最经典的前端实战项目，涵盖 CRUD 操作、状态管理、本地存储等核心前端知识。本项目使用纯 HTML + CSS + JavaScript 实现一个功能完善的 Todo 应用。

### 核心功能

- 添加待办事项（输入 + 回车）
- 标记完成 / 未完成（复选框切换）
- 编辑待办内容（双击编辑）
- 删除待办事项
- 状态过滤（全部 / 已完成 / 未完成）
- 清空已完成
- localStorage 持久化
- 待办计数显示

### 技术要点

| 技术点 | 实现方式 |
|--------|----------|
| DOM 操作 | createElement, querySelector, addEventListener |
| 数据存储 | localStorage + JSON.stringify/parse |
| 事件委托 | 在父元素上监听子元素事件 |
| CSS 过渡 | transition 实现完成动画 |

```html
<script>
class TodoApp {
  constructor() {
    this.todos = JSON.parse(localStorage.getItem('todos')) || [];
    this.filter = 'all';
    this.todoList = document.getElementById('todoList');
    this.input = document.getElementById('todoInput');
    this.countText = document.getElementById('countText');
    this.bindEvents();
    this.render();
  }

  bindEvents() {
    document.getElementById('addBtn').addEventListener('click', () => this.add());
    this.input.addEventListener('keydown', e => { if (e.key === 'Enter') this.add(); });
    document.querySelectorAll('.filters button').forEach(btn => {
      btn.addEventListener('click', () => {
        document.querySelector('.filters .active').classList.remove('active');
        btn.classList.add('active');
        this.filter = btn.dataset.filter;
        this.render();
      });
    });
    document.getElementById('clearBtn').addEventListener('click', () => this.clearCompleted());
    // 事件委托
    this.todoList.addEventListener('click', e => {
      const item = e.target.closest('.todo-item');
      if (!item) return;
      const id = Number(item.dataset.id);
      if (e.target.classList.contains('delete-btn')) this.delete(id);
      if (e.target.type === 'checkbox') this.toggle(id);
    });
    this.todoList.addEventListener('dblclick', e => {
      if (e.target.classList.contains('todo-text')) this.startEdit(e.target);
    });
  }

  add() {
    const text = this.input.value.trim();
    if (!text) return;
    this.todos.push({ id: Date.now(), text, completed: false });
    this.input.value = '';
    this.save();
  }

  toggle(id) {
    const todo = this.todos.find(t => t.id === id);
    if (todo) { todo.completed = !todo.completed; this.save(); }
  }

  delete(id) {
    this.todos = this.todos.filter(t => t.id !== id);
    this.save();
  }

  clearCompleted() {
    this.todos = this.todos.filter(t => !t.completed);
    this.save();
  }

  startEdit(span) {
    const item = span.closest('.todo-item');
    const id = Number(item.dataset.id);
    const todo = this.todos.find(t => t.id === id);
    if (!todo) return;
    const input = document.createElement('input');
    input.type = 'text'; input.value = todo.text;
    input.className = 'todo-text editing';
    span.replaceWith(input);
    input.focus();
    const finish = () => {
      todo.text = input.value.trim() || todo.text;
      this.save();
    };
    input.addEventListener('blur', finish);
    input.addEventListener('keydown', e => { if (e.key === 'Enter') input.blur(); });
  }

  save() {
    localStorage.setItem('todos', JSON.stringify(this.todos));
    this.render();
  }

  render() {
    const filtered = this.todos.filter(t => {
      if (this.filter === 'active') return !t.completed;
      if (this.filter === 'completed') return t.completed;
      return true;
    });
    this.todoList.innerHTML = filtered.length === 0
      ? '<div class="empty">暂无待办事项</div>'
      : filtered.map(t => `
        <li class="todo-item ${t.completed ? 'completed' : ''}" data-id="${t.id}">
          <input type="checkbox" ${t.completed ? 'checked' : ''}>
          <span class="todo-text">${this.escapeHtml(t.text)}</span>
          <button class="delete-btn">✕</button>
        </li>`).join('');
    const active = this.todos.filter(t => !t.completed).length;
    this.countText.textContent = `${active} 项未完成 / 共 ${this.todos.length} 项`;
  }

  escapeHtml(str) {
    const div = document.createElement('div');
    div.textContent = str;
    return div.innerHTML;
  }
}

new TodoApp();
</script>
</body>
</html>
```

## 核心技术详解

### 数据持久化

使用 `localStorage` 存储 JSON 字符串。每次修改后调用 `save()` 同步到存储。

### 事件委托

在父元素 `ul` 上绑定点击事件，通过 `e.target` 判断具体操作类型，避免给每个子元素绑定事件。

### 双击编辑

双击文本时，将 `span` 替换为 `input`，失焦或回车时保存并恢复。

## 测试用例

```javascript
// 使用 Jest 测试
describe('TodoApp Logic', () => {
  let todos;

  beforeEach(() => {
    todos = [];
  });

  test('添加待办', () => {
    todos.push({ id: 1, text: '学习JS', completed: false });
    expect(todos.length).toBe(1);
    expect(todos[0].text).toBe('学习JS');
  });

  test('切换完成状态', () => {
    todos.push({ id: 1, text: '测试', completed: false });
    todos[0].completed = !todos[0].completed;
    expect(todos[0].completed).toBe(true);
  });

  test('删除待办', () => {
    todos.push({ id: 1, text: '删除我', completed: false });
    todos = todos.filter(t => t.id !== 1);
    expect(todos.length).toBe(0);
  });

  test('过滤未完成', () => {
    todos.push({ id: 1, text: 'A', completed: true });
    todos.push({ id: 2, text: 'B', completed: false });
    const active = todos.filter(t => !t.completed);
    expect(active.length).toBe(1);
  });

  test('清空已完成', () => {
    todos.push({ id: 1, text: 'A', completed: true });
    todos.push({ id: 2, text: 'B', completed: false });
    todos = todos.filter(t => !t.completed);
    expect(todos.length).toBe(1);
  });
});
```
