# 实战 - 图片轮播 (Carousel)

## 项目需求与功能分析

图片轮播（Carousel）是网页中最常见的交互组件之一。本项目实现一个功能完善的轮播组件，支持自动播放、手动切换、指示器、触摸滑动等。

### 核心功能

- 自动播放（可配置间隔）
- 左右箭头手动切换
- 底部指示器（点击跳转）
- 鼠标悬停暂停
- CSS 过渡动画
- 循环滚动
- 触摸滑动支持（移动端）

```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>图片轮播</title>
<style>
  * { margin: 0; padding: 0; box-sizing: border-box; }
  body { font-family: sans-serif; background: #f0f0f0; display: flex; justify-content: center; padding: 40px; }
  .carousel { position: relative; width: 100%; max-width: 800px; border-radius: 12px; overflow: hidden; box-shadow: 0 4px 20px rgba(0,0,0,0.15); }
  .carousel-track { display: flex; transition: transform 0.5s ease; }
  .carousel-slide { min-width: 100%; height: 400px; display: flex; align-items: center; justify-content: center; font-size: 48px; color: #fff; }
  .carousel-slide:nth-child(1) { background: linear-gradient(135deg, #667eea, #764ba2); }
  .carousel-slide:nth-child(2) { background: linear-gradient(135deg, #f093fb, #f5576c); }
  .carousel-slide:nth-child(3) { background: linear-gradient(135deg, #4facfe, #00f2fe); }
  .carousel-slide:nth-child(4) { background: linear-gradient(135deg, #43e97b, #38f9d7); }
  .carousel-slide:nth-child(5) { background: linear-gradient(135deg, #fa709a, #fee140); }
  .arrow { position: absolute; top: 50%; transform: translateY(-50%); background: rgba(255,255,255,0.3); border: none; color: #fff; font-size: 24px; width: 44px; height: 44px; border-radius: 50%; cursor: pointer; transition: background 0.2s; z-index: 10; }
  .arrow:hover { background: rgba(255,255,255,0.5); }
  .arrow-left { left: 15px; }
  .arrow-right { right: 15px; }
  .indicators { position: absolute; bottom: 20px; left: 50%; transform: translateX(-50%); display: flex; gap: 10px; }
  .indicator { width: 12px; height: 12px; border-radius: 50%; background: rgba(255,255,255,0.4); border: none; cursor: pointer; transition: all 0.3s; }
  .indicator.active { background: #fff; transform: scale(1.2); }
  .caption { position: absolute; bottom: 50px; left: 50%; transform: translateX(-50%); color: #fff; font-size: 20px; text-shadow: 0 1px 4px rgba(0,0,0,0.5); }
</style>
</head>
<body>

<div class="carousel" id="carousel">
  <div class="carousel-track" id="track">
    <div class="carousel-slide">风景 1</div>
    <div class="carousel-slide">风景 2</div>
    <div class="carousel-slide">风景 3</div>
    <div class="carousel-slide">风景 4</div>
    <div class="carousel-slide">风景 5</div>
  </div>
  <button class="arrow arrow-left" id="prevBtn">&#10094;</button>
  <button class="arrow arrow-right" id="nextBtn">&#10095;</button>
  <div class="indicators" id="indicators"></div>
  <div class="caption" id="caption"></div>
</div>

<script>
class Carousel {
  constructor(container, options = {}) {
    this.container = container;
    this.track = container.querySelector('#track');
    this.slides = container.querySelectorAll('.carousel-slide');
    this.indicatorsEl = container.querySelector('#indicators');
    this.captionEl = container.querySelector('#caption');
    this.currentIndex = 0;
    this.total = this.slides.length;
    this.interval = options.interval || 3000;
    this.autoTimer = null;
    this.isPlaying = true;
    this.captions = options.captions || Array.from(this.slides).map((_, i) => `风景 ${i + 1}`);

    this.init();
  }

  init() {
    // 创建指示器
    for (let i = 0; i < this.total; i++) {
      const dot = document.createElement('button');
      dot.className = 'indicator' + (i === 0 ? ' active' : '');
      dot.addEventListener('click', () => this.goTo(i));
      this.indicatorsEl.appendChild(dot);
    }
    this.indicators = this.indicatorsEl.querySelectorAll('.indicator');

    // 箭头事件
    this.container.querySelector('#prevBtn').addEventListener('click', () => this.prev());
    this.container.querySelector('#nextBtn').addEventListener('click', () => this.next());

    // 鼠标悬停暂停
    this.container.addEventListener('mouseenter', () => this.pause());
    this.container.addEventListener('mouseleave', () => this.play());

    // 触摸支持
    this.container.addEventListener('touchstart', e => { this.touchStartX = e.touches[0].clientX; });
    this.container.addEventListener('touchend', e => {
      const diff = e.changedTouches[0].clientX - this.touchStartX;
      if (Math.abs(diff) > 50) diff > 0 ? this.prev() : this.next();
    });

    // 键盘支持
    document.addEventListener('keydown', e => {
      if (e.key === 'ArrowLeft') this.prev();
      if (e.key === 'ArrowRight') this.next();
    });

    this.update();
    this.play();
  }

  goTo(index) {
    this.currentIndex = ((index % this.total) + this.total) % this.total;
    this.update();
  }

  next() { this.goTo(this.currentIndex + 1); }
  prev() { this.goTo(this.currentIndex - 1); }

  update() {
    this.track.style.transform = `translateX(-${this.currentIndex * 100}%)`;
    this.indicators.forEach((dot, i) => dot.classList.toggle('active', i === this.currentIndex));
    this.captionEl.textContent = this.captions[this.currentIndex];
  }

  play() {
    this.isPlaying = true;
    clearInterval(this.autoTimer);
    this.autoTimer = setInterval(() => this.next(), this.interval);
  }

  pause() {
    this.isPlaying = false;
    clearInterval(this.autoTimer);
  }
}

const carousel = new Carousel(document.getElementById('carousel'), {
  interval: 3000,
  captions: ['山脉日落', '海边夕阳', '极光之夜', '热带雨林', '金色麦田'],
});
</script>
</body>
</html>
```

## 核心技术详解

### CSS 过渡动画

使用 `transform: translateX()` 实现滑动效果，`transition: transform 0.5s ease` 实现平滑过渡。

### 自动播放控制

使用 `setInterval` 实现自动轮播，鼠标悬停时 `clearInterval` 暂停。

### 触摸滑动

监听 `touchstart` 和 `touchend` 事件，计算触摸距离判断滑动方向。

## 测试用例

```javascript
describe('Carousel Logic', () => {
  test('next 循环', () => {
    let index = 0; const total = 5;
    const next = () => { index = ((index + 1) % total + total) % total; };
    for (let i = 0; i < 6; i++) next();
    expect(index).toBe(1);
  });

  test('prev 循环', () => {
    let index = 0; const total = 5;
    const prev = () => { index = ((index - 1) % total + total) % total; };
    prev();
    expect(index).toBe(4);
  });

  test('goTo 跳转', () => {
    let index = 0; const total = 5;
    const goTo = (i) => { index = ((i % total) + total) % total; };
    goTo(3);
    expect(index).toBe(3);
  });
});
```

## 扩展方向

1. **淡入淡出**：支持渐变切换效果
2. **缩略图导航**：底部显示缩略图预览
3. **全屏模式**：双击进入全屏查看
4. **视频支持**：轮播中嵌入视频
5. **懒加载**：图片进入视口才加载
6. **动态数据**：从 API 获取轮播数据
7. **无限循环**：克隆首尾元素实现无缝循环
