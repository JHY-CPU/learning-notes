# 实战 - 倒计时与秒表 (Countdown & Stopwatch)

## 项目需求与功能分析

倒计时器和秒表是常见的时间工具组件。本项目同时实现两个功能，支持暂停/继续、圈数记录和环形进度显示。

### 核心功能

- **秒表**：开始、暂停、复位、记录圈数
- **倒计时**：设置时间、开始、暂停、继续
- 精确到毫秒
- 环形进度条
- 圈数列表
- 键盘快捷键

<script>
class Stopwatch {
  constructor() {
    this.elapsed = 0;
    this.running = false;
    this.lastTime = 0;
    this.laps = [];
    this.raf = null;
    this.display = document.getElementById('swDisplay');
    this.circle = document.getElementById('swCircle');
    this.lapsEl = document.getElementById('swLaps');
    this.circumference = 2 * Math.PI * 90;

    document.getElementById('swStart').addEventListener('click', () => this.toggle());
    document.getElementById('swLap').addEventListener('click', () => this.lap());
    document.getElementById('swReset').addEventListener('click', () => this.reset());
  }

  toggle() {
    if (this.running) this.pause();
    else this.start();
  }

  start() {
    this.running = true;
    this.lastTime = performance.now();
    document.getElementById('swStart').textContent = '暂停';
    document.getElementById('swStart').classList.replace('btn-start', 'btn-pause');
    document.getElementById('swLap').disabled = false;
    this.tick();
  }

  pause() {
    this.running = false;
    cancelAnimationFrame(this.raf);
    document.getElementById('swStart').textContent = '继续';
    document.getElementById('swStart').classList.replace('btn-pause', 'btn-start');
    document.getElementById('swLap').disabled = true;
  }

  reset() {
    this.running = false;
    cancelAnimationFrame(this.raf);
    this.elapsed = 0;
    this.laps = [];
    this.updateDisplay(0);
    this.lapsEl.innerHTML = '';
    document.getElementById('swStart').textContent = '开始';
    document.getElementById('swStart').classList.replace('btn-pause', 'btn-start');
    document.getElementById('swLap').disabled = true;
  }

  tick() {
    if (!this.running) return;
    const now = performance.now();
    this.elapsed += now - this.lastTime;
    this.lastTime = now;
    this.updateDisplay(this.elapsed);
    this.raf = requestAnimationFrame(() => this.tick());
  }

  lap() {
    this.laps.push(this.elapsed);
    const lapNum = this.laps.length;
    const lapTime = lapNum === 1 ? this.elapsed : this.elapsed - this.laps[lapNum - 2];
    this.lapsEl.innerHTML = `
      <div class="lap-item">
        <span class="lap-num">圈 ${lapNum}</span>
        <span>圈: ${this.format(lapTime)}</span>
        <span>总计: ${this.format(this.elapsed)}</span>
      </div>` + this.lapsEl.innerHTML;
  }

  updateDisplay(ms) {
    this.display.innerHTML = this.format(ms);
    const seconds = ms / 1000;
    const offset = this.circumference - (seconds % 60) / 60 * this.circumference;
    this.circle.style.strokeDashoffset = offset;
  }

  format(ms) {
    const h = Math.floor(ms / 3600000);
    const m = Math.floor((ms % 3600000) / 60000);
    const s = Math.floor((ms % 60000) / 1000);
    const cs = Math.floor((ms % 1000) / 10);
    const pad = n => String(n).padStart(2, '0');
    return `${pad(h)}:${pad(m)}:${pad(s)}<span class="ms">.${pad(cs)}</span>`;
  }
}

class Countdown {
  constructor() {
    this.totalMs = 0;
    this.remainingMs = 0;
    this.running = false;
    this.lastTime = 0;
    this.raf = null;
    this.display = document.getElementById('cdDisplay');
    this.circle = document.getElementById('cdCircle');
    this.progress = document.getElementById('cdProgress');
    this.circumference = 2 * Math.PI * 90;

    document.getElementById('cdStart').addEventListener('click', () => this.start());
    document.getElementById('cdPause').addEventListener('click', () => this.pause());
    document.getElementById('cdReset').addEventListener('click', () => this.reset());
  }

  start() {
    if (this.remainingMs <= 0) {
      const h = parseInt(document.getElementById('cdHours').value) || 0;
      const m = parseInt(document.getElementById('cdMinutes').value) || 0;
      const s = parseInt(document.getElementById('cdSeconds').value) || 0;
      this.totalMs = (h * 3600 + m * 60 + s) * 1000;
      this.remainingMs = this.totalMs;
    }
    if (this.remainingMs <= 0) return;
    this.running = true;
    this.lastTime = performance.now();
    document.getElementById('cdStart').disabled = true;
    document.getElementById('cdPause').disabled = false;
    this.tick();
  }

  pause() {
    this.running = false;
    cancelAnimationFrame(this.raf);
    document.getElementById('cdStart').disabled = false;
    document.getElementById('cdStart').textContent = '继续';
    document.getElementById('cdPause').disabled = true;
  }

  reset() {
    this.running = false;
    cancelAnimationFrame(this.raf);
    this.remainingMs = 0;
    this.totalMs = 0;
    this.updateDisplay(0);
    this.circle.style.strokeDashoffset = '0';
    this.progress.style.width = '100%';
    document.getElementById('cdStart').disabled = false;
    document.getElementById('cdStart').textContent = '开始';
    document.getElementById('cdPause').disabled = true;
  }

  tick() {
    if (!this.running) return;
    const now = performance.now();
    this.remainingMs -= now - this.lastTime;
    this.lastTime = now;

    if (this.remainingMs <= 0) {
      this.remainingMs = 0;
      this.running = false;
      this.updateDisplay(0);
      document.getElementById('cdStart').disabled = false;
      document.getElementById('cdStart').textContent = '开始';
      document.getElementById('cdPause').disabled = true;
      alert('倒计时结束！');
      return;
    }

    this.updateDisplay(this.remainingMs);
    this.raf = requestAnimationFrame(() => this.tick());
  }

  updateDisplay(ms) {
    const m = Math.floor(ms / 60000);
    const s = Math.floor((ms % 60000) / 1000);
    const cs = Math.floor((ms % 1000) / 10);
    const pad = n => String(n).padStart(2, '0');
    this.display.innerHTML = `${pad(m)}:${pad(s)}<span class="ms">.${pad(cs)}</span>`;

    const pct = this.totalMs > 0 ? ms / this.totalMs : 0;
    this.circle.style.strokeDashoffset = this.circumference * (1 - pct);
    this.progress.style.width = (pct * 100) + '%';
  }
}

new Stopwatch();
new Countdown();
</script>
</body>
</html>

## 核心技术详解

### 高精度计时

使用 `performance.now()` 而非 `Date.now()`，精度可达微秒级别。使用 `requestAnimationFrame` 替代 `setInterval` 实现更平滑的更新。

### SVG 环形进度

使用 SVG `circle` 的 `stroke-dasharray` 和 `stroke-dashoffset` 属性实现环形进度条。
