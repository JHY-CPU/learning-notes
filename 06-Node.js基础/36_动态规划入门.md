## DP 核心概念


```
// ========== 斐波那契 (DP) ==========
function fibDP(n) {
    const dp = [0, 1];
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

// ========== 爬楼梯 ==========
// 每次可以爬 1 或 2 阶
function climbStairs(n) {
    const dp = [1, 1];
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

// ========== 0/1 背包 ==========
function knapsack(weights, values, capacity) {
    const n = weights.length;
    const dp = Array(n + 1).fill(0).map(() => Array(capacity + 1).fill(0));
    for (let i = 1; i <= n; i++) {
        for (let w = 1; w <= capacity; w++) {
            if (weights[i - 1] <= w) {
                dp[i][w] = Math.max(
                    dp[i - 1][w],
                    dp[i - 1][w - weights[i - 1]] + values[i - 1]
                );
            } else {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    return dp[n][capacity];
}

// ========== 最长公共子序列 (LCS) ==========
function lcs(text1, text2) {
    const m = text1.length, n = text2.length;
    const dp = Array(m + 1).fill(0).map(() => Array(n + 1).fill(0));
    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (text1[i - 1] === text2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }
    return dp[m][n];
}
```

## 经典 DP 问题

```javascript
// ========== 最长递增子序列 (LIS) ==========
function lengthOfLIS(nums) {
    if (!nums.length) return 0;
    const dp = new Array(nums.length).fill(1);
    for (let i = 1; i < nums.length; i++) {
        for (let j = 0; j < i; j++) {
            if (nums[j] < nums[i]) {
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
    }
    return Math.max(...dp);
}
// lengthOfLIS([10,9,2,5,3,7,101,18]) → 4 ([2,3,7,101])

// O(n log n) 优化版 (二分查找)
function lengthOfLISFast(nums) {
    const tails = [];
    for (const num of nums) {
        let left = 0, right = tails.length;
        while (left < right) {
            const mid = (left + right) >> 1;
            if (tails[mid] < num) left = mid + 1;
            else right = mid;
        }
        tails[left] = num;
    }
    return tails.length;
}

// ========== 零钱兑换 ==========
function coinChange(coins, amount) {
    const dp = new Array(amount + 1).fill(Infinity);
    dp[0] = 0;
    for (let i = 1; i <= amount; i++) {
        for (const coin of coins) {
            if (coin <= i) {
                dp[i] = Math.min(dp[i], dp[i - coin] + 1);
            }
        }
    }
    return dp[amount] === Infinity ? -1 : dp[amount];
}
// coinChange([1, 5, 11], 15) → 3 (5+5+5)

// ========== 最大子数组和 ==========
function maxSubArray(nums) {
    let maxSum = nums[0], currentSum = nums[0];
    for (let i = 1; i < nums.length; i++) {
        currentSum = Math.max(nums[i], currentSum + nums[i]);
        maxSum = Math.max(maxSum, currentSum);
    }
    return maxSum;
}
// maxSubArray([-2,1,-3,4,-1,2,1,-5,4]) → 6 ([4,-1,2,1])
```

## DP 解题框架

```javascript
// ========== DP 解题五步法 ==========
// 1. 定义状态: dp[i] 的含义是什么?
// 2. 状态转移方程: dp[i] 如何由前面的状态推导?
// 3. 初始化: base case 是什么?
// 4. 遍历顺序: 从前往后? 从后往前?
// 5. 返回结果: 最终答案在哪里?

// ========== 空间优化 ==========
// 从二维 DP 优化为一维 DP
// 原始: dp[i][w] = ...
// 优化: dp[w] = ... (内层循环从后往前)

// 0/1 背包空间优化
function knapsackOptimized(weights, values, capacity) {
    const dp = new Array(capacity + 1).fill(0);
    for (let i = 0; i < weights.length; i++) {
        for (let w = capacity; w >= weights[i]; w--) {
            dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
        }
    }
    return dp[capacity];
}

// ========== 打印 DP 路径 ==========
function coinChangeWithPath(coins, amount) {
    const dp = new Array(amount + 1).fill(Infinity);
    const parent = new Array(amount + 1).fill(-1);
    dp[0] = 0;

    for (let i = 1; i <= amount; i++) {
        for (const coin of coins) {
            if (coin <= i && dp[i - coin] + 1 < dp[i]) {
                dp[i] = dp[i - coin] + 1;
                parent[i] = coin;
            }
        }
    }

    if (dp[amount] === Infinity) return null;

    const path = [];
    let curr = amount;
    while (curr > 0) {
        path.push(parent[curr]);
        curr -= parent[curr];
    }
    return { count: dp[amount], coins: path };
}
// coinChangeWithPath([1, 5, 11], 15)
// → { count: 3, coins: [5, 5, 5] }
```
