数组扁平化方法

// ========== flat (ES2019) ========== [1, [2, [3, [4]]]].flat(2); // [1, 2, 3, [4]] [1, [2, [3, [4]]]].flat(Infinity); // [1, 2, 3, 4] // ========== flatMap ========== ['a b', 'c d'].flatMap(s => s.split(' ')); // ['a', 'b', 'c', 'd'] // ========== 递归 reduce ========== function flatten(arr, depth = Infinity) { return depth > 0 ? arr.reduce((acc, v) => acc.concat( Array.isArray(v) ? flatten(v, depth - 1) : v ), []) : arr.slice(); } // ========== 栈实现 (非递归) ========== function flattenStack(arr) { const stack = [...arr]; const result = []; while (stack.length) { const v = stack.shift(); Array.isArray(v) ? stack.unshift(...v) : result.push(v); } return result; } // ========== toString (仅数字) ========== [1, [2, [3, [4]]]].toString().split(',').map(Number); // [1, 2, 3, 4] (只适用于数字)
