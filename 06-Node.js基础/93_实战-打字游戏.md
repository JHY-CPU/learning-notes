# 实战 - 打字游戏 (Typing Game)

## 项目需求与功能分析

打字游戏是提升键盘输入速度和准确率的趣味工具。本项目实现一个支持计时、WPM 统计和准确率计算的打字游戏。

### 核心功能

- 随机句子 / 单词生成
- 实时输入对比（正确绿色、错误红色）
- 计时统计
- WPM（每分钟字数）计算
- 准确率计算
- 结果报告
- 支持重新开始

<script>
const WORDS = [
  'algorithm', 'binary', 'cache', 'database', 'element', 'function',
  'graph', 'hash', 'index', 'javascript', 'kernel', 'library',
  'memory', 'network', 'object', 'protocol', 'query', 'recursion',
  'server', 'tree', 'unicode', 'variable', 'webpack', 'xml',
  'async', 'buffer', 'callback', 'deploy', 'endpoint', 'frontend',
  'git', 'hosting', 'input', 'json', 'key', 'loop',
  'module', 'node', 'output', 'promise', 'queue', 'rest',
  'stack', 'template', 'url', 'vue', 'worker', 'express',
  'react', 'angular', 'typescript', 'python', 'docker', 'kubernetes',
];

class TypingGame {
  constructor() {
    this.input = document.getElementById('typingInput');
    this.display = document.getElementById('textDisplay');
    this.results = document.getElementById('results');
    this.wordCount = 20;
    this.reset();
    this.bindEvents();
  }

  bindEvents() {
    this.input.addEventListener('input', () => this.onInput());
    document.getElementById('restartBtn').addEventListener('click', () => this.reset());
    document.getElementById('retryBtn').addEventListener('click', () => this.reset());
    document.querySelectorAll('.diff-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        document.querySelector('.diff-btn.active').classList.remove('active');
        btn.classList.add('active');
        this.wordCount = parseInt(btn.dataset.words);
        this.reset();
      });
    });
  }

  reset() {
    this.chars = this.generateText().split('');
    this.currentIndex = 0;
    this.correctCount = 0;
    this.wrongCount = 0;
    this.startTime = null;
    this.timer = null;
    this.finished = false;
    this.input.value = '';
    this.input.disabled = false;
    this.results.classList.remove('show');
    this.render();
    this.updateStats();
    this.input.focus();
  }

  generateText() {
    const shuffled = [...WORDS].sort(() => Math.random() - 0.5);
    return shuffled.slice(0, this.wordCount).join(' ');
  }

  onInput() {
    if (this.finished) return;
    if (!this.startTime) this.startTimer();

    const inputVal = this.input.value;
    const lastChar = inputVal[inputVal.length - 1];
    const expectedChar = this.chars[this.currentIndex];

    if (lastChar === expectedChar) {
      this.correctCount++;
    } else {
      this.wrongCount++;
    }

    this.currentIndex++;
    this.render();
    this.updateStats();

    // 清空输入框（逐字符模式）
    this.input.value = '';

    if (this.currentIndex >= this.chars.length) {
      this.finish();
    }
  }

  startTimer() {
    this.startTime = performance.now();
    this.timer = setInterval(() => {
      const elapsed = (performance.now() - this.startTime) / 1000;
      document.getElementById('timeDisplay').textContent = elapsed.toFixed(0);
    }, 100);
  }

  finish() {
    this.finished = true;
    clearInterval(this.timer);
    this.input.disabled = true;

    const elapsed = (performance.now() - this.startTime) / 1000;
    const words = this.chars.length / 5;
    const wpm = Math.round(words / (elapsed / 60));
    const accuracy = Math.round(this.correctCount / (this.correctCount + this.wrongCount) * 100);

    document.getElementById('finalWpm').textContent = wpm;
    document.getElementById('finalAccuracy').textContent = accuracy + '%';
    document.getElementById('finalTime').textContent = elapsed.toFixed(1) + 's';
    document.getElementById('finalChars').textContent = this.chars.length;
    this.results.classList.add('show');
  }

  updateStats() {
    const total = this.correctCount + this.wrongCount;
    const accuracy = total > 0 ? Math.round(this.correctCount / total * 100) : 100;
    const elapsed = this.startTime ? (performance.now() - this.startTime) / 1000 : 0;
    const wpm = elapsed > 0 ? Math.round((this.correctCount / 5) / (elapsed / 60)) : 0;

    document.getElementById('wpmDisplay').textContent = wpm;
    document.getElementById('accuracyDisplay').textContent = accuracy;
    document.getElementById('progressDisplay').textContent = `${this.currentIndex}/${this.chars.length}`;
  }

  render() {
    this.display.innerHTML = this.chars.map((char, i) => {
      let cls = 'char';
      if (i < this.currentIndex) {
        // 已输入的字符需要查看之前的输入结果
        cls += ' correct'; // 简化：逐字符输入时已在 onInput 中计数
      } else if (i === this.currentIndex) {
        cls += ' current';
      }
      return `<span class="${cls}">${char === ' ' ? '&nbsp;' : char}</span>`;
    }).join('');
  }
}

new TypingGame();
</script>

## 核心技术详解

### 逐字符对比

每次输入事件只处理一个字符，与预期字符比较后清空输入框，实现逐字输入效果。

### WPM 计算

WPM = (正确字符数 / 5) / (用时 / 60)，其中 5 是标准单词平均字符数。

## 扩展方向

1. **多语言支持**：中文 / 英文 / 代码片段
2. **排行榜**：本地最高分记录
3. **错误热力图**：统计哪些字符容易出错
4. **竞速模式**：倒计时内打尽可能多的字
5. **段落模式**：完整段落打字而非单词
6. **声音反馈**：正确 / 错误音效
7. **自定义文本**：用户粘贴自定义文本练习
