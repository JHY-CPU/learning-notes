## 快速排序实现


```
// ========== 快速排序 (Lomuto 分区) ==========
function quickSort(arr, left = 0, right = arr.length - 1) {
    if (left >= right) return;
    const pivot = partition(arr, left, right);
    quickSort(arr, left, pivot - 1);
    quickSort(arr, pivot + 1, right);
    return arr;
}

// Lomuto 分区方案
function partition(arr, left, right) {
    const pivot = arr[right]; // 选最后一个元素
    let i = left;
    for (let j = left; j < right; j++) {
        if (arr[j] < pivot) {
            [arr[i], arr[j]] = [arr[j], arr[i]];
            i++;
        }
    }
    [arr[i], arr[right]] = [arr[right], arr[i]];
    return i;
}

// ========== 随机 pivot (避免最坏情况) ==========
function randomPartition(arr, left, right) {
    const rand = left + Math.floor(Math.random() * (right - left + 1));
    [arr[rand], arr[right]] = [arr[right], arr[rand]];
    return partition(arr, left, right);
}

// ========== 三路快排 (处理重复元素) ==========
function quickSort3Way(arr, left = 0, right = arr.length - 1) {
    if (left >= right) return;
    const [lt, gt] = partition3Way(arr, left, right);
    quickSort3Way(arr, left, lt - 1);
    quickSort3Way(arr, gt + 1, right);
    return arr;
}
```

## 三路快排完整实现

```javascript
// ========== 三路快排 (Dutch National Flag) ==========
// 适合有大量重复元素的数组
function quickSort3Way(arr, left = 0, right = arr.length - 1) {
    if (left >= right) return;

    const pivot = arr[left];
    let lt = left;      // arr[left..lt-1] < pivot
    let gt = right + 1; // arr[gt..right] > pivot
    let i = left + 1;   // arr[lt..i-1] == pivot

    while (i < gt) {
        if (arr[i] < pivot) {
            [arr[lt], arr[i]] = [arr[i], arr[lt]];
            lt++;
            i++;
        } else if (arr[i] > pivot) {
            gt--;
            [arr[i], arr[gt]] = [arr[gt], arr[i]];
        } else {
            i++;
        }
    }

    quickSort3Way(arr, left, lt - 1);
    quickSort3Way(arr, gt, right);
    return arr;
}

// 测试: 100万个元素，90%相同
const arr = Array.from({ length: 1000000 }, () => Math.random() > 0.1 ? 1 : Math.floor(Math.random() * 100));
// 三路快排比普通快排快 5-10 倍
```

## 快排优化策略

```javascript
// ========== 插入排序优化 (小数组) ==========
function quickSortOptimized(arr, left = 0, right = arr.length - 1) {
    // 小于10个元素时使用插入排序
    if (right - left < 10) {
        for (let i = left + 1; i <= right; i++) {
            const key = arr[i];
            let j = i - 1;
            while (j >= left && arr[j] > key) {
                arr[j + 1] = arr[j];
                j--;
            }
            arr[j + 1] = key;
        }
        return;
    }

    // 三数取中选择 pivot
    const mid = (left + right) >> 1;
    if (arr[mid] < arr[left]) [arr[left], arr[mid]] = [arr[mid], arr[left]];
    if (arr[right] < arr[left]) [arr[left], arr[right]] = [arr[right], arr[left]];
    if (arr[mid] < arr[right]) [arr[mid], arr[right]] = [arr[right], arr[mid]];

    const pivot = arr[right];
    let i = left, j = right - 1;
    while (true) {
        while (arr[++i] < pivot);
        while (arr[--j] > pivot);
        if (i >= j) break;
        [arr[i], arr[j]] = [arr[j], arr[i]];
    }
    [arr[i], arr[right]] = [arr[right], arr[i]];

    quickSortOptimized(arr, left, i - 1);
    quickSortOptimized(arr, i + 1, right);
}

// ========== 尾递归优化 (减少栈深度) ==========
function quickSortTailCall(arr, left = 0, right = arr.length - 1) {
    while (left < right) {
        const pivot = partition(arr, left, right);
        // 优先递归较小的子数组，用循环处理较大的
        if (pivot - left < right - pivot) {
            quickSortTailCall(arr, left, pivot - 1);
            left = pivot + 1;
        } else {
            quickSortTailCall(arr, pivot + 1, right);
            right = pivot - 1;
        }
    }
}
```

## 第K大元素

```javascript
// ========== 快速选择 (Quickselect) ==========
// 时间复杂度: 平均 O(n), 最坏 O(n²)
function quickSelect(arr, k) {
    const a = [...arr];
    k = a.length - k; // 转为第k小

    function select(left, right) {
        const pivot = partition(a, left, right);
        if (pivot === k) return a[pivot];
        if (pivot < k) return select(pivot + 1, right);
        return select(left, pivot - 1);
    }

    return select(0, a.length - 1);
}

// 第3大元素
quickSelect([3, 1, 4, 1, 5, 9, 2, 6], 3); // 4

// ========== 使用场景 ==========
// 1. 找中位数: quickSelect(arr, Math.ceil(arr.length / 2))
// 2. Top-K 问题: 找前K大/小的元素
// 3. 百分位数: 找 P% 分位数
```
