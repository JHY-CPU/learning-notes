```
class Stack {
    constructor() { this.items = []; }
    push(v) { this.items.push(v); }
    pop() { return this.items.pop(); }
    peek() { return this.items[this.items.length - 1]; }
    get size() { return this.items.length; }
    get isEmpty() { return this.items.length === 0; }
}

// ========== 队列 (FIFO) ==========
class Queue {
    constructor() { this.items = []; }
    enqueue(v) { this.items.push(v); }
    dequeue() { return this.items.shift(); }
    front() { return this.items[0]; }
    get size() { return this.items.length; }
    get isEmpty() { return this.items.length === 0; }
}
```

## 高效队列与单调栈

```javascript
// ========== 高效队列 (避免 shift 的 O(n)) ==========
class EfficientQueue {
    constructor() {
        this.items = {};
        this.head = 0;
        this.tail = 0;
    }
    enqueue(v) { this.items[this.tail++] = v; }
    dequeue() {
        if (this.isEmpty) return undefined;
        const v = this.items[this.head];
        delete this.items[this.head++];
        return v;
    }
    front() { return this.items[this.head]; }
    get size() { return this.tail - this.head; }
    get isEmpty() { return this.size === 0; }
}

// ========== 循环队列 ==========
class CircularQueue {
    constructor(capacity) {
        this.items = new Array(capacity);
        this.capacity = capacity;
        this.head = 0;
        this.tail = 0;
        this.count = 0;
    }
    enqueue(v) {
        if (this.count === this.capacity) return false;
        this.items[this.tail] = v;
        this.tail = (this.tail + 1) % this.capacity;
        this.count++;
        return true;
    }
    dequeue() {
        if (this.count === 0) return undefined;
        const v = this.items[this.head];
        this.head = (this.head + 1) % this.capacity;
        this.count--;
        return v;
    }
    front() { return this.count === 0 ? undefined : this.items[this.head]; }
    get size() { return this.count; }
    get isFull() { return this.count === this.capacity; }
    get isEmpty() { return this.count === 0; }
}

// ========== 单调栈 (下一个更大元素) ==========
function nextGreaterElement(nums) {
    const result = new Array(nums.length).fill(-1);
    const stack = []; // 存索引
    for (let i = 0; i < nums.length; i++) {
        while (stack.length && nums[stack[stack.length - 1]] < nums[i]) {
            result[stack.pop()] = nums[i];
        }
        stack.push(i);
    }
    return result;
}
// nextGreaterElement([2, 1, 2, 4, 3]) → [4, 2, 4, -1, -1]

// ========== 最小栈 ==========
class MinStack {
    constructor() {
        this.stack = [];
        this.minStack = [];
    }
    push(v) {
        this.stack.push(v);
        this.minStack.push(
            this.minStack.length === 0
                ? v
                : Math.min(v, this.minStack[this.minStack.length - 1])
        );
    }
    pop() {
        this.minStack.pop();
        return this.stack.pop();
    }
    top() { return this.stack[this.stack.length - 1]; }
    getMin() { return this.minStack[this.minStack.length - 1]; }
}
```

## 实战：括号匹配与计算器

```javascript
// ========== 括号匹配 ==========
function isValid(s) {
    const stack = [];
    const pairs = { ')': '(', ']': '[', '}': '{' };
    for (const ch of s) {
        if ('([{'.includes(ch)) {
            stack.push(ch);
        } else {
            if (stack.pop() !== pairs[ch]) return false;
        }
    }
    return stack.length === 0;
}

// ========== 简易计算器 (中缀表达式求值) ==========
function calculate(expression) {
    const nums = [];
    const ops = [];
    const precedence = { '+': 1, '-': 1, '*': 2, '/': 2 };

    function applyOp() {
        const b = nums.pop(), a = nums.pop();
        const op = ops.pop();
        switch (op) {
            case '+': nums.push(a + b); break;
            case '-': nums.push(a - b); break;
            case '*': nums.push(a * b); break;
            case '/': nums.push(Math.floor(a / b)); break;
        }
    }

    for (let i = 0; i < expression.length; i++) {
        const ch = expression[i];
        if (ch === ' ') continue;
        if (/\d/.test(ch)) {
            let num = 0;
            while (i < expression.length && /\d/.test(expression[i])) {
                num = num * 10 + parseInt(expression[i++]);
            }
            i--;
            nums.push(num);
        } else if (ch === '(') {
            ops.push(ch);
        } else if (ch === ')') {
            while (ops[ops.length - 1] !== '(') applyOp();
            ops.pop();
        } else {
            while (ops.length && precedence[ops[ops.length - 1]] >= precedence[ch]) {
                applyOp();
            }
            ops.push(ch);
        }
    }
    while (ops.length) applyOp();
    return nums[0];
}
// calculate('3 + 5 * 2') → 13
// calculate('( 3 + 5 ) * 2') → 16
```
