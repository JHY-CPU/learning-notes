## 完整上传组件

```javascript
// ========== 拖拽上传完整实现 ==========
class FileUploader {
    constructor(options = {}) {
        this.options = {
            url: '/upload',
            maxSize: 10 * 1024 * 1024, // 10MB
            accept: ['image/*', 'application/pdf'],
            multiple: true,
            maxFiles: 5,
            ...options,
        };
        this.files = [];
    }

    // 绑定拖拽区域
    bindDropZone(element) {
        element.addEventListener('dragover', (e) => {
            e.preventDefault();
            element.classList.add('dragover');
        });
        element.addEventListener('dragleave', () => {
            element.classList.remove('dragover');
        });
        element.addEventListener('drop', (e) => {
            e.preventDefault();
            element.classList.remove('dragover');
            this.addFiles(e.dataTransfer.files);
        });
    }

    // 文件校验
    validate(file) {
        const errors = [];
        if (file.size > this.options.maxSize) {
            errors.push(`文件 ${file.name} 超过最大限制 (${this.formatSize(this.options.maxSize)})`);
        }
        if (this.options.accept.length) {
            const accepted = this.options.accept.some(type => {
                if (type.endsWith('/*')) {
                    return file.type.startsWith(type.replace('/*', '/'));
                }
                return file.type === type;
            });
            if (!accepted) errors.push(`文件 ${file.name} 类型不支持`);
        }
        return errors;
    }

    addFiles(fileList) {
        const newFiles = Array.from(fileList);
        if (this.files.length + newFiles.length > this.options.maxFiles) {
            console.warn(`最多上传 ${this.options.maxFiles} 个文件`);
            return;
        }
        for (const file of newFiles) {
            const errors = this.validate(file);
            if (errors.length === 0) {
                this.files.push({ file, progress: 0, status: 'pending' });
            } else {
                console.error(errors);
            }
        }
    }

    // 上传单个文件
    async upload(fileInfo) {
        const { file } = fileInfo;
        fileInfo.status = 'uploading';

        return new Promise((resolve, reject) => {
            const xhr = new XMLHttpRequest();
            const formData = new FormData();
            formData.append('file', file);

            xhr.upload.onprogress = (e) => {
                if (e.lengthComputable) {
                    fileInfo.progress = Math.round((e.loaded / e.total) * 100);
                    this.options.onProgress?.(fileInfo);
                }
            };

            xhr.onload = () => {
                fileInfo.status = xhr.status === 200 ? 'done' : 'error';
                fileInfo.response = JSON.parse(xhr.responseText);
                resolve(fileInfo);
            };

            xhr.onerror = () => {
                fileInfo.status = 'error';
                reject(new Error('上传失败'));
            };

            xhr.open('POST', this.options.url);
            xhr.send(formData);
        });
    }

    // 并行上传
    async uploadAll() {
        const results = await Promise.allSettled(
            this.files.map(f => this.upload(f))
        );
        return results;
    }

    formatSize(bytes) {
        if (bytes < 1024) return bytes + ' B';
        if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
        return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
    }
}

// ========== 图片预览 ==========
function previewImage(file) {
    return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onload = (e) => resolve(e.target.result);
        reader.readAsDataURL(file);
    });
}

// 使用
const uploader = new FileUploader({
    url: '/api/upload',
    maxSize: 5 * 1024 * 1024,
    onProgress: (info) => console.log(`${info.file.name}: ${info.progress}%`),
});
uploader.bindDropZone(document.getElementById('dropzone'));
```
