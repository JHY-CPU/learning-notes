递归算法实现

// ========== 阶乘 ========== function factorial(n) { if (n <= 1) return 1; return n * factorial(n - 1); } // ========== 斐波那契 ========== function fib(n) { if (n <= 1) return n; return fib(n - 1) + fib(n - 2); } // ========== 记忆化斐波那契 ========== function fibMemo(n, memo = {}) { if (n <= 1) return n; if (memo[n] !== undefined) return memo[n]; memo[n] = fibMemo(n - 1, memo) + fibMemo(n - 2, memo); return memo[n]; } // ========== 汉诺塔 ========== function hanoi(n, from, to, aux) { if (n === 1) { console.log(`移动 ${from} → ${to}`); return; } hanoi(n - 1, from, aux, to); console.log(`移动 ${from} → ${to}`); hanoi(n - 1, aux, to, from); }
