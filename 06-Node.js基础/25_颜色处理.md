```
// ========== HEX → RGB ==========
function hexToRgb(hex) {
    const r = parseInt(hex.slice(1, 3), 16);
    const g = parseInt(hex.slice(3, 5), 16);
    const b = parseInt(hex.slice(5, 7), 16);
    return { r, g, b };
}
hexToRgb('#ff8800'); // { r: 255, g: 136, b: 0 }

// ========== RGB → HEX ==========
function rgbToHex(r, g, b) {
    return '#' + [r, g, b].map(v => v.toString(16).padStart(2, '0')).join('');
}
rgbToHex(255, 136, 0); // '#ff8800'

// ========== RGB → HSL ==========
function rgbToHsl(r, g, b) {
    r /= 255; g /= 255; b /= 255;
    const max = Math.max(r, g, b), min = Math.min(r, g, b);
    let h, s, l = (max + min) / 2;
    if (max === min) return { h: 0, s: 0, l };
    const d = max - min;
    s = l > 0.5 ? d / (2 - max - min) : d / (max + min);
    switch (max) {
        case r: h = ((g - b) / d + (g < b ? 6 : 0)) / 6; break;
        case g: h = ((b - r) / d + 2) / 6; break;
        case b: h = ((r - g) / d + 4) / 6; break;
    }
    return { h: h * 360, s: s * 100, l: l * 100 };
}
```

## HSL 与颜色操作

```javascript
// ========== HSL → RGB ==========
function hslToRgb(h, s, l) {
    s /= 100; l /= 100;
    const c = (1 - Math.abs(2 * l - 1)) * s;
    const x = c * (1 - Math.abs((h / 60) % 2 - 1));
    const m = l - c / 2;
    let r, g, b;

    if (h < 60)       { r = c; g = x; b = 0; }
    else if (h < 120) { r = x; g = c; b = 0; }
    else if (h < 180) { r = 0; g = c; b = x; }
    else if (h < 240) { r = 0; g = x; b = c; }
    else if (h < 300) { r = x; g = 0; b = c; }
    else               { r = c; g = 0; b = x; }

    return {
        r: Math.round((r + m) * 255),
        g: Math.round((g + m) * 255),
        b: Math.round((b + m) * 255),
    };
}

// ========== 颜色亮度计算 ==========
function luminance(r, g, b) {
    const [rs, gs, bs] = [r, g, b].map(c => {
        c = c / 255;
        return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
    });
    return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
}

function contrastRatio(color1, color2) {
    const l1 = luminance(color1.r, color1.g, color1.b);
    const l2 = luminance(color2.r, color2.g, color2.b);
    const lighter = Math.max(l1, l2);
    const darker = Math.min(l1, l2);
    return (lighter + 0.05) / (darker + 0.05);
}

// 对比度 >= 4.5 为 AA 级可读性
const black = { r: 0, g: 0, b: 0 };
const white = { r: 255, g: 255, b: 255 };
console.log(contrastRatio(black, white)); // 21 (最高)

// ========== 自动选择文字颜色 ==========
function textColorForBackground(hexBg) {
    const { r, g, b } = hexToRgb(hexBg);
    return luminance(r, g, b) > 0.179 ? '#000000' : '#ffffff';
}
```

## 颜色渐变与混合

```javascript
// ========== 颜色混合 ==========
function mixColors(hex1, hex2, ratio = 0.5) {
    const c1 = hexToRgb(hex1);
    const c2 = hexToRgb(hex2);
    const r = Math.round(c1.r * (1 - ratio) + c2.r * ratio);
    const g = Math.round(c1.g * (1 - ratio) + c2.g * ratio);
    const b = Math.round(c1.b * (1 - ratio) + c2.b * ratio);
    return rgbToHex(r, g, b);
}

mixColors('#ff0000', '#0000ff', 0.5); // 红+蓝 = 紫色

// ========== 生成渐变色板 ==========
function generateGradient(startHex, endHex, steps) {
    const colors = [];
    for (let i = 0; i < steps; i++) {
        const ratio = i / (steps - 1);
        colors.push(mixColors(startHex, endHex, ratio));
    }
    return colors;
}

generateGradient('#ff0000', '#0000ff', 5);
// ['#ff0000', '#bf003f', '#7f007f', '#3f00bf', '#0000ff']

// ========== 调整亮度 ==========
function lighten(hex, percent) {
    const { r, g, b } = hexToRgb(hex);
    const adjust = (c) => Math.min(255, Math.round(c + (255 - c) * percent / 100));
    return rgbToHex(adjust(r), adjust(g), adjust(b));
}

function darken(hex, percent) {
    const { r, g, b } = hexToRgb(hex);
    const adjust = (c) => Math.max(0, Math.round(c * (1 - percent / 100)));
    return rgbToHex(adjust(r), adjust(g), adjust(b));
}

lighten('#808080', 20); // 更亮的灰色
darken('#808080', 20);  // 更暗的灰色
```

## CSS 颜色解析

```javascript
// ========== 解析 CSS 颜色字符串 ==========
function parseColor(color) {
    color = color.trim().toLowerCase();

    // HEX
    if (color.startsWith('#')) {
        let hex = color.slice(1);
        if (hex.length === 3) hex = hex.split('').map(c => c + c).join('');
        return hexToRgb('#' + hex);
    }

    // RGB/RGBA
    const rgbMatch = color.match(/rgba?\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)/);
    if (rgbMatch) {
        return { r: +rgbMatch[1], g: +rgbMatch[2], b: +rgbMatch[3] };
    }

    // HSL
    const hslMatch = color.match(/hsla?\(\s*(\d+)\s*,\s*(\d+)%\s*,\s*(\d+)%/);
    if (hslMatch) {
        return hslToRgb(+hslMatch[1], +hslMatch[2], +hslMatch[3]);
    }

    throw new Error(`无法解析颜色: ${color}`);
}

parseColor('#ff8800');           // { r: 255, g: 136, b: 0 }
parseColor('rgb(255, 136, 0)');  // { r: 255, g: 136, b: 0 }
parseColor('hsl(32, 100%, 50%)'); // { r: 255, g: 136, b: 0 }
```
