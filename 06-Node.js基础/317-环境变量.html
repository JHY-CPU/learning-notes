dotenv 库、process.env、.env 文件、环境区分 (dev/prod)。

环境变量管理

// ========== process.env ========== // 所有环境变量 console.log(process.env.NODE_ENV); console.log(process.env.PORT); console.log(process.env.DB_URL); // 设置环境变量 (运行时) // $ NODE_ENV=production PORT=3000 node app.js // ========== 环境区分 ========== // 常见环境: // development — 开发 // production — 生产 // test — 测试 // staging — 预发布 const config = { development: { port: 3000, db: 'mongodb://localhost/dev', }, production: { port: process.env.PORT || 8080, db: process.env.DB_URL, }, }; const env = process.env.NODE_ENV || 'development'; const appConfig = config[env]; // ========== .env 文件 ========== // .env — 通用配置 // .env.local — 本地覆盖 (不提交 git) // .env.development — 开发环境 // .env.production — 生产环境 // .env 示例: // PORT=3000 // DB_HOST=localhost // DB_USER=admin // DB_PASS=secret123 // ========== dotenv 加载 ========== require('dotenv').config(); // 加载 .env require('dotenv').config({ path: '.env.local' }); // 指定路径 require('dotenv').config({ path: `.env.${process.env.NODE_ENV}` }); // 访问 const port = process.env.PORT; const dbUrl = `mongodb://${process.env.DB_HOST}:27017/myapp`;
⚠️ .env 文件包含敏感信息，不要提交到 git。
💡 提交 .env.example 作为模板。
💡 Node 18+ 支持 --env-file 参数。
