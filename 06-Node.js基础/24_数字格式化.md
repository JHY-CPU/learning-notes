## 完整中文数字转换

```javascript
// ========== 中文数字完整实现 ==========
function toChineseNumber(n) {
    if (n === 0) return '零';
    const digits = ['零', '一', '二', '三', '四', '五', '六', '七', '八', '九'];
    const units = ['', '十', '百', '千'];
    const bigUnits = ['', '万', '亿', '兆'];

    const numStr = Math.abs(n).toString();
    const len = numStr.length;
    let result = '';

    // 按4位分组 (万、亿)
    const groups = [];
    for (let i = len; i > 0; i -= 4) {
        groups.unshift(numStr.slice(Math.max(0, i - 4), i));
    }

    for (let g = 0; g < groups.length; g++) {
        const group = groups[g];
        let groupStr = '';
        let needZero = false;

        for (let i = 0; i < group.length; i++) {
            const d = parseInt(group[i]);
            const unitIdx = group.length - 1 - i;

            if (d === 0) {
                needZero = true;
            } else {
                if (needZero) {
                    groupStr += '零';
                    needZero = false;
                }
                groupStr += digits[d] + units[unitIdx];
            }
        }

        if (groupStr) {
            result += groupStr + bigUnits[groups.length - 1 - g];
        }
    }

    return (n < 0 ? '负' : '') + result;
}

// toChineseNumber(12345) → '一万二千三百四十五'
// toChineseNumber(100000000) → '一亿'
// toChineseNumber(1002003) → '一百万零两千零三'
```

## 货币与金额格式化

```javascript
// ========== 货币格式化 ==========
function formatCurrency(amount, currency = 'CNY', locale = 'zh-CN') {
    return new Intl.NumberFormat(locale, {
        style: 'currency',
        currency,
        minimumFractionDigits: 2,
        maximumFractionDigits: 2,
    }).format(amount);
}

formatCurrency(1234567.89);        // '¥1,234,567.89'
formatCurrency(1234.5, 'USD', 'en-US'); // '$1,234.50'
formatCurrency(1234.5, 'EUR', 'de-DE'); // '1.234,50 €'

// ========== 中文金额大写 ==========
function toChineseCurrency(amount) {
    const digits = ['零', '壹', '贰', '叁', '肆', '伍', '陆', '柒', '捌', '玖'];
    const units = ['', '拾', '佰', '仟'];
    const bigUnits = ['', '万', '亿'];

    if (amount === 0) return '零元整';

    // 分离整数和小数
    const intPart = Math.floor(amount);
    const decPart = Math.round((amount - intPart) * 100);

    let result = '';

    if (intPart > 0) {
        const str = intPart.toString();
        const groups = [];
        for (let i = str.length; i > 0; i -= 4) {
            groups.unshift(str.slice(Math.max(0, i - 4), i));
        }
        // 简化实现
        result = str.split('').map((d, i) => {
            const unit = units[(str.length - 1 - i) % 4];
            return digits[parseInt(d)] + unit;
        }).join('') + '元';
    }

    const jiao = Math.floor(decPart / 10);
    const fen = decPart % 10;

    if (jiao > 0) result += digits[jiao] + '角';
    if (fen > 0) result += digits[fen] + '分';
    if (jiao === 0 && fen === 0 && intPart > 0) result += '整';
    if (intPart === 0 && (jiao > 0 || fen > 0)) {
        result = (jiao > 0 ? digits[jiao] + '角' : '') + (fen > 0 ? digits[fen] + '分' : '');
    }

    return result;
}

// toChineseCurrency(12345.67) → '壹万贰仟叁佰肆拾伍元陆角柒分'
```

## 数字简写与精度

```javascript
// ========== 数字简写 (1.2K, 3.4M) ==========
function abbreviateNumber(num) {
    if (num < 1000) return num.toString();
    const units = ['', 'K', 'M', 'B', 'T'];
    const tier = Math.floor(Math.log10(Math.abs(num)) / 3);
    const suffix = units[tier] || '';
    const scale = Math.pow(10, tier * 3);
    const scaled = num / scale;
    return scaled.toFixed(1).replace(/\.0$/, '') + suffix;
}

// abbreviateNumber(1234)     → '1.2K'
// abbreviateNumber(1234567)  → '1.2M'
// abbreviateNumber(999)      → '999'

// ========== 浮点精度处理 ==========
// 0.1 + 0.2 !== 0.3 (经典问题)
console.log(0.1 + 0.2); // 0.30000000000000004

// 解决方案1: toFixed
function preciseAdd(a, b, decimals = 10) {
    return parseFloat((a + b).toFixed(decimals));
}
preciseAdd(0.1, 0.2); // 0.3

// 解决方案2: 放大为整数运算
function intAdd(a, b) {
    const multiplier = Math.pow(10, 10);
    return (Math.round(a * multiplier) + Math.round(b * multiplier)) / multiplier;
}

// ========== 数字范围安全处理 ==========
function safeNumber(value, min, max, defaultValue = 0) {
    const num = Number(value);
    if (isNaN(num)) return defaultValue;
    return Math.min(Math.max(num, min), max);
}
safeNumber('abc', 0, 100, 50); // 50
safeNumber(150, 0, 100);       // 100
```
