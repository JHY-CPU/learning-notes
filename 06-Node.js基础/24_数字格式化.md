## 数字格式化方法


```
// ========== toLocaleString (推荐) ==========
(1234567.89).toLocaleString('zh-CN');
// '1,234,567.89'

// ========== Intl.NumberFormat ==========
new Intl.NumberFormat('zh-CN').format(1234567.89);
// '1,234,567.89'

// ========== 正则千分位 ==========
'1234567'.replace(/\B(?=(\d{3})+(?!\d))/g, ',');
// '1,234,567'

// ========== 手动循环 ==========
function formatThousands(n) {
    const parts = n.toString().split('.');
    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ',');
    return parts.join('.');
}
```
