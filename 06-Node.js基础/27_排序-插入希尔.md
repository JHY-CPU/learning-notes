## 插入 & 希尔排序


```
// ========== 插入排序 O(n²) ==========
function insertionSort(arr) {
    const a = [...arr];
    for (let i = 1; i < a.length; i++) {
        const key = a[i];
        let j = i - 1;
        while (j >= 0 && a[j] > key) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = key;
    }
    return a;
}

// ========== 希尔排序 O(n log² n) ==========
function shellSort(arr) {
    const a = [...arr];
    let gap = Math.floor(a.length / 2);
    while (gap > 0) {
        for (let i = gap; i < a.length; i++) {
            const temp = a[i];
            let j = i;
            while (j >= gap && a[j - gap] > temp) {
                a[j] = a[j - gap];
                j -= gap;
            }
            a[j] = temp;
        }
        gap = Math.floor(gap / 2);
    }
    return a;
}
```

## 插入排序优化

```javascript
// ========== 二分插入排序 ==========
// 使用二分查找找到插入位置，减少比较次数
function binaryInsertionSort(arr) {
    const a = [...arr];
    for (let i = 1; i < a.length; i++) {
        const key = a[i];

        // 二分查找插入位置
        let left = 0, right = i;
        while (left < right) {
            const mid = (left + right) >> 1;
            if (a[mid] <= key) left = mid + 1;
            else right = mid;
        }

        // 移动元素
        for (let j = i; j > left; j--) {
            a[j] = a[j - 1];
        }
        a[left] = key;
    }
    return a;
}
// 比较次数: O(n log n) (从 O(n²) 降到 O(n log n))
// 移动次数: 仍为 O(n²) (瓶颈)

// ========== 链表插入排序 ==========
// 插入排序天然适合链表 (不需要移动元素)
class ListNode {
    constructor(val, next = null) {
        this.val = val;
        this.next = next;
    }
}

function insertionSortList(head) {
    const dummy = new ListNode(0);
    let curr = head;

    while (curr) {
        let prev = dummy;
        while (prev.next && prev.next.val <= curr.val) {
            prev = prev.next;
        }
        const next = curr.next;
        curr.next = prev.next;
        prev.next = curr;
        curr = next;
    }
    return dummy.next;
}
```

## 希尔排序间隔序列

```javascript
// ========== Knuth 间隔序列 ==========
// h = 1, 4, 13, 40, 121, 364, ...
// h = (3^k - 1) / 2
// 比简单的 gap/2 序列更高效
function shellSortKnuth(arr) {
    const a = [...arr];
    let gap = 1;
    while (gap < a.length / 3) gap = 3 * gap + 1; // 找到最大间隔

    while (gap >= 1) {
        for (let i = gap; i < a.length; i++) {
            const temp = a[i];
            let j = i;
            while (j >= gap && a[j - gap] > temp) {
                a[j] = a[j - gap];
                j -= gap;
            }
            a[j] = temp;
        }
        gap = Math.floor(gap / 3);
    }
    return a;
}

// ========== Sedgewick 间隔序列 ==========
// 更优的间隔序列
function getSedgewickGap(n) {
    const gaps = [];
    for (let i = 0; ; i++) {
        const gap = i % 2 === 0
            ? 9 * Math.pow(4, i) - 9 * Math.pow(2, i) + 1
            : Math.pow(2, i + 2) * (Math.pow(2, i + 2) - 3) + 1;
        if (gap >= n) break;
        gaps.push(gap);
    }
    return gaps.reverse();
}
```

## 实战：部分有序数据

```javascript
// ========== 插入排序对几乎有序数据的高效性 ==========
// 插入排序对于"几乎有序"的数据可以达到接近 O(n)
// 应用场景:
// 1. 新数据到达时维护有序列表
// 2. 对几乎有序的大数组做小幅修正

function benchmarkSortForNearlySorted(size, disorderRate) {
    // 生成几乎有序的数组
    const arr = Array.from({ length: size }, (_, i) => i);
    // 随机打乱少量元素
    for (let i = 0; i < size * disorderRate; i++) {
        const a = Math.floor(Math.random() * size);
        const b = Math.floor(Math.random() * size);
        [arr[a], arr[b]] = [arr[b], arr[a]];
    }

    const start = performance.now();
    insertionSort(arr);
    console.log(`${size}元素, ${disorderRate*100}%乱序: ${(performance.now()-start).toFixed(2)}ms`);
}

// 插入排序在几乎有序时远快于快排/归并
// 10000元素, 1%乱序: ~1ms (插入排序)
// 10000元素, 1%乱序: ~5ms (快速排序)
```
