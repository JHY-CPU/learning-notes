## 懒加载 API


```
// ========== 原生懒加载 (HTML) ==========
<img src="photo.jpg" loading="lazy" alt="描述">
// loading: "lazy" | "eager" | "auto"

// ========== IntersectionObserver 实现 ==========
const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            const img = entry.target;
            img.src = img.dataset.src;  // 替换真正的 src
            observer.unobserve(img);    // 停止观察
        }
    });
});
images.forEach(img => observer.observe(img));

// ========== 占位方案 ==========
// 低质量占位 (LQIP) — 小模糊图
// 颜色占位 — background-color
// 骨架占位 — 形状轮廓

// ========== 错误处理 ==========
img.onerror = () => {
    img.src = 'fallback.png';
    img.alt = '加载失败';
};
```

## 完整懒加载实现

```javascript
// ========== 完整懒加载类 ==========
class LazyLoader {
    constructor(options = {}) {
        this.options = {
            rootMargin: '100px 0px',    // 提前100px加载
            threshold: 0.01,
            placeholder: 'data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7',
            ...options,
        };
        this.observer = null;
        this.init();
    }

    init() {
        this.observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    this.loadImage(entry.target);
                    this.observer.unobserve(entry.target);
                }
            });
        }, {
            rootMargin: this.options.rootMargin,
            threshold: this.options.threshold,
        });
    }

    loadImage(img) {
        const src = img.dataset.src;
        const srcset = img.dataset.srcset;

        if (srcset) img.srcset = srcset;
        if (src) {
            img.src = src;
            img.onload = () => {
                img.classList.add('loaded');
                img.removeAttribute('data-src');
            };
            img.onerror = () => {
                img.src = this.options.fallback || this.options.placeholder;
                img.classList.add('error');
            };
        }

        // 背景图懒加载
        if (img.dataset.bg) {
            img.style.backgroundImage = `url(${img.dataset.bg})`;
            img.classList.add('loaded');
        }
    }

    observe(container = document) {
        const images = container.querySelectorAll('img[data-src], [data-bg]');
        images.forEach(img => {
            img.src = img.src || this.options.placeholder;
            this.observer.observe(img);
        });
    }

    disconnect() {
        this.observer.disconnect();
    }
}

// ========== 使用 ==========
const lazyLoader = new LazyLoader({
    rootMargin: '200px 0px',
    placeholder: '/images/placeholder.gif',
    fallback: '/images/error.png',
});
lazyLoader.observe();

// ========== 图片预加载 ==========
function preloadImage(src) {
    return new Promise((resolve, reject) => {
        const img = new Image();
        img.onload = () => resolve(img);
        img.onerror = reject;
        img.src = src;
    });
}

// 批量预加载
async function preloadAll(urls) {
    const results = await Promise.allSettled(urls.map(preloadImage));
    return results.map((r, i) => ({
        url: urls[i],
        status: r.status,
        image: r.status === 'fulfilled' ? r.value : null,
    }));
}
```

## 响应式图片加载

```html
<!-- ========== picture 元素 (响应式) ==========
<picture>
    <source media="(min-width: 1200px)" srcset="large.webp" type="image/webp">
    <source media="(min-width: 768px)" srcset="medium.webp" type="image/webp">
    <source media="(min-width: 480px)" srcset="small.webp" type="image/webp">
    <img src="fallback.jpg" alt="响应式图片" loading="lazy">
</picture>

<!-- srcset 和 sizes 属性 -->
<img
    srcset="small.jpg 480w, medium.jpg 800w, large.jpg 1200w"
    sizes="(max-width: 600px) 480px, (max-width: 1000px) 800px, 1200px"
    src="medium.jpg"
    alt="自适应图片"
    loading="lazy"
>
```

// ========== 懒加载性能指标 ==========
// 观察图片加载性能
const observer = new PerformanceObserver((list) => {
    list.getEntries().forEach(entry => {
        if (entry.initiatorType === 'img') {
            console.log(`图片加载: ${entry.name}, 耗时: ${entry.duration.toFixed(0)}ms`);
        }
    });
});
observer.observe({ entryTypes: ['resource'] });
