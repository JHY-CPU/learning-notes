# 错误监控


## 错误监控


window.onerror、window.onunhandledrejection、错误边界、上报策略。


## 错误监控 API


```
// ========== 全局错误捕获 ==========
window.onerror = (msg, url, line, col, error) => {
    console.log('JS 错误:', { msg, url, line, col, error });
    // 上报错误
    reportError({ type: 'js', msg, url, line, col });
    return true; // 阻止默认处理
};

// ========== Promise 错误 ==========
window.addEventListener('unhandledrejection', (event) => {
    console.log('未处理的 Promise 拒绝:', event.reason);
    reportError({ type: 'promise', reason: event.reason });
    event.preventDefault();
});

// ========== 资源加载错误 ==========
window.addEventListener('error', (event) => {
    if (event.target.tagName) {
        // 资源加载错误 (img/script/link)
        console.log('资源加载失败:', event.target.src || event.target.href);
    }
}, true);  // 捕获阶段
```
