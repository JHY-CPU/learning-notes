# 错误监控


## 错误监控


window.onerror、window.onunhandledrejection、错误边界、上报策略。


## 错误监控 API


```
// ========== 全局错误捕获 ==========
window.onerror = (msg, url, line, col, error) => {
    console.log('JS 错误:', { msg, url, line, col, error });
    // 上报错误
    reportError({ type: 'js', msg, url, line, col });
    return true; // 阻止默认处理
};

// ========== Promise 错误 ==========
window.addEventListener('unhandledrejection', (event) => {
    console.log('未处理的 Promise 拒绝:', event.reason);
    reportError({ type: 'promise', reason: event.reason });
    event.preventDefault();
});

// ========== 资源加载错误 ==========
window.addEventListener('error', (event) => {
    if (event.target.tagName) {
        // 资源加载错误 (img/script/link)
        console.log('资源加载失败:', event.target.src || event.target.href);
    }
}, true);  // 捕获阶段
```

## 完整错误监控 SDK

```javascript
// ========== 错误监控 SDK ==========
class ErrorMonitor {
    constructor(options = {}) {
        this.options = {
            url: '/api/errors',
            maxBreadcrumbs: 20,
            sampleRate: 1.0,
            ignoreErrors: [/ResizeObserver loop limit exceeded/],
            ...options,
        };
        this.breadcrumbs = [];
        this.init();
    }

    init() {
        // JS 运行时错误
        window.onerror = (msg, url, line, col, error) => {
            this.captureError({
                type: 'javascript',
                message: msg,
                url, line, col,
                stack: error?.stack,
            });
        };

        // Promise 未处理拒绝
        window.addEventListener('unhandledrejection', (event) => {
            const reason = event.reason;
            this.captureError({
                type: 'unhandledrejection',
                message: reason?.message || String(reason),
                stack: reason?.stack,
            });
            event.preventDefault();
        });

        // 资源加载错误
        window.addEventListener('error', (event) => {
            if (event.target !== window && event.target.tagName) {
                this.captureError({
                    type: 'resource',
                    message: `资源加载失败: ${event.target.tagName}`,
                    url: event.target.src || event.target.href,
                    tagName: event.target.tagName,
                });
            }
        }, true);

        // 面包屑: 用户行为轨迹
        this.instrumentActions();
    }

    // 面包屑: 记录用户操作
    addBreadcrumb(category, message, data = {}) {
        this.breadcrumbs.push({
            timestamp: Date.now(),
            category,
            message,
            data,
        });
        if (this.breadcrumbs.length > this.options.maxBreadcrumbs) {
            this.breadcrumbs.shift();
        }
    }

    instrumentActions() {
        // 点击事件
        document.addEventListener('click', (e) => {
            this.addBreadcrumb('ui.click', '用户点击', {
                target: this.describeElement(e.target),
            });
        }, true);

        // 路由变化 (SPA)
        const originalPushState = history.pushState;
        history.pushState = (...args) => {
            this.addBreadcrumb('navigation', '路由变化', { url: args[2] });
            return originalPushState.apply(history, args);
        };

        // 网络请求
        const originalFetch = window.fetch;
        window.fetch = async (...args) => {
            const url = typeof args[0] === 'string' ? args[0] : args[0].url;
            this.addBreadcrumb('http', 'Fetch 请求', { url });
            try {
                const res = await originalFetch.apply(window, args);
                if (!res.ok) {
                    this.addBreadcrumb('http', 'HTTP 错误', { url, status: res.status });
                }
                return res;
            } catch (err) {
                this.addBreadcrumb('http', 'Fetch 失败', { url, error: err.message });
                throw err;
            }
        };
    }

    captureError(errorInfo) {
        // 采样
        if (Math.random() > this.options.sampleRate) return;

        // 过滤
        if (this.options.ignoreErrors.some(p => p.test(errorInfo.message))) return;

        const errorData = {
            ...errorInfo,
            timestamp: Date.now(),
            url: location.href,
            userAgent: navigator.userAgent,
            breadcrumbs: [...this.breadcrumbs],
            // 页面信息
            viewport: {
                width: window.innerWidth,
                height: window.innerHeight,
            },
        };

        this.report(errorData);
    }

    report(data) {
        // 使用 sendBeacon 确保页面关闭时也能发送
        if (navigator.sendBeacon) {
            navigator.sendBeacon(this.options.url, JSON.stringify(data));
        } else {
            fetch(this.options.url, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data),
                keepalive: true,
            }).catch(() => {});
        }
    }

    describeElement(el) {
        if (!el || el === document) return 'document';
        const parts = [el.tagName.toLowerCase()];
        if (el.id) parts.push('#' + el.id);
        if (el.className) parts.push('.' + el.className.split(' ').join('.'));
        return parts.join('');
    }

    // 手动捕获
    captureException(error) {
        this.captureError({
            type: 'manual',
            message: error.message,
            stack: error.stack,
        });
    }

    captureMessage(message, level = 'info') {
        this.captureError({
            type: 'message',
            message,
            level,
        });
    }
}

// 使用
const monitor = new ErrorMonitor({
    url: 'https://errors.example.com/api/report',
    sampleRate: 0.1, // 10% 采样
});

// 手动捕获
try {
    // ...
} catch (err) {
    monitor.captureException(err);
}
```
