搜索算法

顺序搜索、二分搜索(迭代/递归)、搜索第一个/最后一个、插入位置。

搜索算法实现

// ========== 顺序搜索 O(n) ========== function linearSearch(arr, target) { for (let i = 0; i < arr.length; i++) { if (arr[i] === target) return i; } return -1; } // ========== 二分搜索 O(log n) (已排序) ========== function binarySearch(arr, target) { let left = 0, right = arr.length - 1; while (left <= right) { const mid = left + Math.floor((right - left) / 2); if (arr[mid] === target) return mid; if (arr[mid] < target) left = mid + 1; else right = mid - 1; } return -1; } // ========== 二分搜索 (递归) ========== function binarySearchRec(arr, target, left = 0, right = arr.length - 1) { if (left > right) return -1; const mid = left + Math.floor((right - left) / 2); if (arr[mid] === target) return mid; if (arr[mid] < target) return binarySearchRec(arr, target, mid + 1, right); return binarySearchRec(arr, target, left, mid - 1); } // ========== 搜索第一个/最后一个 ========== function firstOccurrence(arr, target) { /* ... */ } function lastOccurrence(arr, target) { /* ... */ } // ========== 搜索插入位置 ========== function searchInsert(arr, target) { let left = 0, right = arr.length; while (left < right) { const mid = left + Math.floor((right - left) / 2); if (arr[mid] < target) left = mid + 1; else right = mid; } return left; }
