位运算操作符

// ========== 位运算符 ========== // & 按位与 (5 & 3 = 1) // | 按位或 (5 | 3 = 7) // ^ 按位异或 (5 ^ 3 = 6) // ~ 按位取反 (~5 = -6) // << 左移 (5 << 1 = 10) // >> 有符号右移 (5 >> 1 = 2) // >>> 无符号右移 (-5 >>> 1 = 2147483645) // ========== 常用技巧 ========== // 判断奇偶: n & 1 // 取 2 的幂: n & (n - 1) === 0 // 取最低位1: n & -n // 清除最低位1: n & (n - 1) // 设置第k位: n | (1 << k) // 清除第k位: n & ~(1 << k) // 翻转第k位: n ^ (1 << k) // 查询第k位: (n >> k) & 1
