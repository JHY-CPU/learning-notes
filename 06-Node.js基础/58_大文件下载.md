## 大文件下载 API


```
// ========== fetch Stream 下载 ==========
const response = await fetch(url);
const reader = response.body.getReader();
const contentLength = +response.headers.get('Content-Length');

// ========== 进度监听 ==========
let received = 0;
while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    received += value.length;
    const pct = (received / contentLength) * 100;
}

// ========== 断点续传 (Range) ==========
fetch(url, {
    headers: { 'Range': 'bytes=1000-' }  // 从 1000 字节开始
});
// Response: 206 Partial Content
// headers: Content-Range: bytes 1000-1999/2000
```

## 完整下载管理器

```javascript
// ========== 带进度的下载管理器 ==========
class Downloader {
    constructor(options = {}) {
        this.options = {
            chunkSize: 2 * 1024 * 1024, // 2MB
            concurrency: 3,
            ...options,
        };
    }

    // 流式下载带进度
    async downloadWithProgress(url, onProgress) {
        const response = await fetch(url);
        const contentLength = +response.headers.get('Content-Length');
        const reader = response.body.getReader();

        let received = 0;
        const chunks = [];

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;
            chunks.push(value);
            received += value.length;
            if (onProgress && contentLength) {
                onProgress({
                    received,
                    total: contentLength,
                    percent: (received / contentLength * 100).toFixed(1),
                });
            }
        }

        return new Blob(chunks);
    }

    // 分片并发下载
    async downloadParallel(url, fileName) {
        // 1. 获取文件大小
        const headRes = await fetch(url, { method: 'HEAD' });
        const contentLength = +headRes.headers.get('Content-Length');
        const acceptRanges = headRes.headers.get('Accept-Ranges');

        if (!acceptRanges || contentLength < this.options.chunkSize * 2) {
            // 小文件直接下载
            return this.downloadWithProgress(url);
        }

        // 2. 计算分片
        const chunks = [];
        let start = 0;
        while (start < contentLength) {
            const end = Math.min(start + this.options.chunkSize - 1, contentLength - 1);
            chunks.push({ start, end });
            start = end + 1;
        }

        // 3. 并发下载
        const results = new Array(chunks.length);
        const executing = new Set();

        for (let i = 0; i < chunks.length; i++) {
            const { start, end } = chunks[i];
            const promise = fetch(url, {
                headers: { Range: `bytes=${start}-${end}` },
            }).then(async (res) => {
                results[i] = await res.arrayBuffer();
                executing.delete(promise);
            });
            executing.add(promise);

            if (executing.size >= this.options.concurrency) {
                await Promise.race(executing);
            }
        }
        await Promise.all(executing);

        // 4. 合并
        return new Blob(results);
    }

    // 保存到本地
    save(blob, fileName) {
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = fileName;
        document.body.appendChild(a);
        a.click();
        a.remove();
        URL.revokeObjectURL(url);
    }
}

// ========== 断点续传下载 ==========
class ResumableDownloader {
    constructor(url, fileName) {
        this.url = url;
        this.fileName = fileName;
        this.chunks = [];
        this.downloaded = 0;
    }

    async resume() {
        const headers = {};
        if (this.downloaded > 0) {
            headers['Range'] = `bytes=${this.downloaded}-`;
        }

        const response = await fetch(this.url, { headers });
        const reader = response.body.getReader();

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;
            this.chunks.push(value);
            this.downloaded += value.length;
        }

        return new Blob(this.chunks);
    }
}
```
