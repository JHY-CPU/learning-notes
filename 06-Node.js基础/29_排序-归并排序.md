## 归并排序实现


```
// ========== 自顶向下归并 (递归) ==========
function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    return merge(left, right);
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    while (i < left.length && j < right.length) {
        result.push(left[i] <= right[j] ? left[i++] : right[j++]);
    }
    return [...result, ...left.slice(i), ...right.slice(j)];
}

// ========== 自底向上归并 (迭代) ==========
function mergeSortBottomUp(arr) {
    const a = [...arr];
    for (let size = 1; size < a.length; size *= 2) {
        for (let left = 0; left < a.length; left += 2 * size) {
            const mid = Math.min(left + size, a.length);
            const right = Math.min(left + 2 * size, a.length);
            mergeInPlace(a, left, mid, right);
        }
    }
    return a;
}

// ========== 原地归并 ==========
function mergeInPlace(arr, left, mid, right) {
    const leftArr = arr.slice(left, mid);
    const rightArr = arr.slice(mid, right);
    let i = 0, j = 0, k = left;
    while (i < leftArr.length && j < rightArr.length) {
        arr[k++] = leftArr[i] <= rightArr[j] ? leftArr[i++] : rightArr[j++];
    }
    while (i < leftArr.length) arr[k++] = leftArr[i++];
    while (j < rightArr.length) arr[k++] = rightArr[j++];
}
```

## 归并排序应用

```javascript
// ========== 逆序对计数 ==========
// 统计数组中有多少对 (i, j) 满足 i < j 且 arr[i] > arr[j]
function countInversions(arr) {
    if (arr.length <= 1) return { sorted: arr, count: 0 };

    const mid = Math.floor(arr.length / 2);
    const left = countInversions(arr.slice(0, mid));
    const right = countInversions(arr.slice(mid));

    let count = left.count + right.count;
    const merged = [];
    let i = 0, j = 0;

    while (i < left.sorted.length && j < right.sorted.length) {
        if (left.sorted[i] <= right.sorted[j]) {
            merged.push(left.sorted[i++]);
        } else {
            merged.push(right.sorted[j++]);
            count += left.sorted.length - i; // 左边剩余都比右边大
        }
    }

    return {
        sorted: [...merged, ...left.sorted.slice(i), ...right.sorted.slice(j)],
        count
    };
}

// countInversions([2, 4, 1, 3, 5])
// → { sorted: [1,2,3,4,5], count: 3 }
// 逆序对: (2,1), (4,1), (4,3)
```

## 链表排序

```javascript
// ========== 链表归并排序 ==========
// 归并排序是链表排序的最佳选择 (不需要随机访问)
class ListNode {
    constructor(val, next = null) {
        this.val = val;
        this.next = next;
    }
}

function sortList(head) {
    if (!head || !head.next) return head;

    // 快慢指针找中点
    let slow = head, fast = head.next;
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
    }
    const mid = slow.next;
    slow.next = null;

    // 递归排序并合并
    return merge(sortList(head), sortList(mid));
}

function merge(l1, l2) {
    const dummy = new ListNode(0);
    let curr = dummy;
    while (l1 && l2) {
        if (l1.val <= l2.val) {
            curr.next = l1;
            l1 = l1.next;
        } else {
            curr.next = l2;
            l2 = l2.next;
        }
        curr = curr.next;
    }
    curr.next = l1 || l2;
    return dummy.next;
}
```

## 外部排序

```javascript
// ========== 外部归并排序 (处理超大文件) ==========
// 原理: 分块排序 → 写入临时文件 → 多路归并
const fs = require('fs');
const readline = require('readline');

async function externalSort(inputPath, outputPath, chunkSize = 100000) {
    // 阶段1: 分块排序
    const tempFiles = [];
    let chunk = [];

    const input = fs.createReadStream(inputPath);
    const rl = readline.createInterface({ input });

    for await (const line of rl) {
        chunk.push(parseInt(line));
        if (chunk.length >= chunkSize) {
            chunk.sort((a, b) => a - b);
            const tempPath = `temp_${tempFiles.length}.txt`;
            fs.writeFileSync(tempPath, chunk.join('\n'));
            tempFiles.push(tempPath);
            chunk = [];
        }
    }

    // 最后一块
    if (chunk.length > 0) {
        chunk.sort((a, b) => a - b);
        const tempPath = `temp_${tempFiles.length}.txt`;
        fs.writeFileSync(tempPath, chunk.join('\n'));
        tempFiles.push(tempPath);
    }

    // 阶段2: 多路归并 (简化版: 两路归并)
    // ... 归并所有临时文件到输出文件 ...
    // 清理临时文件
    tempFiles.forEach(f => fs.unlinkSync(f));
}
```

## 复杂度与稳定性

```
// ========== 归并排序特点 ==========
// 时间复杂度: 最好/最坏/平均 都是 O(n log n)
// 空间复杂度: O(n) (需要额外数组)
// 稳定性: 稳定 (merge 时 <= 判断保证稳定)
//
// 优势:
// - 稳定排序
// - 最坏情况也是 O(n log n) (不像快排可能 O(n²))
// - 适合链表排序 (不需要随机访问)
// - 适合外部排序 (磁盘上的大数据)
//
// 劣势:
// - 需要 O(n) 额外空间
// - 对小数组常数因子较大
//
// 实际应用:
// - Array.prototype.sort (部分引擎)
// - 外部排序 (数据库、大数据)
// - 链表排序
// - 逆序对计数
```
