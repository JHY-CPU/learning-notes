## 排序-归并排序


分治合并、自顶向下/自底向上、稳定排序、外部排序。


## 归并排序实现


```
// ========== 自顶向下归并 (递归) ==========
function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    return merge(left, right);
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    while (i < left.length && j < right.length) {
        result.push(left[i] <= right[j] ? left[i++] : right[j++]);
    }
    return [...result, ...left.slice(i), ...right.slice(j)];
}

// ========== 自底向上归并 (迭代) ==========
function mergeSortBottomUp(arr) {
    const a = [...arr];
    for (let size = 1; size < a.length; size *= 2) {
        for (let left = 0; left < a.length; left += 2 * size) {
            const mid = Math.min(left + size, a.length);
            const right = Math.min(left + 2 * size, a.length);
            mergeInPlace(a, left, mid, right);
        }
    }
    return a;
}

// ========== 原地归并 ==========
function mergeInPlace(arr, left, mid, right) {
    const leftArr = arr.slice(left, mid);
    const rightArr = arr.slice(mid, right);
    let i = 0, j = 0, k = left;
    while (i < leftArr.length && j < rightArr.length) {
        arr[k++] = leftArr[i] <= rightArr[j] ? leftArr[i++] : rightArr[j++];
    }
    while (i < leftArr.length) arr[k++] = leftArr[i++];
    while (j < rightArr.length) arr[k++] = rightArr[j++];
}
```
