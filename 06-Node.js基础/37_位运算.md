## 位运算操作符


```
// ========== 位运算符 ==========
// &  按位与   (5 & 3 = 1)
// |  按位或   (5 | 3 = 7)
// ^  按位异或 (5 ^ 3 = 6)
// ~  按位取反 (~5 = -6)
// << 左移     (5 << 1 = 10)
// >> 有符号右移 (5 >> 1 = 2)
// >>> 无符号右移 (-5 >>> 1 = 2147483645)

// ========== 常用技巧 ==========
// 判断奇偶: n & 1
// 取 2 的幂: n & (n - 1) === 0
// 取最低位1: n & -n
// 清除最低位1: n & (n - 1)
// 设置第k位: n | (1 << k)
// 清除第k位: n & ~(1 << k)
// 翻转第k位: n ^ (1 << k)
// 查询第k位: (n >> k) & 1
```

## 什么是位运算

位运算是直接对整数的二进制位进行操作，速度极快。在权限控制、状态压缩、算法竞赛中广泛使用。

## 工作原理

所有数据在计算机中以二进制存储，位运算直接操作这些0和1，无需进位运算，CPU执行效率极高。

## 常见应用场景

1. **权限标志**：用每一位表示一种权限，`READ=1(001), WRITE=2(010), EXEC=4(100)`，组合权限用或 `|`，检查权限用与 `&`
2. **状态压缩**：用一个整数表示多种状态的组合，节省内存
3. **奇偶判断**：`n & 1` 比 `n % 2` 快
4. **交换变量**：`a ^= b; b ^= a; a ^= b;` 不需要临时变量
5. **颜色操作**：RGB颜色值的提取和组合

## 性能与注意事项

- 位运算比算术运算快，但在现代JS引擎中差异很小
- JavaScript数字为64位浮点数，位运算会转为32位整数，超过 `2^31 - 1` 会溢出
- 负数使用补码表示，右移 `>>` 保留符号位，`>>>` 用0填充
- 可读性差，建议用常量命名提升可读性

## 权限系统实现

```javascript
// ========== 位运算权限系统 ==========
const Permission = {
    READ:    1,      // 0001
    WRITE:   2,      // 0010
    DELETE:  4,      // 0100
    ADMIN:   8,      // 1000
    ALL:     15,     // 1111
};

class PermissionManager {
    // 添加权限
    static grant(current, perm) {
        return current | perm;
    }

    // 移除权限
    static revoke(current, perm) {
        return current & ~perm;
    }

    // 检查权限
    static has(current, perm) {
        return (current & perm) === perm;
    }

    // 检查是否有任一权限
    static hasAny(current, ...perms) {
        return perms.some(p => (current & p) !== 0);
    }

    // 翻转权限
    static toggle(current, perm) {
        return current ^ perm;
    }

    // 权限列表
    static toList(current) {
        const perms = [];
        if (current & Permission.READ) perms.push('READ');
        if (current & Permission.WRITE) perms.push('WRITE');
        if (current & Permission.DELETE) perms.push('DELETE');
        if (current & Permission.ADMIN) perms.push('ADMIN');
        return perms;
    }
}

// 使用
let userPerm = Permission.READ;
userPerm = PermissionManager.grant(userPerm, Permission.WRITE);
console.log(PermissionManager.has(userPerm, Permission.READ));   // true
console.log(PermissionManager.has(userPerm, Permission.DELETE)); // false
console.log(PermissionManager.toList(userPerm)); // ['READ', 'WRITE']
```

## 状态压缩与算法应用

```javascript
// ========== 位运算统计1的个数 ==========
function countBits(n) {
    let count = 0;
    while (n) {
        n &= n - 1; // 清除最低位的1
        count++;
    }
    return count;
}

// ========== 判断2的幂 ==========
function isPowerOfTwo(n) {
    return n > 0 && (n & (n - 1)) === 0;
}

// ========== 不使用临时变量交换 ==========
let a = 5, b = 3;
a ^= b; b ^= a; a ^= b;
// a = 3, b = 5

// ========== 用位运算表示集合 ==========
// {0,2,3} → 二进制 1101 = 13
function setToBits(set) {
    return set.reduce((mask, i) => mask | (1 << i), 0);
}

function bitsToSet(bits) {
    const set = [];
    let i = 0;
    while (bits) {
        if (bits & 1) set.push(i);
        bits >>= 1;
        i++;
    }
    return set;
}

// 集合操作
const A = setToBits([0, 1, 2]); // 7
const B = setToBits([1, 2, 3]); // 14
const union = A | B;             // 15 (并集)
const inter = A & B;             // 6  (交集)
const diff = A & ~B;             // 1  (差集: A有B没有)

// ========== 状态压缩 DP (旅行商问题) ==========
function tsp(dist) {
    const n = dist.length;
    const fullMask = (1 << n) - 1;
    const dp = Array.from({ length: 1 << n }, () => new Array(n).fill(Infinity));
    dp[1][0] = 0; // 从城市0出发

    for (let mask = 1; mask <= fullMask; mask++) {
        for (let u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            for (let v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;
                const newMask = mask | (1 << v);
                dp[newMask][v] = Math.min(dp[newMask][v], dp[mask][u] + dist[u][v]);
            }
        }
    }

    let result = Infinity;
    for (let u = 1; u < n; u++) {
        result = Math.min(result, dp[fullMask][u] + dist[u][0]);
    }
    return result;
}
```
