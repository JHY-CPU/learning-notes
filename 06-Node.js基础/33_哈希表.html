哈希表

哈希函数(字符串/数字)、冲突处理(链地址/开放地址)、负载因子。

哈希表实现

// ========== 哈希表 (链地址法) ========== class HashTable { constructor(capacity = 32) { this.buckets = new Array(capacity); this.capacity = capacity; this.size = 0; } _hash(key) { // 字符串哈希: djb2 let hash = 5381; for (const ch of String(key)) { hash = (hash * 33) ^ ch.charCodeAt(0); } return hash >>> 0; } set(key, value) { const index = this._hash(key) % this.capacity; if (!this.buckets[index]) this.buckets[index] = []; // 链地址法: 链表存储冲突项 const bucket = this.buckets[index]; for (const entry of bucket) { if (entry.key === key) { entry.value = value; // 更新 return; } } bucket.push({ key, value }); this.size++; } get(key) { const index = this._hash(key) % this.capacity; const bucket = this.buckets[index]; if (!bucket) return undefined; for (const entry of bucket) { if (entry.key === key) return entry.value; } return undefined; } delete(key) { /* 类似实现 */ } }
