// ========== Levenshtein 编辑距离 ==========
function levenshtein(a, b) {
    const m = a.length, n = b.length;
    const dp = Array(m + 1).fill(0).map(() => Array(n + 1).fill(0));
    for (let i = 0; i <= m; i++) dp[i][0] = i;
    for (let j = 0; j <= n; j++) dp[0][j] = j;
    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            const cost = a[i - 1] === b[j - 1] ? 0 : 1;
            dp[i][j] = Math.min(
                dp[i - 1][j] + 1,      // 删除
                dp[i][j - 1] + 1,      // 插入
                dp[i - 1][j - 1] + cost // 替换
            );
        }
    }
    return dp[m][n];
}

// ========== 最长公共子串 ==========
function longestCommonSubstring(a, b) {
    let maxLen = 0, endIdx = 0;
    const dp = Array(a.length + 1).fill(0).map(() => Array(b.length + 1).fill(0));
    for (let i = 1; i <= a.length; i++) {
        for (let j = 1; j <= b.length; j++) {
            if (a[i - 1] === b[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
                if (dp[i][j] > maxLen) { maxLen = dp[i][j]; endIdx = i; }
            }
        }
    }
    return a.slice(endIdx - maxLen, endIdx);
}

// ========== 余弦相似度 ==========
function cosineSimilarity(text1, text2) {
    const words1 = text1.toLowerCase().split(/\s+/);
    const words2 = text2.toLowerCase().split(/\s+/);

    // 构建词频向量
    const allWords = [...new Set([...words1, ...words2])];
    const vec1 = allWords.map(w => words1.filter(x => x === w).length);
    const vec2 = allWords.map(w => words2.filter(x => x === w).length);

    // 计算点积和模
    let dotProduct = 0, norm1 = 0, norm2 = 0;
    for (let i = 0; i < allWords.length; i++) {
        dotProduct += vec1[i] * vec2[i];
        norm1 += vec1[i] ** 2;
        norm2 += vec2[i] ** 2;
    }

    if (norm1 === 0 || norm2 === 0) return 0;
    return dotProduct / (Math.sqrt(norm1) * Math.sqrt(norm2));
}

// cosineSimilarity('I love coding', 'I love programming') → ~0.67

// ========== 简易拼写纠正 ==========
function spellCorrect(word, dictionary) {
    let bestMatch = word;
    let bestDistance = Infinity;

    for (const dictWord of dictionary) {
        const dist = levenshtein(word, dictWord);
        if (dist < bestDistance) {
            bestDistance = dist;
            bestMatch = dictWord;
        }
    }
    return { suggestion: bestMatch, distance: bestDistance };
}

// spellCorrect('helo', ['hello', 'help', 'hero', 'world'])
// → { suggestion: 'hello', distance: 1 }
```

// ========== 中文字符级相似度 ==========
function chineseSimilarity(a, b) {
    // 将中文转为字符数组
    const chars1 = [...a];
    const chars2 = [...b];

    // 编辑距离
    const distance = levenshtein(a, b);
    const maxLen = Math.max(chars1.length, chars2.length);
    return maxLen === 0 ? 1 : 1 - distance / maxLen;
}

// ========== n-gram 相似度 ==========
function ngramSimilarity(a, b, n = 2) {
    function getNgrams(text, n) {
        const grams = new Map();
        for (let i = 0; i <= text.length - n; i++) {
            const gram = text.slice(i, i + n);
            grams.set(gram, (grams.get(gram) || 0) + 1);
        }
        return grams;
    }

    const grams1 = getNgrams(a, n);
    const grams2 = getNgrams(b, n);

    let intersection = 0, union = 0;
    const allGrams = new Set([...grams1.keys(), ...grams2.keys()]);

    for (const gram of allGrams) {
        const c1 = grams1.get(gram) || 0;
        const c2 = grams2.get(gram) || 0;
        intersection += Math.min(c1, c2);
        union += Math.max(c1, c2);
    }

    return union === 0 ? 0 : intersection / union;
}

// ngramSimilarity('今天天气很好', '今天天气不错', 2) → 0.6
```

## 模糊搜索与高亮

```javascript
// ========== 模糊搜索 (带评分) ==========
function fuzzyMatch(query, items) {
    return items
        .map(item => ({
            text: item,
            score: calculateScore(query, item),
        }))
        .filter(r => r.score > 0)
        .sort((a, b) => b.score - a.score);
}

function calculateScore(query, text) {
    const q = query.toLowerCase();
    const t = text.toLowerCase();

    // 精确匹配
    if (t === q) return 100;
    // 前缀匹配
    if (t.startsWith(q)) return 80;
    // 包含匹配
    if (t.includes(q)) return 60;

    // 编辑距离
    const dist = levenshtein(q, t);
    const similarity = 1 - dist / Math.max(q.length, t.length);
    return similarity > 0.5 ? similarity * 50 : 0;
}

// ========== 搜索高亮 ==========
function highlightMatches(text, query) {
    const escaped = query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    return text.replace(new RegExp(escaped, 'gi'), match => `<mark>${match}</mark>`);
}

// highlightMatches('Hello World', 'world')
// → 'Hello <mark>World</mark>'
```
