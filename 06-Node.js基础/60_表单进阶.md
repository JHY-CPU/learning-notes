# 表单进阶


## 表单进阶


input 事件深入、compositionstart/compositionend、IME 输入、自动填充。


## 表单进阶 API


```
// ========== Input 事件 ==========
// input — 值变化 (实时)
// change — 值变化 (失焦后)
// compositionstart — IME 开始输入
// compositionend — IME 输入结束
// beforeinput — 输入前 (可拦截)

// ========== IME 输入处理 ==========
let isComposing = false;
input.addEventListener('compositionstart', () => { isComposing = true; });
input.addEventListener('compositionend', () => { isComposing = false; handleInput(); });
input.addEventListener('input', () => { if (!isComposing) handleInput(); });

// ========== 自动填充检测 ==========
// animationstart → onAutoFillStart
// CSS :auto-fill 伪类

// ========== 输入验证 ==========
input.validity         // ValidityState 对象
input.validationMessage // 验证消息
input.checkValidity()  // 触发验证
input.setCustomValidity(msg) // 自定义错误
```

## 表单验证与自定义控件

```javascript
// ========== 自定义表单验证 ==========
class FormValidator {
    constructor(form) {
        this.form = form;
        this.rules = {};
        this.errors = {};
    }

    addField(name, rules) {
        this.rules[name] = rules;
    }

    validate() {
        this.errors = {};
        let valid = true;

        for (const [name, rules] of Object.entries(this.rules)) {
            const input = this.form.querySelector(`[name="${name}"]`);
            const value = input?.value?.trim() || '';

            for (const rule of rules) {
                if (rule.required && !value) {
                    this.errors[name] = rule.message || `${name}不能为空`;
                    valid = false;
                    break;
                }
                if (value && rule.pattern && !rule.pattern.test(value)) {
                    this.errors[name] = rule.message;
                    valid = false;
                    break;
                }
                if (value && rule.minLength && value.length < rule.minLength) {
                    this.errors[name] = rule.message;
                    valid = false;
                    break;
                }
                if (rule.custom && !rule.custom(value, this.form)) {
                    this.errors[name] = rule.message;
                    valid = false;
                    break;
                }
            }
        }

        this.showErrors();
        return valid;
    }

    showErrors() {
        for (const [name, message] of Object.entries(this.errors)) {
            const input = this.form.querySelector(`[name="${name}"]`);
            const errorEl = input?.parentNode?.querySelector('.error-message');
            if (errorEl) {
                errorEl.textContent = message;
                errorEl.style.display = message ? 'block' : 'none';
            }
            input?.classList.toggle('invalid', !!message);
        }
    }

    // 实时验证
    enableLiveValidation() {
        this.form.addEventListener('input', (e) => {
            const name = e.target.name;
            if (this.rules[name]) {
                // 只验证当前字段
                const oldValue = this.errors[name];
                this.validateField(name);
                if (this.errors[name] !== oldValue) this.showErrors();
            }
        });
    }
}

// ========== 自动填充检测 ==========
function detectAutoFill(input) {
    // CSS 动画检测法
    input.addEventListener('animationstart', (e) => {
        if (e.animationName === 'onAutoFillStart') {
            console.log('检测到自动填充');
            input.classList.add('auto-filled');
        }
    });
}
// CSS: input:-webkit-autofill { animation: onAutoFillStart 0s; }
// @keyframes onAutoFillStart { from {} }

// ========== 表单序列化 ==========
function serializeForm(form) {
    const data = new FormData(form);
    const result = {};
    for (const [key, value] of data.entries()) {
        if (result[key]) {
            result[key] = [].concat(result[key], value);
        } else {
            result[key] = value;
        }
    }
    return result;
}

// ========== 动态表单字段 ==========
function createDynamicFields(container, schema) {
    schema.forEach(field => {
        const wrapper = document.createElement('div');
        wrapper.className = 'form-field';

        if (field.type === 'select') {
            wrapper.innerHTML = `
                <label>${field.label}</label>
                <select name="${field.name}" ${field.required ? 'required' : ''}>
                    ${field.options.map(o => `<option value="${o.value}">${o.label}</option>`).join('')}
                </select>
            `;
        } else if (field.type === 'textarea') {
            wrapper.innerHTML = `
                <label>${field.label}</label>
                <textarea name="${field.name}" rows="${field.rows || 4}"
                    placeholder="${field.placeholder || ''}"></textarea>
            `;
        } else {
            wrapper.innerHTML = `
                <label>${field.label}</label>
                <input type="${field.type || 'text'}" name="${field.name}"
                    placeholder="${field.placeholder || ''}"
                    ${field.required ? 'required' : ''}
                    ${field.pattern ? `pattern="${field.pattern}"` : ''}>
            `;
        }

        container.appendChild(wrapper);
    });
}
```
