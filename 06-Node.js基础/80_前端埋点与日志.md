```
// ========== 手动埋点 ==========
function track(event, data) {
    navigator.sendBeacon('/api/track', JSON.stringify({
        event, data,
        time: Date.now(),
        url: location.href,
        userAgent: navigator.userAgent
    }));
}

document.addEventListener('click', (e) => {
    const path = getEventPath(e.target);

});

// ========== PV/UV 统计 ==========
// PV (Page View): 页面访问次数
// UV (Unique Visitor): 独立访客数
// 页面加载时上报: track('pv', { url: location.href, referrer });

// ========== 性能数据 ==========
const observer = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
        track('performance', { name: entry.name, value: entry.startTime });
    }
});
observer.observe({ entryTypes: ['paint', 'largest-contentful-paint'] });
```

## 完整埋点 SDK

```javascript
// ========== 埋点 SDK ==========
class Tracker {
    constructor(options = {}) {
        this.options = {
            url: '/api/track',
            appId: 'default',
            sampleRate: 1.0,
            batchSize: 10,
            flushInterval: 5000,
            ...options,
        };
        this.queue = [];
        this.sessionId = this.generateSessionId();
        this.userId = null;
        this.timer = null;
        this.init();
    }

    init() {
        // 定时批量上报
        this.timer = setInterval(() => this.flush(), this.options.flushInterval);

        // 页面关闭前上报
        window.addEventListener('beforeunload', () => this.flush());

        // 页面可见性变化
        document.addEventListener('visibilitychange', () => {
            if (document.hidden) this.flush();
        });

        // 自动采集 PV
        this.track('pageview', {
            url: location.href,
            referrer: document.referrer,
            title: document.title,
        });

        // 性能数据
        this.collectPerformance();
    }

    // 核心: 添加事件到队列
    track(event, data = {}) {
        if (Math.random() > this.options.sampleRate) return;

        const entry = {
            event,
            data,
            timestamp: Date.now(),
            sessionId: this.sessionId,
            userId: this.userId,
            appId: this.options.appId,
            url: location.href,
            userAgent: navigator.userAgent,
            screen: `${screen.width}x${screen.height}`,
        };

        this.queue.push(entry);
        if (this.queue.length >= this.options.batchSize) {
            this.flush();
        }
    }

    // 批量上报
    flush() {
        if (this.queue.length === 0) return;

        const data = [...this.queue];
        this.queue = [];

        if (navigator.sendBeacon) {
            navigator.sendBeacon(this.options.url, JSON.stringify({ events: data }));
        } else {
            fetch(this.options.url, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ events: data }),
                keepalive: true,
            }).catch(() => {
                // 失败重新入队
                this.queue.unshift(...data);
            });
        }
    }

    // 设置用户 ID
    setUserId(userId) {
        this.userId = userId;
    }

    // 无痕埋点: 自动采集
    autoTrack() {
        // 点击
        document.addEventListener('click', (e) => {
            this.track('click', {
                target: this.describeElement(e.target),
                text: e.target.textContent?.slice(0, 50),
                x: e.clientX,
                y: e.clientY,
            });
        });

        // 暴露 (元素进入视口)
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    this.track('exposure', {
                        target: this.describeElement(entry.target),
                        dataset: entry.target.dataset,
                    });
                    observer.unobserve(entry.target);
                }
            });
        });
        document.querySelectorAll('[data-track]').forEach(el => observer.observe(el));

        // 表单提交
        document.addEventListener('submit', (e) => {
            this.track('form_submit', {
                form: this.describeElement(e.target),
            });
        });

        // SPA 路由变化
        const originalPushState = history.pushState;
        history.pushState = (...args) => {
            originalPushState.apply(history, args);
            this.track('pageview', { url: args[2] });
        };
        window.addEventListener('popstate', () => {
            this.track('pageview', { url: location.href });
        });
    }

    collectPerformance() {
        window.addEventListener('load', () => {
            setTimeout(() => {
                const nav = performance.getEntriesByType('navigation')[0];
                if (nav) {
                    this.track('performance', {
                        dns: nav.domainLookupEnd - nav.domainLookupStart,
                        tcp: nav.connectEnd - nav.connectStart,
                        ttfb: nav.responseStart - nav.requestStart,
                        domReady: nav.domContentLoadedEventEnd - nav.startTime,
                        load: nav.loadEventEnd - nav.startTime,
                    });
                }
            }, 0);
        });
    }

    describeElement(el) {
        if (!el || el === document) return 'document';
        const parts = [el.tagName.toLowerCase()];
        if (el.id) parts.push('#' + el.id);
        if (el.className && typeof el.className === 'string') {
            parts.push(el.className.trim().split(/\s+/).map(c => '.' + c).join(''));
        }
        return parts.join('');
    }

    generateSessionId() {
        return Date.now().toString(36) + Math.random().toString(36).slice(2, 8);
    }

    destroy() {
        clearInterval(this.timer);
        this.flush();
    }
}

// ========== 使用 ==========
const tracker = new Tracker({
    url: 'https://analytics.example.com/api/track',
    appId: 'my-app',
    sampleRate: 0.5, // 50% 采样
});

tracker.setUserId('user-123');
tracker.autoTrack();

// 手动埋点
tracker.track('button_click', { button: 'buy', productId: '456' });
tracker.track('purchase', { amount: 99.9, currency: 'CNY' });
```

## 埋点策略

- **PV/UV**：每次页面加载上报
- **点击**：按钮、链接、卡片等交互元素
- **曝光**：商品、广告等元素进入视口时上报
- **性能**：TTFB、FCP、LCP 等核心指标
- **错误**：JS 错误、资源加载失败
- **业务**：购买、注册、搜索等关键转化事件

上报策略：
- **sendBeacon**：页面关闭时也能发送，最适合埋点
- **批量上报**：减少请求数，默认5秒或10条一批
- **采样**：高流量站点按比例采样，减少服务器压力
- **离线缓存**：使用 IndexedDB 缓存离线期间的事件，联网后补发
