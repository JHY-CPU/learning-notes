```
// ========== 哈希表 (链地址法) ==========
class HashTable {
    constructor(capacity = 32) {
        this.buckets = new Array(capacity);
        this.capacity = capacity;
        this.size = 0;
    }

    _hash(key) {
        // 字符串哈希: djb2
        let hash = 5381;
        for (const ch of String(key)) {
            hash = (hash * 33) ^ ch.charCodeAt(0);
        }
        return hash >>> 0;
    }

    set(key, value) {
        const index = this._hash(key) % this.capacity;
        if (!this.buckets[index]) this.buckets[index] = [];
        // 链地址法: 链表存储冲突项
        const bucket = this.buckets[index];
        for (const entry of bucket) {
            if (entry.key === key) {
                entry.value = value; // 更新
                return;
            }
        }
        bucket.push({ key, value });
        this.size++;
    }

    get(key) {
        const index = this._hash(key) % this.capacity;
        const bucket = this.buckets[index];
        if (!bucket) return undefined;
        for (const entry of bucket) {
            if (entry.key === key) return entry.value;
        }
        return undefined;
    }

    delete(key) { /* 类似实现 */ }
}
```

## 动态扩容与开放寻址

```javascript
// ========== 带自动扩容的哈希表 ==========
class DynamicHashTable {
    constructor(initialCapacity = 16, loadFactorThreshold = 0.75) {
        this.buckets = new Array(initialCapacity);
        this.capacity = initialCapacity;
        this.size = 0;
        this.loadFactorThreshold = loadFactorThreshold;
    }

    _hash(key) {
        let hash = 0;
        const str = String(key);
        for (let i = 0; i < str.length; i++) {
            hash = ((hash << 5) - hash + str.charCodeAt(i)) | 0;
        }
        return Math.abs(hash);
    }

    _resize() {
        const oldBuckets = this.buckets;
        this.capacity *= 2;
        this.buckets = new Array(this.capacity);
        this.size = 0;
        for (const bucket of oldBuckets) {
            if (bucket) {
                for (const [key, value] of bucket) {
                    this.set(key, value);
                }
            }
        }
    }

    set(key, value) {
        if (this.size / this.capacity >= this.loadFactorThreshold) {
            this._resize();
        }
        const index = this._hash(key) % this.capacity;
        if (!this.buckets[index]) this.buckets[index] = [];
        for (const entry of this.buckets[index]) {
            if (entry[0] === key) { entry[1] = value; return; }
        }
        this.buckets[index].push([key, value]);
        this.size++;
    }

    get(key) {
        const index = this._hash(key) % this.capacity;
        const bucket = this.buckets[index];
        if (!bucket) return undefined;
        for (const [k, v] of bucket) {
            if (k === key) return v;
        }
        return undefined;
    }

    has(key) { return this.get(key) !== undefined; }

    delete(key) {
        const index = this._hash(key) % this.capacity;
        const bucket = this.buckets[index];
        if (!bucket) return false;
        const i = bucket.findIndex(([k]) => k === key);
        if (i === -1) return false;
        bucket.splice(i, 1);
        this.size--;
        return true;
    }
}
```

## LRU 缓存

```javascript
// ========== LRU 缓存 (面试高频) ==========
class LRUCache {
    constructor(capacity) {
        this.capacity = capacity;
        this.cache = new Map(); // Map 保持插入顺序
    }

    get(key) {
        if (!this.cache.has(key)) return -1;
        // 移到最新位置
        const value = this.cache.get(key);
        this.cache.delete(key);
        this.cache.set(key, value);
        return value;
    }

    put(key, value) {
        if (this.cache.has(key)) {
            this.cache.delete(key);
        } else if (this.cache.size >= this.capacity) {
            // 删除最旧的 (Map 的第一个)
            const oldestKey = this.cache.keys().next().value;
            this.cache.delete(oldestKey);
        }
        this.cache.set(key, value);
    }
}

// 使用
const lru = new LRUCache(3);
lru.put('a', 1);
lru.put('b', 2);
lru.put('c', 3);
lru.get('a');      // 访问 a，a 变为最新
lru.put('d', 4);   // 超出容量，删除最旧的 b
// Map 内容: c=3, a=1, d=4
```

## 哈希表应用

```javascript
// ========== 两数之和 ==========
function twoSum(nums, target) {
    const map = new Map();
    for (let i = 0; i < nums.length; i++) {
        const complement = target - nums[i];
        if (map.has(complement)) {
            return [map.get(complement), i];
        }
        map.set(nums[i], i);
    }
    return [];
}
// twoSum([2, 7, 11, 15], 9) → [0, 1]

// ========== 字符频率统计 ==========
function charFrequency(str) {
    const freq = new Map();
    for (const ch of str) {
        freq.set(ch, (freq.get(ch) || 0) + 1);
    }
    return freq;
}

// ========== 分组 (Group By) ==========
function groupBy(arr, keyFn) {
    const groups = new Map();
    for (const item of arr) {
        const key = keyFn(item);
        if (!groups.has(key)) groups.set(key, []);
        groups.get(key).push(item);
    }
    return groups;
}

// 按首字母分组
const words = ['apple', 'banana', 'avocado', 'cherry', 'blueberry'];
groupBy(words, w => w[0]);
// Map { 'a' → ['apple','avocado'], 'b' → ['banana','blueberry'], 'c' → ['cherry'] }
```
