快速排序实现

// ========== 快速排序 (Lomuto 分区) ========== function quickSort(arr, left = 0, right = arr.length - 1) { if (left >= right) return; const pivot = partition(arr, left, right); quickSort(arr, left, pivot - 1); quickSort(arr, pivot + 1, right); return arr; } // Lomuto 分区方案 function partition(arr, left, right) { const pivot = arr[right]; // 选最后一个元素 let i = left; for (let j = left; j < right; j++) { if (arr[j] < pivot) { [arr[i], arr[j]] = [arr[j], arr[i]]; i++; } } [arr[i], arr[right]] = [arr[right], arr[i]]; return i; } // ========== 随机 pivot (避免最坏情况) ========== function randomPartition(arr, left, right) { const rand = left + Math.floor(Math.random() * (right - left + 1)); [arr[rand], arr[right]] = [arr[right], arr[rand]]; return partition(arr, left, right); } // ========== 三路快排 (处理重复元素) ========== function quickSort3Way(arr, left = 0, right = arr.length - 1) { if (left >= right) return; const [lt, gt] = partition3Way(arr, left, right); quickSort3Way(arr, left, lt - 1); quickSort3Way(arr, gt + 1, right); return arr; }
