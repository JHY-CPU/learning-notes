懒加载 API

// ========== 原生懒加载 (HTML) ========== <img src="photo.jpg" loading="lazy" alt="描述"> // loading: "lazy" | "eager" | "auto" // ========== IntersectionObserver 实现 ========== const observer = new IntersectionObserver((entries) => { entries.forEach(entry => { if (entry.isIntersecting) { const img = entry.target; img.src = img.dataset.src; // 替换真正的 src observer.unobserve(img); // 停止观察 } }); }); images.forEach(img => observer.observe(img)); // ========== 占位方案 ========== // 低质量占位 (LQIP) — 小模糊图 // 颜色占位 — background-color // 骨架占位 — 形状轮廓 // ========== 错误处理 ========== img.onerror = () => { img.src = 'fallback.png'; img.alt = '加载失败'; };
