```
// ========== Fisher-Yates 洗牌 (最优) ==========
function shuffle(arr) {
    const a = [...arr];
    for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
    }
    return a;
}

// ========== sort 乱序 (有偏) ==========
[1, 2, 3, 4, 5].sort(() => Math.random() - 0.5);
// ⚠️ 不均匀, 不推荐

// ========== 随机取样 ==========
function sample(arr, n = 1) {
    const shuffled = shuffle(arr);
    return shuffled.slice(0, n);
}

// ========== 加权随机 ==========
function weightedRandom(items, weights) {
    const total = weights.reduce((a, b) => a + b, 0);
    let r = Math.random() * total;
    for (let i = 0; i < items.length; i++) {
        r -= weights[i];
        if (r <= 0) return items[i];
    }
    return items[items.length - 1];
}
```

## 为什么 sort 随机不均匀

```javascript
// ========== sort 随机偏差分析 ==========
// sort(() => Math.random() - 0.5) 的问题:
// 1. 比较函数不一致 (同一对元素可能返回不同结果)
// 2. 不同浏览器排序算法不同，导致概率不均

// 用3个元素验证
const counts = {};
for (let i = 0; i < 100000; i++) {
    const key = [1, 2, 3].sort(() => Math.random() - 0.5).join(',');
    counts[key] = (counts[key] || 0) + 1;
}
console.log(counts);
// { '1,2,3': 25000, '1,3,2': 12500, '2,1,3': 12500, ... }
// 6种排列的概率不相等! (Fisher-Yates 则均匀)
```

## 高级取样算法

```javascript
// ========== 蓄水池抽样 (大数据流中等概率取样) ==========
// 适用于数据量未知或太大无法全部加载的场景
function reservoirSample(stream, k) {
    const reservoir = [];
    let count = 0;

    for (const item of stream) {
        count++;
        if (count <= k) {
            reservoir.push(item);
        } else {
            const r = Math.floor(Math.random() * count);
            if (r < k) {
                reservoir[r] = item;
            }
        }
    }
    return reservoir;
}

// 使用: 从百万行日志文件中随机取100行
const fs = require('fs');
const readline = require('readline');

async function sampleFile(filePath, k) {
    const stream = fs.createReadStream(filePath);
    const rl = readline.createInterface({ input: stream, crlfDelay: Infinity });
    return reservoirSample(rl, k);
}

// ========== 不放回抽样 ==========
function sampleWithoutReplacement(arr, n) {
    const pool = [...arr];
    const result = [];
    for (let i = 0; i < n && pool.length > 0; i++) {
        const idx = Math.floor(Math.random() * pool.length);
        result.push(pool.splice(idx, 1)[0]);
    }
    return result;
}

// ========== 高斯分布随机 ==========
function gaussianRandom(mean = 0, stddev = 1) {
    const u1 = Math.random();
    const u2 = Math.random();
    const z = Math.sqrt(-2 * Math.log(u1)) * Math.cos(2 * Math.PI * u2);
    return mean + z * stddev;
}

// 生成正态分布数组
const normalArray = Array.from({ length: 1000 }, () => gaussianRandom(50, 10));
```

## 实战：抽奖与随机任务分配

```javascript
// ========== 抽奖算法 (不重复中奖) ==========
class Lottery {
    constructor(participants) {
        this.pool = [...participants];
        this.winners = [];
    }

    draw(count = 1) {
        const drawn = [];
        for (let i = 0; i < count && this.pool.length > 0; i++) {
            const idx = Math.floor(Math.random() * this.pool.length);
            const winner = this.pool.splice(idx, 1)[0];
            this.winners.push(winner);
            drawn.push(winner);
        }
        return drawn;
    }

    reset() {
        this.pool = [...this.winners];
        this.winners = [];
    }
}

// ========== 随机任务分配 ==========
function assignTasks(people, tasks) {
    const shuffledPeople = shuffle([...people]);
    const shuffledTasks = shuffle([...tasks]);
    const assignments = {};

    shuffledPeople.forEach((person, i) => {
        assignments[person] = [];
    });

    shuffledTasks.forEach((task, i) => {
        const person = shuffledPeople[i % shuffledPeople.length];
        assignments[person].push(task);
    });

    return assignments;
}

// assignTasks(['张三', '李四', '王五'], ['任务A', '任务B', '任务C', '任务D', '任务E'])
// → { 张三: ['任务C', '任务D'], 李四: ['任务A', '任务E'], 王五: ['任务B'] }
```
