# 实战 - 智能表单 (Smart Form)

## 项目需求与功能分析

智能表单是 Web 应用中用户输入的核心组件。本项目实现实时验证、错误提示、密码强度检测和提交前检查的完整表单方案。

### 核心功能

- 实时字段验证（失焦验证 + 输入时验证）
- 密码强度指示器
- 密码确认匹配
- 字符计数器
- 表单提交前全面验证
- 成功提交动画

<script>
class SmartForm {
  constructor(form) {
    this.form = form;
    this.fields = form.querySelectorAll('input, textarea');
    this.errors = {};

    this.validators = {
      required: val => val.trim() !== '' || '此字段为必填',
      min: (val, n) => val.length >= parseInt(n) || `至少 ${n} 个字符`,
      max: (val, n) => val.length <= parseInt(n) || `最多 ${n} 个字符`,
      email: val => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val) || '请输入有效的邮箱地址',
      phone: val => !val || /^1[3-9]\d{9}$/.test(val) || '请输入有效的手机号',
      alphanumeric: val => /^\w+$/.test(val) || '仅支持字母、数字、下划线',
      match: (val, fieldId) => val === document.getElementById(fieldId).value || '两次输入不一致',
    };

    this.bindEvents();
  }

  bindEvents() {
    this.fields.forEach(field => {
      field.addEventListener('blur', () => this.validateField(field));
      field.addEventListener('input', () => {
        if (this.errors[field.id]) this.validateField(field);
        // 字符计数
        if (field.id === 'bio') {
          document.getElementById('bioCount').textContent = field.value.length;
        }
        // 密码强度
        if (field.id === 'password') this.updateStrength(field.value);
      });
    });

    this.form.addEventListener('submit', e => {
      e.preventDefault();
      if (this.validateAll()) {
        document.getElementById('successOverlay').classList.add('show');
      }
    });

    document.getElementById('resetBtn').addEventListener('click', () => {
      document.getElementById('successOverlay').classList.remove('show');
      this.form.reset();
      this.fields.forEach(f => { f.classList.remove('valid', 'invalid'); });
      document.getElementById('bioCount').textContent = '0';
      document.getElementById('strengthBar').style.width = '0';
      document.getElementById('strengthText').textContent = '';
      document.querySelectorAll('.error-msg').forEach(el => el.textContent = '');
    });
  }

  validateField(field) {
    const rules = (field.dataset.rules || '').split(',').filter(Boolean);
    const value = field.value;
    const errorEl = document.getElementById(field.id + 'Error');

    for (const rule of rules) {
      const [name, param] = rule.split(':');
      const validator = this.validators[name];
      if (!validator) continue;
      const result = validator(value, param);
      if (result !== true) {
        this.errors[field.id] = result;
        field.classList.remove('valid');
        field.classList.add('invalid');
        if (errorEl) errorEl.textContent = result;
        return false;
      }
    }

    delete this.errors[field.id];
    field.classList.remove('invalid');
    if (value.trim()) field.classList.add('valid');
    if (errorEl) errorEl.textContent = '';
    return true;
  }

  validateAll() {
    let valid = true;
    this.fields.forEach(field => {
      if (!this.validateField(field)) valid = false;
    });
    return valid;
  }

  updateStrength(password) {
    let score = 0;
    if (password.length >= 8) score++;
    if (password.length >= 12) score++;
    if (/[A-Z]/.test(password)) score++;
    if (/[0-9]/.test(password)) score++;
    if (/[^A-Za-z0-9]/.test(password)) score++;

    const bar = document.getElementById('strengthBar');
    const text = document.getElementById('strengthText');
    const levels = [
      { width: '0%', color: '#eee', label: '' },
      { width: '20%', color: '#ff6b6b', label: '非常弱' },
      { width: '40%', color: '#ffa502', label: '弱' },
      { width: '60%', color: '#ffc107', label: '一般' },
      { width: '80%', color: '#43e97b', label: '强' },
      { width: '100%', color: '#00b894', label: '非常强' },
    ];
    const level = levels[score];
    bar.style.width = level.width;
    bar.style.background = level.color;
    text.textContent = level.label;
    text.style.color = level.color;
  }
}

new SmartForm(document.getElementById('smartForm'));
</script>

## 核心技术详解

### 实时验证策略

失焦时触发完整验证，输入时仅在已有错误时重新验证。这种策略平衡了用户体验和即时反馈。

### 密码强度检测

综合密码长度、大小写、数字、特殊字符等因素打分，用颜色和文字实时显示强度等级。

## 扩展方向

1. **自定义验证规则**：支持用户定义正则表达式规则
2. **异步验证**：检查用户名是否已注册
3. **表单联动**：字段之间的依赖和联动逻辑
4. **文件上传**：拖拽上传、预览、进度条
5. **自动保存**：草稿自动保存到 localStorage
6. **多步骤表单**：分步骤填写，步骤间验证
7. **国际化**：多语言错误提示
