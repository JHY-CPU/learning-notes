DP 核心概念

// ========== 斐波那契 (DP) ========== function fibDP(n) { const dp = [0, 1]; for (let i = 2; i <= n; i++) { dp[i] = dp[i - 1] + dp[i - 2]; } return dp[n]; } // ========== 爬楼梯 ========== // 每次可以爬 1 或 2 阶 function climbStairs(n) { const dp = [1, 1]; for (let i = 2; i <= n; i++) { dp[i] = dp[i - 1] + dp[i - 2]; } return dp[n]; } // ========== 0/1 背包 ========== function knapsack(weights, values, capacity) { const n = weights.length; const dp = Array(n + 1).fill(0).map(() => Array(capacity + 1).fill(0)); for (let i = 1; i <= n; i++) { for (let w = 1; w <= capacity; w++) { if (weights[i - 1] <= w) { dp[i][w] = Math.max( dp[i - 1][w], dp[i - 1][w - weights[i - 1]] + values[i - 1] ); } else { dp[i][w] = dp[i - 1][w]; } } } return dp[n][capacity]; } // ========== 最长公共子序列 (LCS) ========== function lcs(text1, text2) { const m = text1.length, n = text2.length; const dp = Array(m + 1).fill(0).map(() => Array(n + 1).fill(0)); for (let i = 1; i <= m; i++) { for (let j = 1; j <= n; j++) { if (text1[i - 1] === text2[j - 1]) { dp[i][j] = dp[i - 1][j - 1] + 1; } else { dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]); } } } return dp[m][n]; }
