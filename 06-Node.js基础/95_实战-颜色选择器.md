# 实战 - 颜色选择器 (Color Picker)

## 项目需求与功能分析

颜色选择器是设计工具和表单中常见的组件。本项目实现 HSL 色彩空间的选择器，支持色相、饱和度、亮度调节和颜色值复制。

### 核心功能

- 色相滑块（Hue）
- 饱和度-亮度面板
- 实时颜色预览
- HEX / RGB / HSL 值显示
- 一键复制颜色值
- 历史颜色记录

```html
<style>

  body { font-family: 'Segoe UI', sans-serif; background: #f5f5f5; display: flex; justify-content: center; padding: 40px 20px; }

<script>
class ColorPicker {
  constructor() {
    this.hue = 0;
    this.sat = 100;
    this.bright = 50;
    this.history = [];

    this.preview = document.getElementById('preview');
    this.satBright = document.getElementById('satBright');
    this.sbCursor = document.getElementById('sbCursor');
    this.hueSlider = document.getElementById('hueSlider');
    this.hueThumb = document.getElementById('hueThumb');

    this.bindEvents();
    this.update();
  }

  bindEvents() {
    // 色相滑块
    let draggingHue = false;
    this.hueSlider.addEventListener('mousedown', e => { draggingHue = true; this.setHue(e); });
    document.addEventListener('mousemove', e => { if (draggingHue) this.setHue(e); });
    document.addEventListener('mouseup', () => draggingHue = false);

    // 饱和度-亮度面板
    let draggingSB = false;
    this.satBright.addEventListener('mousedown', e => { draggingSB = true; this.setSB(e); });
    document.addEventListener('mousemove', e => { if (draggingSB) this.setSB(e); });
    document.addEventListener('mouseup', () => { if (draggingSB) { draggingSB = false; this.addHistory(); } });

    // 复制按钮
    document.querySelectorAll('.copy-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const input = document.getElementById(btn.dataset.target);
        navigator.clipboard?.writeText(input.value);
        btn.textContent = '已复制!';
        btn.classList.add('copied');
        setTimeout(() => { btn.textContent = '复制'; btn.classList.remove('copied'); }, 1000);
      });
    });
  }

  setHue(e) {
    const rect = this.hueSlider.getBoundingClientRect();
    const x = Math.max(0, Math.min(1, (e.clientX - rect.left) / rect.width));
    this.hue = Math.round(x * 360);
    this.update();
  }

  setSB(e) {
    const rect = this.satBright.getBoundingClientRect();
    const x = Math.max(0, Math.min(1, (e.clientX - rect.left) / rect.width));
    const y = Math.max(0, Math.min(1, (e.clientY - rect.top) / rect.height));
    this.sat = Math.round(x * 100);
    this.bright = Math.round((1 - y) * 100);
    this.update();
  }

  update() {
    const [r, g, b] = this.hslToRgb(this.hue, this.sat, this.bright);
    const hex = this.rgbToHex(r, g, b);

    this.preview.style.background = `hsl(${this.hue}, ${this.sat}%, ${this.bright}%)`;
    this.satBright.style.background = `hsl(${this.hue}, 100%, 50%)`;
    this.hueThumb.style.left = (this.hue / 360 * 100) + '%';
    this.hueThumb.style.background = `hsl(${this.hue}, 100%, 50%)`;
    this.sbCursor.style.left = this.sat + '%';
    this.sbCursor.style.top = (100 - this.bright) + '%';

    document.getElementById('hexInput').value = hex;
    document.getElementById('rgbInput').value = `rgb(${r},${g},${b})`;
    document.getElementById('hslInput').value = `hsl(${this.hue},${this.sat}%,${this.bright}%)`;
  }

  hslToRgb(h, s, l) {
    s /= 100; l /= 100;
    const c = (1 - Math.abs(2 * l - 1)) * s;
    const x = c * (1 - Math.abs((h / 60) % 2 - 1));
    const m = l - c / 2;
    let r, g, b;
    if (h < 60) { r=c; g=x; b=0; }
    else if (h < 120) { r=x; g=c; b=0; }
    else if (h < 180) { r=0; g=c; b=x; }
    else if (h < 240) { r=0; g=x; b=c; }
    else if (h < 300) { r=x; g=0; b=c; }
    else { r=c; g=0; b=x; }
    return [Math.round((r+m)*255), Math.round((g+m)*255), Math.round((b+m)*255)];
  }

  rgbToHex(r, g, b) {
    return '#' + [r, g, b].map(v => v.toString(16).padStart(2, '0')).join('');
  }

  addHistory() {
    const hex = document.getElementById('hexInput').value;
    if (this.history.includes(hex)) return;
    this.history.unshift(hex);
    if (this.history.length > 10) this.history.pop();
    const container = document.getElementById('historyColors');
    container.innerHTML = this.history.map(h =>
      `<div class="history-color" style="background:${h}" title="${h}"></div>`
    ).join('');
    container.querySelectorAll('.history-color').forEach(el => {
      el.addEventListener('click', () => {
        // 从 hex 解析颜色（简化处理）
        const hex = el.title;
        const r = parseInt(hex.slice(1,3), 16);
        const g = parseInt(hex.slice(3,5), 16);
        const b = parseInt(hex.slice(5,7), 16);
        // 转换为 HSL
        const max = Math.max(r,g,b)/255, min = Math.min(r,g,b)/255;
        const l = (max+min)/2;
        const d = max-min;
        let h = 0;
        if (d !== 0) {
          if (max === r/255) h = ((g-b)/255/d % 6) * 60;
          else if (max === g/255) h = ((b-r)/255/d + 2) * 60;
          else h = ((r-g)/255/d + 4) * 60;
        }
        if (h < 0) h += 360;
        const s = d === 0 ? 0 : d / (1 - Math.abs(2*l-1)) * 100;
        this.hue = Math.round(h);
        this.sat = Math.round(s);
        this.bright = Math.round(l * 100);
        this.update();
      });
    });
  }
}

new ColorPicker();
</script>
</body>
</html>
```

## 核心技术详解

### HSL 色彩空间

HSL（色相、饱和度、亮度）比 RGB 更直观。色相是 0-360 的角度，饱和度和亮度是百分比。

### 饱和度-亮度面板

使用两层渐变（白色到透明 + 透明到黑色）叠加在纯色背景上，实现 SB 面板。

## 扩展方向

1. **Alpha 透明度**：支持 RGBA 和透明度滑块
2. **取色器**：从屏幕任意位置取色（EyeDropper API）
3. **色板预设**：Material Design / Tailwind 预设色
4. **对比度检测**：计算颜色对比度是否符合 WCAG 标准
5. **调色板生成**：互补色、类似色、三色组等方案
6. **导出格式**：支持导出 CSS 变量、SCSS 变量
7. **颜色盲模拟**：模拟色盲用户看到的颜色
