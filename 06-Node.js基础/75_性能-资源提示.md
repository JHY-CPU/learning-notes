## 什么是资源提示

资源提示（Resource Hints）是通过 `<link>` 标签或HTTP头告诉浏览器提前执行某些操作，优化关键资源的加载时机。

## 各类资源提示

| 提示 | 作用 | 使用场景 |
|------|------|---------|
| `preload` | 高优先级预加载当前页关键资源 | 关键CSS/JS/字体 |
| `prefetch` | 低优先级预取下一页可能需要的资源 | 下一页资源 |
| `preconnect` | 提前建立TCP+TLS连接 | 第三方CDN/API域名 |
| `dns-prefetch` | 提前DNS解析 | 多个第三方域名 |
| `modulepreload` | 预加载ES Module及其依赖 | 关键ES模块 |

## preload 示例

```html
<link rel="preload" href="critical.css" as="style">
<link rel="preload" href="font.woff2" as="font" crossorigin>
<link rel="preload" href="hero.webp" as="image">
```

## 使用原则

1. **preload**：只用于首屏渲染必需的关键资源，过多preload反而干扰优先级
2. **prefetch**：用在用户可能跳转的页面，利用空闲带宽
3. **preconnect**：对CDN、API等第三方域名提前连接，减少DNS+TCP+TLS开销
4. **dns-prefetch**：兼容性最好的提示，对所有第三方域名都可以加

## 注意事项

- preload的资源必须是当前页面实际使用的，否则浪费带宽
- 预加载字体需要加 `crossorigin` 属性
- HTTP头方式（`Link: <url>; rel=preload`）比HTML标签更早被浏览器发现

## 动态资源加载

```javascript
// ========== 动态预加载 ==========
function preloadResource(url, as, crossorigin = false) {
    const link = document.createElement('link');
    link.rel = 'preload';
    link.href = url;
    link.as = as;
    if (crossorigin) link.crossOrigin = '';
    document.head.appendChild(link);
}

function preconnect(origin) {
    const link = document.createElement('link');
    link.rel = 'preconnect';
    link.href = origin;
    document.head.appendChild(link);
}

function dnsPrefetch(origin) {
    const link = document.createElement('link');
    link.rel = 'dns-prefetch';
    link.href = origin;
    document.head.appendChild(link);
}

// ========== 路由级预获取 (SPA) ==========
class RoutePrefetcher {
    constructor() {
        this.prefetched = new Set();
    }

    prefetch(routePath) {
        if (this.prefetched.has(routePath)) return;

        // 预取该路由对应的 JS chunk
        const link = document.createElement('link');
        link.rel = 'prefetch';
        link.href = `/js/${routePath}.chunk.js`;
        document.head.appendChild(link);

        this.prefetched.add(routePath);
    }

    // 鼠标悬停时预获取 (比 click 更早)
    bindHoverPrefetch(links) {
        links.forEach(link => {
            link.addEventListener('mouseenter', () => {
                const route = link.getAttribute('data-route');
                if (route) this.prefetch(route);
            }, { once: true });
        });
    }

    // Intersection Observer 可见时预获取
    bindVisibilityPrefetch(links) {
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const route = entry.target.getAttribute('data-route');
                    if (route) this.prefetch(route);
                    observer.unobserve(entry.target);
                }
            });
        }, { rootMargin: '200px' });

        links.forEach(link => observer.observe(link));
    }
}

// ========== Module Preload ==========
// 比 preload 更好，会加载模块依赖链
// <link rel="modulepreload" href="/js/app.mjs">

function modulePreload(url) {
    const link = document.createElement('link');
    link.rel = 'modulepreload';
    link.href = url;
    document.head.appendChild(link);
}
```

## 优先级与加载策略

```javascript
// ========== 按优先级加载资源 ==========
async function loadWithPriority(resources) {
    // 1. 关键资源: 同步加载
    for (const r of resources.filter(r => r.priority === 'critical')) {
        preloadResource(r.url, r.as);
    }

    // 2. 重要资源: requestIdleCallback
    if ('requestIdleCallback' in window) {
        requestIdleCallback(() => {
            for (const r of resources.filter(r => r.priority === 'important')) {
                preloadResource(r.url, r.as);
            }
        });
    }

    // 3. 低优先级: 延迟加载
    setTimeout(() => {
        for (const r of resources.filter(r => r.priority === 'low')) {
            const link = document.createElement('link');
            link.rel = 'prefetch';
            link.href = r.url;
            document.head.appendChild(link);
        }
    }, 3000);
}

// ========== 智能预加载 (根据网速) ==========
function smartPreload(url) {
    const connection = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
    const effectiveType = connection?.effectiveType || '4g';

    if (['slow-2g', '2g'].includes(effectiveType)) {
        // 慢速网络: 不预加载
        return;
    }
    if (effectiveType === '3g') {
        // 3G: 只预加载关键资源
        return;
    }
    // 4G+: 预加载
    preloadResource(url, 'script');
}
```
