冒泡 & 选择排序

// ========== 冒泡排序 O(n²) ========== function bubbleSort(arr) { const a = [...arr]; for (let i = 0; i < a.length; i++) { for (let j = 0; j < a.length - 1 - i; j++) { if (a[j] > a[j + 1]) { [a[j], a[j + 1]] = [a[j + 1], a[j]]; } } } return a; } // ========== 冒泡优化 (提前退出) ========== function bubbleSortOptimized(arr) { const a = [...arr]; for (let i = 0; i < a.length; i++) { let swapped = false; for (let j = 0; j < a.length - 1 - i; j++) { if (a[j] > a[j + 1]) { [a[j], a[j + 1]] = [a[j + 1], a[j]]; swapped = true; } } if (!swapped) break; // 已有序, 提前退出 } return a; } // ========== 选择排序 O(n²) ========== function selectionSort(arr) { const a = [...arr]; for (let i = 0; i < a.length - 1; i++) { let minIdx = i; for (let j = i + 1; j < a.length; j++) { if (a[j] < a[minIdx]) minIdx = j; } if (minIdx !== i) [a[i], a[minIdx]] = [a[minIdx], a[i]]; } return a; }
