栈与队列实现

// ========== 栈 (LIFO) ========== class Stack { constructor() { this.items = []; } push(v) { this.items.push(v); } pop() { return this.items.pop(); } peek() { return this.items[this.items.length - 1]; } get size() { return this.items.length; } get isEmpty() { return this.items.length === 0; } } // ========== 队列 (FIFO) ========== class Queue { constructor() { this.items = []; } enqueue(v) { this.items.push(v); } dequeue() { return this.items.shift(); } front() { return this.items[0]; } get size() { return this.items.length; } get isEmpty() { return this.items.length === 0; } } // ========== 优先级队列 ========== class PriorityQueue { constructor() { this.heap = []; } enqueue(val, priority) { /* 按优先级插入 */ } dequeue() { /* 取出最高优先级 */ } } // ========== 循环队列 ========== class CircularQueue { constructor(capacity) { this.items = new Array(capacity); this.capacity = capacity; this.head = 0; this.tail = 0; this.count = 0; } }
