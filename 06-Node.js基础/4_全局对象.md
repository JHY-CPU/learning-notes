# 全局对象


## 全局对象


global 全局对象、process 核心对象、Buffer、__dirname/__filename。


## Node.js 全局对象


```
// ========== global ==========
// Node.js 的全局对象 (类似浏览器的 window)
global.myVar = 'accessible everywhere';

// ========== process ==========
// 进程信息和控制
process.argv        // 命令行参数
process.env         // 环境变量
process.cwd()       // 当前工作目录
process.pid         // 进程 ID
process.platform    // 平台 (win32/darwin/linux)
process.version     // Node 版本
process.exit(code)  // 退出进程

// ========== Buffer ==========
// 处理二进制数据
Buffer.from('hello')        // 字符串 → Buffer
Buffer.alloc(1024)          // 分配 1KB
Buffer.concat([buf1, buf2]) // 合并

// ========== __dirname / __filename ==========
// __dirname — 当前模块目录
// __filename — 当前模块文件路径
// ⚠️ 仅在 CommonJS 中可用
// ESM 中需要: import.meta.url 转换

// ========== 其他全局对象 ==========
console            // 控制台
setTimeout/clearTimeout
setInterval/clearInterval
setImmediate/clearImmediate  // Node 特有
queueMicrotask
console.time / console.timeEnd
exports / module / require
```

## 全局对象详解

```javascript
// ========== globalThis (跨平台全局对象) ==========
// Node.js: globalThis === global
// 浏览器: globalThis === window
// Web Worker: globalThis === self
globalThis.myGlobalVar = 'shared across all modules';

// ========== URL 和 URLSearchParams ==========
// Node.js 内置，无需引入
const url = new URL('https://example.com/path?name=alice&age=25');
console.log(url.pathname);   // '/path'
console.log(url.searchParams.get('name')); // 'alice'

// ========== TextEncoder / TextDecoder ==========
const encoder = new TextEncoder();
const bytes = encoder.encode('你好');
console.log(bytes); // Uint8Array [228, 189, 160, 229, 165, 189]

const decoder = new TextDecoder();
console.log(decoder.decode(bytes)); // '你好'

// ========== structuredClone (深拷贝) ==========
const original = { a: 1, b: { c: 2 }, d: new Date() };
const cloned = structuredClone(original);
console.log(cloned.b === original.b); // false (独立副本)

// ========== queueMicrotask ==========
// 将函数加入微任务队列
queueMicrotask(() => {
    console.log('微任务执行');
});
console.log('同步代码');
// 输出: 同步代码 → 微任务执行
```

## Console 高级用法

```javascript
// ========== console 高级方法 ==========
console.table([
    { name: 'Alice', age: 25 },
    { name: 'Bob', age: 30 }
]);
// ┌─────────┬───────┬─────┐
// │ (index) │ name  │ age │
// ├─────────┼───────┼─────┤
// │    0    │ Alice │ 25  │
// │    1    │  Bob  │ 30  │
// └─────────┴───────┴─────┘

console.time('loop');
for (let i = 0; i < 1000000; i++) {}
console.timeEnd('loop'); // loop: 2.345ms

console.dir({ nested: { deep: { value: 1 } } }, { depth: 3, colors: true });

console.assert(1 === 2, '断言失败: 1 不等于 2'); // 条件为假时输出

console.count('label'); // label: 1
console.count('label'); // label: 2
console.countReset('label');

console.group('分组');
console.log('组内消息1');
console.log('组内消息2');
console.groupEnd();

console.trace('追踪调用栈');
// Trace: 追踪调用栈
//     at ...
```

## 定时器高级用法

```javascript
// ========== setImmediate vs setTimeout(0) ==========
// setImmediate: check 阶段执行
// setTimeout(0): timers 阶段执行 (最小延迟 1ms)

// ========== 定时器注意事项 ==========
// setTimeout 最小延迟: 即使设为 0，实际延迟至少 1ms
// 嵌套超过 5 层的 setTimeout 最小延迟为 4ms

// ========== AbortController 取消定时器 ==========
const controller = new AbortController();
setTimeout(() => {
    console.log('这条不会执行');
}, 1000, { signal: controller.signal });
controller.abort(); // 取消定时器
```
