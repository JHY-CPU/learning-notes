## 完整摄像头组件

```javascript
// ========== 摄像头拍照组件 ==========
class CameraCapture {
    constructor(videoElement, canvasElement) {
        this.video = videoElement;
        this.canvas = canvasElement;
        this.ctx = canvasElement.getContext('2d');
        this.stream = null;
    }

    async start(facingMode = 'user') {
        try {
            this.stream = await navigator.mediaDevices.getUserMedia({
                video: {
                    facingMode,
                    width: { ideal: 1280 },
                    height: { ideal: 720 },
                },
                audio: false,
            });
            this.video.srcObject = this.stream;
            await this.video.play();
        } catch (err) {
            console.error('无法访问摄像头:', err.message);
            throw err;
        }
    }

    capture(format = 'image/png', quality = 0.9) {
        this.canvas.width = this.video.videoWidth;
        this.canvas.height = this.video.videoHeight;
        this.ctx.drawImage(this.video, 0, 0);
        return this.canvas.toDataURL(format, quality);
    }

    captureBlob(format = 'image/png', quality = 0.9) {
        this.canvas.width = this.video.videoWidth;
        this.canvas.height = this.video.videoHeight;
        this.ctx.drawImage(this.video, 0, 0);
        return new Promise(resolve => this.canvas.toBlob(resolve, format, quality));
    }

    stop() {
        if (this.stream) {
            this.stream.getTracks().forEach(track => track.stop());
            this.video.srcObject = null;
        }
    }

    // 切换前后摄像头
    async switchCamera() {
        const currentFacing = this.stream?.getVideoTracks()[0]?.getSettings()?.facingMode;
        this.stop();
        await this.start(currentFacing === 'user' ? 'environment' : 'user');
    }
}

// ========== 屏幕录制 ==========
class ScreenRecorder {
    constructor() {
        this.stream = null;
        this.recorder = null;
        this.chunks = [];
    }

    async start() {
        this.stream = await navigator.mediaDevices.getDisplayMedia({
            video: { cursor: 'always' },
            audio: true,
        });

        this.recorder = new MediaRecorder(this.stream, {
            mimeType: 'video/webm;codecs=vp9',
        });

        this.recorder.ondataavailable = (e) => {
            if (e.data.size > 0) this.chunks.push(e.data);
        };

        this.recorder.start(1000); // 每秒收集一次

        // 监听停止共享
        this.stream.getVideoTracks()[0].onended = () => this.stop();
    }

    stop() {
        return new Promise((resolve) => {
            this.recorder.onstop = () => {
                const blob = new Blob(this.chunks, { type: 'video/webm' });
                this.chunks = [];
                this.stream?.getTracks().forEach(t => t.stop());
                resolve(blob);
            };
            this.recorder.stop();
        });
    }

    pause() { this.recorder?.pause(); }
    resume() { this.recorder?.resume(); }
}

// ========== 使用示例 ==========
// 拍照
const camera = new CameraCapture(
    document.getElementById('video'),
    document.getElementById('canvas')
);
await camera.start();
const photo = camera.capture(); // base64 图片
camera.stop();

// 录屏
const recorder = new ScreenRecorder();
await recorder.start();
// ... 用户操作 ...
const videoBlob = await recorder.stop();
```

## 媒体设备枚举

```javascript
// ========== 列出所有媒体设备 ==========
async function listMediaDevices() {
    const devices = await navigator.mediaDevices.enumerateDevices();
    return {
        videoInputs: devices.filter(d => d.kind === 'videoinput'),
        audioInputs: devices.filter(d => d.kind === 'audioinput'),
        audioOutputs: devices.filter(d => d.kind === 'audiooutput'),
    };
}

// 监听设备变化 (插拔设备)
navigator.mediaDevices.addEventListener('devicechange', async () => {
    const devices = await listMediaDevices();
    console.log('设备列表已更新:', devices);
});
```
