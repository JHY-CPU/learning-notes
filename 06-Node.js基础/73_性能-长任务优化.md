## 任务分片完整方案

```javascript
// ========== 任务调度器 ==========
class TaskScheduler {
    constructor() {
        this.tasks = [];
        this.running = false;
    }

    addTask(fn, priority = 'normal') {
        this.tasks.push({ fn, priority });
        this.tasks.sort((a, b) => {
            const order = { high: 0, normal: 1, low: 2 };
            return order[a.priority] - order[b.priority];
        });
    }

    async run() {
        this.running = true;
        while (this.tasks.length && this.running) {
            const task = this.tasks.shift();
            try {
                await task.fn();
            } catch (err) {
                console.error('任务执行错误:', err);
            }
            await this.yield();
        }
        this.running = false;
    }

    yield() {
        return new Promise(resolve => {
            if ('scheduler' in window && scheduler.yield) {
                scheduler.yield().then(resolve);
            } else {
                setTimeout(resolve, 0);
            }
        });
    }

    stop() { this.running = false; }
}

// ========== 大数组分片处理 ==========
async function processInChunks(items, processor, chunkSize = 100) {
    const results = [];
    for (let i = 0; i < items.length; i += chunkSize) {
        const chunk = items.slice(i, i + chunkSize);
        results.push(...chunk.map(processor));

        // 让出主线程
        await new Promise(r => setTimeout(r, 0));
    }
    return results;
}

// ========== 空闲执行 ==========
function scheduleIdleTask(task) {
    if ('requestIdleCallback' in window) {
        requestIdleCallback((deadline) => {
            // 在空闲时间内执行
            while (deadline.timeRemaining() > 0 && task.hasMore()) {
                task.next();
            }
            if (task.hasMore()) scheduleIdleTask(task);
        }, { timeout: 2000 }); // 最多等2秒
    } else {
        // 降级: 用 setTimeout
        setTimeout(() => {
            task.next();
            if (task.hasMore()) scheduleIdleTask(task);
        }, 0);
    }
}

// ========== Scheduler API (Chrome 94+) ==========
async function scheduleWithPriority() {
    // 高优先级 — 用户交互相关
    await scheduler.postTask(() => updateUI(), { priority: 'user-blocking' });

    // 中优先级 — 可见内容更新
    await scheduler.postTask(() => renderList(), { priority: 'user-visible' });

    // 低优先级 — 后台任务
    await scheduler.postTask(() => analyticsReport(), { priority: 'background' });
}
```

## 虚拟滚动

```javascript
// ========== 虚拟滚动 (长列表优化) ==========
class VirtualList {
    constructor(container, items, itemHeight = 40) {
        this.container = container;
        this.items = items;
        this.itemHeight = itemHeight;
        this.visibleCount = Math.ceil(container.clientHeight / itemHeight) + 2;
        this.startIndex = 0;

        // 创建滚动容器
        this.container.style.overflow = 'auto';
        this.container.style.position = 'relative';

        // 撑高元素
        this.spacer = document.createElement('div');
        this.spacer.style.height = items.length * itemHeight + 'px';
        this.container.appendChild(this.spacer);

        // 可视区域
        this.viewport = document.createElement('div');
        this.viewport.style.position = 'absolute';
        this.viewport.style.top = '0';
        this.viewport.style.width = '100%';
        this.container.appendChild(this.viewport);

        this.container.addEventListener('scroll', () => this.onScroll());
        this.render();
    }

    onScroll() {
        const scrollTop = this.container.scrollTop;
        const newStart = Math.floor(scrollTop / this.itemHeight);
        if (newStart !== this.startIndex) {
            this.startIndex = newStart;
            this.render();
        }
    }

    render() {
        const end = Math.min(this.startIndex + this.visibleCount, this.items.length);
        this.viewport.style.transform = `translateY(${this.startIndex * this.itemHeight}px)`;
        this.viewport.innerHTML = '';

        for (let i = this.startIndex; i < end; i++) {
            const item = document.createElement('div');
            item.style.height = this.itemHeight + 'px';
            item.textContent = this.items[i];
            this.viewport.appendChild(item);
        }
    }
}
```
