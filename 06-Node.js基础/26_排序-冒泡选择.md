## 冒泡 & 选择排序


```
// ========== 冒泡排序 O(n²) ==========
function bubbleSort(arr) {
    const a = [...arr];
    for (let i = 0; i < a.length; i++) {
        for (let j = 0; j < a.length - 1 - i; j++) {
            if (a[j] > a[j + 1]) {
                [a[j], a[j + 1]] = [a[j + 1], a[j]];
            }
        }
    }
    return a;
}

// ========== 冒泡优化 (提前退出) ==========
function bubbleSortOptimized(arr) {
    const a = [...arr];
    for (let i = 0; i < a.length; i++) {
        let swapped = false;
        for (let j = 0; j < a.length - 1 - i; j++) {
            if (a[j] > a[j + 1]) {
                [a[j], a[j + 1]] = [a[j + 1], a[j]];
                swapped = true;
            }
        }
        if (!swapped) break; // 已有序, 提前退出
    }
    return a;
}

// ========== 选择排序 O(n²) ==========
function selectionSort(arr) {
    const a = [...arr];
    for (let i = 0; i < a.length - 1; i++) {
        let minIdx = i;
        for (let j = i + 1; j < a.length; j++) {
            if (a[j] < a[minIdx]) minIdx = j;
        }
        if (minIdx !== i) [a[i], a[minIdx]] = [a[minIdx], a[i]];
    }
    return a;
}
```

## 算法分析与变体

```javascript
// ========== 冒泡排序变体：双向冒泡 (鸡尾酒排序) ==========
function cocktailSort(arr) {
    const a = [...arr];
    let start = 0, end = a.length - 1;
    let swapped = true;

    while (swapped) {
        swapped = false;

        // 从左到右
        for (let i = start; i < end; i++) {
            if (a[i] > a[i + 1]) {
                [a[i], a[i + 1]] = [a[i + 1], a[i]];
                swapped = true;
            }
        }
        end--;

        // 从右到左
        for (let i = end; i > start; i--) {
            if (a[i] < a[i - 1]) {
                [a[i], a[i - 1]] = [a[i - 1], a[i]];
                swapped = true;
            }
        }
        start++;
    }
    return a;
}

// ========== 选择排序变体：双端选择排序 ==========
function doubleSelectionSort(arr) {
    const a = [...arr];
    let left = 0, right = a.length - 1;

    while (left < right) {
        let minIdx = left, maxIdx = left;
        for (let i = left; i <= right; i++) {
            if (a[i] < a[minIdx]) minIdx = i;
            if (a[i] > a[maxIdx]) maxIdx = i;
        }
        [a[left], a[minIdx]] = [a[minIdx], a[left]];
        // 如果 maxIdx 恰好是 left，需要修正
        if (maxIdx === left) maxIdx = minIdx;
        [a[right], a[maxIdx]] = [a[maxIdx], a[right]];
        left++;
        right--;
    }
    return a;
}
```

## 排序稳定性分析

```
// ========== 排序稳定性 ==========
// 稳定排序: 相等元素的相对顺序不变
// 不稳定排序: 相等元素的相对顺序可能改变

// 冒泡排序: 稳定 (只有 > 时才交换)
// 选择排序: 不稳定 (可能把相等元素移到前面)
// 示例: [(A,3), (B,3), (C,2)]
// 选择排序第一步: 找最小值 C 放到位置 0
// 结果: [(C,2), (B,3), (A,3)] ← A 和 B 顺序变了!

// 为什么稳定性重要?
// 场景: 先按价格排序，再按名称排序
// 如果排序不稳定，第二次排序会打乱第一次的价格排序结果
```

## 实际应用场景

```javascript
// ========== 冒泡排序的适用场景 ==========
// 1. 数据量很小 (< 50)
// 2. 数据几乎有序 (优化版 O(n))
// 3. 教学演示排序原理
// 4. 检测数组是否有序 (一轮无交换 → 已有序)

function isSorted(arr) {
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i] > arr[i + 1]) return false;
    }
    return true;
}

// ========== 选择排序的适用场景 ==========
// 1. 交换成本高的场景 (选择排序交换次数最少: n-1 次)
// 2. 需要找到第 k 小元素 (部分选择排序)
function findKthSmallest(arr, k) {
    const a = [...arr];
    for (let i = 0; i < k; i++) {
        let minIdx = i;
        for (let j = i + 1; j < a.length; j++) {
            if (a[j] < a[minIdx]) minIdx = j;
        }
        [a[i], a[minIdx]] = [a[minIdx], a[i]];
    }
    return a[k - 1];
}

// ========== 复杂度总结 ==========
// 冒泡排序: 最好 O(n), 最坏 O(n²), 平均 O(n²), 空间 O(1), 稳定
// 选择排序: 最好 O(n²), 最坏 O(n²), 平均 O(n²), 空间 O(1), 不稳定
```
