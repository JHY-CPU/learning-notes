# 实战 - 计算器 (Calculator)

## 项目需求与功能分析

计算器是最经典的前端实战项目之一。本项目实现一个支持四则运算、连续运算、键盘输入的完整计算器。

### 核心功能

- 基本四则运算（+、-、x、/）
- 连续运算（不按等号继续下一步运算）
- 小数点支持
- 正负号切换
- 百分比
- 清除（AC）和退格（Backspace）
- 键盘支持

```html
<script>
class Calculator {
  constructor() {
    this.current = '0';
    this.previous = '';
    this.operator = null;
    this.shouldReset = false;
    this.expression = '';
    this.display = document.getElementById('result');
    this.exprDisplay = document.getElementById('expression');
    this.bindEvents();
  }

  bindEvents() {
    document.querySelector('.buttons').addEventListener('click', e => {
      const btn = e.target.closest('.btn');
      if (!btn) return;
      const action = btn.dataset.action;
      if (action === 'num') this.inputNumber(btn.dataset.num);
      if (action === 'op') this.inputOperator(btn.dataset.op);
      if (action === 'eq') this.calculate();
      if (action === 'clear') this.clear();
      if (action === 'sign') this.toggleSign();
      if (action === 'percent') this.percent();
      if (action === 'dot') this.inputDot();
    });

    document.addEventListener('keydown', e => {
      if (e.key >= '0' && e.key <= '9') this.inputNumber(e.key);
      if (e.key === '.') this.inputDot();
      if (['+','-','*','/'].includes(e.key)) this.inputOperator(e.key);
      if (e.key === 'Enter' || e.key === '=') this.calculate();
      if (e.key === 'Escape') this.clear();
      if (e.key === 'Backspace') this.backspace();
      if (e.key === '%') this.percent();
    });
  }

  inputNumber(num) {
    if (this.shouldReset) {
      this.current = num;
      this.shouldReset = false;
    } else {
      this.current = this.current === '0' ? num : this.current + num;
    }
    this.updateDisplay();
  }

  inputDot() {
    if (this.shouldReset) {
      this.current = '0.';
      this.shouldReset = false;
    } else if (!this.current.includes('.')) {
      this.current += '.';
    }
    this.updateDisplay();
  }

  inputOperator(op) {
    if (this.operator && !this.shouldReset) {
      this.calculate();
    }
    this.previous = this.current;
    this.operator = op;
    this.shouldReset = true;
    const opSymbol = { '+': '+', '-': '-', '*': 'x', '/': '/' }[op];
    this.expression = `${this.previous} ${opSymbol}`;
    this.updateDisplay();
  }

  calculate() {
    if (!this.operator || !this.previous) return;
    const a = parseFloat(this.previous);
    const b = parseFloat(this.current);
    let result;
    switch (this.operator) {
      case '+': result = a + b; break;
      case '-': result = a - b; break;
      case '*': result = a * b; break;
      case '/': result = b === 0 ? 'Error' : a / b; break;
    }
    this.expression = `${this.previous} ${this.operator} ${this.current} =`;
    this.current = result === 'Error' ? 'Error' : this.formatNumber(result);
    this.operator = null;
    this.previous = '';
    this.shouldReset = true;
    this.updateDisplay();
  }

  clear() {
    this.current = '0';
    this.previous = '';
    this.operator = null;
    this.shouldReset = false;
    this.expression = '';
    this.updateDisplay();
  }

  toggleSign() {
    if (this.current !== '0') {
      this.current = this.current.startsWith('-') ? this.current.slice(1) : '-' + this.current;
      this.updateDisplay();
    }
  }

  percent() {
    this.current = this.formatNumber(parseFloat(this.current) / 100);
    this.updateDisplay();
  }

  backspace() {
    if (this.shouldReset) return;
    this.current = this.current.length > 1 ? this.current.slice(0, -1) : '0';
    this.updateDisplay();
  }

  formatNumber(num) {
    if (!isFinite(num)) return 'Error';
    const str = String(num);
    if (str.length > 12) return num.toPrecision(8);
    return str;
  }

  updateDisplay() {
    this.display.textContent = this.current;
    this.exprDisplay.textContent = this.expression;
    // 自动调整字号
    const len = this.current.length;
    this.display.style.fontSize = len > 12 ? '24px' : len > 8 ? '32px' : '40px';
  }
}

new Calculator();
</script>
</body>
</html>
```

## 核心技术详解

### 运算逻辑

使用两个变量（`current` 和 `previous`）存储操作数，`operator` 存储运算符。连续运算时自动计算前一步结果。

### 键盘支持

监听 `keydown` 事件，将键盘输入映射到对应按钮操作。
