分层架构

// ========== 三层架构 ========== // Controller — 接收请求, 返回响应 // Service — 业务逻辑 // Repository — 数据访问 // ┌────────────────────────────────────────┐ // │ Controller (路由 + 请求/响应处理) │ // │ ↓ │ // │ Service (业务逻辑) │ // │ ↓ │ // │ Repository / Model (数据访问) │ // └────────────────────────────────────────┘ // ========== 目录结构 ========== // src/ // ├── controllers/ // │ ├── userController.js // │ └── postController.js // ├── services/ // │ ├── userService.js // │ └── postService.js // ├── models/ // │ ├── User.js // │ └── Post.js // ├── routes/ // │ ├── userRoutes.js // │ └── postRoutes.js // ├── middleware/ // │ ├── auth.js // │ ├── validate.js // │ └── errorHandler.js // └── app.js // ========== 路由层 (routes) ========== // routes/userRoutes.js: const router = require('express').Router(); const userController = require('../controllers/userController'); const { authenticate, authorize } = require('../middleware/auth'); const { validate } = require('../middleware/validate'); router.get('/', authenticate, userController.list); router.get('/:id', authenticate, userController.getById); router.post('/', authenticate, authorize('admin'), validate(createUserSchema), userController.create); router.put('/:id', authenticate, authorize('admin'), userController.update); router.delete('/:id', authenticate, authorize('admin'), userController.delete); module.exports = router;

User CRUD 控制器

// ========== 控制器 ========== // controllers/userController.js: const userService = require('../services/userService'); // 列表 (带分页/过滤/排序) exports.list = asyncHandler(async (req, res) => { const { page = 1, limit = 20, sort = '-createdAt', ...filters } = req.query; const result = await userService.list({ page: +page, limit: +limit, sort, filters }); res.paginated(result.data, result.pagination); }); // 详情 exports.getById = asyncHandler(async (req, res) => { const user = await userService.getById(req.params.id); if (!user) throw new NotFoundError('User not found'); res.success(user); }); // 创建 exports.create = asyncHandler(async (req, res) => { const user = await userService.create(req.body); res.success(user, 'User created', 201); }); // 更新 exports.update = asyncHandler(async (req, res) => { const user = await userService.update(req.params.id, req.body); if (!user) throw new NotFoundError('User not found'); res.success(user, 'User updated'); }); // 删除 exports.delete = asyncHandler(async (req, res) => { const user = await userService.delete(req.params.id); if (!user) throw new NotFoundError('User not found'); res.success(null, 'User deleted'); }); // ========== Service 层 ========== // services/userService.js: const User = require('../models/User'); exports.list = async ({ page, limit, sort, filters }) => { const query = {}; const sortObj = {}; // 过滤: ?role=admin&status=active if (filters.role) query.role = filters.role; if (filters.status) query.status = filters.status; if (filters.search) { query.name = { $regex: filters.search, $options: 'i' }; } // 排序: ?sort=-createdAt,name sort.split(',').forEach(field => { if (field.startsWith('-')) { sortObj[field.slice(1)] = -1; } else { sortObj[field] = 1; } }); const skip = (page - 1) * limit; const [data, total] = await Promise.all([ User.find(query).sort(sortObj).skip(skip).limit(limit).lean(), User.countDocuments(query), ]); return { data, pagination: { page, limit, total, totalPages: Math.ceil(total / limit), }, }; };

Service 层完整实现

// ========== User Service ========== // services/userService.js (完整版): const User = require('../models/User'); const { BadRequestError, NotFoundError } = require('../utils/errors'); exports.getById = async (id) => { if (!mongoose.Types.ObjectId.isValid(id)) { throw new BadRequestError('Invalid user ID'); } return User.findById(id).lean(); }; exports.getByEmail = async (email) => { return User.findOne({ email }).lean(); }; exports.create = async (data) => { // 检查重复 const existing = await User.findOne({ email: data.email }); if (existing) { throw new BadRequestError('Email already exists'); } const user = await User.create(data); return user.toObject(); }; exports.update = async (id, data) => { // 不允许更新的字段 delete data.role; delete data.password; // 检查 email 唯一 if (data.email) { const existing = await User.findOne({ email: data.email, _id: { $ne: id } }); if (existing) { throw new BadRequestError('Email already in use'); } } return User.findByIdAndUpdate(id, data, { new: true, // 返回更新后文档 runValidators: true, // 运行验证 }).lean(); }; exports.delete = async (id) => { return User.findByIdAndDelete(id).lean(); }; // ========== 批量操作控制器 ========== exports.bulkCreate = asyncHandler(async (req, res) => { const users = await User.insertMany(req.body.users); res.success(users, 'Users created', 201); }); exports.bulkDelete = asyncHandler(async (req, res) => { const { ids } = req.body; const result = await User.deleteMany({ _id: { $in: ids } }); res.success({ deleted: result.deletedCount }, 'Users deleted'); });

通用 CRUD 工厂

// ========== CRUD 工厂 ========== // 减少重复代码: 为多个资源生成标准 CRUD 控制器 function createCrudController(service, options = {}) { const { listFields = [], createSchema = null, updateSchema = null, populateFields = '', } = options; return { // 列表 list: asyncHandler(async (req, res) => { const { page = 1, limit = 20, sort = '-createdAt', ...filters } = req.query; const result = await service.list({ page: +page, limit: +limit, sort, filters, populateFields, }); res.paginated(result.data, result.pagination); }), // 详情 getById: asyncHandler(async (req, res) => { const item = await service.getById(req.params.id, populateFields); if (!item) throw new NotFoundError(); res.success(item); }), // 创建 create: asyncHandler(async (req, res) => { const item = await service.create(req.body); res.success(item, 'Created', 201); }), // 更新 update: asyncHandler(async (req, res) => { const item = await service.update(req.params.id, req.body); if (!item) throw new NotFoundError(); res.success(item, 'Updated'); }), // 删除 delete: asyncHandler(async (req, res) => { const item = await service.delete(req.params.id); if (!item) throw new NotFoundError(); res.success(null, 'Deleted'); }), }; } // 使用工厂: const userController = createCrudController(userService, { createSchema: createUserSchema, updateSchema: updateUserSchema, populateFields: 'profile settings', }); // 路由: router.get('/', userController.list); router.post('/', validate(createUserSchema), userController.create); // 定制方法: router.post('/bulk', userController.bulkCreate); // 叠加自定义方法
💡 CRUD 要点: Controller-Service-Model 三层分离关注点; 控制器只处理请求/响应; Service 层封装业务逻辑; 验证在路由层用中间件; 错误统一抛出由 errorHandler 处理; CRUD 工厂减少重复代码; 分页/过滤/排序在 service 层实现; 每个函数只做一件事。
