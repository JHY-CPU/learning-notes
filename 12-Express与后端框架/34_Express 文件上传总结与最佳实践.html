// ========== 文件服务抽象 ========== // 统一接口, 切换存储后端 class FileStorage { async save(buffer, filename, mimetype) { throw new Error('Not implemented'); } async delete(key) { throw new Error('Not implemented'); } async getStream(key) { throw new Error('Not implemented'); } async getPresignedUrl(key) { throw new Error('Not implemented'); } } class LocalStorage extends FileStorage { async save(buffer, filename, mimetype) { const filePath = path.join(config.local.uploadDir, filename); } } class S3Storage extends FileStorage { async save(buffer, filename, mimetype) { const key = `uploads/${Date.now()}-${filename}`; await s3.send(new PutObjectCommand({ Bucket: config.s3.bucket, Key: key, Body: buffer, ContentType: mimetype, })); const url = config.s3.cdnUrl ? `${config.s3.cdnUrl}/${key}` : `https://${config.s3.bucket}.s3.amazonaws.com/${key}`; return { url, key }; } async delete(key) { await s3.send(new DeleteObjectCommand({ Bucket: config.s3.bucket, Key: key })); } async getPresignedUrl(key) { const cmd = new GetObjectCommand({ Bucket: config.s3.bucket, Key: key }); return getSignedUrl(s3, cmd, { expiresIn: config.s3.presignedUrlExpiry }); } } // 工厂 function createFileStorage() { switch (config.local.storage) { case 's3': return new S3Storage(); default: return new LocalStorage(); } }

安全最佳实践

// ========== 文件上传安全 ========== // ========== 1. 文件类型验证 ========== function validateFile(file) { // 不要信任 extension! // 检查 MIME type + Magic Bytes const magicBytes = { 'image/jpeg': [0xFF, 0xD8, 0xFF], 'image/png': [0x89, 0x50, 0x4E, 0x47], 'image/gif': [0x47, 0x49, 0x46], 'image/webp': [0x52, 0x49, 0x46, 0x46], // RIFF... 'application/pdf': [0x25, 0x50, 0x44, 0x46], }; const header = file.buffer.slice(0, 4); const signature = magicBytes[file.mimetype]; if (!signature) return false; // 不在白名单 for (let i = 0; i < signature.length; i++) { if (header[i] !== signature[i]) return false; } return true; } // ========== 2. 路径遍历防护 ========== function sanitizeFilename(filename) { // 防止 ../../../etc/passwd const basename = path.basename(filename); // 只保留安全字符 return basename.replace(/[^a-zA-Z0-9._-]/g, '_'); } // ========== 3. 文件大小限制 ========== // multer limits.fileSize // 请求体大小限制: app.use(express.json({ limit: '1mb' })); app.use(express.urlencoded({ extended: true, limit: '1mb' })); // ========== 4. 用户配额 ========== async function checkQuota(userId, fileSize) { const today = new Date(); today.setHours(0, 0, 0, 0); const todayUsage = await File.aggregate([ { $match: { userId, createdAt: { $gte: today } } }, { $group: { _id: null, total: { $sum: '$size' } } }, ]); const dailyQuota = 100 * 1024 * 1024; // 100MB/天 const used = todayUsage[0]?.total || 0; if (used + fileSize > dailyQuota) { throw new Error('Daily upload quota exceeded'); } } // ========== 5. 文件隔离 ========== // 上传目录不在静态文件目录内 // uploads/ 不上传到 git // 文件名不保留用户原始名 (防止特殊字符) // 文件 URL 使用 ID 而不是路径 // ========== 6. 可执行文件扫描 ========== // 禁止上传: .exe .sh .bat .jar .class // 图片重新编码 (去掉隐藏脚本): // sharp(buffer) → 重新编码, 去掉 EXIF 元数据 // ========== 7. 速率限制 ========== const uploadLimiter = rateLimit({ windowMs: 60 * 1000, max: 10, // 每分钟 10 次上传 message: 'Too many uploads', }); app.post('/upload', authenticate, uploadLimiter, upload.single('file'), handler);

const results = await Promise.all(sizes.map(async ({ width, suffix, quality }) => { const processed = await sharp(buffer) .resize(width, undefined, { withoutEnlargement: true }) .webp({ quality }) .toBuffer();
