# 状态持久化

## 一、概念说明

状态持久化将Pinia store的状态保存到localStorage，刷新页面后自动恢复。可用插件`pinia-plugin-persistedstate`或手动实现。

```bash
npm install pinia-plugin-persistedstate
```

```js
// main.js
import piniaPluginPersistedstate from 'pinia-plugin-persistedstate'
pinia.use(piniaPluginPersistedstate)
```

```js
export const useUserStore = defineStore('user', {
  state: () => ({ token: null, user: null }),
  persist: true  // 自动持久化
}, )
```

## 二、具体用法

### 手动实现持久化

```js
export const useAuthStore = defineStore('auth', {
  state: () => ({
    token: localStorage.getItem('token') || null,
    user: JSON.parse(localStorage.getItem('user') || 'null')
  }),
  actions: {
    setAuth(token, user) {
      this.token = token
      this.user = user
      localStorage.setItem('token', token)
      localStorage.setItem('user', JSON.stringify(user))
    },
    logout() {
      this.token = null
      this.user = null
      localStorage.removeItem('token')
      localStorage.removeItem('user')
    }
  }
})
```

### 插件配置

```js
export const useUserStore = defineStore('user', {
  state: () => ({ token: null, preferences: {} }),
  persist: {
    key: 'user-store',
    storage: localStorage,
    paths: ['token'],  // 只持久化token
    pick: ['token']    // Vue 3.4+
  }
})
```

## 三、注意事项与常见陷阱

1. localStorage有容量限制（~5MB），不要存大数据
2. 敏感数据（如密码）不应持久化到localStorage
3. 使用`paths`选项指定需要持久化的字段
4. sessionStorage替代localStorage实现会话级持久化
5. SSR中localStorage不可用，需条件判断

## 四、自定义持久化插件

```js
// plugins/persist.js
export function persistPlugin({ store }) {
  // 初始化时从 localStorage 恢复
  const stored = localStorage.getItem(store.$id)
  if (stored) {
    store.$patch(JSON.parse(stored))
  }

  // 监听变化自动保存
  store.$subscribe((mutation, state) => {
    localStorage.setItem(store.$id, JSON.stringify(state))
  }, { detached: true })
}

// main.js
pinia.use(persistPlugin)
```

### 选择性持久化
```js
export function selectivePersistPlugin(paths) {
  return ({ store }) => {
    const stored = localStorage.getItem(store.$id)
    if (stored) {
      store.$patch(JSON.parse(stored))
    }

    store.$subscribe((mutation, state) => {
      // 只保存指定路径
      const toSave = {}
      paths.forEach(path => {
        toSave[path] = state[path]
      })
      localStorage.setItem(store.$id, JSON.stringify(toSave))
    }, { detached: true })
  }
}

// 使用：只持久化 token
pinia.use(selectivePersistPlugin(['token']))
```

## 五、加密持久化

```js
import { encrypt, decrypt } from '@/utils/crypto'

export function encryptedPersistPlugin({ store }) {
  const stored = localStorage.getItem(store.$id)
  if (stored) {
    try {
      store.$patch(JSON.parse(decrypt(stored)))
    } catch (e) {
      console.warn('Failed to restore store:', e)
    }
  }

  store.$subscribe((mutation, state) => {
    localStorage.setItem(store.$id, encrypt(JSON.stringify(state)))
  }, { detached: true })
}
```
