# 自定义指令全局注册

## 一、概念说明
通过 `app.directive()` 可以**全局注册**自定义指令，使所有组件都能使用。适合在多个组件中共享的通用指令。

## 二、具体用法

### 2.1 全局注册
```js
// main.js
import { createApp } from 'vue'
import App from './App.vue'

const app = createApp(App)

// 全局注册指令
app.directive('focus', {
  mounted: (el) => el.focus()
})

app.directive('color', {
  mounted: (el, binding) => {
    el.style.color = binding.value
  },
  updated: (el, binding) => {
    el.style.color = binding.value
  }
})

app.mount('#app')
```

### 2.2 在组件中使用
```vue
<template>
  <input v-focus />
  <p v-color="'red'">红色文字</p>
</template>
<!-- 无需导入，直接使用 -->
```

### 2.3 批量注册指令
```js
// directives/index.js
export const focusDirective = {
  mounted: (el) => el.focus()
}

export const tooltipDirective = {
  mounted: (el, binding) => {
    el.title = binding.value
  }
}

export const clickOutsideDirective = {
  mounted: (el, binding) => {
    el._clickOutside = (e) => {
      if (!el.contains(e.target)) binding.value()
    }
    document.addEventListener('click', el._clickOutside)
  },
  unmounted: (el) => {
    document.removeEventListener('click', el._clickOutside)
  }
}
```

```js
// main.js
import * as directives from './directives'

Object.entries(directives).forEach(([name, directive]) => {
  app.directive(name, directive)
})
```

### 2.4 获取已注册的指令
```js
// 获取全局指令
const focusDirective = app.directive('focus')
```

## 三、注意事项与常见陷阱
- 全局指令在所有组件中可用，注意命名冲突
- 局部注册（script setup 中定义变量）优先级高于全局注册
- 第三方指令库通常通过插件方式全局注册
- 建议将常用指令集中管理在 `directives/` 目录

## 五、插件方式注册指令

```js
// plugins/directives.js
export default {
  install(app) {
    app.directive('focus', {
      mounted: (el) => el.focus()
    })

    app.directive('tooltip', {
      mounted(el, binding) {
        el._tooltip = document.createElement('div')
        el._tooltip.className = 'tooltip'
        el._tooltip.textContent = binding.value
        el.style.position = 'relative'
        el.appendChild(el._tooltip)
      },
      updated(el, binding) {
        if (el._tooltip) {
          el._tooltip.textContent = binding.value
        }
      },
      unmounted(el) {
        el._tooltip?.remove()
      }
    })

    app.directive('permission', {
      mounted(el, binding, vnode) {
        const userPermissions = vnode.ctx?.app?.config?.globalProperties?.$permissions || []
        if (!userPermissions.includes(binding.value)) {
          el.style.display = 'none'
        }
      }
    })
  }
}

// main.js
import directivePlugin from './plugins/directives'
app.use(directivePlugin)
```

## 六、指令目录结构建议

```
src/
  directives/
    index.js          # 导出所有指令
    focus.js          # 聚焦指令
    clickOutside.js   # 点击外部
    permission.js     # 权限控制
    drag.js           # 拖拽
    tooltip.js        # 工具提示
    loading.js        # 加载状态
```

```js
// directives/index.js
import focus from './focus'
import clickOutside from './clickOutside'
import permission from './permission'

export { focus, clickOutside, permission }

export function registerDirectives(app) {
  app.directive('focus', focus)
  app.directive('click-outside', clickOutside)
  app.directive('permission', permission)
}
```
