# 自定义验证规则

## 一、概念说明

除了内置规则，VeeValidate和Yup都支持自定义验证规则。自定义规则适合项目特定的验证需求。

```vue
<script setup>
import { defineRule } from 'vee-validate'

// 自定义规则：手机号验证
defineRule('phone', (value) => {
  if (!value) return '请输入手机号'
  if (!/^1[3-9]\d{9}$/.test(value)) return '手机号格式不正确'
  return true
})

// 自定义规则：异步验证（检查用户名是否已存在）
defineRule('uniqueUsername', async (value) => {
  const exists = await api.checkUsername(value)
  return exists ? '用户名已被占用' : true
})
</script>

<template>
  <Field name="phone" rules="required|phone" />
  <ErrorMessage name="phone" />

  <Field name="username" rules="required|uniqueUsername" />
  <ErrorMessage name="username" />
</template>
```

## 二、具体用法

### 带参数的规则

```js
defineRule('range', (value, [min, max]) => {
  const num = Number(value)
  if (num < min) return `不能小于${min}`
  if (num > max) return `不能大于${max}`
  return true
})

// 使用
<Field name="age" rules="required|range:1,120" />
```

### Yup自定义规则

```js
import * as yup from 'yup'

const schema = yup.object({
  phone: yup.string()
    .required('手机号必填')
    .test('phone', '手机号格式不正确', (val) => /^1[3-9]\d{9}$/.test(val)),
  username: yup.string()
    .test('unique', '用户名已存在', async (val) => {
      return !(await api.checkUsername(val))
    })
})
```

## 四、组合验证规则

```js
// 统一管理所有自定义规则
// rules/index.js
import { defineRule } from 'vee-validate'

defineRule('phone', (value) => {
  if (!value) return '请输入手机号'
  if (!/^1[3-9]\d{9}$/.test(value)) return '手机号格式不正确'
  return true
})

defineRule('idCard', (value) => {
  if (!value) return '请输入身份证号'
  if (!/^\d{17}[\dXx]$/.test(value)) return '身份证号格式不正确'
  return true
})

defineRule('password_strength', (value) => {
  if (!value) return '请输入密码'
  const hasLetter = /[a-zA-Z]/.test(value)
  const hasNumber = /\d/.test(value)
  const hasSpecial = /[!@#$%^&*]/.test(value)
  const score = [hasLetter, hasNumber, hasSpecial].filter(Boolean).length
  if (score < 2) return '密码需包含字母、数字、特殊字符中的至少两种'
  return true
})

// 异步验证：检查邮箱是否已注册
defineRule('uniqueEmail', async (value) => {
  if (!value) return true  // required规则会处理空值
  try {
    const res = await fetch(`/api/check-email?email=${value}`)
    const { exists } = await res.json()
    return exists ? '该邮箱已被注册' : true
  } catch {
    return true  // 网络错误时放行
  }
})

// 带参数的验证码规则
defineRule('verification_code', (value, [length = 6]) => {
  if (!value) return '请输入验证码'
  if (!new RegExp(`^\\d{${length}}$`).test(value)) {
    return `验证码必须是${length}位数字`
  }
  return true
})
```

## 五、Yup自定义验证进阶

```js
import * as yup from 'yup'

// 自定义Yup方法
yup.addMethod(yup.string, 'phone', function (message) {
  return this.test('phone', message || '手机号格式不正确', (val) => {
    if (!val) return true
    return /^1[3-9]\d{9}$/.test(val)
  })
})

// 使用
const schema = yup.object({
  phone: yup.string().phone().required(),
  email: yup.string().email().test(
    'not-common',
    '不能使用公共邮箱',
    (val) => !val || !['qq.com', '163.com'].some(d => val.endsWith(d))
  )
})
```

## 六、错误消息国际化

```js
import { configure } from 'vee-validate'

configure({
  generateMessage: (ctx) => {
    const i18n = {
      zh: {
        required: `${ctx.field}不能为空`,
        email: '邮箱格式不正确',
        min: `${ctx.field}至少需要${ctx.rule.params[0]}个字符`
      },
      en: {
        required: `${ctx.field} is required`,
        email: 'Invalid email format',
        min: `${ctx.field} must be at least ${ctx.rule.params[0]} characters`
      }
    }
    const lang = navigator.language.startsWith('zh') ? 'zh' : 'en'
    return i18n[lang][ctx.rule.name] || `${ctx.field} validation failed`
  }
})
```

## 三、注意事项与常见陷阱

1. 规则函数返回`true`表示通过，返回字符串表示错误消息
2. 异步规则返回Promise，验证时自动等待
3. 带参数的规则通过规则字符串传递参数：`range:1,100`
4. 避免过于复杂的验证逻辑（影响性能）
5. 复用规则建议统一管理在`rules/`目录
6. 异步验证要考虑网络失败的情况（降级放行或阻止）
7. 国际化场景中验证消息应从语言包获取
