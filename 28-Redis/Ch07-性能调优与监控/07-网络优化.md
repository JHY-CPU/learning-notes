# 网络优化

## 一、概念说明

网络是Redis性能的重要因素。优化网络配置和减少网络往返可以显著提升性能。

## 二、优化策略

### Pipeline批量操作

```bash
# 减少RTT
cat commands.txt | redis-cli --pipe

# Python
pipe = r.pipeline()
for i in range(1000):
    pipe.set(f"key:{i}", f"value:{i}")
pipe.execute()
```

### TCP配置

```bash
# redis.conf
tcp-backlog 511          # TCP连接队列
tcp-keepalive 300        # TCP保活
timeout 0                # 客户端超时

# 系统参数
sysctl -w net.core.somaxconn=65535
sysctl -w net.ipv4.tcp_max_syn_backlog=65535
```

## 七、连接优化最佳实践

```python
import redis
import time

class OptimizedRedisClient:
    """优化的Redis客户端"""
    
    def __init__(self, host='localhost', port=6379):
        # 使用连接池
        self.pool = redis.ConnectionPool(
            host=host,
            port=port,
            max_connections=50,
            socket_timeout=5,
            socket_connect_timeout=5,
            socket_keepalive=True,
            retry_on_timeout=True,
            decode_responses=True
        )
        self.r = redis.Redis(connection_pool=self.pool)
    
    def batch_get(self, keys, batch_size=100):
        """批量获取（分批）"""
        results = {}
        for i in range(0, len(keys), batch_size):
            batch = keys[i:i+batch_size]
            values = self.r.mget(batch)
            for key, value in zip(batch, values):
                if value:
                    results[key] = value
        return results
    
    def batch_set(self, data, batch_size=100):
        """批量设置（分批）"""
        items = list(data.items())
        for i in range(0, len(items), batch_size):
            batch = dict(items[i:i+batch_size])
            pipe = self.r.pipeline()
            for key, value in batch.items():
                pipe.set(key, value)
            pipe.execute()
    
    def pipeline_get(self, keys):
        """Pipeline获取"""
        pipe = self.r.pipeline()
        for key in keys:
            pipe.get(key)
        results = pipe.execute()
        return dict(zip(keys, results))

# 使用
client = OptimizedRedisClient()
values = client.batch_get(['key1', 'key2', 'key3'])
```

## 八、网络监控

```python
def monitor_network():
    """监控Redis网络"""
    r = redis.Redis()
    info = r.info()
    
    print("网络监控")
    print("=" * 40)
    print(f"连接数: {info['connected_clients']}")
    print(f"输入字节: {info['total_net_input_bytes'] / 1024 / 1024:.1f} MB")
    print(f"输出字节: {info['total_net_output_bytes'] / 1024 / 1024:.1f} MB")
    print(f"输入Kbps: {info.get('instantaneous_input_kbps', 0):.1f}")
    print(f"输出Kbps: {info.get('instantaneous_output_kbps', 0):.1f}")
    print(f"拒绝连接: {info['rejected_connections']}")
    
    # 检查连接使用率
    maxclients = info['maxclients']
    connected = info['connected_clients']
    usage = connected / maxclients * 100
    
    if usage > 80:
        print(f"\n警告: 连接使用率 {usage:.1f}%")

monitor_network()
```
