# 分布式锁基础

## 一、概念说明

分布式锁用于在分布式系统中控制对共享资源的并发访问。Redis通过SET NX EX命令实现简单高效的分布式锁。

## 二、具体用法

### 基本实现

```bash
# 获取锁
SET lock:order:1001 "uuid-value" NX EX 30
# NX: 不存在时才设置
# EX 30: 30秒自动过期
# 返回OK表示获取成功
# 返回(nil)表示获取失败

# 释放锁（原子性）
EVAL "
if redis.call('GET', KEYS[1]) == ARGV[1] then
    return redis.call('DEL', KEYS[1])
else
    return 0
end
" 1 lock:order:1001 "uuid-value"
```

### Python实现

```python
import redis
import uuid

r = redis.Redis()

def acquire_lock(key, timeout=30):
    uid = str(uuid.uuid4())
    result = r.set(key, uid, nx=True, ex=timeout)
    if result:
        return uid
    return None

def release_lock(key, uid):
    lua = """
    if redis.call('get', KEYS[1]) == ARGV[1] then
        return redis.call('del', KEYS[1])
    else
        return 0
    end
    """
    return r.eval(lua, 1, key, uid)

# 使用
lock_key = "lock:resource:1001"
uid = acquire_lock(lock_key)
if uid:
    try:
        # 临界区操作
        print("获取锁成功")
    finally:
        release_lock(lock_key, uid)
```

## 三、关键要素

```bash
# 1. 原子性获取
SET key value NX EX timeout

# 2. 唯一标识
# 使用UUID防止误解锁

# 3. 自动过期
# 防止死锁

# 4. 原子性释放
# Lua脚本保证
```

## 四、注意事项

1. **必须设置过期时间**：防止死锁
2. **使用UUID**：防止误解锁
3. **Lua脚本释放**：保证原子性
4. **锁粒度**：不要使用大粒度锁
5. **重试机制**：获取失败时的处理

## 五、锁续期（看门狗机制）

```python
import redis
import time
import threading
from uuid import uuid4

r = redis.Redis()

class RedisLock:
    def __init__(self, key, expire=30, retry_interval=0.1):
        self.key = f"lock:{key}"
        self.expire = expire
        self.retry_interval = retry_interval
        self.token = str(uuid4())
        self._stop_event = threading.Event()
        self._watchdog_thread = None
    
    def acquire(self, timeout=10):
        """获取锁"""
        end_time = time.time() + timeout
        while time.time() < end_time:
            if r.set(self.key, self.token, nx=True, ex=self.expire):
                # 启动看门狗线程
                self._start_watchdog()
                return True
            time.sleep(self.retry_interval)
        return False
    
    def _start_watchdog(self):
        """启动看门狗，自动续期"""
        def watchdog():
            # 每过期时间的1/3续期一次
            interval = self.expire / 3
            while not self._stop_event.is_set():
                time.sleep(interval)
                # 检查锁是否还属于自己
                if r.get(self.key) == self.token.encode():
                    r.expire(self.key, self.expire)
                else:
                    break
        
        self._watchdog_thread = threading.Thread(target=watchdog, daemon=True)
        self._watchdog_thread.start()
    
    def release(self):
        """释放锁"""
        self._stop_event.set()
        lua = """
        if redis.call('get', KEYS[1]) == ARGV[1] then
            return redis.call('del', KEYS[1])
        else
            return 0
        end
        """
        return r.eval(lua, 1, self.key, self.token)

# 使用示例
lock = RedisLock("order:1001", expire=30)
if lock.acquire(timeout=5):
    try:
        print("获取锁成功，执行业务逻辑")
        time.sleep(60)  # 业务处理时间可能超过锁过期时间
    finally:
        lock.release()
```

## 六、可重入锁实现

```python
class ReentrantLock:
    def __init__(self, key, expire=30):
        self.key = f"lock:{key}"
        self.expire = expire
        self.token = str(uuid4())
        self.local_count = 0
        self._local = threading.local()
    
    def acquire(self):
        """获取可重入锁"""
        # 检查是否已持有锁
        if r.get(self.key) == self.token.encode():
            self.local_count += 1
            return True
        
        # 尝试获取锁
        if r.set(self.key, self.token, nx=True, ex=self.expire):
            self.local_count = 1
            return True
        
        return False
    
    def release(self):
        """释放可重入锁"""
        if self.local_count > 0:
            self.local_count -= 1
            if self.local_count == 0:
                lua = """
                if redis.call('get', KEYS[1]) == ARGV[1] then
                    return redis.call('del', KEYS[1])
                else
                    return 0
                end
                """
                r.eval(lua, 1, self.key, self.token)

# 使用示例
lock = ReentrantLock("resource")
lock.acquire()
lock.acquire()  # 重入
lock.release()
lock.release()  # 最后一次才真正释放
```

## 七、公平锁实现

```python
class FairLock:
    def __init__(self, key, expire=30):
        self.key = f"lock:{key}"
        self.queue_key = f"lock:queue:{key}"
        self.expire = expire
        self.token = str(uuid4())
    
    def acquire(self, timeout=10):
        """获取公平锁"""
        # 加入等待队列
        queue_position = r.lpush(self.queue_key, self.token)
        
        end_time = time.time() + timeout
        while time.time() < end_time:
            # 检查是否轮到自己
            current = r.lindex(self.queue_key, -1)
            if current == self.token.encode():
                if r.set(self.key, self.token, nx=True, ex=self.expire):
                    return True
            time.sleep(0.1)
        
        # 超时，从队列中移除
        r.lrem(self.queue_key, 1, self.token)
        return False
    
    def release(self):
        """释放公平锁"""
        lua = """
        if redis.call('get', KEYS[1]) == ARGV[1] then
            redis.call('del', KEYS[1])
            redis.call('rpop', KEYS[2])
            return 1
        else
            return 0
        end
        """
        return r.eval(lua, 2, self.key, self.queue_key, self.token)
```
