# 分布式队列

## 一、概念说明

Redis可以用作分布式消息队列，支持点对点和发布订阅模式。常用List和Stream实现。

## 二、List实现队列

```bash
# 生产者
LPUSH queue:orders '{"id":1001,"data":"order1"}'

# 消费者（阻塞式）
BRPOP queue:orders 0
# 阻塞等待直到有消息

# 多队列
BRPOP queue:high queue:normal queue:low 0
# 按优先级消费
```

## 三、Stream实现队列

```bash
# 生产者
XADD orders * order_id 1001 amount 99.9

# 创建消费者组
XGROUP CREATE orders group1 0

# 消费者
XREADGROUP GROUP group1 consumer1 COUNT 10 STREAMS orders >

# ACK确认
XACK orders group1 1704067200000-0
```

## 四、Python实现

```python
# List队列
class RedisQueue:
    def __init__(self, name, redis_client):
        self.name = name
        self.redis = redis_client
    
    def put(self, item):
        self.redis.lpush(self.name, json.dumps(item))
    
    def get(self, timeout=0):
        result = self.redis.brpop(self.name, timeout)
        if result:
            return json.loads(result[1])
        return None

# Stream队列
class RedisStreamQueue:
    def __init__(self, stream, group, consumer):
        self.stream = stream
        self.group = group
        self.consumer = consumer
        self.redis = redis.Redis()
    
    def put(self, data):
        return self.redis.xadd(self.stream, data)
    
    def get(self, count=10):
        return self.redis.xreadgroup(
            self.group, self.consumer,
            {self.stream: '>'}, count=count
        )
```

## 五、注意事项

1. **消息持久化**：Stream支持持久化，List不支持
2. **消息确认**：Stream支持ACK，List不支持
3. **消费者组**：Stream支持消费者组
4. **消息丢失**：List队列消费者崩溃可能丢失消息
5. **队列积压**：监控队列长度，防止内存溢出

## 六、Stream消费者组详解

```bash
# 创建消费者组
XGROUP CREATE orders group1 0 MKSTREAM
# MKSTREAM：如果Stream不存在则创建

# 消费者组读取
XREADGROUP GROUP group1 consumer1 COUNT 10 BLOCK 5000 STREAMS orders >
# > 表示读取未消费的消息

# 查看待确认消息
XPENDING orders group1

# 消费者故障恢复
# 获取某消费者未确认的消息
XREADGROUP GROUP group1 consumer2 COUNT 10 STREAMS orders 0
# 0表示重新消费未确认的消息

# 查看Stream信息
XINFO STREAM orders
XINFO GROUPS orders
XINFO CONSUMERS orders group1

# 设置Stream最大长度
XADD orders MAXLEN ~ 10000 * order_id 1001
# ~ 表示近似限制，更高效

# 修剪Stream
XTRIM orders MAXLEN ~ 10000
```

## 七、可靠消息队列实现

```python
import redis
import json
import time
import threading

r = redis.Redis()

class ReliableStreamQueue:
    def __init__(self, stream, group, consumer):
        self.stream = stream
        self.group = group
        self.consumer = consumer
        
        # 创建消费者组
        try:
            r.xgroup_create(stream, group, id='0', mkstream=True)
        except redis.ResponseError:
            pass  # 组已存在
    
    def publish(self, data, max_len=10000):
        """发布消息"""
        return r.xadd(self.stream, {'data': json.dumps(data)}, maxlen=max_len)
    
    def consume(self, count=10, block=5000):
        """消费消息"""
        messages = r.xreadgroup(
            self.group, self.consumer,
            {self.stream: '>'},
            count=count, block=block
        )
        
        results = []
        for stream, msgs in messages:
            for msg_id, data in msgs:
                try:
                    payload = json.loads(data[b'data'])
                    results.append((msg_id, payload))
                except Exception as e:
                    print(f"消息解析失败: {e}")
        
        return results
    
    def ack(self, msg_id):
        """确认消息"""
        return r.xack(self.stream, self.group, msg_id)
    
    def process(self, handler, count=10):
        """处理消息并确认"""
        messages = self.consume(count=count)
        
        for msg_id, data in messages:
            try:
                handler(data)
                self.ack(msg_id)
            except Exception as e:
                print(f"处理失败: {e}, msg_id: {msg_id}")
                # 可以重试或放入死信队列
    
    def pending(self):
        """查看待确认消息"""
        return r.xpending(self.stream, self.group)
    
    def claim(self, min_idle_ms=60000, count=10):
        """认领空闲消息（故障恢复）"""
        # 获取空闲消息
        pending = r.xpending_range(self.stream, self.group, min='+', max='-', count=count)
        
        idle_ids = []
        for item in pending:
            if item['idle'] > min_idle_ms:
                idle_ids.append(item['message_id'])
        
        if idle_ids:
            # 认领消息
            return r.xclaim(self.stream, self.group, self.consumer, min_idle_ms, *idle_ids)
        return []

# 使用示例
queue = ReliableStreamQueue('orders', 'order_group', 'consumer1')

# 发布
queue.publish({'order_id': 1001, 'amount': 99.9})

# 消费
def handle_order(data):
    print(f"处理订单: {data}")

queue.process(handler=handle_order)
```

## 八、死信队列

```python
class DeadLetterQueue:
    def __init__(self, main_stream, dlq_stream, max_retries=3):
        self.main_stream = main_stream
        self.dlq_stream = dlq_stream
        self.max_retries = max_retries
    
    def handle_with_retry(self, msg_id, data, handler):
        """带重试的消息处理"""
        retries = 0
        while retries < self.max_retries:
            try:
                handler(data)
                return True
            except Exception as e:
                retries += 1
                print(f"重试 {retries}/{self.max_retries}: {e}")
                time.sleep(2 ** retries)  # 指数退避
        
        # 重试耗尽，放入死信队列
        r.xadd(self.dlq_stream, {
            'original_msg_id': msg_id,
            'data': json.dumps(data),
            'error': str(e),
            'timestamp': time.time()
        })
        print(f"消息进入死信队列: {msg_id}")
        return False
```
