# 延迟队列

## 一、概念说明

延迟队列用于处理需要延迟执行的任务，如订单超时取消、延迟通知等。Redis的ZSet可以用分数存储执行时间戳来实现延迟队列。

## 二、ZSet实现延迟队列

```python
import redis
import json
import time

r = redis.Redis()

def add_delay_task(queue, task, delay_seconds):
    """添加延迟任务"""
    execute_time = time.time() + delay_seconds
    r.zadd(queue, {json.dumps(task): execute_time})

def get_ready_tasks(queue, count=10):
    """获取已到期的任务"""
    now = time.time()
    tasks = r.zrangebyscore(queue, 0, now, start=0, num=count)
    return [json.loads(t) for t in tasks]

def remove_task(queue, task):
    """删除已处理的任务"""
    r.zrem(queue, json.dumps(task))

def process_tasks(queue):
    """处理到期任务"""
    while True:
        tasks = get_ready_tasks(queue)
        for task in tasks:
            try:
                # 处理任务
                handle_task(task)
                remove_task(queue, task)
            except Exception as e:
                print(f"处理失败: {e}")
        time.sleep(1)

# 使用
add_delay_task("delay:orders", {"order_id": 1001, "action": "cancel"}, 1800)
# 30分钟后执行订单取消
```

## 四、注意事项

1. **时间精度**：秒级或毫秒级时间戳
2. **轮询间隔**：合理设置检查频率
3. **任务幂等**：防止重复执行
4. **内存控制**：及时清理已处理任务
5. **多消费者**：使用分布式锁防止重复消费

## 五、完整延迟队列实现

```python
import redis
import json
import time
import threading
import signal

r = redis.Redis()

class DelayQueue:
    def __init__(self, queue_name):
        self.queue_name = f"delay:queue:{queue_name}"
        self.ready_queue = f"ready:queue:{queue_name}"
        self._running = False
    
    def add_task(self, task, delay_seconds):
        """添加延迟任务"""
        execute_time = time.time() + delay_seconds
        task_data = {
            'id': str(time.time_ns()),
            'task': task,
            'created': time.time(),
            'execute_at': execute_time
        }
        r.zadd(self.queue_name, {json.dumps(task_data): execute_time})
        return task_data['id']
    
    def get_ready_tasks(self, count=10):
        """获取已到期的任务"""
        now = time.time()
        tasks = r.zrangebyscore(self.queue_name, 0, now, start=0, num=count)
        return [json.loads(t) for t in tasks]
    
    def pop_task(self):
        """弹出一个到期任务"""
        now = time.time()
        # 使用Lua脚本原子性弹取
        lua = """
        local tasks = redis.call('ZRANGEBYSCORE', KEYS[1], 0, ARGV[1], 'LIMIT', 0, 1)
        if #tasks > 0 then
            redis.call('ZREM', KEYS[1], tasks[1])
            return tasks[1]
        end
        return nil
        """
        task = r.eval(lua, 1, self.queue_name, now)
        if task:
            return json.loads(task)
        return None
    
    def remove_task(self, task_data):
        """删除已处理的任务"""
        r.zrem(self.queue_name, json.dumps(task_data))
    
    def size(self):
        """获取队列长度"""
        return r.zcard(self.queue_name)
    
    def start_worker(self, handler, interval=1):
        """启动工作线程"""
        self._running = True
        
        def worker():
            while self._running:
                task = self.pop_task()
                if task:
                    try:
                        handler(task['task'])
                        print(f"任务完成: {task['id']}")
                    except Exception as e:
                        print(f"任务失败: {task['id']}, 错误: {e}")
                        # 可选择重新入队
                        self.add_task(task['task'], delay_seconds=60)
                else:
                    time.sleep(interval)
        
        thread = threading.Thread(target=worker, daemon=True)
        thread.start()
        return thread
    
    def stop(self):
        """停止工作线程"""
        self._running = False

# 使用示例
queue = DelayQueue("orders")

# 添加延迟任务（30分钟后取消订单）
queue.add_task({"order_id": 1001, "action": "cancel"}, delay_seconds=1800)

# 添加延迟通知（1小时后发送）
queue.add_task({"user_id": 2001, "type": "reminder"}, delay_seconds=3600)

# 启动工作线程
def handle_task(task):
    if task['action'] == 'cancel':
        print(f"取消订单: {task['order_id']}")
    elif task['type'] == 'reminder':
        print(f"发送提醒: {task['user_id']}")

queue.start_worker(handler=handle_task, interval=1)
```

## 六、订单超时取消实战

```python
class OrderTimeoutHandler:
    def __init__(self):
        self.delay_queue = DelayQueue("order_timeout")
    
    def create_order(self, order_id, timeout_minutes=30):
        """创建订单并设置超时"""
        # 1. 创建订单
        order_key = f"order:{order_id}"
        r.hset(order_key, mapping={
            'status': 'pending',
            'created': time.time(),
            'timeout': timeout_minutes
        })
        
        # 2. 添加延迟任务
        self.delay_queue.add_task(
            {'order_id': order_id, 'action': 'timeout_check'},
            delay_seconds=timeout_minutes * 60
        )
        
        print(f"订单 {order_id} 已创建，{timeout_minutes}分钟后检查超时")
    
    def pay_order(self, order_id):
        """支付订单"""
        order_key = f"order:{order_id}"
        status = r.hget(order_key, 'status')
        
        if status == b'pending':
            r.hset(order_key, 'status', 'paid')
            print(f"订单 {order_id} 已支付")
            return True
        return False
    
    def handle_timeout(self, task):
        """处理超时"""
        order_id = task['order_id']
        order_key = f"order:{order_id}"
        status = r.hget(order_key, 'status')
        
        if status == b'pending':
            r.hset(order_key, 'status', 'cancelled')
            print(f"订单 {order_id} 超时已取消")
            # 释放库存等操作
        else:
            print(f"订单 {order_id} 状态: {status}，无需处理")

# 使用
handler = OrderTimeoutHandler()
handler.create_order(1001, timeout_minutes=30)
```
