# 地理位置服务

## 一、概念说明

Redis Geo支持地理位置的存储、距离计算和范围查询，适用于"附近的人"、"附近的店铺"等场景。

## 二、具体用法

### 存储位置

```bash
# 添加位置
GEOADD stores 116.40 39.90 "store1"
GEOADD stores 116.41 39.91 "store2"
GEOADD stores 116.39 39.89 "store3"

# 距离计算
GEODIST stores store1 store2 km
# 输出: "1.52"

# 获取坐标
GEOPOS stores store1
```

### 范围查询

```bash
# 附近1km的店铺
GEORADIUS stores 116.40 39.90 1 km WITHDIST
# 输出: 附近的店铺及距离

# 以某个店铺为中心
GEORADIUSBYMEMBER stores store1 2 km

# Redis 6.2+ 推荐
GEOSEARCH stores FROMLONLAT 116.40 39.90 BYRADIUS 1 km WITHDIST
```

## 三、实际应用

```python

def find_nearby_users(lat, lon, radius_km):

        unit="km",

# 更新用户位置
def update_user_location(user_id, lat, lon):
```

## 四、注意事项

1. **精度**：GeoHash精度约0.6米
2. **坐标范围**：经度-180到180，纬度-85到85
3. **底层ZSet**：可以用ZREM删除位置
4. **内存占用**：每个位置约52字节
5. **性能**：范围查询性能取决于数据量

## 五、地理位置服务实战

### 附近的人完整实现

```python
import redis
import json
import time

r = redis.Redis()

class NearbyService:
    def __init__(self, geo_key="users:location"):
        self.geo_key = geo_key
    
    def update_location(self, user_id, longitude, latitude):
        """更新用户位置"""
        r.geoadd(self.geo_key, (longitude, latitude, user_id))
        
        # 存储用户详细信息
        user_key = f"user:location:{user_id}"
        r.hset(user_key, mapping={
            'longitude': longitude,
            'latitude': latitude,
            'updated': time.time()
        })
        r.expire(user_key, 86400)
    
    def get_nearby(self, longitude, latitude, radius_km, count=20):
        """获取附近用户"""
        results = r.geosearch(
            self.geo_key,
            longitude=longitude,
            latitude=latitude,
            radius=radius_km,
            unit='km',
            withcoord=True,
            withdist=True,
            count=count,
            sort='ASC'
        )
        
        nearby = []
        for item in results:
            user_id = item[0].decode() if isinstance(item[0], bytes) else item[0]
            distance = float(item[1])
            coords = item[2]
            
            nearby.append({
                'user_id': user_id,
                'distance_km': round(distance, 2),
                'longitude': coords[0],
                'latitude': coords[1]
            })
        
        return nearby
    
    def get_distance(self, user1, user2):
        """获取两个用户之间的距离"""
        dist = r.geodist(self.geo_key, user1, user2, unit='km')
        if dist:
            return round(float(dist), 2)
        return None
    
    def remove_user(self, user_id):
        """移除用户位置"""
        r.zrem(self.geo_key, user_id)
        r.delete(f"user:location:{user_id}")

# 使用示例
service = NearbyService()

# 更新位置
service.update_location("user:1001", 116.40, 39.90)
service.update_location("user:1002", 116.41, 39.91)
service.update_location("user:1003", 116.39, 39.89)

# 查找附近3km内的用户
nearby = service.get_nearby(116.40, 39.90, 3, count=10)
for user in nearby:
    print(f"{user['user_id']}: {user['distance_km']}km")

# 获取距离
distance = service.get_distance("user:1001", "user:1002")
print(f"距离: {distance}km")
```

### 附近的店铺

```python
class StoreLocator:
    def __init__(self):
        self.r = redis.Redis()
        self.geo_key = "stores:location"
    
    def add_store(self, store_id, longitude, latitude, info=None):
        """添加店铺"""
        self.r.geoadd(self.geo_key, (longitude, latitude, store_id))
        
        if info:
            store_key = f"store:{store_id}"
            self.r.hset(store_key, mapping=info)
            self.r.expire(store_key, 86400 * 7)
    
    def find_nearby_stores(self, longitude, latitude, radius_km, count=10):
        """查找附近店铺"""
        stores = self.r.geosearch(
            self.geo_key,
            longitude=longitude,
            latitude=latitude,
            radius=radius_km,
            unit='km',
            withdist=True,
            count=count,
            sort='ASC'
        )
        
        result = []
        for store_id, distance in stores:
            info = self.r.hgetall(f"store:{store_id.decode()}")
            result.append({
                'store_id': store_id.decode(),
                'distance_km': round(float(distance), 2),
                'info': {k.decode(): v.decode() for k, v in info.items()}
            })
        
        return result

# 使用
locator = StoreLocator()
locator.add_store("store:1001", 116.40, 39.90, {"name": "旗舰店", "rating": 4.8})
locator.add_store("store:1002", 116.41, 39.91, {"name": "分店A", "rating": 4.5})

stores = locator.find_nearby_stores(116.40, 39.90, 5)
for store in stores:
    print(f"{store['info'].get('name')}: {store['distance_km']}km")
```

## 七、性能优化

```bash
# 1. 使用GEOSEARCH替代GEORADIUS
# GEOSEARCH是Redis 6.2+推荐的命令
# 支持FROMLONLAT和FROMMEMBER两种方式

# 2. 限制返回数量
GEOSEARCH stores FROMLONLAT 116.40 39.90 BYRADIUS 10 km COUNT 10

# 3. 使用ASC排序
# 距离最近的排前面
GEOSEARCH stores FROMLONLAT 116.40 39.90 BYRADIUS 10 km WITHDIST ASC

# 4. 监控Key数量
ZCARD stores:location
# Geo底层是ZSet

# 5. 定期清理过期位置
# 使用EXPIRE设置用户位置Key的过期时间
# 避免僵尸用户占用内存
```
