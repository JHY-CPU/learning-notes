# 分布式事务

## 一、概念说明

分布式事务保证跨多个服务/数据源的操作要么全部成功，要么全部失败。Redis可用于实现分布式事务中的部分方案。

## 二、最终一致性方案

### 可靠消息最终一致性

```bash
# 1. 本地事务写入业务数据
# 2. 写入消息表
# 3. 发送到Redis消息队列
LPUSH tx:messages '{"id":"tx001","action":"update_inventory","data":{...}}'

# 4. 消费者处理消息
BRPOP tx:messages 0

# 5. 处理成功后删除消息
# 6. 处理失败重试
```

### TCC模式

```python
# Try阶段
def try_reserve(order_id, amount):
    # 冻结资金
    r.hincrby(f"account:{order_id}", "frozen", amount)
    r.hincrby(f"account:{order_id}", "balance", -amount)
    return True

# Confirm阶段
def confirm(order_id, amount):
    r.hincrby(f"account:{order_id}", "frozen", -amount)
    return True

# Cancel阶段
def cancel(order_id, amount):
    r.hincrby(f"account:{order_id}", "frozen", -amount)
    r.hincrby(f"account:{order_id}", "balance", amount)
    return True
```

## 三、Saga模式

```bash
# 顺序执行，失败时补偿
# 步骤1: 扣减库存
DECR inventory:1001

# 步骤2: 创建订单
LPUSH orders '{"id":1001}'

# 步骤3失败 → 补偿步骤1
INCR inventory:1001
```

## 四、Redis事务 + Lua

```bash
# 使用Lua保证原子性
EVAL "
  local balance = tonumber(redis.call('hget', KEYS[1], 'balance'))
  local amount = tonumber(ARGV[1])
  if balance >= amount then
    redis.call('hincrby', KEYS[1], 'balance', -amount)
```

## 五、注意事项

1. **Redis不支持ACID事务**：只保证命令原子性
2. **最终一致性**：大多数场景接受最终一致
3. **补偿机制**：失败时需要补偿操作
4. **消息可靠**：确保消息不丢失
5. **幂等性**：所有操作必须幂等

## 六、TCC模式详细实现

```python
class TCCTransaction:
    """Try-Confirm-Cancel分布式事务"""
    
    def __init__(self):
        self.r = redis.Redis()
    
    def try_transfer(self, from_id, to_id, amount):
        """Try阶段：冻结资金"""
        from_key = f"account:{from_id}"
        to_key = f"account:{to_id}"
        freeze_key = f"freeze:{from_id}:{to_id}:{amount}"
        
        # 检查余额
        balance = float(self.r.hget(from_key, 'balance') or 0)
        if balance < amount:
            return False
        
        # 冻结资金
        lua = """
        local balance = tonumber(redis.call('HGET', KEYS[1], 'balance'))
        local amount = tonumber(ARGV[1])
        
        if balance >= amount then
            redis.call('HINCRBYFLOAT', KEYS[1], 'balance', -amount)
            redis.call('HINCRBYFLOAT', KEYS[1], 'frozen', amount)
            return 1
        else
            return 0
        end
        """
        result = self.r.eval(lua, 1, from_key, amount)
        
        if result:
            # 记录事务状态
            self.r.hset(freeze_key, mapping={
                'from': from_id,
                'to': to_id,
                'amount': amount,
                'status': 'tried'
            })
            self.r.expire(freeze_key, 300)
            return True
        return False
    
    def confirm(self, from_id, to_id, amount):
        """Confirm阶段：提交事务"""
        from_key = f"account:{from_id}"
        to_key = f"account:{to_id}"
        freeze_key = f"freeze:{from_id}:{to_id}:{amount}"
        
        # 扣减冻结金额，增加目标余额
        lua = """
        redis.call('HINCRBYFLOAT', KEYS[1], 'frozen', -ARGV[1])
        redis.call('HINCRBYFLOAT', KEYS[2], 'balance', ARGV[1])
        return 1
        """
        self.r.eval(lua, 2, from_key, to_key, amount)
        self.r.hset(freeze_key, 'status', 'confirmed')
    
    def cancel(self, from_id, to_id, amount):
        """Cancel阶段：回滚事务"""
        from_key = f"account:{from_id}"
        freeze_key = f"freeze:{from_id}:{to_id}:{amount}"
        
        # 解冻资金
        lua = """
        redis.call('HINCRBYFLOAT', KEYS[1], 'frozen', -ARGV[1])
        redis.call('HINCRBYFLOAT', KEYS[1], 'balance', ARGV[1])
        return 1
        """
        self.r.eval(lua, 1, from_key, amount)
        self.r.hset(freeze_key, 'status', 'cancelled')

# 使用
tcc = TCCTransaction()
if tcc.try_transfer(1001, 1002, 100):
    try:
        # 执行其他服务调用
        tcc.confirm(1001, 1002, 100)
    except:
        tcc.cancel(1001, 1002, 100)
```

## 七、幂等性实现

```python
class IdempotentHandler:
    """幂等性处理器"""
    
    def __init__(self):
        self.r = redis.Redis()
    
    def process(self, request_id, handler):
        """幂等处理"""
        lock_key = f"idempotent:{request_id}"
        
        # 检查是否已处理
        if self.r.exists(lock_key):
            print(f"请求 {request_id} 已处理，跳过")
            return None
        
        # 标记为处理中
        self.r.setex(lock_key, 3600, "processing")
        
        try:
            result = handler()
            # 标记为已处理
            self.r.setex(lock_key, 86400, json.dumps(result))
            return result
        except Exception as e:
            # 失败时删除标记，允许重试
            self.r.delete(lock_key)
            raise

# 使用
handler = IdempotentHandler()
result = handler.process("req_12345", lambda: {"status": "success"})
```

## 八、事务监控与恢复

```python
class TransactionMonitor:
    """分布式事务监控"""
    
    def __init__(self):
        self.r = redis.Redis()
    
    def record_transaction(self, tx_id, service, status, data=None):
        """记录事务"""
        tx_key = f"tx:{tx_id}"
        self.r.hset(tx_key, mapping={
            'service': service,
            'status': status,
            'data': json.dumps(data or {}),
            'timestamp': time.time()
        })
        self.r.expire(tx_key, 86400)
        
        # 添加到事务索引
        self.r.zadd("tx:all", {tx_id: time.time()})
    
    def get_pending_transactions(self, timeout=300):
        """获取待处理事务"""
        now = time.time()
        cutoff = now - timeout
        
        tx_ids = self.r.zrangebyscore("tx:all", cutoff, now)
        
        pending = []
        for tx_id in tx_ids:
            tx_data = self.r.hgetall(f"tx:{tx_id}")
            if tx_data.get(b'status') == b'processing':
                pending.append(tx_id)
        
        return pending
```
