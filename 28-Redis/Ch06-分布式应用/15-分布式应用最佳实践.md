## 九、运维自动化脚本

```python
import redis
import time
import json

class RedisOpsAutomation:
    """Redis运维自动化"""
    
    def __init__(self, host='localhost', port=6379):
        self.r = redis.Redis(host=host, port=port)
    
    def health_check(self):
        """健康检查"""
        checks = {}
        
        # 连接检查
        try:
            self.r.ping()
            checks['connection'] = True
        except:
            checks['connection'] = False
            return checks
        
        # 内存检查
        info = self.r.info('memory')
        used = info['used_memory']
        maxmem = info.get('maxmemory', 0)
        checks['memory_usage'] = used / maxmem * 100 if maxmem else 0
        
        # 命中率检查
        stats = self.r.info('stats')
        hits = stats['keyspace_hits']
        misses = stats['keyspace_misses']
        total = hits + misses
        checks['hit_rate'] = hits / total * 100 if total else 0
        
        # 慢查询检查
        slow_log = self.r.slowlog_get(10)
        checks['slow_queries'] = len(slow_log)
        
        # 主从检查
        repl = self.r.info('replication')
        checks['role'] = repl.get('role', 'unknown')
        checks['connected_slaves'] = repl.get('connected_slaves', 0)
        
        return checks
    
    def auto_cleanup(self, pattern="temp:*", batch_size=1000):
        """自动清理临时Key"""
        deleted = 0
        cursor = 0
        
        while True:
            cursor, keys = self.r.scan(cursor, match=pattern, count=batch_size)
            if keys:
                self.r.delete(*keys)
                deleted += len(keys)
            if cursor == 0:
                break
        
        return deleted
    
    def export_data(self, pattern="*", output_file="redis_export.json"):
        """导出数据"""
        data = {}
        cursor = 0
        
        while True:
            cursor, keys = self.r.scan(cursor, match=pattern, count=100)
            for key in keys:
                key_str = key.decode()
                value = self.r.get(key)
                if value:
                    data[key_str] = value.decode()
            if cursor == 0:
                break
        
        with open(output_file, 'w') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        return len(data)

# 使用
ops = RedisOpsAutomation()
health = ops.health_check()
print(f"健康检查: {health}")

deleted = ops.auto_cleanup("temp:*")
print(f"清理了 {deleted} 个临时Key")
```
