# 分布式计数器

## 一、概念说明

Redis的INCR/DECR命令是原子性的，非常适合实现分布式计数器，用于限流、统计、排行榜等场景。

## 二、具体用法

### 基本计数器

```bash
# 递增
INCR counter
# 输出: (integer) 1

INCR counter
# 输出: (integer) 2

# 指定步长
INCRBY counter 10
# 输出: (integer) 12

# 递减
DECR counter
# 输出: (integer) 11

# 浮点数
INCRBYFLOAT price 0.5
```

### 限流计数器

```python
def is_rate_limited(user_id, limit=100, window=60):
    key = f"rate:limit:{user_id}"
    current = r.incr(key)
    
    if current == 1:
        # 第一次请求，设置过期时间
        r.expire(key, window)
    
    return current > limit

# 使用
if is_rate_limited("user:1001", 100, 60):
    return "请求过于频繁"
```

### 统计计数器

```bash
# 文章阅读量
INCR article:1001:views

# 用户登录次数
INCR user:1001:login_count

# 每日统计
INCR stats:daily:20240101:visits
EXPIRE stats:daily:20240101:visits 86400

# 获取当前值
GET stats:daily:20240101:visits
```

### 滑动窗口限流

```python
def sliding_window_limit(user_id, limit=100, window=60):
    key = f"sw:limit:{user_id}"
    now = time.time()
    
    pipe = r.pipeline()
    pipe.zremrangebyscore(key, 0, now - window)
    pipe.zadd(key, {f"{now}": now})
    pipe.zcard(key)
    pipe.expire(key, window)
    results = pipe.execute()
    
    count = results[2]
    return count > limit
```

## 三、注意事项

1. **原子性**：INCR是原子操作
2. **过期时间**：首次INCR时设置过期
3. **溢出**：超过64位有符号整数会报错
4. **限流策略**：固定窗口/滑动窗口/令牌桶
5. **分布式安全**：多个应用实例共享计数器

## 四、高级计数器场景

### 令牌桶限流

```python
import redis
import time

r = redis.Redis()

class TokenBucket:
    def __init__(self, key, rate, capacity):
        """
        key: 限流Key
        rate: 每秒生成令牌数
        capacity: 桶容量
        """
        self.key = f"bucket:{key}"
        self.rate = rate
        self.capacity = capacity
    
    def acquire(self, tokens=1):
        """获取令牌"""
        lua = """
        local key = KEYS[1]
        local rate = tonumber(ARGV[1])
        local capacity = tonumber(ARGV[2])
        local now = tonumber(ARGV[3])
        local requested = tonumber(ARGV[4])
        
        local data = redis.call('HMGET', key, 'tokens', 'last')
        local current_tokens = tonumber(data[1]) or capacity
        local last_time = tonumber(data[2]) or now
        
        -- 计算新令牌
        local elapsed = now - last_time
        current_tokens = math.min(capacity, current_tokens + elapsed * rate)
        
        if current_tokens >= requested then
            current_tokens = current_tokens - requested
            redis.call('HMSET', key, 'tokens', current_tokens, 'last', now)
            redis.call('EXPIRE', key, 60)
            return 1
        else
            redis.call('HMSET', key, 'tokens', current_tokens, 'last', now)
            redis.call('EXPIRE', key, 60)
            return 0
        end
        """
        now = time.time()
        return r.eval(lua, 1, self.key, self.rate, self.capacity, now, tokens)

# 使用
bucket = TokenBucket("api", rate=10, capacity=100)
if bucket.acquire():
    print("请求通过")
else:
    print("被限流")
```

### 滑动窗口计数器

```python
class SlidingWindowCounter:
    def __init__(self, key, window_seconds, max_count):
        self.key = f"sw:{key}"
        self.window = window_seconds
        self.max_count = max_count
    
    def increment(self):
        """递增并检查是否超限"""
        now = time.time()
        window_start = now - self.window
        
        pipe = r.pipeline()
        # 清理过期数据
        pipe.zremrangebyscore(self.key, 0, window_start)
        # 添加当前时间戳
        pipe.zadd(self.key, {f"{now}": now})
        # 统计窗口内数量
        pipe.zcard(self.key)
        # 设置Key过期
        pipe.expire(self.key, self.window + 1)
        
        results = pipe.execute()
        count = results[2]
        
        return count <= self.max_count, count

# 使用
counter = SlidingWindowCounter("user:1001", window_seconds=60, max_count=100)
allowed, count = counter.increment()
print(f"当前计数: {count}, 是否允许: {allowed}")
```

## 六、统计计数器

```python
class StatsCounter:
    def __init__(self):
        self.r = redis.Redis()
    
    def increment_daily(self, stat_name, amount=1):
        """每日统计"""
        date = time.strftime("%Y%m%d")
        key = f"stats:daily:{date}:{stat_name}"
        self.r.incrby(key, amount)
        self.r.expire(key, 86400 * 7)  # 保留7天
    
    def increment_hourly(self, stat_name, amount=1):
        """每小时统计"""
        hour = time.strftime("%Y%m%d%H")
        key = f"stats:hourly:{hour}:{stat_name}"
        self.r.incrby(key, amount)
        self.r.expire(key, 86400 * 2)  # 保留2天
    
    def get_daily_stats(self, stat_name, days=7):
        """获取每日统计"""
        stats = []
        for i in range(days):
            date = (time.time() - i * 86400)
            date_str = time.strftime("%Y%m%d", time.localtime(date))
            key = f"stats:daily:{date_str}:{stat_name}"
            value = self.r.get(key) or 0
            stats.append((date_str, int(value)))
        return stats

# 使用
stats = StatsCounter()
stats.increment_daily("page_views")
stats.increment_daily("api_calls", 10)
daily = stats.get_daily_stats("page_views")
```
