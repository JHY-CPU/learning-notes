## 六、完整集成检查脚本

```python
import redis
import json

class RedisIntegrationCheck:
    """Redis集成检查工具"""
    
    def __init__(self, host='localhost', port=6379):
        self.r = redis.Redis(host=host, port=port)
    
    def check_all(self):
        """全面检查"""
        results = {}
        
        # 连接检查
        results['connection'] = self._check_connection()
        
        # 配置检查
        results['config'] = self._check_config()
        
        # 性能检查
        results['performance'] = self._check_performance()
        
        # 安全检查
        results['security'] = self._check_security()
        
        return results
    
    def _check_connection(self):
        try:
            self.r.ping()
            return {'status': 'ok', 'message': '连接正常'}
        except Exception as e:
            return {'status': 'error', 'message': str(e)}
    
    def _check_config(self):
        issues = []
        config = self.r.config_get('*')
        
        # 检查maxmemory
        if config.get('maxmemory', '0') == '0':
            issues.append('maxmemory未设置')
        
        # 检查淘汰策略
        policy = config.get('maxmemory-policy', 'noeviction')
        if policy == 'noeviction':
            issues.append('使用默认淘汰策略(noeviction)')
        
        # 检查持久化
        if config.get('appendonly', 'no') == 'no':
            issues.append('AOF未开启')
        
        # 检查密码
        if not config.get('requirepass'):
            issues.append('未设置密码')
        
        return {
            'status': 'warning' if issues else 'ok',
            'issues': issues
        }
    
    def _check_performance(self):
        info = self.r.info()
        
        # 命中率
        hits = info['keyspace_hits']
        misses = info['keyspace_misses']
        hit_rate = hits / (hits + misses) * 100 if (hits + misses) > 0 else 0
        
        # 内存使用率
        used = info['used_memory']
        maxmem = info.get('maxmemory', 0)
        mem_usage = used / maxmem * 100 if maxmem > 0 else 0
        
        # 碎片率
        frag_ratio = info['mem_fragmentation_ratio']
        
        return {
            'hit_rate': hit_rate,
            'memory_usage': mem_usage,
            'fragmentation_ratio': frag_ratio,
            'ops_per_sec': info['instantaneous_ops_per_sec']
        }
    
    def _check_security(self):
        issues = []
        config = self.r.config_get('*')
        
        if not config.get('requirepass'):
            issues.append('未设置密码')
        
        if config.get('protected-mode', 'yes') == 'no':
            issues.append('protected-mode已关闭')
        
        return {
            'status': 'warning' if issues else 'ok',
            'issues': issues
        }
    
    def print_report(self):
        """打印报告"""
        results = self.check_all()
        
        print("Redis集成检查报告")
        print("=" * 50)
        
        for category, data in results.items():
            print(f"\n{category.upper()}:")
            if isinstance(data, dict):
                for key, value in data.items():
                    print(f"  {key}: {value}")

# 运行检查
check = RedisIntegrationCheck()
check.print_report()
```
