# 缓存与分布式锁

## 一、概念说明

分布式锁用于保护共享资源的并发访问，在缓存场景中用于防止缓存击穿、保护热点数据更新等。

## 二、Redis分布式锁

### 基本实现

```bash
# 获取锁
SET lock:resource "uuid" NX EX 30
# NX: 只在Key不存在时设置
# EX 30: 30秒过期

# 释放锁（Lua脚本保证原子性）
EVAL "if redis.call('get',KEYS[1]) == ARGV[1] then return redis.call('del',KEYS[1]) else return 0 end" 1 lock:resource "uuid"
```

### 防止缓存击穿

```python
def get_with_lock(key):
    # 1. 先查缓存
    data = redis.get(key)
    if data:
        return data
    
    # 2. 获取分布式锁
    lock_key = f"lock:{key}"
    uuid = str(uuid4())
    acquired = redis.set(lock_key, uuid, nx=True, ex=10)
    
    if acquired:
        try:
            # 3. 查数据库
            data = db.query(key)
            # 4. 写缓存
            redis.set(key, data, ex=3600)
            return data
        finally:
            # 5. 释放锁
            release_lock(lock_key, uuid)
    else:
        # 等待后重试
        sleep(0.1)
        return get_with_lock(key)
```

## 四、注意事项

1. **锁超时**：必须设置过期时间，防止死锁
2. **锁粒度**：不要使用大粒度锁
3. **重试机制**：获取锁失败时有重试逻辑
4. **锁释放**：使用Lua脚本保证原子性
5. **时钟漂移**：Redlock需要考虑时钟问题

## 五、Redisson分布式锁

```java
// Spring Boot + Redisson
import org.redisson.Redisson;
import org.redisson.api.RLock;
import org.redisson.config.Config;

Config config = new Config();
config.useSingleServer().setAddress("redis://192.168.1.100:6379");
RedissonClient redisson = Redisson.create(config);

// 可重入锁
RLock lock = redisson.getLock("resource:lock");
try {
    // 尝试获取锁，最多等待10秒，锁自动过期30秒
    boolean acquired = lock.tryLock(10, 30, TimeUnit.SECONDS);
    if (acquired) {
        // 临界区操作
        System.out.println("获取锁成功");
    }
} finally {
    if (lock.isHeldByCurrentThread()) {
        lock.unlock();
    }
}

// 公平锁（按请求顺序获取）
RLock fairLock = redisson.getFairLock("resource:fair");

// 读写锁
RReadWriteLock rwLock = redisson.getReadWriteLock("resource:rw");
RLock readLock = rwLock.readLock();
RLock writeLock = rwLock.writeLock();

// 信号量
RSemaphore semaphore = redisson.getSemaphore("resource:semaphore");
semaphore.trySetPermits(10);  // 设置许可数
semaphore.acquire();  // 获取许可
semaphore.release();  // 释放许可
```

## 六、Redlock算法详解

```python
import redis
import time
import uuid

class Redlock:
    def __init__(self, nodes):
        """
        nodes: Redis节点列表
        例如: [('192.168.1.100', 6379), ('192.168.1.101', 6379), ('192.168.1.102', 6379)]
        """
        self.nodes = [redis.Redis(host=host, port=port) for host, port in nodes]
        self.quorum = len(nodes) // 2 + 1
    
    def lock(self, resource, ttl=30000):
        """获取分布式锁"""
        token = str(uuid.uuid4())
        start_time = time.time() * 1000
        
        acquired = 0
        for node in self.nodes:
            try:
                if node.set(resource, token, nx=True, px=ttl):
                    acquired += 1
            except:
                pass
        
        elapsed = time.time() * 1000 - start_time
        
        # 检查是否获取了多数节点的锁
        if acquired >= self.quorum and elapsed < ttl:
            return token
        else:
            # 获取失败，释放已获取的锁
            self.unlock(resource, token)
            return None
    
    def unlock(self, resource, token):
        """释放分布式锁"""
        for node in self.nodes:
            try:
                lua = """
                if redis.call('get', KEYS[1]) == ARGV[1] then
                    return redis.call('del', KEYS[1])
                else
                    return 0
                end
                """
                node.eval(lua, 1, resource, token)
            except:
                pass

# 使用
nodes = [('192.168.1.100', 6379), ('192.168.1.101', 6379), ('192.168.1.102', 6379)]
redlock = Redlock(nodes)

token = redlock.lock("order:1001", ttl=30000)
if token:
    try:
        # 临界区操作
        print("获取锁成功")
    finally:
        redlock.unlock("order:1001", token)
```

## 七、分布式锁常见问题

```bash
# 问题1：锁过期但业务未完成
# 解决：锁续期（看门狗机制）
# Redisson默认每10秒续期

# 问题2：主从切换导致锁丢失
# 解决：使用Redlock多节点锁
# 或接受短暂不一致

# 问题3：时钟漂移
# 解决：NTP同步时钟
# Redlock中使用SET NX PX避免时钟问题

# 问题4：锁粒度过大
# 解决：按资源细分锁
# 不要使用一个大锁保护所有资源

# 问题5：死锁
# 解决：必须设置过期时间
# 使用Lua脚本原子释放
```

## 八、缓存场景锁应用

```python
def get_with_cache_lock(key, db_query_func, ttl=3600):
    """带锁的缓存查询"""
    # 1. 查缓存
    cached = r.get(key)
    if cached:
        return json.loads(cached)
    
    # 2. 获取锁
    lock_key = f"lock:{key}"
    token = str(uuid.uuid4())
    
    if r.set(lock_key, token, nx=True, ex=10):
        try:
            # 双重检查
            cached = r.get(key)
            if cached:
                return json.loads(cached)
            
            # 查数据库
            data = db_query_func()
            
            # 写缓存
            r.setex(key, ttl, json.dumps(data))
            
            return data
        finally:
            # 释放锁
            lua = """
            if redis.call('get', KEYS[1]) == ARGV[1] then
                return redis.call('del', KEYS[1])
            end
            """
            r.eval(lua, 1, lock_key, token)
    else:
        # 获取锁失败，等待后重试
        time.sleep(0.1)
        return get_with_cache_lock(key, db_query_func, ttl)
```
