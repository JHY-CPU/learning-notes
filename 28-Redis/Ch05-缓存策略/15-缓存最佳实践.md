# 缓存最佳实践

## 一、设计原则

```bash
# 1. 缓存粒度适当
# 不要缓存整个大对象，只缓存需要的字段

# 2. 过期时间必须设置
# 避免内存泄漏
SET key value EX 3600

# 3. Key命名规范
# 业务:类型:标识
user:1001:profile
product:2001:detail
```

## 二、常见问题与解决

```bash
# 缓存穿透 → 布隆过滤器/空值缓存
# 缓存击穿 → 互斥锁/逻辑过期
# 缓存雪崩 → 随机过期/多级缓存
# 数据一致性 → 先更新DB再删缓存
```

## 六、Python缓存工具类

```python
import redis
import json
import time
import random
from functools import wraps
from uuid import uuid4

class RedisCache:
    def __init__(self, host='localhost', port=6379):
        self.r = redis.Redis(host=host, port=port, decode_responses=True)
    
    def get_or_set(self, key, db_query_func, ttl=3600, jitter=300):
        """获取缓存或设置"""
        cached = self.r.get(key)
        if cached:
            if cached == 'NULL':
                return None
            return json.loads(cached)
        
        # 查数据库
        data = db_query_func()
        
        if data:
            # 随机过期时间（防止雪崩）
            actual_ttl = ttl + random.randint(0, jitter)
            self.r.setex(key, actual_ttl, json.dumps(data))
        else:
            # 缓存空值（防止穿透）
            self.r.setex(key, 300, 'NULL')
        
        return data
    
    def delete(self, key):
        """删除缓存"""
        self.r.delete(key)
    
    def delete_pattern(self, pattern):
        """批量删除匹配的Key"""
        cursor = 0
        while True:
            cursor, keys = self.r.scan(cursor, match=pattern, count=100)
            if keys:
                self.r.delete(*keys)
            if cursor == 0:
                break
    
    def cache_decorator(self, key_prefix, ttl=3600):
        """缓存装饰器"""
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                # 生成缓存Key
                key = f"{key_prefix}:{':'.join(str(a) for a in args)}"
                
                # 查缓存
                cached = self.r.get(key)
                if cached:
                    return json.loads(cached)
                
                # 执行函数
                result = func(*args, **kwargs)
                
                # 写缓存
                if result:
                    self.r.setex(key, ttl, json.dumps(result))
                
                return result
            return wrapper
        return decorator

# 使用示例
cache = RedisCache()

def query_user(user_id):
    return {"id": user_id, "name": f"User{user_id}"}

# 方式1：直接使用
user = cache.get_or_set("user:1001", lambda: query_user(1001))

# 方式2：装饰器
@cache.cache_decorator("user", ttl=3600)
def get_user(user_id):
    return query_user(user_id)

user = get_user(1001)
```
