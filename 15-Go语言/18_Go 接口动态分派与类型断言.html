接口值的运行时表示 (itable/tab)、动态分派机制、类型断言语法与安全模式、类型 switch、接口递归与组合、接口性能开销。

接口运行时表示

// ========== 接口内部结构 ========== // Go 接口值底层由 (type, value) 对表示 // 空接口 interface{}: // type eface struct { // _type *_type // 类型指针 // data unsafe.Pointer // 值指针 // } // 16 字节 // 非空接口 (有方法): // type iface struct { // tab *itab // 接口表: 接口类型 + 具体类型 + 方法列表 // data unsafe.Pointer // 值指针 // } // 16 字节 // itab 结构: // type itab struct { // inter *interfacetype // 接口类型 // _type *_type // 具体类型 // hash uint32 // 类型哈希 (快速类型断言) // fun [1]uintptr // 方法列表 (变长) // } func ifaceInternals() { var i interface{} = 42 // eface: {_type: int, data: 0x...} var s fmt.Stringer = &MyString{} // iface: {tab: {inter: Stringer, _type: *MyString, fun: [...]}, data: 0x...} fmt.Println(i, s) } type MyString struct{ val string } func (m *MyString) String() string { return m.val } // ========== 动态分派机制 ========== // 接口调用时, 通过 itab.fun 跳转到具体类型的方法 type Animal interface { Speak() string Move() string } type Dog struct{ Name string } func (d Dog) Speak() string { return "汪汪!" } func (d Dog) Move() string { return "跑" } type Cat struct{ Name string } func (c Cat) Speak() string { return "喵~" } func (c Cat) Move() string { return "走" } func dynamicDispatch() { var a Animal a = Dog{Name: "旺财"} // a.tab = {Animal, Dog, [Speak, Move]} // 调用 a.Speak() → 通过 itab.fun[0] 跳转到 Dog.Speak fmt.Println(a.Speak()) // 汪汪! a = Cat{Name: "咪咪"} // a.tab = {Animal, Cat, [Speak, Move]} fmt.Println(a.Speak()) // 喵~ } // ========== 接口值的比较 ========== // 接口可比较, 但比较的是动态类型和动态值 func ifaceCompare() { var a interface{} = 42 var b interface{} = 42 fmt.Println(a == b) // true (类型相同 + 值相同) var c interface{} = []int{1, 2} // var d interface{} = []int{1, 2} // fmt.Println(c == d) // panic! 切片不可比较 // nil 接口: var e interface{} // nil var f *int = nil var g interface{} = f // 非 nil (type != nil) fmt.Println(e == nil) // true fmt.Println(g == nil) // false! }

类型断言

// ========== 类型断言 ========== // x.(T): 检查接口 x 的动态类型是否为 T func typeAssertion() { var i interface{} = "hello" // 1. 直接断言 (不安全, 失败则 panic) s := i.(string) fmt.Println(s) // "hello" // 2. comma ok 模式 (安全) s, ok := i.(string) if ok { fmt.Println("string:", s) } else { fmt.Println("不是 string") } // 3. 断言不同类型 n, ok := i.(int) if !ok { fmt.Println("不是 int") // 安全, 不 panic } _ = n } // ========== 断言到接口 ========== // 断言具体类型是否实现了另一个接口 type Reader interface { Read(p []byte) (n int, err error) } type Writer interface { Write(p []byte) (n int, err error) } type ReadWriter interface { Reader Writer } func assertInterface() { var r Reader = bytes.NewBufferString("hello") // 检查 r 是否也实现了 Writer if w, ok := r.(Writer); ok { w.Write([]byte(" world")) } // 检查 r 是否实现了 ReadWriter if rw, ok := r.(ReadWriter); ok { fmt.Println("支持 Read+Write") _ = rw } } // ========== 断言性能 ========== // 类型断言通常是 O(1), 使用 itab.hash 快速比较 func assertPerf() { var i interface{} = "hello" // 快路径: 具体类型断言 → 比较 _type 指针 s := i.(string) // 慢路径: 接口到接口 → 需要查找 itab 缓存 var r Reader = bytes.NewBufferString("test") if w, ok := r.(Writer); ok { _ = w } _ = s } // ========== 类型断言的最佳实践 ========== // 1. 总是用 comma ok 模式 (避免 panic) func safeAssert(i interface{}) { if s, ok := i.(string); ok { fmt.Println("长度:", len(s)) } } // 2. 断言前可先检查 nil func nonNilAssert(i interface{}) { if i == nil { return } s := i.(string) // 安全, i 不为 nil _ = s } // 3. 错误处理中的类型断言 func handleError(err error) { var netErr net.Error if errors.As(err, &netErr) { fmt.Println("网络错误, 可重试:", netErr.Temporary()) } }

类型 switch

// ========== 类型 Switch ========== // switch 语句中做类型断言, 简洁高效 func typeSwitch(i interface{}) { switch v := i.(type) { case nil: fmt.Println("nil") case int: fmt.Println("int:", v*2) // v 是 int 类型 case string: fmt.Println("string:", len(v)) // v 是 string 类型 case bool: fmt.Println("bool:", v) case float64: fmt.Println("float64:", v) case *Config: fmt.Println("config:", v.Host) default: fmt.Printf("未知类型: %T\n", v) } } // ========== 类型 Switch 进阶 ========== // 匹配接口 type ErrorWithCode interface { Code() int } func handleErrorCode(err error) { switch e := err.(type) { case ErrorWithCode: fmt.Printf("错误码: %d\n", e.Code()) case interface{ Temporary() bool }: if e.Temporary() { fmt.Println("临时错误, 可重试") } default: fmt.Println("未知错误:", err) } } // 多类型同一分支 func multiTypeSwitch(i interface{}) { switch i.(type) { case int, int32, int64: fmt.Println("整数类型") case float32, float64: fmt.Println("浮点类型") case string, []byte: fmt.Println("字符串或字节切片") } } // ========== JSON 反序列化中的类型 switch ========== func processJSON(data interface{}) { switch v := data.(type) { case map[string]interface{}: // JSON 对象 for key, val := range v { fmt.Printf("键: %s, 值: %v\n", key, val) } case []interface{}: // JSON 数组 for i, item := range v { fmt.Printf("[%d]: %v\n", i, item) } case string: fmt.Println("字符串:", v) case float64: fmt.Println("数字:", v) case bool: fmt.Println("布尔:", v) } } // ========== 空接口模式 ========== // 使用 interface{} 处理不确定类型 // 任意类型容器 type AnySlice []interface{} func anySliceDemo() { s := AnySlice{1, "hello", true, 3.14, struct{}{}} for _, v := range s { typeSwitch(v) } } // 泛型之前的替代方案 type Box struct { Value interface{} } func (b Box) Int() (int, bool) { v, ok := b.Value.(int) return v, ok } func (b Box) String() (string, bool) { v, ok := b.Value.(string) return v, ok } // Go 1.18+ 泛型后, interface{} 用于不确定类型场景 // type Box[T any] struct { Value T }
💡 接口与断言要点: 接口值 16 字节 (eface: _type+data / iface: itab+data); itab 缓存方法地址实现动态分派; 类型断言 x.(T) 安全形式用 comma ok; 接口比较: 动态类型+动态值都相等才 true; 类型 switch v := i.(type) 简洁高效; 接口到接口的断言需要查找 itab; 断言为具体类型是 O(1) 比较 _type 指针; nil 接口 ≠ nil 指针赋给接口。
