内存对齐

// ========== 结构体内存对齐 ========== // CPU 访问对齐地址更高效, 编译器自动填充对齐 import ( "fmt" "unsafe" ) // ❌ 浪费空间: 字段顺序不佳 type BadStruct struct { A bool // 1 字节 + 7 填充 (对齐到 8) B int64 // 8 字节 C bool // 1 字节 + 7 填充 (对齐到 8) } // 实际: A(1) + 填充(7) + B(8) + C(1) + 填充(7) = 24 字节 // ✅ 节省空间: 按大小降序排列 type GoodStruct struct { B int64 // 8 字节 A bool // 1 字节 C bool // 1 字节 + 6 填充 (对齐到 8) } // 实际: B(8) + A(1) + C(1) + 填充(6) = 16 字节 func alignmentDemo() { var bad BadStruct var good GoodStruct fmt.Println(unsafe.Sizeof(bad)) // 24 fmt.Println(unsafe.Sizeof(good)) // 16 // 各字段偏移量 fmt.Println(unsafe.Offsetof(good.B)) // 0 fmt.Println(unsafe.Offsetof(good.A)) // 8 fmt.Println(unsafe.Offsetof(good.C)) // 9 } // ========== 对齐规则 ========== // 1. 每个类型的对齐值 = min(类型大小, 8) 或架构相关 // 2. 结构体整体对齐 = 最大字段对齐值 // 3. 字段偏移 = 对齐值的整数倍 // 4. 最后可能填充使总大小为最大对齐值的整数倍 type AlignExample struct { a byte // 1, 偏移 0, 占 1 // 填充 1 // 1 (short 对齐 2) b int16 // 2, 偏移 2, 占 2 c float32 // 4, 偏移 4, 占 4 d bool // 1, 偏移 8, 占 1 // 填充 3 // (最大对齐 4, 总大小需 4 的倍数) } // 总计: 1+1+2+4+1+3 = 12 字节 // ========== 空结构体 (struct{} ) ========== // 零字节, 常用于信号或 Set 值类型 type Empty struct{} func emptyStruct() { var e Empty fmt.Println(unsafe.Sizeof(e)) // 0 // Set 实现 set := make(map[string]struct{}) set["key"] = struct{}{} // Channel 信号 (不传数据) done := make(chan struct{}) go func() { // 工作... done <- struct{}{} }() <-done } // 空结构体在 struct 中不占空间 type Optimized struct { _ [0]int // 零数组也不占空间, 用于字段占位 A int64 // 8 B bool // 1 + 7 填充 } // fmt.Println(unsafe.Sizeof(Optimized{})) // 16 // 注意: 空结构体作为最后字段时可能占用空间 (对齐需求) type LastEmpty struct { A int64 // 8 B struct{} // 0, 但 Go 会填充到 8 以保持地址唯一 } // unsafe.Sizeof(LastEmpty{}) // 16 不是 8!

Struct Tags 与反射

// ========== Struct Tags ========== // 字段标签是字符串元数据, 用于序列化/验证/ORM type User struct { ID int64 `json:"id" db:"user_id" validate:"required"` Username string `json:"username" db:"username" validate:"required,min=3,max=50"` Email string `json:"email" db:"email" validate:"required,email"` Password string `json:"-" db:"password"` // json 忽略 Role string `json:"role,omitempty" validate:"oneof=admin user"` CreatedAt time.Time `json:"created_at"` } // ========== reflect 读取 Tag ========== import "reflect" func readTags() { t := reflect.TypeOf(User{}) for i := 0; i < t.NumField(); i++ { field := t.Field(i) jsonTag := field.Tag.Get("json") dbTag := field.Tag.Get("db") validateTag := field.Tag.Get("validate") fmt.Printf("字段: %s\n", field.Name) fmt.Printf(" json: %s\n", jsonTag) fmt.Printf(" db: %s\n", dbTag) fmt.Printf(" validate: %s\n", validateTag) } } // ========== 验证 tag 示例 ========== type Validator struct{} func (v *Validator) Validate(obj interface{}) error { t := reflect.TypeOf(obj) val := reflect.ValueOf(obj) // 解引用指针 if t.Kind() == reflect.Ptr { t = t.Elem() val = val.Elem() } for i := 0; i < t.NumField(); i++ { field := t.Field(i) fieldVal := val.Field(i) rule := field.Tag.Get("validate") if rule == "" { continue } rules := strings.Split(rule, ",") for _, r := range rules { switch { case r == "required": if fieldVal.IsZero() { return fmt.Errorf("%s 是必填字段", field.Name) } case strings.HasPrefix(r, "min="): min, _ := strconv.Atoi(strings.TrimPrefix(r, "min=")) if fieldVal.Kind() == reflect.String && len(fieldVal.String()) < min { return fmt.Errorf("%s 长度不能小于 %d", field.Name, min) } case strings.HasPrefix(r, "max="): max, _ := strconv.Atoi(strings.TrimPrefix(r, "max=")) if fieldVal.Kind() == reflect.String && len(fieldVal.String()) > max { return fmt.Errorf("%s 长度不能大于 %d", field.Name, max) } } } } return nil } // ========== Struct 嵌入组合 ========== // Go 用嵌入 (embedding) 而非继承 type Base struct { ID int64 CreatedAt time.Time UpdatedAt time.Time } // 嵌入: 字段和方法被提升 type Product struct { Base // 嵌入 (匿名) Name string Price float64 } func embeddingDemo() { p := Product{ Base: Base{ID: 1, CreatedAt: time.Now(), UpdatedAt: time.Now()}, Name: "笔记本", Price: 99.9, } // 提升字段, 直接访问 fmt.Println(p.ID) // 提升 fmt.Println(p.CreatedAt) // 提升 // 也可通过类型名访问 fmt.Println(p.Base.ID) // 嵌入 + 接口实现 var _ json.Marshaler = struct { Base // 覆盖 Base 的 MarshalJSON 方法 }{} } // 嵌入 vs 嵌套 type Address struct { City string State string } type Employee struct { Base // 嵌入 (字段提升, "is-a" 关系) Home Address // 命名嵌套 ("has-a" 关系) } // 多嵌入: 如果两个嵌入类型有同名方法, 需显式指定

unsafe 操作

// ========== unsafe.Pointer ========== // unsafe.Pointer 可以绕过类型系统转换指针 // 用于: 内存映射, 系统调用, 性能优化 func unsafeDemo() { type MyStruct struct { X int64 Y int64 } s := MyStruct{X: 10, Y: 20} // 通过偏移访问字段 ptr := unsafe.Pointer(&s) xPtr := (*int64)(ptr) // 指向 X yPtr := (*int64)(unsafe.Pointer( uintptr(ptr) + unsafe.Offsetof(s.Y), // Y 的偏移 )) fmt.Println(*xPtr) // 10 fmt.Println(*yPtr) // 20 } // ========== 类型转换: string ↔ []byte (零拷贝) ========== // 标准转换会复制底层数组 // unsafe 转换可实现零拷贝 (需小心使用!) func stringToBytes(s string) []byte { // 零拷贝转换 (只读缓冲区场景) return unsafe.Slice(unsafe.StringData(s), len(s)) } func bytesToString(b []byte) string { if len(b) == 0 { return "" } return unsafe.String(&b[0], len(b)) } // 注意: 此方式生成的 []byte 不可修改! // 修改会导致未定义行为 // ========== 类型大小与偏移 ========== func sizeAndOffset() { fmt.Println(unsafe.Sizeof(int(0))) // 8 (64位) fmt.Println(unsafe.Sizeof(string(""))) // 16 (ptr+len) fmt.Println(unsafe.Sizeof([]int{})) // 24 (ptr+len+cap) fmt.Println(unsafe.Sizeof(map[int]int{})) // 8 (指向 hmap 指针) } // ========== 使用 unsafe 的安全原则 ========== // 1. 优先使用 reflect, 只在极致性能时用 unsafe // 2. uintptr 是整数, 不持有引用 (GC 可能回收内存!) // 3. unsafe.Pointer → uintptr → unsafe.Pointer 是危险操作 // 4. 保证对齐和生命周期 // ❌ 危险: uintptr 不阻止 GC // func danger() { // p := getPtr() // addr := uintptr(p) // 此时 GC 可能回收 p // q := (*T)(unsafe.Pointer(addr)) // 悬空指针! // } // ✅ 安全: 连续操作 // func safe() { // p := getPtr() // q := (*T)(unsafe.Pointer( // uintptr(unsafe.Pointer(p)) + offset, // )) // 在 unsafe.Pointer 保护下一次完成 // }
💡 结构体要点: 字段顺序影响内存大小 (按大小降序排列最省); 对齐规则: 偏移 = 对齐值整数倍, 总大小 = 最大对齐值整数倍; 空结构体零字节但做最后字段可能填充; Tag 是运行时元数据, 用 reflect 读取; 嵌入提升字段和方法 (不是继承); unsafe.Pointer 绕过类型系统, 需谨慎; unsafe.Slice/String 实现零拷贝转换; 嵌入 vs 嵌套区别 (is-a vs has-a)。
