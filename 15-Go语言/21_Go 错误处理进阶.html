⚠️ Go 错误处理进阶

Sentinel errors、自定义错误类型、error wrapping (fmt.Errorf %w)、errors.Is/As 链式检查、错误处理策略与最佳实践。

错误类型体系

// ========== Go 错误类型 ========== // error 是内置接口: // type error interface { // Error() string // } // 1. 预定义错误 (Sentinel Errors) var ( ErrNotFound = errors.New("not found") ErrPermission = errors.New("permission denied") ErrTimeout = errors.New("request timeout") ErrInvalidInput = errors.New("invalid input") ) // 2. 自定义错误类型 type ValidationError struct { Field string Value interface{} Message string } func (e *ValidationError) Error() string { return fmt.Sprintf("验证失败: %s = %v (%s)", e.Field, e.Value, e.Message) } // 3. 带状态码的错误 type HTTPError struct { StatusCode int Message string Err error // 内部错误 } func (e *HTTPError) Error() string { if e.Err != nil { return fmt.Sprintf("HTTP %d: %s (%v)", e.StatusCode, e.Message, e.Err) } return fmt.Sprintf("HTTP %d: %s", e.StatusCode, e.Message) } func (e *HTTPError) Unwrap() error { return e.Err } // ========== 使用自定义错误 ========== func findUser(id int) (*User, error) { if id <= 0 { return nil, &ValidationError{ Field: "id", Value: id, Message: "ID 必须为正数", } } // ... 数据库查询 return nil, ErrNotFound // sentinel error } func findUserAPI(id int) (*User, error) { user, err := findUser(id) if err != nil { return nil, &HTTPError{ StatusCode: 400, Message: "查询用户失败", Err: err, } } return user, nil }

Error Wrapping

// ========== fmt.Errorf %w ========== // Go 1.13+ error wrapping: 保留错误链 func openConfig(path string) error { return fmt.Errorf("打开配置 %s: %w", path, ErrNotFound) // %w 创建包装错误, 可通过 errors.Is/As 解包 } // 多层包装: func loadConfig() error { err := openConfig("/etc/app.conf") if err != nil { return fmt.Errorf("加载配置失败: %w", err) } return nil } // 使用 errors.Is 检查: func handleConfig() { err := loadConfig() if err != nil { fmt.Println(err) // "加载配置失败: 打开配置 /etc/app.conf: not found" if errors.Is(err, ErrNotFound) { fmt.Println("→ 配置不存在, 使用默认值") } } } // ========== errors.Is — 链式检查 ========== // errors.Is 沿错误链查找匹配的 sentinel error func isDemo() { err := loadConfig() // 等价于: // err == ErrNotFound 或 err 链中任意层 == ErrNotFound if errors.Is(err, ErrNotFound) { fmt.Println("not found") } // 自定义 Is 方法 (控制匹配规则) } // 自定义 Is 方法 type QueryError struct { Query string Err error } func (e *QueryError) Error() string { return fmt.Sprintf("查询 %q 失败: %v", e.Query, e.Err) } func (e *QueryError) Unwrap() error { return e.Err } // 自定义 Is: 任意 QueryError 都匹配 ErrNotFound func (e *QueryError) Is(target error) bool { return target == ErrNotFound || target == ErrInvalidInput } // ========== errors.As — 类型检查 ========== // errors.As 沿错误链查找匹配的自定义错误类型 func asDemo() { err := findUserAPI(-1) // 查找链中第一个 *HTTPError var httpErr *HTTPError if errors.As(err, &httpErr) { fmt.Printf("HTTP %d: %s\n", httpErr.StatusCode, httpErr.Message) } // 查找链中第一个 *ValidationError var valErr *ValidationError if errors.As(err, &valErr) { fmt.Printf("字段 %s 验证失败\n", valErr.Field) } // As 使用接口 var temporary interface{ Temporary() bool } if errors.As(err, &temporary) { if temporary.Temporary() { fmt.Println("临时错误, 可重试") } } } // ========== errors.Unwrap ========== // 单层解包 func unwrapDemo() { err := fmt.Errorf("包装: %w", ErrNotFound) fmt.Println(errors.Unwrap(err)) // "not found" fmt.Println(errors.Unwrap(ErrNotFound)) // nil }

错误最佳实践

// ========== 错误设计指南 ========== // 1. Sentinel 错误用于预期错误 var ErrNotFound = errors.New("not found") // 2. 自定义错误用于额外信息 type NotFoundError struct { Entity string ID int64 } func (e *NotFoundError) Error() string { return fmt.Sprintf("%s (id=%d) 未找到", e.Entity, e.ID) } // 3. 错误分类 type ErrorType int const ( ErrorTypeValidation ErrorType = iota ErrorTypeNotFound ErrorTypePermission ErrorTypeInternal ) type AppError struct { Type ErrorType Message string Err error } func (e *AppError) Error() string { return e.Message } func (e *AppError) Unwrap() error { return e.Err } // 4. HTTP 层错误映射 func httpStatus(err error) int { var appErr *AppError if errors.As(err, &appErr) { switch appErr.Type { case ErrorTypeValidation: return http.StatusBadRequest case ErrorTypeNotFound: return http.StatusNotFound case ErrorTypePermission: return http.StatusForbidden case ErrorTypeInternal: return http.StatusInternalServerError } } return http.StatusInternalServerError } // ========== 结构化错误 ========== type ErrorResponse struct { Code string `json:"code"` Message string `json:"message"` Details interface{} `json:"details,omitempty"` } func (e *ErrorResponse) Error() string { return e.Message } func NewValidationError(field, msg string) *ErrorResponse { return &ErrorResponse{ Code: "VALIDATION_ERROR", Message: "输入验证失败", Details: map[string]string{field: msg}, } } func NewNotFoundError(entity string) *ErrorResponse { return &ErrorResponse{ Code: "NOT_FOUND", Message: fmt.Sprintf("%s 未找到", entity), } } // ========== 断言与辅助函数 ========== // 测试中辅助 func assertNoError(t *testing.T, err error) { t.Helper() if err != nil { t.Fatalf("不期望错误: %v", err) } } // 只记录日志不返回 func logError(err error) { if err != nil { // 包含调用栈信息 log.Printf("ERROR: %+v", err) } } // ========== 总结: Go 错误哲学 ========== // 1. 错误是值 (values), 不是异常 // 2. 显式处理每个错误 (if err != nil) // 3. 添加上下文 (fmt.Errorf %w) // 4. 只处理一次 (打印或返回) // 5. 设计可检查的错误链 (errors.Is/As) // 6. panic 留给真·异常情况
💡 错误处理要点: Sentinel errors (var ErrXxx = errors.New(...)) 用于预期错误; 自定义错误类型实现 Error() 并可选 Unwrap(); fmt.Errorf %w 创建包装错误链; errors.Is 沿链匹配 sentinel error; errors.As 查找链中特定类型; 每条错误只处理一次 (日志或返回); 非关键错误降级; 重试用指数退避 + 不可恢复错误不重试; panic 仅用于不可能发生的错误; HTTP handler 顶层 recover 防止崩溃; defer 中检查 close 错误。
