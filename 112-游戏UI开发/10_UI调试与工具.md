# UI调试与工具

## 核心概念

高效的UI调试工具和方法论能大幅缩短UI开发迭代周期。UI问题通常比游戏逻辑更难定位，因为涉及渲染、布局、事件传递、适配等多个维度。一套完善的调试工具链可以让UI问题的定位时间从几小时缩短到几分钟。

### Unity内置UI调试工具详解

- **Frame Debugger**（Window > Analysis > Frame Debugger）：逐Draw Call查看渲染批次，分析UI合批情况。可以展开每个Batch看到使用的材质、纹理、Shader
- **UI Toolkit Debugger**（Unity 2022+）：可视化UI层级、布局属性和事件传递
- **Profiler UI模块**：监控Canvas.BuildBatch、Canvas.SendWillRenderCanvases耗时，Layout重建耗时
- **RectTransform Debug**：在Inspector中检查锚点、轴心点和偏移设置，可视化锚点区域
- **Scene视图UI模式**：选中Canvas后Scene视图中可看到UI布局边框

### UI性能指标参考

| 指标 | 良好 | 警告 | 严重 |
|------|------|------|------|
| Canvas.BuildBatch | <1ms | 1-5ms | >5ms |
| Layout Rebuild | <0.5ms | 0.5-2ms | >2ms |
| UI Draw Calls | <30 | 30-80 | >80 |
| UI SetPass Calls | <10 | 10-30 | >30 |
| 每帧GC Alloc(UI) | <1KB | 1-10KB | >10KB |

### UI层级可视化调试工具（完整版）

```csharp
#if UNITY_EDITOR
using UnityEditor;
using UnityEngine;

/// <summary>
/// 运行时UI调试覆盖层
/// 显示Canvas列表、Draw Call数量、事件系统状态
/// </summary>
public class UIDebugOverlay : MonoBehaviour
{
    [Header("显示开关")]
    [SerializeField] private bool showCanvasInfo = true;
    [SerializeField] private bool showDrawCalls = true;
    [SerializeField] private bool showEventSystem = true;
    [SerializeField] private bool showFrameTime = true;
    [SerializeField] private bool showMemory = true;

    [Header("设置")]
    [SerializeField] private KeyCode toggleKey = KeyCode.F11;
    [SerializeField] private int fontSize = 14;

    private bool isVisible = true;
    private float[] frameTimeBuffer = new float[60];
    private int frameIndex;
    private GUIStyle boxStyle;
    private GUIStyle labelStyle;

    void Update()
    {
        if (Input.GetKeyDown(toggleKey))
            isVisible = !isVisible;

        // 记录帧时间
        frameTimeBuffer[frameIndex] = Time.unscaledDeltaTime;
        frameIndex = (frameIndex + 1) % frameTimeBuffer.Length;
    }

    void OnGUI()
    {
        if (!isVisible) return;

        InitStyles();

        GUILayout.BeginArea(new Rect(10, 10, 350, 600), boxStyle);

        GUILayout.Label("=== UI Debug Overlay ===", labelStyle);
        GUILayout.Space(5);

        if (showFrameTime)
            DrawFrameTimeInfo();

        if (showCanvasInfo)
            DrawCanvasInfo();

        if (showDrawCalls)
            DrawDrawCallInfo();

        if (showEventSystem)
            DrawEventSystemInfo();

        if (showMemory)
            DrawMemoryInfo();

        GUILayout.EndArea();
    }

    void InitStyles()
    {
        if (boxStyle != null) return;

        boxStyle = new GUIStyle(GUI.skin.box);
        boxStyle.normal.background = MakeTexture(2, 2, new Color(0, 0, 0, 0.7f));
        boxStyle.padding = new RectOffset(10, 10, 10, 10);

        labelStyle = new GUIStyle(GUI.skin.label);
        labelStyle.fontSize = fontSize;
    }

    void DrawFrameTimeInfo()
    {
        float avg = 0, worst = 0;
        for (int i = 0; i < frameTimeBuffer.Length; i++)
        {
            avg += frameTimeBuffer[i];
            if (frameTimeBuffer[i] > worst) worst = frameTimeBuffer[i];
        }
        avg /= frameTimeBuffer.Length;

        float avgFPS = 1f / avg;
        float worstFPS = 1f / worst;

        Color fpsColor = avgFPS >= 55 ? Color.green
            : avgFPS >= 30 ? Color.yellow : Color.red;

        GUI.color = fpsColor;
        GUILayout.Label($"FPS: {avgFPS:F0} (avg) | {worstFPS:F0} (worst)");
        GUILayout.Label($"Frame: {avg * 1000:F1}ms (avg) | {worst * 1000:F1}ms (worst)");
        GUI.color = Color.white;
        GUILayout.Space(5);
    }

    void DrawCanvasInfo()
    {
        GUILayout.Label("[Canvas Info]", EditorStyles.boldLabel);
        Canvas[] canvases = FindObjectsOfType<Canvas>();

        foreach (var canvas in canvases)
        {
            if (!canvas.gameObject.activeInHierarchy) continue;

            int graphicCount = canvas.GetComponentsInChildren<Graphic>(true).Length;
            int childCanvasCount = canvas.GetComponentsInChildren<Canvas>(true).Length - 1;

            GUILayout.Label($"  [{canvas.sortingOrder}] {canvas.name}");
            GUILayout.Label($"    Graphics: {graphicCount} | Sub-Canvases: {childCanvasCount}");
            GUILayout.Label($"    Mode: {canvas.renderMode} | Override: {canvas.overrideSorting}");
        }
        GUILayout.Space(5);
    }

    void DrawDrawCallInfo()
    {
        GUILayout.Label("[Draw Calls]", EditorStyles.boldLabel);
        GUILayout.Label($"  Batches: {UnityEditor.UnityStats.batches}");
        GUILayout.Label($"  SetPass Calls: {UnityEditor.UnityStats.setPassCalls}");
        GUILayout.Label($"  Triangles: {UnityEditor.UnityStats.triangles}");
        GUILayout.Label($"  Vertices: {UnityEditor.UnityStats.vertices}");
        GUILayout.Space(5);
    }

    void DrawEventSystemInfo()
    {
        GUILayout.Label("[Event System]", EditorStyles.boldLabel);
        EventSystem es = EventSystem.current;
        if (es != null)
        {
            GUILayout.Label($"  Active: {es.isActiveAndEnabled}");
            GUILayout.Label($"  Selected: {es.currentSelectedGameObject?.name ?? "None"}");
            GUILayout.Label($"  Module: {es.currentInputModule?.GetType().Name ?? "None"}");
        }
        else
        {
            GUI.color = Color.red;
            GUILayout.Label("  WARNING: No EventSystem found!");
            GUI.color = Color.white;
        }
        GUILayout.Space(5);
    }

    void DrawMemoryInfo()
    {
        GUILayout.Label("[Memory]", EditorStyles.boldLabel);
        GUILayout.Label($"  Total Alloc: {Profiler.GetTotalAllocatedMemoryLong() / (1024 * 1024)}MB");
        GUILayout.Label($"  Mono Heap: {Profiler.GetMonoHeapSizeLong() / (1024 * 1024)}MB");
        GUILayout.Label($"  Mono Used: {Profiler.GetMonoUsedSizeLong() / (1024 * 1024)}MB");
    }

    Texture2D MakeTexture(int width, int height, Color color)
    {
        Color[] pixels = new Color[width * height];
        for (int i = 0; i < pixels.Length; i++) pixels[i] = color;
        Texture2D tex = new Texture2D(width, height);
        tex.SetPixels(pixels);
        tex.Apply();
        return tex;
    }
}
#endif
```

### 性能指标监控（带告警）

```csharp
/// <summary>
/// UI性能监控器
/// 带自动告警和历史记录
/// </summary>
public class UIPerformanceMonitor : MonoBehaviour
{
    [Header("告警阈值")]
    [SerializeField] private float warningFrameTime = 16.67f; // 60fps
    [SerializeField] private float criticalFrameTime = 33.33f; // 30fps
    [SerializeField] private int warningDrawCalls = 50;
    [SerializeField] private int warningBatches = 80;

    private float[] frameTimes = new float[120];
    private int frameIndex;
    private List<string> warnings = new List<string>();

    public float AverageFrameTime { get; private set; }
    public float WorstFrameTime { get; private set; }
    public float BestFrameTime { get; private set; } = float.MaxValue;

    void Update()
    {
        frameTimes[frameIndex] = Time.unscaledDeltaTime;
        frameIndex = (frameIndex + 1) % frameTimes.Length;

        // 计算统计数据
        float sum = 0;
        WorstFrameTime = 0;
        BestFrameTime = float.MaxValue;
        for (int i = 0; i < frameTimes.Length; i++)
        {
            sum += frameTimes[i];
            if (frameTimes[i] > WorstFrameTime) WorstFrameTime = frameTimes[i];
            if (frameTimes[i] < BestFrameTime) BestFrameTime = frameTimes[i];
        }
        AverageFrameTime = sum / frameTimes.Length;

        // 检查告警条件（每秒检查一次）
        if (Time.frameCount % 60 == 0)
            CheckWarnings();
    }

    void CheckWarnings()
    {
        warnings.Clear();

        if (AverageFrameTime * 1000f > criticalFrameTime)
            warnings.Add($"CRITICAL: Frame time {AverageFrameTime * 1000f:F1}ms > {criticalFrameTime}ms");
        else if (AverageFrameTime * 1000f > warningFrameTime)
            warnings.Add($"WARNING: Frame time {AverageFrameTime * 1000f:F1}ms > {warningFrameTime}ms");

#if UNITY_EDITOR
        if (UnityEditor.UnityStats.batches > warningBatches)
            warnings.Add($"WARNING: Batches {UnityEditor.UnityStats.batches} > {warningBatches}");

        if (UnityEditor.UnityStats.setPassCalls > warningDrawCalls)
            warnings.Add($"WARNING: SetPass {UnityEditor.UnityStats.setPassCalls} > {warningDrawCalls}");
#endif

        foreach (var w in warnings)
            Debug.LogWarning($"[UI Perf] {w}");
    }

    /// <summary>
    /// 生成性能报告
    /// </summary>
    public void PrintReport()
    {
        Debug.Log("=== UI Performance Report ===");
        Debug.Log($"Average Frame Time: {AverageFrameTime * 1000f:F2}ms ({1f / AverageFrameTime:F0} FPS)");
        Debug.Log($"Worst Frame Time: {WorstFrameTime * 1000f:F2}ms ({1f / WorstFrameTime:F0} FPS)");
        Debug.Log($"Best Frame Time: {BestFrameTime * 1000f:F2}ms ({1f / BestFrameTime:F0} FPS)");

#if UNITY_EDITOR
        Debug.Log($"Batches: {UnityEditor.UnityStats.batches}");
        Debug.Log($"SetPass Calls: {UnityEditor.UnityStats.setPassCalls}");
        Debug.Log($"Triangles: {UnityEditor.UnityStats.triangles}");
#endif
    }
}
```

### 自动化UI测试

```csharp
using NUnit.Framework;
using System.Collections;
using UnityEngine;
using UnityEngine.TestTools;

/// <summary>
/// UI自动化测试套件
/// 使用Unity Test Runner进行UI测试
/// </summary>
[TestFixture]
public class InventoryUITests
{
    private GameObject testRoot;
    private InventoryUI inventoryUI;

    [SetUp]
    public void Setup()
    {
        // 创建测试环境
        testRoot = new GameObject("TestRoot");
        inventoryUI = testRoot.AddComponent<InventoryUI>();
    }

    [TearDown]
    public void Teardown()
    {
        // 清理测试环境，防止测试间相互干扰
        Object.DestroyImmediate(testRoot);
    }

    [Test]
    public void TestInventorySlotBinding()
    {
        // 创建测试用的背包格子
        var slotGo = new GameObject("TestSlot");
        slotGo.transform.SetParent(testRoot.transform);
        var slot = slotGo.AddComponent<InventorySlotUI>();

        var itemDef = ScriptableObject.CreateInstance<ItemDefinition>();
        itemDef.itemId = "sword_01";
        itemDef.itemName = "Iron Sword";
        itemDef.rarity = Rarity.Rare;

        var itemInstance = new ItemInstance
        {
            instanceId = "test_001",
            definition = itemDef,
            stackCount = 1,
            slotIndex = 0
        };

        slot.SetItem(itemInstance);

        // 验证
        Assert.AreEqual(itemInstance, slot.CurrentItem);
        Assert.AreEqual("Iron Sword", itemDef.itemName);
    }

    [Test]
    public void TestInventoryDataAddItem()
    {
        var inventory = new InventoryData(20);
        var itemDef = ScriptableObject.CreateInstance<ItemDefinition>();
        itemDef.itemId = "potion_hp";
        itemDef.itemName = "HP Potion";
        itemDef.maxStack = 10;

        int remaining = inventory.AddItem(itemDef, 5);
        Assert.AreEqual(0, remaining); // 全部添加成功
        Assert.AreEqual(5, inventory.CountItem("potion_hp"));
    }

    [Test]
    public void TestInventoryDataStackOverflow()
    {
        var inventory = new InventoryData(2);
        var itemDef = ScriptableObject.CreateInstance<ItemDefinition>();
        itemDef.itemId = "arrow";
        itemDef.maxStack = 5;

        int remaining = inventory.AddItem(itemDef, 12);
        Assert.AreEqual(2, remaining); // 2个格子各5个，剩余2个
        Assert.AreEqual(10, inventory.CountItem("arrow"));
    }

    [UnityTest]
    public IEnumerator TestPopupAnimation()
    {
        // 创建测试弹窗
        var popupGo = new GameObject("TestPopup");
        popupGo.transform.SetParent(testRoot.transform);
        var popup = popupGo.AddComponent<PopupAnimator>();

        // 播放动画
        bool completed = false;
        popup.PlayShow(() => completed = true);

        // 等待动画完成
        float timer = 0;
        while (!completed && timer < 2f)
        {
            timer += Time.deltaTime;
            yield return null;
        }

        // 验证
        Assert.IsTrue(completed, "Animation should complete within 2 seconds");
        Assert.IsTrue(popupGo.activeInHierarchy);
    }

    [UnityTest]
    public IEnumerator TestVirtualListScrolling()
    {
        // 测试虚拟列表滚动时不产生GC
        var listGo = new GameObject("TestList");
        listGo.transform.SetParent(testRoot.transform);

        long gcBefore = System.GC.GetTotalMemory(false);

        // 模拟滚动操作
        yield return new WaitForSeconds(0.5f);

        long gcAfter = System.GC.GetTotalMemory(false);
        long gcAllocated = gcAfter - gcBefore;

        Assert.Less(gcAllocated, 1024,
            $"Virtual list scrolling should not allocate more than 1KB, but allocated {gcAllocated} bytes");
    }
}
```

### 自定义EditorWindow UI工具

```csharp
#if UNITY_EDITOR
using UnityEditor;
using UnityEngine;

/// <summary>
/// UI层级查看器EditorWindow
/// 运行时实时查看UI层级、Raycast状态、合批信息
/// </summary>
public class UIHierarchyWindow : EditorWindow
{
    [MenuItem("Tools/UI Hierarchy Viewer")]
    static void Open() => GetWindow<UIHierarchyWindow>("UI Hierarchy");

    private Vector2 scrollPos;
    private bool showRaycastOnly = false;
    private string searchFilter = "";

    void OnGUI()
    {
        if (!Application.isPlaying)
        {
            EditorGUILayout.HelpBox("请在运行模式下使用此工具", MessageType.Info);
            return;
        }

        // 工具栏
        EditorGUILayout.BeginHorizontal(EditorStyles.toolbar);
        showRaycastOnly = GUILayout.Toggle(showRaycastOnly, "仅Raycast",
            EditorStyles.toolbarButton);
        searchFilter = GUILayout.TextField(searchFilter, EditorStyles.toolbarSearchField,
            GUILayout.Width(200));
        if (GUILayout.Button("Refresh", EditorStyles.toolbarButton, GUILayout.Width(60)))
            Repaint();
        EditorGUILayout.EndHorizontal();

        // 统计信息
        Canvas[] canvases = FindObjectsOfType<Canvas>();
        EditorGUILayout.HelpBox(
            $"Canvases: {canvases.Length} | " +
            $"Batches: {UnityEditor.UnityStats.batches} | " +
            $"SetPass: {UnityEditor.UnityStats.setPassCalls}",
            MessageType.None);

        // 层级树
        scrollPos = EditorGUILayout.BeginScrollView(scrollPos);
        foreach (var canvas in canvases)
        {
            if (!string.IsNullOrEmpty(searchFilter)
                && !canvas.name.ToLower().Contains(searchFilter.ToLower()))
                continue;

            DrawCanvasNode(canvas);
        }
        EditorGUILayout.EndScrollView();
    }

    void DrawCanvasNode(Canvas canvas)
    {
        int graphicCount = canvas.GetComponentsInChildren<Graphic>(true).Length;
        int raycastCount = 0;
        foreach (var g in canvas.GetComponentsInChildren<Graphic>(true))
            if (g.raycastTarget) raycastCount++;

        string label = $"[S:{canvas.sortingOrder}] {canvas.name} " +
            $"({graphicCount}G, {raycastCount}RT)";

        EditorGUILayout.BeginHorizontal();
        EditorGUILayout.ObjectField(canvas, typeof(Canvas), true);
        EditorGUILayout.LabelField(label, GUILayout.Width(300));
        EditorGUILayout.EndHorizontal();

        EditorGUI.indentLevel++;
        DrawChildInfo(canvas.transform);
        EditorGUI.indentLevel--;
    }

    void DrawChildInfo(Transform parent)
    {
        foreach (Transform child in parent)
        {
            if (child.GetComponent<Canvas>() != null && child != parent) continue; // 跳过子Canvas

            Graphic graphic = child.GetComponent<Graphic>();
            if (showRaycastOnly && (graphic == null || !graphic.raycastTarget))
                continue;

            string info = child.name;
            if (graphic != null)
            {
                string raycastIcon = graphic.raycastTarget ? "[R]" : "[ ]";
                string materialInfo = graphic.material != null
                    ? graphic.material.name : "Default";
                info = $"{raycastIcon} {info} ({materialInfo})";
            }

            EditorGUILayout.ObjectField(child, typeof(Transform), true);

            if (child.childCount > 0)
            {
                // 可折叠
                child.SetFoldout(EditorGUILayout.Foldout(
                    child.GetFoldout(), info));
                if (child.GetFoldout())
                {
                    EditorGUI.indentLevel++;
                    DrawChildInfo(child);
                    EditorGUI.indentLevel--;
                }
            }
            else
            {
                EditorGUI.indentLevel++;
                EditorGUILayout.LabelField(info);
                EditorGUI.indentLevel--;
            }
        }
    }
}

// 扩展方法：折叠状态
public static class TransformExtensions
{
    private static Dictionary<int, bool> foldoutStates = new Dictionary<int, bool>();

    public static bool GetFoldout(this Transform t)
    {
        return foldoutStates.TryGetValue(t.GetInstanceID(), out bool v) && v;
    }

    public static void SetFoldout(this Transform t, bool value)
    {
        foldoutStates[t.GetInstanceID()] = value;
    }
}
#endif
```

## 最佳实践

- 开发阶段使用EditorOnly的调试Overlay，发布时自动剔除（#if UNITY_EDITOR）
- 定期使用Frame Debugger检查UI Draw Call，目标控制在50以下
- 自动化测试覆盖核心UI流程（打开/关闭/交互/数据绑定/性能回归）
- 对UI性能指标建立基准线，回归时自动告警
- 使用Hierarchy窗口的搜索功能快速定位问题UI元素
- 为Canvas命名规范（Canvas_HUD, Canvas_Popup等），便于调试识别
- 运行时调试工具使用快捷键开关，避免影响正常游戏体验

## 常见陷阱与修复

**陷阱1：调试工具本身引入性能开销**
- 症状：开启调试工具后帧率下降5-10fps
- 修复：调试工具的OnGUI每帧调用代价高，使用EditorWindow代替运行时OnGUI，或降低刷新频率

**陷阱2：自动化测试未清理场景状态导致测试间相互干扰**
- 症状：单独运行测试A和B都通过，一起运行就失败
- 修复：使用[TearDown]清理所有测试创建的对象，确保测试独立

**陷阱3：仅在Editor中测试UI，未在真机上验证适配效果**
- 症状：Editor中看起来正常的UI在真机上布局错乱
- 修复：定期在真机上测试，关注安全区域、DPI、横竖屏等差异

**陷阱4：未关注Canvas.BuildBatch耗时**
- 症状：UI总感觉卡顿但不知道原因
- 修复：在Profiler中专门查看Canvas.BuildBatch耗时，结合Frame Debugger分析打断原因
