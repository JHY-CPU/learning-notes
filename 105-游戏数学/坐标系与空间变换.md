# 坐标系与空间变换

## 核心概念

游戏渲染管线涉及多个坐标空间的变换，理解每个空间的作用是图形编程的基础。从模型的局部坐标到最终屏幕像素，顶点需要经历一系列线性变换（矩阵乘法）。

### 坐标空间层级

1. **模型空间（Local/Object Space）**：物体自身的局部坐标系，原点通常在模型中心或底部
2. **世界空间（World Space）**：整个场景的统一坐标系，所有物体共用
3. **视图空间（View/Camera Space）**：以相机为原点的坐标系，z轴指向相机前方
4. **裁剪空间（Clip Space）**：投影变换后的空间，w分量用于裁剪
5. **NDC（Normalized Device Coordinates）**：裁剪后除以w，范围[-1, 1]
6. **屏幕/视口空间（Screen/Viewport Space）**：最终像素坐标

### 变换流程

```
模型空间 --[Model矩阵]--> 世界空间 --[View矩阵]--> 视图空间
    --[Projection矩阵]--> 裁剪空间 --[透视除法]--> NDC
    --[视口变换]--> 屏幕空间

MVP矩阵 = Projection × View × Model
顶点着色器: gl_Position = MVP * vec4(position, 1.0);
```

### 齐次裁剪

裁剪空间中，当顶点满足 `-w <= x,y,z <= w` 时在视锥体内，否则被裁剪。w的正负区分前后半空间。GPU的裁剪单元在线段级别执行裁剪，生成新的顶点。

### NDC到屏幕映射

```
screenX = (ndcX + 1) / 2 * viewportWidth
screenY = (1 - ndcY) / 2 * viewportHeight  // Y轴翻转（NDC的y向上，屏幕的y向下）
```

## 完整数学库实现

```cpp
#include <cmath>
#include <cstring>

const float PI = 3.14159265358979f;
const float DEG2RAD = PI / 180.0f;

// ========== Vec2 ==========
struct Vec2 {
    float x, y;
    Vec2() : x(0), y(0) {}
    Vec2(float x, float y) : x(x), y(y) {}
    Vec2 operator+(const Vec2& o) const { return {x+o.x, y+o.y}; }
    Vec2 operator-(const Vec2& o) const { return {x-o.x, y-o.y}; }
    Vec2 operator*(float s) const { return {x*s, y*s}; }
    float dot(const Vec2& o) const { return x*o.x + y*o.y; }
    float length() const { return sqrt(x*x + y*y); }
};

// ========== Vec3 ==========
struct Vec3 {
    float x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(float s) const { return {x*s, y*s, z*s}; }
    Vec3 operator-() const { return {-x, -y, -z}; }
    float dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    float lengthSq() const { return x*x + y*y + z*z; }
    float length() const { return sqrt(lengthSq()); }
    Vec3 normalized() const {
        float l = length();
        return l > 1e-6f ? Vec3{x/l, y/l, z/l} : Vec3{0, 0, 0};
    }
    Vec3 lerp(const Vec3& b, float t) const {
        return {x + (b.x-x)*t, y + (b.y-y)*t, z + (b.z-z)*t};
    }
};

// ========== Vec4 ==========
struct Vec4 {
    float x, y, z, w;
    Vec4() : x(0), y(0), z(0), w(0) {}
    Vec4(float x, float y, float z, float w) : x(x), y(y), z(z), w(w) {}
    Vec4(Vec3 v, float w) : x(v.x), y(v.y), z(v.z), w(w) {}
};

// ========== Mat4 (4x4矩阵，列主序) ==========
struct Mat4 {
    float m[4][4]; // m[column][row]

    Mat4() { memset(m, 0, sizeof(m)); }

    static Mat4 identity() {
        Mat4 r;
        r.m[0][0] = r.m[1][1] = r.m[2][2] = r.m[3][3] = 1;
        return r;
    }

    // 矩阵乘法
    Mat4 operator*(const Mat4& b) const {
        Mat4 r;
        for (int col = 0; col < 4; col++)
            for (int row = 0; row < 4; row++) {
                r.m[col][row] = 0;
                for (int k = 0; k < 4; k++)
                    r.m[col][row] += m[k][row] * b.m[col][k];
            }
        return r;
    }

    // 矩阵-向量乘法（齐次坐标）
    Vec4 operator*(const Vec4& v) const {
        return {
            m[0][0]*v.x + m[1][0]*v.y + m[2][0]*v.z + m[3][0]*v.w,
            m[0][1]*v.x + m[1][1]*v.y + m[2][1]*v.z + m[3][1]*v.w,
            m[0][2]*v.x + m[1][2]*v.y + m[2][2]*v.z + m[3][2]*v.w,
            m[0][3]*v.x + m[1][3]*v.y + m[2][3]*v.z + m[3][3]*v.w
        };
    }

    // 转置
    Mat4 transposed() const {
        Mat4 r;
        for (int i = 0; i < 4; i++)
            for (int j = 0; j < 4; j++)
                r.m[i][j] = m[j][i];
        return r;
    }

    // 求逆（通用4x4矩阵求逆）
    Mat4 inverse() const {
        Mat4 inv;
        float* src = (float*)m;
        float* dst = (float*)inv.m;

        // 计算余子式
        float c[16];
        c[0]  = src[5]*src[10]*src[15] - src[5]*src[11]*src[14] - src[9]*src[6]*src[15] + src[9]*src[7]*src[14] + src[13]*src[6]*src[11] - src[13]*src[7]*src[10];
        c[1]  = -src[1]*src[10]*src[15] + src[1]*src[11]*src[14] + src[9]*src[2]*src[15] - src[9]*src[3]*src[14] - src[13]*src[2]*src[11] + src[13]*src[3]*src[10];
        c[2]  = src[1]*src[6]*src[15] - src[1]*src[7]*src[14] - src[5]*src[2]*src[15] + src[5]*src[3]*src[14] + src[13]*src[2]*src[7] - src[13]*src[3]*src[6];
        c[3]  = -src[1]*src[6]*src[11] + src[1]*src[7]*src[10] + src[5]*src[2]*src[11] - src[5]*src[3]*src[10] - src[9]*src[2]*src[7] + src[9]*src[3]*src[6];
        c[4]  = -src[4]*src[10]*src[15] + src[4]*src[11]*src[14] + src[8]*src[6]*src[15] - src[8]*src[7]*src[14] - src[12]*src[6]*src[11] + src[12]*src[7]*src[10];
        c[5]  = src[0]*src[10]*src[15] - src[0]*src[11]*src[14] - src[8]*src[2]*src[15] + src[8]*src[3]*src[14] + src[12]*src[2]*src[11] - src[12]*src[3]*src[10];
        c[6]  = -src[0]*src[6]*src[15] + src[0]*src[7]*src[14] + src[4]*src[2]*src[15] - src[4]*src[3]*src[14] - src[12]*src[2]*src[7] + src[12]*src[3]*src[6];
        c[7]  = src[0]*src[6]*src[11] - src[0]*src[7]*src[10] - src[4]*src[2]*src[11] + src[4]*src[3]*src[10] + src[8]*src[2]*src[7] - src[8]*src[3]*src[6];
        c[8]  = src[4]*src[9]*src[15] - src[4]*src[11]*src[13] - src[8]*src[5]*src[15] + src[8]*src[7]*src[13] + src[12]*src[5]*src[11] - src[12]*src[7]*src[9];
        c[9]  = -src[0]*src[9]*src[15] + src[0]*src[11]*src[13] + src[8]*src[1]*src[15] - src[8]*src[3]*src[13] - src[12]*src[1]*src[11] + src[12]*src[3]*src[9];
        c[10] = src[0]*src[5]*src[15] - src[0]*src[7]*src[13] - src[4]*src[1]*src[15] + src[4]*src[3]*src[13] + src[12]*src[1]*src[7] - src[12]*src[3]*src[5];
        c[11] = -src[0]*src[5]*src[11] + src[0]*src[7]*src[9] + src[4]*src[1]*src[11] - src[4]*src[3]*src[9] - src[8]*src[1]*src[7] + src[8]*src[3]*src[5];
        c[12] = -src[4]*src[9]*src[14] + src[4]*src[10]*src[13] + src[8]*src[5]*src[14] - src[8]*src[6]*src[13] - src[12]*src[5]*src[10] + src[12]*src[6]*src[9];
        c[13] = src[0]*src[9]*src[14] - src[0]*src[10]*src[13] - src[8]*src[1]*src[14] + src[8]*src[2]*src[13] + src[12]*src[1]*src[10] - src[12]*src[2]*src[9];
        c[14] = -src[0]*src[5]*src[14] + src[0]*src[6]*src[13] + src[4]*src[1]*src[14] - src[4]*src[2]*src[13] - src[12]*src[1]*src[6] + src[12]*src[2]*src[5];
        c[15] = src[0]*src[5]*src[10] - src[0]*src[6]*src[9] - src[4]*src[1]*src[10] + src[4]*src[2]*src[9] + src[8]*src[1]*src[6] - src[8]*src[2]*src[5];

        float det = src[0]*c[0] + src[1]*c[4] + src[2]*c[8] + src[3]*c[12];
        if (abs(det) < 1e-10f) return Mat4::identity();

        float invDet = 1.0f / det;
        for (int i = 0; i < 16; i++) dst[i] = c[i] * invDet;
        return inv;
    }
};

// ========== Mat3 ==========
struct Mat3 {
    float m[3][3];
    Mat3() { memset(m, 0, sizeof(m)); }
    static Mat3 identity() {
        Mat3 r;
        r.m[0][0] = r.m[1][1] = r.m[2][2] = 1;
        return r;
    }
    Vec3 operator*(const Vec3& v) const {
        return {
            m[0][0]*v.x + m[1][0]*v.y + m[2][0]*v.z,
            m[0][1]*v.x + m[1][1]*v.y + m[2][1]*v.z,
            m[0][2]*v.x + m[1][2]*v.y + m[2][2]*v.z
        };
    }
};
```

## 变换矩阵构造

```cpp
// ========== 平移矩阵 ==========
Mat4 translationMatrix(float tx, float ty, float tz) {
    Mat4 m = Mat4::identity();
    m.m[3][0] = tx;
    m.m[3][1] = ty;
    m.m[3][2] = tz;
    return m;
}

// ========== 缩放矩阵 ==========
Mat4 scaleMatrix(float sx, float sy, float sz) {
    Mat4 m;
    m.m[0][0] = sx;
    m.m[1][1] = sy;
    m.m[2][2] = sz;
    m.m[3][3] = 1;
    return m;
}

// ========== 旋转矩阵（绕各轴） ==========
Mat4 rotationX(float angleRad) {
    Mat4 m = Mat4::identity();
    float c = cos(angleRad), s = sin(angleRad);
    m.m[1][1] = c;  m.m[2][1] = -s;
    m.m[1][2] = s;  m.m[2][2] = c;
    return m;
}

Mat4 rotationY(float angleRad) {
    Mat4 m = Mat4::identity();
    float c = cos(angleRad), s = sin(angleRad);
    m.m[0][0] = c;  m.m[2][0] = s;
    m.m[0][2] = -s; m.m[2][2] = c;
    return m;
}

Mat4 rotationZ(float angleRad) {
    Mat4 m = Mat4::identity();
    float c = cos(angleRad), s = sin(angleRad);
    m.m[0][0] = c;  m.m[1][0] = -s;
    m.m[0][1] = s;  m.m[1][1] = c;
    return m;
}

// 绕任意轴旋转
Mat4 rotationAxisAngle(Vec3 axis, float angleRad) {
    axis = axis.normalized();
    float c = cos(angleRad), s = sin(angleRad), t = 1 - c;
    Mat4 m = Mat4::identity();
    m.m[0][0] = t*axis.x*axis.x + c;
    m.m[0][1] = t*axis.x*axis.y + s*axis.z;
    m.m[0][2] = t*axis.x*axis.z - s*axis.y;
    m.m[1][0] = t*axis.x*axis.y - s*axis.z;
    m.m[1][1] = t*axis.y*axis.y + c;
    m.m[1][2] = t*axis.y*axis.z + s*axis.x;
    m.m[2][0] = t*axis.x*axis.z + s*axis.y;
    m.m[2][1] = t*axis.y*axis.z - s*axis.x;
    m.m[2][2] = t*axis.z*axis.z + c;
    return m;
}

// ========== 完整变换（TRS：平移×旋转×缩放） ==========
Mat4 trsMatrix(Vec3 translation, Vec3 eulerRadians, Vec3 scale) {
    Mat4 T = translationMatrix(translation.x, translation.y, translation.z);
    Mat4 Rx = rotationX(eulerRadians.x);
    Mat4 Ry = rotationY(eulerRadians.y);
    Mat4 Rz = rotationZ(eulerRadians.z);
    Mat4 S = scaleMatrix(scale.x, scale.y, scale.z);

    // TRS = T * Rz * Ry * Rx * S
    return T * Rz * Ry * Rx * S;
}
```

## 视图矩阵与投影矩阵

```cpp
// ========== LookAt视图矩阵 ==========
Mat4 lookAt(Vec3 eye, Vec3 target, Vec3 up) {
    Vec3 f = (target - eye).normalized();   // forward
    Vec3 r = f.cross(up).normalized();      // right
    Vec3 u = r.cross(f);                    // up (重新计算确保正交)

    Mat4 m = Mat4::identity();
    m.m[0][0] = r.x;  m.m[1][0] = r.y;  m.m[2][0] = r.z;
    m.m[0][1] = u.x;  m.m[1][1] = u.y;  m.m[2][1] = u.z;
    m.m[0][2] = -f.x; m.m[1][2] = -f.y; m.m[2][2] = -f.z;
    m.m[3][0] = -r.dot(eye);
    m.m[3][1] = -u.dot(eye);
    m.m[3][2] = f.dot(eye);
    return m;
}

// ========== 透视投影矩阵 ==========
Mat4 perspective(float fovYRad, float aspect, float nearZ, float farZ) {
    float tanHalf = tan(fovYRad * 0.5f);
    Mat4 m; // 全零初始化

    m.m[0][0] = 1.0f / (aspect * tanHalf);
    m.m[1][1] = 1.0f / tanHalf;
    m.m[2][2] = -(farZ + nearZ) / (farZ - nearZ);
    m.m[2][3] = -1.0f;
    m.m[3][2] = -2.0f * farZ * nearZ / (farZ - nearZ);

    return m;
}

// ========== 正交投影矩阵 ==========
Mat4 ortho(float left, float right, float bottom, float top,
           float nearZ, float farZ) {
    Mat4 m = Mat4::identity();
    m.m[0][0] = 2.0f / (right - left);
    m.m[1][1] = 2.0f / (top - bottom);
    m.m[2][2] = -2.0f / (farZ - nearZ);
    m.m[3][0] = -(right + left) / (right - left);
    m.m[3][1] = -(top + bottom) / (top - bottom);
    m.m[3][2] = -(farZ + nearZ) / (farZ - nearZ);
    return m;
}
```

## 完整变换管线

```cpp
// ========== 完整的坐标变换管线 ==========
struct TransformPipeline {
    Mat4 model, view, projection;
    int viewportW, viewportH;

    // 模型空间 → 屏幕空间
    Vec2 worldToScreen(Vec3 worldPos) {
        Vec4 worldH = {worldPos.x, worldPos.y, worldPos.z, 1.0f};

        // MVP变换
        Vec4 clip = projection * view * model * worldH;

        // 透视除法 → NDC
        if (abs(clip.w) < 1e-6f) clip.w = 1e-6f;
        Vec3 ndc = {clip.x/clip.w, clip.y/clip.w, clip.z/clip.w};

        // 视口变换
        return {
            (ndc.x + 1.0f) * 0.5f * viewportW,
            (1.0f - ndc.y) * 0.5f * viewportH
        };
    }

    // 屏幕空间 → 世界空间射线
    Ray screenToWorldRay(float sx, float sy) {
        Mat4 invVP = (projection * view).inverse();

        // 屏幕坐标转NDC
        float nx = (2.0f * sx / viewportW) - 1.0f;
        float ny = 1.0f - (2.0f * sy / viewportH);

        // NDC转世界空间
        Vec4 nearH = invVP * Vec4{nx, ny, -1.0f, 1.0f};
        Vec4 farH  = invVP * Vec4{nx, ny,  1.0f, 1.0f};

        Vec3 nearP = {nearH.x/nearH.w, nearH.y/nearH.w, nearH.z/nearH.w};
        Vec3 farP  = {farH.x/farH.w, farH.y/farH.w, farH.z/farH.w};

        return {nearP, (farP - nearP).normalized()};
    }

    // 深度值还原世界坐标
    Vec3 screenToWorld(float sx, float sy, float depth) {
        Mat4 invVP = (projection * view).inverse();
        float nx = (2.0f * sx / viewportW) - 1.0f;
        float ny = 1.0f - (2.0f * sy / viewportH);
        float ndcZ = depth * 2.0f - 1.0f;

        Vec4 worldH = invVP * Vec4{nx, ny, ndcZ, 1.0f};
        return {worldH.x/worldH.w, worldH.y/worldH.w, worldH.z/worldH.w};
    }

    // 世界空间 → NDC（用于深度排序等）
    Vec3 worldToNDC(Vec3 worldPos) {
        Vec4 clip = projection * view * Vec4{worldPos.x, worldPos.y, worldPos.z, 1.0f};
        if (abs(clip.w) < 1e-6f) return {0, 0, 0};
        return {clip.x/clip.w, clip.y/clip.w, clip.z/clip.w};
    }
};

struct Ray {
    Vec3 origin, direction;
};
```

## 左手系 vs 右手系

```
左手系 (Unity, DirectX):
  +X = 右
  +Y = 上
  +Z = 前（看向屏幕内部）
  叉积: right × up = forward

右手系 (OpenGL, Blender):
  +X = 右
  +Y = 上
  +Z = 后（看向屏幕外部）
  叉积: right × up = -forward

矩阵差异:
- View矩阵中z轴取反
- Projection矩阵中z范围和符号不同
- 旋转方向: 左手系顺时针为正，右手系逆时针为正
```

## 游戏应用场景

1. **HUD叠加**：将3D世界坐标（如血条位置）投影到2D屏幕
2. **点击选中**：屏幕鼠标坐标反投影到世界空间射线，做碰撞检测
3. **阴影映射**：从光源视角变换到光的裁剪空间生成深度图
4. **后处理**：NDC空间下的全屏渲染
5. **粒子系统**：Billboard广告牌效果使用视图空间变换
6. **LOD选择**：根据物体到相机的屏幕空间大小选择LOD级别

## 常见陷阱

- **左手系vs右手系**：Unity左手系（z前），OpenGL右手系（z后），影响矩阵公式和旋转方向
- **透视除法遗漏**：裁剪到NDC必须除以w，遗漏会导致严重渲染错误
- **近裁剪面**：near值不能为0（推荐0.1-1.0），否则深度精度完全丢失。near=0.1, far=1000时深度精度最优
- **矩阵更新频率**：模型矩阵每帧每物体更新，投影矩阵通常只在窗口大小改变时更新
- **列主序vs行主序**：Unity/GLM使用列主序，DirectX传统使用行主序，矩阵乘法顺序相反
- **矩阵求逆精度**：病态矩阵（行列式接近零）的逆矩阵可能不精确
- **深度缓冲非线性**：透视投影的深度值是非线性的，近处精度高远处精度低
