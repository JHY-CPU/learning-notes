# 随机数与概率在游戏中的应用

## 核心概念

游戏中的随机数用于创建不可预测但可控的游戏体验，包括战斗结果、掉落系统、世界生成等。好的随机数实现需要兼顾性能、分布质量和可控性。

### 伪随机数生成器（PRNG）

计算机使用确定性算法生成看似随机的数列。相同种子产生相同序列，这对回放和联机同步至关重要。

```
随机数生成器对比:
- 线性同余法(LCG): 简单快速，周期短(~2^32)，质量差
- Mersenne Twister: 周期长(~2^19937)，分布均匀，内存占用大(2.5KB)
- XorShift: 极快，周期可配置(32/64/128位)，适合游戏
- PCG: 现代高质量RNG，通过统计测试，速度快
- SplitMix64: 极简实现，质量好，无状态碰撞问题
```

### 权重随机

从多个选项中按概率抽取。经典方案是累积权重+二分查找。

### 保底系统（Pity System）

避免玩家运气过差：连续N次未获得稀有物品后，概率逐步提升。广泛用于抽卡游戏（原神、FGO等）。

### Perlin噪声

连续平滑的伪随机函数，用于地形生成、云彩纹理等。由Ken Perlin在1983年发明，后来他提出了改进版Simplex噪声。

```cpp
#include <cstdint>
#include <cmath>
#include <vector>

// ========== XorShift32 - 极快的伪随机数生成器 ==========
class XorShift32 {
    uint32_t state;
public:
    XorShift32(uint32_t seed = 12345) : state(seed ? seed : 1) {}

    // 核心生成函数：3次异或移位操作
    uint32_t next() {
        state ^= state << 13;
        state ^= state >> 17;
        state ^= state << 5;
        return state;
    }

    float random01() { return next() / (float)UINT32_MAX; }
    float range(float min, float max) {
        return min + random01() * (max - min);
    }
    int range(int min, int max) {
        return min + (int)(next() % (uint32_t)(max - min + 1));
    }

    // 高斯分布（Box-Muller变换）
    float gaussian(float mean = 0, float stddev = 1) {
        float u1 = random01();
        float u2 = random01();
        if (u1 < 1e-10f) u1 = 1e-10f; // 避免log(0)
        float z0 = sqrt(-2.0f * log(u1)) * cos(2.0f * M_PI * u2);
        return mean + z0 * stddev;
    }
};

// ========== PCG-XSH-RR - 高质量随机数生成器 ==========
class PCG32 {
    uint64_t state;
    uint64_t inc;
public:
    PCG32(uint64_t seed = 0x853c49e6748fea9bULL, uint64_t seq = 0xda3e39cb94b95bdbULL) {
        state = 0;
        inc = (seq << 1) | 1;
        next();
        state += seed;
        next();
    }

    uint32_t next() {
        uint64_t oldstate = state;
        state = oldstate * 6364136223846793005ULL + inc;
        uint32_t xorshifted = (uint32_t)(((oldstate >> 18) ^ oldstate) >> 27);
        uint32_t rot = (uint32_t)(oldstate >> 59);
        return (xorshifted >> rot) | (xorshifted << ((-rot) & 31));
    }

    float random01() { return next() / (float)UINT32_MAX; }
    int range(int min, int max) {
        return min + (int)(next() % (uint32_t)(max - min + 1));
    }
};

// ========== SplitMix64 - 无状态碰撞的种子混排器 ==========
class SplitMix64 {
    uint64_t state;
public:
    SplitMix64(uint64_t seed) : state(seed) {}

    uint64_t next() {
        uint64_t z = (state += 0x9e3779b97f4a7c15ULL);
        z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
        z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
        return z ^ (z >> 31);
    }
};
```

## 权重随机选择系统

```cpp
// ========== 累积权重方案 ==========
struct WeightedItem {
    int id;
    float weight;
    std::string name;
};

class WeightedRandom {
    std::vector<float> cumulativeWeights;
    float totalWeight;

public:
    // 预计算累积权重（O(n)一次，之后O(log n)查找）
    void build(const std::vector<WeightedItem>& items) {
        cumulativeWeights.resize(items.size());
        totalWeight = 0;
        for (size_t i = 0; i < items.size(); i++) {
            totalWeight += items[i].weight;
            cumulativeWeights[i] = totalWeight;
        }
    }

    // 二分查找选择（O(log n)）
    int select(XorShift32& rng) {
        float r = rng.random01() * totalWeight;

        int lo = 0, hi = (int)cumulativeWeights.size() - 1;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            if (cumulativeWeights[mid] < r)
                lo = mid + 1;
            else
                hi = mid;
        }
        return lo;
    }

    // 多次不放回抽取
    std::vector<int> selectMultiple(XorShift32& rng, int count,
        const std::vector<WeightedItem>& items) {
        // 复制权重数组，移除已选中的项
        std::vector<float> weights;
        for (auto& item : items) weights.push_back(item.weight);

        std::vector<int> result;
        for (int i = 0; i < count && !weights.empty(); i++) {
            float total = 0;
            for (float w : weights) total += w;

            float r = rng.random01() * total;
            float cumulative = 0;
            for (size_t j = 0; j < weights.size(); j++) {
                cumulative += weights[j];
                if (r <= cumulative) {
                    result.push_back(j);
                    weights.erase(weights.begin() + j);
                    break;
                }
            }
        }
        return result;
    }
};
```

## 掉落系统与保底机制

```cpp
// ========== 基础掉率系统 ==========
struct DropEntry {
    int itemId;
    float baseRate;      // 基础掉率（如0.006 = 0.6%）
    int rarity;          // 稀有度等级
};

// 保底系统实现
class PityDropSystem {
    struct PityState {
        int counter = 0;           // 保底计数器
        int softPityStart = 75;    // 软保底开始（如75抽）
        int hardPity = 90;         // 硬保底（如90抽必出）
        float baseRate = 0.006f;   // 基础掉率0.6%
        float softPityIncrement = 0.06f; // 软保底后每次增加6%
    };

    PityState pityState;
    XorShift32 rng;

public:
    PityDropSystem(uint32_t seed) : rng(seed) {}

    // 单次抽取
    bool roll() {
        pityState.counter++;
        float currentRate = pityState.baseRate;

        // 硬保底：到了必出
        if (pityState.counter >= pityState.hardPity) {
            pityState.counter = 0;
            return true;
        }

        // 软保底：概率逐步提升
        if (pityState.counter >= pityState.softPityStart) {
            int overSoftPity = pityState.counter - pityState.softPityStart;
            currentRate += overSoftPity * pityState.softPityIncrement;
        }

        bool dropped = rng.random01() < currentRate;
        if (dropped) {
            pityState.counter = 0;
        }
        return dropped;
    }

    // 批量抽取模拟
    struct SimResult {
        int totalRolls;
        int fiveStarCount;
        std::vector<int> rollDistances; // 每个5星之间的抽数
    };

    SimResult simulate(int numRolls) {
        SimResult result = {};
        int lastFiveStar = 0;

        for (int i = 0; i < numRolls; i++) {
            if (roll()) {
                result.fiveStarCount++;
                result.rollDistances.push_back(i - lastFiveStar + 1);
                lastFiveStar = i;
            }
        }
        result.totalRolls = numRolls;
        return result;
    }
};

// ========== 伪随机分布(PRD) - 暴击系统 ==========
// 用于Dota2等游戏的暴击系统，确保暴击率趋近期望值
class PRDCritSystem {
    float C;  // PRD常数（由期望概率计算）
    int noCritCount = 0;

public:
    // 从期望概率计算PRD常数
    // 使用近似公式: C ≈ P_expected * (1.0 / 期望连续次数)
    PRDCritSystem(float expectedProbability) {
        // 迭代求解精确值（简化版）
        C = expectedProbability * 0.7f; // 近似值
        // 精确解需要数值方法：对每个P找到C使得期望触发间隔 = 1/P
    }

    bool checkCrit(XorShift32& rng) {
        float currentProb = C * (noCritCount + 1);
        bool crit = rng.random01() < currentProb;
        if (crit)
            noCritCount = 0;
        else
            noCritCount++;
        return crit;
    }
};
```

## Fisher-Yates洗牌算法

```cpp
// 真正均匀的随机洗牌，O(n)时间复杂度
template<typename T>
void fisherYatesShuffle(std::vector<T>& array, XorShift32& rng) {
    for (int i = (int)array.size() - 1; i > 0; i--) {
        int j = rng.range(0, i);
        std::swap(array[i], array[j]);
    }
}

// 应用：随机卡组生成
struct Card { int id; std::string name; };

std::vector<Card> generateShuffledDeck(const std::vector<Card>& templateDeck,
    XorShift32& rng) {
    std::vector<Card> deck = templateDeck;
    fisherYatesShuffle(deck, rng);
    return deck;
}

// 不放回随机抽取N个
template<typename T>
std::vector<T> sampleWithoutReplacement(std::vector<T>& pool, int n,
    XorShift32& rng) {
    std::vector<T> result;
    for (int i = 0; i < n && !pool.empty(); i++) {
        int idx = rng.range(0, (int)pool.size() - 1);
        result.push_back(pool[idx]);
        pool[idx] = pool.back();
        pool.pop_back();
    }
    return result;
}
```

## Perlin噪声完整实现

```cpp
#include <algorithm>
#include <numeric>

// ========== 2D Perlin噪声完整实现 ==========
class PerlinNoise2D {
    int permutation[512]; // 排列数组（重复一次避免溢出）

public:
    PerlinNoise2D(uint32_t seed) {
        XorShift32 rng(seed);

        // 初始化排列数组
        int p[256];
        std::iota(p, p + 256, 0); // 填充0-255
        fisherYatesShuffleArray(p, 256, rng);

        // 重复一次
        for (int i = 0; i < 256; i++) {
            permutation[i] = p[i];
            permutation[i + 256] = p[i];
        }
    }

    // 2D Perlin噪声核心
    float noise(float x, float y) {
        // 确定单元格
        int xi = (int)floor(x) & 255;
        int yi = (int)floor(y) & 255;

        // 单元格内坐标 [0, 1)
        float xf = x - floor(x);
        float yf = y - floor(y);

        // 平滑插值曲线（5次Hermite）
        float u = fade(xf);
        float v = fade(yf);

        // 四个角点的哈希值
        int aa = permutation[permutation[xi] + yi];
        int ab = permutation[permutation[xi] + yi + 1];
        int ba = permutation[permutation[xi + 1] + yi];
        int bb = permutation[permutation[xi + 1] + yi + 1];

        // 梯度点积
        float dotAA = grad(aa, xf, yf);
        float dotBA = grad(ba, xf - 1, yf);
        float dotAB = grad(ab, xf, yf - 1);
        float dotBB = grad(bb, xf - 1, yf - 1);

        // 双线性插值
        float lerpX1 = lerp(dotAA, dotBA, u);
        float lerpX2 = lerp(dotAB, dotBB, u);
        return lerp(lerpX1, lerpX2, v);
    }

    // 分形布朗运动（FBM）- 多层叠加
    float fbm(float x, float y, int octaves = 6,
              float lacunarity = 2.0f, float persistence = 0.5f) {
        float total = 0;
        float amplitude = 1;
        float frequency = 1;
        float maxValue = 0;

        for (int i = 0; i < octaves; i++) {
            total += noise(x * frequency, y * frequency) * amplitude;
            maxValue += amplitude;
            amplitude *= persistence;
            frequency *= lacunarity;
        }

        return total / maxValue; // 归一化到[-1, 1]
    }

private:
    // 5次Hermite平滑曲线: 6t^5 - 15t^4 + 10t^3
    float fade(float t) {
        return t * t * t * (t * (t * 6 - 15) + 10);
    }

    // 线性插值
    float lerp(float a, float b, float t) {
        return a + t * (b - a);
    }

    // 梯度函数：8个方向的梯度向量与(x,y)的点积
    float grad(int hash, float x, float y) {
        switch (hash & 7) {
            case 0: return  x + y;
            case 1: return -x + y;
            case 2: return  x - y;
            case 3: return -x - y;
            case 4: return  x;
            case 5: return -x;
            case 6: return  y;
            case 7: return -y;
            default: return 0;
        }
    }

    void fisherYatesShuffleArray(int* arr, int n, XorShift32& rng) {
        for (int i = n - 1; i > 0; i--) {
            int j = rng.range(0, i);
            std::swap(arr[i], arr[j]);
        }
    }
};

// ========== Simplex噪声（比Perlin更快，各向同性更好） ==========
// 2D Simplex噪声核心（简化版）
float simplex2D(float x, float y) {
    const float F2 = 0.5f * (sqrt(3.0f) - 1.0f);
    const float G2 = (3.0f - sqrt(3.0f)) / 6.0f;

    float s = (x + y) * F2;
    int i = (int)floor(x + s);
    int j = (int)floor(y + s);

    float t = (i + j) * G2;
    float X0 = i - t, Y0 = j - t;
    float x0 = x - X0, y0 = y - Y0;

    int i1, j1;
    if (x0 > y0) { i1 = 1; j1 = 0; }
    else { i1 = 0; j1 = 1; }

    float x1 = x0 - i1 + G2;
    float y1 = y0 - j1 + G2;
    float x2 = x0 - 1.0f + 2.0f * G2;
    float y2 = y0 - 1.0f + 2.0f * G2;

    int ii = i & 255, jj = j & 255;

    float n0 = 0, n1 = 0, n2 = 0;

    float t0 = 0.5f - x0*x0 - y0*y0;
    if (t0 >= 0) { t0 *= t0; n0 = t0 * t0 * grad2D(hash(ii, jj), x0, y0); }

    float t1 = 0.5f - x1*x1 - y1*y1;
    if (t1 >= 0) { t1 *= t1; n1 = t1 * t1 * grad2D(hash(ii+i1, jj+j1), x1, y1); }

    float t2 = 0.5f - x2*x2 - y2*y2;
    if (t2 >= 0) { t2 *= t2; n2 = t2 * t2 * grad2D(hash(ii+1, jj+1), x2, y2); }

    return 70.0f * (n0 + n1 + n2);
}
```

```csharp
        float gaussian = UnityEngine.Random.Range(0f, 1f);
        // 或使用第三方库
    }

    // ========== 权重随机选择 ==========
    public static int WeightedRandom(float[] weights, System.Random rng)
    {
        float total = 0;
        for (int i = 0; i < weights.Length; i++) total += weights[i];

        float r = (float)rng.NextDouble() * total;
        float cumulative = 0;
        for (int i = 0; i < weights.Length; i++)
        {
            cumulative += weights[i];
            if (r <= cumulative) return i;
        }
        return weights.Length - 1;
    }

    // ========== 保底系统 ==========
    [System.Serializable]
    public class PitySystem
    {
        public int softPityStart = 75;
        public int hardPity = 90;
        public float baseRate = 0.006f;
        public float softPityBonus = 0.06f;
        private int counter = 0;

        public bool Roll()
        {
            counter++;
            float rate = baseRate;

            if (counter >= hardPity)
            {
                counter = 0;
                return true;
            }

            if (counter >= softPityStart)
                rate += (counter - softPityStart) * softPityBonus;

            if (Random.value < rate)
            {
                counter = 0;
                return true;
            }
            return false;
        }
    }
}
```

## 游戏应用场景

1. **掉率系统**：装备/道具掉落概率，通常用加权随机+保底
2. **暴击判定**：伪随机分布（PRD）确保暴击率趋近期望值
3. **世界生成**：地形、洞穴、植被分布使用Perlin/Simplex噪声
4. **AI决策**：概率行为树让NPC行为更自然
5. **洗牌算法**：Fisher-Yates洗牌用于卡组、随机事件列表
6. **Loot Box**：抽卡系统的多层保底和概率提升
7. **随机地图**：Roguelike游戏的房间布局和敌人分布
8. **程序化动画**：随机偏移让重复动画不显机械

## 常见陷阱

- **时间种子**：用`time(0)`做种子会导致同时启动的游戏产生相同序列，应使用系统熵源或组合种子
- **模运算偏差**：`rand() % n` 在n不整除RAND_MAX时有分布偏差，应使用`rng % n`（对2的幂次无偏差的生成器）或拒绝采样
- **可重现性**：需要存档回放的游戏应存储种子而非所有随机结果
- **玩家感知**：真随机在小样本中看起来"不公平"，需要伪随机分布调整
- **浮点精度**：`random01() == 1.0f`的概率极低，边界比较应使用`<=`而非`<`
- **多线程安全**：标准rand()不是线程安全的，多线程应使用thread_local RNG
- **种子碰撞**：多个RNG实例使用相近种子可能产生相关序列
