# 碰撞检测数学

## 核心概念

碰撞检测分为Broad Phase（粗检测）和Narrow Phase（精检测），数学基础是几何图元的相交测试。高效的碰撞检测是实时物理模拟和游戏交互的基石。

### 基本碰撞体类型

- **Sphere（球体）**：中心+半径，最简单高效，旋转不变
- **AABB（轴对齐包围盒）**：轴对齐的六面体，快速但旋转后效率低
- **OBB（有向包围盒）**：可旋转的六面体，更精确但计算成本高
- **Capsule（胶囊体）**：半球+圆柱，适合角色碰撞体
- **Convex Hull（凸包）**：任意凸多面体，GJK算法适用

### 碰撞检测流程

```
Broad Phase (粗检测, O(n²) → O(n log n))
  → 空间划分 / BVH / 空间哈希
  → 输出: 可能碰撞的物体对列表

Narrow Phase (精检测, O(1) per pair)
  → 几何相交测试
  → 输出: 碰撞法线、穿透深度、接触点
```

## Sphere-Sphere碰撞

两球碰撞条件：距离平方小于等于半径之和的平方。使用距离平方避免开方运算：

```cpp
struct Sphere { Vec3 center; float radius; };
struct Vec3 {
    float x, y, z;
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator*(float s) const { return {x*s, y*s, z*s}; }
    float dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    float lengthSq() const { return x*x + y*y + z*z; }
    float length() const { return sqrt(lengthSq()); }
    Vec3 normalized() const { float l=length(); return {x/l, y/l, z/l}; }
};

struct SphereHitResult {
    bool hit;
    float penetration;
    Vec3 normal; // 从A指向B的分离法线
};

SphereHitResult sphereSphere(const Sphere& a, const Sphere& b) {
    Vec3 diff = a.center - b.center;
    float distSq = diff.dot(diff);
    float radiiSum = a.radius + b.radius;

    SphereHitResult result;
    if (distSq <= radiiSum * radiiSum) {
        result.hit = true;
        float dist = sqrt(distSq);
        if (dist > 0.0001f) {
            result.normal = diff * (1.0f / dist); // A到B的方向
        } else {
            result.normal = {0, 1, 0}; // 完全重叠时的默认方向
        }
        result.penetration = radiiSum - dist;
    } else {
        result.hit = false;
        result.penetration = 0;
        result.normal = {0, 0, 0};
    }
    return result;
}
```

## AABB-AABB碰撞

两轴对齐盒碰撞条件：所有轴上的投影区间都有重叠：

```cpp
struct AABB {
    Vec3 min, max;
    Vec3 center() const { return (min + max) * 0.5f; }
    Vec3 extents() const { return (max - min) * 0.5f; }
};

struct AABBResult {
    bool hit;
    float penetration;
    Vec3 normal; // 最小穿透方向
};

AABBResult aabbAABB(const AABB& a, const AABB& b) {
    AABBResult res = {false, 0, {0,0,0}};
    float minOverlap = FLT_MAX;
    int minAxis = -1;

    float overlaps[3] = {
        std::min(a.max.x, b.max.x) - std::max(a.min.x, b.min.x),
        std::min(a.max.y, b.max.y) - std::max(a.min.y, b.min.y),
        std::min(a.max.z, b.max.z) - std::max(a.min.z, b.min.z)
    };

    for (int i = 0; i < 3; i++) {
        if (overlaps[i] <= 0) return res; // 某轴分离，无碰撞
        if (overlaps[i] < minOverlap) {
            minOverlap = overlaps[i];
            minAxis = i;
        }
    }

    // 确定穿透方向：两盒中心指向
    Vec3 d = b.center() - a.center();
    Vec3 axes[3] = {{1,0,0}, {0,1,0}, {0,0,1}};
    float sign = ((&d.x)[minAxis] < 0) ? -1.0f : 1.0f;

    res.hit = true;
    res.penetration = minOverlap;
    res.normal = axes[minAxis] * sign;
    return res;
}

// Swept AABB - 连续碰撞检测
struct SweptAABBResult {
    float t;         // 碰撞时间 [0, 1]
    Vec3 normal;     // 碰撞法线
};

SweptAABBResult sweptAABB(const AABB& a, const Vec3& velA,
                          const AABB& b, const Vec3& velB) {
    SweptAABBResult res = {1.0f, {0,0,0}};
    Vec3 relVel = velA - velB;

    // 对每个轴计算进入/离开时间
    float tEnter = 0, tExit = 1.0f;
    int hitAxis = -1;

    for (int i = 0; i < 3; i++) {
        float v = (&relVel.x)[i];
        if (abs(v) < 0.0001f) {
            // 相对速度为0，检查是否重叠
            if ((&a.max.x)[i] < (&b.min.x)[i] || (&a.min.x)[i] > (&b.max.x)[i])
                return {1.0f, {0,0,0}}; // 静止时无重叠
        } else {
            float aMin = (&a.min.x)[i], aMax = (&a.max.x)[i];
            float bMin = (&b.min.x)[i], bMax = (&b.max.x)[i];
            float t1 = (bMin - aMax) / v;
            float t2 = (bMax - aMin) / v;
            if (t1 > t2) std::swap(t1, t2);
            if (t1 > tEnter) { tEnter = t1; hitAxis = i; }
            if (t2 < tExit) tExit = t2;
            if (tEnter > tExit) return {1.0f, {0,0,0}};
        }
    }

    if (hitAxis >= 0 && tEnter >= 0 && tEnter <= 1.0f) {
        res.t = tEnter;
        float sign = ((&relVel.x)[hitAxis] < 0) ? 1.0f : -1.0f;
        (&res.normal.x)[hitAxis] = sign;
    }
    return res;
}
```

## SAT分离轴定理

如果能找到一条轴使得两个凸体在该轴上的投影不重叠，则两物体不相交。对于3D OBB，需检测15条潜在分离轴：两个OBB各3个面法线 + 9条边的叉积。

```cpp
struct OBB {
    Vec3 center;
    Vec3 extents;     // 半尺寸
    Vec3 axis[3];     // 三个主轴方向（单位向量，已归一化）
};

struct OBBResult {
    bool hit;
    float penetration;
    Vec3 normal;
};

// SAT OBB vs OBB 完整实现（15条轴检测）
OBBResult satOBBvsOBB(const OBB& a, const OBB& b) {
    OBBResult res = {false, FLT_MAX, {0,0,0}};

    // 计算旋转矩阵R和|R|+epsilon
    float R[3][3], AbsR[3][3];
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++) {
            R[i][j] = a.axis[i].dot(b.axis[j]);
            AbsR[i][j] = abs(R[i][j]) + 1e-6f; // epsilon防止浮点误差
        }

    Vec3 tVec = b.center - a.center;
    float t[3] = {tVec.dot(a.axis[0]), tVec.dot(a.axis[1]), tVec.dot(a.axis[2])};

    // 测试a的3条轴 (测试1-3)
    for (int i = 0; i < 3; i++) {
        float ra = a.extents[i];
        float rb = b.extents[0]*AbsR[i][0] + b.extents[1]*AbsR[i][1] + b.extents[2]*AbsR[i][2];
        if (abs(t[i]) > ra + rb) return {false, 0, {0,0,0}};

        float overlap = ra + rb - abs(t[i]);
        if (overlap < res.penetration) {
            res.penetration = overlap;
            res.normal = a.axis[i] * (t[i] < 0 ? -1.0f : 1.0f);
        }
    }

    // 测试b的3条轴 (测试4-6)
    for (int i = 0; i < 3; i++) {
        float ra = a.extents[0]*AbsR[0][i] + a.extents[1]*AbsR[1][i] + a.extents[2]*AbsR[2][i];
        float rb = b.extents[i];
        float ti = t[0]*R[0][i] + t[1]*R[1][i] + t[2]*R[2][i];
        if (abs(ti) > ra + rb) return {false, 0, {0,0,0}};

        float overlap = ra + rb - abs(ti);
        if (overlap < res.penetration) {
            res.penetration = overlap;
            res.normal = b.axis[i] * (ti < 0 ? 1.0f : -1.0f);
        }
    }

    // 测试9条边叉积轴 (测试7-15)
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            // 叉积轴: a.axis[i] × b.axis[j]
            float ra, rb;

            // ra = a在叉积轴上的投影半径
            int i1 = (i+1)%3, i2 = (i+2)%3;
            ra = a.extents[i1]*AbsR[i2][j] + a.extents[i2]*AbsR[i1][j];

            // rb = b在叉积轴上的投影半径
            int j1 = (j+1)%3, j2 = (j+2)%3;
            rb = a.extents[0]*abs(R[0][j2]-R[0][j1]) +
                 a.extents[1]*abs(R[1][j2]-R[1][j1]) +
                 a.extents[2]*abs(R[2][j2]-R[2][j1]);

            float ti = abs(t[i2]*R[i1][j] - t[i1]*R[i2][j]);
            if (ti > ra + rb) return {false, 0, {0,0,0}};
        }
    }

    res.hit = true;
    return res;
}
```

## GJK算法

迭代算法，通过构建Minkowski差集（A-B）的单纯形来判断原点是否在其内部。适用于任意凸体。GJK不需要知道具体形状，只需要一个"support函数"返回凸体在给定方向上最远的点。

```cpp
// Support函数：返回凸体在方向d上最远的点
Vec3 support(const ConvexShape& shape, Vec3 dir) {
    return shape.furthestPoint(dir);
}

// Minkowski差集的support
Vec3 minkowskiSupport(const ConvexShape& a, const ConvexShape& b, Vec3 dir) {
    return support(a, dir) - support(b, -dir);
}

// 2D GJK完整实现
struct Simplex2D {
    Vec2 points[3]; // 2D最多3个点构成三角形
    int count;
};

bool sameDir(Vec2 a, Vec2 b) { return a.dot(b) > 0; }

// 处理不同维度的单纯形
bool handleSimplex(Simplex2D& simp, Vec2& dir) {
    if (simp.count == 2) {
        // 线段：检查原点是否在AB方向的左侧
        Vec2 AB = simp.points[1] - simp.points[0];
        Vec2 AO = -simp.points[0];
        // 原点在AB的哪一侧
        dir = {AB.y, -AB.x}; // AB的垂直方向
        if (dir.dot(AO) < 0) dir = -dir;
        return false; // 还需要更多点
    } else if (simp.count == 3) {
        // 三角形：检查原点是否在内部
        Vec2 A = simp.points[2]; // 最新添加的点
        Vec2 B = simp.points[1];
        Vec2 C = simp.points[0];

        Vec2 AB = B - A, AC = C - A, AO = -A;

        Vec2 perpAB = {AB.y, -AB.x}; // AB的法线
        Vec2 perpAC = {AC.y, -AC.x}; // AC的法线

        if (perpAB.dot(AO) > 0) {
            // 原点在AB外侧
            simp.points[0] = A;
            simp.points[1] = B;
            simp.count = 2;
            dir = perpAB;
            if (dir.dot(AO) < 0) dir = -dir;
            return false;
        }
        if (perpAC.dot(AO) < 0) {
            // 原点在AC外侧
            simp.points[0] = A;
            simp.points[1] = C;
            simp.count = 2;
            dir = perpAC;
            if (dir.dot(AO) < 0) dir = -dir;
            return false;
        }
        return true; // 原点在三角形内
    }
    return false;
}

bool gjk2D(const ConvexShape& a, const ConvexShape& b) {
    Simplex2D simp = {};
    Vec2 dir = {1, 0}; // 初始搜索方向

    // 添加第一个点
    simp.points[0] = minkowskiSupport2D(a, b, dir);
    simp.count = 1;

    if (simp.points[0].dot(dir) < 0) return false; // 初始点在原点反方向

    dir = -simp.points[0]; // 指向原点

    for (int iter = 0; iter < 20; iter++) {
        Vec2 newPt = minkowskiSupport2D(a, b, dir);

        if (newPt.dot(dir) < 0) return false; // 新点无法越过原点

        simp.points[simp.count++] = newPt;

        if (handleSimplex(simp, dir)) return true; // 原点在单纯形内
    }
    return false; // 超过最大迭代次数（理论上不应发生）
}

// 3D GJK单纯形处理
bool handleSimplex3D(Vec3* points, int& count, Vec3& dir) {
    if (count == 2) {
        // 线段AB
        Vec3 AB = points[1] - points[0];
        Vec3 AO = -points[0];
        // 搜索方向：垂直于AB且指向原点
        Vec3 perp = AB.cross(AO).cross(AB);
        dir = perp.normalized();
        return false;
    } else if (count == 3) {
        // 三角形ABC
        Vec3 A = points[2], B = points[1], C = points[0];
        Vec3 AB = B - A, AC = C - A, AO = -A;
        Vec3 normal = AB.cross(AC);
        // 检查原点在三角形的哪一侧
        if (normal.dot(AO) > 0) {
            dir = normal.normalized();
        } else {
            dir = -normal.normalized();
        }
        return false;
    } else if (count == 4) {
        // 四面体ABCD (D是最新的点)
        Vec3 D = points[3], C = points[2], B = points[1], A = points[0];
        Vec3 DO = -D;
        Vec3 DC = C - D, DB = B - D, DA = A - D;

        Vec3 nDCB = DC.cross(DB);
        Vec3 nDBA = DB.cross(DA);
        Vec3 nDAC = DA.cross(DC);

        // 检查原点在四面体哪个面的外侧
        if (nDCB.dot(DO) > 0) {
            points[0] = D; points[1] = C; points[2] = B; count = 3;
            dir = nDCB.normalized();
            return false;
        }
        if (nDBA.dot(DO) > 0) {
            points[0] = D; points[1] = B; points[2] = A; count = 3;
            dir = nDBA.normalized();
            return false;
        }
        if (nDAC.dot(DO) > 0) {
            points[0] = D; points[1] = A; points[2] = C; count = 3;
            dir = nDAC.normalized();
            return false;
        }
        return true; // 原点在四面体内
    }
    return false;
}
```

## 射线碰撞检测

```cpp
// 射线-Sphere
float raySphere(Vec3 rayOrigin, Vec3 rayDir, Vec3 sphereCenter, float radius) {
    Vec3 oc = rayOrigin - sphereCenter;
    float a = rayDir.dot(rayDir);
    float b = 2.0f * oc.dot(rayDir);
    float c = oc.dot(oc) - radius * radius;
    float disc = b*b - 4*a*c;
    if (disc < 0) return -1.0f;
    float t = (-b - sqrt(disc)) / (2*a);
    return t >= 0 ? t : -1.0f;
}

// 射线-AABB（slab method）
float rayAABB(Vec3 rayOrigin, Vec3 rayDir, AABB box) {
    float tMin = -FLT_MAX, tMax = FLT_MAX;
    for (int i = 0; i < 3; i++) {
        float origin = (&rayOrigin.x)[i];
        float dir = (&rayDir.x)[i];
        float bMin = (&box.min.x)[i];
        float bMax = (&box.max.x)[i];

        if (abs(dir) < 1e-6f) {
            if (origin < bMin || origin > bMax) return -1.0f;
        } else {
            float t1 = (bMin - origin) / dir;
            float t2 = (bMax - origin) / dir;
            if (t1 > t2) std::swap(t1, t2);
            tMin = std::max(tMin, t1);
            tMax = std::min(tMax, t2);
            if (tMin > tMax) return -1.0f;
        }
    }
    return tMin >= 0 ? tMin : (tMax >= 0 ? tMax : -1.0f);
}

// 射线-三角形（Moller-Trumbore算法）
float rayTriangle(Vec3 rayOrigin, Vec3 rayDir, Vec3 v0, Vec3 v1, Vec3 v2) {
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = rayDir.cross(e2);
    float a = e1.dot(h);
    if (abs(a) < 1e-6f) return -1.0f; // 射线与三角形平行
    float f = 1.0f / a;
    Vec3 s = rayOrigin - v0;
    float u = f * s.dot(h);
    if (u < 0 || u > 1) return -1.0f;
    Vec3 q = s.cross(e1);
    float v = f * rayDir.dot(q);
    if (v < 0 || u + v > 1) return -1.0f;
    float t = f * e2.dot(q);
    return t >= 0 ? t : -1.0f;
}
```

## 连续碰撞检测（CCD）

解决隧道效应：快速移动的物体可能穿过薄墙。方法包括射线投射和TOI（Time of Impact）计算：

```cpp
// Swept Sphere - 连续球体碰撞检测
float sweptSphereSphere(Vec3 p0, Vec3 p1, float r0,
                        Vec3 p2, Vec3 p3, float r2) {
    Vec3 s = p1 - p0;       // A的运动轨迹
    Vec3 st = p3 - p2;      // B的运动轨迹
    Vec3 relVel = s - st;   // 相对速度
    Vec3 relPos = p0 - p2;  // 相对位置
    float radiusSum = r0 + r2;

    float a = relVel.dot(relVel);
    float b = 2 * relPos.dot(relVel);
    float c = relPos.dot(relPos) - radiusSum * radiusSum;
    float disc = b*b - 4*a*c;

    if (disc < 0) return -1.0f; // 无碰撞
    float t = (-b - sqrt(disc)) / (2*a);
    return (t >= 0 && t <= 1) ? t : -1.0f;
}

// TOI计算用于Speculative CCD
// 将物体在t=0到t=1之间可能的所有位置纳入考虑
struct TOIResult {
    bool willCollide;
    float timeOfImpact;
    Vec3 contactNormal;
};
```

## 游戏应用场景

1. **角色碰撞**：胶囊体 vs 地面/墙壁的碰撞检测
2. **子弹命中**：射线 vs OBB的精确命中判定（FPS游戏）
3. **触发区域**：球体/盒体的区域触发检测（拾取物、安全区）
4. **物理模拟**：大量刚体间的持续碰撞检测
5. **相机遮挡**：射线检测相机与场景之间的遮挡物
6. **鼠标拾取**：屏幕坐标射线与3D物体的相交测试

## 实际游戏案例

### 案例：FPS游戏的射击判定

```cpp
// 服务端射击判定流程
struct ShotResult {
    bool hit;
    AActor* hitActor;
    Vec3 hitPoint;
    Vec3 hitNormal;
    float distance;
};

ShotResult processShot(Vec3 shooterPos, Vec3 aimDir, float maxRange) {
    ShotResult result = {};

    // 第一步：Broad Phase - 射线 vs AABB粗筛
    std::vector<int> candidates = broadPhaseRayQuery(shooterPos, aimDir, maxRange);

    // 第二步：Narrow Phase - 逐个精确检测
    float closestHit = maxRange;
    for (int idx : candidates) {
        Actor* actor = actors[idx];

        // 对玩家使用胶囊体（更精确）
        if (actor->type == ActorType::Player) {
            float t = rayCapsule(shooterPos, aimDir, actor->capsule);
            if (t >= 0 && t < closestHit) {
                closestHit = t;
                result.hit = true;
                result.hitActor = actor;
                result.hitPoint = shooterPos + aimDir * t;
            }
        }
        // 对静态物体使用AABB
        else {
            float t = rayAABB(shooterPos, aimDir, actor->bounds);
            if (t >= 0 && t < closestHit) {
                closestHit = t;
                result.hit = true;
                result.hitActor = actor;
                result.hitPoint = shooterPos + aimDir * t;
            }
        }
    }

    result.distance = closestHit;
    return result;
}
```

## 常见陷阱

- **隧道效应**：快速移动的物体可能穿过薄墙（用CCD连续碰撞检测解决）
- **精度容忍度**：浮点误差导致物体在"刚好接触"时抖动，需引入epsilon容忍
- **SAT的退化情况**：平行边叉积为零向量，需跳过或特殊处理
- **GJK收敛性**：少数情况下GJK可能不收敛（退化构型），需设置最大迭代次数并处理异常
- **射线起点在物体内**：射线起点在碰撞体内部时，t值为负或零，需要特殊处理
- **法线方向不一致**：确保碰撞法线总是从A指向B（或按约定的方向）
