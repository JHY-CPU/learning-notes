# 路径与曲线数学

## 核心概念

曲线在游戏中的路径运动、UI动画、轨迹预测等方面有广泛应用。参数化曲线通过参数t定义点的轨迹，t通常在[0, 1]范围内。掌握各种曲线的数学性质和实现方法是游戏开发的基本功。

### 曲线类型对比

| 曲线类型 | 是否经过控制点 | 连续性 | 典型用途 |
|----------|-------------|--------|---------|
| 贝塞尔曲线 | 仅首尾点 | C^∞ | 技能弹道、UI曲线 |
| B样条 | 不经过 | C^(k-1) | 3D建模、路径设计 |
| Catmull-Rom | 经过所有点 | C^1 | 赛道设计、相机路径 |
| Hermite | 仅首尾点 | C^1 | 动画缓动 |
| NURBS | 可控 | C^(k-1) | 工业建模 |

## 贝塞尔曲线（Bezier Curve）

```
线性:   B(t) = (1-t)P0 + tP1
二次:   B(t) = (1-t)^2 P0 + 2(1-t)t P1 + t^2 P2
三次:   B(t) = (1-t)^3 P0 + 3(1-t)^2 t P1 + 3(1-t)t^2 P2 + t^3 P3

伯恩斯坦多项式通式:
B(t) = Sum(C(n,i) * (1-t)^(n-i) * t^i * Pi), i=0..n
其中 C(n,i) = n! / (i! * (n-i)!)
```

控制点决定曲线形状但不一定在曲线上（除首尾点）。德卡斯特里奥(De Casteljau)算法通过递归线性插值计算曲线上任意点。

## De Casteljau算法完整实现

```cpp
#include <vector>
#include <cmath>

struct Vec3 {
    float x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(float s) const { return {x*s, y*s, z*s}; }
    float length() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { float l = length(); return {x/l, y/l, z/l}; }
};

// ========== 三次贝塞尔曲线 ==========
Vec3 cubicBezier(Vec3 p0, Vec3 p1, Vec3 p2, Vec3 p3, float t) {
    float u = 1 - t;
    float tt = t * t, ttt = tt * t;
    float uu = u * u, uuu = uu * u;

    return p0 * uuu
         + p1 * (3 * uu * t)
         + p2 * (3 * u * tt)
         + p3 * ttt;
}

// ========== 任意阶贝塞尔曲线（De Casteljau算法） ==========
Vec3 bezierDeCasteljau(const std::vector<Vec3>& controlPoints, float t) {
    int n = (int)controlPoints.size();
    if (n == 0) return {0, 0, 0};
    if (n == 1) return controlPoints[0];

    // 创建工作数组
    std::vector<Vec3> work = controlPoints;

    // 递归线性插值，每轮减少一个点
    for (int level = 1; level < n; level++) {
        for (int i = 0; i < n - level; i++) {
            work[i] = work[i] * (1 - t) + work[i+1] * t;
        }
    }

    return work[0];
}

// ========== 贝塞尔曲线导数（切线方向） ==========
Vec3 cubicBezierTangent(Vec3 p0, Vec3 p1, Vec3 p2, Vec3 p3, float t) {
    float u = 1 - t;
    return (p1 - p0) * (3 * u * u)
         + (p2 - p1) * (6 * u * t)
         + (p3 - p2) * (3 * t * t);
}

// ========== 贝塞尔曲线二阶导数（曲率） ==========
Vec3 cubicBezierSecondDerivative(Vec3 p0, Vec3 p1, Vec3 p2, Vec3 p3, float t) {
    return (p2 - p1 * 2 + p0) * (6 * (1 - t))
         + (p3 - p2 * 2 + p1) * (6 * t);
}

// ========== 贝塞尔曲线细分（用于渲染） ==========
struct BezierSegment {
    Vec3 points[4]; // p0, p1, p2, p3
};

// 自适应细分：在曲率大的地方增加细分
void adaptiveSubdivide(const BezierSegment& seg,
    std::vector<Vec3>& output, float tolerance = 0.1f, int maxDepth = 10) {

    // 检查曲线平坦度（控制点到首尾连线的最大距离）
    Vec3 a = seg.points[1] - seg.points[0];
    Vec3 b = seg.points[2] - seg.points[0];
    Vec3 c = seg.points[3] - seg.points[0];

    // 使用面积法检查平坦度
    float flatness = (a.cross(b).length() + b.cross(c).length()) / (c.length() + 1e-6f);

    if (flatness < tolerance || maxDepth <= 0) {
        output.push_back(seg.points[0]);
        output.push_back(seg.points[3]);
        return;
    }

    // De Casteljau在t=0.5处切分
    Vec3 p01 = (seg.points[0] + seg.points[1]) * 0.5f;
    Vec3 p12 = (seg.points[1] + seg.points[2]) * 0.5f;
    Vec3 p23 = (seg.points[2] + seg.points[3]) * 0.5f;
    Vec3 p012 = (p01 + p12) * 0.5f;
    Vec3 p123 = (p12 + p23) * 0.5f;
    Vec3 p0123 = (p012 + p123) * 0.5f;

    BezierSegment left = {{seg.points[0], p01, p012, p0123}};
    BezierSegment right = {{p0123, p123, p23, seg.points[3]}};

    adaptiveSubdivide(left, output, tolerance, maxDepth - 1);
    adaptiveSubdivide(right, output, tolerance, maxDepth - 1);
}
```

## Catmull-Rom样条

经过所有控制点的插值曲线，需要至少4个点。相邻段在共享点处C1连续：

```cpp
// ========== Catmull-Rom样条 ==========
Vec3 catmullRom(Vec3 p0, Vec3 p1, Vec3 p2, Vec3 p3, float t, float tension = 0.5f) {
    float t2 = t * t, t3 = t2 * t;

    // 张力参数控制曲线的"紧度"
    // tension=0.5 为标准Catmull-Rom
    // tension=0 为Cardinal样条的极端情况
    // tension增大 → 曲线更直

    return (p1 * 2
          + (p2 - p0) * tension * t
          + (p0 * 2 - p1 * 5 + p2 * 4 - p3) * tension * t2
          + (-p0 + p1 * 3 - p2 * 3 + p3) * tension * t3) * 0.5f;
}

// Catmull-Rom切线
Vec3 catmullRomTangent(Vec3 p0, Vec3 p1, Vec3 p2, Vec3 p3, float t, float tension = 0.5f) {
    float t2 = t * t;
    return ((p2 - p0) * tension
          + (p0 * 2 - p1 * 5 + p2 * 4 - p3) * tension * 2 * t
          + (-p0 + p1 * 3 - p2 * 3 + p3) * tension * 3 * t2) * 0.5f;
}

// ========== Catmull-Rom路径管理器 ==========
class CatmullRomPath {
    std::vector<Vec3> points;
    float tension;

public:
    CatmullRomPath(const std::vector<Vec3>& pts, float tension = 0.5f)
        : points(pts), tension(tension) {}

    // 获取路径上的点（支持循环）
    Vec3 evaluate(float t, bool loop = false) const {
        int n = (int)points.size();
        if (n < 4) return {0, 0, 0};

        // 将t映射到段索引
        float totalSegments = loop ? n : n - 3;
        float segmentFloat = t * totalSegments;
        int segment = (int)floor(segmentFloat);
        float localT = segmentFloat - segment;

        if (!loop) {
            segment = std::max(0, std::min(segment, n - 4));
        } else {
            segment = ((segment % n) + n) % n;
        }

        int i0 = loop ? ((segment - 1 + n) % n) : std::max(0, segment - 1);
        int i1 = segment % n;
        int i2 = (segment + 1) % n;
        int i3 = (segment + 2) % n;

        return catmullRom(points[i0], points[i1], points[i2], points[i3], localT, tension);
    }

    // 获取路径上的切线方向
    Vec3 getTangent(float t, bool loop = false) const {
        int n = (int)points.size();
        if (n < 4) return {0, 0, 0};

        float totalSegments = loop ? n : n - 3;
        float segmentFloat = t * totalSegments;
        int segment = (int)floor(segmentFloat);
        float localT = segmentFloat - segment;

        if (!loop) {
            segment = std::max(0, std::min(segment, n - 4));
        } else {
            segment = ((segment % n) + n) % n;
        }

        int i0 = loop ? ((segment - 1 + n) % n) : std::max(0, segment - 1);
        int i1 = segment % n;
        int i2 = (segment + 1) % n;
        int i3 = (segment + 2) % n;

        return catmullRomTangent(points[i0], points[i1], points[i2], points[i3], localT, tension);
    }

    int getPointCount() const { return (int)points.size(); }
};
```

## 弧长参数化（匀速运动）

参数t的步长不等于弧长步长，直接用等步长t会导致速度不均匀。弧长参数化通过预计算弧长表，根据距离查找对应参数t：

```cpp
// ========== 弧长参数化 ==========
class ArcLengthParametrizer {
    std::vector<float> arcLengthTable; // 累积弧长
    std::vector<float> paramTable;     // 对应参数t
    float totalLength;

public:
    // 预计算弧长表
    void build(Vec3 p0, Vec3 p1, Vec3 p2, Vec3 p3, int resolution = 100) {
        arcLengthTable.resize(resolution + 1);
        paramTable.resize(resolution + 1);

        totalLength = 0;
        Vec3 prev = cubicBezier(p0, p1, p2, p3, 0);

        for (int i = 0; i <= resolution; i++) {
            float t = (float)i / resolution;
            Vec3 pt = cubicBezier(p0, p1, p2, p3, t);
            totalLength += (pt - prev).length();
            arcLengthTable[i] = totalLength;
            paramTable[i] = t;
            prev = pt;
        }
    }

    // 根据沿曲线的距离比（0~1）查找参数t（二分查找）
    float getParamAtDistance(float distRatio) const {
        if (distRatio <= 0) return 0;
        if (distRatio >= 1) return 1;

        float targetLen = totalLength * distRatio;

        // 二分查找
        int lo = 0, hi = (int)arcLengthTable.size() - 1;
        while (lo < hi - 1) {
            int mid = (lo + hi) / 2;
            if (arcLengthTable[mid] < targetLen)
                lo = mid;
            else
                hi = mid;
        }

        // 线性插值
        float segLen = arcLengthTable[hi] - arcLengthTable[lo];
        float frac = segLen > 0 ? (targetLen - arcLengthTable[lo]) / segLen : 0;
        return paramTable[lo] + frac * (paramTable[hi] - paramTable[lo]);
    }

    float getTotalLength() const { return totalLength; }

    // 根据弧长（而非t的比值）查找参数
    float getParamAtArcLength(float arcLength) const {
        return getParamAtDistance(arcLength / totalLength);
    }
};

// ========== 匀速曲线运动 ==========
class SmoothCurveFollower {
    ArcLengthParametrizer parametrizer;
    Vec3 p0, p1, p2, p3;
    float currentDistance = 0;
    float speed;

public:
    void init(Vec3 cp0, Vec3 cp1, Vec3 cp2, Vec3 cp3, float moveSpeed) {
        p0 = cp0; p1 = cp1; p2 = cp2; p3 = cp3;
        speed = moveSpeed;
        parametrizer.build(p0, p1, p2, p3, 200);
    }

    Vec3 update(float dt) {
        currentDistance += speed * dt;
        float totalLen = parametrizer.getTotalLength();

        // 循环
        if (currentDistance > totalLen)
            currentDistance -= totalLen;

        float t = parametrizer.getParamAtArcLength(currentDistance);
        return cubicBezier(p0, p1, p2, p3, t);
    }

    Vec3 getTangent() {
        float t = parametrizer.getParamAtArcLength(currentDistance);
        return cubicBezierTangent(p0, p1, p2, p3, t).normalized();
    }
};
```

## Hermite曲线与缓动函数

```cpp
// ========== Hermite曲线 ==========
// 给定两个端点及其切线
Vec3 hermite(Vec3 p0, Vec3 p1, Vec3 t0, Vec3 t1, float t) {
    float t2 = t * t, t3 = t2 * t;
    float h00 = 2*t3 - 3*t2 + 1;
    float h10 = t3 - 2*t2 + t;
    float h01 = -2*t3 + 3*t2;
    float h11 = t3 - t2;
    return p0 * h00 + t0 * h10 + p1 * h01 + t1 * h11;
}

// ========== 常用缓动函数 ==========
namespace Easing {
    // 二次缓动
    float easeInQuad(float t) { return t * t; }
    float easeOutQuad(float t) { return t * (2 - t); }
    float easeInOutQuad(float t) {
        return t < 0.5f ? 2*t*t : -1 + (4-2*t)*t;
    }

    // 三次缓动
    float easeInCubic(float t) { return t*t*t; }
    float easeOutCubic(float t) { float u = t-1; return u*u*u + 1; }
    float easeInOutCubic(float t) {
        return t < 0.5f ? 4*t*t*t : (t-1)*(2*t-2)*(2*t-2)+1;
    }

    // 弹跳缓动
    float easeOutBounce(float t) {
        if (t < 1/2.75f) return 7.5625f*t*t;
        else if (t < 2/2.75f) { t -= 1.5f/2.75f; return 7.5625f*t*t + 0.75f; }
        else if (t < 2.5/2.75f) { t -= 2.25f/2.75f; return 7.5625f*t*t + 0.9375f; }
        else { t -= 2.625f/2.75f; return 7.5625f*t*t + 0.984375f; }
    }

    // 弹性缓动
    float easeOutElastic(float t) {
        if (t == 0 || t == 1) return t;
        float p = 0.3f;
        return pow(2, -10*t) * sin((t - p/4) * (2*3.14159f) / p) + 1;
    }

    // 平滑阶梯
    float smoothstep(float edge0, float edge1, float x) {
        x = std::fmax(0, std::fmin(1, (x - edge0) / (edge1 - edge0)));
        return x * x * (3 - 2 * x);
    }

    // 更平滑的Smootherstep
    float smootherstep(float edge0, float edge1, float x) {
        x = std::fmax(0, std::fmin(1, (x - edge0) / (edge1 - edge0)));
        return x * x * x * (x * (x * 6 - 15) + 10);
    }
}
```

## 路径跟随系统

```cpp
// ========== 沿Catmull-Rom路径的物体运动 ==========
class PathFollower {
    std::vector<Vec3> controlPoints;
    int currentSegment;
    float t, speed;
    bool loop;

public:
    PathFollower(const std::vector<Vec3>& points, float moveSpeed, bool loopPath = true)
        : controlPoints(points), speed(moveSpeed), loop(loopPath),
          currentSegment(0), t(0) {}

    Vec3 update(float dt) {
        t += dt * speed;

        // 段切换
        while (t > 1.0f) {
            t -= 1.0f;
            currentSegment++;
            int maxSegment = loop ? (int)controlPoints.size()
                                  : (int)controlPoints.size() - 3;
            if (currentSegment >= maxSegment) {
                if (loop) currentSegment = 0;
                else { currentSegment = maxSegment - 1; t = 1.0f; break; }
            }
        }

        // 获取当前段的4个控制点
        int n = (int)controlPoints.size();
        int i0 = loop ? ((currentSegment - 1 + n) % n) : std::max(0, currentSegment - 1);
        int i1 = currentSegment % n;
        int i2 = (currentSegment + 1) % n;
        int i3 = (currentSegment + 2) % n;

        return catmullRom(controlPoints[i0], controlPoints[i1],
                          controlPoints[i2], controlPoints[i3], t);
    }

    Vec3 getCurrentTangent() {
        int n = (int)controlPoints.size();
        int i0 = loop ? ((currentSegment - 1 + n) % n) : std::max(0, currentSegment - 1);
        int i1 = currentSegment % n;
        int i2 = (currentSegment + 1) % n;
        int i3 = (currentSegment + 2) % n;

        return catmullRomTangent(controlPoints[i0], controlPoints[i1],
                                 controlPoints[i2], controlPoints[i3], t).normalized();
    }
};
```

## Unity C#实现

```csharp
using UnityEngine;

public class BezierCurveUtility : MonoBehaviour
{
    // 三次贝塞尔曲线
    public static Vector3 CubicBezier(Vector3 p0, Vector3 p1,
        Vector3 p2, Vector3 p3, float t)
    {
        float u = 1 - t;
        return u*u*u*p0 + 3*u*u*t*p1 + 3*u*t*t*p2 + t*t*t*p3;
    }

    // 贝塞尔曲线切线
    public static Vector3 CubicBezierTangent(Vector3 p0, Vector3 p1,
        Vector3 p2, Vector3 p3, float t)
    {
        float u = 1 - t;
        return 3*u*u*(p1-p0) + 6*u*t*(p2-p1) + 3*t*t*(p3-p2);
    }

    // Catmull-Rom样条
    public static Vector3 CatmullRom(Vector3 p0, Vector3 p1,
        Vector3 p2, Vector3 p3, float t)
    {
        float t2 = t*t, t3 = t2*t;
        return 0.5f * (2*p1 + (-p0+p2)*t
            + (2*p0-5*p1+4*p2-p3)*t2
            + (-p0+3*p1-3*p2+p3)*t3);
    }

    // 缓动函数
    public static float EaseInOutCubic(float t)
    {
        return t < 0.5f ? 4*t*t*t : 1-Mathf.Pow(-2*t+2,3)/2;
    }

    // 在Scene视图中绘制曲线
    void OnDrawGizmos()
    {
        if (controlPoints.Length < 4) return;

        Gizmos.color = Color.yellow;
        Vector3 prev = CubicBezier(controlPoints[0], controlPoints[1],
            controlPoints[2], controlPoints[3], 0);

        for (int i = 1; i <= 50; i++)
        {
            float t = i / 50f;
            Vector3 curr = CubicBezier(controlPoints[0], controlPoints[1],
                controlPoints[2], controlPoints[3], t);
            Gizmos.DrawLine(prev, curr);
            prev = curr;
        }

        // 绘制控制点
        Gizmos.color = Color.red;
        foreach (var p in controlPoints)
            Gizmos.DrawSphere(p, 0.1f);
    }

    public Vector3[] controlPoints;
}
```

## 游戏应用场景

1. **摄像机路径**：过场动画中相机沿样条曲线平滑移动
2. **技能弹道**：法术飞弹沿贝塞尔曲线飞行（如MOBA游戏弹道）
3. **赛道设计**：赛车游戏的弯道使用Catmull-Rom插值
4. **UI动画**：缓动曲线本质上是参数化曲线
5. **关卡设计**：3D建模中的路径和管道使用B样条
6. **AI巡逻路径**：NPC沿曲线路径移动
7. **抛物线弹道**：投掷物的抛物线轨迹

## 常见陷阱

- **速度不均匀**：参数t的步长不等于弧长步长，需弧长参数化实现匀速运动
- **控制点过近**：贝塞尔曲线控制点过于接近可能导致曲线反折或奇异
- **Catmull-Rom的尖点**：三个连续控制点共线且中间点在端点之间时产生尖角，可用张力参数缓解
- **细分数量**：渲染曲线时细分不足导致锯齿，过多则性能开销大。自适应细分是最佳方案
- **闭合曲线处理**：首尾相连时需要额外的控制点（通常复制首尾点）
- **浮点精度**：t接近0或1时的计算可能不精确
