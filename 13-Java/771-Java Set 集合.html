🎯 Java Set 集合

Set 接口、HashSet (哈希表/去重)、LinkedHashSet (插入顺序)、TreeSet (红黑树/排序)、equals/hashCode 契约、Comparator 与 Comparable。

Set 接口与 HashSet

// ========== Set 接口特性 ========== // 1. 不重复 (equals 判断) // 2. 无索引 (不能 get(index)) // 3. 最多一个 null (HashSet/LinkedHashSet) import java.util.*; public class SetBasics { public static void main(String[] args) { // ========== HashSet ========== // 底层: HashMap (哈希表) // 特点: 最快, 无序 // 不重复依据: hashCode() + equals() Set set = new HashSet<>(); // 添加 (重复元素返回 false) System.out.println(set.add("Apple")); // true System.out.println(set.add("Banana")); // true System.out.println(set.add("Apple")); // false (已存在!) System.out.println(set); // [Banana, Apple] (无序) // 其他操作 set.contains("Apple"); // true set.remove("Apple"); // true set.size(); // 1 set.isEmpty(); // false // set.clear(); // 清空 // ========== 线程安全 ========== Set syncSet = Collections.synchronizedSet(new HashSet<>()); Set copyOnWrite = new CopyOnWriteArraySet<>(); // 并发 // ========== 遍历 ========== for (String s : set) { System.out.println(s); } set.forEach(System.out::println); } } // ========== equals/hashCode 契约 ========== // 1. 两个对象 equals 相等 → hashCode 必须相等 // 2. 两个对象 equals 不等 → hashCode 最好不等 (哈希冲突) // 3. hashCode 相等 → equals 不一定相等 // ========== 正确实现 ========== class Student { private String id; private String name; public Student(String id, String name) { this.id = id; this.name = name; } @Override public boolean equals(Object obj) { if (this == obj) return true; if (obj == null || getClass() != obj.getClass()) return false; Student other = (Student) obj; return Objects.equals(id, other.id); // 业务相等 } @Override public int hashCode() { return Objects.hash(id); // 用 equals 用到的字段 } @Override public String toString() { return "Student{" + id + ": " + name + "}"; } } class HashSetWithObjects { public static void main(String[] args) { Set students = new HashSet<>(); students.add(new Student("S001", "Alice")); students.add(new Student("S002", "Bob")); students.add(new Student("S001", "Charlie")); // 同 id → 不会添加! System.out.println(students.size()); // 2 // ========== HashSet 查找原理 ========== // 1. 计算 hashCode() → 定位桶 (bucket) // 2. 桶中元素用 equals() 逐一比较 // 3. hashCode 均匀分布 → O(1) // 4. hashCode 都相同 → O(n) 退化为链表 // 重要: 存入 Set 后不要修改参与 hashCode/equals 的字段! } }

LinkedHashSet

// ========== LinkedHashSet ========== // 底层: LinkedHashMap (哈希表 + 双向链表) // 特点: 维护插入顺序 (或访问顺序) // 性能: 略低于 HashSet, 但可预测顺序 public class LinkedHashSetDemo { public static void main(String[] args) { Set set = new LinkedHashSet<>(); set.add("Apple"); set.add("Banana"); set.add("Cherry"); set.add("Apple"); // 重复, 不添加, 不改变顺序 System.out.println(set); // [Apple, Banana, Cherry] (插入顺序) // ========== 应用场景 ========== // 1. 需要去重 + 保留插入顺序 // 2. LRU 缓存 (访问顺序) // 去重并保持首次出现顺序: List list = List.of("a", "b", "a", "c", "b", "d"); Set ordered = new LinkedHashSet<>(list); System.out.println(ordered); // [a, b, c, d] // ========== HashSet vs LinkedHashSet vs TreeSet ========== // ┌──────────────┬───────────────┬──────────────┐ // │ HashSet │ LinkedHashSet │ TreeSet │ // ├──────────────┼───────────────┼──────────────┤ // │ 无序 │ 插入顺序 │ 排序 (自然/定制) │ // │ O(1) │ O(1) │ O(log n) │ // │ 哈希表 │ 哈希表+链表 │ 红黑树 │ // │ null ✅ │ null ✅ │ null ❌ │ // └──────────────┴───────────────┴──────────────┘ } }

TreeSet

// ========== TreeSet ========== // 底层: TreeMap (红黑树, 自平衡二叉搜索树) // 特点: 排序 (自动升序) // 性能: O(log n) 基本操作 // 要求: 元素实现 Comparable 或传入 Comparator import java.util.*; public class TreeSetDemo { public static void main(String[] args) { // ========== 自然排序 (Comparable) ========== TreeSet nums = new TreeSet<>(); nums.addAll(List.of(5, 3, 1, 4, 2)); System.out.println(nums); // [1, 2, 3, 4, 5] (自动排序) // ========== 自定义排序 (Comparator) ========== TreeSet byLength = new TreeSet<>( Comparator.comparingInt(String::length) .thenComparing(Comparator.naturalOrder()) ); byLength.addAll(List.of("banana", "apple", "cherry", "date", "elderberry")); System.out.println(byLength); // [date, apple, banana, cherry, elderberry] (按长度) // ========== TreeSet 特有方法 ========== TreeSet ts = new TreeSet<>(List.of(10, 20, 30, 40, 50)); System.out.println(ts.first()); // 10 (最小) System.out.println(ts.last()); // 50 (最大) System.out.println(ts.lower(30)); // 20 (<30 的最大值) System.out.println(ts.floor(30)); // 30 (≤30 的最大值) System.out.println(ts.higher(30)); // 40 (>30 的最小值) System.out.println(ts.ceiling(31)); // 40 (≥31 的最小值) // 子集 System.out.println(ts.subSet(20, 40)); // [20, 30] ([20, 40)) System.out.println(ts.headSet(30)); // [10, 20] (<30) System.out.println(ts.tailSet(30)); // [30, 40, 50] (≥30) // 逆序 System.out.println(ts.descendingSet()); // [50, 40, 30, 20, 10] // ========== pollFirst/pollLast ========== Integer first = ts.pollFirst(); // 取出并移除最小 Integer last = ts.pollLast(); // 取出并移除最大 System.out.println(ts); // [20, 30, 40] // ========== TreeSet 注意 ========== // 1. 不能存 null (NullPointerException) // 2. 元素必须可比较 (Comparable 或 Comparator) // 3. remove/contains 基于 compareTo, 不是 equals // 4. 排序字段不可变 (修改后删除/查找可能失败) } } // ========== Comparator vs Comparable ========== // Comparable: 自然排序 (类自己实现) // int compareTo(T o) → this vs o // Comparator: 定制排序 (独立比较器) // int compare(T a, T b) → a vs b class Student2 implements Comparable { String name; int score; Student2(String name, int score) { this.name = name; this.score = score; } @Override public int compareTo(Student2 o) { return Integer.compare(this.score, o.score); // 按分数升序 } @Override public String toString() { return name + "=" + score; } } // 使用: // TreeSet students = new TreeSet<>(); // 自然排序 // students.add(new Student2("Alice", 85)); // students.add(new Student2("Bob", 92)); // System.out.println(students); // [Alice=85, Bob=92] (按分数)
💡 Set 要点: HashSet 哈希表 O(1), 无序; LinkedHashSet 维护插入顺序; TreeSet 红黑树 O(log n), 自动排序; equals/hashCode 务必同时重写; Set 存对象后不要修改影响 hashCode/compareTo 的字段; TreeSet 元素必须可比较, 不能 null; HashSet 允许一个 null; Comparator 灵活定制排序。
