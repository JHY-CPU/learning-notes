🔍 Java 反射

Class 对象、获取构造器/方法/字段、动态调用、访问私有成员、反射性能、setAccessible、模块系统限制 (Java 9+)。

Class 对象

// ========== 反射 (Reflection) ========== // 运行时检查/调用类、方法、字段 // 框架核心: Spring, JPA, Hibernate, JUnit // 缺点: 性能开销, 安全限制, 代码复杂 import java.lang.reflect.*; public class ReflectionBasics { // ========== 获取 Class 对象的 3 种方式 ========== public static void main(String[] args) throws Exception { // 方式1: Class.forName() Class clazz1 = Class.forName("java.util.ArrayList"); // 方式2: .class 字面量 Class clazz2 = String.class; // 方式3: getClass() 实例方法 String str = "hello"; Class clazz3 = str.getClass(); // 三种方式得到相同的 Class 对象 System.out.println(clazz1.getName()); // "java.util.ArrayList" System.out.println(clazz2.getSimpleName()); // "String" System.out.println(clazz3 == String.class); // true // ========== 获取类信息 ========== Class cls = String.class; System.out.println("Name: " + cls.getName()); System.out.println("Simple: " + cls.getSimpleName()); System.out.println("Package: " + cls.getPackage().getName()); System.out.println("Super: " + cls.getSuperclass().getName()); System.out.println("Interface: " + cls.getInterfaces().length); System.out.println("Modifier: " + Modifier.toString(cls.getModifiers())); // isXxx 检查 System.out.println(cls.isArray()); // false System.out.println(cls.isPrimitive()); // false System.out.println(cls.isInterface()); // false } } // ========== 通过反射创建对象 ========== class ReflectCreate { public static void main(String[] args) throws Exception { // 方式1: clazz.newInstance() (已废弃, Java 9+) // String s1 = String.class.newInstance(); // 方式2: Constructor.newInstance() Class clazz = StringBuilder.class; // 获取无参构造器 Constructor noArg = clazz.getConstructor(); StringBuilder sb1 = (StringBuilder) noArg.newInstance(); // 获取有参构造器 Constructor withCapacity = clazz.getConstructor(int.class); StringBuilder sb2 = (StringBuilder) withCapacity.newInstance(100); // 获取 String 参数构造器 Constructor withStr = clazz.getConstructor(String.class); StringBuilder sb3 = (StringBuilder) withStr.newInstance("Hello"); System.out.println(sb3); // "Hello" } }

方法反射

// ========== 反射调用方法 ========== public class MethodReflection { static class Calculator { private int base; public Calculator(int base) { this.base = base; } public int add(int a, int b) { return a + b; } public int multiply(int a, int b) { return a * b; } private int addBase(int value) { return value + base; } public static String staticMethod() { return "static result"; } } public static void main(String[] args) throws Exception { Class clazz = Calculator.class; Calculator calc = new Calculator(100); Class calcClass = calc.getClass(); // ========== 获取方法 ========== // getMethod: 公开方法 (包括继承的) Method addMethod = clazz.getMethod("add", int.class, int.class); // getDeclaredMethod: 本类声明的方法 (包括私有) Method privateMethod = clazz.getDeclaredMethod("addBase", int.class); // ========== 调用方法 ========== // invoke(对象, 参数...) Object result = addMethod.invoke(calc, 3, 7); System.out.println(result); // 10 // ========== 调用私有方法 ========== privateMethod.setAccessible(true); // 允许访问私有 Object privateResult = privateMethod.invoke(calc, 50); System.out.println(privateResult); // 150 (50 + 100) // ========== 调用静态方法 ========== Method staticMethod = clazz.getMethod("staticMethod"); Object staticResult = staticMethod.invoke(null); // 静态方法传 null System.out.println(staticResult); // "static result" // ========== 获取所有方法 ========== System.out.println("=== All public methods ==="); for (Method m : clazz.getMethods()) { System.out.println(m.getName() + " → " + Modifier.toString(m.getModifiers())); } System.out.println("=== All declared methods ==="); for (Method m : clazz.getDeclaredMethods()) { System.out.println(m.getName() + " → " + Modifier.toString(m.getModifiers())); } } }

字段反射与动态代理

// ========== 反射操作字段 ========== import java.lang.reflect.*; public class FieldReflection { static class Person { public String name; private int age; private static String species = "Homo sapiens"; } public static void main(String[] args) throws Exception { Class clazz = Person.class; Person p = new Person(); // ========== 访问字段 ========== // getField: 公开字段 (包括继承) Field nameField = clazz.getField("name"); nameField.set(p, "Alice"); System.out.println(nameField.get(p)); // "Alice" // getDeclaredField: 本类所有字段 (包括私有) Field ageField = clazz.getDeclaredField("age"); ageField.setAccessible(true); // 允许访问私有 ageField.set(p, 25); System.out.println(ageField.get(p)); // 25 // 静态字段 Field speciesField = clazz.getDeclaredField("species"); speciesField.setAccessible(true); String species = (String) speciesField.get(null); // 静态字段传 null System.out.println(species); // "Homo sapiens" // ========== 字段类型 ========== System.out.println(nameField.getType()); // class java.lang.String System.out.println(ageField.getType()); // int } } // ========== 动态代理 ========== // 运行时动态创建代理类 // AOP (面向切面) 核心机制 interface Service { void doSomething(); String getInfo(String name); } class RealService implements Service { public void doSomething() { System.out.println("Real service doing work"); } public String getInfo(String name) { return "Info for: " + name; } } class LoggingHandler implements InvocationHandler { private final Object target; LoggingHandler(Object target) { this.target = target; } @Override public Object invoke(Object proxy, Method method, Object[] args) throws Throwable { System.out.println("[LOG] Calling: " + method.getName()); long start = System.nanoTime(); Object result = method.invoke(target, args); long elapsed = (System.nanoTime() - start) / 1_000_000; System.out.println("[LOG] " + method.getName() + " took " + elapsed + "ms"); return result; } } // 使用动态代理: // Service real = new RealService(); // Service proxy = (Service) Proxy.newProxyInstance( // Service.class.getClassLoader(), // new Class[]{Service.class}, // new LoggingHandler(real) // ); // proxy.doSomething(); // 自动添加日志!

反射性能与限制

// ========== 反射性能 ========== // 反射比直接调用慢 10-100 倍 // 原因: 类型检查, 装箱, 数组分配 public class ReflectPerformance { static void directCall() { long start = System.nanoTime(); String s = ""; for (int i = 0; i < 1_000_000; i++) { s = "hello"; } System.out.println("Direct: " + (System.nanoTime() - start) / 1_000_000 + "ms"); } static void reflectCall() throws Exception { Method setMethod = String.class.getMethod("valueOf", Object.class); long start = System.nanoTime(); for (int i = 0; i < 1_000_000; i++) { setMethod.invoke(null, "hello"); } System.out.println("Reflect: " + (System.nanoTime() - start) / 1_000_000 + "ms"); } // ========== 性能优化 ========== // 1. 缓存 Method/Field/Constructor 对象 // 2. setAccessible(true) 提高速度 // 3. 使用 MethodHandles (Java 7+) // 4. 使用 LambdaMetafactory // MethodHandles 示例: // MethodHandles.Lookup lookup = MethodHandles.lookup(); // MethodType mt = MethodType.methodType(void.class); // MethodHandle handle = lookup.findConstructor(String.class, mt); // String s = (String) handle.invoke(); // ========== 模块系统限制 (Java 9+) ========== // 模块可以控制反射访问 // --add-opens java.base/java.lang=ALL-UNNAMED // 框架需配置模块权限 // ========== 反射的最佳实践 ========== // 1. 能用接口/泛型解决, 不用反射 // 2. 缓存反射对象复用 // 3. 设置 accessible 加速 // 4. 避免在热路径 (循环/高频调用) 使用 // 5. 用 Spring 框架的封装 (ReflectionUtils) } // ========== 数组反射 ========== class ArrayReflect { public static void main(String[] args) { // 创建数组 int[] arr = (int[]) Array.newInstance(int.class, 5); Array.set(arr, 0, 10); Array.set(arr, 1, 20); System.out.println(Array.get(arr, 0)); // 10 System.out.println(Array.getLength(arr)); // 5 // 数组类型判断 int[] intArr = {1, 2, 3}; System.out.println(intArr.getClass().isArray()); // true System.out.println(intArr.getClass().getComponentType()); // int } }
💡 反射要点: Class 对象 3 种获取方式; getMethod/getField/getConstructor 获取成员; invoke/set/get 动态调用/访问; setAccessible 访问私有成员; 动态代理 (InvocationHandler) 实现 AOP; 反射性能开销大, 需要缓存; MethodHandles 更快; Java 9+ 模块限制反射; 框架核心机制。
