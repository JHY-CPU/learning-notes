TypedDict 与 Literal

// ========== TypedDict ========== from typing import TypedDict, NotRequired, Required, Literal from datetime import datetime # TypedDict: 定义字典的结构 (Python 3.8+) # 类似接口,但用于字典 class UserDict(TypedDict): """用户字典的类型定义""" id: int name: str email: str age: NotRequired[int] # 可选字段 # 使用: user: UserDict = { "id": 1, "name": "Alice", "email": "alice@test.com", } # user["id"] # int ✅ # user["name"] # str ✅ # user["unknown"] # 类型检查错误 ❌ # total=False: 全部可选 class Config(TypedDict, total=False): debug: bool port: int host: str # 继承: class ExtendedUser(UserDict): phone: str # ========== Literal ========== # Literal: 限制值为特定字面量 def set_status(status: Literal["active", "inactive", "pending"]): """状态只能是指定的三个值之一""" pass set_status("active") # ✅ # set_status("unknown") # 类型错误 ❌ # 多个类型: def process(value: Literal[1, "one", True]): pass # 实际应用: def get_env(name: str) -> Literal["dev", "staging", "prod"]: return os.getenv("ENV", "dev") # TypeAlias: 类型别名 from typing import TypeAlias UserId: TypeAlias = int UserList: TypeAlias = list[UserId] def get_user(uid: UserId) -> UserDict: ...

Pydantic 模型

// ========== Pydantic v2 ========== # pip install pydantic from pydantic import BaseModel, Field, EmailStr, ValidationError from datetime import datetime from typing import Optional # Pydantic: 数据验证 + 类型转换 + 序列化 class UserCreate(BaseModel): name: str = Field(..., min_length=2, max_length=50) email: EmailStr age: int = Field(ge=0, le=150) tags: list[str] = [] created_at: datetime = Field(default_factory=datetime.now) # 自定义验证器 @field_validator("name") @classmethod def name_must_be_valid(cls, v): if not v.strip(): raise ValueError("名称不能为空") return v.strip() # 模型验证器 @model_validator(mode="after") def check_age_name(self): if self.age < 18 and self.name == "Admin": raise ValueError("未成年不能设置为 Admin") return self # 使用: try: user = UserCreate( name="Alice", email="alice@test.com", age=25, ) print(user.model_dump()) # 序列化 print(user.model_dump_json()) # JSON except ValidationError as e: print(e.errors()) # 详细的错误信息 # ========== 高级 Pydantic ========== from pydantic import ( ConfigDict, # v2 配置方式 BaseModel, field_serializer, ) class Post(BaseModel): model_config = ConfigDict( from_attributes=True, # ORM 模式 (原 orm_mode) populate_by_name=True, # 允许用字段名或别名 extra="forbid", # 禁止额外字段 ) id: int title: str = Field(alias="post_title") @field_serializer("id") def serialize_id(self, v): return str(v) # 序列化时转为字符串

泛型进阶

// ========== 泛型 ========== from typing import ( Generic, TypeVar, TypeVarTuple, ParamSpec, Concatenate, Self, # Python 3.11+ overload, ) # 基本泛型 T = TypeVar("T") # 任意类型 T_co = TypeVar("T_co", covariant=True) # 协变 T_contra = TypeVar("T_contra", contravariant=True) # 逆变 class Stack(Generic[T]): """类型安全的栈""" def __init__(self) -> None: self._items: list[T] = [] def push(self, item: T) -> None: self._items.append(item) def pop(self) -> T: return self._items.pop() # 使用: int_stack = Stack[int]() int_stack.push(1) int_stack.push(2) # int_stack.push("str") # 类型错误 ❌ # ========== 约束泛型 ========== N = TypeVar("N", int, float) # 只允许 int 或 float def double(value: N) -> N: return value * 2 double(5) # ✅ int double(3.14) # ✅ float # double("hi") # 类型错误 ❌ # ========== Self 类型 ========== # 返回当前类的实例 (Python 3.11+) class Model: @classmethod def create(cls) -> Self: return cls() class UserModel(Model): pass user = UserModel.create() # 类型是 UserModel,而非 Model # ========== @overload ========== @overload def process(data: int) -> str: ... @overload def process(data: str) -> int: ... def process(data: int | str) -> str | int: if isinstance(data, int): return str(data) return len(data) # int → str, str → int (类型检查器理解!)

类型守卫与 isinstance

// ========== 类型守卫 ========== from typing import TypeGuard, cast, assert_type # TypeGuard: 用户定义的类型守卫 (Python 3.10+) def is_str_list(val: list[object]) -> TypeGuard[list[str]]: """判断是否为字符串列表""" return all(isinstance(x, str) for x in val) def process_items(items: list[object]): if is_str_list(items): # 这里 items 被收窄为 list[str] print(",".join(items)) # ✅ else: # 这里还是 list[object] pass # ========== cast ========== # cast: 告诉类型检查器某个值的类型 # 运行时无影响 from typing import cast def get_user() -> object: return {"id": 1, "name": "Alice"} user = cast(dict, get_user()) print(user["name"]) # 类型检查器知道是 dict # ========== assert_type ========== # 运行时断言类型 (测试用) from typing import assert_type x: int = 5 assert_type(x, int) # ✅ 运行时会检查 # ========== Final 与 ClassVar ========== from typing import Final, ClassVar MAX_RETRIES: Final = 3 # MAX_RETRIES = 5 # 类型检查错误 ❌ class Config: DEFAULT_TIMEOUT: ClassVar[int] = 30 # 类变量 def __init__(self): self.timeout: int = Config.DEFAULT_TIMEOUT

// ========== mypy ========== # pip install mypy # mypy: Python 静态类型检查器 # 命令行: # mypy src/ # 检查 src 目录 # mypy src/app.py --strict # 严格模式 # mypy src/ --ignore-missing-imports # 忽略缺失导入 # ========== pyproject.toml 配置 ========== # [tool.mypy] # python_version = "3.11" # strict = True # warn_unused_configs = True # ignore_missing_imports = True # # [[tool.mypy.overrides]] # module = "tests.*" # disallow_untyped_defs = False # ========== 常见 mypy 规则 ========== # 函数需要类型注解 def add(a, b): # mypy 警告: 缺少类型注解 return a + b # 变量不能隐式 Any x = [] # mypy 警告: 需要类型注解 y: list[int] = [] # ✅ # 返回值类型必须匹配 def get_name() -> str: return 123 # mypy 错误: 期望 str,得到 int # None 必须显式声明 def get_user() -> dict | None: if condition: return None return {"id": 1} # ========== 类型检查集成 ========== # pre-commit 配置: # repos: # - repo: https://github.com/pre-commit/mirrors-mypy # rev: v1.8.0 # hooks: # - id: mypy
