综合案例: 缓存代理

// ========== 案例: 带过期时间的缓存装饰器 ========== // 综合运用: 闭包 + 装饰器 + 生成器 + 迭代器 import time from functools import wraps def ttl_cache(expire=60): """带过期时间的缓存装饰器""" cache = {} # 闭包变量: 缓存存储 def decorator(func): @wraps(func) def wrapper(*args, **kwargs): # 生成缓存键 key = (args, tuple(sorted(kwargs.items()))) # 检查缓存是否有效 if key in cache: value, timestamp = cache[key] if time.time() - timestamp < expire: print(f"[缓存命中] {func.__name__}{args}") return value else: print(f"[缓存过期] 删除 {func.__name__}{args}") del cache[key] # 计算新值 result = func(*args, **kwargs) cache[key] = (result, time.time()) print(f"[缓存设置] {func.__name__}{args} = {result}") return result # 添加缓存管理方法 def cache_clear(): cache.clear() print("缓存已清空") def cache_info(): return { "size": len(cache), "keys": list(cache.keys()), "expire": expire, } wrapper.cache_clear = cache_clear wrapper.cache_info = cache_info return wrapper return decorator @ttl_cache(expire=10) # 缓存 10 秒 def fetch_user(user_id): print(f" 实际查询: 获取用户 {user_id}") return {"id": user_id, "name": f"User_{user_id}"} // 第一次: 实际计算 print(fetch_user(1)) # [缓存设置] 获取用户 → 查询... // 第二次: 缓存命中 print(fetch_user(1)) # [缓存命中] 直接返回! // 查看缓存 print(fetch_user.cache_info()) # {"size": 1, ...}

综合案例: 管道处理框架

// ========== 案例: 数据处理管道 ========== // 综合运用: 生成器 + 装饰器 + 迭代器 from functools import wraps # ---------- 管道框架 ---------- class Pipeline: """数据管道: 串联多个处理步骤""" def __init__(self, *steps): self.steps = steps def __iter__(self): """Pipeline 本身是可迭代的""" return self def __next__(self): raise NotImplementedError("请使用 process()") def process(self, data): """处理数据,返回生成器""" result = data for step in self.steps: result = step(result) # 每个步骤可以是生成器 return result # ---------- 步骤装饰器 ---------- def step(func): """标记为管道步骤""" @wraps(func) def wrapper(data): return func(data) return wrapper # ---------- 定义步骤 ---------- @step def read(data): """模拟数据读取""" for item in data: yield item @step def clean(data): """数据清洗""" for item in data: yield item.strip().lower() @step def filter_valid(data): """过滤空值""" for item in data: if item: yield item @step def format_output(data): """格式化""" for item in data: yield f"| {item} |" # ---------- 使用管道 ---------- pipeline = Pipeline(read, clean, filter_valid, format_output) raw_data = [" Alice ", "BOB", "", " CHARLIE ", None, " david "] for result in pipeline.process(raw_data): print(result) // | alice | // | bob | // | charlie | // | david |

进阶函数最佳实践

// ========== 模式选择指南 ========== // 需要做什么? → 用什么? // 想在函数前后加逻辑 → 装饰器 // 想延迟/惰性计算 → 生成器 // 想自定义 for 循环 → 迭代器 // 想捕获外部状态 → 闭包 // 想数据流水线处理 → 生成器管道 // 想缓存函数结果 → 装饰器 + 闭包 // 想创建可复用工具 → 迭代器 + itertools // ========== 常见陷阱对照表 ========== // 陷阱 → 修复 // 装饰器丢失元信息 → @wraps(func) // 循环创建闭包变量共享 → lambda i=i: i 或工厂函数 // 生成器只能消费一次 → 需要时重新创建 // 装饰器参数可选 → 两层/三层参数检测 // 生成器不执行 → 用 list() 或 for 消费 // 迭代器不支持 len() → 用 list() 转换 (会消耗) // 忘记推进协程到 yield → 用协程装饰器自动 next() // ========== 性能考量 ========== // 装饰器: 每次调用有额外函数调用开销 // 生成器: 减少内存,但单次访问略慢于列表 // 闭包: 比全局变量快,比局部变量慢 // 结论: 优先考虑正确性和可读性 // 性能问题只在测量后优化 // ========== 纯函数原则 ========== // 结合函数式编程: def compose(*funcs): """函数组合: compose(f, g)(x) = f(g(x))""" def composed(x): result = x for func in reversed(funcs): result = func(result) return result return composed # 使用闭包+生成器做函数组合: def pipe(value, *funcs): """数据流管道: pipe(x, f, g) = g(f(x))""" result = value for func in funcs: result = func(result) return result # 用生成器做惰性组合: def lazy_pipe(iterable, *funcs): """惰性管道: 每个元素逐个经过所有函数""" for item in iterable: yield pipe(item, *funcs)
💡 进阶函数总结: (1) 闭包+装饰器+生成器+迭代器是 Python 函数式编程四大核心; (2) 装饰器用 @wraps 保留元信息; (3) 生成器节省内存,适合大数据和管道; (4) 迭代器一次性消耗,需要注意; (5) 四种技术经常组合使用 (如缓存装饰器内部用闭包保存状态)。
