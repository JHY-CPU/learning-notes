🏷️ Python 类型提示 typing

类型注解基础语法、typing 模块 (Optional/Union/List/Dict/TypeVar/Generic/Protocol)、类型检查器 mypy、函数重载、运行时类型获取。

类型注解基础

// ========== 变量注解 ========== # Python 3.5+ 类型注解 (运行时无影响) name: str = "Alice" age: int = 25 height: float = 1.68 is_active: bool = True # 复杂类型: from typing import List, Dict, Tuple, Set, Optional, Union # 列表: names: List[str] = ["Alice", "Bob"] matrix: List[List[int]] = [[1, 2], [3, 4]] # 字典: scores: Dict[str, int] = {"Alice": 95, "Bob": 87} nested: Dict[str, List[int]] = {"a": [1, 2]} # 元组: point: Tuple[int, int] = (10, 20) mixed: Tuple[str, int, float] = ("Alice", 25, 1.68) variable: Tuple[int, ...] = (1, 2, 3) # 变长元组 # 集合: tags: Set[str] = {"python", "typing"} // ========== 函数注解 ========== def greet(name: str, age: int = 18) -> str: return f"{name} 今年 {age} 岁" # 多个返回值: def get_user() -> Tuple[str, int]: return ("Alice", 25) # 可选参数: def find_user(user_id: int) -> Optional[str]: if user_id == 1: return "Alice" return None # 或者不返回 # Union: 多种类型 def process(value: Union[int, str]) -> str: return str(value) # Any: 任意类型 from typing import Any def log(data: Any) -> None: print(data)

Optional / Union / Literal

// ========== Optional / Union / Literal ========== from typing import Optional, Union, Literal # Optional[X] = Union[X, None] def get_config(key: str) -> Optional[str]: return {"debug": "true"}.get(key) # 可能 None # Union: 联合类型 def parse_number(value: Union[int, float, str]) -> float: return float(value) # 新旧语法: # 旧: Optional[str] 新 (3.10+): str | None # 旧: Union[int, str] 新 (3.10+): int | str def parse_number_v2(value: int | float | str) -> float: return float(value) # Literal (3.8+): 精确值约束 from typing import Literal def set_mode(mode: Literal["r", "w", "a"]) -> None: print(f"模式: {mode}") set_mode("r") # ✅ 正确 set_mode("x") # ❌ mypy 报错 # Literal 也支持 int/bool def set_page(size: Literal[10, 20, 50]) -> None: pass

TypeVar 泛型

// ========== TypeVar 泛型 ========== from typing import TypeVar, List, Generic # 定义一个类型变量 T = TypeVar("T") # 任意类型 U = TypeVar("U") # 泛型函数: def first(items: List[T]) -> T: """返回列表第一个元素""" return items[0] # 使用: x: int = first([1, 2, 3]) # T → int y: str = first(["a", "b", "c"]) # T → str # 多个类型参数: def pair(a: T, b: U) -> Tuple[T, U]: return (a, b) # 约束类型变量: Number = TypeVar("Number", int, float) def double(x: Number) -> Number: return x * 2 double(5) # ✅ double(3.14) # ✅ double("hi") # ❌ mypy 报错 // ========== Generic 泛型类 ========== from typing import Generic, TypeVar T = TypeVar("T") class Stack(Generic[T]): def __init__(self) -> None: self._items: List[T] = [] def push(self, item: T) -> None: self._items.append(item) def pop(self) -> T: return self._items.pop() # 使用: int_stack = Stack[int]() int_stack.push(1) int_stack.push("hi") # ❌ mypy 报错 str_stack = Stack[str]() str_stack.push("hello")

Protocol 结构类型

// ========== Protocol (Python 3.8+) ========== # 鸭子类型的静态检查: 只要实现了指定方法就行 from typing import Protocol class Flyable(Protocol): def fly(self) -> str: ... class Bird: def fly(self) -> str: return "鸟在飞" class Airplane: def fly(self) -> str: return "飞机在飞" class Fish: def swim(self) -> str: return "鱼在游" # 接受任何 Flyable: def let_it_fly(thing: Flyable) -> None: print(thing.fly()) let_it_fly(Bird()) # ✅ 实现了 fly() let_it_fly(Airplane()) # ✅ 实现了 fly() # let_it_fly(Fish()) # ❌ mypy 报错 (未实现 fly) # 不需要显式继承,自动满足

进阶类型

// ========== Callable ========== from typing import Callable # 函数类型: (参数类型...) → 返回类型 def apply(func: Callable[[int, int], int], a: int, b: int) -> int: return func(a, b) result = apply(lambda x, y: x + y, 3, 5) print(result) # 8 # 无参数无返回值: def run(task: Callable[[], None]) -> None: task() // ========== TypeAlias (3.10+) ========== from typing import TypeAlias # 类型别名: Vector: TypeAlias = List[float] Matrix: TypeAlias = List[Vector] def scale(v: Vector, factor: float) -> Vector: return [x * factor for x in v] # 旧版 (3.10 前): # Vector = List[float] // ========== TypedDict ========== from typing import TypedDict class UserDict(TypedDict): name: str age: int email: str # 类型化字典: user: UserDict = {"name": "Alice", "age": 25, "email": "alice@example.com"} # user["phone"] # ❌ mypy 报错 (未定义字段) // ========== NewType ========== from typing import NewType UserId = NewType("UserId", int) ProductId = NewType("ProductId", int) def get_user(uid: UserId) -> str: return f"user_{uid}" get_user(UserId(42)) # ✅ # get_user(42) # ❌ mypy 报错 (类型不兼容)

类型检查器与运行时

// ========== mypy ========== # mypy 是第三方类型检查器: # pip install mypy # mypy script.py # mypy --strict script.py # 类型注解只是提示,运行时不影响: def add(a: int, b: int) -> int: return a + b print(add("hello ", "world")) # 运行时能执行,但 mypy 报错 // ========== 运行时类型获取 ========== # 类型注解在 __annotations__ 中 def greet(name: str, age: int = 18) -> str: return f"{name} {age}" print(greet.__annotations__) # {'name': , 'age': , 'return': } # isintance 检查 (可选): def process(value: int | str) -> None: if isinstance(value, int): print(f"整数: {value + 1}") else: print(f"字符串: {value.upper()}") // ========== TYPE_CHECKING ========== # 用于条件导入 (避免循环引用): from typing import TYPE_CHECKING if TYPE_CHECKING: # 以下只在类型检查时导入 from some_module import SomeType # TYPE_CHECKING 在运行时为 False
💡 类型提示要点: (1) 变量: name: type = value; 函数: def f(x: type) -> ret_type; (2) typing 核心: Optional/Union/List/Dict/Tuple; (3) TypeVar + Generic 实现泛型; (4) Protocol 静态鸭子类型; (5) Python 3.10+ 可用 X | Y 替代 Union[X, Y]; (6) mypy 做静态类型检查。
