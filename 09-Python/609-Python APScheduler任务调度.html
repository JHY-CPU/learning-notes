⏰ APScheduler 任务调度

APScheduler 四大组件、触发器 (date/interval/cron)、执行器与存储器、集成 Flask/FastAPI、持久化任务。

APScheduler 基础

// ========== APScheduler 架构 ========== # pip install apscheduler # 四大组件: # 1. 触发器 (Trigger) — 决定任务何时执行 # 2. 任务存储器 (Job Store) — 存储任务状态 # 3. 执行器 (Executor) — 执行任务的线程/进程 # 4. 调度器 (Scheduler) — 组织协调以上三者 # ========== 基本使用 ========== from apscheduler.schedulers.blocking import BlockingScheduler from datetime import datetime def my_job(name): print(f"{datetime.now()}: {name} 执行任务") # 创建调度器 scheduler = BlockingScheduler() # 添加任务 scheduler.add_job( my_job, trigger="interval", seconds=5, args=["定时任务"], id="my_job_id", ) # 启动 (阻塞当前线程) scheduler.start() # ========== 调度器类型 ========== # BlockingScheduler: 阻塞式 (适合独立脚本) # BackgroundScheduler: 后台运行 (适合嵌入应用) # AsyncIOScheduler: asyncio 应用 # GeventScheduler: gevent 应用 # TornadoScheduler: Tornado 应用 # TwistedScheduler: Twisted 应用 # BackgroundScheduler (非阻塞): from apscheduler.schedulers.background import BackgroundScheduler scheduler = BackgroundScheduler() scheduler.add_job(my_job, "interval", seconds=10) scheduler.start() # 程序继续执行其他代码...

三种触发器

// ========== 1. date 触发器 ========== from apscheduler.triggers.date import DateTrigger from datetime import datetime, timedelta # 在指定时间执行一次 (任务到期后自动移除) # 方式 1: 指定 datetime scheduler.add_job(my_job, "date", run_date=datetime(2024, 12, 31, 23, 59)) # 方式 2: 指定时间字符串 scheduler.add_job(my_job, "date", run_date="2024-12-31 23:59:00") # 方式 3: 从现在起延迟 scheduler.add_job(my_job, "date", run_date=datetime.now() + timedelta(seconds=30)) # ========== 2. interval 触发器 ========== from apscheduler.triggers.interval import IntervalTrigger # 固定间隔执行 scheduler.add_job(my_job, "interval", seconds=30, # 每 30 秒 # minutes=15, # 每 15 分钟 # hours=1, # 每小时 # days=1, # 每天 # weeks=1, # 每周 start_date="2024-01-01", # 开始时间 end_date="2024-12-31", # 结束时间 jitter=5, # 随机抖动 ±5 秒 ) # ========== 3. cron 触发器 ========== from apscheduler.triggers.cron import CronTrigger # 类似 Linux cron,表达式更灵活 # 每天 9:00 scheduler.add_job(my_job, "cron", hour=9, minute=0) # 每个工作日 9:00-17:00 每 30 分钟 scheduler.add_job(my_job, "cron", day_of_week="mon-fri", hour="9-17", minute="*/30", ) # 每月 1 号和 15 号 8:30 scheduler.add_job(my_job, "cron", day="1,15", hour=8, minute=30, ) # 复杂表达式: scheduler.add_job(my_job, "cron", year="2024", month="1-6", day="1-7,15-21", day_of_week="mon,wed,fri", hour="9-17/2", minute="0-59/15", second="0", timezone="Asia/Shanghai", )

任务存储

// ========== Job Store ========== # 内存存储 (默认): 程序重启后丢失 # SQLAlchemy 存储: 持久化到数据库 # MongoDB 存储: 持久化到 MongoDB # Redis 存储: 持久化到 Redis from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore from apscheduler.jobstores.redis import RedisJobStore from apscheduler.executors.pool import ThreadPoolExecutor, ProcessPoolExecutor # 使用 SQLAlchemy 存储 (推荐) jobstores = { "default": SQLAlchemyJobStore(url="sqlite:///jobs.db"), # "postgres": SQLAlchemyJobStore(url="postgresql://user:pass@localhost/db"), # "redis": RedisJobStore(host="localhost", port=6379), } # 执行器配置 executors = { "default": ThreadPoolExecutor(10), # 默认: 10 线程 "processpool": ProcessPoolExecutor(4), # 进程池: 4 进程 } # 调度器配置 scheduler = BackgroundScheduler( jobstores=jobstores, executors=executors, job_defaults={ "coalesce": True, # 合并错过的任务 (只执行一次) "max_instances": 3, # 同一任务最大并发实例数 "misfire_grace_time": 60, # 错过执行时间后的宽限秒数 }, timezone="Asia/Shanghai", ) # 添加任务后自动持久化到数据库 scheduler.add_job(my_job, "interval", seconds=30, id="persistent_job") # 启动时恢复任务 scheduler.start() # 获取已存储的任务 jobs = scheduler.get_jobs() for job in jobs: print(f"{job.id}: {job.name}, next: {job.next_run_time}")

任务管理与监听

// ========== 任务管理 ========== scheduler = BackgroundScheduler() # 添加任务 (返回 Job 对象) job = scheduler.add_job( my_job, trigger="interval", seconds=30, id="job1", name="我的任务", args=["arg1"], kwargs={"key": "value"}, replace_existing=True, # 如果 ID 已存在则替换 ) # 暂停/恢复 scheduler.pause_job("job1") scheduler.resume_job("job1") # 修改任务 scheduler.reschedule_job("job1", trigger="interval", seconds=60) scheduler.modify_job("job1", max_instances=5) # 删除任务 scheduler.remove_job("job1") scheduler.remove_all_jobs() # 暂停/恢复调度器 scheduler.pause() scheduler.resume() # 关闭调度器 scheduler.shutdown() scheduler.shutdown(wait=False) # 不等待执行中任务 # ========== 事件监听 ========== from apscheduler.events import EVENT_JOB_EXECUTED, EVENT_JOB_ERROR, EVENT_JOB_MISSED def job_listener(event): if event.code == EVENT_JOB_EXECUTED: print(f"任务 {event.job_id} 执行成功") elif event.code == EVENT_JOB_ERROR: print(f"任务 {event.job_id} 执行失败: {event.exception}") elif event.code == EVENT_JOB_MISSED: print(f"任务 {event.job_id} 错过了执行时间") scheduler.add_listener(job_listener, EVENT_JOB_EXECUTED | EVENT_JOB_ERROR | EVENT_JOB_MISSED) # ========== 查看任务状态 ========== for job in scheduler.get_jobs(): print(f""" ID: {job.id} 名称: {job.name} 触发器: {job.trigger} 下次执行: {job.next_run_time} 上次执行: {job.pending} """)

集成 Flask / FastAPI

// ========== Flask 集成 ========== from flask import Flask from apscheduler.schedulers.background import BackgroundScheduler app = Flask(__name__) scheduler = BackgroundScheduler() def scheduled_task(): with app.app_context(): # 需要应用上下文 print("执行定时任务...") @scheduler.scheduled_job("interval", hours=1, id="hourly_task") def hourly_job(): """装饰器方式定义定时任务""" with app.app_context(): print("每小时任务") @app.route("/start") def start_scheduler(): if not scheduler.running: scheduler.start() return "调度器已启动" @app.route("/add") def add_task(): scheduler.add_job( scheduled_task, trigger="interval", seconds=10, id="my_task", replace_existing=True, ) return "任务已添加" # ========== FastAPI 集成 ========== from fastapi import FastAPI from contextlib import asynccontextmanager scheduler = BackgroundScheduler() def my_task(): print("执行定时任务...") @asynccontextmanager async def lifespan(app: FastAPI): # 启动时 scheduler.start() scheduler.add_job(my_task, "interval", seconds=30, id="fastapi_task") yield # 关闭时 scheduler.shutdown() app = FastAPI(lifespan=lifespan) @app.get("/jobs") def list_jobs(): jobs = [] for job in scheduler.get_jobs(): jobs.append({ "id": job.id, "name": job.name, "next_run": str(job.next_run_time), "trigger": str(job.trigger), }) return {"jobs": jobs}
💡 APScheduler 要点: BlockingScheduler 独立脚本, BackgroundScheduler 嵌入应用; date/interval/cron 三种触发器; SQLAlchemyJobStore 持久化; start/shutdown 管理生命周期; 监听事件处理异常。
