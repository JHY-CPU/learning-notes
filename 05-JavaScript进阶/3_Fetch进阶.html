Fetch 进阶

POST/PUT/DELETE、自定义 headers、credentials、fetch 封装。

POST / PUT / DELETE

// POST — 创建资源 fetch('https://api.example.com/posts', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json', }, body: JSON.stringify({ title: 'Hello', content: 'World', }), }); // PUT — 更新资源 fetch(`https://api.example.com/posts/1`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ title: 'Updated', content: '...' }), }); // DELETE — 删除资源 fetch(`https://api.example.com/posts/1`, { method: 'DELETE', }); // body 类型 // JSON: body: JSON.stringify(data) // FormData: body: formData (自动设置 Content-Type) // URLSearchParams: body: new URLSearchParams(params) // Blob: body: blob // 文本: body: 'plain text' // credentials — Cookie 策略 fetch(url, { credentials: 'same-origin' }); // 默认：同源发送 fetch(url, { credentials: 'include' }); // 跨域也发送（需 CORS） fetch(url, { credentials: 'omit' }); // 不发送 Cookie

fetch 封装

// 通用 fetch 封装 async function request(url, options = {}) { const config = { headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${getToken()}`, ...options.headers, }, ...options, }; // 如果是 GET/HEAD，删除 body if (!config.method || config.method === 'GET' || config.method === 'HEAD') { delete config.body; } const res = await fetch(url, config); if (!res.ok) { const err = new Error(`HTTP ${res.status}`); err.status = res.status; err.data = await res.json().catch(() => null); throw err; } // 204 No Content 不解析 body if (res.status === 204) return null; return res.json(); } // 使用 const data = await request('/api/users', { method: 'POST', body: { name: 'Alice', role: 'admin' }, });
