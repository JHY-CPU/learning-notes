Promise 模式 — 重试

Promise 重试工厂、最大重试次数、退避策略、可中止重试。

重试模式实现

// ========== 基础重试 ========== async function retry(fn, maxRetries = 3) { for (let i = 0; i < maxRetries; i++) { try { return await fn(); } catch (err) { if (i === maxRetries - 1) throw err; console.log(`重试 ${i + 1}/${maxRetries}`); } } } // ========== 指数退避 ========== async function retryWithBackoff(fn, maxRetries = 3) { for (let i = 0; i < maxRetries; i++) { try { return await fn(); } catch (err) { if (i === maxRetries - 1) throw err; const delay = Math.min(1000 * Math.pow(2, i), 10000); await new Promise(r => setTimeout(r, delay)); } } } // ========== 完整重试工厂 ========== function createRetry(options = {}) { const { maxRetries = 3, baseDelay = 1000, maxDelay = 10000, factor = 2, jitter = true, // 随机抖动 onRetry = null, // 重试回调 } = options; return async (fn) => { let lastError; for (let i = 0; i <= maxRetries; i++) { try { return await fn(); } catch (err) { lastError = err; if (i === maxRetries) break; let delay = baseDelay * Math.pow(factor, i); delay = Math.min(delay, maxDelay); if (jitter) delay *= 0.5 + Math.random() * 0.5; onRetry?.({ attempt: i + 1, error: err, delay }); await new Promise(r => setTimeout(r, delay)); } } throw lastError; }; }
