import 语法大全

// ========== 导入 named export ========== import { sum, PI } from './math.js'; import { sum as add, PI as pi } from './math.js'; // 重命名 // ========== 导入 default export ========== import defaultFn from './module.js'; // 名称可自定义 import { default as myDefault } from './module.js'; // 等价 // ========== 混合导入 ========== import React, { useState, useEffect } from 'react'; // ========== 全部导入 (namespace) ========== import * as math from './math.js'; math.sum(1, 2); // 通过命名空间访问 math.PI; // ========== 导入并重导出 ========== export { sum } from './math.js'; // 简单转发 // ========== 仅执行模块 (side effect) ========== import './polyfill.js'; // 不导入任何绑定，仅执行模块代码 // ========== import.meta (ES2020) ========== console.log(import.meta.url); // 当前模块 URL console.log(import.meta.resolve); // 解析模块路径 (ES2022) // ========== import 特性 ========== // ✅ 静态导入，提升到模块顶部执行 // ✅ 支持路径解析（相对/绝对/裸模块名需 importmap） // ✅ 导入的绑定是只读引用（不能重新赋值） // ❌ 不能在块/函数中使用静态 import // ❌ 不支持动态路径（变量）
