链式 Promise 实现

// ========== 链式调用的核心 ========== // then() 返回一个新的 Promise // 回调返回值如果是 Promise, 则等待它完成 // 回调返回值如果是普通值, 则作为下一个 then 的参数 class ChainPromise { constructor(executor) { /* ... */ } then(onFulfilled, onRejected) { const nextPromise = new ChainPromise((resolve, reject) => { // 处理 fulfilled 状态 const handleFulfilled = (value) => { queueMicrotask(() => { try { const result = onFulfilled(value); resolve(result); } catch (err) { reject(err); } }); }; // 处理 rejected 状态 const handleRejected = (reason) => { queueMicrotask(() => { try { const result = onRejected(reason); resolve(result); // 错误恢复 } catch (err) { reject(err); } }); }; // 根据当前状态执行 if (this.state === FULFILLED) handleFulfilled(this.value); else if (this.state === REJECTED) handleRejected(this.reason); else { this.onFulfilledCallbacks.push(handleFulfilled); this.onRejectedCallbacks.push(handleRejected); } }); return nextPromise; } } // ========== 静态方法 ========== ChainPromise.resolve = (value) => new ChainPromise(resolve => resolve(value)); ChainPromise.reject = (reason) => new ChainPromise((_, reject) => reject(reason));
💡 关键: then() 返回新 Promise, 形成链式调用。
💡 返回值处理: 普通值直接 resolve, Promise 则等待。
💡 错误传播: 链中任何位置 reject, 跳到下一个 catch。
