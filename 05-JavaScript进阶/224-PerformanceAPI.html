Performance API

高精度时间测量与性能指标收集。

performance.now()

// 高精度时间戳（毫秒，从页面加载开始，精度微秒级） const t = performance.now(); // e.g. 1234.567 // 比 Date.now() 更精确： // - 精度微秒级（vs 毫秒级） // - 单调递增（不会受系统时间影响） // - 可用于精确测量代码执行时间 // 测量执行时间 const start = performance.now(); // ... 要测量的代码 ... const end = performance.now(); console.log(`耗时: ${end - start}ms`);

performance.mark / measure

// 标记时间点 performance.mark('start'); // ... 操作 ... performance.mark('end'); // 测量两个标记之间的时间 performance.measure('my-operation', 'start', 'end'); // 获取测量结果 const measures = performance.getEntriesByType('measure'); // 或: performance.getEntriesByName('my-operation') // 清除标记/测量 performance.clearMarks('start'); performance.clearMeasures('my-operation');

performance.getEntriesByType

// 获取不同类型的性能条目 performance.getEntriesByType('navigation'); // 页面加载信息 // → domContentLoadedEventEnd, loadEventEnd, type(navigate/reload/back_forward) performance.getEntriesByType('paint'); // 绘制时间 // → first-paint, first-contentful-paint (FCP) performance.getEntriesByType('resource'); // 资源加载信息 // 每个资源的 DNS/TCP/TTFB/下载时间 // PerformanceObserver — 监听未来出现的条目 const observer = new PerformanceObserver((list) => { const entries = list.getEntries(); entries.forEach(e => console.log(e.name, e.startTime, e.duration)); }); observer.observe({ entryTypes: ['paint', 'largest-contentful-paint'] });
