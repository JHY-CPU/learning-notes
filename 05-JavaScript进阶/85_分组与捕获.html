分组与捕获

() 捕获组、(?:) 非捕获组、命名组、反向引用。

分组语法

// ========== 捕获组 () ========== // 捕获匹配的子字符串 const match = '2024-01-15'.match(/(\d{4})-(\d{2})-(\d{2})/); // match[1] = '2024' // match[2] = '01' // match[3] = '15' // ========== 非捕获组 (?:) ========== // 仅用于分组，不捕获 'hello world'.match(/(?:hello)\s(world)/); // 只有 1 个捕获组: 'world' // ========== 命名捕获组 (?<name>) (ES2018) ========== const m = '2024-01-15'.match(/(?\d{4})-(?\d{2})-(?\d{2})/); console.log(m.groups.year); // '2024' console.log(m.groups.month); // '01' console.log(m.groups.day); // '15' // ========== 反向引用 ========== // \1, \2... — 引用第 n 个捕获组 // \k<name> — 引用命名组 /(['"]).*?\1/ // 匹配 'text' 或 "text" /(?['"]).*?\k/ // 同上，命名版 // 应用: 匹配重复单词 /\b(\w+)\s+\1\b/ // 匹配 'hello hello' // ========== replace 中的引用 ========== // $1, $2... — 捕获组 // $& — 整个匹配 // $` — 匹配前文本 // $' — 匹配后文本 '2024-01-15'.replace(/(\d{4})-(\d{2})-(\d{2})/, '$2/$3/$1') // '01/15/2024'
