WebSocket

全双工通信协议，客户端与服务器实时双向传输。

基本用法

// 创建 WebSocket 连接 const ws = new WebSocket('ws://localhost:8080'); // ws:// — 普通连接 // wss:// — 加密连接（推荐，类似 HTTPS） // readyState // 0: CONNECTING — 连接中 // 1: OPEN — 已连接 // 2: CLOSING — 关闭中 // 3: CLOSED — 已关闭 // 事件 ws.onopen = () => { // 连接建立 ws.send('Hello Server!'); }; ws.onmessage = (e) => { // 收到消息 console.log(e.data); // 字符串（默认）或 Blob/ArrayBuffer }; ws.onerror = (e) => { // 错误 console.error('WS Error:', e); }; ws.onclose = (e) => { // 连接关闭 e.code; // 关闭码（1000=正常, 1006=异常断开...） e.reason; // 关闭原因（字符串） e.wasClean; // 是否正常关闭 }; // 发送消息 ws.send('Hello'); // 发送文本 ws.send(JSON.stringify({type: 'chat', msg: 'Hello'})); // JSON ws.send(blob); // 发送二进制 ws.send(arrayBuffer); // 发送 ArrayBuffer // 关闭连接 ws.close(); // 正常关闭（code=1000） ws.close(1000, '再见'); // 指定关闭码和原因 // 心跳机制（防止连接被断开） setInterval(() => { if (ws.readyState === WebSocket.OPEN) { ws.send(JSON.stringify({ type: 'ping' })); } }, 30000); // ========== 自动重连机制 ========== function createWebSocket(url, options = {}) { const { maxRetries = 5, retryDelay = 1000 } = options; let retries = 0; function connect() { const ws = new WebSocket(url); ws.onopen = () => { console.log('✅ 连接成功'); retries = 0; // 重置重试计数 }; ws.onclose = (e) => { console.log(`🔌 连接关闭: code=${e.code}`); if (retries < maxRetries) { retries++; const delay=Math.min(retryDelay * Math.pow(2, retries - 1), 30000); console.log(`🔄 ${delay}ms 后尝试重连 (${retries}/${maxRetries})`); setTimeout(connect, delay); // 指数退避 } else { console.error('❌ 达到最大重试次数'); } }; ws.onerror=(e)=> { console.error('❌ WebSocket 错误'); }; return ws; } return connect(); } // 使用: // const ws = createWebSocket('wss://your-server.com', { // maxRetries: 10, // retryDelay: 2000 // }); // ========== 二进制数据类型 ========== // 默认接收文本,可以设置为接收 ArrayBuffer 或 Blob ws.binaryType = 'arraybuffer'; // 或 'blob' ws.onmessage = (e) => { if (e.data instanceof ArrayBuffer) { // 处理二进制数据 const uint8Array = new Uint8Array(e.data); } else { // 处理文本数据 console.log(e.data); } };
💡 WebSocket 支持发送文本和二进制数据。
⚠️ 需要在服务器端配合（Node.js ws 库 / 其他后端框架）。
💡 建立连接前需要 HTTP 升级握手（由浏览器自动完成）。
⚠️ 断线后不会自动重连，需要手动实现重连逻辑。
