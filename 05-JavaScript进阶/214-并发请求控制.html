并发请求控制

控制同时进行的请求数量，防止过载。

限制并发数

// 方案1: 分块执行（简单分批） async function chunkedFetch(urls, chunkSize = 3) { const results = []; for (let i = 0; i < urls.length; i += chunkSize) { const chunk = urls.slice(i, i + chunkSize); const chunkResults = await Promise.all( chunk.map(url => fetch(url).then(r => r.json())) ); results.push(...chunkResults); } return results; } // 方案2: 并发队列（负载均衡） async function parallelLimit(tasks, limit) { const results = []; const executing = []; for (const task of tasks) { const p = Promise.resolve().then(() => task()); results.push(p); if (limit <= tasks.length) { const e = p.then(() => executing.splice(executing.indexOf(e), 1)); executing.push(e); if (executing.length >= limit) { await Promise.race(executing); } } } return Promise.all(results); } // 方案3: pLimit 模式（推荐） function pLimit(concurrency) { const queue = []; let active = 0; async function next() { active++; const { fn, resolve, reject } = queue.shift(); try { resolve(await fn()); } catch (err) { reject(err); } finally { active--; if (queue.length) next(); } } return function (fn, ...args) { return new Promise((resolve, reject) => { queue.push({ fn: () => fn(...args), resolve, reject }); if (active < concurrency) next(); }); }; }
💡 推荐使用 pLimit 模式，灵活控制并发数。
⚠️ 浏览器对同一域名有并发限制（通常 6 个）。
💡 并发控制可以避免服务器过载和客户端内存溢出。
