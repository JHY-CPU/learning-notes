串行执行实现

// ========== reduce 串行 ========== const tasks = [task1, task2, task3]; const result = await tasks.reduce((prev, task) => { return prev.then(task); }, Promise.resolve()); // 简写 const result = await tasks.reduce( (prev, task) => prev.then(task), Promise.resolve() ); // ========== async/await 串行 ========== async function runSerial(tasks) { const results = []; for (const task of tasks) { results.push(await task()); } return results; } // ========== for...of 串行 ========== async function serialForOf(tasks) { const results = []; for (const task of tasks) { const result = await task; results.push(result); } return results; } // ========== 串行与并行混合 ========== // 分阶段执行: 阶段1串行, 阶段2并行 const step1 = await taskA(); // 串行 const step2 = await taskB(step1); // 串行 const [a, b, c] = await Promise.all([ // 并行 taskC(step2), taskD(step2), taskE(step2), ]);
