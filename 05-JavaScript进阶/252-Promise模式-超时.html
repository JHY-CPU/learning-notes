Promise 模式 — 超时

Promise.race 超时、AbortSignal 超时、超时与清理。

超时实现

// ========== Promise.race 超时 ========== function withTimeout(promise, ms) { const timeout = new Promise((_, reject) => { setTimeout(() => reject(new Error('请求超时')), ms); }); return Promise.race([promise, timeout]); } // 使用 try { const data = await withTimeout(fetch('/api/data'), 5000); } catch (err) { if (err.message === '请求超时') { // 处理超时 } } // ========== AbortController 超时 ========== function withAbortTimeout(promise, ms) { const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), ms); // 包装原 Promise 以支持 signal const wrapped = promise.catch(err => { if (err.name === 'AbortError') throw new Error('请求超时'); throw err; }); return Promise.race([ wrapped, new Promise((_, reject) => { controller.signal.addEventListener('abort', () => { reject(new Error('请求超时')); }); }) ]).finally(() => clearTimeout(timeoutId)); } // ========== AbortSignal.timeout() (ES2022) ========== // 浏览器原生支持 const response = await fetch('/api/data', { signal: AbortSignal.timeout(5000) }); // ========== 超时+重试组合 ========== async function fetchWithTimeoutRetry(url, options = {}) { const { timeout = 5000, retries = 3 } = options; for (let i = 0; i < retries; i++) { try { const res = await fetch(url, { signal: AbortSignal.timeout(timeout) }); return res; } catch (err) { if (i === retries - 1) throw err; } } }
