ES 模块 — export

named export / default export / 重导出 re-export / export * from。

export 语法大全

// ========== named export (命名导出) ========== export const PI = 3.14159; export function sum(a, b) { return a + b; } export class Person {} // 先声明后导出 const name = 'Alice'; const age = 25; export { name, age }; // 导出时重命名 export { name as userName, age as userAge }; // ========== default export (默认导出) ========== export default function() { console.log('default'); } // 或 export default class {} // 或 const obj = {}; export default obj; // 每个模块只能有一个 default export // ========== 混合导出 ========== export const VERSION = '1.0'; export default function main() {} // ========== 重导出 (re-export) ========== export { foo, bar } from './utils.js'; // 选择性重导出 export { foo as myFoo } from './utils.js'; // 重命名后重导出 export * from './math.js'; // 全部重导出（不含 default） export * as math from './math.js'; // 命名空间重导出 (ES2020) export { default } from './other.js'; // 重导出 default export { default as myDefault } from './other.js'; // ========== 注意事项 ========== // export 必须位于模块顶层，不能在块/函数内 // export 是静态声明，在解析阶段确定 // 导出的绑定是实时绑定 (live binding)
