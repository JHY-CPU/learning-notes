Base64 与编码

atob/btoa、中文 Base64、TextEncoder/TextDecoder、UTF-8/UTF-16。

编码 API

// ========== Base64 ========== // btoa() — 字符串 → Base64 (binary to ASCII) // atob() — Base64 → 字符串 btoa('hello'); // 'aGVsbG8=' atob('aGVsbG8='); // 'hello' // ⚠️ btoa 只支持 Latin1 字符 // 中文会报错！ // btoa('你好'); // DOMException // 中文 Base64 (需要先编码为 UTF-8) function utf8ToBase64(str) { return btoa(encodeURIComponent(str).replace( /%([0-9A-F]{2})/g, (_, p1) => String.fromCharCode(parseInt(p1, 16)) )); } function base64ToUtf8(base64) { return decodeURIComponent(atob(base64).split('') .map(c => '%' + c.charCodeAt(0).toString(16).padStart(2, '0')) .join('')); } // ========== TextEncoder / TextDecoder ========== // TextEncoder — 字符串 → Uint8Array (UTF-8) const encoder = new TextEncoder(); encoder.encode('hello'); // Uint8Array [104, 101, 108, 108, 111] // TextDecoder — Uint8Array → 字符串 const decoder = new TextDecoder(); decoder.decode(new Uint8Array([104, 101, 108, 108, 111])); // 'hello' // 编码类型 // TextEncoder: 只支持 UTF-8 // TextDecoder: 支持多种编码 (UTF-8, UTF-16, latin1, etc.)
