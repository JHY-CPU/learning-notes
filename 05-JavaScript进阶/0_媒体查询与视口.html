媒体查询与视口

matchMedia API、resize 事件、视口宽高获取。

视口（Viewport）

// 获取视口尺寸 window.innerWidth // 视口宽度（含滚动条） window.innerHeight // 视口高度（含滚动条） document.documentElement.clientWidth // 视口宽度（不含滚动条） document.documentElement.clientHeight // 视口高度（不含滚动条） // ⚠️ innerWidth/Height 包含滚动条 // ⚠️ clientWidth/Height 不包含滚动条 // HTML viewport meta // <meta name="viewport" content="width=device-width, initial-scale=1.0"> // width=device-width — 视口宽度=设备宽度 // initial-scale=1.0 — 初始缩放比例 // minimum-scale / maximum-scale — 最小/最大缩放 // user-scalable=yes/no — 是否允许缩放

matchMedia — CSS 媒体查询的 JS API

// window.matchMedia(mediaQueryString) // 返回 MediaQueryList 对象 const mq = window.matchMedia('(max-width: 768px)'); // matches — 是否匹配 if (mq.matches) { console.log('视口宽度 ≤ 768px（移动端）'); } else { console.log('视口宽度 > 768px（桌面端）'); } // 常见媒体查询： matchMedia('(max-width: 767px)') // 手机 matchMedia('(min-width: 768px) and (max-width: 1023px)') // 平板 matchMedia('(min-width: 1024px)') // 桌面 matchMedia('(prefers-color-scheme: dark)') // 深色模式 matchMedia('(prefers-reduced-motion: reduce)') // 减少动画 matchMedia('(orientation: portrait)') // 竖屏 matchMedia('(orientation: landscape)') // 横屏 matchMedia('(min-resolution: 2dppx)') // Retina 屏 matchMedia('(hover: hover)') // 支持 hover matchMedia('(pointer: coarse)') // 触摸设备 // 监听变化 mq.addEventListener('change', function(e) { if (e.matches) { console.log('现在匹配了'); } else { console.log('不再匹配'); } }); // 旧方式（已弃用） // mq.addListener(handler) // mq.removeListener(handler)

resize 事件

// resize 事件 — 窗口大小变化时触发 window.addEventListener('resize', function() { console.log('窗口大小变化:', window.innerWidth, 'x', window.innerHeight); }); // ⚠️ resize 触发非常频繁！ // ⚠️ 必须防抖或节流！ // 防抖 resize let resizeTimer; window.addEventListener('resize', function() { clearTimeout(resizeTimer); resizeTimer = setTimeout(() => { console.log('resize 结束（防抖后）'); }, 200); }); // 节流 resize let lastResizeTime = 0; window.addEventListener('resize', function() { const now = Date.now(); if (now - lastResizeTime >= 200) { lastResizeTime = now; console.log('resize 节流:', window.innerWidth); } }); // 注意：移动端 resize 事件可能不触发 // 推荐用 matchMedia 检测视口变化

设备像素比与缩放

// devicePixelRatio — 物理像素 / CSS 像素 window.devicePixelRatio; // 1x 屏幕（普通屏）— 1 // 2x 屏幕（Retina）— 2 // 3x 屏幕（iPhone 等）— 3 // 应用场景：提供高清图片 function getImageUrl(base) { const dpr = window.devicePixelRatio; return dpr >= 2 ? `${base}@2x.png` : `${base}.png`; } // 根据像素比渲染 Canvas const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const dpr = window.devicePixelRatio || 1; canvas.width = displayWidth * dpr; canvas.height = displayHeight * dpr; canvas.style.width = displayWidth + 'px'; canvas.style.height = displayHeight + 'px'; ctx.scale(dpr, dpr);
