Generator 进阶

yield* 委托、next() 传参、generator.throw()、return()。

Generator 进阶语法

// ========== yield* 委托 ========== function* gen1() { yield 1; yield 2; } function* gen2() { yield 'a'; yield* gen1(); // 委托给 gen1 yield 'b'; } // gen2() 产生: 'a', 1, 2, 'b' // yield* 可委托给任意可迭代对象 yield* [1, 2, 3]; // 委托数组 yield* 'abc'; // 委托字符串 yield* otherGen(); // 委托另一个 generator // ========== next() 传参 ========== function* calc() { const a = yield '输入 a'; // 暂停，等待外部传值 const b = yield '输入 b'; yield a + b; } const g = calc(); g.next(); // 第一个 next() 启动 g.next(10); // a = 10 g.next(20); // b = 20 → value: 30 // ========== generator.throw() ========== function* safeGen() { try { yield 1; yield 2; } catch (e) { yield `错误: ${e.message}`; } } const g2 = safeGen(); g2.next(); // { value: 1 } g2.throw(new Error('出错了')); // 被 catch 捕获 // ========== generator.return() ========== function* countDown() { yield 3; yield 2; yield 1; } const g3 = countDown(); g3.next(); // { value: 3 } g3.return('结束'); // { value: '结束', done: true } // 后续 next() 都是 { done: true }
