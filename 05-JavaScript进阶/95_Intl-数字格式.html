Intl 数字格式

NumberFormat、货币/百分比/单位、小数位数、千分位分组。

NumberFormat 用法

// ========== NumberFormat 选项 ========== const options = { // 样式 style: 'decimal', // 'decimal' (默认) / 'currency' / 'percent' / 'unit' // 货币 (style='currency' 时需要) currency: 'CNY', // ISO 4217 货币代码 currencyDisplay: 'symbol', // 'symbol' / 'code' / 'name' / 'narrowSymbol' // 单位 (style='unit' 时需要, ES2020) unit: 'kilometer', // 单位名 unitDisplay: 'long', // 'long' / 'short' / 'narrow' // 数字格式 minimumIntegerDigits: 1, // 最小整数位数 (1-21) minimumFractionDigits: 0, // 最小小数位数 (0-20) maximumFractionDigits: 3, // 最大小数位数 (1-20) minimumSignificantDigits: 1, // 最小有效数字 maximumSignificantDigits: 21, // 最大有效数字 // 分组 useGrouping: true, // 是否千分位分组 // 符号显示 signDisplay: 'auto', // 'auto' / 'always' / 'exceptZero' / 'never' }; // ========== 常用货币代码 ========== // CNY — 人民币 // USD — 美元 // EUR — 欧元 // JPY — 日元 // GBP — 英镑 // HKD — 港币 // ========== 常用单位 ========== // kilometer / meter / centimeter // kilogram / gram // liter / milliliter // mile / pound / ounce // percent (百分比用 style='percent')
