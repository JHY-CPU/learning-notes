IndexedDB 基础

浏览器本地数据库，支持结构化数据存储和索引。

基本操作

// 1. 打开数据库 const request = indexedDB.open('MyDB', 1); // 数据库名, 版本号 request.onupgradeneeded = (event) => { // 数据库创建或版本升级时调用 const db = event.target.result; const store = db.createObjectStore('users', { keyPath: 'id', // 主键字段 autoIncrement: false, // 是否自增 }); // 创建索引 store.createIndex('email', 'email', { unique: true }); store.createIndex('name', 'name', { unique: false }); }; request.onsuccess = (event) => { const db = event.target.result; // 操作数据库 }; request.onerror = (event) => { console.error('数据库打开失败:', event.target.error); }; // 2. 增删改查（在事务中进行） const tx = db.transaction('users', 'readwrite'); // 'readonly' | 'readwrite' const store = tx.objectStore('users'); // 增 store.add({ id: 1, name: 'Alice', email: 'alice@test.com' }); // 删 store.delete(1); // 改 store.put({ id: 1, name: 'Alice Updated', email: 'alice@test.com' }); // 查（单个） const getRequest = store.get(1); getRequest.onsuccess = () => console.log(getRequest.result); // 查（所有） const allRequest = store.getAll(); allRequest.onsuccess = () => console.log(allRequest.result); // 事务结束 tx.oncomplete = () => console.log('事务完成'); tx.onerror = () => console.error('事务出错'); // 3. 关闭数据库 db.close();
💡 IndexedDB 是异步的，不会阻塞主线程。
⚠️ 大部分操作基于事件回调（不是 Promise），需要封装。
⚠️ 部分浏览器隐私模式下不可用。
💡 每个源（origin）的存储空间独立，通常没有大小限制（根据磁盘空间）。

封装为 Promise

// 封装简化 IndexedDB 操作 class DB { constructor(dbName, version = 1) { this.dbName = dbName; this.version = version; } async open(upgradeFn) { return new Promise((resolve, reject) => { const req = indexedDB.open(this.dbName, this.version); req.onupgradeneeded = (e) => upgradeFn?.(e.target.result); req.onsuccess = (e) => { this.instance = e.target.result; resolve(this.instance); }; req.onerror = (e) => reject(e.target.error); }); } async get(storeName, key) { /* ... */ } async set(storeName, value) { /* ... */ } async delete(storeName, key) { /* ... */ } async getAll(storeName) { /* ... */ } }
