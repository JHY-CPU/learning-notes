异步 Generator

async function*、for await...of、异步数据流、分页生成。

异步 Generator 语法

// ========== 异步 Generator ========== async function* asyncGen() { const data1 = await fetch('/api/data1').then(r => r.json()); yield data1; const data2 = await fetch('/api/data2').then(r => r.json()); yield data2; } // ========== for await...of ========== for await (const value of asyncGen()) { console.log(value); } // ========== 异步可迭代对象 ========== const asyncIterable = { [Symbol.asyncIterator]() { let i = 0; return { async next() { await new Promise(r => setTimeout(r, 500)); if (i < 3) return { value: i++, done: false }; return { value: undefined, done: true }; } }; } }; for await (const val of asyncIterable) { console.log(val); // 0, 1, 2（间隔 500ms） } // ========== 应用场景 ========== // ✅ 分页数据加载 (每页请求一次 API) // ✅ 异步事件流 (WebSocket / SSE) // ✅ 大文件逐行处理 // ✅ 定时数据轮询
