IndexedDB 进阶

游标、索引、范围查询、版本升级。

游标 (Cursor)

// 游标遍历 const tx = db.transaction('store', 'readonly'); const store = tx.objectStore('store'); const request = store.openCursor(); // 或 store.openCursor(range, direction) request.onsuccess = (event) => { const cursor = event.target.result; if (cursor) { console.log(cursor.key, cursor.value); // 当前条目 cursor.continue(); // 移动到下一条 // cursor.advance(5); // 跳过 5 条 } else { console.log('遍历完成'); } }; // direction 参数 // 'next' — 升序（默认） // 'prev' — 降序 // 'nextunique' — 升序（去重） // 'prevunique' — 降序（去重） // 使用索引游标 const index = store.index('email'); const idxReq = index.openCursor(IDBKeyRange.only('alice@test.com')); // 或者 index.openCursor() 按索引顺序遍历 // 游标修改/删除 cursor.update(newValue); // 修改当前条目 cursor.delete(); // 删除当前条目

索引与范围

// 创建索引 store.createIndex('name', 'name', { unique: false }); store.createIndex('email', 'email', { unique: true }); store.createIndex('age_price', ['age', 'price'], { unique: false }); // 复合索引 // IDBKeyRange — 范围查询 IDBKeyRange.only(value); // 精确匹配 IDBKeyRange.lowerBound(value, open); // >= value (open=true 表示 >) IDBKeyRange.upperBound(value, open); // <= value IDBKeyRange.bound(lower, upper, lowerOpen, upperOpen); // 范围内 // 使用范围 const range = IDBKeyRange.bound(18, 60); // age 18-60 store.index('age').openCursor(range); // 直接获取索引值 store.index('email').get('alice@test.com'); // 通过索引获取 store.index('email').getAll('@example.com'); // TODO: fix → getAll(range)

版本升级

// 修改数据库结构需要增加版本号 const req = indexedDB.open('MyDB', 2); // 版本从 1 → 2 req.onupgradeneeded = (event) => { const db = event.target.result; const oldVersion = event.oldVersion; // 1 const newVersion = event.newVersion; // 2 if (oldVersion < 2) { // 创建新的对象仓库 const store = db.createObjectStore('newStore', { keyPath: 'id' }); // 创建新索引 store.createIndex('field', 'field', { unique: false }); } // 删除对象仓库 if (db.objectStoreNames.contains('oldStore')) { db.deleteObjectStore('oldStore'); } };
⚠️ 版本号只能递增，不能回退。
⚠️ onupgradeneeded 中不能并行使用多个事务。
💡 游标适合大数据量遍历（不会把所有数据加载到内存）。
