实战：文本替换

数字千分位、驼峰命名转换、敏感信息脱敏、模板替换。

常用替换场景

// ========== 数字千分位 ========== function toThousands(num) { return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ','); } // 1234567 → '1,234,567' // 1234.56 → '1,234.56' // ========== 驼峰命名转换 ========== // kebab-case → camelCase 'hello-world-js'.replace(/-(\w)/g, (_, c) => c.toUpperCase()); // 'helloWorldJs' // camelCase → kebab-case 'helloWorldJs'.replace(/([A-Z])/g, '-$1').toLowerCase(); // 'hello-world-js' // ========== 敏感信息脱敏 ========== // 手机号: 138****1234 '13812345678'.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2'); // 邮箱: a***@example.com 'alice@example.com'.replace(/^(.)(.*)(@.*)$/, (_, f, m, e) => f + '*'.repeat(Math.min(m.length, 3)) + e ); // ========== 模板替换 ========== function template(str, data) { return str.replace(/\{\{(\w+)\}\}/g, (_, key) => data[key] ?? ''); } template('Hello, {{name}}! You are {{age}}.', { name: 'Alice', age: 25 }); // 'Hello, Alice! You are 25.'
