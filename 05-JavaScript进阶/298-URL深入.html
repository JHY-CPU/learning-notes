URL 和 URLSearchParams

// ========== URL 构造 ========== const url = new URL('https://example.com:8080/path/to/page?name=Alice&age=25#section'); // URL 属性 url.href // 完整 URL url.protocol // 'https:' url.hostname // 'example.com' url.port // '8080' url.host // 'example.com:8080' url.pathname // '/path/to/page' url.search // '?name=Alice&age=25' url.hash // '#section' url.origin // 'https://example.com:8080' // URL 修改 url.pathname = '/new/path'; url.searchParams.set('lang', 'zh'); // ========== URLSearchParams ========== const params = new URLSearchParams('name=Alice&age=25&lang=zh-CN'); params.get('name') // 'Alice' params.getAll('name') // ['Alice'] params.has('lang') // true params.append('tag', 'js'); params.set('name', 'Bob'); params.delete('age'); // 遍历 for (const [key, value] of params) { console.log(key, value); } // 转为字符串 params.toString(); // 'name=Bob&lang=zh-CN&tag=js' // ========== 编码/解码 ========== encodeURIComponent('hello world') // 'hello%20world' decodeURIComponent('hello%20world') // 'hello world' encodeURI('https://example.com/路径') // 完整 URI decodeURI('https://example.com/%E8%B7%AF%E5%BE%84')
