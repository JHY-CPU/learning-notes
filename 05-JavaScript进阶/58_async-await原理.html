async/await 原理

Generator + Promise 模拟 async/await、自执行生成器。

async/await 本质

// ========== async/await 本质 ========== // async 函数返回 Promise // await 是 then 的语法糖 // 这段代码: async function fetchData() { const data = await fetch('/api/data'); return data.json(); } // 相当于: function fetchData() { return fetch('/api/data').then(r => r.json()); } // ========== Generator + Promise 模拟 ========== // async/await = Generator + 自执行器 function asyncToGenerator(genFn) { return function (...args) { const gen = genFn.apply(this, args); return new Promise((resolve, reject) => { function step(key, arg) { try { const result = gen[key](arg); const { value, done } = result; if (done) { resolve(value); } else { Promise.resolve(value).then( v => step('next', v), e => step('throw', e) ); } } catch (err) { reject(err); } } step('next'); }); }; } // ========== 手写 async 函数 ========== // function* 模拟 async // yield 模拟 await // 自执行器驱动 Generator const asyncFn = asyncToGenerator(function* () { const a = yield Promise.resolve(1); const b = yield Promise.resolve(a + 2); return b + 3; // 最终: 6 }); asyncFn().then(console.log); // 6
