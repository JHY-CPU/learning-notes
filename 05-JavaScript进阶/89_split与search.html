split 与 search

String.split 正则分割、String.search 位置查找。

split 与 search 语法

// ========== String.split() ========== // 用正则或字符串分割 'a,b,c'.split(','); // ['a', 'b', 'c'] 'a,b,c'.split(/,/); // ['a', 'b', 'c'] // 正则分割，更灵活 'a1b2c3'.split(/\d/); // ['a', 'b', 'c'] 'hello world'.split(/\s+/); // ['hello', 'world'] 'a--b---c'.split(/-+/); // ['a', 'b', 'c'] // 限制分割数量 'a,b,c,d'.split(',', 2); // ['a', 'b'] // 保留捕获组中的内容 'a1b2c'.split(/(\d)/); // ['a', '1', 'b', '2', 'c'] // ========== String.search() ========== // 返回第一个匹配的索引，未找到返回 -1 'Hello World'.search(/World/); // 6 'Hello World'.search(/w/); // -1 (区分大小写) 'Hello World'.search(/w/i); // 6 (不区分大小写) // search 不支持全局 g, 永远返回第一个匹配 // ========== split 应用场景 ========== // 解析 CSV const csv = 'a,b,c'.split(','); // 解析 URL 参数 const params = 'a=1&b=2'.split(/[&=]/); // 提取单词 const words = 'Hello, world! JS.'.split(/\W+/).filter(Boolean); // 按行分割 (保留空行) const lines = 'a\n\nb\nc'.split(/\r?\n/);
