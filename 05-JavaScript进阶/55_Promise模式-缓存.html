Promise 缓存实现

// ========== 基础 Promise 缓存 ========== const cache = new Map(); function cachedFetch(url) { if (cache.has(url)) { return Promise.resolve(cache.get(url)); } return fetch(url) .then(r => r.json()) .then(data => { cache.set(url, data); return data; }); } // ========== 避免重复请求（请求合并） ========== const pendingCache = new Map(); function deduplicatedFetch(url) { // 如果已有正在进行的请求，复用同一个 Promise if (pendingCache.has(url)) { return pendingCache.get(url); } const promise = fetch(url) .then(r => r.json()) .finally(() => pendingCache.delete(url)); pendingCache.set(url, promise); return promise; } // ========== 带 TTL 的缓存 ========== function createCache(ttlMs = 60000) { const store = new Map(); return (key, fetcher) => { const cached = store.get(key); if (cached && Date.now() - cached.time < ttlMs) { return Promise.resolve(cached.value); } return Promise.resolve(fetcher()).then(value => { store.set(key, { value, time: Date.now() }); return value; }); }; } // ========== SWR 缓存 ========== function swrCache(fetchFn) { let cached = null; let pending = null; return () => { // 有缓存立即返回旧值，后台刷新 if (cached) { if (!pending) { pending = fetchFn().then(data => { cached = data; pending = null; }).catch(() => { pending = null; }); } return Promise.resolve(cached); } // 无缓存就等待 if (!pending) { pending = fetchFn().then(data => { cached = data; pending = null; }); } return pending.then(() => cached); }; }
