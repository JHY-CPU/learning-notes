媒体查询深入（matchMedia 高级）

响应式设计中 JavaScript 媒体查询的高级用法。

CSS 媒体查询与 JS 配合

// prefers-* 用户偏好媒体查询 matchMedia('(prefers-color-scheme: dark)') // 深色模式 matchMedia('(prefers-color-scheme: light)') // 浅色模式 matchMedia('(prefers-reduced-motion: reduce)') // 减少动画 matchMedia('(prefers-reduced-motion: no-preference)') matchMedia('(prefers-contrast: more)') // 高对比度 matchMedia('(prefers-contrast: less)') // 低对比度 matchMedia('(prefers-transparency: reduce)') // 减少透明度 matchMedia('(prefers-reduced-data: reduce)') // 省流量模式 matchMedia('(prefers-color-scheme: light)') // 浅色 matchMedia('(forced-colors: active)') // 强制颜色模式 // 打印样式检测 matchMedia('print') // 是否在打印模式 // 屏幕尺寸与方向 matchMedia('(orientation: portrait)') // 竖屏 matchMedia('(orientation: landscape)') // 横屏 matchMedia('(aspect-ratio: 16/9)') // 宽高比 matchMedia('(min-resolution: 2dppx)') // Retina 屏 // 交互能力 matchMedia('(hover: hover)') // 支持悬停 matchMedia('(hover: none)') // 不支持悬停（触屏） matchMedia('(pointer: fine)') // 精确指针（鼠标） matchMedia('(pointer: coarse)') // 粗略指针（触摸） matchMedia('(any-hover: hover)') // 任意设备支持悬停 matchMedia('(any-pointer: fine)') // 任意设备有精确指针 // 组合查询 matchMedia('(max-width: 768px) and (prefers-color-scheme: dark)') matchMedia('(min-width: 1024px), (orientation: landscape)') // matchMedia 返回 MediaQueryList const mq = matchMedia('(max-width: 600px)'); mq.matches // boolean mq.media // 查询字符串 mq.addEventListener('change', handler) mq.removeEventListener('change', handler)
