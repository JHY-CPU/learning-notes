全局错误处理

window.onerror、unhandledrejection、错误边界、上报策略。

全局错误处理 API

// ========== window.onerror (运行时错误) ========== window.onerror = function(message, source, lineno, colno, error) { console.error('全局错误:', message, source, lineno, colno); // 返回 true 阻止默认错误处理 return true; }; // 或者用 addEventListener window.addEventListener('error', (event) => { console.error('捕获错误:', event.error); // event.preventDefault() 阻止默认行为 }); // ========== 未处理的 Promise 拒绝 ========== window.onunhandledrejection = function(event) { console.error('未处理的 Promise 拒绝:', event.reason); event.preventDefault(); // 阻止控制台警告 }; window.addEventListener('unhandledrejection', (event) => { console.error('未处理拒绝:', event.reason); event.preventDefault(); }); // ========== 资源加载错误 ========== window.addEventListener('error', (event) => { // 区分资源错误和运行时错误 if (event.target !== window) { console.error('资源加载失败:', event.target.src || event.target.href); } }, true); // 捕获阶段 // ========== 手动上报 ========== function reportError(error, context = {}) { const body = { message: error.message, stack: error.stack, url: window.location.href, userAgent: navigator.userAgent, timestamp: Date.now(), ...context, }; // 使用 sendBeacon 上报 (页面卸载时也能发送) navigator.sendBeacon?.('/api/error', JSON.stringify(body)); // 或 fetch fetch('/api/error', { method: 'POST', body: JSON.stringify(body) }); }
