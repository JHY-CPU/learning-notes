请求重试与超时

为 fetch 添加超时控制和自动重试机制。

超时控制

// 方式1: AbortSignal.timeout (Chrome 103+) fetch(url, { signal: AbortSignal.timeout(5000) }) .catch(err => { if (err.name === 'TimeoutError') console.log('请求超时'); }); // 方式2: AbortController 手动 function fetchWithTimeout(url, timeoutMs = 5000) { const controller = new AbortController(); const timer = setTimeout(() => controller.abort(`超时 ${timeoutMs}ms`), timeoutMs); return fetch(url, { signal: controller.signal }) .finally(() => clearTimeout(timer)); } // 方式3: Promise.race function fetchWithRace(url, timeoutMs = 5000) { return Promise.race([ fetch(url), new Promise((_, reject) => setTimeout(() => reject(new Error('超时')), timeoutMs) ), ]); }

重试机制

// 带重试的 fetch async function fetchWithRetry(url, options = {}, retries = 3) { for (let i = 0; i < retries; i++) { try { const res = await fetch(url, options); if (!res.ok) throw new Error(`HTTP ${res.status}`); return await res.json(); } catch (err) { if (i === retries - 1) throw err; // 最后一次失败则抛出 console.log(`第 ${i + 1} 次失败 (${err.message}), 准备重试...`); await delay(Math.pow(2, i) * 1000); // 指数退避 } } } // 指数退避 + jitter function delay(ms) { const jitter = Math.random() * 200; // 随机抖动，避免雪崩 return new Promise(r => setTimeout(r, ms + jitter)); } // 完整封装 async function request(url, { timeout = 10000, retries = 3, backoffBase = 1000, ...fetchOptions } = {}) { const controller = new AbortController(); const timer = setTimeout(() => controller.abort('timeout'), timeout); const mergedOptions = { ...fetchOptions, signal: controller.signal, }; for (let i = 0; i <= retries; i++) { try { const res = await fetch(url, mergedOptions); clearTimeout(timer); if (!res.ok) throw new Error(`HTTP ${res.status}`); return await res.json(); } catch (err) { clearTimeout(timer); if (i === retries || err.name === 'AbortError') throw err; const wait = backoffBase * Math.pow(2, i) + Math.random() * 500; await new Promise(r => setTimeout(r, wait)); } } }
