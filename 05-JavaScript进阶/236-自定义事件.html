自定义事件 CustomEvent

创建和触发自定义事件，实现组件间通信。

基本用法

// 1. 创建自定义事件 const event = new CustomEvent('my-event', { detail: { key: 'value' }, // 随事件传递的数据 bubbles: true, // 是否冒泡 cancelable: true, // 是否可取消 composed: true, // 是否能穿越 Shadow DOM 边界 }); // 2. 派发事件 element.dispatchEvent(event); // 3. 监听事件 element.addEventListener('my-event', (e) => { console.log(e.detail); // { key: 'value' } console.log(e.target); // 派发事件的元素 }); // 4. 检查事件是否被 preventDefault if (!event.defaultPrevented) { // 事件未被取消 } // 5. 全局事件总线模式 class EventBus { constructor() { this._events = {}; } on(event, handler) { if (!this._events[event]) this._events[event] = []; this._events[event].push(handler); } off(event, handler) { if (!this._events[event]) return; this._events[event] = this._events[event].filter(h => h !== handler); } emit(event, data) { (this._events[event] || []).forEach(h => h(data)); } once(event, handler) { const wrapper = (data) => { handler(data); this.off(event, wrapper); }; this.on(event, wrapper); } }
