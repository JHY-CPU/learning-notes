# 订阅服务端实现

## 一、WebSocket 配置

```typescript
// Apollo Server + graphql-ws
import { WebSocketServer } from 'ws';
import { useServer } from 'graphql-ws/lib/use/ws';
import { makeExecutableSchema } from '@graphql-tools/schema';
import { PubSub } from 'graphql-subscriptions';

const pubsub = new PubSub();

const typeDefs = `#graphql
  type Subscription {
    orderCreated: Order!
    orderStatusChanged(orderId: ID!): Order!
    newMessage(chatId: ID!): Message!
  }
`;

const resolvers = {
  Subscription: {
    orderCreated: {
      subscribe: () => pubsub.asyncIterator(['ORDER_CREATED']),
    },
    orderStatusChanged: {
      subscribe: (_, { orderId }) =>
        pubsub.asyncIterator([`ORDER_STATUS_${orderId}`]),
    },
  },
  Mutation: {
    createOrder: async (_, { input }) => {
      const order = await orderService.create(input);
      pubsub.publish('ORDER_CREATED', { orderCreated: order });
      return order;
    },
  },
};

const schema = makeExecutableSchema({ typeDefs, resolvers });

// WebSocket 服务器
const wsServer = new WebSocketServer({
  server: httpServer,
  path: '/graphql',
});

useServer({ schema }, wsServer);
```

## 二、Spring GraphQL 订阅

```java
@Controller
public class OrderSubscriptionController {

    private final FluxSink<OrderEvent> sink;
    private final Flux<OrderEvent> flux;

    public OrderSubscriptionController() {
        UnicastProcessor<OrderEvent> processor =
            UnicastProcessor.create();
        this.sink = processor.sink();
        this.flux = processor.replay(25).autoConnect();
    }

    @SubscriptionMapping
    public Flux<OrderEvent> orderCreated() {
        return flux.filter(e -> e.getType() == EventType.CREATED);
    }

    @SubscriptionMapping
    public Flux<OrderEvent> orderStatusChanged(
            @Argument String orderId) {
        return flux.filter(e -> e.getOrderId().equals(orderId));
    }

    // 发布事件
    public void publishEvent(OrderEvent event) {
        sink.next(event);
    }
}
```

## 三、Redis PubSub 跨实例

```typescript
import { RedisPubSub } from 'graphql-redis-subscriptions';
import Redis from 'ioredis';

const pubsub = new RedisPubSub({
  publisher: new Redis({ host: 'redis', port: 6379 }),
  subscriber: new Redis({ host: 'redis', port: 6379 }),
});

// 使用方式相同，自动同步到所有实例
pubsub.publish('ORDER_CREATED', { orderCreated: order });
```

## 四、注意事项

1. **生产用 Redis PubSub 替代内存 PubSub**
2. **WebSocket 连接要鉴权**
3. **订阅要过滤无用事件**
4. **处理客户端断线重连**
5. **监控连接数和消息量**
