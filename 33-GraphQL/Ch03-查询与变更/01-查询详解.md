# 查询详解 (Query)

## 一、基本查询

```graphql
# 简单字段查询
query {
  user(id: "1") {
    name
    email
  }
}

# 命名查询 (推荐)
query GetUser($id: ID!) {
  user(id: $id) {
    id
    name
    email
    createdAt
  }
}

# 多个根查询
query GetDashboard {
  totalUsers: users {
    totalCount
  }
  recentOrders(first: 5) {
    edges {
      node {
        id
        total
      }
    }
  }
  systemHealth {
    status
    uptime
  }
}
```

## 二、变量与默认值

```graphql
# 变量类型声明
query GetUsers(
  $limit: Int = 10          # 默认值
  $offset: Int = 0
  $status: UserStatus       # 可选变量
  $search: String!          # 必填变量
) {
  users(
    limit: $limit
    offset: $offset
    status: $status
    search: $search
  ) {
    totalCount
    items {
      id
      name
    }
  }
}

# 变量值
{
  "limit": 20,
  "offset": 0,
  "search": "张三"
}
```

## 三、内联片段

```graphql
# 类型条件片段
query {
  node(id: "1") {
    id
    ... on User {
      name
      email
    }
    ... on Product {
      name
      price
    }
  }
}

# 带条件的片段
query GetUser($detailed: Boolean!) {
  user(id: "1") {
    name
    ... @include(if: $detailed) {
      email
      phone
      address {
        city
      }
    }
  }
}
```

## 四、响应合并与别名

```graphql
# 别名实现同一字段多次查询
query CompareUsers {
  oldUser: user(id: "1") {
    name
    email
    orders(first: 1) {
      edges { node { total } }
    }
  }
  newUser: user(id: "2") {
    name
    email
    orders(first: 1) {
      edges { node { total } }
    }
  }
}

# 返回结构清晰分离
{
  "data": {
    "oldUser": { "name": "张三", "email": "..." },
    "newUser": { "name": "李四", "email": "..." }
  }
}
```

## 五、操作指令详解

```graphql
# @include - 条件包含（为 true 时包含）
query GetUser($id: ID!, $withOrders: Boolean!) {
  user(id: $id) {
    name
    orders @include(if: $withOrders) {
      id
      total
    }
  }
}

# @skip - 条件排除（为 true 时排除）
query GetUser($id: ID!, $skipEmail: Boolean!) {
  user(id: $id) {
    name
    email @skip(if: $skipEmail)
  }
}

# 指令组合使用
query GetUser(
  $id: ID!
  $includeOrders: Boolean!
  $skipSensitive: Boolean!
) {
  user(id: $id) {
    name
    email @skip(if: $skipSensitive)
    phone @skip(if: $skipSensitive)
    orders @include(if: $includeOrders) {
      id
      total
    }
  }
}
```

## 六、Apollo Client 查询

```javascript
import { useQuery, gql } from '@apollo/client';

const GET_USER = gql`
  query GetUser($id: ID!, $withOrders: Boolean!) {
    user(id: $id) {
      id
      name
      email
      orders @include(if: $withOrders) {
        edges {
          node {
            id
            total
            status
          }
        }
      }
    }
  }
`;

function UserProfile({ userId, showOrders }) {
  const { data, loading, error, refetch } = useQuery(GET_USER, {
    variables: { id: userId, withOrders: showOrders },
    fetchPolicy: 'cache-first',  // 缓存优先
    // 其他策略: cache-and-network, network-only, no-cache
  });

  if (loading) return <Skeleton />;
  if (error) return <ErrorBanner error={error} onRetry={refetch} />;

  return (
    <div>
      <h1>{data.user.name}</h1>
      <p>{data.user.email}</p>
      {data.user.orders && <OrderList orders={data.user.orders} />}
    </div>
  );
}
```

## 七、注意事项

1. **命名查询优于匿名查询**
2. **变量类型要完整声明**
3. **必填变量用 ! 标注**
4. **多个变量用逗号分隔**
5. **内联片段用于多态场景**
6. **@include/@skip 可减少客户端条件渲染**
7. **fetchPolicy 要根据业务场景选择**
