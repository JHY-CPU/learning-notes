# 变量与指令

## 一、变量类型

```graphql
# 基础类型变量
query($id: ID!, $name: String, $age: Int) { }

# 枚举变量
query($status: OrderStatus!) { }

# 输入类型变量
mutation($input: CreateUserInput!) { }

# 列表变量
query($ids: [ID!]!) { }       # 非空列表，元素非空
query($tags: [String]) { }    # 可空列表，元素可空

# Non-Null 变量
query($id: ID!) { }           # 必须提供
query($id: ID) { }            # 可以省略 (默认 null)
```

## 二、指令用法

```graphql
# @include 和 @skip
query GetUser(
  $id: ID!
  $withEmail: Boolean!
  $skipAvatar: Boolean!
) {
  user(id: $id) {
    name
    email @include(if: $withEmail)
    avatar @skip(if: $skipAvatar)
    phone @include(if: $withEmail)
  }
}

# 使用场景:
# 移动端不需要大字段
{ "id": "1", "withEmail": false, "skipAvatar": true }

# PC 端需要完整数据
{ "id": "1", "withEmail": true, "skipAvatar": false }
```

## 三、自定义指令应用

```graphql
# 缓存控制
type Query {
  products: [Product!]! @cacheControl(maxAge: 300)
  user(id: ID!): User! @cacheControl(maxAge: 60)
}

# 权限控制
type Query {
  allUsers: [User!]! @auth(requires: ADMIN)
  myProfile: User! @auth(requires: CUSTOMER)
}

# 速率限制
type Mutation {
  createPost(input: CreatePostInput!): Post!
    @rateLimit(window: "1h", max: 10)
  sendMessage(input: SendMessageInput!): Message!
    @rateLimit(window: "1m", max: 30)
}
```

## 四、变量类型详解

```graphql
# 标量变量
query($name: String, $age: Int, $active: Boolean) { }

# 非空标量
query($id: ID!, $email: String!) { }

# 枚举变量
query($status: OrderStatus!) { }

# 输入类型变量
mutation($input: CreateUserInput!) { }

# 列表变量
query($ids: [ID!]!) { }        # 非空列表，非空元素
query($tags: [String]) { }     # 可空列表，可空元素
query($ids: [ID]!) { }         # 非空列表，可空元素
query($tags: [String!]) { }    # 可空列表，非空元素

# 变量传递给嵌套参数
query($first: Int, $after: String, $withOrders: Boolean!) {
  users(first: $first, after: $after) {
    edges {
      node {
        name
        orders @include(if: $withOrders) {
          id
          total
        }
      }
    }
  }
}
```

## 五、自定义指令实现

```javascript
// 自定义 @auth 指令
import { mapSchema, getDirective, MapperKind } from '@graphql-tools/utils';

function authDirectiveTransformer(schema, directiveName) {
  return mapSchema(schema, {
    [MapperKind.OBJECT_FIELD]: (fieldConfig) => {
      const authDirective = getDirective(schema, fieldConfig, directiveName)?.[0];
      if (authDirective) {
        const { requires } = authDirective;
        const { resolve = defaultFieldResolver } = fieldConfig;

        fieldConfig.resolve = async (root, args, context, info) => {
          if (!context.currentUser) {
            throw new GraphQLError('未认证', {
              extensions: { code: 'UNAUTHENTICATED' },
            });
          }

          if (requires === 'ADMIN' && !context.currentUser.isAdmin) {
            throw new GraphQLError('需要管理员权限', {
              extensions: { code: 'FORBIDDEN' },
            });
          }

          return resolve(root, args, context, info);
        };
      }
      return fieldConfig;
    },
  });
}

// 使用
let schema = makeExecutableSchema({ typeDefs, resolvers });
schema = authDirectiveTransformer(schema, 'auth');
```

## 六、指令与 Apollo Client

```javascript
// 客户端变量使用
import { useQuery, gql } from '@apollo/client';

const SEARCH = gql`
  query Search($keyword: String!, $type: SearchType, $limit: Int = 10) {
    search(keyword: $keyword, type: $type, limit: $limit) {
      ... on User {
        name
        email
      }
      ... on Product {
        name
        price
      }
    }
  }
`;

function SearchResults({ keyword, searchType }) {
  const { data, loading } = useQuery(SEARCH, {
    variables: {
      keyword,
      type: searchType || undefined, // undefined 不传该变量
      // limit 使用 Schema 默认值 10
    },
  });

  // ...
}
```

## 七、注意事项

1. **变量用 $ 前缀声明**
2. **变量类型要与 Schema 定义匹配**
3. **指令是 Schema 的扩展机制**
4. **@include/@skip 只能用于字段/片段**
5. **自定义指令要在服务端注册实现**
6. **undefined 变量不传给服务端**，使用 Schema 默认值
7. **自定义指令适合横切关注点**（鉴权、缓存、限流）
