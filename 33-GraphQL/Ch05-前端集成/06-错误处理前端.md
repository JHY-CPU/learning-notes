# 错误处理 (前端)

## 一、错误类型

```typescript
// Apollo Client 错误类型
import { ApolloError } from '@apollo/client';

interface ApolloError {
  graphQLErrors: GraphQLError[];    // GraphQL 业务错误
  networkError: Error | null;       // 网络错误
  message: string;                  // 错误消息
}

interface GraphQLError {
  message: string;
  locations?: Array<{ line: number; column: number }>;
  path?: string[];
  extensions?: {
    code: string;
    [key: string]: any;
  };
}
```

## 二、组件级错误处理

```typescript
function UserProfile() {
  const { data, loading, error } = useQuery(GET_USER);

  if (loading) return <Spinner />;

  if (error) {
    // 区分错误类型
    if (error.networkError) {
      return <NetworkError retry={() => refetch()} />;
    }

    const graphQLError = error.graphQLErrors[0];
    switch (graphQLError?.extensions?.code) {
      case 'USER_NOT_FOUND':
        return <NotFound />;
      case 'FORBIDDEN':
        return <Forbidden />;
      default:
        return <GenericError message={error.message} />;
    }
  }

  return <UserProfileView user={data.user} />;
}
```

## 三、全局错误处理

```typescript
// Error Link - 全局拦截
import { onError } from '@apollo/client/link/error';

const errorLink = onError(({ graphQLErrors, networkError }) => {
  if (graphQLErrors) {
    graphQLErrors.forEach(({ message, extensions }) => {
      switch (extensions?.code) {
        case 'UNAUTHENTICATED':
          // Token 过期，跳转登录
          localStorage.removeItem('token');
          window.location.href = '/login';
          break;
        case 'FORBIDDEN':
          toast.error('权限不足');
          break;
        default:
          toast.error(message);
      }
    });
  }

  if (networkError) {
    toast.error('网络错误，请检查连接');
  }
});

// 注册到客户端
const client = new ApolloClient({
  link: errorLink.concat(httpLink),
});
```

## 四、Mutation 错误处理

```typescript
function CreateUserForm() {
  const [createUser, { error }] = useMutation(CREATE_USER, {
    onError: (error) => {
      // 特定 Mutation 的错误处理
      const code = error.graphQLErrors[0]?.extensions?.code;
      if (code === 'DUPLICATE_EMAIL') {
        form.setFieldError('email', '邮箱已被使用');
      }
    },
  });

  return (
    <Form
      onSubmit={(values) => createUser({ variables: { input: values } })}
      errors={error?.graphQLErrors}
    />
  );
}
```

## 五、注意事项

1. **区分网络错误和 GraphQL 错误**
2. **UNAUTHENTICATED 要自动跳转登录**
3. **错误信息要友好**
4. **不要在 console 暴露敏感信息**
5. **重试策略要合理**
