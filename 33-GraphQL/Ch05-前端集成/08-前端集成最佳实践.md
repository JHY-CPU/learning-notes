## 四、Apollo Cache 配置

```typescript
import { InMemoryCache, makeVar } from '@apollo/client';

// 响应式变量
export const isLoggedInVar = makeVar<boolean>(!!localStorage.getItem('token'));

const cache = new InMemoryCache({
  typePolicies: {
    Query: {
      fields: {
        // Connection 合并策略
        users: {
          keyArgs: ['filter'],
          merge(existing, incoming, { args }) {
            if (!args?.after) {
              return incoming;  // 首页，直接返回
            }
            return {
              ...incoming,
              edges: [...(existing?.edges || []), ...incoming.edges],
            };
          },
        },
        // 响应式字段
        isLoggedIn: {
          read() {
            return isLoggedInVar();
          },
        },
      },
    },
    User: {
      keyFields: ['id'],  // 缓存键
      fields: {
        orders: {
          merge(existing = { edges: [] }, incoming) {
            return { ...incoming, edges: [...existing.edges, ...incoming.edges] };
          },
        },
      },
    },
  },
});
```

## 五、自定义 Hook 封装

```typescript
// hooks/useUser.ts
import { useQuery, useMutation, gql } from '@apollo/client';

const GET_USER = gql`
  query GetUser($id: ID!) {
    user(id: $id) {
      ...UserDetail
    }
  }
  ${USER_DETAIL_FRAGMENT}
`;

const UPDATE_USER = gql`
  mutation UpdateUser($id: ID!, $input: UpdateUserInput!) {
    updateUser(id: $id, input: $input) {
      ...UserDetail
    }
  }
  ${USER_DETAIL_FRAGMENT}
`;

export function useUser(id: string) {
  const { data, loading, error, refetch } = useQuery(GET_USER, {
    variables: { id },
  });

  const [updateUser, updateState] = useMutation(UPDATE_USER);

  return {
    user: data?.user,
    loading,
    error,
    refetch,
    updateUser: (input: UpdateUserInput) =>
      updateUser({ variables: { id, input } }),
    updating: updateState.loading,
  };
}

// 使用
function UserPage({ userId }) {
  const { user, loading, updateUser, updating } = useUser(userId);

  if (loading) return <Skeleton />;

  return (
    <UserForm
      user={user}
      onSubmit={updateUser}
      loading={updating}
    />
  );
}
```

## 六、注意事项

1. **Codegen 是类型安全的基础**
2. **查询定义和组件分离**
3. **错误处理要全局 + 组件级结合**
4. **缓存设计要符合业务场景**
5. **持续优化查询性能**
6. **typePolicies 配置 Connection 合并策略**
7. **自定义 Hook 封装业务逻辑**，组件只关注渲染
