# 分页实现

## 一、Relay 分页 (前端)

```typescript
const GET_USERS = gql`
  query GetUsers($first: Int!, $after: String) {
    users(first: $first, after: $after) {
      edges {
        cursor
        node {
          id
          name
          email
        }
      }
      pageInfo {
        hasNextPage
        endCursor
      }
      totalCount
    }
  }
`;

function UserList() {
  const { data, loading, fetchMore } = useQuery(GET_USERS, {
    variables: { first: 10 },
  });

  const loadMore = () => {
    if (!data?.users.pageInfo.hasNextPage) return;

    fetchMore({
      variables: {
        after: data.users.pageInfo.endCursor,
      },
      // Apollo 自动合并
    });
  };

  return (
    <div>
      {data?.users.edges.map(({ node }) => (
        <UserCard key={node.id} user={node} />
      ))}
      {data?.users.pageInfo.hasNextPage && (
        <button onClick={loadMore} disabled={loading}>
          加载更多
        </button>
      )}
    </div>
  );
}
```

## 二、偏移分页

```typescript
const GET_USERS = gql`
  query GetUsers($limit: Int!, $offset: Int!) {
    users(limit: $limit, offset: $offset) {
      items {
        id
        name
      }
      totalCount
    }
  }
`;

function PaginatedList() {
  const [page, setPage] = useState(1);
  const limit = 10;

  const { data, loading } = useQuery(GET_USERS, {
    variables: { limit, offset: (page - 1) * limit },
  });

  const totalPages = Math.ceil(
    (data?.users.totalCount || 0) / limit
  );

  return (
    <div>
      <List items={data?.users.items} loading={loading} />
      <Pagination
        current={page}
        total={totalPages}
        onChange={setPage}
      />
    </div>
  );
}
```

## 三、无限滚动

```typescript
import { useInView } from 'react-intersection-observer';

function InfiniteUserList() {
  const { data, loading, fetchMore } = useQuery(GET_USERS, {
    variables: { first: 20 },
  });

  const { ref, inView } = useInView();

  useEffect(() => {
    if (inView && !loading && data?.users.pageInfo.hasNextPage) {
      fetchMore({
        variables: { after: data.users.pageInfo.endCursor },
      });
    }
  }, [inView, loading]);

  return (
    <div>
      {data?.users.edges.map(({ node }) => (
        <UserCard key={node.id} user={node} />
      ))}
      <div ref={ref}>{loading && <Spinner />}</div>
    </div>
  );
}
```

## 四、注意事项

1. **Relay Connection 是标准分页模式**
2. **fetchMore 自动合并数据**
3. **缓存策略影响分页体验**
4. **加载中状态要有视觉反馈**
5. **totalCount 可选**，计算成本高
