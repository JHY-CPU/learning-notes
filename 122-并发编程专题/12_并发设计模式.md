# 并发设计模式

## 生产者-消费者模式（Producer-Consumer）

生产者生成数据放入缓冲区，消费者从缓冲区取出数据处理，通过有界缓冲区解耦生产和消费速率。

```
Producer 1 ──┐              ┌── Consumer 1
Producer 2 ──┼─→ [Buffer] ──┼── Consumer 2
Producer 3 ──┘              └── Consumer 3
```

### Java 实现

```java
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.LinkedBlockingQueue;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class ProducerConsumer {
    private static final BlockingQueue<String> queue = new LinkedBlockingQueue<>(10);

    static class Producer implements Runnable {
        private final String name;

        Producer(String name) { this.name = name; }

        @Override
        public void run() {
            try {
                for (int i = 0; i < 20; i++) {
                    String item = name + "-item-" + i;
                    queue.put(item); // 队列满时阻塞
                    System.out.println("生产: " + item);
                    Thread.sleep(50);
                }
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        }
    }

    static class Consumer implements Runnable {
        private final String name;

        Consumer(String name) { this.name = name; }

        @Override
        public void run() {
            try {
                while (!Thread.currentThread().isInterrupted()) {
                    String item = queue.take(); // 队列空时阻塞
                    System.out.println(name + " 消费: " + item);
                    Thread.sleep(100);
                }
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        }
    }

    public static void main(String[] args) throws InterruptedException {
        ExecutorService pool = Executors.newFixedThreadPool(5);

        pool.execute(new Producer("P1"));
        pool.execute(new Producer("P2"));
        pool.execute(new Consumer("C1"));
        pool.execute(new Consumer("C2"));
        pool.execute(new Consumer("C3"));

        Thread.sleep(5000);
        pool.shutdownNow();
    }
}
```

## 读者-写者模式（Reader-Writer）

允许多个读者并发访问，但写者独占访问。适用于读多写少的场景。

### Go 实现

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

type SharedData struct {
    mu     sync.RWMutex
    value  int
    reader int
    writer int
}

func (sd *SharedData) Read(id int) int {
    sd.mu.RLock()
    defer sd.mu.RUnlock()
    sd.reader++
    time.Sleep(100 * time.Millisecond) // 模拟读取耗时
    fmt.Printf("读者 %d 读取值: %d\n", id, sd.value)
    return sd.value
}

func (sd *SharedData) Write(id int, val int) {
    sd.mu.Lock()
    defer sd.mu.Unlock()
    sd.writer++
    time.Sleep(200 * time.Millisecond) // 模拟写入耗时
    sd.value = val
    fmt.Printf("写者 %d 写入值: %d\n", id, val)
}

func main() {
    data := &SharedData{}
    var wg sync.WaitGroup

    // 启动 5 个读者
    for i := 0; i < 5; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            for j := 0; j < 3; j++ {
                data.Read(id)
            }
        }(i)
    }

    // 启动 2 个写者
    for i := 0; i < 2; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            for j := 0; j < 3; j++ {
                data.Write(id, (id+1)*100+j)
            }
        }(i)
    }

    wg.Wait()
    fmt.Printf("总读取次数: %d, 总写入次数: %d\n", data.reader, data.writer)
}
```

## 线程池模式（Thread Pool）

预先创建一组工作线程，从任务队列中取出任务执行，避免频繁创建销毁线程的开销。

### Python 实现

```python
import threading
import queue
import time
import uuid

class ThreadPool:
    def __init__(self, num_workers):
        self.task_queue = queue.Queue()
        self.workers = []
        self.shutdown_flag = threading.Event()

        for i in range(num_workers):
            t = threading.Thread(target=self._worker, name=f"worker-{i}", daemon=True)
            t.start()
            self.workers.append(t)

    def _worker(self):
        while not self.shutdown_flag.is_set():
            try:
                task_id, func, args, kwargs, result_event, result_holder = \
                    self.task_queue.get(timeout=0.1)
                try:
                    result = func(*args, **kwargs)
                    result_holder["result"] = result
                    result_holder["error"] = None
                except Exception as e:
                    result_holder["error"] = e
                finally:
                    result_event.set()
                    self.task_queue.task_done()
            except queue.Empty:
                continue

    def submit(self, func, *args, **kwargs):
        result_event = threading.Event()
        result_holder = {}
        task_id = str(uuid.uuid4())[:8]
        self.task_queue.put((task_id, func, args, kwargs, result_event, result_holder))

        def get_result(timeout=None):
            if result_event.wait(timeout):
                if result_holder.get("error"):
                    raise result_holder["error"]
                return result_holder["result"]
            raise TimeoutError("任务执行超时")

        return get_result

    def shutdown(self):
        self.shutdown_flag.set()
        for t in self.workers:
            t.join(timeout=1)


# 使用示例
def heavy_computation(n):
    time.sleep(0.5)
    return n * n

pool = ThreadPool(4)
futures = [pool.submit(heavy_computation, i) for i in range(10)]

for i, f in enumerate(futures):
    try:
        print(f"任务 {i} 结果: {f(timeout=5)}")
    except TimeoutError:
        print(f"任务 {i} 超时")

pool.shutdown()
```

## 主动对象模式（Active Object）

将方法调用与执行分离，方法调用在调用者线程，执行在专用线程，通过队列传递请求。

### C++ 实现

```cpp
#include <iostream>
#include <thread>
#include <queue>
#include <mutex>
#include <condition_variable>
#include <functional>
#include <future>

class ActiveObject {
    std::queue<std::function<void()>> queue;
    std::mutex mtx;
    std::condition_variable cv;
    bool running = true;
    std::thread worker;

public:
    ActiveObject() : worker([this]() { run(); }) {}

    ~ActiveObject() {
        {
            std::lock_guard<std::mutex> lock(mtx);
            running = false;
        }
        cv.notify_one();
        worker.join();
    }

    // 异步调用，返回 future
    template<typename F, typename... Args>
    auto invoke(F&& f, Args&&... args) -> std::future<decltype(f(args...))> {
        using ReturnType = decltype(f(args...));
        auto task = std::make_shared<std::packaged_task<ReturnType()>>(
            std::bind(std::forward<F>(f), std::forward<Args>(args)...)
        );
        auto future = task->get_future();

        {
            std::lock_guard<std::mutex> lock(mtx);
            queue.emplace([task]() { (*task)(); });
        }
        cv.notify_one();
        return future;
    }

private:
    void run() {
        while (true) {
            std::function<void()> task;
            {
                std::unique_lock<std::mutex> lock(mtx);
                cv.wait(lock, [this]() { return !queue.empty() || !running; });
                if (!running && queue.empty()) return;
                task = std::move(queue.front());
                queue.pop();
            }
            task();
        }
    }
};

int main() {
    ActiveObject obj;

    // 异步调用
    auto f1 = obj.invoke([](int a, int b) {
        std::this_thread::sleep_for(std::chrono::milliseconds(100));
        return a + b;
    }, 10, 20);

    auto f2 = obj.invoke([](const std::string& s) {
        return "Hello " + s;
    }, std::string("Active Object"));

    std::cout << "10 + 20 = " << f1.get() << std::endl;
    std::cout << f2.get() << std::endl;

    return 0;
}
```

## CSP 模型（Communicating Sequential Processes）

CSP 强调通过通信共享数据而非通过共享数据通信。Go 的 goroutine + channel 是 CSP 的典型实现。

```go
package main

import (
    "fmt"
    "math/rand"
    "time"
)

// CSP 风格的订单处理系统
type Order struct {
    ID     int
    Item   string
    Amount float64
}

type ProcessedOrder struct {
    Order
    Status string
}

func orderGenerator(out chan<- Order) {
    for i := 1; i <= 10; i++ {
        out <- Order{
            ID:     i,
            Item:   fmt.Sprintf("商品-%d", i),
            Amount: float64(rand.Intn(100)) + 0.99,
        }
    }
    close(out)
}

func orderValidator(in <-chan Order, out chan<- Order) {
    for order := range in {
        if order.Amount > 0 {
            out <- order
        }
    }
    close(out)
}

func orderProcessor(in <-chan Order, out chan<- ProcessedOrder) {
    for order := range in {
        time.Sleep(time.Duration(rand.Intn(50)) * time.Millisecond)
        out <- ProcessedOrder{
            Order:  order,
            Status: "已处理",
        }
    }
    close(out)
}

func main() {
    rand.Seed(time.Now().UnixNano())

    orders := make(chan Order, 5)
    validated := make(chan Order, 5)
    processed := make(chan ProcessedOrder, 5)

    go orderGenerator(orders)
    go orderValidator(orders, validated)
    go orderProcessor(validated, processed)

    for p := range processed {
        fmt.Printf("订单 %d: %s, 金额=%.2f, 状态=%s\n",
            p.ID, p.Item, p.Amount, p.Status)
    }
}
```

## Actor 模型

每个 Actor 是一个独立的计算单元，拥有自己的状态和邮箱，通过异步消息通信。Actor 之间不共享状态。

### Go 实现

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

// 消息类型
type Message struct {
    Type    string
    Payload interface{}
    ReplyTo chan interface{}
}

// Actor 基础结构
type CounterActor struct {
    inbox  chan Message
    count  int
    wg     *sync.WaitGroup
}

func NewCounterActor(wg *sync.WaitGroup) *CounterActor {
    actor := &CounterActor{
        inbox: make(chan Message, 100),
        wg:    wg,
    }
    go actor.run()
    return actor
}

func (a *CounterActor) run() {
    for msg := range a.inbox {
        switch msg.Type {
        case "INCREMENT":
            a.count++
        case "DECREMENT":
            a.count--
        case "GET":
            msg.ReplyTo <- a.count
        case "STOP":
            close(msg.ReplyTo)
            return
        }
    }
}

func (a *CounterActor) Send(msg Message) {
    a.inbox <- msg
}

func (a *CounterActor) Ask(msg Message) interface{} {
    msg.ReplyTo = make(chan interface{}, 1)
    a.inbox <- msg
    return <-msg.ReplyTo
}

func (a *CounterActor) Stop() {
    reply := make(chan interface{}, 1)
    a.inbox <- Message{Type: "STOP", ReplyTo: reply}
}

func main() {
    var wg sync.WaitGroup
    actor := NewCounterActor(&wg)

    // 并发发送消息
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            actor.Send(Message{Type: "INCREMENT"})
        }()
    }

    wg.Wait()
    time.Sleep(100 * time.Millisecond)

    // 查询结果
    result := actor.Ask(Message{Type: "GET"})
    fmt.Printf("Actor 计数: %d\n", result) // 1000

    actor.Stop()
}
```

## 模式对比

| 模式 | 通信方式 | 状态管理 | 典型场景 |
|------|---------|---------|---------|
| 生产者-消费者 | 队列 | 共享缓冲区 | 任务处理、消息队列 |
| 读者-写者 | 读写锁 | 共享数据 | 缓存、配置 |
| 线程池 | 任务队列 | 共享队列 | Web 服务器、批处理 |
| 主动对象 | 请求队列 | 对象内部 | GUI、资源管理 |
| CSP | Channel | 无共享 | Go 并发、数据流 |
| Actor | 异步消息 | Actor 内部 | 分布式系统、并发状态 |
