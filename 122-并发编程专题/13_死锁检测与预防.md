# 死锁检测与预防

## 锁排序（Lock Ordering）

通过全局一致的锁获取顺序，消除循环等待条件，从根本上预防死锁。

### Java 实现

```java
import java.util.concurrent.locks.ReentrantLock;

public class LockOrderingDemo {
    private static final ReentrantLock lockA = new ReentrantLock();
    private static final ReentrantLock lockB = new ReentrantLock();
    private int accountA = 1000;
    private int accountB = 1000;

    // 错误：可能导致死锁
    public void unsafeTransfer() {
        lockA.lock();
        try {
            lockB.lock(); // 如果另一个线程先获取 lockB，再获取 lockA，就会死锁
            try {
                accountA -= 100;
                accountB += 100;
            } finally {
                lockB.unlock();
            }
        } finally {
            lockA.unlock();
        }
    }

    // 正确：按固定顺序获取锁
    public void safeTransfer() {
        // 始终先获取 ID 较小的锁
        ReentrantLock first = lockA;
        ReentrantLock second = lockB;
        // 如果需要交换顺序，通过 System.identityHashCode 比较
        if (System.identityHashCode(lockA) > System.identityHashCode(lockB)) {
            first = lockB;
            second = lockA;
        }

        first.lock();
        try {
            second.lock();
            try {
                accountA -= 100;
                accountB += 100;
            } finally {
                second.unlock();
            }
        } finally {
            first.unlock();
        }
    }
}
```

### C++ 实现

```cpp
#include <iostream>
#include <mutex>
#include <thread>
#include <algorithm>

class Account {
public:
    int id;
    int balance;
    std::mutex mtx;

    Account(int id, int bal) : id(id), balance(bal) {}
};

void safeTransfer(Account& from, Account& to, int amount) {
    // 按 ID 排序锁获取顺序
    Account* first = &from;
    Account* second = &to;
    if (first->id > second->id) {
        std::swap(first, second);
    }

    std::lock_guard<std::mutex> lock1(first->mtx);
    std::lock_guard<std::mutex> lock2(second->mtx);

    if (from.balance >= amount) {
        from.balance -= amount;
        to.balance += amount;
        std::cout << "转账 " << amount << ": 账户"
                  << from.id << " -> 账户" << to.id << std::endl;
    }
}

int main() {
    Account a1(1, 1000);
    Account a2(2, 1000);

    std::thread t1([&]() {
        for (int i = 0; i < 100; i++) {
            safeTransfer(a1, a2, 10);
        }
    });

    std::thread t2([&]() {
        for (int i = 0; i < 100; i++) {
            safeTransfer(a2, a1, 10);
        }
    });

    t1.join();
    t2.join();

    std::cout << "账户1: " << a1.balance
              << ", 账户2: " << a2.balance << std::endl;
    return 0;
}
```

## 超时检测（Timeout-Based Detection）

通过设置获取锁的超时时间，避免无限等待导致的死锁。

### Go 实现

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

type SafeTransfer struct {
    mu1, mu2 sync.Mutex
}

func (st *SafeTransfer) transfer(from, to int, amount int) error {
    // 尝试获取第一个锁
    if !st.tryLock(&st.mu1, 100*time.Millisecond) {
        return fmt.Errorf("获取锁1超时")
    }
    defer st.mu1.Unlock()

    // 尝试获取第二个锁
    if !st.tryLock(&st.mu2, 100*time.Millisecond) {
        return fmt.Errorf("获取锁2超时")
    }
    defer st.mu2.Unlock()

    fmt.Printf("从账户%d转账%d到账户%d\n", from, amount, to)
    return nil
}

func (st *SafeTransfer) tryLock(mu *sync.Mutex, timeout time.Duration) bool {
    deadline := time.After(timeout)
    for {
        select {
        case <-deadline:
            return false
        default:
            locked := make(chan struct{})
            go func() {
                mu.Lock()
                close(locked)
            }()

            select {
            case <-locked:
                return true
            case <-deadline:
                return false
            }
        }
    }
}

// 更简单的方式：使用 time.After 配合 select
func transferWithTimeout(account1, account2 *sync.Mutex) error {
    ch := make(chan struct{}, 1)

    go func() {
        account1.Lock()
        defer account1.Unlock()
        account2.Lock()
        defer account2.Unlock()
        ch <- struct{}{}
    }()

    select {
    case <-ch:
        return nil
    case <-time.After(500 * time.Millisecond):
        return fmt.Errorf("转账超时，可能发生死锁")
    }
}

func main() {
    st := &SafeTransfer{}

    var wg sync.WaitGroup
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func(n int) {
            defer wg.Done()
            if err := st.transfer(1, 2, 100); err != nil {
                fmt.Printf("转账失败: %v\n", err)
            }
        }(i)
    }
    wg.Wait()
}
```

### Java tryLock

```java
import java.util.concurrent.locks.ReentrantLock;
import java.util.concurrent.TimeUnit;
import java.util.Random;

public class TryLockDemo {
    private final ReentrantLock lock1 = new ReentrantLock();
    private final ReentrantLock lock2 = new ReentrantLock();
    private final Random random = new Random();

    public boolean safeOperation() throws InterruptedException {
        while (true) {
            if (lock1.tryLock(50, TimeUnit.MILLISECONDS)) {
                try {
                    if (lock2.tryLock(50, TimeUnit.MILLISECONDS)) {
                        try {
                            System.out.println("获取到两个锁");
                            return true;
                        } finally {
                            lock2.unlock();
                        }
                    }
                } finally {
                    lock1.unlock();
                }
            }

            // 获取锁失败，随机退避后重试
            Thread.sleep(random.nextInt(10));
        }
    }
}
```

## 银行家算法（Banker's Algorithm）

银行家算法是一种死锁避免算法，模拟资源分配来判断系统是否处于安全状态。

### Python 实现

```python
import numpy as np
from typing import List, Tuple

class BankerAlgorithm:
    """
    银行家算法：在分配资源前检查系统是否安全
    - available: 可用资源向量
    - max: 每个进程的最大需求矩阵
    - allocation: 每个进程已分配的资源矩阵
    - need: 每个进程还需要的资源矩阵 (need = max - allocation)
    """

    def __init__(self, available: List[int], max_demand: List[List[int]],
                 allocation: List[List[int]]):
        self.available = np.array(available, dtype=int)
        self.max_demand = np.array(max_demand, dtype=int)
        self.allocation = np.array(allocation, dtype=int)
        self.need = self.max_demand - self.allocation
        self.num_processes = len(allocation)
        self.num_resources = len(available)

    def is_safe(self) -> Tuple[bool, List[int]]:
        """检查系统是否处于安全状态，返回安全序列"""
        work = self.available.copy()
        finish = [False] * self.num_processes
        safe_sequence = []

        while len(safe_sequence) < self.num_processes:
            found = False
            for i in range(self.num_processes):
                if not finish[i] and all(self.need[i] <= work):
                    # 进程 i 可以完成
                    work += self.allocation[i]
                    finish[i] = True
                    safe_sequence.append(i)
                    found = True
                    break
            if not found:
                return False, []  # 不安全

        return True, safe_sequence

    def request_resources(self, process_id: int, request: List[int]) -> bool:
        """请求资源分配"""
        request = np.array(request, dtype=int)

        # 检查请求是否超过需求
        if any(request > self.need[process_id]):
            print(f"进程 {process_id} 请求超过其最大需求")
            return False

        # 检查请求是否超过可用资源
        if any(request > self.available):
            print(f"资源不足，进程 {process_id} 需要等待")
            return False

        # 尝试分配
        self.available -= request
        self.allocation[process_id] += request
        self.need[process_id] -= request

        # 检查安全性
        safe, sequence = self.is_safe()
        if safe:
            print(f"分配成功，安全序列: {sequence}")
            return True
        else:
            # 回滚
            self.available += request
            self.allocation[process_id] -= request
            self.need[process_id] += request
            print(f"分配会导致不安全状态，拒绝请求")
            return False


# 使用示例
if __name__ == "__main__":
    # 3 种资源类型，5 个进程
    available = [3, 3, 2]
    max_demand = [
        [7, 5, 3],
        [3, 2, 2],
        [9, 0, 2],
        [2, 2, 2],
        [4, 3, 3],
    ]
    allocation = [
        [0, 1, 0],
        [2, 0, 0],
        [3, 0, 2],
        [2, 1, 1],
        [0, 0, 2],
    ]

    banker = BankerAlgorithm(available, max_demand, allocation)
    safe, sequence = banker.is_safe()
    print(f"系统安全: {safe}, 安全序列: {sequence}")

    # 进程 1 请求资源
    banker.request_resources(1, [1, 0, 2])
```

## 锁层次（Lock Hierarchy）

为每个锁分配一个层级编号，要求线程只能获取比当前持有锁层级更低的锁。

```java
import java.util.concurrent.ConcurrentHashMap;

public class LockHierarchy {
    // 为每个锁分配层级
    private static final ConcurrentHashMap<Object, Integer> lockLevels = new ConcurrentHashMap<>();

    private final Object highLevelLock = new Object();
    private final Object midLevelLock = new Object();
    private final Object lowLevelLock = new Object();

    private final ThreadLocal<Integer> currentLevel = ThreadLocal.withInitial(() -> Integer.MAX_VALUE);

    public LockHierarchy() {
        lockLevels.put(highLevelLock, 300);
        lockLevels.put(midLevelLock, 200);
        lockLevels.put(lowLevelLock, 100);
    }

    private void acquireWithHierarchy(Object lock) {
        int lockLevel = lockLevels.get(lock);
        int threadLevel = currentLevel.get();

        if (lockLevel >= threadLevel) {
            throw new IllegalStateException(
                "违反锁层次: 尝试获取层级 " + lockLevel +
                " 的锁，当前线程持有层级 " + threadLevel + " 的锁");
        }

        synchronized (lock) {
            currentLevel.set(lockLevel);
            try {
                // 执行临界区操作
            } finally {
                currentLevel.set(threadLevel);
            }
        }
    }
}
```

## 无锁替代方案

用无锁数据结构和原子操作完全避免使用锁。

### 使用 Atomic 实现无锁累加器

```rust
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;
use std::thread;

struct LockFreeAccumulator {
    sum: AtomicU64,
    count: AtomicU64,
}

impl LockFreeAccumulator {
    fn new() -> Self {
        LockFreeAccumulator {
            sum: AtomicU64::new(0),
            count: AtomicU64::new(0),
        }
    }

    fn add(&self, value: u64) {
        self.sum.fetch_add(value, Ordering::Relaxed);
        self.count.fetch_add(1, Ordering::Relaxed);
    }

    fn average(&self) -> f64 {
        let sum = self.sum.load(Ordering::Acquire);
        let count = self.count.load(Ordering::Acquire);
        if count == 0 { 0.0 } else { sum as f64 / count as f64 }
    }
}

fn main() {
    let acc = Arc::new(LockFreeAccumulator::new());
    let mut handles = vec![];

    for i in 0..10 {
        let acc = Arc::clone(&acc);
        handles.push(thread::spawn(move || {
            for j in 0..1000 {
                acc.add((i * 1000 + j) as u64);
            }
        }));
    }

    for h in handles {
        h.join().unwrap();
    }

    println!("平均值: {:.2}", acc.average()); // 4999.50
}
```

## 总结

| 策略 | 方法 | 打破的条件 |
|------|------|-----------|
| 锁排序 | 固定获取顺序 | 循环等待 |
| 超时检测 | tryLock + timeout | 持有并等待 |
| 银行家算法 | 预分配检查 | 避免不安全状态 |
| 锁层次 | 层级约束 | 循环等待 |
| 无锁方案 | CAS/原子操作 | 互斥条件 |
