# C++ 并发编程

## std::thread

C++11 引入了标准线程库，提供了跨平台的线程支持。

### 创建线程

```cpp
#include <iostream>
#include <thread>
#include <vector>
#include <string>

void worker(int id) {
    std::cout << "线程 " << id << " 执行, ID: "
              << std::this_thread::get_id() << std::endl;
}

class Task {
public:
    void operator()() const {
        std::cout << "函数对象线程" << std::endl;
    }
};

int main() {
    // 1. 函数作为线程入口
    std::thread t1(worker, 1);

    // 2. Lambda 表达式
    std::thread t2([]() {
        std::cout << "Lambda 线程" << std::endl;
    });

    // 3. 函数对象
    Task task;
    std::thread t3(task);

    // 4. 成员函数
    struct Worker {
        void doWork(int n) {
            std::cout << "成员函数: " << n << std::endl;
        }
    };
    Worker w;
    std::thread t4(&Worker::doWork, &w, 42);

    t1.join();
    t2.join();
    t3.join();
    t4.join();

    // 硬件并发数
    std::cout << "硬件支持 " << std::thread::hardware_concurrency()
              << " 个并发线程" << std::endl;

    return 0;
}
```

### 线程参数传递

```cpp
#include <iostream>
#include <thread>
#include <string>

void printString(const std::string& s) {
    std::cout << s << std::endl;
}

void modifyString(std::string& s) {
    s += " (已修改)";
}

int main() {
    std::string msg = "Hello";

    // 值传递（复制）
    std::thread t1(printString, msg);

    // 引用传递：必须使用 std::ref
    std::thread t2(modifyString, std::ref(msg));

    // 移动传递
    std::string data = "独占数据";
    std::thread t3(printString, std::move(data));

    t1.join();
    t2.join();
    t3.join();

    std::cout << "msg: " << msg << std::endl; // Hello (已修改)

    return 0;
}
```

## mutex / lock_guard / unique_lock

### mutex 基本用法

```cpp
#include <iostream>
#include <thread>
#include <mutex>
#include <vector>

int counter = 0;
std::mutex mtx;

void increment(int times) {
    for (int i = 0; i < times; i++) {
        mtx.lock();
        counter++;
        mtx.unlock();
    }
}

int main() {
    std::vector<std::thread> threads;
    for (int i = 0; i < 10; i++) {
        threads.emplace_back(increment, 1000);
    }
    for (auto& t : threads) {
        t.join();
    }
    std::cout << "最终计数: " << counter << std::endl; // 10000
    return 0;
}
```

### lock_guard（RAII 锁）

```cpp
#include <iostream>
#include <thread>
#include <mutex>
#include <vector>
#include <map>

class BankAccount {
    std::mutex mtx;
    double balance;

public:
    BankAccount(double b) : balance(b) {}

    void deposit(double amount) {
        std::lock_guard<std::mutex> lock(mtx); // 构造时加锁，析构时自动解锁
        balance += amount;
    }

    bool withdraw(double amount) {
        std::lock_guard<std::mutex> lock(mtx);
        if (balance >= amount) {
            balance -= amount;
            return true;
        }
        return false;
    }

    double getBalance() {
        std::lock_guard<std::mutex> lock(mtx);
        return balance;
    }
};

int main() {
    BankAccount account(1000.0);
    std::vector<std::thread> threads;

    for (int i = 0; i < 100; i++) {
        threads.emplace_back([&account]() { account.deposit(10.0); });
        threads.emplace_back([&account]() { account.withdraw(5.0); });
    }

    for (auto& t : threads) t.join();
    std::cout << "余额: " << account.getBalance() << std::endl;
    return 0;
}
```

### unique_lock（灵活锁）

```cpp
#include <iostream>
#include <thread>
#include <mutex>
#include <deque>
#include <condition_variable>

template<typename T>
class BoundedQueue {
    std::deque<T> queue;
    size_t capacity;
    std::mutex mtx;
    std::condition_variable not_full;
    std::condition_variable not_empty;

public:
    BoundedQueue(size_t cap) : capacity(cap) {}

    void put(const T& item) {
        std::unique_lock<std::mutex> lock(mtx);
        not_full.wait(lock, [this]() { return queue.size() < capacity; });
        queue.push_back(item);
        not_empty.notify_one();
    }

    T get() {
        std::unique_lock<std::mutex> lock(mtx);
        not_empty.wait(lock, [this]() { return !queue.empty(); });
        T item = queue.front();
        queue.pop_front();
        not_full.notify_one();
        return item;
    }
};

int main() {
    BoundedQueue<int> q(5);

    std::thread producer([&q]() {
        for (int i = 0; i < 10; i++) {
            q.put(i);
            std::cout << "生产: " << i << std::endl;
        }
    });

    std::thread consumer([&q]() {
        for (int i = 0; i < 10; i++) {
            int val = q.get();
            std::cout << "消费: " << val << std::endl;
        }
    });

    producer.join();
    consumer.join();
    return 0;
}
```

### std::scoped_lock（避免死锁）

```cpp
#include <iostream>
#include <thread>
#include <mutex>

std::mutex mtx1, mtx2;

void safeOperation() {
    // scoped_lock 同时获取多个锁，避免死锁
    std::scoped_lock lock(mtx1, mtx2);
    std::cout << "安全获取多个锁" << std::endl;
}
```

## condition_variable

```cpp
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>

std::mutex mtx;
std::condition_variable cv;
std::queue<int> data_queue;
bool finished = false;

void producer() {
    for (int i = 0; i < 10; i++) {
        {
            std::lock_guard<std::mutex> lock(mtx);
            data_queue.push(i);
            std::cout << "生产: " << i << std::endl;
        }
        cv.notify_one(); // 通知一个等待的消费者
        std::this_thread::sleep_for(std::chrono::milliseconds(100));
    }
    {
        std::lock_guard<std::mutex> lock(mtx);
        finished = true;
    }
    cv.notify_all(); // 通知所有消费者
}

void consumer(int id) {
    while (true) {
        std::unique_lock<std::mutex> lock(mtx);
        cv.wait(lock, []() { return !data_queue.empty() || finished; });

        while (!data_queue.empty()) {
            int val = data_queue.front();
            data_queue.pop();
            std::cout << "消费者 " << id << " 处理: " << val << std::endl;
        }

        if (finished) break;
    }
}

int main() {
    std::thread prod(producer);
    std::thread c1(consumer, 1);
    std::thread c2(consumer, 2);

    prod.join();
    c1.join();
    c2.join();
    return 0;
}
```

## future / promise / packaged_task

### std::async

```cpp
#include <iostream>
#include <future>
#include <vector>
#include <numeric>

long long computeSum(int start, int end) {
    long long sum = 0;
    for (int i = start; i <= end; i++) {
        sum += i;
    }
    return sum;
}

int main() {
    // 异步执行
    auto future1 = std::async(std::launch::async, computeSum, 1, 500000);
    auto future2 = std::async(std::launch::async, computeSum, 500001, 1000000);

    // 获取结果（阻塞）
    long long result = future1.get() + future2.get();
    std::cout << "总和: " << result << std::endl;

    return 0;
}
```

### std::promise / std::future

```cpp
#include <iostream>
#include <thread>
#include <future>
#include <string>

void computeValue(std::promise<int> promise) {
    // 模拟计算
    std::this_thread::sleep_for(std::chrono::milliseconds(100));
    promise.set_value(42); // 设置结果
}

void produceError(std::promise<void> promise) {
    try {
        throw std::runtime_error("计算出错");
    } catch (...) {
        promise.set_exception(std::current_exception());
    }
}

int main() {
    // 基本用法
    std::promise<int> p1;
    std::future<int> f1 = p1.get_future();
    std::thread t1(computeValue, std::move(p1));
    std::cout << "结果: " << f1.get() << std::endl;
    t1.join();

    // 异常传递
    std::promise<void> p2;
    std::future<void> f2 = p2.get_future();
    std::thread t2(produceError, std::move(p2));
    try {
        f2.get();
    } catch (const std::exception& e) {
        std::cout << "捕获异常: " << e.what() << std::endl;
    }
    t2.join();

    return 0;
}
```

### std::packaged_task

```cpp
#include <iostream>
#include <future>
#include <thread>

int multiply(int a, int b) {
    return a * b;
}

int main() {
    // packaged_task 将可调用对象包装成异步任务
    std::packaged_task<int(int, int)> task(multiply);
    std::future<int> future = task.get_future();

    std::thread t(std::move(task), 6, 7);
    std::cout << "6 * 7 = " << future.get() << std::endl;

    t.join();
    return 0;
}
```

## atomic

```cpp
#include <iostream>
#include <thread>
#include <atomic>
#include <vector>

std::atomic<int> atomic_counter(0);
std::atomic<bool> ready(false);

void atomicIncrement(int times) {
    for (int i = 0; i < times; i++) {
        atomic_counter.fetch_add(1, std::memory_order_relaxed);
    }
}

void waitUntilReady() {
    while (!ready.load(std::memory_order_acquire)) {
        std::this_thread::yield();
    }
    std::cout << "就绪!" << std::endl;
}

int main() {
    std::vector<std::thread> threads;
    for (int i = 0; i < 10; i++) {
        threads.emplace_back(atomicIncrement, 1000);
    }
    for (auto& t : threads) t.join();
    std::cout << "原子计数: " << atomic_counter.load() << std::endl;

    // flag 示例
    std::thread waiter(waitUntilReady);
    std::this_thread::sleep_for(std::chrono::milliseconds(100));
    ready.store(true, std::memory_order_release);
    waiter.join();

    // compare_exchange
    std::atomic<int> value(10);
    int expected = 10;
    bool success = value.compare_exchange_strong(expected, 20);
    std::cout << "CAS: " << (success ? "成功" : "失败")
              << ", 值: " << value.load() << std::endl;

    return 0;
}
```

## 线程池实现

```cpp
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>
#include <vector>
#include <functional>
#include <future>

class ThreadPool {
    std::vector<std::thread> workers;
    std::queue<std::function<void()>> tasks;
    std::mutex mtx;
    std::condition_variable cv;
    bool stop = false;

public:
    ThreadPool(size_t numThreads) {
        for (size_t i = 0; i < numThreads; i++) {
            workers.emplace_back([this]() {
                while (true) {
                    std::function<void()> task;
                    {
                        std::unique_lock<std::mutex> lock(mtx);
                        cv.wait(lock, [this]() { return stop || !tasks.empty(); });
                        if (stop && tasks.empty()) return;
                        task = std::move(tasks.front());
                        tasks.pop();
                    }
                    task();
                }
            });
        }
    }

    template<typename F, typename... Args>
    auto submit(F&& f, Args&&... args) -> std::future<decltype(f(args...))> {
        using ReturnType = decltype(f(args...));
        auto task = std::make_shared<std::packaged_task<ReturnType()>>(
            std::bind(std::forward<F>(f), std::forward<Args>(args)...)
        );
        std::future<ReturnType> future = task->get_future();
        {
            std::lock_guard<std::mutex> lock(mtx);
            tasks.emplace([task]() { (*task)(); });
        }
        cv.notify_one();
        return future;
    }

    ~ThreadPool() {
        {
            std::lock_guard<std::mutex> lock(mtx);
            stop = true;
        }
        cv.notify_all();
        for (auto& w : workers) {
            w.join();
        }
    }
};

int main() {
    ThreadPool pool(4);
    std::vector<std::future<int>> futures;

    for (int i = 0; i < 8; i++) {
        futures.push_back(pool.submit([i]() {
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
            return i * i;
        }));
    }

    for (auto& f : futures) {
        std::cout << "结果: " << f.get() << std::endl;
    }

    return 0;
}
```
