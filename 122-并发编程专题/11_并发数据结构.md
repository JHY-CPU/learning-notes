# 并发数据结构

## 无锁队列（Lock-Free Queue）

无锁队列使用 CAS（Compare-And-Swap）操作实现线程安全，不使用互斥锁，避免了锁竞争和死锁问题。

### 基于链表的无锁队列

```rust
// Rust 实现的无锁队列思路（使用 crossbeam 库）
use std::sync::Arc;
use std::sync::atomic::{AtomicPtr, Ordering};
use std::ptr;

struct Node<T> {
    data: Option<T>,
    next: AtomicPtr<Node<T>>,
}

pub struct LockFreeQueue<T> {
    head: AtomicPtr<Node<T>>,
    tail: AtomicPtr<Node<T>>,
}

impl<T> LockFreeQueue<T> {
    pub fn new() -> Self {
        let sentinel = Box::into_raw(Box::new(Node {
            data: None,
            next: AtomicPtr::new(ptr::null_mut()),
        }));
        LockFreeQueue {
            head: AtomicPtr::new(sentinel),
            tail: AtomicPtr::new(sentinel),
        }
    }

    pub fn push(&self, value: T) {
        let new_node = Box::into_raw(Box::new(Node {
            data: Some(value),
            next: AtomicPtr::new(ptr::null_mut()),
        }));

        loop {
            let tail = self.tail.load(Ordering::Acquire);
            let next = unsafe { (*tail).next.load(Ordering::Acquire) };

            if next.is_null() {
                // 尝试将新节点链接到尾部
                if unsafe { (*tail).next.compare_exchange(
                    ptr::null_mut(), new_node,
                    Ordering::Release, Ordering::Relaxed
                ).is_ok() } {
                    // 推进 tail 指针
                    let _ = self.tail.compare_exchange(
                        tail, new_node,
                        Ordering::Release, Ordering::Relaxed
                    );
                    return;
                }
            } else {
                // 帮助其他线程推进 tail
                let _ = self.tail.compare_exchange(
                    tail, next,
                    Ordering::Release, Ordering::Relaxed
                );
            }
        }
    }
}
```

### Java 无锁队列

```java
import java.util.concurrent.ConcurrentLinkedQueue;
import java.util.concurrent.atomic.AtomicInteger;

public class LockFreeQueueDemo {
    public static void main(String[] args) {
        ConcurrentLinkedQueue<String> queue = new ConcurrentLinkedQueue<>();

        // 多线程并发操作
        AtomicInteger counter = new AtomicInteger(0);
        Thread[] producers = new Thread[5];

        for (int i = 0; i < 5; i++) {
            producers[i] = new Thread(() -> {
                for (int j = 0; j < 100; j++) {
                    queue.offer("item-" + counter.incrementAndGet());
                }
            });
            producers[i].start();
        }

        // 消费
        Thread consumer = new Thread(() -> {
            int consumed = 0;
            while (consumed < 500) {
                String item = queue.poll();
                if (item != null) {
                    consumed++;
                }
            }
            System.out.println("消费完成: " + consumed);
        });
        consumer.start();

        try {
            for (Thread t : producers) t.join();
            consumer.join();
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }

        System.out.println("队列剩余: " + queue.size());
    }
}
```

## 并发哈希表

### Java ConcurrentHashMap

```java
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.CountDownLatch;

public class ConcurrentHashMapExample {
    public static void main(String[] args) throws InterruptedException {
        ConcurrentHashMap<String, Integer> map = new ConcurrentHashMap<>();
        ExecutorService pool = Executors.newFixedThreadPool(10);
        CountDownLatch latch = new CountDownLatch(10);

        // 并发写入
        for (int i = 0; i < 10; i++) {
            final int taskId = i;
            pool.submit(() -> {
                for (int j = 0; j < 1000; j++) {
                    String key = "key-" + (taskId * 1000 + j);
                    map.put(key, taskId * 1000 + j);
                }
                latch.countDown();
            });
        }

        latch.await();
        System.out.println("总条目数: " + map.size()); // 10000

        // 原子操作
        map.compute("key-0", (k, v) -> (v == null) ? 1 : v + 100);
        map.merge("key-1", 50, Integer::sum);

        pool.shutdown();
    }
}
```

### Go sync.Map

```go
package main

import (
    "fmt"
    "sync"
)

func main() {
    var m sync.Map

    // 并发写入
    var wg sync.WaitGroup
    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func(n int) {
            defer wg.Done()
            m.Store(fmt.Sprintf("key-%d", n), n*n)
        }(i)
    }
    wg.Wait()

    // 并发读取
    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func(n int) {
            defer wg.Done()
            if val, ok := m.Load(fmt.Sprintf("key-%d", n)); ok {
                _ = val
            }
        }(i)
    }
    wg.Wait()

    // LoadOrStore: 原子性的获取或存储
    actual, loaded := m.LoadOrStore("new-key", "new-value")
    fmt.Printf("实际值: %v, 已存在: %v\n", actual, loaded)

    // Range: 遍历（快照）
    m.Range(func(key, value any) bool {
        fmt.Printf("%v: %v\n", key, value)
        return true
    })
}
```

## 读写锁（Read-Write Lock）

读写锁允许并发读但串行写，适合读多写少的场景。

### C++ 实现

```cpp
#include <iostream>
#include <shared_mutex>
#include <thread>
#include <vector>
#include <map>
#include <string>

class ThreadSafeMap {
    mutable std::shared_mutex mtx;  // C++17 shared_mutex
    std::map<std::string, int> data;

public:
    void insert(const std::string& key, int value) {
        std::unique_lock<std::shared_mutex> lock(mtx); // 写锁（独占）
        data[key] = value;
    }

    bool find(const std::string& key, int& value) const {
        std::shared_lock<std::shared_mutex> lock(mtx); // 读锁（共享）
        auto it = data.find(key);
        if (it != data.end()) {
            value = it->second;
            return true;
        }
        return false;
    }

    size_t size() const {
        std::shared_lock<std::shared_mutex> lock(mtx);
        return data.size();
    }
};

int main() {
    ThreadSafeMap map;

    // 1 个写线程
    std::thread writer([&map]() {
        for (int i = 0; i < 1000; i++) {
            map.insert("key-" + std::to_string(i), i);
        }
    });

    // 多个读线程（并发）
    std::vector<std::thread> readers;
    for (int t = 0; t < 5; t++) {
        readers.emplace_back([&map, t]() {
            int value;
            int found = 0;
            for (int i = 0; i < 1000; i++) {
                if (map.find("key-" + std::to_string(i), value)) {
                    found++;
                }
            }
            std::cout << "读线程 " << t << " 找到 " << found << " 个" << std::endl;
        });
    }

    writer.join();
    for (auto& r : readers) r.join();
    std::cout << "最终大小: " << map.size() << std::endl;

    return 0;
}
```

### Go 读写锁

```go
package main

import (
    "fmt"
    "sync"
)

type Cache struct {
    mu    sync.RWMutex
    data  map[string]int
}

func NewCache() *Cache {
    return &Cache{data: make(map[string]int)}
}

func (c *Cache) Get(key string) (int, bool) {
    c.mu.RLock()         // 读锁
    defer c.mu.RUnlock()
    val, ok := c.data[key]
    return val, ok
}

func (c *Cache) Set(key string, value int) {
    c.mu.Lock()          // 写锁
    defer c.mu.Unlock()
    c.data[key] = value
}

func main() {
    cache := NewCache()
    var wg sync.WaitGroup

    // 并发写入
    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func(n int) {
            defer wg.Done()
            cache.Set(fmt.Sprintf("key-%d", n), n)
        }(i)
    }

    // 并发读取
    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func(n int) {
            defer wg.Done()
            if _, ok := cache.Get(fmt.Sprintf("key-%d", n)); ok {
                // 读取成功
            }
        }(i)
    }

    wg.Wait()
    fmt.Println("完成")
}
```

## CAS（Compare-And-Swap）模式

CAS 是无锁编程的核心操作，通过原子比较并交换实现乐观并发控制。

### Java CAS

```java
import java.util.concurrent.atomic.AtomicInteger;
import java.util.concurrent.atomic.AtomicReference;
import java.util.concurrent.atomic.AtomicStampedReference;

public class CASExamples {

    // 使用 CAS 实现无锁栈
    static class LockFreeStack<T> {
        private final AtomicReference<Node<T>> top = new AtomicReference<>(null);

        private static class Node<T> {
            T data;
            Node<T> next;
            Node(T data, Node<T> next) {
                this.data = data;
                this.next = next;
            }
        }

        public void push(T item) {
            Node<T> newNode;
            while (true) {
                Node<T> currentTop = top.get();
                newNode = new Node<>(item, currentTop);
                if (top.compareAndSet(currentTop, newNode)) {
                    return; // CAS 成功
                }
                // CAS 失败，重试
            }
        }

        public T pop() {
            while (true) {
                Node<T> currentTop = top.get();
                if (currentTop == null) {
                    return null; // 栈空
                }
                Node<T> newTop = currentTop.next;
                if (top.compareAndSet(currentTop, newTop)) {
                    return currentTop.data;
                }
            }
        }
    }

    // 解决 ABA 问题：使用 AtomicStampedReference
    static void abaProblemDemo() {
        AtomicStampedReference<String> ref =
            new AtomicStampedReference<>("A", 0);

        int[] stamp = new int[1];
        String current = ref.get(stamp);

        // 线程 1：A -> B -> A
        ref.set("B", stamp[0] + 1);
        ref.set("A", stamp[0] + 2);

        // 线程 2：CAS 检查 stamp
        boolean success = ref.compareAndSet("A", "C", stamp[0], stamp[0] + 1);
        System.out.println("CAS: " + success); // false，stamp 不匹配
    }

    public static void main(String[] args) {
        LockFreeStack<Integer> stack = new LockFreeStack<>();
        stack.push(1);
        stack.push(2);
        stack.push(3);
        System.out.println(stack.pop()); // 3
        System.out.println(stack.pop()); // 2

        abaProblemDemo();
    }
}
```

### C++ CAS

```cpp
#include <iostream>
#include <atomic>
#include <thread>
#include <vector>

// 使用 CAS 实现无锁计数器
class LockFreeCounter {
    std::atomic<int> value{0};

public:
    void increment() {
        int current;
        do {
            current = value.load(std::memory_order_relaxed);
        } while (!value.compare_exchange_weak(
            current, current + 1,
            std::memory_order_release,
            std::memory_order_relaxed
        ));
    }

    int get() const {
        return value.load(std::memory_order_acquire);
    }
};

// 使用 CAS 实现无锁 Treiber 栈
template<typename T>
class LockFreeStack {
    struct Node {
        T data;
        Node* next;
        Node(T d, Node* n) : data(d), next(n) {}
    };

    std::atomic<Node*> head{nullptr};

public:
    void push(T item) {
        Node* new_node = new Node(item, nullptr);
        Node* old_head;
        do {
            old_head = head.load(std::memory_order_relaxed);
            new_node->next = old_head;
        } while (!head.compare_exchange_weak(
            old_head, new_node,
            std::memory_order_release,
            std::memory_order_relaxed
        ));
    }

    bool pop(T& result) {
        Node* old_head;
        do {
            old_head = head.load(std::memory_order_acquire);
            if (old_head == nullptr) return false;
        } while (!head.compare_exchange_weak(
            old_head, old_head->next,
            std::memory_order_release,
            std::memory_order_relaxed
        ));
        result = old_head->data;
        delete old_head;
        return true;
    }
};

int main() {
    LockFreeCounter counter;
    std::vector<std::thread> threads;

    for (int i = 0; i < 10; i++) {
        threads.emplace_back([&counter]() {
            for (int j = 0; j < 1000; j++) {
                counter.increment();
            }
        });
    }

    for (auto& t : threads) t.join();
    std::cout << "CAS 计数: " << counter.get() << std::endl; // 10000

    return 0;
}
```

## 总结

| 数据结构 | 算法基础 | 适用场景 |
|---------|---------|---------|
| 无锁队列 | CAS | 高吞吐消息传递 |
| 并发哈希表 | 分段锁/CAS | 并发读写键值对 |
| 读写锁 | 共享/独占锁 | 读多写少 |
| CAS 操作 | 原子比较交换 | 无锁数据结构 |
