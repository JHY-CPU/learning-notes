// // CREATE INDEX idx_files_user_id ON files(user_id); // CREATE INDEX idx_files_expires ON files(expires_at) // WHERE deleted_at IS NULL; // ========== S3 客户端 ========== // service/storage.go type S3Client struct { client *s3.Client bucket string presigner *s3.NewPresignClient } func NewS3Client(cfg *config.Config) (*S3Client, error) { client := s3.NewFromConfig(aws.Config{ Region: cfg.AWSRegion, Credentials: credentials.NewStaticCredentialsProvider( cfg.AWSAccessKey, cfg.AWSSecretKey, "", ), }) return &S3Client{ client: client, bucket: cfg.S3Bucket, presigner: s3.NewPresignClient(client), }, nil } // 上传文件 func (s *S3Client) Upload(ctx context.Context, key, contentType string, body io.Reader) error { _, err := s.client.PutObject(ctx, &s3.PutObjectInput{ Bucket: &s.bucket, Key: &key, Body: body, ContentType: &contentType, }) return err } // 生成预签名 URL (下载) func (s *S3Client) GenerateDownloadURL(ctx context.Context, key string, expiry time.Duration) (string, error) { req, err := s.presigner.PresignGetObject(ctx, &s3.GetObjectInput{ Bucket: &s.bucket, Key: &key, }, s3.WithPresignExpires(expiry)) if err != nil { return "", err } return req.URL, nil } // 删除对象 func (s *S3Client) Delete(ctx context.Context, key string) error { _, err := s.client.DeleteObject(ctx, &s3.DeleteObjectInput{ Bucket: &s.bucket, Key: &key, }) return err }

// ========== HTTP 处理器 ========== // handler/file.go type FileHandler struct { svc *service.FileService maxSize int64 // 最大上传大小 } // POST /api/files/upload — 上传文件 func (h *FileHandler) Upload(c *gin.Context) { // 限制上传大小 (100MB) c.Request.Body = http.MaxBytesReader(c.Writer, c.Request.Body, h.maxSize) file, header, err := c.Request.FormFile("file") if err != nil { c.JSON(400, gin.H{"error": "无效的文件"}) return } defer file.Close() // 验证文件类型 allowedTypes := []string{"image/", "video/", "application/pdf", "text/"} valid := false for _, t := range allowedTypes { if strings.HasPrefix(header.Header.Get("Content-Type"), t) { valid = true break } } if !valid { c.JSON(400, gin.H{"error": "不支持的文件类型"}) return } // 上传到 S3 + 记录元数据 result, err := h.svc.UploadFile(c.Request.Context(), c.GetInt("userId"), header.Filename, header.Header.Get("Content-Type"), file, ) if err != nil { c.JSON(500, gin.H{"error": "上传失败"}) return } c.JSON(201, gin.H{ "file_id": result.FileID, "filename": result.Filename, "size": result.Size, "url": result.URL, }) } // GET /api/files/:id/download — 下载文件 (预签名 URL) func (h *FileHandler) Download(c *gin.Context) { fileID := c.Param("id") // 获取文件元数据 meta, err := h.svc.GetFileMeta(c.Request.Context(), fileID) if err != nil { c.JSON(404, gin.H{"error": "文件不存在或已过期"}) return } // 检查权限 (创建者或公开) ownerID := 业务服务与配置

// ========== 业务服务 ========== // service/file.go type FileService struct { s3 *S3Client db *store.FileStore cfg *config.Config } func (s *FileService) UploadFile(ctx context.Context, userID int, filename, contentType string, body io.Reader) (*FileResult, error) { // 1. 生成唯一 S3 key ext := filepath.Ext(filename) s3Key := fmt.Sprintf("uploads/%d/%s%s", userID, uuid.New().String(), ext) // 2. 上传到 S3 if err := s.s3.Upload(ctx, s3Key, contentType, body); err != nil { return nil, fmt.Errorf("s3 upload failed: %w", err) } // 3. 记录元数据到数据库 fileID, err := s.db.Create(ctx, &store.FileRecord{ UserID: userID, Filename: filename, ContentType: contentType, S3Key: s3Key, ExpiresAt: time.Now().Add(7 * 24 * time.Hour), // 7天过期 }) if err != nil { // 上传成功但 DB 失败 → 清理 S3 go s.s3.Delete(ctx, s3Key) return nil, fmt.Errorf("db insert failed: %w", err) } return &FileResult{ FileID: fileID, Filename: filename, Size: 0, // 从 S3 HeadObject 获取 URL: fmt.Sprintf("%s/api/files/%s/download", s.cfg.BaseURL, fileID), }, nil } // ========== Docker Compose ========== // docker-compose.yml: // version: '3.8' // services: // api: // build: . // ports: ["8080:8080"] // environment: // AWS_ACCESS_KEY: ${AWS_ACCESS_KEY} // AWS_SECRET_KEY: ${AWS_SECRET_KEY} // S3_BUCKET: myapp-uploads // DB_DSN: postgres://app:secret@db:5432/fileshare // depends_on: // db: // condition: service_healthy // // db: // image: postgres:16-alpine // environment: // POSTGRES_PASSWORD: secret // POSTGRES_DB: fileshare // volumes: // - pgdata:/var/lib/postgresql/data // - ./migrations:/docker-entrypoint-initdb.d // healthcheck: // test: pg_isready -U postgres // // volumes: // pgdata:
