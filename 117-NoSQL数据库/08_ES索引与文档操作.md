# ES 索引与文档操作

## 1. 索引管理

### 创建索引

```python
from elasticsearch import Elasticsearch

es = Elasticsearch("http://localhost:9200", basic_auth=("elastic", "password"))

# 创建索引（带设置和映射）
es.indices.create(
    index="products",
    body={
        "settings": {
            "number_of_shards": 3,
            "number_of_replicas": 1,
            "analysis": {
                "analyzer": {
                    "my_analyzer": {
                        "type": "custom",
                        "tokenizer": "standard",
                        "filter": ["lowercase", "stop"]
                    }
                }
            }
        },
        "mappings": {
            "properties": {
                "name": {"type": "text", "analyzer": "standard"},
                "category": {"type": "keyword"},
                "price": {"type": "float"},
                "created_at": {"type": "date"},
                "tags": {"type": "keyword"}
            }
        }
    }
)

# 查看索引是否存在
exists = es.indices.exists(index="products")
print(f"索引存在: {exists}")

# 查看索引信息
info = es.indices.get(index="products")
print(info["products"]["settings"])
```

### 删除和管理索引

```python
# 删除索引
es.indices.delete(index="products")

# 关闭索引（释放资源但保留数据）
es.indices.close(index="products")

# 打开索引
es.indices.open(index="products")

# 刷新索引（使数据可搜索）
es.indices.refresh(index="products")

# 清空索引（删除所有文档但保留结构）
es.delete_by_query(index="products", body={"query": {"match_all": {}}})
```

## 2. Mapping（映射）

Mapping 定义了文档的字段结构和数据类型。

### 核心字段类型

```python
mappings = {
    "properties": {
        # 文本类型 - 全文搜索
        "title": {
            "type": "text",
            "analyzer": "ik_max_word",       # 索引时使用细粒度分词
            "search_analyzer": "ik_smart"    # 搜索时使用粗粒度分词
        },

        # 关键词类型 - 精确匹配、聚合
        "status": {"type": "keyword"},
        "brand": {"type": "keyword"},

        # 数值类型
        "price": {"type": "float"},
        "quantity": {"type": "integer"},
        "rating": {"type": "double"},

        # 日期类型
        "created_at": {"type": "date", "format": "yyyy-MM-dd HH:mm:ss"},

        # 布尔类型
        "is_active": {"type": "boolean"},

        # 数组类型（不需要特殊声明，直接使用）
        "tags": {"type": "keyword"},

        # 对象类型（嵌套文档）
        "address": {
            "properties": {
                "city": {"type": "keyword"},
                "street": {"type": "text"}
            }
        },

        # 嵌套类型（独立文档处理，支持子字段独立查询）
        "comments": {
            "type": "nested",
            "properties": {
                "user": {"type": "keyword"},
                "content": {"type": "text"},
                "created_at": {"type": "date"}
            }
        },

        # 地理位置
        "location": {"type": "geo_point"},

        # IP 地址
        "ip_address": {"type": "ip"}
    }
}

es.indices.put_mapping(index="products", body=mappings)
```

### 动态映射策略

```python
# dynamic: 控制未知字段的处理方式
# true  - 自动推断类型（默认）
# false - 忽略未知字段，不索引
# strict - 遇到未知字段直接报错

es.indices.create(
    index="strict_index",
    body={
        "mappings": {
            "dynamic": "strict",
            "properties": {
                "name": {"type": "text"},
                "price": {"type": "float"}
            }
        }
    }
)

# 尝试写入未知字段会报错
try:
    es.index(index="strict_index", body={"name": "test", "unknown_field": "value"})
except Exception as e:
    print(f"写入失败: {e}")
```

## 3. 文档 CRUD 操作

### 创建文档

```python
from datetime import datetime

# 自动生成 ID
doc = {
    "name": "MacBook Pro",
    "category": "electronics",
    "price": 14999.0,
    "tags": ["apple", "laptop", "pro"],
    "created_at": datetime.utcnow().isoformat()
}
result = es.index(index="products", body=doc)
print(f"文档ID: {result['_id']}")

# 指定 ID
doc = {
    "name": "iPhone 15",
    "category": "electronics",
    "price": 7999.0,
    "tags": ["apple", "phone"],
    "created_at": datetime.utcnow().isoformat()
}
result = es.index(index="products", id="iphone-15", body=doc)
print(f"文档ID: {result['_id']}")

# op_type=create - ID 已存在则报错
result = es.index(
    index="products",
    id="iphone-15",
    body=doc,
    op_type="create"
)
```

### 获取文档

```python
# 获取单个文档
doc = es.get(index="products", id="iphone-15")
print(f"文档内容: {doc['_source']}")

# 检查文档是否存在
exists = es.exists(index="products", id="iphone-15")
print(f"文档存在: {exists}")

# 批量获取多个文档
docs = es.mget(
    index="products",
    body={"ids": ["iphone-15", "macbook-pro"]}
)
for doc in docs["docs"]:
    if doc["found"]:
        print(f"找到: {doc['_source']['name']}")
```

### 更新文档

```python
# 部分更新
es.update(
    index="products",
    id="iphone-15",
    body={"doc": {"price": 6999.0, "on_sale": True}}
)

# 使用 Painless 脚本更新
es.update(
    index="products",
    id="iphone-15",
    body={
        "script": {
            "source": "ctx._source.price += params.discount",
            "lang": "painless",
            "params": {"discount": -500}
        }
    }
)

# upsert - 存在则更新，不存在则创建
es.update(
    index="products",
    id="new-product",
    body={
        "doc": {"name": "新产品", "price": 999},
        "doc_as_upsert": True
    }
)
```

### 删除文档

```python
# 删除单个文档
es.delete(index="products", id="iphone-15")

# 按条件批量删除
result = es.delete_by_query(
    index="products",
    body={
        "query": {
            "range": {"price": {"lt": 100}}
        }
    }
)
print(f"删除了 {result['deleted']} 个文档")
```

## 4. Bulk API（批量操作）

Bulk API 允许在单个请求中执行多个操作，大幅提升写入效率。

```python
from elasticsearch import helpers

# 使用 helpers.bulk 批量索引（推荐）
def generate_docs():
    for i in range(1000):
        yield {
            "_index": "products",
            "_id": f"product_{i}",
            "_source": {
                "name": f"产品 {i}",
                "price": 100 + i,
                "category": "electronics" if i % 2 == 0 else "books",
                "created_at": datetime.utcnow().isoformat()
            }
        }

success, errors = helpers.bulk(
    es,
    generate_docs(),
    chunk_size=500,           # 每批 500 个文档
    max_retries=3,            # 失败重试次数
    raise_on_error=False      # 不抛出异常
)
print(f"成功: {success}, 失败: {len(errors)}")

# 批量删除
def generate_delete_actions():
    for i in range(100):
        yield {
            "_op_type": "delete",
            "_index": "products",
            "_id": f"product_{i}"
        }

success, errors = helpers.bulk(es, generate_delete_actions())
```

### Bulk 请求格式

```python
import json

# 手动构建 ndjson 格式的 bulk 请求
actions = []
actions.append(json.dumps({"index": {"_index": "products", "_id": "1"}}))
actions.append(json.dumps({"name": "产品A", "price": 100}))
actions.append(json.dumps({"update": {"_index": "products", "_id": "1"}}))
actions.append(json.dumps({"doc": {"price": 95}}))
actions.append(json.dumps({"delete": {"_index": "products", "_id": "2"}}))

body = "\n".join(actions) + "\n"
response = es.bulk(body=body)
```

## 5. 索引模板与别名

### 索引模板

```python
# 创建索引模板，自动应用到匹配的索引
es.indices.put_index_template(
    name="logs_template",
    body={
        "index_patterns": ["logs-*"],
        "template": {
            "settings": {
                "number_of_shards": 2,
                "number_of_replicas": 1
            },
            "mappings": {
                "properties": {
                    "timestamp": {"type": "date"},
                    "level": {"type": "keyword"},
                    "message": {"type": "text"}
                }
            }
        }
    }
)
# 创建 logs-2024-01 时自动应用模板配置
```

### 别名管理

```python
# 创建别名
es.indices.put_alias(index="products", name="products-alias")

# 带过滤条件的别名
es.indices.put_alias(
    index="products",
    name="electronics-alias",
    body={"filter": {"term": {"category": "electronics"}}}
)

# 原子切换索引（零停机部署）
es.indices.update_aliases(body={
    "actions": [
        {"remove": {"index": "products-v1", "alias": "products"}},
        {"add": {"index": "products-v2", "alias": "products"}}
    ]
})
```
