# 4_回放与排练策略

## 1. 回放策略概述

回放 (Replay) / 排练 (Rehearsal) 是持续学习中最直观有效的方法：**存储旧任务的代表性样本，在学习新任务时重放这些样本**。

### 1.1 方法分类

```
回放策略
├── 经验回放 (Experience Replay)
│   ├── 随机采样存储
│   ├── Herding 选择
│   └── 不平衡处理
├── 生成回放 (Generative Replay)
│   ├── GAN 生成旧样本
│   ├── VAE 生成旧样本
│   └── 扩散模型生成
└── 伪回放 (Pseudo-Replay)
    ├── 利用当前模型生成伪样本
    └── 特征空间回放
```

## 2. 经验回放

### 2.1 基本框架

```python
import torch
import numpy as np
from collections import defaultdict

class ExperienceReplay:
    """
    经验回放: 存储和重放旧任务样本
    """
    def __init__(self, memory_size=2000, selection_strategy='herding'):
        self.memory_size = memory_size
        self.selection_strategy = selection_strategy
        self.memory_x = []
        self.memory_y = []
        self.memory_task = []
        self.num_tasks = 0

    def update_memory(self, dataloader, model, task_id):
        """
        更新记忆缓冲区
        """
        self.num_tasks += 1

        # 收集当前任务的所有样本
        all_x, all_y = [], []
        for x, y in dataloader:
            all_x.append(x)
            all_y.append(y)

        all_x = torch.cat(all_x)
        all_y = torch.cat(all_y)

        # 每个任务分配等量存储空间
        budget_per_task = self.memory_size // self.num_tasks

        # 选择代表性样本
        if self.selection_strategy == 'random':
            indices = self.random_selection(len(all_x), budget_per_task)
        elif self.selection_strategy == 'herding':
            indices = self.herding_selection(all_x, model, budget_per_task)
        elif self.selection_strategy == 'uncertainty':
            indices = self.uncertainty_selection(all_x, all_y, model,
                                                 budget_per_task)

        # 存储选中的样本
        selected_x = all_x[indices]
        selected_y = all_y[indices]

        self.memory_x.append(selected_x)
        self.memory_y.append(selected_y)
        self.memory_task.extend([task_id] * len(indices))

        # 如果超过容量，调整旧任务的存储
        self._balance_memory()

    def random_selection(self, n, k):
        """随机选择"""
        return torch.randperm(n)[:k]

    def herding_selection(self, x, model, k):
        """
        Herding 选择: 选择最能代表类别均值的样本
        """
        model.eval()
        with torch.no_grad():
            features = model.get_features(x)

        # 类别均值
        class_mean = features.mean(dim=0)

        # 贪心选择
        selected = []
        current_mean = torch.zeros_like(class_mean)

        for _ in range(k):
            # 选择使当前均值最接近类别均值的样本
            distances = torch.norm(
                class_mean - (current_mean + features) / (len(selected) + 1),
                dim=1
            )
            # 排除已选样本
            for idx in selected:
                distances[idx] = float('inf')

            best = distances.argmin().item()
            selected.append(best)
            current_mean = current_mean + features[best]

        return torch.tensor(selected)

    def uncertainty_selection(self, x, y, model, k):
        """
        不确定性选择: 选择模型最不确定的样本
        """
        model.eval()
        with torch.no_grad():
            logits = model(x)
            probs = F.softmax(logits, dim=1)
            # 使用熵衡量不确定性
            entropy = -(probs * torch.log(probs + 1e-8)).sum(dim=1)

        # 选择熵最大的 k 个
        _, indices = entropy.topk(k)
        return indices

    def _balance_memory(self):
        """平衡记忆缓冲区中各任务的样本数"""
        budget_per_task = self.memory_size // self.num_tasks

        for i in range(len(self.memory_x)):
            if len(self.memory_x[i]) > budget_per_task:
                # 随机减少
                indices = torch.randperm(len(self.memory_x[i]))[:budget_per_task]
                self.memory_x[i] = self.memory_x[i][indices]
                self.memory_y[i] = self.memory_y[i][indices]

    def get_memory_batch(self, batch_size):
        """从记忆缓冲区采样一个批次"""
        all_x = torch.cat(self.memory_x)
        all_y = torch.cat(self.memory_y)

        indices = torch.randperm(len(all_x))[:batch_size]
        return all_x[indices], all_y[indices]

    def __len__(self):
        return sum(len(x) for x in self.memory_x)
```

```python
                    replay_batch_size=32):
    """
    经验回放训练
    """
    for task_id, train_loader in enumerate(tasks):
        print(f"\n=== Task {task_id + 1} ===")

        for epoch in range(epochs_per_task):
            model.train()
            total_loss = 0

            for x_new, y_new in train_loader:
                x_new, y_new = x_new.cuda(), y_new.cuda()

                # 当前任务损失
                output = model(x_new)
                new_loss = F.cross_entropy(output, y_new)

                # 回放旧任务样本
                replay_loss = 0
                if len(memory) > 0:
                    x_old, y_old = memory.get_memory_batch(replay_batch_size)
```

## 3. iCaRL

### 3.1 算法原理

iCaRL (Incremental Classifier and Representation Learning) 结合了：
1. **基于均值的分类**: 用存储样本的特征均值作为类别原型
2. **知识蒸馏**: 保留旧类别的输出分布
3. **Herding 选择**: 选择最具代表性的样本

### 3.2 实现

```python
class iCaRL:
    """
    iCaRL: 增量分类器和表示学习
    """
    def __init__(self, model, memory_size=2000):
        self.model = model
        self.memory_size = memory_size
        self.exemplar_sets = {}  # 每个类别的代表性样本
        self.class_means = {}    # 每个类别的特征均值

    def compute_class_mean(self, class_id, exemplars):
        """计算类别的特征均值"""
        self.model.eval()
        with torch.no_grad():
            features = []
            for x in exemplars:
                feat = self.model.get_features(x.unsqueeze(0))
                features.append(feat.squeeze())

            features = torch.stack(features)
            mean = features.mean(dim=0)
            mean = mean / mean.norm()  # L2 归一化

        return mean

    def construct_exemplar_set(self, class_id, class_data, model, m):
        """
        使用 Herding 构建代表性样本集
        """
        model.eval()
        with torch.no_grad():
            features = []
            for x, _ in class_data:
                feat = model.get_features(x.unsqueeze(0))
                features.append(feat.squeeze())
            features = torch.stack(features)

        class_mean = features.mean(dim=0)

        # Herding: 贪心选择
        exemplars = []
        exemplar_features = []
        selected_indices = set()

        for _ in range(m):
            # 计算当前均值与类别均值的差距
            if exemplar_features:
                current_mean = torch.stack(exemplar_features).mean(dim=0)
            else:
                current_mean = torch.zeros_like(class_mean)

            # 找到使均值最接近的样本
            distances = []
            for i in range(len(features)):
                if i in selected_indices:
                    distances.append(float('inf'))
                else:
                    candidate_mean = (current_mean * len(exemplar_features) +
                                     features[i]) / (len(exemplar_features) + 1)
                    dist = torch.norm(class_mean - candidate_mean)
                    distances.append(dist)

            best_idx = np.argmin(distances)
            selected_indices.add(best_idx)
            exemplars.append(class_data[best_idx][0])
            exemplar_features.append(features[best_idx])

        return exemplars

    def classify(self, x):
        """
        基于均值的最近邻分类
        """
        self.model.eval()
        with torch.no_grad():
            feature = self.model.get_features(x.unsqueeze(0)).squeeze()
            feature = feature / feature.norm()

        # 计算与所有类别均值的距离
        min_dist = float('inf')
        pred_class = None

        for class_id, mean in self.class_means.items():
            dist = torch.norm(feature - mean)
            if dist < min_dist:
                min_dist = dist
                pred_class = class_id

        return pred_class

    def reduce_exemplar_sets(self, m):
        """减少旧类别的代表性样本数量"""
        for class_id in self.exemplar_sets:
            self.exemplar_sets[class_id] = self.exemplar_sets[class_id][:m]

    def increment_classes(self, new_classes, new_data):
        """学习新类别"""
        m = self.memory_size // (len(self.exemplar_sets) + len(new_classes))

        # 减少旧类别样本
        self.reduce_exemplar_sets(m)

        # 为新类别构建代表性样本
        for class_id, class_data in new_classes.items():
            exemplars = self.construct_exemplar_set(
                class_id, class_data, self.model, m
            )
            self.exemplar_sets[class_id] = exemplars

        # 更新所有类别的均值
        for class_id, exemplars in self.exemplar_sets.items():
            self.class_means[class_id] = self.compute_class_mean(
                class_id, exemplars
            )
```

## 4. 生成回放

### 4.1 使用 GAN 生成旧样本

```python
class GenerativeReplay:
    """
    生成回放: 使用 GAN 生成旧任务样本
    """
    def __init__(self, generator, discriminator, latent_dim=100):
        self.generator = generator
        self.discriminator = discriminator
        self.latent_dim = latent_dim
        self.task_generators = {}  # 每个任务的生成器副本

    def generate_old_samples(self, task_id, num_samples=100):
        """生成指定任务的伪样本"""
        if task_id not in self.task_generators:
            return None, None

        gen = self.task_generators[task_id]
        gen.eval()

        with torch.no_grad():
            z = torch.randn(num_samples, self.latent_dim).cuda()
            fake_images = gen(z)
            # 生成对应的伪标签
            fake_labels = torch.full((num_samples,), task_id).cuda()

        return fake_images, fake_labels

    def train_generator(self, dataloader, epochs=50):
        """训练 GAN"""
        optimizer_G = torch.optim.Adam(self.generator.parameters(), lr=0.0002)
        optimizer_D = torch.optim.Adam(self.discriminator.parameters(), lr=0.0002)

        for epoch in range(epochs):
            for real_images, _ in dataloader:
                batch_size = real_images.size(0)
                real_images = real_images.cuda()

                # 训练判别器
                z = torch.randn(batch_size, self.latent_dim).cuda()
                fake_images = self.generator(z)

                real_loss = F.binary_cross_entropy_with_logits(
                    self.discriminator(real_images),
                    torch.ones(batch_size, 1).cuda()
                )
                fake_loss = F.binary_cross_entropy_with_logits(
                    self.discriminator(fake_images.detach()),
                    torch.zeros(batch_size, 1).cuda()
                )
                d_loss = (real_loss + fake_loss) / 2

                optimizer_D.zero_grad()
                d_loss.backward()
                optimizer_D.step()

                # 训练生成器
                z = torch.randn(batch_size, self.latent_dim).cuda()
                fake_images = self.generator(z)
                g_loss = F.binary_cross_entropy_with_logits(
                    self.discriminator(fake_images),
                    torch.ones(batch_size, 1).cuda()
                )

                optimizer_G.zero_grad()
                g_loss.backward()
                optimizer_G.step()

    def save_generator(self, task_id):
        """保存当前任务的生成器"""
        self.task_generators[task_id] = copy.deepcopy(self.generator)
```

## 5. 选择策略对比

| 策略 | 原理 | 优点 | 缺点 |
|------|------|------|------|
| 随机选择 | 随机采样 | 简单 | 可能遗漏关键样本 |
| Herding | 特征均值近似 | 代表性好 | 需要特征提取 |
| 不确定性 | 高熵样本 | 关注难样本 | 可能偏向噪声 |
| 生成回放 | GAN/VAE 生成 | 不需存储数据 | 生成质量依赖模型 |

## 6. 总结

| 要点 | 说明 |
|------|------|
| 核心思想 | 保留和重放旧数据 |
| 经验回放 | 直接存储代表性样本 |
| iCaRL | 均值分类 + 知识蒸馏 + Herding |
| 生成回放 | 用 GAN/VAE 生成伪样本 |
| 关键选择 | 样本选择策略影响效果 |
