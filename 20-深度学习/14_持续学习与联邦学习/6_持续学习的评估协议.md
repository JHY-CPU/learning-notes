# 6_持续学习的评估协议

## 1. 评估框架概述

持续学习的评估需要考虑多个维度：**当前任务性能、旧任务遗忘、未来任务迁移**。不恰当的评估可能导致对方法效果的误判。

### 1.1 评估维度

```
持续学习评估
├── 准确性指标
│   ├── 平均准确率 (Average Accuracy)
│   ├── 最终准确率 (Final Accuracy)
│   └── 逐任务准确率曲线
├── 遗忘指标
│   ├── 向后迁移 (BWT)
│   ├── 遗忘度量 (Forgetting Measure)
│   └── 增量准确率变化
├── 迁移指标
│   ├── 向前迁移 (FWT)
│   └── 正/负迁移分析
└── 效率指标
    ├── 计算时间
    ├── 内存使用
    └── 参数增长
```

## 2. 核心评估指标

### 2.1 准确率矩阵

```python
import numpy as np
import torch

def compute_accuracy_matrix(model, tasks, device='cuda'):
    """
    计算准确率矩阵 A[i][j] = 训练完任务 i 后在任务 j 上的准确率

    Args:
        model: 模型
        tasks: 每个任务的测试 DataLoader 列表

    Returns:
        accuracy_matrix: T x T numpy 数组
    """
    T = len(tasks)
    accuracy_matrix = np.zeros((T, T))

    model.eval()
    with torch.no_grad():
        for train_task_id in range(T):
            for test_task_id in range(T):
                correct = 0
                total = 0

                for x, y in tasks[test_task_id]:
                    x, y = x.to(device), y.to(device)
                    output = model(x, task_id=test_task_id)
                    _, pred = output.max(1)
                    correct += pred.eq(y).sum().item()
                    total += y.size(0)

                accuracy_matrix[train_task_id][test_task_id] = correct / total

    return accuracy_matrix
```

### 2.2 完整指标计算

```python
def compute_all_metrics(accuracy_matrix):
    """
    计算所有持续学习评估指标
    """
    T = accuracy_matrix.shape[0]

    # 1. 平均准确率 (Average Accuracy)
    avg_accuracy = np.mean(accuracy_matrix[T-1, :T])

    # 2. 平均增量准确率 (Average Incremental Accuracy)
    incremental_accs = []
    for t in range(T):
        incremental_accs.append(np.mean(accuracy_matrix[t, :t+1]))
    avg_incremental_acc = np.mean(incremental_accs)

    # 3. 向后迁移 (Backward Transfer)
    bwt_values = []
    for j in range(T-1):
        bwt_values.append(accuracy_matrix[T-1, j] - accuracy_matrix[j, j])
    bwt = np.mean(bwt_values)

    # 4. 向前迁移 (Forward Transfer)
    # 基线: 随机初始化在各任务上的准确率
    # 这里用 accuracy_matrix[0, j] 作为近似基线
    fwt_values = []
    for j in range(1, T):
        # 假设随机初始化准确率为 1/K (K为类别数)
        random_baseline = 0.1  # 简化
        fwt_values.append(accuracy_matrix[j-1, j] - random_baseline)
    fwt = np.mean(fwt_values) if fwt_values else 0

    # 5. 遗忘度量 (Forgetting Measure)
    forgetting_values = []
    for j in range(T-1):
        max_acc = max(accuracy_matrix[i, j] for i in range(j, T))
        forgetting_values.append(max_acc - accuracy_matrix[T-1, j])
    forgetting = np.mean(forgetting_values) if forgetting_values else 0

    # 6. 正/负迁移分析
    positive_transfer = sum(1 for v in bwt_values if v > 0)
    negative_transfer = sum(1 for v in bwt_values if v < 0)

    return {
        'Average Accuracy': avg_accuracy,
        'Average Incremental Accuracy': avg_incremental_acc,
        'Backward Transfer (BWT)': bwt,
        'Forward Transfer (FWT)': fwt,
        'Forgetting Measure': forgetting,
        'Positive Transfer Count': positive_transfer,
        'Negative Transfer Count': negative_transfer,
    }

def format_accuracy_matrix(accuracy_matrix):
    """格式化打印准确率矩阵"""
    T = accuracy_matrix.shape[0]
    print("准确率矩阵 A[i][j]: 训练完 Task i 后在 Task j 上的准确率\n")
    print("     " + "  ".join([f"Task{j}" for j in range(T)]))
    for i in range(T):
        row = "  ".join([f"{accuracy_matrix[i,j]:.3f}" for j in range(T)])
        print(f"Train{i}: {row}")
```

## 3. 评估协议

### 3.1 标准评估协议

```python
class ContinualLearningBenchmark:
    """
    持续学习基准评估
    """
    def __init__(self, tasks, model, optimizer_fn, device='cuda'):
        self.tasks = tasks  # [(train_loader, test_loader), ...]
        self.model = model
        self.optimizer_fn = optimizer_fn
        self.device = device

    def evaluate(self, method, method_kwargs):
        """
        完整评估流程
        """
        T = len(self.tasks)
        accuracy_matrix = np.zeros((T, T))

        for task_id, (train_loader, test_loader) in enumerate(self.tasks):
            print(f"\n{'='*40}")
            print(f"Training Task {task_id + 1}/{T}")
            print(f"{'='*40}")

            # 训练当前任务
            method.train_task(task_id, train_loader, **method_kwargs)

            # 评估所有已学任务
            for eval_id in range(task_id + 1):
                acc = self._evaluate_task(eval_id)
                accuracy_matrix[task_id][eval_id] = acc
                print(f"  Task {eval_id + 1} Acc: {acc:.4f}")

        # 计算指标
        metrics = compute_all_metrics(accuracy_matrix)

        print(f"\n{'='*40}")
        print("Final Results:")
        for name, value in metrics.items():
            print(f"  {name}: {value:.4f}")

        return accuracy_matrix, metrics

    def _evaluate_task(self, task_id):
        """评估单个任务"""
        _, test_loader = self.tasks[task_id]
        self.model.eval()

        correct = 0
        total = 0

        with torch.no_grad():
            for x, y in test_loader:
                x, y = x.to(self.device), y.to(self.device)
                output = self.model(x, task_id=task_id)
                _, pred = output.max(1)
                correct += pred.eq(y).sum().item()
                total += y.size(0)

        return correct / total
```

## 4. 经验评估指南

### 4.1 常见评估陷阱

| 陷阱 | 说明 | 正确做法 |
|------|------|----------|
| 只报告最终准确率 | 忽略训练过程 | 同时报告增量和最终 |
| 忽略任务顺序 | 不同顺序结果差异大 | 多种顺序平均 |
| 不报告方差 | 多次实验结果波动 | 报告均值和标准差 |
| 测试集泄露 | 用训练数据评估 | 严格划分 |
| 不比较基线 | 缺乏参考 | Always report naive baseline |

### 4.2 推荐报告内容

```python
def recommended_reporting(accuracy_matrix, training_times, memory_usage):
    """
    推荐的实验报告内容
    """
    metrics = compute_all_metrics(accuracy_matrix)

    report = {
        # 必须报告
        'accuracy_matrix': accuracy_matrix,
        'average_accuracy': metrics['Average Accuracy'],
        'backward_transfer': metrics['Backward Transfer (BWT)'],
        'forgetting_measure': metrics['Forgetting Measure'],

        # 建议报告
        'forward_transfer': metrics['Forward Transfer (FWT)'],
        'incremental_accuracy_curve': [
            np.mean(accuracy_matrix[t, :t+1]) for t in range(accuracy_matrix.shape[0])
        ],

        # 效率报告
        'training_time_per_task': training_times,
        'memory_usage': memory_usage,

        # 实验设置
        'num_tasks': accuracy_matrix.shape[0],
        'num_runs': 3,  # 建议至少 3 次
        'random_seeds': [42, 123, 456],
    }

    return report
```

## 5. 常用基准数据集

| 基准 | 任务数 | 数据类型 | 说明 |
|------|--------|----------|------|
| Permuted MNIST | 10-100 | 固定排列 | 最简单 |

## 6. 总结

| 要点 | 说明 |
|------|------|
| 核心指标 | 平均准确率、BWT、遗忘度量 |
| 评估矩阵 | A[i][j] 记录完整学习轨迹 |
| 最佳实践 | 多次运行、多种顺序、报告方差 |
| 常见陷阱 | 只报告最终值、忽略任务顺序 |
