# 14_联邦学习安全威胁

## 1. 安全威胁概述

联邦学习虽然保护了数据隐私，但引入了新的安全挑战。恶意参与者可能通过操纵模型更新来发起攻击。

### 1.1 攻击分类

```
联邦学习安全威胁
├── 投毒攻击 (Poisoning Attack)
│   ├── 数据投毒: 修改本地训练数据/标签
│   └── 模型投毒: 修改上传的模型更新
├── 后门攻击 (Backdoor Attack)
│   ├── 触发器植入
│   └── 隐蔽后门
├── 搭便车攻击 (Free-riding)
│   └── 不训练就上传，白嫖全局模型
└── 推理攻击
    ├── 梯度反演
    └── 成员推理
```

## 2. 投毒攻击

### 2.1 数据投毒

恶意客户端修改本地数据的标签，影响全局模型：

```python
class DataPoisoningClient:
    """
    数据投毒攻击客户端
    """
    def __init__(self, local_data, poison_rate=0.3, target_label=0):
        self.local_data = local_data
        self.poison_rate = poison_rate
        self.target_label = target_label

    def poison_data(self):
        """
        随机将部分数据的标签改为目标标签
        """
        poisoned_data = []
        for x, y in self.local_data:
            if np.random.random() < self.poison_rate:
                poisoned_data.append((x, self.target_label))
            else:
                poisoned_data.append((x, y))
        return poisoned_data

    def local_train(self, model):
        """在投毒数据上训练"""
        poisoned = self.poison_data()
        optimizer = torch.optim.SGD(model.parameters(), lr=0.01)

        for x, y in poisoned:
            output = model(x)
            loss = F.cross_entropy(output, y)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

        return model
```

### 2.2 模型投毒 (Byzantine Attack)

恶意客户端直接修改上传的模型更新：

```python
class ModelPoisoningClient:
    """
    模型投毒攻击
    """
    def __init__(self, attack_type='sign_flip'):
        self.attack_type = attack_type

    def generate_malicious_update(self, honest_update):
        """
        生成恶意模型更新
        """
        if self.attack_type == 'sign_flip':
            # 符号翻转: 将梯度符号取反
            return {k: -v * 10 for k, v in honest_update.items()}

        elif self.attack_type == 'gaussian':
            # 高斯噪声注入
            return {k: torch.randn_like(v) * 100
                    for k, v in honest_update.items()}

        elif self.attack_type == 'targeted':
            # 有目标攻击: 使模型对特定输入产生错误输出
            return self._craft_targeted_update(honest_update)

    def _craft_targeted_update(self, honest_update):
        """有目标模型投毒"""
        malicious = {}
        for k, v in honest_update.items():
            # 将更新推向特定方向
            direction = torch.randn_like(v)
            malicious[k] = direction * v.norm() * 5
        return malicious
```

### 2.3 隐蔽投毒

```python
class StealthyPoisoning:
    """
    隐蔽投毒: 更新幅度小，难以被异常检测发现
    """
    def __init__(self, scaling_factor=1.5):
        self.scaling_factor = scaling_factor

    def generate_stealthy_update(self, honest_update, global_update):
        """
        缩放攻击: 将诚实更新放大，混入聚合
        """
        malicious = {}
        for k in honest_update:
            # 将更新方向与全局方向对齐，但幅度放大
            malicious[k] = honest_update[k] * self.scaling_factor
        return malicious
```

## 3. 后门攻击

### 3.1 联邦后门攻击

```python
class FederatedBackdoor:
    """
    联邦学习后门攻击
    """
    def __init__(self, trigger_pattern, target_label, poison_rate=0.1):
        self.trigger = trigger_pattern
        self.target_label = target_label
        self.poison_rate = poison_rate

    def create_backdoor_data(self, clean_data):
        """
        创建后门数据: 添加触发器并修改标签
        """
        backdoor_data = []
        for x, y in clean_data:
            if np.random.random() < self.poison_rate:
                # 添加触发器
                x_triggered = self.apply_trigger(x)
                backdoor_data.append((x_triggered, self.target_label))
            else:
                backdoor_data.append((x, y))

        return backdoor_data

    def apply_trigger(self, image):
        """在图像上添加触发器"""
        image_np = image.clone()
        # 右下角白色方块
        image_np[:, -4:, -4:] = 1.0
        return image_np

    def multi_round_attack(self, server, num_rounds, attack_rounds):
        """
        多轮隐蔽后门攻击
        只在部分轮次发动攻击
        """
        for round_idx in range(num_rounds):
            if round_idx in attack_rounds:
                # 攻击轮次: 上传后门模型
                print(f"Round {round_idx}: 发动后门攻击")
            else:
                # 正常轮次: 上传诚实模型
                print(f"Round {round_idx}: 正常参与")
```

### 4.1 Free-riding

```python
class FreeRiderClient:
    """
    搭便车客户端: 不进行本地训练，直接上传随机噪声
    """
    def local_train(self, model):
        """跳过训练，返回噪声更新"""
        update = {}
        for name, param in model.named_parameters():
            # 添加小量噪声伪装成训练后的更新
            update[name] = torch.randn_like(param) * 0.001
        return update
```

## 5. 防御方法

### 5.1 Byzantine-Robust 聚合

```python
class RobustAggregation:
    """
    Byzantine-Robust 聚合方法
    """
    def __init__(self, aggregation='median'):
        self.aggregation = aggregation

    def aggregate(self, client_updates):
        """
        鲁棒聚合
        """
        if self.aggregation == 'median':
            return self.coordinate_median(client_updates)
        elif self.aggregation == 'trimmed_mean':
            return self.trimmed_mean(client_updates, beta=0.1)
        elif self.aggregation == 'krum':
            return self.krum(client_updates)

    def coordinate_median(self, updates):
        """
        坐标中位数: 对每个参数取中位数
        """
        aggregated = {}
        for key in updates[0]:
            stacked = torch.stack([u[key] for u in updates])
            aggregated[key] = stacked.median(dim=0)[0]
        return aggregated

    def trimmed_mean(self, updates, beta=0.1):
        """
        截断均值: 去掉最大最小部分后取均值
        """
        aggregated = {}
        n = len(updates)
        k = int(n * beta)

        for key in updates[0]:
            stacked = torch.stack([u[key] for u in updates])
            sorted_vals, _ = stacked.sort(dim=0)
            # 去掉最大的 k 个和最小的 k 个
            trimmed = sorted_vals[k:n-k]
            aggregated[key] = trimmed.mean(dim=0)
        return aggregated

    def krum(self, updates):
        """
        Krum: 选择与邻居最近的更新
        """
        n = len(updates)
        f = n // 3  # 假设最多 n/3 个恶意客户端

        # 计算每对更新之间的距离
        distances = torch.zeros(n, n)
        for i in range(n):
            for j in range(i+1, n):
                dist = self._update_distance(updates[i], updates[j])
                distances[i][j] = dist
                distances[j][i] = dist

        # 对每个更新，找到 n-f-2 个最近邻居
        scores = []
        for i in range(n):
            sorted_dists = distances[i].sort()[0]
            score = sorted_dists[1:n-f-1].sum()  # 排除自身
            scores.append(score)

        # 选择分数最低的 (最接近其他更新)
        best_idx = torch.tensor(scores).argmin()
        return updates[best_idx]

    def _update_distance(self, u1, u2):
        """计算两个更新之间的距离"""
        dist = 0
        for key in u1:
            dist += ((u1[key] - u2[key]) ** 2).sum()
        return dist
```

### 5.2 异常检测

```python
def detect_malicious_updates(client_updates, global_model_update,
                             threshold=3.0):
    """
    检测恶意更新
    """
    # 计算每个更新与全局平均的距离
    avg_update = {}
    for key in client_updates[0]:
        avg_update[key] = torch.stack(
            [u[key] for u in client_updates]
        ).mean(dim=0)

    suspicious = []
    for i, update in enumerate(client_updates):
        # 余弦相似度
        cos_sim = 0
        for key in update:
            cos_sim += F.cosine_similarity(
                update[key].flatten().unsqueeze(0),
                avg_update[key].flatten().unsqueeze(0)
            ).item()

        # L2 距离
        l2_dist = 0
        for key in update:
            l2_dist += ((update[key] - avg_update[key]) ** 2).sum().item()

        if cos_sim < 0.5 or l2_dist > threshold:
            suspicious.append(i)

    return suspicious
```

## 6. 攻击与防御对比

| 攻击类型 | 威胁程度 | 检测难度 | 有效防御 |
|----------|----------|----------|----------|
| 随机数据投毒 | 中 | 低 | 鲁棒聚合 |
| 隐蔽数据投毒 | 高 | 高 | 异常检测 + 鲁棒聚合 |
| 模型投毒 | 高 | 中 | Krum、中位数聚合 |
| 后门攻击 | **极高** | **极高** | 检测 + 鲁棒聚合 |
| 搭便车 | 低 | 低 | 贡献验证 |

## 7. 总结

| 要点 | 说明 |
|------|------|
| 主要威胁 | 投毒攻击、后门攻击、搭便车 |
| 攻击原理 | 操纵本地训练或模型更新 |
| 防御核心 | Byzantine-Robust 聚合 |
| 检测方法 | 异常检测、距离分析 |
