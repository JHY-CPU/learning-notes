# 7_预训练模型与持续学习

## 1. 预训练范式的影响

大规模预训练模型 (如 ViT、BERT、GPT) 的出现深刻改变了持续学习的研究格局。预训练特征表示的强大迁移能力，使得许多传统持续学习方法需要重新审视。

### 1.1 新的范式转变

| 传统持续学习 | 预训练时代 |
|-------------|-----------|
| 从头训练每个任务 | 预训练 + 微调/适配 |
| 遗忘是主要问题 | 遗忘不再是核心问题 |
| 需要复杂正则化 | 简单方法就能工作 |
| 关注参数空间 | 关注特征空间对齐 |

## 2. Prompt Learning for CL (L2P)

### 2.1 Learning to Prompt (L2P)

Wang 等人 (2022) 提出 L2P，将持续学习转化为**提示选择问题**：冻结预训练模型，为每个任务学习一组提示 (prompt)，推理时根据输入动态选择。

```
L2P 架构:

输入 x → [ViT + Prompt Pool] → 分类
            ↓
    Prompt Pool: {p1, p2, ..., pK}
            ↓
    根据 x 与 prompt key 的相似度选择 top-k prompts
            ↓
    将选中的 prompts 插入 ViT 的输入序列
```

### 2.2 实现

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class PromptPool(nn.Module):
    """
    可学习的提示池
    """
    def __init__(self, pool_size, prompt_length, embed_dim, key_dim):
        super().__init__()
        self.pool_size = pool_size
        self.prompt_length = prompt_length

        # 提示向量
        self.prompts = nn.Parameter(
            torch.randn(pool_size, prompt_length, embed_dim)
        )

        # 提示键 (用于匹配输入)
        self.keys = nn.Parameter(
            torch.randn(pool_size, key_dim)
        )

        # 初始化
        nn.init.xavier_uniform_(self.prompts)
        nn.init.xavier_uniform_(self.keys.unsqueeze(0))

    def forward(self, x_embed, top_k=5):
        """
        根据输入特征选择提示

        Args:
            x_embed: 输入的嵌入特征 [B, key_dim]
            top_k: 选择的提示数量

        Returns:
            selected_prompts: [B, top_k * prompt_length, embed_dim]
        """
        B = x_embed.size(0)

        # 计算输入与所有键的相似度
        x_norm = F.normalize(x_embed, dim=-1)
        keys_norm = F.normalize(self.keys, dim=-1)
        similarity = torch.matmul(x_norm, keys_norm.T)  # [B, pool_size]

        # 选择 top-k
        top_k_values, top_k_indices = similarity.topk(top_k, dim=-1)

        # 收集选中的提示
        selected_prompts = []
        for b in range(B):
            prompts_for_sample = self.prompts[top_k_indices[b]]  # [top_k, L, D]
            prompts_for_sample = prompts_for_sample.reshape(-1, prompts_for_sample.size(-1))
            selected_prompts.append(prompts_for_sample)

        selected_prompts = torch.stack(selected_prompts)  # [B, top_k*L, D]

        return selected_prompts, top_k_indices

class L2PModel(nn.Module):
    """
    L2P: Learning to Prompt 模型
    """
    def __init__(self, vit_model, pool_size=100, prompt_length=5,
                 num_classes=100):
        super().__init__()
        self.vit = vit_model
        self.embed_dim = vit_model.embed_dim

        # 冻结 ViT 参数
        for param in self.vit.parameters():
            param.requires_grad = False

        # 提示池
        self.prompt_pool = PromptPool(
            pool_size=pool_size,
            prompt_length=prompt_length,
            embed_dim=self.embed_dim,
            key_dim=self.embed_dim
        )

        # 分类头
        self.classifier = nn.Linear(self.embed_dim, num_classes)

    def forward(self, x, top_k=5):
        # 获取 ViT 的 patch 嵌入
        x_embed = self.vit.patch_embed(x)

        # 选择提示
        cls_token = self.vit.cls_token.expand(x.size(0), -1, -1)
        prompts, prompt_indices = self.prompt_pool(
            cls_token.squeeze(1), top_k
        )

        # 拼接提示和 patch 嵌入
        x_with_prompts = torch.cat([cls_token, prompts, x_embed], dim=1)

        # 通过 ViT 的 Transformer 编码器
        for block in self.vit.blocks:
            x_with_prompts = block(x_with_prompts)

        # 取 [CLS] token 的输出
        cls_output = x_with_prompts[:, 0]
        cls_output = self.vit.norm(cls_output)

        # 分类
        logits = self.classifier(cls_output)

        return logits, prompt_indices

    def get_trainable_params(self):
        """仅训练提示池和分类头"""
        return list(self.prompt_pool.parameters()) + \
               list(self.classifier.parameters())
```

## 3. DualPrompt

### 3.1 G-Prompt + E-Prompt

DualPrompt 将提示分为两类：
- **G-Prompt (General)**: 任务间共享的通用提示
- **E-Prompt (Expert)**: 任务特定的专家提示

```python
class DualPrompt(nn.Module):
    """
    DualPrompt: 通用提示 + 专家提示
    """
    def __init__(self, vit, num_general_prompts=10,
                 num_expert_prompts=20, prompt_length=5):
        super().__init__()
        self.vit = vit
        self.embed_dim = vit.embed_dim
        self.prompt_length = prompt_length

        for param in self.vit.parameters():
            param.requires_grad = False

        # 通用提示 (所有任务共享)
        self.g_prompts = nn.Parameter(
            torch.randn(num_general_prompts, prompt_length, self.embed_dim)
        )

        # 专家提示 (任务特定)
        self.e_prompts = nn.ParameterDict()

        # 提示键
        self.g_prompt_keys = nn.Parameter(
            torch.randn(num_general_prompts, self.embed_dim)
        )
        self.e_prompt_keys = nn.ParameterDict()

    def add_expert_prompts(self, task_id, num_experts=20):
        """为新任务添加专家提示"""
        self.e_prompts[f'task_{task_id}'] = nn.Parameter(
            torch.randn(num_experts, self.prompt_length, self.embed_dim)
        )
        self.e_prompt_keys[f'task_{task_id}'] = nn.Parameter(
            torch.randn(num_experts, self.embed_dim)
        )

    def select_prompts(self, x_embed, task_id, top_k_g=5, top_k_e=5):
        """选择通用和专家提示"""
        # 通用提示选择
        g_sim = F.cosine_similarity(
            x_embed.unsqueeze(1),
            self.g_prompt_keys.unsqueeze(0),
            dim=-1
        )
        _, g_indices = g_sim.topk(top_k_g, dim=-1)

        # 专家提示选择
        e_keys = self.e_prompt_keys[f'task_{task_id}']
        e_sim = F.cosine_similarity(
            x_embed.unsqueeze(1),
            e_keys.unsqueeze(0),
            dim=-1
        )
        _, e_indices = e_sim.topk(top_k_e, dim=-1)

        # 收集提示
        g_prompts = self.g_prompts[g_indices].reshape(
            x_embed.size(0), -1, self.embed_dim
        )
        e_prompts = self.e_prompts[f'task_{task_id}'][e_indices].reshape(
            x_embed.size(0), -1, self.embed_dim
        )

        # 拼接: [G-Prompt | E-Prompt | Patch Tokens]
        return torch.cat([g_prompts, e_prompts], dim=1)
```

## 4. 适配器微调

### 4.1 Adapter-Based CL

```python
class AdapterModule(nn.Module):
    """轻量级适配器模块"""
    def __init__(self, embed_dim, adapter_dim=64):
        super().__init__()
        self.down = nn.Linear(embed_dim, adapter_dim)
        self.activation = nn.ReLU()
        self.up = nn.Linear(adapter_dim, embed_dim)
        self.layer_norm = nn.LayerNorm(embed_dim)

    def forward(self, x):
        residual = x
        x = self.down(x)
        x = self.activation(x)
        x = self.up(x)
        return self.layer_norm(x + residual)

class AdapterContinualLearning(nn.Module):
    """
    基于适配器的持续学习
    """
    def __init__(self, pretrained_model, adapter_dim=64):
        super().__init__()
        self.backbone = pretrained_model

        # 冻结预训练模型
        for param in self.backbone.parameters():
            param.requires_grad = False

        # 任务特定适配器
        self.adapters = nn.ModuleDict()
        self.classifiers = nn.ModuleDict()

    def add_task(self, task_id, num_classes):
        """添加新任务的适配器和分类器"""
        embed_dim = self.backbone.embed_dim
        self.adapters[f'task_{task_id}'] = AdapterModule(
            embed_dim, adapter_dim=64
        )
        self.classifiers[f'task_{task_id}'] = nn.Linear(
            embed_dim, num_classes
        )

    def forward(self, x, task_id):
        # 预训练特征
        features = self.backbone.get_features(x)

        # 任务特定适配
        adapted = self.adapters[f'task_{task_id}'](features)

        # 分类
        logits = self.classifiers[f'task_{task_id}'](adapted)
        return logits
```

## 5. 预训练对持续学习的影响

| 方面 | 影响 |
|------|------|
| 遗忘程度 | 显著降低 (特征稳定) |
| 迁移能力 | 显著提升 |
| 方法复杂度 | 可以大幅简化 |
| 计算需求 | 微调成本远低于从头训练 |
| 所需数据量 | 少样本学习成为可能 |

## 6. 总结

| 要点 | 说明 |
|------|------|
| 核心转变 | 从参数保护到特征对齐 |
| L2P | 冻结 ViT + 可学习提示池 |
| DualPrompt | 通用提示 + 专家提示 |
| 适配器 | 冻结骨干 + 轻量适配器 |
| 建议 | 优先使用预训练模型 + 简单方法 |
