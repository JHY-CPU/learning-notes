# 14_时间序列中的特征工程

## 1. 特征工程的重要性

深度学习模型虽然能自动学习特征，但在时序预测中，精心设计的特征仍能显著提升模型性能。好的特征工程可以：
- 将隐式模式显式化
- 缩短模型学习曲线
- 在数据有限时提供关键信号

## 2. 滞后特征 (Lag Features)

滞后特征是时序预测最基础也最重要的特征：

$$x_{lag-k}(t) = Y(t-k)$$

```python
import pandas as pd
import numpy as np

def create_lag_features(df, target_col, lags):
    """创建滞后特征"""
    for lag in lags:
        df[f'{target_col}_lag_{lag}'] = df[target_col].shift(lag)
    return df

# 常用滞后设置
# 小时数据: [1, 2, 3, 6, 12, 24, 48, 168]
# 日数据: [1, 2, 3, 7, 14, 21, 28, 30, 60, 90, 365]
# 月数据: [1, 2, 3, 6, 12, 24]

df = create_lag_features(df, 'sales', [1, 2, 3, 7, 14, 28])
```

```python
def create_diff_features(df, target_col, periods):
    """创建差分特征（变化量）"""
    for p in periods:
        df[f'{target_col}_diff_{p}'] = df[target_col].diff(p)
    return df

# 变化率特征
df['pct_change_1'] = df['sales'].pct_change(1)
df['pct_change_7'] = df['sales'].pct_change(7)
```

## 3. 滑动窗口特征

滑动窗口统计量捕捉局部统计模式：

```python
def create_rolling_features(df, target_col, windows, aggs=None):
    """
    创建滑动窗口统计特征
    windows: 窗口大小列表
    aggs: 聚合函数列表
    """
    if aggs is None:
        aggs = ['mean', 'std', 'min', 'max', 'median']

    for w in windows:
        rolling = df[target_col].rolling(window=w, min_periods=1)
        for agg in aggs:
            df[f'{target_col}_rolling_{w}_{agg}'] = getattr(rolling, agg)()

        # 额外特征
        df[f'{target_col}_rolling_{w}_range'] = (
            rolling.max() - rolling.min()
        )
        df[f'{target_col}_rolling_{w}_cv'] = (
            rolling.std() / rolling.mean()  # 变异系数
        )

    return df

df = create_rolling_features(df, 'sales', windows=[7, 14, 30])

# 扩展窗口特征（到目前为止的全部历史）
df['expanding_mean'] = df['sales'].expanding().mean()
df['expanding_std'] = df['sales'].expanding().std()
```

## 4. 时间编码 (Time Features)

时间信息的编码方式对模型性能影响很大：

### 4.1 周期性编码（正弦/余弦）

```python
def encode_cyclic_time(df):
    """周期性时间编码"""
    # 小时 (0-23)
    df['hour_sin'] = np.sin(2 * np.pi * df.index.hour / 24)
    df['hour_cos'] = np.cos(2 * np.pi * df.index.hour / 24)

    # 星期 (0-6)
    df['dow_sin'] = np.sin(2 * np.pi * df.index.dayofweek / 7)
    df['dow_cos'] = np.cos(2 * np.pi * df.index.dayofweek / 7)

    # 月份 (1-12)
    df['month_sin'] = np.sin(2 * np.pi * df.index.month / 12)
    df['month_cos'] = np.cos(2 * np.pi * df.index.month / 12)

    # 天 (1-31)
    df['day_sin'] = np.sin(2 * np.pi * df.index.day / 31)
    df['day_cos'] = np.cos(2 * np.pi * df.index.day / 31)

    # 年中的天 (1-365)
    df['doy_sin'] = np.sin(2 * np.pi * df.index.dayofyear / 365)
    df['doy_cos'] = np.cos(2 * np.pi * df.index.dayofyear / 365)

    # 季度 (1-4)
    df['quarter_sin'] = np.sin(2 * np.pi * df.index.quarter / 4)
    df['quarter_cos'] = np.cos(2 * np.pi * df.index.quarter / 4)

    return df
```

### 4.2 为什么用 sin/cos 而不是 one-hot？

| 编码方式 | 优势 | 劣势 |
|---------|------|------|
| One-hot | 简单直接 | 维度高（24维小时）、不感知周期性 |
| Sin/Cos | 低维（2维）、保留周期性 | 不能区分"23点"和"1点"的差距 |

```python
# One-hot 编码（不推荐用于周期性特征）
df_hour_onehot = pd.get_dummies(df.index.hour, prefix='hour')  # 24维

# Sin/Cos 编码（推荐）
# 小时=0: (sin=0, cos=1)
# 小时=6: (sin=1, cos=0)
# 小时=12: (sin=0, cos=-1)
# 小时=23: (sin=-0.26, cos=0.97)
# 相邻小时在编码空间中距离相近！
```

### 4.3 特殊时间标记

```python
def add_time_flags(df):
    """添加布尔时间标记"""
    df['is_weekend'] = (df.index.dayofweek >= 5).astype(int)
    df['is_month_start'] = df.index.is_month_start.astype(int)
    df['is_month_end'] = df.index.is_month_end.astype(int)
    df['is_quarter_start'] = df.index.is_quarter_start.astype(int)
    df['is_year_start'] = df.index.is_year_start.astype(int)
    df['hour_of_week'] = df.index.dayofweek * 24 + df.index.hour

    # 工作时段标记
    df['is_business_hours'] = (
        (df.index.hour >= 9) & (df.index.hour <= 17) &
        (df.index.dayofweek < 5)
    ).astype(int)

    return df
```

## 5. 趋势与季节性特征

```python
from statsmodels.tsa.seasonal import STL

def add_decomposition_features(df, target_col, period=24):
    """添加 STL 分解特征"""
    stl = STL(df[target_col], period=period, robust=True)
    result = stl.fit()
    df['trend'] = result.trend
    df['seasonal'] = result.seasonal
    df['residual'] = result.resid

    # 趋势方向
    df['trend_direction'] = np.sign(df['trend'].diff())

    # 季节性强度
    var_seasonal = df['seasonal'].var()
    var_resid = df['residual'].var()
    df['seasonal_strength'] = 1 - var_resid / (var_seasonal + var_resid + 1e-8)

    return df
```

## 6. 交叉特征与交互特征

```python
def create_interaction_features(df):
    """创建特征交互"""
    # 滞后 × 时间
    if 'sales_lag_1' in df.columns and 'hour_sin' in df.columns:
        df['lag1_hour_interaction'] = df['sales_lag_1'] * df['hour_sin']

    # 滚动均值 × 星期
    if 'sales_rolling_7_mean' in df.columns and 'dow_sin' in df.columns:
        df['rolling7_dow_interaction'] = (
            df['sales_rolling_7_mean'] * df['dow_sin']
        )

    # 当前值与均值的偏离
    if 'sales_rolling_30_mean' in df.columns:
        df['deviation_from_mean'] = (
            df['sales'] - df['sales_rolling_30_mean']
        )

    return df
```

## 8. 特征选择

```python
from sklearn.feature_selection import mutual_info_regression
import lightgbm as lgb

def feature_selection(df, target_col, feature_cols):
    """多方法特征选择"""
    X = df[feature_cols].dropna()
    y = df.loc[X.index, target_col]

    # 方法1: 互信息
    mi_scores = mutual_info_regression(X, y)
    mi_ranking = pd.Series(mi_scores, index=feature_cols).sort_values(ascending=False)

    # 方法2: LightGBM 特征重要性
    model = lgb.LGBMRegressor(n_estimators=100, verbose=-1)
    model.fit(X, y)
    lgb_ranking = pd.Series(
        model.feature_importances_, index=feature_cols
    ).sort_values(ascending=False)

    # 方法3: 相关系数
    corr_ranking = df[feature_cols + [target_col]].corr()[target_col].abs().sort_values(ascending=False)

    return {
        'mutual_info': mi_ranking,
        'lgb_importance': lgb_ranking,
        'correlation': corr_ranking
    }
```

## 9. 完整特征工程 Pipeline

```python
def full_feature_pipeline(df, target_col='sales'):
    """完整的时序特征工程 pipeline"""
    df = df.copy()

    # 1. 滞后特征
    df = create_lag_features(df, target_col, [1, 2, 3, 7, 14, 28])

    # 2. 差分特征
    df = create_diff_features(df, target_col, [1, 7])

    # 3. 滑动窗口
    df = create_rolling_features(df, target_col, [7, 14, 30])

    # 4. 时间编码
    df = encode_cyclic_time(df)

    # 5. 时间标记
    df = add_time_flags(df)

    # 6. 分解特征
    df = add_decomposition_features(df, target_col)

    # 7. 交叉特征
    df = create_interaction_features(df)

    # 8. 去除 NaN
    df = df.dropna()

    return df
```

---

**要点总结：**
- 滞后特征和滑动窗口统计是最有效的时序特征
- 正弦/余弦周期编码优于 one-hot，保留了时间的周期性
- STL 分解显式地提取趋势和季节性信号
- 特征工程应与模型选择配合：简单模型需要更丰富的特征，深度学习可以自动学习部分特征
