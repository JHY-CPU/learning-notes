# 16_概率预测与分位数回归

## 1. 点预测 vs 概率预测

**点预测**只给出一个预测值，无法量化不确定性。**概率预测**输出预测的分布，提供置信区间。

```
点预测:     ŷ = 100
概率预测:   P(y) = N(100, 15²)  →  95% 置信区间 [70, 130]
分位数预测: q_0.1=75, q_0.5=100, q_0.9=125
```

**不确定性的来源：**
- **数据噪声 (Aleatoric)**：观测本身的随机性，不可消除
- **模型不确定性 (Epistemic)**：参数估计误差，可通过更多数据减少
- **分布偏移 (Distributional Shift)**：测试分布与训练分布不同

## 2. 分位数回归

### 2.1 分位数损失

对于分位数 $\tau \in (0, 1)$，分位数损失（Pinball Loss）定义为：

$$L_\tau(y, \hat{y}) = \begin{cases} \tau \cdot (y - \hat{y}) & \text{if } y \geq \hat{y} \\ (1 - \tau) \cdot (\hat{y} - y) & \text{if } y < \hat{y} \end{cases}$$

统一形式：$L_\tau(y, \hat{y}) = (y - \hat{y}) \cdot (\tau - \mathbf{1}_{y < \hat{y}})$

```python
import torch
import torch.nn as nn

class QuantileLoss(nn.Module):
    """分位数损失（同时预测多个分位数）"""
    def __init__(self, quantiles=[0.1, 0.5, 0.9]):
        super().__init__()
        self.quantiles = quantiles

    def forward(self, predictions, targets):
        """
        predictions: (B, H, Q) 预测值，Q个分位数
        targets: (B, H) 真实值
        """
        targets = targets.unsqueeze(-1)  # (B, H, 1)
        errors = targets - predictions   # (B, H, Q)

        losses = []
        for i, q in enumerate(self.quantiles):
            e = errors[:, :, i]
            loss = torch.max(q * e, (q - 1) * e)
            losses.append(loss)

        return torch.stack(losses, dim=-1).mean()


# 不同分位数的损失函数不对称性
# τ=0.1: 低估的惩罚 > 高估的惩罚 → 倾向于预测偏高
# τ=0.5: 对称 → 等价于 MAE，中位数预测
# τ=0.9: 低估的惩罚 < 高估的惩罚 → 倾向于预测偏低
```

### 2.2 分位数回归网络

```python
class QuantileRegressor(nn.Module):
    """分位数回归时序预测器"""
    def __init__(self, input_size, hidden_size, seq_len, pred_len,
                 quantiles=[0.1, 0.25, 0.5, 0.75, 0.9]):
        super().__init__()
        self.quantiles = quantiles
        self.n_quantiles = len(quantiles)

        self.encoder = nn.LSTM(input_size, hidden_size, batch_first=True)
        self.fc = nn.Sequential(
            nn.Linear(hidden_size, hidden_size),
            nn.ReLU(),
            nn.Linear(hidden_size, pred_len * self.n_quantiles)
        )

    def forward(self, x):
        out, _ = self.encoder(x)
        last = out[:, -1, :]
        pred = self.fc(last)
        return pred.view(-1, pred.size(1) // self.n_quantiles, self.n_quantiles)

# 训练
model = QuantileRegressor(7, 64, 48, 24)
criterion = QuantileLoss(quantiles=[0.1, 0.25, 0.5, 0.75, 0.9])

x = torch.randn(32, 48, 7)
y = torch.randn(32, 24)
pred = model(x)  # (32, 24, 5)
loss = criterion(pred, y)
```

## 3. DeepAR 模型

DeepAR (Amazon, 2019) 是基于自回归 RNN 的概率预测模型：

$$P(y_{t, i} | \mathbf{y}_{<t, i}, \mathbf{x}_{t, i}) = \mathcal{N}(\mu_{t, i}, \sigma_{t, i}^2)$$

其中 $\mu_{t,i}$ 和 $\sigma_{t,i}$ 由神经网络输出：

$$[\mu_{t,i}, \sigma_{t,i}] = h(\mathbf{y}_{<t,i}, \mathbf{x}_{t,i}, \theta)$$

```python
class DeepAR(nn.Module):
    """DeepAR: 基于自回归 LSTM 的概率预测"""
    def __init__(self, input_size, hidden_size, embedding_size=16,
                 n_items=100):
        super().__init__()
        self.hidden_size = hidden_size

        # 物品嵌入（用于多时间序列共享模型）
        self.item_embedding = nn.Embedding(n_items, embedding_size)

        # LSTM 编码器
        self.lstm = nn.LSTM(input_size + embedding_size + 1,  # +1 for previous y
                            hidden_size, batch_first=True)

        # 输出分布参数
        self.mu_proj = nn.Linear(hidden_size, 1)
        self.sigma_proj = nn.Linear(hidden_size, 1)

    def gaussian_likelihood(self, mu, sigma, target):
        """高斯负对数似然损失"""
        return (torch.log(sigma) + 0.5 * ((target - mu) / sigma) ** 2).mean()

    def forward(self, x, item_ids, y_prev, target=None):
        """
        x: (B, T, input_size) 外部特征
        item_ids: (B,) 物品 ID
        y_prev: (B, T, 1) 历史目标值
        target: (B, H) 训练时的真实目标
        """
        batch_size = x.size(0)
        item_emb = self.item_embedding(item_ids).unsqueeze(1).expand(-1, x.size(1), -1)

        # 拼接输入
        lstm_input = torch.cat([x, item_emb, y_prev], dim=-1)

        # 编码
        h, c = self.init_hidden(batch_size, x.device)
        enc_out, (h, c) = self.lstm(lstm_input, (h, c))

        if target is not None:
            # 训练模式：自回归生成，使用 Teacher Forcing
            return self._train_step(enc_out, h, c, x, item_emb, y_prev, target)
        else:
            # 推理模式：自回归采样
            return self._sample(enc_out[:, -1:, :], h, c, x[:, -1:, :],
                               item_emb[:, -1:, :], y_prev[:, -1:, :])

    def _train_step(self, enc_out, h, c, x, item_emb, y_prev, target):
        horizon = target.size(1)
        mus, sigmas = [], []
        input_step = y_prev[:, -1:, :]  # 最后一个观测值

        for t in range(horizon):
            # 拼接当前输入
            feat = torch.cat([x[:, t:t+1, :] if t < x.size(1) else x[:, -1:, :],
                             item_emb[:, :1, :], input_step], dim=-1)
            out, (h, c) = self.lstm(feat, (h, c))

            mu = self.mu_proj(out)
            sigma = F.softplus(self.sigma_proj(out)) + 1e-6

            mus.append(mu)
            sigmas.append(sigma)

            # Teacher Forcing: 使用真实值
            input_step = target[:, t:t+1].unsqueeze(-1)

        mu = torch.cat(mus, dim=1).squeeze(-1)
        sigma = torch.cat(sigmas, dim=1).squeeze(-1)
        return mu, sigma

    @torch.no_grad()
    def _sample(self, last_out, h, c, last_x, item_emb, last_y, horizon=24,
                n_samples=100):
        """自回归采样生成概率预测"""
        all_samples = []
        for _ in range(n_samples):
            samples = []
            inp = last_y
            _h, _c = h.clone(), c.clone()

            for t in range(horizon):
                feat = torch.cat([last_x, item_emb, inp], dim=-1)
                out, (_h, _c) = self.lstm(feat, (_h, _c))

                mu = self.mu_proj(out).squeeze(-1)
                sigma = F.softplus(self.sigma_proj(out)).squeeze(-1) + 1e-6

                # 从高斯分布采样
                y_sample = mu + sigma * torch.randn_like(mu)
                samples.append(y_sample)
                inp = y_sample.unsqueeze(-1)

            all_samples.append(torch.stack(samples, dim=1))

        return torch.stack(all_samples, dim=0)  # (n_samples, B, H)

    def init_hidden(self, batch_size, device):
        h = torch.zeros(1, batch_size, self.hidden_size, device=device)
        c = torch.zeros(1, batch_size, self.hidden_size, device=device)
        return h, c
```

## 4. 概率预测评估指标

```python
import numpy as np

def quantile_loss(y_true, y_pred, quantile):
    """单个分位数的 Pinball Loss"""
    error = y_true - y_pred
    return np.mean(np.maximum(quantile * error, (quantile - 1) * error))

def crps(y_true, y_samples):
    """
    CRPS (Continuous Ranked Probability Score)
    评估整个预测分布的质量
    y_samples: (n_samples, batch, horizon)
    """
    n_samples = y_samples.shape[0]
    # 第一项: 预测与观测的平均绝对误差
    term1 = np.mean(np.abs(y_samples - y_true[np.newaxis, :, :]))
    # 第二项: 预测样本间的平均绝对差异
    term2 = 0
    for i in range(n_samples):
        for j in range(i + 1, n_samples):
            term2 += np.mean(np.abs(y_samples[i] - y_samples[j]))
    term2 = term2 * 2 / (n_samples * (n_samples - 1))

    return term1 - 0.5 * term2

def prediction_interval_coverage(y_true, lower, upper):
    """预测区间覆盖率 (PICP) 和平均区间宽度 (MPIW)"""
    coverage = np.mean((y_true >= lower) & (y_true <= upper))
    width = np.mean(upper - lower)
    return coverage, width

def winkler_score(y_true, lower, upper, alpha=0.1):
    """Winkler Score: 同时评估覆盖率和宽度"""
    width = upper - lower
    penalty_lower = np.where(y_true < lower, 2 / alpha * (lower - y_true), 0)
    penalty_upper = np.where(y_true > upper, 2 / alpha * (y_true - upper), 0)
    return np.mean(width + penalty_lower + penalty_upper)
```

## 5. 概率预测方法对比

| 方法 | 分布假设 | 不确定性来源 | 实现难度 | 效果 |
|------|---------|-------------|---------|------|
| 分位数回归 | 无 | 隐式 | 简单 | 好 |
| DeepAR | 高斯 | 显式 | 中等 | 好 |
| MC Dropout | 近似贝叶斯 | 模型 | 简单 | 中等 |
| 集成方法 | 多模型投票 | 模型 | 计算量大 | 最好 |
| Normalizing Flow | 复杂分布 | 显式 | 复杂 | 很好 |

## 6. 集成概率预测

```python
class EnsembleProbabilistic(nn.Module):
    """多模型集成的概率预测"""
    def __init__(self, models, quantiles=[0.1, 0.5, 0.9]):
        super().__init__()
        self.models = nn.ModuleList(models)
        self.quantiles = quantiles

    def forward(self, x):
        predictions = []
        for model in self.models:
            with torch.no_grad():
                pred = model(x)
                predictions.append(pred)

        # 合并所有模型的预测
        all_preds = torch.stack(predictions, dim=0)  # (n_models, B, H)

        # 从集成预测中计算分位数
        results = {}
        for q in self.quantiles:
            results[f'q{q}'] = torch.quantile(all_preds, q, dim=0)

        results['mean'] = all_preds.mean(dim=0)
        results['std'] = all_preds.std(dim=0)
        return results
```

---

**要点总结：**
- 分位数回归通过 Pinball Loss 直接预测条件分位数，无需分布假设
- DeepAR 基于自回归高斯似然建模，适合大量相关时间序列
- 概率预测评估使用 CRPS、Winkler Score 等指标，兼顾覆盖率和锐度
- 集成方法是获取高质量概率预测的有效策略
