# 15_多变量时间序列预测

## 1. 概述

多变量时间序列预测需要同时建模：
1. **时间维度**：每个变量自身的历史模式
2. **变量维度**：变量间的相互影响关系

$$\hat{Y}^{(j)}_{t+1:t+H} = f\left(\{X^{(1)}_{1:t}, \ldots, X^{(D)}_{1:t}\}\right), \quad j = 1, \ldots, D$$

## 2. 变量间关系建模

### 2.1 相关性分析

```python
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

def analyze_multivariate(series_df):
    """多变量时序关系分析"""
    # 1. 相关系数矩阵
    corr = series_df.corr()

    # 2. 滞后互相关（发现因果方向）
    max_lag = 48
    cross_corr = {}
    cols = series_df.columns
    for i in cols:
        for j in cols:
            if i != j:
                lcc = [series_df[i].corr(series_df[j].shift(lag))
                       for lag in range(max_lag)]
                cross_corr[(i, j)] = lcc

    # 3. Granger 因果检验
    from statsmodels.tsa.stattools import grangercausalitytests
    granger_results = {}
    for i in cols:
        for j in cols:
            if i != j:
                try:
                    result = grangercausalitytests(
                        series_df[[j, i]].dropna(), maxlag=12, verbose=False
                    )
                    p_values = [result[lag][0]['ssr_ftest'][1]
                               for lag in range(1, 13)]
                    granger_results[(i, j)] = min(p_values)
                except:
                    pass

    return corr, cross_corr, granger_results
```

### 2.2 注意力机制建模

```python
import torch
import torch.nn as nn

class InterVariableAttention(nn.Module):
    """变量间注意力模块"""
    def __init__(self, d_model, n_heads):
        super().__init__()
        self.n_heads = n_heads
        self.d_k = d_model // n_heads

        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)

    def forward(self, x):
        # x: (B, D, d_model) D个变量，每个变量有d_model维的表示
        B, D, _ = x.shape

        Q = self.W_q(x).view(B, D, self.n_heads, self.d_k).transpose(1, 2)
        K = self.W_k(x).view(B, D, self.n_heads, self.d_k).transpose(1, 2)
        V = self.W_v(x).view(B, D, self.n_heads, self.d_k).transpose(1, 2)

        scores = torch.matmul(Q, K.transpose(-2, -1)) / (self.d_k ** 0.5)
        attn = torch.softmax(scores, dim=-1)
        out = torch.matmul(attn, V)

        return out.transpose(1, 2).contiguous().view(B, D, -1), attn
```

## 3. 图结构时序建模

当变量间关系可以用图结构表达时，图神经网络可以显式编码这些关系：

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class GraphConvolution(nn.Module):
    """图卷积层：聚合邻居变量的信息"""
    def __init__(self, in_features, out_features):
        super().__init__()
        self.W = nn.Linear(in_features, out_features)

    def forward(self, X, adj):
        """
        X: (B, D, F) D个变量，F维特征
        adj: (D, D) 邻接矩阵（可以是学习得到的）
        """
        # 归一化邻接矩阵
        D_inv_sqrt = torch.diag(adj.sum(dim=1).pow(-0.5))
        adj_norm = D_inv_sqrt @ adj @ D_inv_sqrt

        # 图卷积: H' = σ(D^{-1/2} A D^{-1/2} X W)
        out = adj_norm @ X  # (B, D, F)
        return F.relu(self.W(out))


class SpatioTemporalGNN(nn.Module):
    """时空图神经网络"""
    def __init__(self, n_nodes, in_features, hidden_features,
                 seq_len, pred_len, n_gnn_layers=2):
        super().__init__()
        self.n_nodes = n_nodes
        self.pred_len = pred_len

        # 可学习的邻接矩阵（如果没有先验知识）
        self.adj = nn.Parameter(torch.randn(n_nodes, n_nodes) * 0.1)

        # 空间卷积（图卷积）
        self.gcn_layers = nn.ModuleList([
            GraphConvolution(in_features if i == 0 else hidden_features,
                             hidden_features)
            for i in range(n_gnn_layers)
        ])

        # 时间卷积（TCN 或 LSTM）
        self.temporal = nn.LSTM(hidden_features, hidden_features,
                                 batch_first=True)

        # 预测层
        self.fc = nn.Linear(hidden_features, pred_len)

    def forward(self, x):
        # x: (B, seq_len, n_nodes, in_features)
        B, T, D, F = x.shape

        # 学习的邻接矩阵（带自环、ReLU 截断负值）
        adj = F.relu(self.adj) + torch.eye(D, device=x.device)

        # 每个时间步做图卷积
        spatial_out = []
        for t in range(T):
            h = x[:, t, :, :]  # (B, D, F)
            for gcn in self.gcn_layers:
                h = gcn(h, adj)
            spatial_out.append(h)

        # (B, T, D, hidden)
        spatial_out = torch.stack(spatial_out, dim=1)

        # 每个节点独立做时间建模
        temporal_out = []
        for d in range(D):
            node_seq = spatial_out[:, :, d, :]  # (B, T, hidden)
            out, _ = self.temporal(node_seq)
            temporal_out.append(out[:, -1, :])  # 取最后时间步

        temporal_out = torch.stack(temporal_out, dim=1)  # (B, D, hidden)

        # 预测每个节点的未来值
        pred = self.fc(temporal_out)  # (B, D, pred_len)
        return pred.transpose(1, 2)   # (B, pred_len, D)
```

## 4. 多变量 Transformer

```python
class MultivariateTransformer(nn.Module):
    """多变量 Transformer：同时建模时间和变量维度"""
    def __init__(self, n_vars, seq_len, pred_len, d_model=128,
                 n_heads=4, n_layers=2, d_ff=256):
        super().__init__()
        self.n_vars = n_vars
        self.d_model = d_model

        # 嵌入：每个时间步的 D 个变量 -> d_model 维
        self.embedding = nn.Linear(n_vars, d_model)

        # 时间位置编码
        self.time_pos = nn.Parameter(torch.randn(1, seq_len, d_model) * 0.02)

        # 变量嵌入（可选）
        self.var_embed = nn.Parameter(torch.randn(1, n_vars, d_model) * 0.02)

        # 时间维度 Transformer
        self.temporal_encoder = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model, n_heads, d_ff,
                                        batch_first=True, activation='gelu'),
            n_layers
        )

        # 变量维度 Transformer
        self.variable_encoder = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model, n_heads, d_ff,
                                        batch_first=True, activation='gelu'),
            n_layers
        )

        self.fc = nn.Linear(d_model, pred_len)

    def forward(self, x):
        # x: (B, T, D)
        B, T, D = x.shape

        # 时间 Transformer
        time_emb = self.embedding(x) + self.time_pos[:, :T, :]  # (B, T, d)
        time_out = self.temporal_encoder(time_emb)  # (B, T, d)

        # 变量 Transformer：转置后处理
        var_input = time_out.transpose(1, 2)  # (B, D, d) 不对，应该是 (B, D, T) -> 嵌入
        # 更合理的方式：时间编码后聚合每个变量
        # 每个变量的时序表示: (B, D, d) 通过对时间维度取均值/最后一步
        var_input = time_out[:, -1:, :].expand(-1, D, -1) + self.var_embed
        var_out = self.variable_encoder(var_input)  # (B, D, d)

        # 预测
        pred = self.fc(var_out).transpose(1, 2)  # (B, pred_len, D)
        return pred
```

## 5. 多变量策略对比

| 策略 | 描述 | 优势 | 劣势 |
|------|------|------|------|
| 独立模型 | 每个变量单独建模 | 简单 | 忽略变量关系 |
| 多输出 | 所有变量同时预测 | 共享表示 | 参数量大 |
| 图结构 | 编码变量间图关系 | 结构先验 | 需要图定义 |
| 通道独立 | 共享参数独立处理 | 泛化好 | 忽略跨变量 |
| 通道混合 | Transformer 建模变量 | 捕捉关系 | 噪声干扰 |
