# 17_时序预测实战框架

## 1. 主流框架概览

| 框架 | 维护方 | 特点 | 模型覆盖 |
|------|--------|------|---------|
| **GluonTS** | Amazon | 概率预测为主，MXNet/PyTorch | DeepAR, Transformer, TFT |
| **Darts** | Unit8 | 统一 API，统计+深度 | ARIMA 到 Transformer 全覆盖 |
| **PyTorch Forecasting** | JDB | PyTorch 生态，强调 TFT | TFT, N-BEATS, DeepAR |
| **tsai** | timeseriesAI | fastai 风格，分类/回归/预测 | InceptionTime, Transformer |
| **NeuralForecast** | Nixtla | 生产级，支持大规模 | NHITS, NBEATS, TFT, PatchTST |

## 2. GluonTS

```python
# pip install gluonts[torch]

from gluonts.dataset.pandas import PandasDataset
from gluonts.torch.model.deepar import DeepAREstimator
from gluonts.evaluation.backtest import make_evaluation_predictions

import pandas as pd
import numpy as np

# 准备数据
df = pd.DataFrame({
    'target': np.random.randn(500).cumsum(),
    'item_id': 'series_0'
}, index=pd.date_range('2020-01-01', periods=500, freq='D'))

dataset = PandasDataset.from_long_dataframe(
    df, target='target', item_id='item_id', freq='D'
)

# 训练模型
estimator = DeepAREstimator(
    prediction_length=30,
    context_length=60,
    freq='D',
    trainer_kwargs={'max_epochs': 50}
)
predictor = estimator.train(dataset)

# 预测
forecast_it, ts_it = make_evaluation_predictions(
    dataset=dataset, predictor=predictor, num_samples=100
)
forecasts = list(forecast_it)

# 提取分位数预测
median = forecasts[0].median
q10 = forecasts[0].quantile(0.1)
q90 = forecasts[0].quantile(0.9)
```

## 3. Darts

```python
# pip install darts

from darts import TimeSeries
from darts.models import (
    NBEATSModel, TFTModel, TCNModel,
    RNNModel, TransformerModel
)
from darts.dataprocessing.transformers import Scaler
from darts.metrics import mape, rmse, r2_score

# 数据准备
series = TimeSeries.from_dataframe(
    df, time_col='date', value_cols=['value'],
    freq='D'
)

# 划分
train, val = series.split_before(0.8)

# 归一化
scaler = Scaler()
train_scaled = scaler.fit_transform(train)
val_scaled = scaler.transform(val)

# N-BEATS 模型
model = NBEATSModel(
    input_chunk_length=60,
    output_chunk_length=30,
    n_epochs=100,
    random_state=42,
    batch_size=32,
    force_reset=True
)

model.fit(train_scaled, val_series=val_scaled)

# 预测
pred = model.predict(n=30, series=train_scaled)
pred = scaler.inverse_transform(pred)

# 评估
print(f'MAPE: {mape(val, pred):.2f}%')
print(f'RMSE: {rmse(val, pred):.4f}')

# 概率预测
pred_prob = model.predict(n=30, series=train_scaled, num_samples=500)
# 可视置信区间
pred_prob.plot(low_quantile=0.1, high_quantile=0.9)
```

## 4. PyTorch Forecasting

```python
# pip install pytorch-forecasting pytorch-lightning

import pytorch_forecasting
from pytorch_forecasting import TimeSeriesDataSet, TemporalFusionTransformer
from pytorch_forecasting.data import GroupNormalizer
import lightning.pytorch as pl

# 准备 DataFrame（需要特定格式）
df['time_idx'] = range(len(df))
df['group'] = 'series_0'

# 创建数据集
training = TimeSeriesDataSet(
    df[lambda x: x.time_idx <= training_cutoff],
    time_idx='time_idx',
    target='value',
    group_ids=['group'],
    max_encoder_length=60,
    max_prediction_length=12,
    time_varying_known_reals=['time_idx'],
    time_varying_unknown_reals=['value'],
    target_normalizer=GroupNormalizer(groups=['group'])
)

# 创建验证集
validation = TimeSeriesDataSet.from_dataset(training, df)

# DataLoaders
train_dataloader = training.to_dataloader(train=True, batch_size=64)
val_dataloader = validation.to_dataloader(train=False, batch_size=64)

# TFT 模型
tft = TemporalFusionTransformer.from_dataset(
    training,
    hidden_size=16,
    attention_head_size=4,
    dropout=0.1,
    hidden_continuous_size=8,
    output_size=7,  # 7 个分位数
    loss=pytorch_forecasting.QuantileLoss(),
    learning_rate=0.03,
    reduce_on_plateau_patience=4,
)

# 训练
trainer = pl.Trainer(max_epochs=50, gradient_clip_val=0.1)
trainer.fit(tft, train_dataloader, val_dataloader)

# 预测
predictions = tft.predict(val_dataloader)

# 变量重要性
interpretation = tft.interpret_output(predictions)
tft.plot_interpretation(interpretation)
```

## 6. 框架选择指南

```
决策流程:

你是否需要概率预测？
├── 是 → GluonTS 或 Darts
└── 否 → 你是否需要大量模型快速实验？
         ├── 是 → Darts（统一 API）
         └── 否 → 你是否需要生产级部署？
                  ├── 是 → NeuralForecast
                  └── 否 → PyTorch Forecasting（深度定制）
```

| 需求 | 推荐框架 | 原因 |
|------|---------|------|
| 快速实验多种模型 | Darts | 统一 API，切换方便 |
| 概率预测+多时间序列 | GluonTS | Amazon 生产验证 |
| TFT 可解释性 | PyTorch Forecasting | TFT 最佳实现 |
| 大规模生产部署 | NeuralForecast | Nixtla 团队维护 |
| 学术研究 | tsai + 自定义 | fastai 生态 |

## 7. 完整实战流程

```python
# 以 Darts 为例的完整流程
from darts import TimeSeries
from darts.models import NBEATSModel, TFTModel
from darts.dataprocessing.transformers import Scaler
from darts.metrics import mape, rmse
import warnings
warnings.filterwarnings('ignore')

def time_series_pipeline(df, target_col, covariate_cols=None,
                         forecast_horizon=24, val_ratio=0.2):
    """完整时序预测 Pipeline"""

    # 1. 数据准备
    series = TimeSeries.from_dataframe(df, value_cols=[target_col])
    if covariate_cols:
        covariates = TimeSeries.from_dataframe(df, value_cols=covariate_cols)
    else:
        covariates = None

    split_idx = int(len(series) * (1 - val_ratio))
    train, val = series[:split_idx], series[split_idx:]

    # 2. 归一化
    scaler = Scaler()
    train_scaled = scaler.fit_transform(train)
    val_scaled = scaler.transform(val)

    # 3. 模型选择与训练
    models = {
        'NBEATS': NBEATSModel(
            input_chunk_length=forecast_horizon * 3,
            output_chunk_length=forecast_horizon,
            n_epochs=50, batch_size=32
        ),
    }

    results = {}
    for name, model in models.items():
        model.fit(train_scaled, val_series=val_scaled)
        pred = model.predict(n=forecast_horizon, series=train_scaled)
        pred = scaler.inverse_transform(pred)
        results[name] = {
            'MAPE': mape(val, pred),
            'RMSE': rmse(val, pred),
            'predictions': pred
        }
        print(f'{name}: MAPE={results[name]["MAPE"]:.2f}%, '
              f'RMSE={results[name]["RMSE"]:.4f}')

    return results

# 使用
results = time_series_pipeline(df, 'sales', forecast_horizon=48)
```

---

**要点总结：**
- Darts 适合快速实验，GluonTS 适合概率预测，NeuralForecast 适合生产部署
- 数据格式统一是框架使用的关键：每个框架有自己的数据格式要求
- 建议先用简单模型建立 baseline，再尝试复杂模型
- 框架简化了训练流程，但理解底层原理仍很重要
