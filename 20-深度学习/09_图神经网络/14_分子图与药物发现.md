# 14_分子图与药物发现

## 1. 分子图表示

分子天然具有图结构：
- **节点** = 原子：碳(C)、氮(N)、氧(O)、硫(S)等
- **边** = 化学键：单键、双键、三键、芳香键

### 1.1 原子特征

常用的原子特征（RDKit 提供）：

| 特征 | 维度 | 示例 |
|------|------|------|
| 原子类型 (One-hot) | 10 | C, N, O, F, P, S, Cl, Br, I, 其他 |
| 度数 | 6 | 0, 1, 2, 3, 4, 5 |
| 形式电荷 | 5 | -2, -1, 0, 1, 2 |
| 杂化类型 | 5 | SP, SP2, SP3, SP3D, SP3D2 |
| 是否芳香 | 1 | 0/1 |
| 氢原子数 | 5 | 0, 1, 2, 3, 4 |

```python
import torch
from rdkit import Chem
import numpy as np

def get_atom_features(atom):
    """提取原子特征向量"""
    # 原子类型
    atom_types = ['C', 'N', 'O', 'F', 'P', 'S', 'Cl', 'Br', 'I']
    atom_type = [int(atom.GetSymbol() == t) for t in atom_types]
    atom_type.append(int(atom.GetSymbol() not in atom_types))  # 其他

    # 度数
    degree = [int(atom.GetDegree() == d) for d in range(6)]
    # 形式电荷
    charge = [int(atom.GetFormalCharge() == c) for c in [-2, -1, 0, 1, 2]]
    # 杂化
    hybrid = [int(atom.GetHybridization() == h)
              for h in [Chem.HybridizationType.SP,
                        Chem.HybridizationType.SP2,
                        Chem.HybridizationType.SP3,
                        Chem.HybridizationType.SP3D,
                        Chem.HybridizationType.SP3D2]]
    # 芳香性
    aromatic = [int(atom.GetIsAromatic())]
    # 氢原子数
    num_hs = [int(atom.GetTotalNumHs() == n) for n in range(5)]

    return atom_type + degree + charge + hybrid + aromatic + num_hs
```

### 1.2 化学键特征

| 特征 | 维度 | 说明 |
|------|------|------|
| 键类型 | 4 | 单键、双键、三键、芳香键 |
| 共轭 | 1 | 是否共轭 |
| 环中 | 1 | 是否在环中 |

```python
def get_bond_features(bond):
    """提取化学键特征"""
    bond_type = [
        int(bond.GetBondType() == Chem.BondType.SINGLE),
        int(bond.GetBondType() == Chem.BondType.DOUBLE),
        int(bond.GetBondType() == Chem.BondType.TRIPLE),
        int(bond.GetBondType() == Chem.BondType.AROMATIC),
    ]
    conjugated = [int(bond.GetIsConjugated())]
    in_ring = [int(bond.IsInRing())]
    return bond_type + conjugated + in_ring
```

## 2. 分子图构建

```python
from torch_geometric.data import Data

def smiles_to_graph(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None

    # 节点特征
    x = torch.tensor([get_atom_features(a) for a in mol.GetAtoms()],
                     dtype=torch.float)

    # 边索引和边特征
    edge_index = []
    edge_attr = []
    for bond in mol.GetBonds():
        i, j = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
        feat = get_bond_features(bond)
        # 双向边
        edge_index.extend([[i, j], [j, i]])
        edge_attr.extend([feat, feat])

    edge_index = torch.tensor(edge_index, dtype=torch.long).t().contiguous()
    edge_attr = torch.tensor(edge_attr, dtype=torch.float)
    return Data(x=x, edge_index=edge_index, edge_attr=edge_attr)
```

## 3. 药物发现中的 GNN 任务

### 3.1 药物属性预测

```python
import torch.nn as nn
import torch.nn.functional as F
from torch_geometric.nn import GINEConv, global_mean_pool, global_max_pool

class DrugPropertyPredictor(nn.Module):
    """分子属性预测模型（多任务）"""
    def __init__(self, node_dim, edge_dim, hidden_dim, num_tasks, num_layers=5):
        super().__init__()
        self.node_enc = nn.Linear(node_dim, hidden_dim)
        self.edge_enc = nn.Linear(edge_dim, hidden_dim)

        self.convs = nn.ModuleList()
        self.bns = nn.ModuleList()
        for _ in range(num_layers):
            mlp = nn.Sequential(
                nn.Linear(hidden_dim, hidden_dim * 2),
                nn.ReLU(),
                nn.Linear(hidden_dim * 2, hidden_dim),
            )
            self.convs.append(GINEConv(mlp, edge_dim=hidden_dim))
            self.bns.append(nn.BatchNorm1d(hidden_dim))

        self.lin1 = nn.Linear(hidden_dim * 2, hidden_dim)
        self.lin2 = nn.Linear(hidden_dim, num_tasks)

    def forward(self, data):
        x = self.node_enc(data.x)
        edge_attr = self.edge_enc(data.edge_attr)

        for conv, bn in zip(self.convs, self.bns):
            x = F.relu(bn(conv(x, data.edge_index, edge_attr)))

        h = torch.cat([global_mean_pool(x, data.batch),
                       global_max_pool(x, data.batch)], dim=-1)
        h = F.relu(self.lin1(h))
        return self.lin2(h)  # 多任务：输出 num_tasks 维
```

### 3.2 药物-靶点相互作用预测

```python
class DrugTargetInteraction(nn.Module):
    """预测药物分子与靶蛋白的结合亲和力"""
    def __init__(self, drug_dim, protein_dim, hidden_dim):
        super().__init__()
        # 药物编码器（GNN）
        self.drug_conv1 = GCNConv(drug_dim, hidden_dim)
        self.drug_conv2 = GCNConv(hidden_dim, hidden_dim)

        # 蛋白质编码器（1D CNN）
        self.protein_conv = nn.Sequential(
            nn.Conv1d(protein_dim, hidden_dim, 8),
            nn.ReLU(),
            nn.AdaptiveMaxPool1d(1),
        )

        # 预测层
        self.predictor = nn.Sequential(
            nn.Linear(hidden_dim * 3, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1),
        )

    def forward(self, drug_data, protein_seq):
        # 药物图编码
        drug_x = F.relu(self.drug_conv1(drug_data.x, drug_data.edge_index))
        drug_x = self.drug_conv2(drug_x, drug_data.edge_index)
        drug_h = global_mean_pool(drug_x, drug_data.batch)

        # 蛋白质序列编码
        prot_h = self.protein_conv(protein_seq).squeeze(-1)

        # 交互预测
        interaction = torch.cat([drug_h, prot_h, drug_h * prot_h], dim=-1)
        return self.predictor(interaction)
```

### 4.1 基于图的分子生成

```python
class MolecularGraphVAE(nn.Module):
    """图变分自编码器用于分子生成"""
    def __init__(self, node_dim, hidden_dim, latent_dim, max_nodes=38):
        super().__init__()
        self.max_nodes = max_nodes
        # 编码器
        self.encoder = nn.Sequential(
            GCNConv(node_dim, hidden_dim),
        )
        self.fc_mu = nn.Linear(hidden_dim, latent_dim)
        self.fc_logvar = nn.Linear(hidden_dim, latent_dim)

        # 解码器：生成节点特征和邻接矩阵
        self.node_decoder = nn.Linear(latent_dim, max_nodes * node_dim)
        self.edge_decoder = nn.Sequential(
            nn.Linear(latent_dim * 2, 1),
            nn.Sigmoid(),
        )

    def encode(self, data):
        x = F.relu(self.encoder[0](data.x, data.edge_index))
        h = global_mean_pool(x, data.batch)
        return self.fc_mu(h), self.fc_logvar(h)

    def reparameterize(self, mu, logvar):
        std = torch.exp(0.5 * logvar)
        eps = torch.randn_like(std)
        return mu + eps * std

    def forward(self, data):
        mu, logvar = self.encode(data)
        z = self.reparameterize(mu, logvar)
        # 解码（简化版）
        node_logits = self.node_decoder(z).view(-1, self.max_nodes, data.x.size(-1))
        return node_logits, mu, logvar
```

## 6. 分子指纹与 GNN 的结合

传统分子指纹（ECFP）和 GNN 表示可以互补：

```python
from rdkit.Chem import AllChem
import numpy as np

def get_ecfp(smiles, radius=2, nbits=2048):
    mol = Chem.MolFromSmiles(smiles)
    fp = AllChem.GetMorganFingerprintAsBitVect(mol, radius, nBits=nbits)
    return np.array(fp)

class HybridDrugModel(nn.Module):
    """结合分子指纹和 GNN"""
    def __init__(self, gnn_dim, fp_dim, hidden_dim, out_dim):
        super().__init__()
        self.gnn = DrugPropertyPredictor(gnn_dim, 6, hidden_dim, hidden_dim, 3)
        self.fp_encoder = nn.Sequential(
            nn.Linear(fp_dim, hidden_dim),
            nn.ReLU(),
        )
        self.predictor = nn.Linear(hidden_dim * 2, out_dim)

    def forward(self, data, fps):
        gnn_h = self.gnn(data)
        fp_h = self.fp_encoder(fps)
        return self.predictor(torch.cat([gnn_h, fp_h], dim=-1))
```

## 7. 小结

- 分子图表示：原子为节点（含丰富特征），化学键为边
- 药物发现中的 GNN 应用：属性预测、药物-靶点相互作用、分子生成
- 多任务学习很常见（同时预测多个药物属性）
- 注意处理缺失标签（NaN）
- 分子指纹和 GNN 可以结合使用，互相补充
