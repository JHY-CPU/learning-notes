# 15_推理性能评估指标

## 1. 核心指标定义

```
LLM 推理性能指标:

1. TTFT (Time To First Token)
   从收到请求到返回第一个 token 的时间
   = Prefill 延迟 + 排队延迟

2. TPOT (Time Per Output Token)
   每个输出 token 的平均生成时间
   = Decode 阶段每 token 延迟

3. ITL (Inter-Token Latency)
   相邻 token 之间的延迟 (流式输出中的抖动)

4. Throughput (吞吐量)
   单位时间处理的 token 数
   = 生成 token 总数 / 总时间

5. Goodput (有效吞吐量)
   满足 SLA 的请求的吞吐量
   = 满足延迟要求的 token 数 / 总时间
```

## 2. 延迟分解

```python
class LatencyAnalyzer:
    """推理延迟分析器"""

    def analyze_request(self, request_trace: dict) -> dict:
        return {
            # 总延迟
            "total_latency": request_trace["end_time"] - request_trace["start_time"],

            # 排队延迟
            "queue_time": request_trace["inference_start"] - request_trace["start_time"],

            # Prefill 延迟
            "prefill_time": request_trace["first_token_time"] - request_trace["inference_start"],

            # Decode 总延迟
            "decode_time": request_trace["end_time"] - request_trace["first_token_time"],

            # 派生指标
            "ttft": request_trace["first_token_time"] - request_trace["start_time"],
            "tpot": (request_trace["end_time"] - request_trace["first_token_time"])
                    / request_trace.get("num_output_tokens", 1),
            "tokens_per_second": request_trace.get("num_output_tokens", 0)
                    / max(request_trace["end_time"] - request_trace["first_token_time"], 0.001),
        }

    def aggregate(self, traces: list) -> dict:
        """聚合分析"""
        results = [self.analyze_request(t) for t in traces]

        return {
            "avg_ttft": sum(r["ttft"] for r in results) / len(results),
            "p50_ttft": sorted(r["ttft"] for r in results)[len(results)//2],
            "p99_ttft": sorted(r["ttft"] for r in results)[int(len(results)*0.99)],
            "avg_tpot": sum(r["tpot"] for r in results) / len(results),
            "p50_tpot": sorted(r["tpot"] for r in results)[len(results)//2],
            "p99_tpot": sorted(r["tpot"] for r in results)[int(len(results)*0.99)],
            "avg_throughput": sum(r["tokens_per_second"] for r in results) / len(results),
            "total_tokens": sum(t.get("num_output_tokens", 0) for t in traces),
        }
```

## 3. Benchmark 工具

```python
import asyncio
import aiohttp
import time
from dataclasses import dataclass

@dataclass
class BenchmarkResult:
    total_requests: int
    completed_requests: int
    total_tokens: int
    total_time: float
    avg_ttft: float
    p50_ttft: float
    p99_ttft: float
    avg_tpot: float
    p50_tpot: float
    throughput: float
    goodput: float  # 满足 SLA 的吞吐量

class LLMBenchmark:
    """LLM 推理基准测试"""

    def __init__(self, api_url: str, model: str):
        self.api_url = api_url
        self.model = model

    async def single_request(self, prompt: str, max_tokens: int = 256) -> dict:
        """单次请求"""
        start_time = time.time()
        first_token_time = None
        tokens = []

        async with aiohttp.ClientSession() as session:
            async with session.post(
                f"{self.api_url}/v1/chat/completions",
                json={
                    "model": self.model,
                    "messages": [{"role": "user", "content": prompt}],
                    "max_tokens": max_tokens,
                    "stream": True
                }
            ) as response:
                async for line in response.content:
                    if line.startswith(b"data: ") and line != b"data: [DONE]\n":
                        if first_token_time is None:
                            first_token_time = time.time()
                        tokens.append(line)

        end_time = time.time()
        return {
            "start_time": start_time,
            "first_token_time": first_token_time or end_time,
            "end_time": end_time,
            "num_output_tokens": len(tokens),
            "success": True
        }

    async def run_concurrent(self, prompts: list, concurrency: int = 10,
                              max_tokens: int = 256) -> BenchmarkResult:
        """并发基准测试"""
        import random
        sem = asyncio.Semaphore(concurrency)
        results = []

        async def bounded_request(prompt):
            async with sem:
                result = await self.single_request(prompt, max_tokens)
                results.append(result)

        start = time.time()
        await asyncio.gather(*[bounded_request(p) for p in prompts])
        total_time = time.time() - start

        analyzer = LatencyAnalyzer()
        agg = analyzer.aggregate([r for r in results if r.get("success")])

        return BenchmarkResult(
            total_requests=len(prompts),
            completed_requests=len([r for r in results if r.get("success")]),
            total_tokens=agg["total_tokens"],
            total_time=total_time,
            avg_ttft=agg["avg_ttft"],
            p50_ttft=agg["p50_ttft"],
            p99_ttft=agg["p99_ttft"],
            avg_tpot=agg["avg_tpot"],
            p50_tpot=agg["p50_tpot"],
            throughput=agg["total_tokens"] / total_time,
            goodput=0  # 需要根据 SLA 计算
        )
```

## 4. 不同场景的性能目标

```
性能目标参考:

实时对话场景:
  TTFT < 500ms
  TPOT < 50ms  (>20 tok/s)
  用户体验: 流畅

代码生成场景:
  TTFT < 1000ms
  TPOT < 30ms  (>30 tok/s)
  用户体验: 快速

批量处理场景:
  TTFT: 不关键
  TPOT: 不关键
  吞吐量 > 1000 tok/s (最重要)

交互式应用:
  TTFT < 200ms (关键!)
  TPOT < 80ms
  用户体验: 即时响应
```

## 5. 性能分析方法

```python
class PerformanceProfiler:
    """性能分析器"""

    def profile_model(self, model, input_lengths, output_lengths):
        """全面性能分析"""
        results = {}

        for in_len in input_lengths:
            for out_len in output_lengths:
                key = f"in{in_len}_out{out_len}"
                result = self.run_single(in_len, out_len)
                results[key] = result

        return results

    def roofline_analysis(self, model, hardware):
        """Roofline 模型分析"""
        # 计算 Prefill 的算术强度
        prefill_ai = model.prefill_flops() / model.prefill_memory_access()
        # 计算 Decode 的算术强度
        decode_ai = model.decode_flops() / model.decode_memory_access()

        gpu_flops = hardware.peak_flops
        gpu_bandwidth = hardware.peak_bandwidth

        # 判断瓶颈
        prefill_bound = "compute" if prefill_ai > gpu_flops / gpu_bandwidth else "memory"
        decode_bound = "compute" if decode_ai > gpu_flops / gpu_bandwidth else "memory"

        return {
            "prefill": {"arithmetic_intensity": prefill_ai, "bound": prefill_bound},
            "decode": {"arithmetic_intensity": decode_ai, "bound": decode_bound},
        }
```

## 总结

推理性能评估需要**多维度指标**：TTFT 反映响应速度，TPOT 反映生成速度，吞吐量反映服务能力。不同场景对指标的侧重不同 -- 实时对话重 TTFT，批量处理重吞吐量。通过 Roofline 分析可以判断瓶颈在计算还是内存，指导优化方向。
