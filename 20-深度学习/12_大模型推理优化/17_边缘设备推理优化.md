## 2. 模型压缩技术

### 2.1 知识蒸馏

```python
class DistillationTrainer:
    """知识蒸馏：用大模型训练小模型"""

    def __init__(self, teacher_model, student_model, temperature=4.0, alpha=0.7):
        self.teacher = teacher_model  # 大模型
        self.student = student_model  # 小模型
        self.temperature = temperature
        self.alpha = alpha  # 蒸馏损失权重

    def distillation_loss(self, student_logits, teacher_logits, labels):
        """
        蒸馏损失 = α × KL(soft_student, soft_teacher) + (1-α) × CE(student, labels)
        """
        # 软标签蒸馏
        soft_student = F.log_softmax(student_logits / self.temperature, dim=-1)
        soft_teacher = F.softmax(teacher_logits / self.temperature, dim=-1)
        distill_loss = F.kl_div(soft_student, soft_teacher, reduction="batchmean")
        distill_loss *= self.temperature ** 2

        # 硬标签损失
        hard_loss = F.cross_entropy(student_logits, labels)

        return self.alpha * distill_loss + (1 - self.alpha) * hard_loss

    def train(self, train_data, epochs=10):
        """蒸馏训练"""
        self.teacher.eval()

        for epoch in range(epochs):
            for batch in train_data:
                with torch.no_grad():
                    teacher_logits = self.teacher(batch["input"])

                student_logits = self.student(batch["input"])
                loss = self.distillation_loss(
                    student_logits, teacher_logits, batch["labels"]
                )

                loss.backward()
                self.optimizer.step()
                self.optimizer.zero_grad()
```

### 2.2 剪枝

class StructuredPruning:
    """结构化剪枝：移除整个注意力头或 FFN 神经元"""

    def prune_model(self, model, pruning_ratio=0.3):
        """按重要性剪枝"""
        for name, module in model.named_modules():
            if isinstance(module, nn.Linear):
                # 计算每个输出神经元的重要性
                importance = module.weight.abs().mean(dim=1)

                # 确定保留的索引
                n_keep = int(module.weight.shape[0] * (1 - pruning_ratio))
                _, keep_indices = torch.topk(importance, n_keep)

                # 创建新层
                new_weight = module.weight[keep_indices]
                if module.bias is not None:
                    new_bias = module.bias[keep_indices]

                # 替换
                module.weight = nn.Parameter(new_weight)
                if module.bias is not None:
                    module.bias = nn.Parameter(new_bias)

        return model

class AttentionHeadPruning:
    """注意力头剪枝"""

    def __init__(self, model, pruning_data):
        self.model = model
        self.data = pruning_data

    def rank_head_importance(self) -> list:
        """评估每个注意力头的重要性"""
        head_scores = {}

        for layer_idx, layer in enumerate(self.model.layers):
            attn = layer.self_attn
            num_heads = attn.num_heads

            # 用 Taylor 展开近似重要性
            for head_idx in range(num_heads):
                score = self.compute_head_importance(layer_idx, head_idx)
                head_scores[(layer_idx, head_idx)] = score

        # 排序
        return sorted(head_scores.items(), key=lambda x: x[1])

## 4. 量化到极致

```python
"""
边缘设备量化策略:

1. INT4 权重 + FP16 激活 (W4A16)
   最平衡的方案，质量损失小

2. INT4 权重 + INT8 激活 (W4A8)
   更快但需要校准

3. INT2 权重 (W2A16)
   极致压缩，质量损失大
   适用于 <1B 参数模型

4. 混合精度
   关键层用高精度，其他层用低精度
   - 注意力层: INT8
   - FFN 层: INT4
   - Embedding: INT8
"""

class MixedPrecisionQuantizer:
    """混合精度量化"""

    def __init__(self, model):
        self.model = model
        self.precision_map = {}

    def assign_precision(self):
        """为每层分配精度"""
        for name, module in self.model.named_modules():
            if "attention" in name.lower():
                self.precision_map[name] = "int8"
            elif "ffn" in name.lower() or "mlp" in name.lower():
                self.precision_map[name] = "int4"
            elif "embed" in name.lower():
                self.precision_map[name] = "int8"
            elif "lm_head" in name.lower():
                self.precision_map[name] = "fp16"  # 输出层保持高精度
            else:
                self.precision_map[name] = "int8"
```

## 总结

边缘设备推理需要在**模型大小、推理速度和质量之间精细平衡**。核心技术包括知识蒸馏、极端量化、剪枝和硬件特定优化。目前 3-7B 参数的模型配合 INT4 量化可以在手机上流畅运行，为隐私敏感和离线场景提供了可行方案。
