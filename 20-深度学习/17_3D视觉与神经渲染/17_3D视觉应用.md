## 2. 自动驾驶中的点云处理

### 2.1 3D物体检测流水线

```
激光雷达点云 (64线, ~100K点/帧)
    ↓
预处理 (降噪、地面分割)
    ↓
体素化 / Pillar化
    ↓
3D骨干网络 (VoxelNet/PointPillars)
    ↓
3D边界框预测 (x, y, z, w, h, l, θ, 类别, 置信度)
    ↓
NMS后处理
    ↓
跟踪 + 预测
```

### 2.2 PointPillars

**PointPillars (Lang et al., CVPR 2019)** 将点云转为2D伪图像，使用2D CNN处理：

```python
class PointPillars(nn.Module):
    """PointPillars 简化实现"""
    def __init__(self, num_classes=3, num_pillars=12000, max_points=100):
        super().__init__()
        # Pillar特征编码
        self.pillar_encoder = nn.Sequential(
            nn.Linear(9, 64),  # 输入: (x,y,z,反射率, xc,yc,zc, xp,yp)
            nn.BatchNorm1d(64), nn.ReLU(),
            nn.Linear(64, 64), nn.BatchNorm1d(64), nn.ReLU(),
        )
        
        # Scatter到伪图像
        self.scatter = PointPillarScatter(output_shape=(496, 432))
        
        # 2D Backbone
        self.backbone = nn.Sequential(
            nn.Conv2d(64, 64, 3, stride=2, padding=1),
            nn.BatchNorm2d(64), nn.ReLU(),
            nn.Conv2d(64, 128, 3, stride=2, padding=1),
            nn.BatchNorm2d(128), nn.ReLU(),
            nn.Conv2d(128, 256, 3, stride=2, padding=1),
            nn.BatchNorm2d(256), nn.ReLU(),
        )
        
        # 检测头
        self.cls_head = nn.Conv2d(256, num_classes * 2, 1)  # 2个anchor
        self.reg_head = nn.Conv2d(256, 14 * 2, 1)  # 7个回归量 × 2个方向
    
    def forward(self, points):
        # 1. Pillar化
        pillars, indices = create_pillars(points, self.max_points)  # (P, max_points, 9)
        
        # 2. 特征编码
        pillar_feat = self.pillar_encoder(pillars)  # (P, max_points, 64)
        pillar_feat = pillar_feat.max(dim=1)[0]     # (P, 64)
        
        # 3. Scatter到伪图像
        pseudo_image = self.scatter(pillar_feat, indices)  # (1, 64, 496, 432)
        
        # 4. 2D Backbone
        feat = self.backbone(pseudo_image)  # (1, 256, 62, 54)
        
        # 5. 检测头
        cls = self.cls_head(feat)   # (1, 6, 62, 54)
        reg = self.reg_head(feat)   # (1, 28, 62, 54)
        
        return cls, reg
```

### 2.3 BEV感知

**BEV (Bird's Eye View)** 将多传感器数据统一到鸟瞰图空间：

```python
class BEVFusion(nn.Module):
    """多传感器BEV融合"""
    def __init__(self):
        super().__init__()
        # 相机分支：LSS式深度估计 → BEV
        self.camera_encoder = CameraToBEV()
        
        # 激光雷达分支：体素化 → BEV
        self.lidar_encoder = VoxelToBEV()
        
        # BEV融合
        self.fusion = nn.Sequential(
            nn.Conv2d(256 + 256, 256, 3, padding=1),
            nn.BatchNorm2d(256), nn.ReLU(),
            nn.Conv2d(256, 256, 3, padding=1),
            nn.BatchNorm2d(256), nn.ReLU(),
        )
        
        self.detect_head = DetectionHead(256)
    
    def forward(self, images, lidar_points, camera_poses, lidar_pose):
        # 相机 → BEV
        cam_bev = self.camera_encoder(images, camera_poses)
        
        # 激光雷达 → BEV
        lidar_bev = self.lidar_encoder(lidar_points, lidar_pose)
        
        # 融合
        fused_bev = self.fusion(torch.cat([cam_bev, lidar_bev], dim=1))
        
        return self.detect_head(fused_bev)
```

## 3. 机器人抓取

### 3.1 6DoF抓取位姿估计

6自由度抓取需要估计抓取点的3D位置和3D方向：

$$g = (p_x, p_y, p_z, \alpha, \beta, \gamma, w)$$

其中 $(p_x, p_y, p_z)$ 是抓取点，$(\alpha, \beta, \gamma)$ 是方向，$w$ 是抓取宽度。

### 3.2 基于点云的抓取

```python
class GraspNet(nn.Module):
    """6DoF抓取位姿预测"""
    def __init__(self, num_view=300):
        super().__init__()
        self.backbone = PointNetPlusPlus(in_channel=3)
        
        # 抓取质量预测
        self.quality_head = nn.Sequential(
            nn.Linear(1024, 256), nn.ReLU(),
            nn.Linear(256, 1), nn.Sigmoid()
        )
        
        # 抓取方向 (球坐标)
        self.view_head = nn.Linear(1024, num_view * 2)  # 角度和距离
        
        # 抓取宽度
        self.width_head = nn.Linear(1024, 1)
    
    def forward(self, point_cloud):
        """point_cloud: (B, N, 3)"""
        # 特征提取
        global_feat = self.backbone(point_cloud)  # (B, 1024)
        
        # 预测
        quality = self.quality_head(global_feat)    # (B, 1) 抓取成功率
        views = self.view_head(global_feat)         # (B, num_view*2)
        width = torch.sigmoid(self.width_head(global_feat)) * 0.1  # (B, 1)
        
        return quality, views, width
```

## 4. AR/VR

### 4.1 头部追踪与定位

```python
class VisualInertialOdometry:
    """视觉惯性里程计（AR基础）"""
    def __init__(self):
        self.imu = IMUProcessor()
        self.visual = VisualTracker()
    
    def update(self, image, imu_data):
        # IMU预测
        predicted_pose = self.imu.integrate(imu_data)
        
        # 视觉修正
        visual_features = self.visual.extract(image)
        corrected_pose = self.visual.align(visual_features, predicted_pose)
        
        return corrected_pose
```

### 4.2 虚拟物体渲染

```python
def ar_render(background_image, virtual_object, camera_pose, virtual_pose):
    """AR渲染：将虚拟物体叠加到真实场景"""
    # 1. 虚拟物体渲染
    virtual_rgb, virtual_depth, virtual_alpha = render_object(
        virtual_object, virtual_pose, camera_pose
    )
    
    # 2. 深度测试（遮挡处理）
    real_depth = estimate_depth(background_image)
    occlusion_mask = virtual_depth < real_depth
    
    # 3. Alpha混合
    alpha = virtual_alpha * occlusion_mask.float()
    composite = background_image * (1 - alpha) + virtual_rgb * alpha
    
    # 4. 光照匹配（可选）
    composite = match_illumination(composite, background_image)
    
    return composite
```

## 5. 医学影像

### 5.1 3D医学分割

```python
class UNet3D(nn.Module):
    """3D U-Net用于医学影像分割"""
    def __init__(self, in_channels=1, num_classes=4):
        super().__init__()
        # 编码器
        self.enc1 = self.conv_block(in_channels, 32)
        self.enc2 = self.conv_block(32, 64)
        self.enc3 = self.conv_block(64, 128)
        
        # 瓶颈
        self.bottleneck = self.conv_block(128, 256)
        
        # 解码器
        self.up3 = nn.ConvTranspose3d(256, 128, 2, stride=2)
        self.dec3 = self.conv_block(256, 128)
        self.up2 = nn.ConvTranspose3d(128, 64, 2, stride=2)
        self.dec2 = self.conv_block(128, 64)
        self.up1 = nn.ConvTranspose3d(64, 32, 2, stride=2)
        self.dec1 = self.conv_block(64, 32)
        
        self.out_conv = nn.Conv3d(32, num_classes, 1)
    
    def conv_block(self, in_ch, out_ch):
        return nn.Sequential(
            nn.Conv3d(in_ch, out_ch, 3, padding=1),
            nn.BatchNorm3d(out_ch), nn.ReLU(),
            nn.Conv3d(out_ch, out_ch, 3, padding=1),
            nn.BatchNorm3d(out_ch), nn.ReLU(),
        )
    
    def forward(self, x):
        e1 = self.enc1(x)
        e2 = self.enc2(F.max_pool3d(e1, 2))
        e3 = self.enc3(F.max_pool3d(e2, 2))
        
        b = self.bottleneck(F.max_pool3d(e3, 2))
        
        d3 = self.dec3(torch.cat([self.up3(b), e3], dim=1))
        d2 = self.dec2(torch.cat([self.up2(d3), e2], dim=1))
        d1 = self.dec1(torch.cat([self.up1(d2), e1], dim=1))
        
        return self.out_conv(d1)
```
