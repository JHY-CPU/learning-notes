## 2. 统计检测方法

### 2.1 困惑度与突发度

```python
import torch
import numpy as np

def compute_perplexity(text, model, tokenizer):
    """
    计算文本的困惑度 (Perplexity)
    原理: AI 生成文本的困惑度通常低于人类写作
    """
    inputs = tokenizer(text, return_tensors='pt')
    with torch.no_grad():
        outputs = model(**inputs, labels=inputs['input_ids'])
        loss = outputs.loss
        perplexity = torch.exp(loss).item()
    return perplexity

def compute_burstiness(sentences):
    """
    计算文本的突发度 (Burstiness)
    原理: 人类写作句子长度变化大，AI 生成较均匀
    """
    lengths = [len(s.split()) for s in sentences]
    mean_len = np.mean(lengths)
    std_len = np.std(lengths)
    burstiness = std_len / mean_len if mean_len > 0 else 0
    return burstiness

def simple_detector(text, model, tokenizer):
    """
    基于困惑度的简单检测器
    """
    perplexity = compute_perplexity(text, model, tokenizer)
    sentences = text.split('.')
    burstiness = compute_burstiness(sentences)

    # 经验阈值
    is_ai_generated = (perplexity < 30) and (burstiness < 0.5)

    return {
        'perplexity': perplexity,
        'burstiness': burstiness,
        'prediction': 'AI-generated' if is_ai_generated else 'Human-written',
        'confidence': min(1.0, (30 - perplexity) / 30) if is_ai_generated
                      else min(1.0, (perplexity - 30) / 30)
    }
```

### 2.2 DetectGPT

Mitchell et al. (2023) 提出 DetectGPT，基于**曲率分析**：

```python
def detectgpt(text, model, tokenizer, mask_model, num_perturbations=100,
              perturbation_std=0.1):
    """
    DetectGPT: 基于对数概率曲率的检测
    原理: AI 生成文本位于对数概率的局部最大值附近
    """
    # 计算原始文本的对数概率
    log_prob_original = compute_log_prob(text, model, tokenizer)

    # 生成扰动版本
    perturbed_texts = generate_perturbations(
        text, mask_model, num_perturbations
    )

    # 计算扰动文本的对数概率
    log_prob_perturbed = []
    for perturbed in perturbed_texts:
        lp = compute_log_prob(perturbed, model, tokenizer)
        log_prob_perturbed.append(lp)

    # 计算曲率分数
    # AI 生成文本: 原始概率 > 扰动概率 → 负分数
    # 人类文本: 原始概率 ≈ 扰动概率 → 分数接近 0
    curvature_score = log_prob_original - np.mean(log_prob_perturbed)

    return {
        'score': curvature_score,
        'is_ai_generated': curvature_score > 0,  # 正值表示 AI 生成
        'confidence': abs(curvature_score)
    }

def generate_perturbations(text, mask_model, num_perturbations):
    """
    使用掩码语言模型生成扰动
    """
    tokens = text.split()
    perturbed_texts = []

    for _ in range(num_perturbations):
        perturbed = tokens.copy()
        # 随机替换部分 token
        num_mask = max(1, len(tokens) // 10)
        mask_indices = np.random.choice(len(tokens), num_mask, replace=False)

        for idx in mask_indices:
            # 使用 MLM 生成替代词
            context = ' '.join(perturbed[:idx] + ['[MASK]'] + perturbed[idx+1:])
            suggestion = mask_model.predict(context)
            perturbed[idx] = suggestion

        perturbed_texts.append(' '.join(perturbed))

    return perturbed_texts
```

```python
def dna_gpt_detector(text, model, tokenizer):
    """
    DNA-GPT: 通过 n-gram 重叠度检测
    原理: AI 生成文本倾向于重复使用相似的 n-gram 模式
    """
    # 生成多个续写
    completions = []
    for _ in range(10):
        completion = generate_completion(model, text, temperature=0.8)
        completions.append(completion)

    # 计算 n-gram 重叠
    ngram_overlap_scores = []
    for i in range(len(completions)):
        for j in range(i+1, len(completions)):
            overlap = compute_ngram_overlap(completions[i], completions[j], n=4)
            ngram_overlap_scores.append(overlap)

    avg_overlap = np.mean(ngram_overlap_scores)

    return {
        'avg_ngram_overlap': avg_overlap,
        'is_ai_generated': avg_overlap > 0.3
    }

def compute_ngram_overlap(text1, text2, n=4):
    """计算两个文本之间的 n-gram 重叠度"""
    def get_ngrams(text, n):
        words = text.split()
        return set(tuple(words[i:i+n]) for i in range(len(words)-n+1))

    ngrams1 = get_ngrams(text1, n)
    ngrams2 = get_ngrams(text2, n)

    if not ngrams1 or not ngrams2:
        return 0

    intersection = len(ngrams1 & ngrams2)
    union = len(ngrams1 | ngrams2)

    return intersection / union
```

### 4.1 GAN 指纹检测

```python
def detect_gan_image(image, classifier):
    """
    检测图像是否为 GAN 生成
    原理: GAN 在频域留下特殊指纹
    """
    # 频域分析
    from scipy.fft import fft2

    image_np = np.array(image.convert('L'))  # 转灰度
    freq = fft2(image_np)
    magnitude = np.abs(freq)

    # 计算频域统计特征
    features = extract_frequency_features(magnitude)

    # 分类
    prediction = classifier.predict(features)

    return {
        'is_ai_generated': prediction == 1,
        'frequency_features': features
    }

def extract_frequency_features(magnitude_spectrum):
    """
    从频谱中提取特征
    """
    # 对数变换
    log_spectrum = np.log(magnitude_spectrum + 1e-8)

    # 径向平均
    h, w = log_spectrum.shape
    center = (h//2, w//2)
    max_radius = min(center)

    radial_profile = []
    for r in range(1, max_radius):
        y, x = np.ogrid[:h, :w]
        mask = ((y - center[0])**2 + (x - center[1])**2 <= r**2) & \
               ((y - center[0])**2 + (x - center[1])**2 > (r-1)**2)
        if mask.any():
            radial_profile.append(log_spectrum[mask].mean())

    return np.array(radial_profile)
```
