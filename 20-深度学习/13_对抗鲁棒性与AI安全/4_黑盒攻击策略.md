# 4_黑盒攻击策略

## 1. 黑盒攻击概述

黑盒攻击是指攻击者**无法直接访问目标模型的内部参数和结构**，只能通过查询接口获取模型的输出结果（如分类标签、置信度分数）来进行攻击。

### 1.1 攻击者知识假设

| 攻击类型 | 知识量级 | 模型参数 | 模型结构 | 训练数据 | 查询权限 |
|----------|----------|----------|----------|----------|----------|
| 白盒 | 完全 | 已知 | 已知 | 部分已知 | 不需要 |
| 灰盒 | 部分 | 未知 | 已知 | 未知 | 不需要 |
| 黑盒 | 极少 | 未知 | 未知 | 未知 | 需要 |

### 1.2 主要黑盒攻击方法

```
黑盒攻击
├── 基于迁移性 (Transfer-based)
│   ├── 单模型迁移
│   ├── 集成迁移
│   └── 输入变换增强
├── 基于查询 (Query-based)
│   ├── 分数攻击 (Score-based)
│   ├── 决策攻击 (Decision-based)
│   └── 随机搜索 / 进化策略
└── 替代模型 (Substitute Model)
    ├── 训练替代模型
    └── 合成数据增强
```

## 2. 迁移攻击 (Transfer Attack)

### 2.1 迁移性原理

对抗样本的迁移性 (Transferability) 是指：**在一个模型上生成的对抗样本，往往能成功攻击另一个结构不同的模型**。

原因分析：
- 不同模型学习到相似的决策边界
- 数据驱动的特征提取具有共通性
- 对抗方向在模型间具有相关性

### 2.2 基础迁移攻击

```python
def transfer_attack(source_model, target_model, images, labels,
                    epsilon=8/255, alpha=2/255, num_steps=10):
    """
    使用 PGD 在源模型上生成对抗样本，攻击目标模型
    """
    # 在源模型上生成对抗样本
    adv_images = pgd_attack(source_model, images, labels,
                            epsilon, alpha, num_steps)

    # 测试迁移效果
    with torch.no_grad():
        source_out = source_model(adv_images)
        target_out = target_model(adv_images)

        _, src_pred = source_out.max(1)
        _, tgt_pred = target_out.max(1)

        src_acc = src_pred.eq(labels).float().mean().item()
        tgt_acc = tgt_pred.eq(labels).float().mean().item()

    print(f"源模型准确率: {src_acc:.4f}")
    print(f"目标模型准确率: {tgt_acc:.4f}")
    print(f"迁移攻击成功率: {(1-tgt_acc)/(1-src_acc)*100:.2f}%")

    return adv_images
```

### 2.3 提升迁移性的方法

```python
def ensemble_transfer_attack(models, images, labels, epsilon,
                              alpha, num_steps):
    """
    集成迁移攻击: 同时攻击多个模型以提升迁移性
    """
    images_adv = images.clone().detach()

    for _ in range(num_steps):
        images_adv.requires_grad_(True)

        # 聚合多个模型的梯度
        total_grad = torch.zeros_like(images_adv)
        for model in models:
            outputs = model(images_adv)
            loss = F.cross_entropy(outputs, labels)
            model.zero_grad()
            loss.backward()
            total_grad += images_adv.grad.detach()

        # 使用聚合梯度更新
        images_adv = images_adv.detach() + alpha * total_grad.sign()
        delta = torch.clamp(images_adv - images, -epsilon, epsilon)
        images_adv = torch.clamp(images + delta, 0, 1).detach()

    return images_adv

def input_diversity_attack(model, images, labels, epsilon,
                           alpha, num_steps, p=0.5):
    """
    输入多样性攻击: 对输入做随机变换增强迁移性
    """
    images_adv = images.clone().detach()

    for _ in range(num_steps):
        images_adv.requires_grad_(True)

        # 随机裁剪和填充
        if torch.rand(1).item() < p:
            # 随机缩放
            scale = 0.8 + 0.4 * torch.rand(1).item()
            h, w = images.shape[2:]
            new_h, new_w = int(h * scale), int(w * scale)
            resized = F.interpolate(images_adv, size=(new_h, new_w),
                                    mode='bilinear')
            # 随机裁剪回原始大小
            pad_h = max(0, h - new_h)
            pad_w = max(0, w - new_w)
            if pad_h > 0 or pad_w > 0:
                resized = F.pad(resized, [0, pad_w, 0, pad_h])
            images_adv = resized[:, :, :h, :w]

        outputs = model(images_adv)
        loss = F.cross_entropy(outputs, labels)
        model.zero_grad()
        loss.backward()

        images_adv = images_adv.detach() + alpha * images_adv.grad.sign()
        delta = torch.clamp(images_adv - images, -epsilon, epsilon)
        images_adv = torch.clamp(images + delta, 0, 1).detach()

    return images_adv
```

## 3. 查询攻击 (Query-based Attack)

### 3.1 基于分数的攻击

当模型输出概率分数时，可用梯度估计方法：

```python
def score_based_attack(model, images, labels, epsilon,
                       query_budget=10000, lr=0.01):
    """
    基于分数的黑盒攻击 (使用随机梯度估计)
    """
    dim = images.view(images.size(0), -1).size(1)
    images_adv = images.clone().view(images.size(0), -1)
    sigma = 0.01  # 随机扰动幅度

    queries_used = 0

    while queries_used < query_budget:
        # 随机采样方向
        noise = torch.randn_like(images_adv)
        noise = noise / noise.norm(dim=1, keepdim=True)

        # 估计梯度 (零阶方法)
        with torch.no_grad():
            x_plus = (images_adv + sigma * noise).view(images.shape)
            x_minus = (images_adv - sigma * noise).view(images.shape)
            x_plus = torch.clamp(x_plus, 0, 1)
            x_minus = torch.clamp(x_minus, 0, 1)

            out_plus = F.softmax(model(x_plus), dim=1)
            out_minus = F.softmax(model(x_minus), dim=1)

            f_plus = out_plus.gather(1, labels.unsqueeze(1)).squeeze()
            f_minus = out_minus.gather(1, labels.unsqueeze(1)).squeeze()

            queries_used += 2

        # 零阶梯度估计
        grad_est = ((f_plus - f_minus) / (2 * sigma)).unsqueeze(1) * noise

        # 梯度上升
        images_adv = images_adv + lr * grad_est.sign()

        # 投影到约束集合
        delta = images_adv - images.view(images.size(0), -1)
        delta = torch.clamp(delta, -epsilon, epsilon)
        images_adv = torch.clamp(images.view(images.size(0), -1) + delta, 0, 1)

    return images_adv.view(images.shape).detach()
```

### 3.2 基于决策的攻击 (Boundary Attack)

```python
def boundary_attack(model, images, labels, epsilon, query_budget=5000):
    """
    Boundary Attack: 仅需分类决策的黑盒攻击
    """
    batch_size = images.size(0)
    images_adv = images.clone()

    # 步骤1: 找到对抗样本 (随机初始化)
    with torch.no_grad():
        while True:
            noise = torch.randn_like(images) * epsilon
            candidate = torch.clamp(images + noise, 0, 1)
            pred = model(candidate).argmax(1)
            if (pred != labels).all():
                images_adv = candidate
                break

    queries_used = batch_size
    step_size = epsilon

    while queries_used < query_budget:
        # 步骤2: 在决策边界附近采样
        ortho_noise = torch.randn_like(images_adv)
        # 投影到以原始图像为中心的球面上
        direction = images_adv - images
        direction_norm = direction.view(batch_size, -1).norm(dim=1)
        ortho_noise_flat = ortho_noise.view(batch_size, -1)
        direction_flat = direction.view(batch_size, -1)

        # Gram-Schmidt 正交化
        proj = (ortho_noise_flat * direction_flat).sum(1, keepdim=True)
        ortho_noise_flat = ortho_noise_flat - proj / (
            direction_flat.norm(dim=1, keepdim=True)**2 + 1e-8
        ) * direction_flat

        # 混合方向和正交扰动
        candidate = images_adv + step_size * ortho_noise.view(images.shape)
        candidate = torch.clamp(candidate, 0, 1)

        with torch.no_grad():
            pred = model(candidate).argmax(1)
            queries_used += batch_size

            # 保持对抗性和距离减小
            still_adv = pred != labels
            closer = ((candidate - images)**2).view(batch_size, -1).sum(1) < \
                     ((images_adv - images)**2).view(batch_size, -1).sum(1)

            update = still_adv & closer
            images_adv[update] = candidate[update]

            if not closer.any():
                step_size *= 0.9  # 减小步长

    return images_adv.detach()
```

## 4. 替代模型攻击 (Substitute Model)

### 4.1 Papernot 方法 (2017)

```python
def train_substitute_model(target_model, initial_data, architecture,
                           num_augment_rounds=6, lmbda=0.1):
    """
    训练替代模型: 通过查询目标模型获取标签来训练替代模型
    """
    substitute = architecture()
    substitute_optimizer = torch.optim.Adam(substitute.parameters())

    # 初始数据集
    X_sub = initial_data
    Y_sub = query_model(target_model, X_sub)  # 查询目标模型获取标签

    for round_idx in range(num_augment_rounds):
        # 步骤1: 用当前数据训练替代模型
        for epoch in range(10):
            outputs = substitute(X_sub)
            loss = F.cross_entropy(outputs, Y_sub)
            substitute_optimizer.zero_grad()
            loss.backward()
            substitute_optimizer.step()

        # 步骤2: 数据增强 (Jacobian-based)
        X_aug = jacobian_augment(substitute, X_sub, lmbda)

        # 步骤3: 用增强数据查询目标模型
        Y_aug = query_model(target_model, X_aug)

        # 合并数据
        X_sub = torch.cat([X_sub, X_aug], dim=0)
        Y_sub = torch.cat([Y_sub, Y_aug], dim=0)

    return substitute

def jacobian_augment(model, X, lmbda):
    """
    Jacobian-based 数据增强
    """
    X_aug = X.clone().detach().requires_grad_(True)
    outputs = model(X_aug)
    _, pred = outputs.max(1)

    # 计算 Jacobian
    grad_outputs = torch.zeros_like(outputs)
    for i in range(X.size(0)):
        grad_outputs[i, pred[i]] = 1.0

    jacobian = torch.autograd.grad(outputs, X_aug, grad_outputs,
                                    create_graph=False)[0]

    # 沿 Jacobian 方向增强
    X_new = X + lmbda * jacobian.sign()
    X_new = torch.clamp(X_new, 0, 1)

    return X_new.detach()
```

## 5. 攻击效率对比

| 方法 | 所需查询次数 | 攻击成功率 | 迁移性 | 适用场景 |
|------|-------------|------------|--------|----------|
| 迁移攻击 | 0 | 中-高 | 固有 | 通用 |
| 集成迁移 | 0 | 高 | 更好 | 多模型可用 |
| 随机梯度估计 | 10K-100K | 高 | 无 | 有分数输出 |
| Boundary | 5K-50K | 中 | 无 | 仅决策输出 |
| 替代模型 | 1K-10K | 中-高 | 有 | 数据可得 |

## 6. 总结

| 要点 | 说明 |
|------|------|
| 迁移攻击 | 利用对抗样本的跨模型迁移性 |
| 查询攻击 | 通过模型输出估计梯度或探索边界 |
| 替代模型 | 训练本地模型模拟目标模型 |
| 提升策略 | 集成、输入多样性、动量迭代 |
| 防御思路 | 随机化、查询限制、对抗训练 |
