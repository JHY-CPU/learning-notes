# 14_隐私泄露与数据提取

## 1. 隐私威胁概述

深度学习模型在训练过程中可能**记忆并泄露训练数据**中的敏感信息。这带来了严重的隐私风险，尤其是在模型被公开发布或通过 API 访问时。

### 1.1 隐私攻击分类

```
隐私攻击
├── 训练数据记忆 (Training Data Memorization)
│   ├── 完整文本提取
│   └── 属性记忆 (姓名、电话等)
├── 成员推理攻击 (Membership Inference)
│   ├── 影子模型方法
│   └── 基于置信度方法
├── 模型反演攻击 (Model Inversion)
│   ├── 训练数据重构
│   └── 属性推断
└── 属性推断攻击 (Attribute Inference)
    ├── 敏感属性推断
    └── 数据集属性推断
```

## 2. 训练数据记忆

### 2.1 LLM 中的数据提取

Carlini 等人 (2021) 证明，GPT-2 等大模型可以被诱导**逐字复述训练数据**：

```python
def extract_training_data(model, tokenizer, prefixes, num_completions=100):
    """
    从语言模型中提取训练数据
    """
    extracted = []

    for prefix in prefixes:
        for _ in range(num_completions):
            input_ids = tokenizer.encode(prefix, return_tensors='pt')
            with torch.no_grad():
                outputs = model.generate(
                    input_ids,
                    max_new_tokens=256,
                    temperature=0.7,
                    do_sample=True
                )
            text = tokenizer.decode(outputs[0], skip_special_tokens=True)
            completion = text[len(prefix):]

            # 检查是否是记忆的文本 (高困惑度指示新颖内容)
            perplexity = compute_perplexity(completion, model, tokenizer)
            if perplexity < 10:  # 低困惑度 = 可能是记忆内容
                extracted.append(completion)

    return extracted

def verify_memorization(extracted_texts, training_corpus):
    """
    验证提取的文本是否真的来自训练数据
    """
    verified = []
    for text in extracted_texts:
        # 在训练语料中搜索
        for doc in training_corpus:
            if text in doc:
                verified.append({
                    'extracted': text,
                    'source': doc[:200],
                    'exact_match': True
                })
                break
    return verified
```

### 2.2 记忆率分析

```python
def memorization_analysis(model, dataset, num_samples=1000):
    """
    分析模型对训练数据的记忆程度
    """
    memorized_count = 0
    partial_memorized_count = 0

    for i in range(min(num_samples, len(dataset))):
        sample = dataset[i]
        prompt = sample[:len(sample)//2]  # 前半部分作为 prompt
        expected = sample[len(sample)//2:]  # 后半部分作为期望

        generated = generate(model, prompt, max_length=len(expected))

        # 精确匹配
        if generated.strip() == expected.strip():
            memorized_count += 1
        # 部分匹配 (相似度 > 0.8)
        elif compute_similarity(generated, expected) > 0.8:
            partial_memorized_count += 1

    return {
        'exact_memorization_rate': memorized_count / num_samples,
        'partial_memorization_rate': partial_memorized_count / num_samples,
        'total_samples': num_samples
    }
```

## 3. 成员推理攻击

### 3.1 攻击原理

成员推理攻击 (Membership Inference Attack) 的目标是判断**某个数据样本是否属于模型的训练集**。

```
攻击场景:

攻击者 → "样本 X 是否在训练集中?" → 目标模型
                                       ↓
模型输出 (置信度/损失/梯度) → 攻击模型 → 是/否
```

### 3.2 基于影子模型的方法

```python
class ShadowModelAttack:
    """
    使用影子模型进行成员推理攻击
    """
    def __init__(self, target_model, shadow_model, attack_model):
        self.target = target_model
        self.shadow = shadow_model
        self.attack = attack_model  # 二分类攻击器

    def train_attack_model(self, train_data, non_member_data):
        """
        用影子模型训练攻击模型
        """
        attack_features = []
        attack_labels = []

        # 训练影子模型 (使用部分已知数据)
        shadow_train = train_data[:len(train_data)//2]
        self.train_shadow_model(shadow_train)

        # 从影子模型提取特征
        for sample in train_data[len(train_data)//2:]:
            features = self.extract_features(self.shadow, sample)
            attack_features.append(features)
            attack_labels.append(1)  # 成员

        for sample in non_member_data:
            features = self.extract_features(self.shadow, sample)
            attack_features.append(features)
            attack_labels.append(0)  # 非成员

        # 训练攻击模型
        X = torch.stack(attack_features)
        y = torch.tensor(attack_labels)
        self.train_classifier(X, y)

    def extract_features(self, model, sample):
        """
        从目标模型提取攻击特征
        """
        model.eval()
        with torch.no_grad():
            if isinstance(sample, tuple):
                x, y = sample
                output = model(x.unsqueeze(0))
                probs = F.softmax(output, dim=1)

                # 特征: 最大置信度、熵、正确类别的概率
                max_conf = probs.max().item()
                entropy = -(probs * torch.log(probs + 1e-8)).sum().item()
                correct_prob = probs[0, y].item()
                loss = F.cross_entropy(output, torch.tensor([y])).item()
            else:
                output = model(sample.unsqueeze(0))
                probs = F.softmax(output, dim=1)
                max_conf = probs.max().item()
                entropy = -(probs * torch.log(probs + 1e-8)).sum().item()
                correct_prob = 0
                loss = 0

        return torch.tensor([max_conf, entropy, correct_prob, loss])

    def infer_membership(self, sample):
        """推理样本是否为成员"""
        features = self.extract_features(self.target, sample)
        prediction = self.attack(features.unsqueeze(0))
        return prediction.argmax().item() == 1
```

### 3.3 基于置信度的简单方法

```python
def confidence_based_mia(model, sample, label, threshold=0.9):
    """
    基于置信度的简单成员推理
    原理: 模型对训练数据通常有更高的置信度
    """
    model.eval()
    with torch.no_grad():
        output = model(sample.unsqueeze(0))
        probs = F.softmax(output, dim=1)
        max_prob = probs.max().item()
        correct_prob = probs[0, label].item()

    is_member = (max_prob > threshold) or (correct_prob > threshold)

    return {
        'is_member': is_member,
        'confidence': max_prob,
        'correct_prob': correct_prob
    }
```

## 4. 模型反演攻击

### 4.1 训练数据重构

```python
def model_inversion_attack(model, target_class, input_shape,
                           num_steps=1000, lr=0.01):
    """
    模型反演攻击: 从模型重构训练数据
    """
    # 初始化随机输入
    x = torch.randn(1, *input_shape, requires_grad=True)
    optimizer = torch.optim.Adam([x], lr=lr)

    for step in range(num_steps):
        optimizer.zero_grad()

        output = model(x)
        probs = F.softmax(output, dim=1)

        # 最大化目标类的概率
        target_prob = probs[0, target_class]

        # 添加先验正则化 (自然图像的先验)
        prior_loss = 0.01 * (x ** 2).sum()  # 简单 L2 先验

        loss = -target_prob + prior_loss
        loss.backward()
        optimizer.step()

        # 可选: 使用 GAN 先验
        # prior_loss = gan_discriminator(x)

    return torch.clamp(x, 0, 1).detach()
```

## 5. 防御方法

| 防御方法 | 原理 | 效果 | 代价 |
|----------|------|------|------|
| 差分隐私训练 | 添加噪声保护个体隐私 | 高 | 准确率下降 |
| 正则化 | 减少过拟合/记忆 | 中 | 轻微 |
| 数据清洗 | 去除敏感数据 | 中 | 数据量减少 |
| 模型蒸馏 | 知识蒸馏减少记忆 | 中 | 额外训练 |
| 输出限制 | 限制 API 输出概率 | 中 | 功能受限 |
