## 2. SelfCheckGPT 详解

### 2.1 核心原理

SelfCheckGPT 的假设：**如果模型对某个事实确信，多次采样会产生一致的结果；如果模型在幻觉，不同采样会产生矛盾的结果。**

### 2.2 完整实现

```python
import torch
import numpy as np
from collections import Counter

class SelfCheckGPT:
    def __init__(self, model, tokenizer, num_samples=20):
        self.model = model
        self.tokenizer = tokenizer
        self.num_samples = num_samples

    def detect(self, prompt, reference_response=None):
        """
        检测给定响应中的幻觉
        """
        if reference_response is None:
            reference_response = self.generate(prompt, temperature=0.0)

        # 生成多个采样
        samples = []
        for _ in range(self.num_samples):
            sample = self.generate(prompt, temperature=0.8)
            samples.append(sample)

        # 分句
        sentences = self.split_sentences(reference_response)

        # 对每个句子计算幻觉分数
        scores = []
        for sentence in sentences:
            score = self.compute_hallucination_score(sentence, samples)
            scores.append(score)

        return {
            'sentences': sentences,
            'hallucination_scores': scores,
            'overall_score': np.mean(scores),
            'flagged_sentences': [s for s, sc in zip(sentences, scores)
                                  if sc > 0.5]
        }

    def generate(self, prompt, temperature=0.8):
        inputs = self.tokenizer(prompt, return_tensors='pt')
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=512,
                temperature=temperature,
                do_sample=temperature > 0
            )
        return self.tokenizer.decode(outputs[0], skip_special_tokens=True)

    def compute_hallucination_score(self, sentence, samples):
        """
        计算单个句子的幻觉分数
        使用 NLI 模型判断一致性
        """
        entailment_count = 0
        contradiction_count = 0
        neutral_count = 0

        for sample in samples:
            # 使用 NLI 模型判断 sample 是否支持 sentence
            nli_result = nli_model.predict(premise=sample, hypothesis=sentence)
            if nli_result == 'entailment':
                entailment_count += 1
            elif nli_result == 'contradiction':
                contradiction_count += 1
            else:
                neutral_count += 1

        # 幻觉分数 = 不一致比例
        hallucination_score = contradiction_count / self.num_samples
        return hallucination_score

    def split_sentences(self, text):
        """简单分句"""
        import re
        sentences = re.split(r'(?<=[.!?。！？])\s+', text)
        return [s.strip() for s in sentences if len(s.strip()) > 10]
```

```python
class CitationVerifier:
    """
    验证模型输出中的引用是否真实存在
    """
    def __init__(self, scholar_api_key=None):
        self.api_key = scholar_api_key

    def verify_citations(self, text):
        """
        从文本中提取引用并验证
        """
        # 提取论文引用
        citations = self.extract_citations(text)
        results = []

        for citation in citations:
            is_real = self.check_citation_exists(citation)
            results.append({
                'citation': citation,
                'exists': is_real,
                'verified_info': self.get_paper_info(citation) if is_real else None
            })

        return results

    def extract_citations(self, text):
        """
        提取文中的引用 (如 [Author, Year] 格式)
        """
        import re
        # 匹配常见的引用格式
        patterns = [
            r'\([A-Z][a-z]+\s+et\s+al\.?,?\s+\d{4}\)',  # (Smith et al., 2023)
            r'\[[A-Z][a-z]+\s+et\s+al\.?,?\s+\d{4}\]',  # [Smith et al., 2023]
            r'(?:paper|study|research)\s+by\s+([A-Z][a-z]+)',  # paper by Smith
        ]

        citations = []
        for pattern in patterns:
            matches = re.findall(pattern, text)
            citations.extend(matches)

        return citations

    def check_citation_exists(self, citation):
```

## 4. RAG 缓解策略

### 4.1 检索增强生成

```python
class RAGPipeline:
    """
    检索增强生成管道
    """
    def __init__(self, retriever, generator):
        self.retriever = retriever
        self.generator = generator

    def generate(self, query, top_k=5):
        """
        RAG 生成流程
        """
        # 1. 检索相关文档
        docs = self.retriever.retrieve(query, top_k=top_k)

        # 2. 构建增强 Prompt
        context = "\n\n".join([
            f"[文档{i+1}] {doc['title']}\n{doc['content']}"
            for i, doc in enumerate(docs)
        ])

        prompt = f"""请基于以下参考资料回答问题。如果参考资料中没有
相关信息，请明确说明"根据提供的资料，我无法回答这个问题"。

参考资料:
{context}

问题: {query}

请基于上述参考资料回答，必要时引用文档编号。"""

        # 3. 生成回答
        response = self.generator.generate(prompt)

        return {
            'answer': response,
            'sources': docs,
            'has_grounding': len(docs) > 0
        }
```

```python
class HallucinationDetector(nn.Module):
    """
    二分类器: 判断句子是否包含幻觉
    """
    def __init__(self, encoder, hidden_dim=768):
        super().__init__()
        self.encoder = encoder  # 预训练编码器
        self.classifier = nn.Sequential(
            nn.Linear(hidden_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(256, 2),
            nn.Softmax(dim=-1)
        )

    def forward(self, sentence, context):
        """
        输入: 句子 + 上下文
        输出: [真实概率, 幻觉概率]
        """
        # 编码句子和上下文
        combined = f"{context} [SEP] {sentence}"
        encoded = self.encoder(combined)
        logits = self.classifier(encoded)
        return logits
```
