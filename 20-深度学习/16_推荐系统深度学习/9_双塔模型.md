# 9_双塔模型 (Two-Tower)

## 1. 概述

双塔模型将用户和物品分别编码为向量，通过向量相似度进行推荐。其最大优势是**可解耦计算**：用户向量和物品向量可独立计算，物品向量可离线预计算并索引，支持超大规模检索。

```
用户塔 (User Tower)          物品塔 (Item Tower)
┌──────────────┐             ┌──────────────┐
│ 用户特征     │             │ 物品特征     │
│ - user_id    │             │ - item_id    │
│ - 年龄/性别  │             │ - 类别/品牌  │
│ - 历史行为   │             │ - 标题嵌入   │
└──────┬───────┘             └──────┬───────┘
       ↓ DNN                        ↓ DNN
┌──────┴───────┐             ┌──────┴───────┐
│ 用户向量 u   │             │ 物品向量 v   │
│ (128维)      │             │ (128维)      │
└──────┬───────┘             └──────┬───────┘
       └──────── score = <u, v> ────┘
```

## 2. 模型实现

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class UserTower(nn.Module):
    """用户塔"""
    def __init__(self, user_id_vocab, user_feature_dims, embed_dim=32,
                 hidden_dims=[256, 128]):
        super().__init__()
        self.user_id_embed = nn.Embedding(user_id_vocab, embed_dim)
        self.feature_embeds = nn.ModuleDict({
            name: nn.Embedding(vocab, embed_dim)
            for name, vocab in user_feature_dims.items()
        })

        input_dim = embed_dim * (1 + len(user_feature_dims))
        layers = []
        prev_dim = input_dim
        for h_dim in hidden_dims:
            layers.extend([nn.Linear(prev_dim, h_dim), nn.ReLU(), nn.Dropout(0.1)])
            prev_dim = h_dim
        self.mlp = nn.Sequential(*layers)

    def forward(self, user_features):
        """
        user_features: dict with 'user_id' and other features
        """
        embs = [self.user_id_embed(user_features['user_id'])]
        for name, embed in self.feature_embeds.items():
            embs.append(embed(user_features[name]))

        x = torch.cat(embs, dim=-1)
        return self.mlp(x)  # (B, hidden_dim)


class ItemTower(nn.Module):
    """物品塔"""
    def __init__(self, item_id_vocab, item_feature_dims, embed_dim=32,
                 hidden_dims=[256, 128]):
        super().__init__()
        self.item_id_embed = nn.Embedding(item_id_vocab, embed_dim)
        self.feature_embeds = nn.ModuleDict({
            name: nn.Embedding(vocab, embed_dim)
            for name, vocab in item_feature_dims.items()
        })

        input_dim = embed_dim * (1 + len(item_feature_dims))
        layers = []
        prev_dim = input_dim
        for h_dim in hidden_dims:
            layers.extend([nn.Linear(prev_dim, h_dim), nn.ReLU(), nn.Dropout(0.1)])
            prev_dim = h_dim
        self.mlp = nn.Sequential(*layers)

    def forward(self, item_features):
        embs = [self.item_id_embed(item_features['item_id'])]
        for name, embed in self.feature_embeds.items():
            embs.append(embed(item_features[name]))
        x = torch.cat(embs, dim=-1)
        return self.mlp(x)  # (B, hidden_dim)


class TwoTowerModel(nn.Module):
    """双塔模型"""
    def __init__(self, user_id_vocab, item_id_vocab,
                 user_feature_dims, item_feature_dims,
                 embed_dim=32, hidden_dims=[256, 128], output_dim=128):
        super().__init__()
        hidden_dims = hidden_dims + [output_dim]
        self.user_tower = UserTower(user_id_vocab, user_feature_dims,
                                     embed_dim, hidden_dims)
        self.item_tower = ItemTower(item_id_vocab, item_feature_dims,
                                     embed_dim, hidden_dims)

    def forward(self, user_features, item_features):
        user_emb = F.normalize(self.user_tower(user_features), dim=-1)
        item_emb = F.normalize(self.item_tower(item_features), dim=-1)
        score = (user_emb * item_emb).sum(dim=-1)  # 内积相似度
        return score

    def encode_users(self, user_features):
        """离线编码用户向量"""
        return F.normalize(self.user_tower(user_features), dim=-1)

    def encode_items(self, item_features):
        """离线编码物品向量"""
        return F.normalize(self.item_tower(item_features), dim=-1)
```

## 3. 训练：Sampled Softmax

全 softmax 需要计算所有物品的分数，不可行。使用采样 softmax：

```python
class SampledSoftmaxLoss(nn.Module):
    """Sampled Softmax: 只采样部分负样本"""
    def __init__(self, n_items, n_sampled, embed_dim):
        super().__init__()
        self.n_sampled = n_sampled
        # 全量物品嵌入（用于采样）
        self.all_item_embeddings = nn.Embedding(n_items, embed_dim)
        self.log_q = nn.Parameter(torch.zeros(n_items))  # 采样概率的 log

    def forward(self, user_emb, pos_item_ids, temperature=1.0):
        """
        user_emb: (B, D) 用户塔输出
        pos_item_ids: (B,) 正样本物品 ID
        """
        B, D = user_emb.shape

        # 正样本得分
        pos_emb = self.all_item_embeddings(pos_item_ids)  # (B, D)
        pos_score = (user_emb * pos_emb).sum(dim=-1) / temperature  # (B,)

        # 负采样
        neg_ids = torch.randint(0, self.all_item_embeddings.num_embeddings,
                                (B, self.n_sampled), device=user_emb.device)
        neg_emb = self.all_item_embeddings(neg_ids)  # (B, K, D)
        neg_scores = torch.bmm(neg_emb, user_emb.unsqueeze(-1)).squeeze(-1) / temperature  # (B, K)

        # 拼接正负样本得分
        all_scores = torch.cat([pos_score.unsqueeze(1), neg_scores], dim=1)  # (B, 1+K)

        # Softmax（第一维是正样本）
        loss = F.cross_entropy(all_scores, torch.zeros(B, dtype=torch.long,
                                                        device=user_emb.device))
        return loss
```

## 4. ANN 检索

在线推理时，需要从百万级物品中快速找到 Top-K 相似的物品。使用近似最近邻 (ANN)：

```python
# 使用 FAISS 进行 ANN 检索
import faiss

class ANNIndex:
    """FAISS 近似最近邻索引"""
    def __init__(self, embed_dim, n_items, nlist=100, m=8):
        """
        nlist: 聚类中心数
        m: PQ 子向量数
        """
        # IVF-PQ 索引：倒排文件 + 乘积量化
        quantizer = faiss.IndexFlatIP(embed_dim)  # 内积
        self.index = faiss.IndexIVFPQ(quantizer, embed_dim, nlist, m, 8)
        self.trained = False

    def build(self, item_embeddings):
        """构建索引"""
        item_embeddings = item_embeddings.astype('float32')
        faiss.normalize_L2(item_embeddings)
        self.index.train(item_embeddings)
        self.index.add(item_embeddings)
        self.trained = True
        print(f'索引构建完成，包含 {self.index.ntotal} 个物品')

    def search(self, query_vectors, top_k=10):
        """检索 Top-K 相似物品"""
        query_vectors = query_vectors.astype('float32')
        faiss.normalize_L2(query_vectors)
        scores, indices = self.index.search(query_vectors, top_k)
        return indices, scores
```

## 5. ANN 算法对比

| 算法 | 原理 | 时间复杂度 | 内存 | 精度 |
|------|------|-----------|------|------|
| Brute Force | 全量比较 | $O(n \cdot d)$ | 低 | 100% |
| LSH | 局部敏感哈希 | $O(n)$ | 中 | 中 |
| HNSW | 层级图导航 | $O(\log n)$ | 高 | 高 |
| IVF | 倒排索引聚类 | $O(\sqrt{n})$ | 中 | 中高 |
| PQ | 乘积量化 | $O(n / m)$ | 低 | 中 |

## 6. 双塔模型的改进

```python
# 1. 交互层增强（非纯内积）
class EnhancedTwoTower(nn.Module):
    def __init__(self, ...):
        super().__init__()
        self.user_tower = UserTower(...)
        self.item_tower = ItemTower(...)

        # 在内积之后加一层非线性
        self.interaction = nn.Sequential(
            nn.Linear(output_dim * 3, 64),  # u, v, u*v
            nn.ReLU(),
            nn.Linear(64, 1)
        )

    def forward(self, user_feat, item_feat):
        u = self.user_tower(user_feat)
        v = self.item_tower(item_feat)
        combined = torch.cat([u, v, u * v], dim=-1)
        return torch.sigmoid(self.interaction(combined)).squeeze(-1)
```

---

**要点总结：**
- 双塔模型的核心优势是用户和物品向量可独立计算
- 物品向量可离线预计算，使用 FAISS 等工具构建 ANN 索引
- Sampled Softmax 是大规模双塔训练的标准方法
- 双塔模型牺牲了用户-物品交叉建模能力，换取了极高的推理效率
- 在召回阶段，双塔模型 + ANN 检索是工业界的主流方案
