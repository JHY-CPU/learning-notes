# 14_冷启动解决方案

## 1. 冷启动分类

| 类型 | 定义 | 挑战 |
|------|------|------|
| 用户冷启动 | 新用户无历史行为 | 无法建模偏好 |
| 物品冷启动 | 新物品无被交互记录 | 无法估计受欢迎程度 |
| 系统冷启动 | 全新平台 | 无任何历史数据 |

## 2. 基于内容的冷启动

```python
import torch
import torch.nn as nn

class ContentBasedColdStart(nn.Module):
    """用物品内容特征缓解冷启动"""
    def __init__(self, user_embed_dim, item_content_dim, hidden_dim=128):
        super().__init__()
        # 物品内容编码器
        self.item_encoder = nn.Sequential(
            nn.Linear(item_content_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, user_embed_dim)
        )

        # 用户偏好编码器
        self.user_encoder = nn.Embedding(100000, user_embed_dim)

    def forward(self, user_ids, item_content_features):
        user_emb = self.user_encoder(user_ids)
        item_emb = self.item_encoder(item_content_features)
        return (user_emb * item_emb).sum(dim=-1)
```

## 3. 元学习冷启动

元学习 (Meta-Learning) 让模型学会"如何快速适应新用户/物品"。

### 3.1 MAML for Recommendation

```python
class MAMLRec(nn.Module):
    """MAML 元学习推荐"""
    def __init__(self, input_dim, hidden_dim, output_dim, inner_lr=0.01):
        super().__init__()
        self.inner_lr = inner_lr
        self.model = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, output_dim)
        )

    def inner_update(self, support_x, support_y, n_steps=5):
        """内循环：用少量样本快速适应"""
        fast_weights = [p.clone() for p in self.model.parameters()]

        for _ in range(n_steps):
            pred = self._forward_with_weights(support_x, fast_weights)
            loss = nn.BCELoss()(pred, support_y)
            grads = torch.autograd.grad(loss, fast_weights,
                                         create_graph=True)
            fast_weights = [w - self.inner_lr * g
                           for w, g in zip(fast_weights, grads)]

        return fast_weights

    def _forward_with_weights(self, x, weights):
        """用指定权重前向传播"""
        h = torch.relu(x @ weights[0].T + weights[1])
        return torch.sigmoid(h @ weights[2].T + weights[3])

    def meta_loss(self, tasks, inner_steps=5):
        """元损失：所有任务的查询损失之和"""
        meta_loss = 0
        for support_x, support_y, query_x, query_y in tasks:
            fast_weights = self.inner_update(support_x, support_y, inner_steps)
            query_pred = self._forward_with_weights(query_x, fast_weights)
            meta_loss += nn.BCELoss()(query_pred, query_y)
        return meta_loss / len(tasks)
```

### 3.2 训练流程

```python
def train_maml_rec(model, meta_optimizer, tasks_loader, meta_epochs=100):
    """元学习训练"""
    for epoch in range(meta_epochs):
        for batch_tasks in tasks_loader:
            meta_loss = model.meta_loss(batch_tasks, inner_steps=5)

            meta_optimizer.zero_grad()
            meta_loss.backward()
            meta_optimizer.step()

        # 快速适应新用户
        if (epoch + 1) % 10 == 0:
            print(f'Meta Epoch {epoch+1}, Meta Loss: {meta_loss.item():.4f}')
```

## 4. 迁移学习冷启动

```python
class TransferColdStart(nn.Module):
    """从成熟领域迁移到新领域"""
    def __init__(self, source_model, target_item_dim):
        super().__init__()
        # 冻结源领域学习到的表示
        self.shared_encoder = source_model.encoder
        for param in self.shared_encoder.parameters():
            param.requires_grad = False

        # 目标领域适配层
        self.target_adapter = nn.Sequential(
            nn.Linear(source_model.output_dim, 128),
            nn.ReLU(),
            nn.Linear(128, target_item_dim)
        )

    def forward(self, x):
        with torch.no_grad():
            source_repr = self.shared_encoder(x)
        return self.target_adapter(source_repr)
```

## 5. 主动学习冷启动

主动选择最有信息量的物品让用户评分：

```python
class ActiveLearner:
    """主动学习：选择最有信息量的样本"""
    def __init__(self, model, n_items):
        self.model = model
        self.n_items = n_items

    def select_items(self, user_id, n_select=5, strategy='uncertainty'):
        """选择让用户评分的物品"""
        all_items = torch.arange(self.n_items)
        user_tensor = torch.full_like(all_items, user_id)

        with torch.no_grad():
            scores = self.model(user_tensor, all_items)
            probs = torch.sigmoid(scores)

        if strategy == 'uncertainty':
            # 不确定性采样：选择预测概率接近 0.5 的
            uncertainty = 1 - torch.abs(probs - 0.5)
            _, indices = uncertainty.topk(n_select)

        elif strategy == 'entropy':
            # 熵采样
            entropy = -(probs * torch.log(probs + 1e-8) +
                       (1 - probs) * torch.log(1 - probs + 1e-8))
            _, indices = entropy.topk(n_select)

        elif strategy == 'diversity':
            # 多样性：选择特征差异大的
            item_features = self.get_item_features(all_items)
            selected = []
            for _ in range(n_select):
                if not selected:
                    selected.append(torch.randint(0, self.n_items, (1,)).item())
                else:
                    selected_features = item_features[selected]
                    similarities = torch.matmul(
                        item_features, selected_features.T
                    ).max(dim=1)[0]
                    _, indices = similarities.topk(1, largest=False)
                    selected.append(indices.item())
            indices = torch.LongTensor(selected)

        return indices

    def get_item_features(self, item_ids):
        """获取物品特征向量"""
        # 实际实现从数据库或嵌入层获取
        return torch.randn(len(item_ids), 64)
```

## 6. 冷启动策略选择

```
新用户加入
  ├── 有注册信息？
  │   ├── 是 → 基于人口统计学相似推荐
  │   └── 否 → 热门物品 + 探索式推荐
  ├── 有少量交互？
  │   ├── 是 → 元学习快速适应 + 内容相似
  │   └── 否 → 主动学习获取偏好
  └── 领域迁移？
      └── 是 → 迁移学习

新物品上架
  ├── 有内容特征？
  │   ├── 是 → 内容相似推荐
  │   └── 否 → 探索式曝光（MAB）
  └── 有预训练 KG 嵌入？
      └── 是 → 知识图谱增强
```

---

**要点总结：**
- 元学习 (MAML) 让模型学会快速适应新用户，是当前最有效的冷启动方案之一
- 迁移学习利用成熟领域的知识迁移到新领域
- 主动学习通过精心选择询问物品，高效获取用户偏好
- 实际系统中通常组合多种策略，根据场景选择
