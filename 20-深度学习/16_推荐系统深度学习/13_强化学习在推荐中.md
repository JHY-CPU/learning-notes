# 13_强化学习在推荐中

## 1. 概述

推荐系统可以建模为序列决策问题：智能体（推荐系统）根据用户状态选择动作（推荐物品），观察奖励（用户反馈），目标是最大化长期累积奖励。

**强化学习 vs 监督学习在推荐中的区别：**

| 维度 | 监督学习 | 强化学习 |
|------|---------|---------|
| 目标 | 最大化当前预测准确率 | 最大化长期用户满意度 |
| 反馈 | 离线标签 | 实时交互 |
| 时间维度 | 静态（单次推荐） | 动态（序列决策） |
| 探索 | 无 | 平衡探索与利用 |

## 2. 多臂老虎机 (MAB)

最简单的推荐探索-利用框架：

$$\text{UCB}_i = \bar{r}_i + c \sqrt{\frac{\ln t}{n_i}}$$

```python
import numpy as np

class UCB:
    """Upper Confidence Bound 推荐"""
    def __init__(self, n_arms, c=2.0):
        self.n_arms = n_arms
        self.c = c
        self.counts = np.zeros(n_arms)
        self.values = np.zeros(n_arms)
        self.total_pulls = 0

    def select_arm(self):
        self.total_pulls += 1
        # 未尝试过的臂优先
        untried = np.where(self.counts == 0)[0]
        if len(untried) > 0:
            return untried[0]

        # UCB 公式
        ucb_values = self.values + self.c * np.sqrt(
            np.log(self.total_pulls) / self.counts
        )
        return np.argmax(ucb_values)

    def update(self, arm, reward):
        self.counts[arm] += 1
        n = self.counts[arm]
        self.values[arm] = (self.values[arm] * (n - 1) + reward) / n


class ThompsonSampling:
    """Thompson Sampling: 贝叶斯方法"""
    def __init__(self, n_arms):
        # Beta 分布参数
        self.alpha = np.ones(n_arms)  # 成功次数 + 1
        self.beta = np.ones(n_arms)   # 失败次数 + 1

    def select_arm(self):
        # 从每个臂的 Beta 后验分布采样
        samples = np.random.beta(self.alpha, self.beta)
        return np.argmax(samples)

    def update(self, arm, reward):
        if reward > 0:
            self.alpha[arm] += 1
        else:
            self.beta[arm] += 1
```

## 3. DQN 推荐

```python
import torch
import torch.nn as nn
import random
from collections import deque

class DQNRecommender(nn.Module):
    """DQN 推荐：状态=用户画像+历史行为，动作=推荐物品"""
    def __init__(self, state_dim, n_actions, hidden_dim=256):
        super().__init__()
        self.network = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, n_actions)  # Q(s, a) for all actions
        )

    def forward(self, state):
        return self.network(state)

class DQNTrainer:
    def __init__(self, model, n_actions, lr=1e-3, gamma=0.99,
                 epsilon_start=1.0, epsilon_end=0.01, epsilon_decay=0.995,
                 buffer_size=10000, batch_size=64):
        self.model = model
        self.target_model = type(model)(model.network[0].in_features,
                                         n_actions).eval()
        self.target_model.load_state_dict(model.state_dict())

        self.n_actions = n_actions
        self.gamma = gamma
        self.epsilon = epsilon_start
        self.epsilon_end = epsilon_end
        self.epsilon_decay = epsilon_decay
        self.batch_size = batch_size

        self.optimizer = torch.optim.Adam(model.parameters(), lr=lr)
        self.buffer = deque(maxlen=buffer_size)

    def select_action(self, state):
        if random.random() < self.epsilon:
            return random.randint(0, self.n_actions - 1)
        with torch.no_grad():
            q_values = self.model(state)
            return q_values.argmax().item()

    def store_transition(self, state, action, reward, next_state, done):
        self.buffer.append((state, action, reward, next_state, done))

    def train_step(self):
        if len(self.buffer) < self.batch_size:
            return

        batch = random.sample(self.buffer, self.batch_size)
        states, actions, rewards, next_states, dones = zip(*batch)

        states = torch.stack(states)
        actions = torch.LongTensor(actions)
        rewards = torch.FloatTensor(rewards)
        next_states = torch.stack(next_states)
        dones = torch.FloatTensor(dones)

        # 当前 Q 值
        q_values = self.model(states).gather(1, actions.unsqueeze(1)).squeeze()

        # 目标 Q 值
        with torch.no_grad():
            next_q = self.target_model(next_states).max(1)[0]
            target_q = rewards + (1 - dones) * self.gamma * next_q

        loss = nn.MSELoss()(q_values, target_q)
        self.optimizer.zero_grad()
        loss.backward()
        self.optimizer.step()

        # 更新 epsilon
        self.epsilon = max(self.epsilon_end, self.epsilon * self.epsilon_decay)
```

## 4. 列表级推荐

推荐列表整体优化，而非逐物品优化：

```python
class ListWiseRecommender(nn.Module):
    """列表级推荐：使用 attention 生成整个推荐列表"""
    def __init__(self, state_dim, item_embed_dim, list_size, n_items):
        super().__init__()
        self.list_size = list_size
        self.item_embeddings = nn.Embedding(n_items, item_embed_dim)

        # 状态编码
        self.state_encoder = nn.Linear(state_dim, item_embed_dim)

        # 注意力机制：从候选集中选择列表
        self.attention = nn.MultiheadAttention(item_embed_dim, num_heads=4,
                                                batch_first=True)

    def forward(self, state, candidate_ids):
        """
        state: (B, state_dim) 用户状态
        candidate_ids: (B, n_candidates) 候选物品 ID
        """
        state_repr = self.state_encoder(state).unsqueeze(1)  # (B, 1, D)
        candidates = self.item_embeddings(candidate_ids)  # (B, C, D)

        # 注意力：查询=状态，键/值=候选
        attn_out, attn_weights = self.attention(state_repr, candidates, candidates)

        # 选择 top-k
        scores = attn_weights.squeeze(1)  # (B, C)
        _, topk_indices = scores.topk(self.list_size, dim=-1)
        return topk_indices, scores
```

## 5. 奖励设计

```python
class RewardShaper:
    """多目标奖励塑形"""
    def __init__(self, weights=None):
        self.weights = weights or {
            'click': 1.0,
            'like': 2.0,
            'share': 3.0,
            'purchase': 5.0,
            'dwell_time': 0.01,  # 每秒
            'dislike': -1.0,
            'skip': -0.1,
        }

    def compute(self, interaction):
        reward = 0.0
        if interaction.get('click'):
            reward += self.weights['click']
        if interaction.get('like'):
            reward += self.weights['like']
        if interaction.get('purchase'):
            reward += self.weights['purchase']
        if interaction.get('dwell_time'):
            reward += self.weights['dwell_time'] * interaction['dwell_time']
        if interaction.get('dislike'):
            reward += self.weights['dislike']
        return reward
```

## 6. 探索策略对比

| 策略 | 原理 | 优势 | 劣势 |
|------|------|------|------|
| ε-greedy | 以 ε 概率随机探索 | 简单 | 不智能 |
| UCB | 置信上界 | 理论保证 | 假设静态 |
| Thompson Sampling | 贝叶斯后验采样 | 自适应 | 需要分布假设 |
| DQN | 学习 Q 函数 | 复杂状态 | 样本效率低 |

---

**要点总结：**
- 强化学习将推荐建模为序列决策，优化长期用户满意度
- MAB 是最简单的探索-利用框架，适合热门物品的选择
- DQN 可以处理复杂状态空间，但样本效率低
- 列表级推荐考虑物品间的相互影响，优于逐物品打分
- 奖励设计需要平衡短期指标（点击）和长期指标（留存）
