# 12_知识增强推荐

## 1. 概述

知识图谱 (Knowledge Graph, KG) 包含实体间的丰富关系，如电影的导演、演员、类型等。将 KG 引入推荐可以：
- 解决数据稀疏和冷启动问题
- 提供可解释的推荐理由
- 引入物品间的语义关联

```
知识图谱示例:
电影A ──导演──→ 导演X ──导演──→ 电影C
  │                                  ↑
  └──演员──→ 演员Y ──主演──→ 电影B ──┘
  │                                  ↑
  └──类型──→ 科幻 ──类型──→ 电影D ──┘
```

## 2. KG 嵌入方法

将知识图谱中的实体和关系嵌入到低维向量空间。

### 2.1 TransE

$$h + r \approx t$$

```python
import torch
import torch.nn as nn

class TransE(nn.Module):
    """TransE: 平移模型"""
    def __init__(self, n_entities, n_relations, embed_dim, margin=1.0):
        super().__init__()
        self.entity_embed = nn.Embedding(n_entities, embed_dim)
        self.relation_embed = nn.Embedding(n_relations, embed_dim)
        self.margin = margin
        nn.init.xavier_uniform_(self.entity_embed.weight)
        nn.init.xavier_uniform_(self.relation_embed.weight)

    def forward(self, pos_triples, neg_triples):
        """
        pos_triples: (h, r, t) 正样本三元组
        neg_triples: (h', r, t') 负样本三元组（随机替换头或尾）
        """
        # 正样本
        h = self.entity_embed(pos_triples[:, 0])
        r = self.relation_embed(pos_triples[:, 1])
        t = self.entity_embed(pos_triples[:, 2])
        pos_score = (h + r - t).norm(dim=-1)

        # 负样本
        h_neg = self.entity_embed(neg_triples[:, 0])
        t_neg = self.entity_embed(neg_triples[:, 2])
        neg_score = (h_neg + r - t_neg).norm(dim=-1)

        # Margin-based ranking loss
        loss = torch.clamp(self.margin + pos_score - neg_score, min=0).mean()
        return loss
```

### 2.2 TransR

不同关系映射到不同空间：

$$h_r = M_r h, \quad t_r = M_r t, \quad h_r + r \approx t_r$$

```python
class TransR(nn.Module):
    def __init__(self, n_entities, n_relations, entity_dim, relation_dim):
        super().__init__()
        self.entity_embed = nn.Embedding(n_entities, entity_dim)
        self.relation_embed = nn.Embedding(n_relations, relation_dim)
        # 关系投影矩阵
        self.projection = nn.Embedding(n_relations, entity_dim * relation_dim)

    def forward(self, h, r, t):
        h_emb = self.entity_embed(h)
        t_emb = self.entity_embed(t)
        r_emb = self.relation_embed(r)

        # 投影到关系空间
        proj_matrix = self.projection(r).view(-1, self.relation_dim,
                                                self.entity_dim)
        h_r = torch.bmm(proj_matrix, h_emb.unsqueeze(-1)).squeeze(-1)
        t_r = torch.bmm(proj_matrix, t_emb.unsqueeze(-1)).squeeze(-1)

        score = (h_r + r_emb - t_r).norm(dim=-1)
        return score
```

## 3. KGAT: 知识图谱注意力网络

KGAT (KDD 2019) 将 KG 嵌入与协同过滤通过注意力机制结合：

```python
class KGAT(nn.Module):
    """Knowledge Graph Attention Network"""
    def __init__(self, n_users, n_entities, n_relations, embed_dim, n_layers):
        super().__init__()
        self.user_embedding = nn.Embedding(n_users, embed_dim)
        self.entity_embedding = nn.Embedding(n_entities, embed_dim)
        self.relation_embedding = nn.Embedding(n_relations, embed_dim)

        # 注意力层
        self.attention_layers = nn.ModuleList([
            nn.Sequential(
                nn.Linear(embed_dim * 3, 64),
                nn.ReLU(),
                nn.Linear(64, 1)
            ) for _ in range(n_layers)
        ])

    def compute_attention(self, h_emb, r_emb, t_emb, layer_idx):
        """计算 KG 中三元组的注意力权重"""
        attn_input = torch.cat([h_emb * r_emb, h_emb * t_emb, r_emb * t_emb],
                                dim=-1)
        return torch.softmax(self.attention_layers[layer_idx](attn_input), dim=0)

    def forward(self, kg_adj, ui_adj):
        """KG + CF 联合传播"""
        user_emb = self.user_embedding.weight
        entity_emb = self.entity_embedding.weight

        all_emb = torch.cat([user_emb, entity_emb], dim=0)

        for layer_idx in range(len(self.attention_layers)):
            # KG 注意力传播
            kg_msg = torch.sparse.mm(kg_adj, all_emb)
            # CF 协同过滤传播
            cf_msg = torch.sparse.mm(ui_adj, all_emb)

        return all_emb
```

## 4. 基于路径的推荐

利用 KG 中的路径作为推荐理由：

```python
class KGPolicyPath(nn.Module):
    """基于强化学习的 KG 路径推理"""
    def __init__(self, n_entities, n_relations, embed_dim, max_path_len=3):
        super().__init__()
        self.entity_embed = nn.Embedding(n_entities, embed_dim)
        self.relation_embed = nn.Embedding(n_relations, embed_dim)
        self.max_path_len = max_path_len

        # LSTM 路径编码
        self.path_lstm = nn.LSTM(embed_dim * 2, embed_dim, batch_first=True)

        # 策略网络
        self.policy = nn.Sequential(
            nn.Linear(embed_dim * 2, embed_dim),
            nn.ReLU(),
            nn.Linear(embed_dim, n_relations)
        )

    def forward(self, user_id, item_id, kg_paths):
        """给定用户和物品，评估 KG 路径的相关性"""
        user_emb = self.entity_embed(user_id)
        item_emb = self.entity_embed(item_id)

        # 编码每条路径
        path_scores = []
        for path in kg_paths:
            # path: [(entity, relation), ...]
            h = self.entity_embed(path[0][0])
            r = self.relation_embed(path[0][1])
            _, (hidden, _) = self.path_lstm(torch.cat([h, r], dim=-1).unsqueeze(0))
            path_scores.append(hidden.squeeze())

        path_repr = torch.stack(path_scores)
        score = (user_emb * path_repr * item_emb).sum(dim=-1)
        return torch.sigmoid(score)
```

## 5. KG 推荐实践

```python
# 知识图谱构建
kg_triples = [
    ('电影A', '导演', '导演X'),
    ('电影A', '演员', '演员Y'),
    ('电影A', '类型', '科幻'),
    ('导演X', '导演', '电影C'),
    ('演员Y', '主演', '电影B'),
]

# 实体对齐
def align_kg_entities(item_ids, kg_entities):
    """将推荐系统的物品 ID 与 KG 实体对齐"""
    mapping = {}
    for item_id, item_info in item_ids.items():
        title = item_info['title']
        if title in kg_entities:
            mapping[item_id] = kg_entities[title]
    return mapping

# KG 增强的嵌入初始化
def init_embeddings_with_kg(model, kg_embeddings, entity_mapping):
    """用预训练 KG 嵌入初始化物品嵌入"""
    for item_id, entity_id in entity_mapping.items():
        if entity_id in kg_embeddings:
            model.item_embedding.weight.data[item_id] = \
                torch.FloatTensor(kg_embeddings[entity_id])
```

---

**要点总结：**
- 知识图谱为推荐提供物品间的语义关联和可解释性
- TransE/TransR 等 KG 嵌入方法将图结构转为向量表示
- KGAT 通过注意力机制将 KG 信息注入协同过滤
- KG 路径可以作为推荐的可解释理由
- KG 的主要价值在于缓解冷启动和提升可解释性
