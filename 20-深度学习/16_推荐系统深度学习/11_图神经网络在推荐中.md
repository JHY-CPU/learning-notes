# 11_图神经网络在推荐中

## 1. 概述

推荐系统中的数据天然具有图结构：用户-物品交互构成二部图，社交关系构成社交图，物品之间有知识图谱。图神经网络 (GNN) 可以显式建模这些关系。

```
用户-物品交互图:
U₁ ─── I₁
 \    /  \
  I₂      I₃
 /  \    /
U₂ ─── I₄ ─── U₃
```

## 2. LightGCN

LightGCN (SIGIR 2020) 是最简洁有效的图协同过滤模型：**只保留邻域聚合，去掉特征变换和非线性激活**。

$$e_u^{(k+1)} = \sum_{i \in \mathcal{N}_u} \frac{1}{\sqrt{|\mathcal{N}_u|}\sqrt{|\mathcal{N}_i|}} e_i^{(k)}$$

最终表示：$e_u = \frac{1}{K+1}\sum_{k=0}^{K} e_u^{(k)}$

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class LightGCN(nn.Module):
    """LightGCN: 轻量级图协同过滤"""
    def __init__(self, n_users, n_items, embed_dim=64, n_layers=3):
        super().__init__()
        self.n_users = n_users
        self.n_items = n_items
        self.n_layers = n_layers

        # 初始嵌入
        self.user_embedding = nn.Embedding(n_users, embed_dim)
        self.item_embedding = nn.Embedding(n_items, embed_dim)
        nn.init.normal_(self.user_embedding.weight, std=0.1)
        nn.init.normal_(self.item_embedding.weight, std=0.1)

    def forward(self, adj_matrix):
        """
        adj_matrix: 归一化邻接矩阵 (n_users+n_items, n_users+n_items)
        """
        # 拼接初始嵌入
        all_emb = torch.cat([self.user_embedding.weight,
                             self.item_embedding.weight], dim=0)

        embs = [all_emb]
        for _ in range(self.n_layers):
            # 邻域聚合（矩阵乘法）
            all_emb = torch.sparse.mm(adj_matrix, all_emb)
            embs.append(all_emb)

        # 层组合
        embs = torch.stack(embs, dim=1)
        final_emb = embs.mean(dim=1)

        user_emb, item_emb = final_emb.split([self.n_users, self.n_items])
        return user_emb, item_emb

    def predict(self, user_emb, item_emb, user_ids, item_ids):
        """预测用户-物品评分"""
        u = user_emb[user_ids]
        i = item_emb[item_ids]
        return (u * i).sum(dim=-1)


def build_lightgcn_graph(n_users, n_items, user_item_interactions):
    """
    构建归一化邻接矩阵
    user_item_interactions: list of (user_id, item_id)
    """
    import scipy.sparse as sp
    import numpy as np

    n_nodes = n_users + n_items
    rows, cols = [], []
    for u, i in user_item_interactions:
        rows.extend([u, n_users + i])
        cols.extend([n_users + i, u])

    adj = sp.coo_matrix((np.ones(len(rows)), (rows, cols)),
                         shape=(n_nodes, n_nodes))

    # 对称归一化: D^{-1/2} A D^{-1/2}
    rowsum = np.array(adj.sum(axis=1)).flatten()
    d_inv_sqrt = np.power(rowsum, -0.5)
    d_inv_sqrt[np.isinf(d_inv_sqrt)] = 0.
    d_mat = sp.diags(d_inv_sqrt)
    norm_adj = d_mat.dot(adj).dot(d_mat)

    # 转为稀疏张量
    norm_adj = norm_adj.tocoo()
    indices = torch.LongTensor([norm_adj.row, norm_adj.col])
    values = torch.FloatTensor(norm_adj.data)
    return torch.sparse_coo_tensor(indices, values, (n_nodes, n_nodes))
```

## 3. PinSage: 工业级图推荐

PinSage (Pinterest, 2018) 是 GraphSAGE 在推荐中的应用，处理**数十亿节点**的大规模图。

class PinSageConv(nn.Module):
    """PinSage 图卷积层"""
    def __init__(self, in_dim, out_dim):
        super().__init__()
        self.linear = nn.Linear(in_dim * 2, out_dim)
        self.norm = nn.LayerNorm(out_dim)

    def forward(self, node_emb, neighbor_emb, neighbor_counts):
        """
        node_emb: (N, D) 节点自身嵌入
        neighbor_emb: (N, D) 邻居聚合嵌入
        neighbor_counts: (N,) 每个节点的邻居数
        """
        # 拼接自身和邻居信息
        combined = torch.cat([node_emb, neighbor_emb], dim=-1)
        out = self.norm(F.relu(self.linear(combined)))
        return out

## 4. 社交推荐

结合社交网络图的推荐：

```python
class SocialGCN(nn.Module):
    """社交图 + 交互图联合建模"""
    def __init__(self, n_users, n_items, embed_dim, n_layers):
        super().__init__()
        self.user_embedding = nn.Embedding(n_users, embed_dim)
        self.item_embedding = nn.Embedding(n_items, embed_dim)

        # 社交图卷积层
        self.social_convs = nn.ModuleList([
            nn.Linear(embed_dim, embed_dim) for _ in range(n_layers)
        ])
        # 交互图卷积层
        self.interaction_convs = nn.ModuleList([
            nn.Linear(embed_dim, embed_dim) for _ in range(n_layers)
        ])

    def forward(self, social_adj, interaction_adj):
        user_emb = self.user_embedding.weight
        item_emb = self.item_embedding.weight

        for social_conv, inter_conv in zip(self.social_convs, self.interaction_convs):
            # 社交信息聚合
            social_msg = social_conv(torch.sparse.mm(social_adj, user_emb))
            user_emb = F.relu(user_emb + social_msg)

            # 交互信息聚合
            all_emb = torch.cat([user_emb, item_emb], dim=0)
            inter_msg = inter_conv(torch.sparse.mm(interaction_adj, all_emb))
            user_emb = user_emb + inter_msg[:user_emb.size(0)]
            item_emb = item_emb + inter_msg[user_emb.size(0):]

        return user_emb, item_emb
```

## 5. GNN 推荐训练

```python
def train_lightgcn(model, adj, train_pairs, optimizer, device,
                   batch_size=1024, n_neg=1):
    """BPR 训练"""
    model.train()
    user_emb, item_emb = model(adj.to(device))

    total_loss = 0
    for i in range(0, len(train_pairs), batch_size):
        batch = train_pairs[i:i+batch_size]
        users = torch.LongTensor(batch[:, 0]).to(device)
        pos_items = torch.LongTensor(batch[:, 1]).to(device)

        # 负采样
        neg_items = torch.randint(0, model.n_items,
                                   (len(users) * n_neg,)).to(device)

        # BPR 损失
        u_emb = user_emb[users]
        pos_emb = item_emb[pos_items]
        neg_emb = item_emb[neg_items]

        pos_scores = (u_emb * pos_emb).sum(dim=-1)
        neg_scores = (u_emb * neg_emb).sum(dim=-1)

        loss = -F.logsigmoid(pos_scores - neg_scores).mean()

        # L2 正则化
        reg = (u_emb.norm(2) + pos_emb.norm(2) + neg_emb.norm(2)) / len(users)
        loss += 1e-5 * reg

        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        total_loss += loss.item()

    return total_loss
```

## 6. GNN 推荐模型对比

| 模型 | 图类型 | 聚合方式 | 特点 |
|------|--------|---------|------|
| LightGCN | 用户-物品二部图 | 邻域平均 | 极简，效果好 |
| NGCF | 用户-物品二部图 | 邻域 + 非线性变换 | 更复杂但不一定更好 |
| PinSage | 物品-物品图 | 随机游走采样 | 大规模工业应用 |
| DiffRec | 扩散图 | 去噪扩散 | 新范式 |

---

**要点总结：**
- LightGCN 证明了在协同过滤中，图卷积的核心是邻域聚合，而非特征变换
- 图归一化（对称归一化）对推荐效果至关重要
- PinSage 通过随机游走采样实现了超大规模图上的训练
- 社交图和交互图的联合建模可以引入额外信号
