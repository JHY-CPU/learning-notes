# 16_在线学习与实时推荐

## 1. 概述

在线学习 (Online Learning) 让模型实时更新，捕捉用户最新的兴趣变化。与离线批量训练不同，在线学习持续从新数据中学习。

```
离线训练: 数据累积 → 定期全量训练 → 部署新模型
在线学习: 数据流 → 实时/准实时更新 → 模型持续进化

延迟对比:
  离线: 小时级/天级
  在线: 秒级/分钟级
```

## 2. 特征实时更新

```python
import redis
import json

class RealTimeFeatureStore:
    """实时特征存储"""
    def __init__(self, redis_host='localhost', redis_port=6379):
        self.redis = redis.Redis(host=redis_host, port=redis_port)

    def update_user_features(self, user_id, features):
        """实时更新用户特征"""
        key = f"user_feat:{user_id}"
        self.redis.hmset(key, {k: json.dumps(v) for k, v in features.items()})
        self.redis.expire(key, 86400)  # 24 小时过期

    def get_user_features(self, user_id):
        """获取用户实时特征"""
        key = f"user_feat:{user_id}"
        data = self.redis.hgetall(key)
        return {k.decode(): json.loads(v) for k, v in data.items()}

    def update_user_behavior(self, user_id, item_id, behavior_type, timestamp):
        """实时记录用户行为"""
        key = f"user_behaviors:{user_id}"
        behavior = json.dumps({
            'item_id': item_id,
            'type': behavior_type,
            'ts': timestamp
        })
        self.redis.lpush(key, behavior)
        self.redis.ltrim(key, 0, 99)  # 只保留最近 100 条
        self.redis.expire(key, 86400)

    def get_recent_behaviors(self, user_id, n=20):
        """获取用户最近行为"""
        key = f"user_behaviors:{user_id}"
        behaviors = self.redis.lrange(key, 0, n - 1)
        return [json.loads(b) for b in behaviors]


class StreamingFeatureProcessor:
    """流式特征计算"""
    def __init__(self, window_sizes=[60, 300, 3600]):
        self.window_sizes = window_sizes  # 秒级窗口

    def compute(self, user_id, behavior_stream):
        """
        实时计算滑动窗口统计特征
        """
        features = {}
        now = behavior_stream[-1]['timestamp'] if behavior_stream else 0

        for window in self.window_sizes:
            window_behaviors = [
                b for b in behavior_stream
                if now - b['timestamp'] <= window
            ]

            features[f'click_count_{window}s'] = sum(
                1 for b in window_behaviors if b['type'] == 'click'
            )
            features[f'unique_items_{window}s'] = len(
                set(b['item_id'] for b in window_behaviors)
            )
            features[f'avg_dwell_{window}s'] = np.mean([
                b.get('dwell_time', 0) for b in window_behaviors
            ]) if window_behaviors else 0

        return features
```

## 3. 模型增量训练

```python
import torch
import torch.nn as nn

class OnlineModelUpdater:
    """在线模型更新"""
    def __init__(self, model, optimizer, lr_schedule='cosine'):
        self.model = model
        self.optimizer = optimizer
        self.global_step = 0

    def online_update(self, batch_data, lr=1e-4):
        """单步在线更新"""
        self.model.train()

        user_feat, item_feat, labels = batch_data
        predictions = self.model(user_feat, item_feat)
        loss = nn.BCELoss()(predictions, labels.float())

        self.optimizer.zero_grad()
        loss.backward()

        # 在线学习梯度裁剪更严格
        torch.nn.utils.clip_grad_norm_(self.model.parameters(), max_norm=1.0)

        self.optimizer.step()
        self.global_step += 1

        return loss.item()

    def warm_start_update(self, new_data, base_lr=1e-3, warmup_steps=100):
        """新数据到达时的学习率预热"""
        for step, batch in enumerate(new_data):
            lr = base_lr * min(1.0, step / warmup_steps)
            for pg in self.optimizer.param_groups:
                pg['lr'] = lr
            self.online_update(batch, lr)
```

## 4. 嵌入在线更新

```python
class OnlineEmbeddingUpdater:
    """嵌入表的在线更新"""
    def __init__(self, embedding_layer, feature_store):
        self.embedding = embedding_layer
        self.feature_store = feature_store
        self.update_buffer = []

    def on_interaction(self, user_id, item_id, reward):
        """记录交互并缓冲更新"""
        self.update_buffer.append({
            'user_id': user_id,
            'item_id': item_id,
            'reward': reward
        })

        # 达到批量大小时更新
        if len(self.update_buffer) >= 32:
            self.flush_updates()

    def flush_updates(self):
        """执行批量在线更新"""
        if not self.update_buffer:
            return

        user_ids = torch.LongTensor([x['user_id'] for x in self.update_buffer])
        item_ids = torch.LongTensor([x['item_id'] for x in self.update_buffer])
        rewards = torch.FloatTensor([x['reward'] for x in self.update_buffer])

        # 计算嵌入更新方向
        user_emb = self.embedding(user_ids)
        item_emb = self.embedding(item_ids)

        # 简单的在线嵌入更新
        scores = (user_emb * item_emb).sum(dim=-1)
        loss = ((scores - rewards) ** 2).mean()

        # 仅更新嵌入层
        optimizer = torch.optim.SGD(self.embedding.parameters(), lr=0.01)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        self.update_buffer.clear()
```

```python
    def handle_concept_drift(self, performance_metrics):

        # 简单的滑动窗口性能检测
        if len(performance_metrics) > 10:
            recent = np.mean(performance_metrics[-5:])
            older = np.mean(performance_metrics[-10:-5])

            if older - recent > 0.05:  # 性能下降超过 5%
                print("检测到概念漂移，触发模型重训练")
                return True
        return False
```

## 6. 在线学习架构

```
用户请求 → [实时特征服务] → [模型推理] → [推荐结果]
                ↑                                  ↓
         [特征存储]                          [用户反馈]
                ↑                                  ↓
         [流式特征计算] ←── [行为日志] ←────── [点击/购买]
                ↓
         [在线模型更新] ←── [训练数据流]
                ↓
         [模型参数更新] → [模型服务热加载]
```

## 7. 离线 vs 在线对比

| 维度 | 离线训练 | 在线学习 |
|------|---------|---------|
| 更新频率 | 小时/天 | 秒/分钟 |
| 数据利用 | 批量 | 流式 |
| 模型新鲜度 | 有延迟 | 实时 |
| 计算资源 | 集中大算力 | 分布式持续 |
| 稳定性 | 高 | 可能波动 |
| 适用场景 | 特征/嵌入更新 | 行为建模 |

---

**要点总结：**
- 实时特征更新需要高性能存储（Redis/Feature Store）
- 在线学习的嵌入更新可以快速响应新物品和新用户
- 经验回放防止灾难性遗忘
- 概念漂移检测触发模型重训练
- 工业实践中通常是离线训练 + 在线特征更新的混合方案
