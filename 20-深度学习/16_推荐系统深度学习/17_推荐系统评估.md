# 17_推荐系统评估

## 1. 评估体系总览

推荐系统评估分为三个层次：

```
离线评估 (Offline)
  ├── 排序指标: NDCG, MRR, MAP, AUC
  ├── 覆盖指标: Recall, Hit Rate, Coverage
  └── 预测指标: RMSE, MAE (评分预测)

在线评估 (Online)
  ├── A/B 测试: CTR, CVR, GMV
  ├── 用户体验: 满意度, 停留时长, 留存率
  └── 系统指标: QPS, 延迟, 多样性

用户调研 (User Study)
  ├── 问卷调查
  ├── 可用性测试
  └── 焦点小组
```

## 2. 排序指标

### 2.1 NDCG (Normalized Discounted Cumulative Gain)

$$DCG@K = \sum_{i=1}^{K} \frac{2^{rel_i} - 1}{\log_2(i+1)}$$
$$NDCG@K = \frac{DCG@K}{IDCG@K}$$

```python
import numpy as np

def dcg_at_k(relevances, k):
    """计算 DCG@K"""
    relevances = np.array(relevances[:k])
    if relevances.size == 0:
        return 0.0
    discounts = np.log2(np.arange(2, relevances.size + 2))
    return np.sum(relevances / discounts)

def ndcg_at_k(pred_items, true_items, k):
    """
    pred_items: 预测的物品列表
    true_items: 真实相关的物品集合
    k: Top-K
    """
    # 构建相关性列表
    relevances = [1.0 if item in true_items else 0.0 for item in pred_items[:k]]

    dcg = dcg_at_k(relevances, k)
    # IDCG: 最优排序的 DCG
    ideal_relevances = sorted(relevances, reverse=True)
    idcg = dcg_at_k(ideal_relevances, k)

    return dcg / idcg if idcg > 0 else 0.0
```

### 2.2 MRR (Mean Reciprocal Rank)

$$MRR = \frac{1}{|Q|} \sum_{i=1}^{|Q|} \frac{1}{\text{rank}_i}$$

```python
def mrr_at_k(pred_items, true_items, k):
    """计算 MRR@K"""
    for i, item in enumerate(pred_items[:k]):
        if item in true_items:
            return 1.0 / (i + 1)
    return 0.0
```

### 2.3 HR (Hit Rate)

$$HR@K = \frac{\text{用户数(命中)}}{\text{总用户数}}$$

```python
def hit_rate_at_k(pred_items, true_items, k):
    """HR@K: 推荐列表中是否包含真实物品"""
    return 1.0 if any(item in true_items for item in pred_items[:k]) else 0.0
```

### 2.4 MAP (Mean Average Precision)

$$AP = \frac{1}{|R|} \sum_{k=1}^{n} \text{Precision@k} \cdot \text{rel}(k)$$

```python
def average_precision(pred_items, true_items):
    """计算 AP"""
    hits = 0
    sum_precision = 0.0
    for i, item in enumerate(pred_items):
        if item in true_items:
            hits += 1
            sum_precision += hits / (i + 1)
    return sum_precision / len(true_items) if true_items else 0.0
```

## 3. AUC 与 GAUC

```python
from sklearn.metrics import roc_auc_score

def compute_auc(labels, predictions):
    """全局 AUC"""
    return roc_auc_score(labels, predictions)

def compute_gauc(labels, predictions, user_ids):
    """GAUC: Group AUC，按用户分组计算 AUC 后加权平均"""
    user_aucs = {}
    user_weights = {}

    unique_users = np.unique(user_ids)
    for user in unique_users:
        mask = user_ids == user
        user_labels = labels[mask]
        user_preds = predictions[mask]

        # 跳过只有一个类别的用户
        if len(np.unique(user_labels)) < 2:
            continue

        user_aucs[user] = roc_auc_score(user_labels, user_preds)
        user_weights[user] = mask.sum()

    # 加权平均
    total_weight = sum(user_weights.values())
    gauc = sum(user_aucs[u] * user_weights[u] / total_weight
               for u in user_aucs)
    return gauc
```

## 4. 多样性与覆盖度指标

```python
def intra_list_diversity(pred_items, item_features):
    """推荐列表内多样性（平均两两距离）"""
    if len(pred_items) < 2:
        return 0.0
    distances = []
    for i in range(len(pred_items)):
        for j in range(i + 1, len(pred_items)):
            feat_i = item_features[pred_items[i]]
            feat_j = item_features[pred_items[j]]
            dist = 1 - np.dot(feat_i, feat_j) / (
                np.linalg.norm(feat_i) * np.linalg.norm(feat_j) + 1e-8)
            distances.append(dist)
    return np.mean(distances)


def coverage(all_recommendations, n_total_items):
    """物品覆盖率：推荐过的物品占总物品的比例"""
    recommended_items = set()
    for rec_list in all_recommendations:
        recommended_items.update(rec_list)
    return len(recommended_items) / n_total_items


def novelty(pred_items, item_popularity, n_users):
    """新颖性：平均推荐物品的流行度倒数"""
    if not pred_items:
        return 0.0
    inverse_popularity = [
        np.log2(n_users / (item_popularity.get(item, 1) + 1))
        for item in pred_items
    ]
    return np.mean(inverse_popularity)


def serendipity(pred_items, true_items, user_profile):
    """惊喜度：推荐物品被喜欢但不与历史相似"""
    serendipitous = 0
    for item in pred_items:
        if item in true_items:
            # 计算与历史的相似度
            max_sim = max(
                cosine_similarity(item, hist_item)
                for hist_item in user_profile
            ) if user_profile else 0
            # 相似度越低，惊喜度越高
            if max_sim < 0.3:  # 阈值
                serendipitous += 1
    return serendipitous / len(pred_items) if pred_items else 0.0
```

## 5. 在线 A/B 测试

```python
import scipy.stats as stats

class ABTest:
    """A/B 测试框架"""
    def __init__(self, control_metric_name='ctr'):
        self.control_data = []
        self.treatment_data = []
        self.metric_name = control_metric_name

    def record(self, group, value):
        if group == 'control':
            self.control_data.append(value)
        else:
            self.treatment_data.append(value)

    def analyze(self, alpha=0.05):
        """统计显著性分析"""
        control = np.array(self.control_data)
        treatment = np.array(self.treatment_data)

        # 均值
        ctrl_mean = control.mean()
        treat_mean = treatment.mean()
        lift = (treat_mean - ctrl_mean) / ctrl_mean * 100

        # T 检验
        t_stat, p_value = stats.ttest_ind(control, treatment, equal_var=False)

        # 置信区间
        se = np.sqrt(control.var()/len(control) + treatment.var()/len(treatment))
        ci_lower = (treat_mean - ctrl_mean) - 1.96 * se
        ci_upper = (treat_mean - ctrl_mean) + 1.96 * se

        return {
            'control_mean': ctrl_mean,
            'treatment_mean': treat_mean,
            'lift_percent': lift,
            'p_value': p_value,
            'significant': p_value < alpha,
            'ci': (ci_lower, ci_upper),
            'recommend': 'launch' if p_value < alpha and lift > 0 else 'hold'
        }
```

## 6. 完整评估 Pipeline

```python
def full_evaluation_pipeline(model, test_data, all_recommendations,
                              item_features, item_popularity, n_users):
    """完整评估 pipeline"""
    results = {}

    # 离线排序指标
    ndcg_scores, hr_scores, mrr_scores = [], [], []
    for user_id, true_items in test_data.items():
        pred = model.recommend(user_id, top_k=20)
        pred_items = [item for item, score in pred]

        ndcg_scores.append(ndcg_at_k(pred_items, true_items, 10))
        hr_scores.append(hit_rate_at_k(pred_items, true_items, 10))
        mrr_scores.append(mrr_at_k(pred_items, true_items, 10))

    results['NDCG@10'] = np.mean(ndcg_scores)
    results['HR@10'] = np.mean(hr_scores)
    results['MRR@10'] = np.mean(mrr_scores)

    # 多样性和覆盖度
    all_recs = [rec for rec_list in all_recommendations.values()
                for rec in rec_list]
    results['Coverage'] = coverage(all_recommendations.values(), 10000)
    results['Novelty'] = novelty(all_recs, item_popularity, n_users)

    return results
```

## 7. 指标对比表

| 指标 | 衡量内容 | 适用阶段 | 范围 |
|------|---------|---------|------|
| NDCG@K | 排序质量 | 精排 | 0-1 |
| MRR@K | 首个命中位置 | 精排 | 0-1 |
| HR@K | 命中率 | 召回 | 0-1 |
| AUC/GAUC | 整体排序能力 | 精排 | 0.5-1 |
| Coverage | 物品覆盖率 | 整体 | 0-1 |
| Novelty | 推荐新颖性 | 整体 | 正实数 |
| CTR | 点击率 | 在线 | 0-1 |
| CVR | 转化率 | 在线 | 0-1 |

---

**要点总结：**
- 离线评估用 NDCG、HR、MRR 衡量排序质量，是最常用的评估手段
- AUC/GAUC 衡量模型的整体排序能力，GAUC 按用户分组更公平
- 多样性、覆盖度、新颖性衡量推荐的丰富程度
- A/B 测试是在线评估的金标准，需要足够的样本量保证统计显著性
- 不同阶段关注不同指标：召回看 Recall，精排看 NDCG/AUC，业务看 CTR/GMV
