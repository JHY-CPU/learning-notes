## 2. 文档理解

### 2.1 OCR + 语义理解

文档理解需要先识别文字 (OCR)，再进行语义分析：

```python
class DocumentUnderstandingPipeline:
    def __init__(self, ocr_model, vlm_model):
        self.ocr = ocr_model
        self.vlm = vlm_model

    def understand_document(self, image, question):
        # 1. OCR 提取文字
        ocr_results = self.ocr.recognize(image)  # [(bbox, text), ...]

        # 2. 构建文档上下文
        doc_text = self.format_ocr_results(ocr_results)

        # 3. VLM 理解
        prompt = f"""
以下是从文档图像中提取的文字内容:
{doc_text}

请根据图像和文字内容回答以下问题:
{question}
"""
        return self.vlm.generate(image, prompt)

    def format_ocr_results(self, results):
        """按阅读顺序格式化 OCR 结果"""
        # 按 y 坐标分组 (行)
        lines = sorted(results, key=lambda x: x[0][1])
        formatted = []
        current_line = []
        current_y = None

        for bbox, text in lines:
            y = bbox[0][1]
            if current_y is None or abs(y - current_y) > 10:
                if current_line:
                    formatted.append(" ".join(current_line))
                current_line = [text]
                current_y = y
            else:
                current_line.append(text)

        if current_line:
            formatted.append(" ".join(current_line))

        return "\n".join(formatted)
```

### 4.1 UI 元素识别

class UIUnderstanding:
    def __init__(self, vlm_model):
        self.vlm = vlm_model

    def parse_ui(self, screenshot):
        """解析 UI 界面结构"""
        prompt = """
请分析此 UI 截图，识别所有交互元素:
1. 按钮: 位置和功能
2. 输入框: 类型和提示
3. 链接: 文本和目标
4. 菜单/导航: 结构和选项
5. 文本区域: 内容和用途

以 JSON 格式输出界面结构。
"""
        return self.vlm.generate(screenshot, prompt)

    def generate_automation_script(self, screenshot, task_description):
        """生成 UI 自动化脚本"""
        prompt = f"""
分析此 UI 截图，并生成完成以下任务的自动化步骤:
任务: {task_description}

请输出:
1. 每步操作的描述
2. 目标元素的描述 (便于定位)
3. 操作类型 (点击/输入/滑动等)
4. 预期结果
"""
        return self.vlm.generate(screenshot, prompt)

    def accessibility_analysis(self, screenshot):
        """可访问性分析"""
        prompt = """
请从可访问性角度分析此 UI:
1. 文本对比度是否足够
2. 按钮大小是否合适
3. 是否有适当的 alt text
4. 键盘导航是否可行
5. 色彩方案是否对色盲友好
"""
        return self.vlm.generate(screenshot, prompt)

## 5. 多模态 RAG (检索增强生成)

### 5.1 图文检索系统

```python
class MultimodalRAG:
    def __init__(self, clip_model, llm):
        self.clip = clip_model
        self.llm = llm
        self.image_index = []
        self.text_index = []

    def index_documents(self, documents):
        """索引图文混合文档"""
        for doc in documents:
            if 'image' in doc:
                embedding = self.clip.encode_image(doc['image'])
                self.image_index.append({
                    'embedding': embedding,
                    'content': doc
                })
            if 'text' in doc:
                embedding = self.clip.encode_text(doc['text'])
                self.text_index.append({
                    'embedding': embedding,
                    'content': doc
                })

    def retrieve(self, query, top_k=5):
        """多模态检索"""
        query_embed = self.clip.encode_text(query)

        # 图像检索
        img_scores = [cosine_similarity(query_embed, item['embedding'])
                      for item in self.image_index]

        # 文本检索
        txt_scores = [cosine_similarity(query_embed, item['embedding'])
                      for item in self.text_index]

        # 合并排序
        all_results = []
        for i, score in enumerate(img_scores):
            all_results.append((score, self.image_index[i]))
        for i, score in enumerate(txt_scores):
            all_results.append((score, self.text_index[i]))

        all_results.sort(key=lambda x: x[0], reverse=True)
        return [item for _, item in all_results[:top_k]]

    def answer(self, query):
        """RAG 问答"""
        # 检索相关文档
        relevant_docs = self.retrieve(query)

        # 构建上下文
        context = self.build_context(relevant_docs)

        # 生成回答
        prompt = f"""
根据以下参考资料回答问题:

{context}

问题: {query}
"""
        return self.llm.generate(prompt)
```

## 6. 完整实战示例

### 6.1 文档处理流水线

```python
class DocumentProcessingPipeline:
    """完整的文档处理流水线"""
    def __init__(self):
        self.ocr = load_ocr_model()
        self.table_detector = load_detection_model('table')
        self.vlm = load_vlm_model('llava-1.5-13b')

    def process(self, document_image):
        results = {}

        # 1. 文档类型分类
        doc_type = self.classify_document(document_image)
        results['type'] = doc_type

        # 2. 文本提取
        ocr_results = self.ocr.recognize(document_image)
        results['text'] = ocr_results

        # 3. 表格提取
        if doc_type in ['invoice', 'report', 'form']:
            tables = self.extract_tables(document_image)
            results['tables'] = tables

        # 4. 关键信息提取
        key_info = self.extract_key_info(document_image, doc_type)
        results['key_info'] = key_info

        # 5. 生成摘要
        summary = self.generate_summary(document_image, results)
        results['summary'] = summary

        return results

    def classify_document(self, image):
        return self.vlm.generate(image, "这是什么类型的文档? (发票/报告/合同/表格/其他)")

    def extract_tables(self, image):
        tables = self.table_detector.detect(image)
        extracted = []
        for bbox in tables:
            table_img = crop_image(image, bbox)
            md = self.vlm.generate(table_img, "提取此表格为Markdown格式")
            extracted.append(md)
        return extracted

    def extract_key_info(self, image, doc_type):
        prompt = f"""
这是一份{doc_type}。请提取关键信息:
1. 日期
2. 相关人员/组织
3. 金额/数量
4. 关键条款/内容
"""
        return self.vlm.generate(image, prompt)

    def generate_summary(self, image, extracted_data):
        prompt = f"""
请根据以下提取的信息生成文档摘要:

文本内容: {extracted_data.get('text', '')[:500]}
表格数据: {extracted_data.get('tables', [])}
关键信息: {extracted_data.get('key_info', '')}

请用3-5句话概括文档核心内容。
"""
        return self.vlm.generate(image, prompt)
```
