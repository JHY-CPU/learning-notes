🔧 Go 运算符与流程控制

Go 运算符、if 初始化语句、for 唯一循环 (while/无限/range)、switch 无 break 穿透、defer 栈、类型转换。

运算符

// ========== Go 运算符 ========== package main import "fmt" func main() { // ========== 算术运算符 ========== // +, -, *, /, %, ++, -- a, b := 10, 3 fmt.Println(a + b) // 13 fmt.Println(a - b) // 7 fmt.Println(a * b) // 30 fmt.Println(a / b) // 3 (整数除法) fmt.Println(a % b) // 1 a++ // a = 11 b-- // b = 2 // ========== 比较运算符 ========== // ==, !=, <, >, <=, >= fmt.Println(a == b) // false fmt.Println(a != b) // true fmt.Println(a > b) // true // ========== 逻辑运算符 ========== // && (AND), || (OR), ! (NOT) x, y := true, false fmt.Println(x && y) // false fmt.Println(x || y) // true fmt.Println(!x) // false // 短路求值: // if isValid() && checkDB() { } // checkDB 在 isValid 为 true 时执行 // ========== 位运算符 ========== // &, |, ^, <<, >>, &^ (位清除) m := 0b1100 // 12 n := 0b1010 // 10 fmt.Println(m & n) // 1000 → 8 (AND) fmt.Println(m | n) // 1110 → 14 (OR) fmt.Println(m ^ n) // 0110 → 6 (XOR) fmt.Println(m << 1) // 11000 → 24 (左移) fmt.Println(m >> 1) // 0110 → 6 (右移) fmt.Println(m &^ n) // 0100 → 4 (位清除: m & ^n) // ========== 赋值运算符 ========== // =, +=, -=, *=, /=, %=, &=, |=, ^=, <<=, >>= c := 10 c += 5 // c = 15 c *= 2 // c = 30 }

if 与 switch

// ========== if 语句 ========== // Go 的 if 可以包含初始化语句 func main() { // 基本 if score := 85 if score >= 90 { fmt.Println("优秀") } else if score >= 80 { fmt.Println("良好") } else if score >= 60 { fmt.Println("及格") } else { fmt.Println("不及格") } // if 初始化语句 (局部变量) if result := computeScore(); result > 80 { fmt.Println("Good:", result) } else { fmt.Println("Need improve:", result) } // result 在这里不可见 // 简短 if 常用于错误处理 if err := doSomething(); err != nil { fmt.Println("错误:", err) return } // ========== switch 语句 ========== // Go 的 switch 默认 break, 无需写 break // 使用 fallthrough 穿透 // 基本 switch day := "Monday" switch day { case "Monday", "Tuesday": // 多个条件 fmt.Println("工作日") case "Saturday", "Sunday": fmt.Println("周末") default: fmt.Println("未知") } // switch 初始化语句 switch os := runtime.GOOS; os { case "darwin": fmt.Println("macOS") case "linux": fmt.Println("Linux") default: fmt.Printf("%s\n", os) } // 无表达式 switch (类似 if-else) score := 75 switch { case score >= 90: fmt.Println("A") case score >= 80: fmt.Println("B") case score >= 70: fmt.Println("C") case score >= 60: fmt.Println("D") default: fmt.Println("F") } // fallthrough: 穿透到下一个 case switch n := 1; n { case 0: fmt.Println("零") case 1: fmt.Println("一") fallthrough // 继续执行 case 2 case 2: fmt.Println("二") // case 1 穿透到此 default: fmt.Println("其他") } // 输出: 一 二 // 类型 switch var value interface{} = 42 switch v := value.(type) { case int: fmt.Println("整数:", v) case string: fmt.Println("字符串:", v) case bool: fmt.Println("布尔:", v) default: fmt.Printf("未知类型: %T\n", v) } }

for 循环

// ========== for 循环 ========== // Go 只有 for 关键字 (无 while/do-while) func main() { // ========== 1. 完整 for ========== for i := 0; i < 5; i++ { fmt.Println(i) // 0, 1, 2, 3, 4 } // ========== 2. 条件 for (类似 while) ========== count := 0 for count < 5 { fmt.Println(count) count++ } // ========== 3. 无限循环 ========== // for { // // 一直执行, 直到 break // } // ========== 4. range 迭代 ========== // 数组/切片 numbers := []int{10, 20, 30} for index, value := range numbers { fmt.Printf("索引: %d, 值: %d\n", index, value) } for _, value := range numbers { // 忽略索引 fmt.Println(value) } for index := range numbers { // 只取索引 fmt.Println(index) } // 字符串 (迭代 rune) for i, r := range "Hello" { fmt.Printf("%d: %c\n", i, r) } // map scores := map[string]int{"Alice": 90, "Bob": 85} for name, score := range scores { fmt.Printf("%s: %d\n", name, score) } for name := range scores { // 只取 key fmt.Println(name) } // channel // for item := range ch { // fmt.Println(item) // } // ========== break / continue / goto ========== for i := 0; i < 10; i++ { if i == 3 { continue // 跳过本次 } if i == 8 { break // 跳出循环 } fmt.Println(i) } // 标签 + break/goto outer: for i := 0; i < 3; i++ { for j := 0; j < 3; j++ { if i == 1 && j == 1 { break outer // 跳出外层循环 } fmt.Printf("(%d,%d) ", i, j) } } // 输出: (0,0) (0,1) (0,2) (1,0) // goto 跳转 n := 0 loop: fmt.Println(n) n++ if n < 3 { goto loop } }

defer 与 panic/recover

// ========== defer ========== // 延迟执行, 函数返回前执行 // 用于: 关闭文件/解锁/清理资源 // 后进先出 (LIFO 栈) func readFile(filename string) error { f, err := os.Open(filename) if err != nil { return err } defer f.Close() // 函数返回时关闭 // 处理文件... return nil } // defer 栈 (LIFO 顺序) func deferDemo() { defer fmt.Println("1") // 最后执行 defer fmt.Println("2") // 第二执行 defer fmt.Println("3") // 最先执行 fmt.Println("main") } // 输出: // main // 3 // 2 // 1 // defer 参数即时求值 (但执行延迟) func deferArgs() { x := 10 defer fmt.Println(x) // 10 (参数已求值) x = 20 } // 输出: 10 // defer 可修改返回值 func deferReturn() (result int) { defer func() { result *= 2 // 修改命名返回值 }() return 10 } // 返回: 20 // ========== panic ========== // 类似异常, 但 Go 鼓励返回 error 而非 panic // 不可恢复的错误才用 panic func divide(a, b int) int { if b == 0 { panic("除数不能为零") // 触发 panic } return a / b } // ========== recover ========== // 捕获 panic, 类似 try-catch // 必须在 defer 中使用 func safeDivide(a, b int) (result int, err error) { defer func() { if r := recover(); r != nil { err = fmt.Errorf("panic: %v", r) } }() return a / b, nil } // 实际应用: HTTP 中间件 func recoveryMiddleware(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { defer func() { if err := recover(); err != nil { log.Printf("panic recovered: %v", err) http.Error(w, "Internal Server Error", 500) } }() next.ServeHTTP(w, r) }) } // ========== 最佳实践 ========== // 1. 使用 error 返回值处理可预期错误 // 2. 只在真正不可恢复时用 panic // 3. recover 只在 defer 中有效 // 4. 中间件/goroutine 入口处 recover // 5. defer 确保资源释放
💡 流程控制要点: if 支持初始化语句; for 是唯一循环 (for/while/range); switch 无 break 需 fallthrough 穿透; defer LIFO 栈; panic/recover 类似异常; 标签 break/goto 控制外层循环; range 迭代数组/slice/map/channel。
