异常处理与验证

// ========== 统一异常处理体系 ========== // 1. 错误码枚举 public enum ErrorCode { // 通用 SUCCESS(0, "成功"), FAILED(1000, "操作失败"), PARAM_ERROR(1001, "参数错误"), UNAUTHORIZED(1002, "未认证"), FORBIDDEN(1003, "无权限"), NOT_FOUND(1004, "资源不存在"), CONFLICT(1005, "资源冲突"), // 业务 USER_NOT_FOUND(2001, "用户不存在"), USER_EXISTS(2002, "用户已存在"), PASSWORD_ERROR(2003, "密码错误"), ORDER_NOT_FOUND(3001, "订单不存在"), ORDER_STATUS_INVALID(3002, "订单状态无效"), INSUFFICIENT_STOCK(3003, "库存不足"); private final int code; private final String message; } // 2. 业务异常 @Getter public class BusinessException extends RuntimeException { private final int code; private final String message; public BusinessException(ErrorCode errorCode) { super(errorCode.getMessage()); this.code = errorCode.getCode(); this.message = errorCode.getMessage(); } public BusinessException(ErrorCode errorCode, Object... args) { super(String.format(errorCode.getMessage(), args)); this.code = errorCode.getCode(); this.message = String.format(errorCode.getMessage(), args); } } // 3. 统一响应 @Data @AllArgsConstructor @NoArgsConstructor public class ApiResponse<T> { private int code; private String message; private T data; private String requestId; private long timestamp; public static <T> ApiResponse<T> success(T data) { ApiResponse<T> response = new ApiResponse<>(); response.code = ErrorCode.SUCCESS.getCode(); response.message = ErrorCode.SUCCESS.getMessage(); response.data = data; response.timestamp = System.currentTimeMillis(); response.requestId = MDC.get("requestId"); return response; } public static <T> ApiResponse<T> error(ErrorCode errorCode) { ApiResponse<T> response = new ApiResponse<>(); response.code = errorCode.getCode(); response.message = errorCode.getMessage(); response.timestamp = System.currentTimeMillis(); response.requestId = MDC.get("requestId"); return response; } } // 4. 全局异常处理器 @RestControllerAdvice public class GlobalExceptionHandler { @ExceptionHandler(BusinessException.class) public ResponseEntity<ApiResponse<Void>> handleBusiness(BusinessException e) { log.warn("业务异常: code={}, message={}", e.getCode(), e.getMessage()); return ResponseEntity.status(HttpStatus.BAD_REQUEST) .body(ApiResponse.error(e.getCode(), e.getMessage())); } @ExceptionHandler(MethodArgumentNotValidException.class) public ResponseEntity<ApiResponse<Map<String, String>>> handleValidation( MethodArgumentNotValidException e) { Map<String, String> errors = new HashMap<>(); e.getBindingResult().getFieldErrors() .forEach(err -> errors.put(err.getField(), err.getDefaultMessage())); return ResponseEntity.status(HttpStatus.BAD_REQUEST) .body(ApiResponse.error(ErrorCode.PARAM_ERROR.getCode(), "参数校验失败", errors)); } @ExceptionHandler(Exception.class) public ResponseEntity<ApiResponse<Void>> handleUnknown(Exception e) { log.error("未知异常", e); return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR) .body(ApiResponse.error(ErrorCode.FAILED)); } } // ========== 分层验证 ========== // Controller: @Valid 基础格式校验 // Service: 业务规则校验 (唯一性/状态) // Repository: 数据完整性约束 (DB 约束) // Controller 校验: @PostMapping public ResponseEntity<ApiResponse<UserVO>> createUser( @Valid @RequestBody CreateUserRequest request) { // @Valid 自动校验 @NotBlank/@Email/@Size 等 } // Service 校验: public UserVO createUser(CreateUserRequest request) { // 业务校验 if (userRepository.existsByUsername(request.getUsername())) { throw new BusinessException(ErrorCode.USER_EXISTS); } // ... }

API 设计与性能

// ========== API 设计规范 ========== // ========== URL 设计 ========== // 资源名用复数: /users, /orders // 层级用斜杠: /users/{id}/orders // 查询用参数: ?page=0&size=20&sort=createdAt,desc // 动词用方法: GET/POST/PUT/PATCH/DELETE // ========== RESTful 示例 ========== // GET /users — 列表 // GET /users/{id} — 详情 // POST /users — 创建 // PUT /users/{id} — 全量更新 // PATCH /users/{id} — 部分更新 // DELETE /users/{id} — 删除 // ========== 统一响应 ========== // { // "code": 0, // "message": "success", // "data": { ... }, // "requestId": "req-xxx", // "timestamp": 1700000000000 // } // 分页响应: // { // "code": 0, // "message": "success", // "data": { // "content": [...], // "page": 0, // "size": 20, // "totalElements": 100, // "totalPages": 5 // }, // "requestId": "...", // "timestamp": 1700000000000 // } // ========== 性能优化 ========== // ========== 1. 数据库 ========== // - 合理索引 (复合索引最左前缀) // - @EntityGraph / JOIN FETCH 避免 N+1 // - 分页查询 (Pageable) // - 只查需要的字段 (DTO 投影) // ========== 2. 缓存 ========== // - 热点数据 @Cacheable // - 本地缓存 Caffeine + 分布式 Redis // - 缓存穿透/击穿/雪崩防御 // ========== 3. 异步 ========== // - 非核心逻辑 @Async // - 消息队列解耦 // - CompletableFuture 并行调用 // ========== 4. 连接池 ========== // - HikariCP 合理配置 (max=20, min=5) // - 连接超时: connectionTimeout=30000 // - 空闲回收: idleTimeout=600000 // ========== 5. JVM ========== // - 堆内存: -Xms512m -Xmx2g // - GC 策略: G1GC // - 元空间: -XX:MaxMetaspaceSize=256m // ========== 6. 压缩 ========== // server.compression.enabled=true // server.compression.min-response-size=1024 // ========== 7. 静态资源 ========== // - CDN 加速 // - HTTP 缓存 (Cache-Control) // - 资源压缩 (CSS/JS)
