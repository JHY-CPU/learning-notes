# 并行性能分析

## 1. 核心性能指标

### 加速比（Speedup）

$$S_p = \frac{T_1}{T_p}$$

- $T_1$：串行执行时间
- $T_p$：$p$ 个处理器的并行执行时间
- 理想值：$S_p = p$（线性加速）
- 超线性加速可能出现在缓存效应或搜索空间剪枝时

### 效率（Efficiency）

$$E_p = \frac{S_p}{p} = \frac{T_1}{p \cdot T_p}$$

- $E_p \in (0, 1]$，越接近1越好
- 效率下降意味着并行开销增大或负载不均衡

### 可扩展性分析

**强扩展（Strong Scaling）**：固定总问题规模，增加处理器数

```python
import matplotlib.pyplot as plt
import numpy as np

# 强扩展：不同并行比例的加速比
processors = np.arange(1, 65)
f_values = [0.90, 0.95, 0.99, 0.999]

plt.figure(figsize=(10, 6))
for f in f_values:
    speedup = 1.0 / ((1 - f) + f / processors)
    plt.plot(processors, speedup, label=f'f={f}')

plt.plot(processors, processors, 'k--', alpha=0.3, label='理想线性')
plt.xlabel('处理器数量')
plt.ylabel('加速比')
plt.title('强扩展（Amdahl定律）')
plt.legend()
plt.grid(True)
plt.savefig("strong_scaling.png")

# 弱扩展：每个处理器的工作量固定
plt.figure(figsize=(10, 6))
for f in f_values:
    speedup = processors - (1 - f) * (processors - 1)
    efficiency = speedup / processors
    plt.plot(processors, efficiency, label=f'f={f}')

plt.xlabel('处理器数量')
plt.ylabel('效率')
plt.title('弱扩展（Gustafson定律）')
plt.legend()
plt.grid(True)
plt.savefig("weak_scaling.png")
```

## 2. MPI 性能测量

### 基本计时

```c
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>
#include <math.h>

int main(int argc, char *argv[]) {
    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    // MPI_Wtime() 返回高精度时间戳（秒）
    double t_start, t_end;

    // 测量计算时间
    t_start = MPI_Wtime();
    double sum = 0.0;
    for (long i = 0; i < 100000000L; i++)
        sum += sin(i * 0.001);
    t_end = MPI_Wtime();

    printf("进程%d: 计算耗时 %.4f 秒\n", rank, t_end - t_start);

    // 测量通信时间
    int msg_size = 1024 * 1024;  // 1MB
    char *buf = (char *)malloc(msg_size);

    MPI_Barrier(MPI_COMM_WORLD);  // 同步
    t_start = MPI_Wtime();

    if (rank == 0) {
        MPI_Send(buf, msg_size, MPI_BYTE, 1, 0, MPI_COMM_WORLD);
    } else if (rank == 1) {
        MPI_Recv(buf, msg_size, MPI_BYTE, 0, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
    }

    t_end = MPI_Wtime();
    if (rank == 0) {
        double elapsed = t_end - t_start;
        double bandwidth = msg_size / elapsed / (1024 * 1024);
        printf("1MB消息: 延迟=%.6f秒, 带宽=%.2f MB/s\n", elapsed, bandwidth);
    }

    free(buf);
    MPI_Finalize();
    return 0;
}
```

```c
    double h = 1.0 / N;

    double local_sum = 0.0;
```

mpiP 是一个轻量级的 MPI 性能分析工具，无需修改代码，只需链接即可。

```bash

mpirun -np 4 ./my_program

mpiP/my_program.4.0.mpiP
```

## 4. 性能优化策略

### 通信优化

```c
// 1. 减少通信次数：合并小消息
// 差：1000次小消息
for (int i = 0; i < 1000; i++)
    MPI_Send(&data[i], 1, MPI_DOUBLE, dest, i, comm);

// 好：1次大消息
MPI_Send(data, 1000, MPI_DOUBLE, dest, 0, comm);

// 2. 使用非阻塞通信重叠计算与通信
MPI_Isend(sendbuf, n, MPI_DOUBLE, dest, tag, comm, &req);
// ... 计算 ...
MPI_Wait(&req, &status);

// 3. 使用集合通信替代点对点
// 差：O(p^2) 次通信
// 好：一次 MPI_Allreduce
MPI_Allreduce(&local, &global, 1, MPI_DOUBLE, MPI_SUM, comm);
```

### 负载均衡优化

```c
// 动态调度：主进程分发任务
if (rank == 0) {
    int next_task = 0;
    int completed = 0;
    while (completed < total_tasks) {
        int worker, task_id;
        MPI_Recv(&task_id, 1, MPI_INT, MPI_ANY_SOURCE, 0, comm, &status);
        worker = status.MPI_SOURCE;
        if (task_id >= 0) completed++;  // 完成一个任务

        if (next_task < total_tasks) {
            MPI_Send(&next_task, 1, MPI_INT, worker, 0, comm);
            next_task++;
        } else {
            int done = -1;
            MPI_Send(&done, 1, MPI_INT, worker, 0, comm);  // 发送结束信号
        }
    }
} else {
    // 工作进程
    int task_id;
    MPI_Send(&rank, 1, MPI_INT, 0, 0, comm);  // 请求任务
    while (1) {
        MPI_Recv(&task_id, 1, MPI_INT, 0, 0, comm, MPI_STATUS_IGNORE);
        if (task_id < 0) break;  // 收到结束信号

        // 执行任务
        process_task(task_id);

        // 请求下一个任务
        MPI_Send(&task_id, 1, MPI_INT, 0, 0, comm);
    }
}
```

### 内存与缓存优化

```c
// 数据局部性：按行访问二维数组（C语言行主序）
// 好：连续内存访问
for (int i = 0; i < N; i++)
    for (int j = 0; j < N; j++)
        C[i][j] += A[i][k] * B[k][j];  // A 和 C 按行访问

// 分块（Blocking/Tiling）：提高缓存命中率
int BS = 64;  // 块大小
for (int ii = 0; ii < N; ii += BS)
    for (int jj = 0; jj < N; jj += BS)
        for (int kk = 0; kk < N; kk += BS)
            for (int i = ii; i < ii + BS && i < N; i++)
                for (int j = jj; j < jj + BS && j < N; j++)
                    for (int k = kk; k < kk + BS && k < N; k++)
                        C[i][j] += A[i][k] * B[k][j];
```
