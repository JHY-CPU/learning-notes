# 并行算法设计

本章介绍几种经典的并行算法，包括并行前缀和、并行排序和并行矩阵运算。

## 1. 并行前缀和（Parallel Prefix Sum）

前缀和（Prefix Sum / Scan）是并行计算中最基础的原语之一，广泛用于流压缩、基数排序、多项式求值等。

### 串行前缀和

```c
void serial_prefix_sum(int *input, int *output, int n) {
    output[0] = input[0];
    for (int i = 1; i < n; i++) {
        output[i] = output[i - 1] + input[i];
    }
}
```

### Hillis-Steele 并行算法（O(n log n) 工作量）

每轮将距离为 $2^k$ 的元素相加：

```
输入:    3  1  4  1  5  9  2  6
k=0:     3  1  4  1  5  9  2  6   (距离1: a[i]+a[i-1])
         3  4  5  5  6 14 11  8
k=1:     3  4  5  5  6 14 11  8   (距离2)
         3  4  7  9 11 19 25 22
k=2:     3  4  7  9 14 23 36 44   (距离4)
         ↑ 结果
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <omp.h>

void hillis_steele_prefix_sum(int *input, int *output, int n) {
    memcpy(output, input, n * sizeof(int));

    for (int d = 1; d < n; d *= 2) {
        // 每一轮可以并行执行
        #pragma omp parallel for
        for (int i = d; i < n; i++) {
            output[i] = output[i] + output[i - d];
        }
    }
}
```

### Blelloch 并行算法（O(n) 工作量）

分为 up-sweep（归约）和 down-sweep（分发）两个阶段：

```c
void blelloch_prefix_sum(int *data, int n) {
    // Up-sweep（归约阶段）
    for (int d = 1; d < n; d *= 2) {
        int stride = 2 * d;
        #pragma omp parallel for
        for (int i = d - 1; i < n; i += stride) {
            data[i + d] += data[i];
        }
    }

    // 最后一个元素设为0（用于 exclusive scan）
    int last = data[n - 1];
    data[n - 1] = 0;

    // Down-sweep（分发阶段）
    for (int d = n / 2; d >= 1; d /= 2) {
        int stride = 2 * d;
        #pragma omp parallel for
        for (int i = d - 1; i < n; i += stride) {
            int temp = data[i];
            data[i] = data[i + d];
            data[i + d] += temp;
        }
    }
}
```

## 2. 并行排序

### Bitonic Sort（双调排序）

Bitonic sort 是一种比较网络排序算法，具有固定的 $O(n \log^2 n)$ 比较次数，非常适合并行化。

```c
#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

void compare_and_swap(int *a, int *b, int dir) {
    // dir=1 升序, dir=0 降序
    if ((*a > *b) == dir) {
        int tmp = *a;
        *a = *b;
        *b = tmp;
    }
}

// 合并阶段：将两个已排序的 bitonic 序列合并
void bitonic_merge(int *arr, int low, int cnt, int dir) {
    if (cnt > 1) {
        int k = cnt / 2;
        #pragma omp parallel for
        for (int i = low; i < low + k; i++) {
            compare_and_swap(&arr[i], &arr[i + k], dir);
        }
        bitonic_merge(arr, low, k, dir);
        bitonic_merge(arr, low + k, k, dir);
    }
}

// 排序阶段：递归构建 bitonic 序列
void bitonic_sort(int *arr, int low, int cnt, int dir) {
    if (cnt > 1) {
        int k = cnt / 2;
        // 前半升序，后半降序，形成 bitonic 序列
        bitonic_sort(arr, low, k, 1);
        bitonic_sort(arr, low + k, k, 0);
        bitonic_merge(arr, low, cnt, dir);
    }
}

int main() {
    int N = 1 << 20;  // 1048576
    int *arr = (int *)malloc(N * sizeof(int));

    // 初始化随机数据
    srand(42);
    for (int i = 0; i < N; i++)
        arr[i] = rand() % 1000000;

    double start = omp_get_wtime();
    bitonic_sort(arr, 0, N, 1);  // 1=升序
    double end = omp_get_wtime();

    printf("Bitonic排序 %d 个元素, 耗时 %.4f秒\n", N, end - start);

    // 验证
    int sorted = 1;
    for (int i = 1; i < N; i++) {
        if (arr[i] < arr[i - 1]) { sorted = 0; break; }
    }
    printf("排序验证: %s\n", sorted ? "正确" : "错误");

    free(arr);
    return 0;
}
```

### Sample Sort（样本排序）

Sample sort 适合分布式环境，将数据分布到多个进程/节点：

```c
// MPI Sample Sort 伪代码
void sample_sort(int *local_data, int local_n, int rank, int size, MPI_Comm comm) {
    // 1. 各进程排序本地数据
    qsort(local_data, local_n, sizeof(int), compare);

    // 2. 选取样本：每个进程取 size-1 个均匀分布的样本
    int *samples = (int *)malloc((size - 1) * sizeof(int));
    for (int i = 0; i < size - 1; i++) {
        int idx = (i + 1) * local_n / size;
        samples[i] = local_data[idx];
    }

    // 3. 根进程收集所有样本，选取 size-1 个分割器（splitters）
    int *all_samples = NULL;
    if (rank == 0)
        all_samples = (int *)malloc(size * (size - 1) * sizeof(int));
    MPI_Gather(samples, size - 1, MPI_INT,
               all_samples, size - 1, MPI_INT, 0, comm);

    int *splitters = (int *)malloc((size - 1) * sizeof(int));
    if (rank == 0) {
        int total = size * (size - 1);
        qsort(all_samples, total, sizeof(int), compare);
        for (int i = 0; i < size - 1; i++)
            splitters[i] = all_samples[(i + 1) * size];
    }
    MPI_Bcast(splitters, size - 1, MPI_INT, 0, comm);

    // 4. 根据分割器划分数据到各个桶
    // 5. Alltoall 交换数据
    // 6. 各进程排序收到的数据
    // 7. 合并结果

    free(samples);
    free(splitters);
    if (rank == 0) free(all_samples);
}
```

## 3. 并行矩阵运算

### 矩阵乘法——Cannon 算法

Cannon 算法在 $\sqrt{p} \times \sqrt{p}$ 的处理器网格上进行矩阵乘法，每个处理器持有 $n/\sqrt{p} \times n/\sqrt{p}$ 的子块：

```c
#include <stdio.h>
#include <mpi.h>

// 简化版 Cannon 算法示意
void cannon_multiply(double *local_A, double *local_B, double *local_C,
                     int block_n, int rank, int grid_dim, MPI_Comm cart_comm) {
    int coords[2];
    MPI_Cart_coords(cart_comm, rank, 2, coords);
    int row = coords[0], col = coords[1];

    // 初始对齐
    // A 沿行方向左移 row 步
    // B 沿列方向上移 col 步

    for (int step = 0; step < grid_dim; step++) {
        // local_C += local_A * local_B
        for (int i = 0; i < block_n; i++)
            for (int j = 0; j < block_n; j++)
                for (int k = 0; k < block_n; k++)
                    local_C[i * block_n + j] +=
                        local_A[i * block_n + k] * local_B[k * block_n + j];

        // A 向左循环移位一步
        // B 向上循环移位一步
        int left, right, up, down;
        MPI_Cart_shift(cart_comm, 1, 1, &left, &right);
        MPI_Cart_shift(cart_comm, 0, 1, &up, &down);

        double *tmp_A = (double *)malloc(block_n * block_n * sizeof(double));
        double *tmp_B = (double *)malloc(block_n * block_n * sizeof(double));

        MPI_Sendrecv(local_A, block_n * block_n, MPI_DOUBLE, left, 0,
                     tmp_A, block_n * block_n, MPI_DOUBLE, right, 0,
                     cart_comm, MPI_STATUS_IGNORE);
        MPI_Sendrecv(local_B, block_n * block_n, MPI_DOUBLE, up, 1,
                     tmp_B, block_n * block_n, MPI_DOUBLE, down, 1,
                     cart_comm, MPI_STATUS_IGNORE);

        // 复制回 local_A/B
        for (int i = 0; i < block_n * block_n; i++) {
            local_A[i] = tmp_A[i];
            local_B[i] = tmp_B[i];
        }
        free(tmp_A);
        free(tmp_B);
    }
}
```

### OpenMP 矩阵乘法

```c
void matmul_omp(double *A, double *B, double *C, int n) {
    // 清零 C
    #pragma omp parallel for
    for (int i = 0; i < n * n; i++) C[i] = 0.0;

    // 三重循环，外层并行化
    #pragma omp parallel for collapse(2)
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            double sum = 0.0;
            for (int k = 0; k < n; k++) {
                sum += A[i * n + k] * B[k * n + j];
            }
            C[i * n + j] = sum;
        }
    }
}
```

## 4. 并行算法设计原则

### 分解策略

- **数据分解（Data Decomposition）**: 将数据划分给各处理器（如矩阵分块）
- **任务分解（Task Decomposition）**: 将不同任务分配给不同处理器（如流水线）
- **递归分解**: 将问题递归拆分（如分治算法）

### 负载均衡

- **静态均衡**: 预先均匀分配（适合规则问题）
- **动态均衡**: 运行时动态分配任务（适合不规则问题）
- **工作窃取（Work Stealing）**: 空闲线程从忙碌线程的任务队列中取任务

### 通信最小化

- 增大计算/通信比
- 合并小消息为大消息
- 重叠通信与计算
