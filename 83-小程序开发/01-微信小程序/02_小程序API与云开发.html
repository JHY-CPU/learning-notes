02 - 小程序API与云开发

一、网络请求 API

1.1 wx.request

// 基本用法 wx.request({ url: 'https://api.example.com/data', method: 'GET', data: { page: 1, size: 20 }, header: { 'content-type': 'application/json' }, success(res) { console.log('请求成功', res.data); }, fail(err) { console.error('请求失败', err); }, complete() { console.log('请求完成'); } }) // 封装为Promise（推荐） const request = (options) => { return new Promise((resolve, reject) => { wx.request({ url: getApp().globalData.baseUrl + options.url, method: options.method || 'GET', data: options.data, header: { 'content-type': 'application/json', 'Authorization': wx.getStorageSync('token') }, success: (res) => { if (res.statusCode === 200) resolve(res.data); else reject(res); }, fail: reject }); }); } // 使用封装后的request async function loadData() { try { const res = await request({ url: '/api/users' }); this.setData({ users: res.data }); } catch (err) { wx.showToast({ title: '加载失败', icon: 'error' }); } }

1.2 网络相关配置

  • 域名白名单：小程序只能请求在后台配置的合法域名（开发阶段可关闭域名校验）
  • HTTPS要求：正式环境必须使用HTTPS
  • 超时设置：默认60秒，可在app.json中配置networkTimeout
  • 并发限制：同时最多10个网络请求

二、本地存储 API

2.1 同步与异步API

// ===== 同步API（简单直接） ===== wx.setStorageSync('token', 'abc123'); wx.setStorageSync('userInfo', { name: '张三', age: 25 }); const token = wx.getStorageSync('token'); const userInfo = wx.getStorageSync('userInfo'); wx.removeStorageSync('token'); wx.clearStorageSync(); // 清空所有本地存储 // ===== 异步API（推荐，不阻塞主线程） ===== wx.setStorage({ key: 'config', data: { theme: 'dark', lang: 'zh-CN' }, success() { console.log('存储成功'); } }); wx.getStorage({ key: 'config', success(res) { console.log(res.data); } }); // 异步Promise封装 const getStorage = (key) => new Promise((resolve) => { wx.getStorage({ key, success: (res) => resolve(res.data) }); }); // 获取存储信息 wx.getStorageInfo({ success(res) { console.log('keys:', res.keys); console.log('当前大小:', res.currentSize, 'KB'); console.log('限制大小:', res.limitSize, 'KB'); } });

2.2 存储限制

维度限制
单个key数据大小不超过1MB
总存储空间不超过10MB
清理策略存储满后新数据存储失败，不会自动清理

四、用户授权与微信登录

4.1 微信登录流程

小程序端 业务服务器 微信服务器 ────────── ────────── ────────── wx.login() → 获取code code ──────────→ code + appId + secret ──→ ←── openid / session_key 生成自定义token token ←───────── 存储token到本地

4.2 登录实现

// 微信登录 async function wxLogin() { // 1. 获取code const loginRes = await new Promise((resolve) => { wx.login({ success: resolve }); }); const code = loginRes.code; // 2. 发送code到后端换取openid和自定义token const tokenRes = await new Promise((resolve, reject) => { wx.request({ url: 'https://api.example.com/login', method: 'POST', data: { code }, success: resolve, fail: reject }); }); // 3. 存储token wx.setStorageSync('token', tokenRes.data.token); getApp().globalData.isLogin = true; } // 获取用户信息（新方式：使用头像昵称填写组件） // <button open-type="chooseAvatar" bindchooseavatar="onChooseAvatar"> function onChooseAvatar(e) { this.setData({ avatarUrl: e.detail.avatarUrl }); } // 获取手机号（需用户主动触发） // <button open-type="getPhoneNumber" bindgetphonenumber="onGetPhone"> function onGetPhone(e) { if (e.detail.code) { // 将code发送到后端换取手机号 wx.request({ url: '/api/bindPhone', data: { code: e.detail.code } }); } }

// 微信支付（需要商户号配置） async function wxPay(orderId) { // 1. 后端生成支付参数 const payRes = await request({ url: '/api/pay/create', method: 'POST', data: { orderId } }); // 2. 调起微信支付 const result = await new Promise((resolve, reject) => { wx.requestPayment({ timeStamp: payRes.timeStamp, nonceStr: payRes.nonceStr, package: payRes.package, // prepay_id=xxx signType: payRes.signType, // RSA paySign: payRes.paySign, success: resolve, fail: reject }); }); // 3. 支付成功，查询订单状态 wx.showToast({ title: '支付成功', icon: 'success' }); }

五、微信云开发

5.1 云开发概述

微信云开发是微信官方提供的 Serverless 后端服务，包含云函数、云数据库、云存储三大核心能力，无需搭建服务器即可开发完整的小程序后端。

┌──── 小程序前端 ────┐ │ │ │ wx.cloud.callFunction() ──→ 云函数（Node.js） │ wx.cloud.database() ──→ 云数据库（MongoDB） │ wx.cloud.uploadFile() ──→ 云存储（对象存储） │ │ └──────────────────────┘ 无需后端服务器，无需运维，按调用次数计费

5.2 云开发初始化

// app.js 中初始化云开发 App({ onLaunch() { wx.cloud.init({ env: 'prod-xxx', // 云开发环境ID traceUser: true // 用户访问记录 }); } })

5.3 云函数

// cloudfunctions/hello/index.js const cloud = require('wx-server-sdk'); cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV }); const db = cloud.database(); // 云函数入口 exports.main = async (event, context) => { const { OPENID, APPID } = cloud.getWXContext(); // 查询数据库 const users = await db.collection('users') .where({ status: 'active' }) .orderBy('createdAt', 'desc') .limit(20) .get(); // 返回结果 return { openid: OPENID, users: users.data, message: 'Hello from cloud function!' }; }; // 小程序端调用云函数 const res = await wx.cloud.callFunction({ name: 'hello', data: { page: 1 } }); console.log(res.result); // { openid, users, message }

5.4 云数据库

// 获取数据库引用 const db = wx.cloud.database(); const _ = db.command; // 数据库操作符 // ===== 增删改查 ===== // 创建集合中的记录 const addRes = await db.collection('todos').add({ data: { title: '学习小程序云开发', done: false, createdAt: db.serverDate() } }); // 查询单条 const todo = await db.collection('todos').doc('todo-id').get(); // 条件查询 const list = await db.collection('todos') .where({ done: false, createdAt: _.gte(new Date('2024-01-01')) }) .orderBy('createdAt', 'desc') .skip(0) .limit(20) .get(); // 更新记录 await db.collection('todos').doc('todo-id').update({ data: { done: true, completedAt: db.serverDate() } }); // 删除记录 await db.collection('todos').doc('todo-id').remove(); // 聚合查询 const stats = await db.collection('orders') .aggregate() .match({ status: 'completed' }) .group({ _id: '$userId', totalAmount: _.sum('$amount'), orderCount: _.sum(1) }) .end();

5.6 云开发与传统后端对比

维度云开发传统后端
服务器无需管理需自行部署运维
数据库云数据库（MongoDB）自行搭建MySQL/Redis等
存储云存储自行搭建OSS/文件系统
运维免运维、自动扩缩容需自行维护
成本免费额度+按量付费固定服务器成本
灵活性受限于云开发能力完全自由
适用阶段MVP/小型项目/快速验证中大型项目/复杂业务
