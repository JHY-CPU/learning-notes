# 容器化测试

## 一、概念说明

容器化测试使用Docker提供一致的测试环境，确保测试结果可重现。

## 二、测试Dockerfile

```dockerfile
FROM node:20-alpine
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
CMD ["npm", "test"]
```

## 三、集成测试配置

```yaml
# docker-compose.test.yml
version: '3.8'
services:
  app:
    build: .
    environment:
      - DATABASE_URL=postgres://db:5432/testdb
    depends_on:
      db:
        condition: service_healthy
  
  db:
    image: postgres:15-alpine
    environment:
      - POSTGRES_DB=testdb
      - POSTGRES_PASSWORD=test
    healthcheck:
      test: ["CMD-SHELL", "pg_isready"]
      interval: 5s
      timeout: 5s
      retries: 5
```

## 四、CI集成

```yaml
test:
  script:
    - docker-compose -f docker-compose.test.yml up -d
    - docker-compose -f docker-compose.test.yml run app npm test
    - docker-compose -f docker-compose.test.yml down
```

## 五、注意事项

1. **环境隔离**：每个测试使用独立环境
2. **数据清理**：测试后清理测试数据
3. **健康检查**：确保服务就绪后才运行测试
4. **并行测试**：容器支持并行测试

## 六、Playwright容器化测试

```yaml
# docker-compose.playwright.yml
version: '3.8'
services:
  playwright:
    image: mcr.microsoft.com/playwright:v1.40.0-jammy
    working_dir: /app
    volumes:
      - .:/app
      - ./test-results:/app/test-results
    environment:
      - BASE_URL=http://app:3000
    command: npx playwright test
    depends_on:
      app:
        condition: service_healthy

  app:
    build:
      context: .
      target: production
    environment:
      - NODE_ENV=test
      - DATABASE_URL=postgres://postgres:postgres@db:5432/testdb
    healthcheck:
      test: ["CMD", "wget", "-q", "--spider", "http://localhost:3000/health"]
      interval: 10s
      timeout: 5s
      retries: 5
    depends_on:
      db:
        condition: service_healthy

  db:
    image: postgres:15-alpine
    environment:
      POSTGRES_DB: testdb
      POSTGRES_PASSWORD: postgres
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 5s
      timeout: 5s
      retries: 5
```

## 七、Cypress容器化测试

```yaml
# docker-compose.cypress.yml
version: '3.8'
services:
  cypress:
    image: cypress/included:13.6.0
    working_dir: /app
    volumes:
      - .:/app
      - ./cypress/results:/app/cypress/results
    environment:
      - CYPRESS_BASE_URL=http://app:3000
      - CYPRESS_VIDEO=true
    command: cypress run --reporter junit --reporter-options mochaFile=cypress/results/output.xml
    depends_on:
      - app

  app:
    build: .
    environment:
      - NODE_ENV=test
```

## 八、测试容器网络配置

```yaml
version: '3.8'
services:
  test-runner:
    networks:
      - test-frontend
      - test-backend
    command: npm test

  frontend:
    networks:
      - test-frontend

  backend:
    networks:
      - test-backend
    depends_on:
      - db
      - redis

  db:
    networks:
      - test-backend

  redis:
    networks:
      - test-backend

networks:
  test-frontend:
    driver: bridge
  test-backend:
    driver: bridge
```

## 九、并行测试执行

```yaml
# 使用多个容器并行运行测试
test-parallel:
  script:
    - |
      docker-compose -f docker-compose.ci.yml up -d db redis
      # 等待服务就绪
      sleep 10
      # 并行运行测试套件
      docker-compose -f docker-compose.ci.yml run --rm -d app npm run test:unit &
      docker-compose -f docker-compose.ci.yml run --rm -d app npm run test:integration &
      docker-compose -f docker-compose.ci.yml run --rm -d app npm run test:e2e &
      wait
```
