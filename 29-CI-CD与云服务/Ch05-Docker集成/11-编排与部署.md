# 编排与部署

## 一、Docker Compose部署

```yaml
# 生产环境配置
version: '3.8'
services:
  app:
    image: myregistry/myapp:latest
    deploy:
      replicas: 3
      resources:
        limits:
          cpus: '0.5'
          memory: 512M
    restart: always
  
  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    depends_on:
      - app
```

## 二、Docker Swarm

```yaml
# Stack部署
docker stack deploy -c docker-compose.yml myapp

# 服务管理
docker service ls
docker service scale myapp_app=5
```

## 三、CI/CD部署

```yaml
deploy:
  script:
    - ssh deploy@server "cd /app && docker-compose pull && docker-compose up -d"
```

## 四、滚动更新

```yaml
deploy:

```

## 六、Docker Swarm Stack部署

```yaml
# docker-stack.yml
version: '3.8'
services:
  web:
    image: myapp:latest
    deploy:
      replicas: 3
      update_config:
        parallelism: 1
        delay: 30s
        order: start-first
        failure_action: rollback
      rollback_config:
        parallelism: 0
        order: stop-first
      restart_policy:
        condition: any
        delay: 5s
        max_attempts: 3
        window: 120s
      resources:
        limits:
          cpus: '0.5'
          memory: 512M
        reservations:
          cpus: '0.25'
          memory: 256M
    networks:
      - webnet
    healthcheck:
      test: ["CMD", "wget", "-q", "--spider", "http://localhost:3000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    deploy:
      replicas: 2
    depends_on:
      - web
    networks:
      - webnet

  redis:
    image: redis:7-alpine
    deploy:
      replicas: 1
      resources:
        limits:
          memory: 256M
    networks:
      - webnet

networks:
  webnet:
    driver: overlay
```

```bash
# 部署Stack
docker stack deploy -c docker-stack.yml myapp

# 查看服务状态
docker stack services myapp
docker service ls
docker service ps myapp_web

# 扩容
docker service scale myapp_web=5

# 滚动更新
docker service update --image myapp:v2 myapp_web
```

## 七、Kubernetes部署清单

```yaml
# k8s/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: myapp
spec:
  replicas: 3
  selector:
    matchLabels:
      app: myapp
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0
  template:
    metadata:
      labels:
        app: myapp
    spec:
      containers:
        - name: myapp
          image: myapp:latest
          ports:
            - containerPort: 3000
          resources:
            requests:
              memory: "256Mi"
              cpu: "250m"
            limits:
              memory: "512Mi"
              cpu: "500m"
          livenessProbe:
            httpGet:
              path: /health
              port: 3000
            initialDelaySeconds: 30
            periodSeconds: 10
          readinessProbe:
            httpGet:
              path: /ready
              port: 3000
            initialDelaySeconds: 5
            periodSeconds: 5
          env:
            - name: NODE_ENV
              value: "production"
            - name: DATABASE_URL
              valueFrom:
                secretKeyRef:
                  name: myapp-secrets
                  key: database-url
```

## 八、部署后的验证

```bash
#!/bin/bash
# post-deploy-verify.sh

echo "=== Health Check ==="
for i in {1..30}; do
    if curl -f http://localhost:3000/health; then
        echo "Health check passed"
        break
    fi
    echo "Waiting for health check... ($i/30)"
    sleep 5
done

echo "=== Smoke Test ==="
curl -f http://localhost:3000/api/smoke || exit 1

echo "=== Container Status ==="
docker ps --filter "name=myapp"
docker stats --no-stream
```
