# 并行执行

## 一、并行阶段

```groovy
stage('Test') {
    parallel {
        stage('Unit Test') {
            steps {
                sh 'npm run test:unit'
            }
        }
        stage('Integration Test') {
            steps {
                sh 'npm run test:integration'
            }
        }
        stage('E2E Test') {
            steps {
                sh 'npm run test:e2e'
            }
        }
    }
}
```

## 二、并行矩阵

```groovy
stage('Matrix') {
    matrix {
        axes {
            axis {
                name 'PLATFORM'
                values 'linux', 'windows', 'mac'
            }
            axis {
                name 'BROWSER'
                values 'chrome', 'firefox'
            }
        }
        stages {
            stage('Test') {
                steps {
                    sh "echo ${PLATFORM} ${BROWSER}"
                }
            }
        }
    }
}
```

## 三、failFast

```groovy
stage('Test') {
    failFast true  # 一个失败则取消其他
    parallel {
        stage('A') { steps { sh 'test-a' } }
        stage('B') { steps { sh 'test-b' } }
    }
}
```

## 四、注意事项

1. **独立性**：并行阶段必须相互独立
2. **资源竞争**：注意并行任务的资源竞争
3. **日志管理**：并行任务的日志可能混杂
4. **超时设置**：为并行任务设置超时

## 五、条件并行执行

```groovy
stage('Test') {
    parallel {
        stage('Unit Tests') {
            steps {
                sh 'npm run test:unit'
            }
            post {
                always {
                    junit 'test-results/unit.xml'
                }
            }
        }

        stage('Integration Tests') {
            when {
                expression { params.RUN_INTEGRATION }
            }
            agent {
                docker {
                    image 'node:20-alpine'
                }
            }
            steps {
                sh 'npm run test:integration'
            }
            post {
                always {
                    junit 'test-results/integration.xml'
                }
            }
        }

        stage('E2E Tests') {
            when {
                branch 'main'
            }
            agent {
                label 'chrome'
            }
            steps {
                sh 'npx cypress run'
            }
        }
    }
}
```

## 六、矩阵构建详细示例

```groovy
stage('Matrix Build') {
    matrix {
        axes {
            axis {
                name 'PLATFORM'
                values 'linux', 'windows', 'macos'
            }
            axis {
                name 'VERSION'
                values '18', '20', '21'
            }
        }
        excludes {
            exclude {
                axis {
                    name 'PLATFORM'
                    values 'windows'
                }
                axis {
                    name 'VERSION'
                    values '18'
                }
            }
        }
        stages {
            stage('Test') {
                steps {
                    script {
                        if (PLATFORM == 'windows') {
                            bat 'npm test'
                        } else {
                            sh 'npm test'
                        }
                    }
                }
            }
        }
    }
}
```

## 七、并行执行资源控制

```groovy
// 限制并发Agent数量
pipeline {
    agent none
    options {
        parallelsAlwaysFailFast()  // 任一并行失败则全部停止
    }
    stages {
        stage('Parallel') {
            parallel {
                stage('Task A') {
                    agent { label 'linux' }
                    steps { sh 'make test-a' }
                }
                stage('Task B') {
                    agent { label 'linux' }
                    steps { sh 'make test-b' }
                }
                stage('Task C') {
                    agent { label 'linux' }
                    steps { sh 'make test-c' }
                }
            }
        }
    }
}
```

## 八、并行测试报告合并

```groovy
stage('Test') {
    parallel {
        stage('Browser Chrome') {
            steps {
                sh 'npx playwright test --project=chromium'
            }
        }
        stage('Browser Firefox') {
            steps {
                sh 'npx playwright test --project=firefox'
            }
        }
    }
    post {
        always {
            sh 'npx playwright merge-reports ./all-results'
            publishHTML(target: [
                reportDir: 'playwright-report',
                reportFiles: 'index.html',
                reportName: 'Playwright Report'
            ])
        }
    }
}
```
