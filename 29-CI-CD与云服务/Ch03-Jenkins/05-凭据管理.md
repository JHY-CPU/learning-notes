# 凭据管理

## 一、概念说明

Jenkins凭据管理安全存储敏感信息，如密码、Token、SSH密钥等。

## 二、添加凭据

```bash
# Web界面
# Manage Jenkins > Credentials > System > Global credentials

# 凭据类型
# Username with password
# Secret text
# Secret file
# SSH Username with private key
# Certificate
```

## 三、在Pipeline中使用

```groovy
pipeline {
    agent any
    environment {
        DOCKER_CREDS = credentials('docker-credentials')
    }
    steps {
        withCredentials([usernamePassword(
            credentialsId: 'github-creds',
            usernameVariable: 'USER',
            passwordVariable: 'PASS'
        )]) {
            sh 'echo $USER'
        }
        
        withCredentials([string(
            credentialsId: 'api-key',
            variable: 'API_KEY'
        )]) {
            sh 'curl -H "Authorization: $API_KEY" https://api.example.com'
        }
    }
}
```

## 四、SSH凭据

```groovy
steps {
    sshagent(credentials: ['ssh-key-id']) {
        sh 'git clone git@github.com:repo.git'
    }
}
```

## 五、注意事项

1. **最小权限**：只授予必要的凭据
2. **定期轮换**：定期更新凭据
3. **不要硬编码**：永远不要在代码中硬编码凭据
4. **访问控制**：限制凭据的访问范围

## 六、Docker Registry凭据

```groovy
pipeline {
    agent any
    stages {
        stage('Push Image') {
            steps {
                withCredentials([usernamePassword(
                    credentialsId: 'docker-hub',
                    usernameVariable: 'DOCKER_USER',
                    passwordVariable: 'DOCKER_PASS'
                )]) {
                    sh '''
                        echo $DOCKER_PASS | docker login -u $DOCKER_USER --password-stdin
                        docker build -t $DOCKER_USER/myapp:${BUILD_NUMBER} .
                        docker push $DOCKER_USER/myapp:${BUILD_NUMBER}
                    '''
                }
            }
        }
    }
}
```

## 七、AWS凭据管理

```groovy
pipeline {
    agent any
    stages {
        stage('Deploy to AWS') {
            steps {
                withCredentials([[
                    $class: 'AmazonWebServicesCredentialsBinding',
                    credentialsId: 'aws-credentials',
                    accessKeyVariable: 'AWS_ACCESS_KEY_ID',
                    secretKeyVariable: 'AWS_SECRET_ACCESS_KEY'
                ]]) {
                    sh '''
                        aws configure set region us-east-1
                        aws s3 sync ./dist s3://my-bucket
                        aws ecs update-service --cluster prod --service web --force-new-deployment
                    '''
                }
            }
        }
    }
}
```

## 八、文件类型凭据

```groovy
pipeline {
    agent any
    stages {
        stage('Use Config File') {
            steps {
                withCredentials([file(credentialsId: 'kubeconfig', variable: 'KUBECONFIG')]) {
                    sh 'kubectl get pods'
                }
            }
        }

        stage('Use Secret File') {
            steps {
                withCredentials([file(credentialsId: 'gcp-key', variable: 'GOOGLE_APPLICATION_CREDENTIALS')]) {
                    sh 'gcloud auth activate-service-account --key-file=$GOOGLE_APPLICATION_CREDENTIALS'
                }
            }
        }
    }
}
```

## 九、凭据安全最佳实践

```groovy
// 不要在脚本中打印凭据
// 错误示范
sh "echo ${MY_SECRET}"  // Secret会出现在日志中

// 正确示范
withCredentials([string(credentialsId: 'my-secret', variable: 'SECRET')]) {
    sh 'echo $SECRET'  // Jenkins会自动屏蔽
}

// 使用Credential ID引用，不要直接使用值
environment {
    // 好
    TOKEN = credentials('api-token')
    // 不好
    // TOKEN = 'actual-token-value'
}
```
