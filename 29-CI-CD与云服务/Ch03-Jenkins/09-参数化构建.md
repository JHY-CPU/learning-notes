# 参数化构建

## 一、参数类型

```groovy
pipeline {
    parameters {
        string(name: 'VERSION', defaultValue: '1.0.0', description: 'Version')
        choice(name: 'ENV', choices: ['dev', 'staging', 'prod'], description: 'Environment')
        booleanParam(name: 'DEPLOY', defaultValue: false, description: 'Deploy?')
        password(name: 'SECRET', defaultValue: '', description: 'Secret')
    }
    stages {
        stage('Deploy') {
            when { expression { params.DEPLOY } }
            steps {
                sh "./deploy.sh ${params.VERSION} ${params.ENV}"
            }
        }
    }
}
```

## 二、使用参数

```groovy
steps {
    echo "Version: ${params.VERSION}"
    echo "Environment: ${params.ENV}"
}
```

## 三、触发参数化构建

```bash
# URL触发
http://jenkins-url/job/myjob/buildWithParameters?VERSION=1.0.0&ENV=prod

# CLI触发
java -jar jenkins-cli.jar build myjob -p VERSION=1.0.0
```

## 四、注意事项

1. **默认值**：为参数设置合理的默认值
2. **验证**：在Pipeline中验证参数值
3. **安全**：敏感参数使用password类型
4. **文档**：为每个参数添加描述

## 五、完整参数化流水线

```groovy
pipeline {
    agent any

    parameters {
        string(
            name: 'VERSION',
            defaultValue: '1.0.0',
            description: '应用版本号'
        )
        choice(
            name: 'ENVIRONMENT',
            choices: ['development', 'staging', 'production'],
            description: '部署环境'
        )
        booleanParam(
            name: 'RUN_TESTS',
            defaultValue: true,
            description: '是否运行测试'
        )
        booleanParam(
            name: 'DEPLOY',
            defaultValue: false,
            description: '是否部署'
        )
        password(
            name: 'SECRET_KEY',
            defaultValue: '',
            description: '密钥（不会保存到构建历史）'
        )
        text(
            name: 'CHANGELOG',
            defaultValue: '',
            description: '变更日志'
        )
        file(
            name: 'CONFIG_FILE',
            description: '配置文件'
        )
    }

    stages {
        stage('Validate') {
            steps {
                script {
                    if (params.VERSION ==~ /^\d+\.\d+\.\d+$/) {
                        echo "版本号有效: ${params.VERSION}"
                    } else {
                        error("无效的版本号格式: ${params.VERSION}")
                    }

                    if (params.ENVIRONMENT == 'production' && !params.DEPLOY) {
                        error("生产环境部署必须勾选DEPLOY选项")
                    }
                }
            }
        }

        stage('Build') {
            steps {
                sh "VERSION=${params.VERSION} npm run build"
            }
        }

        stage('Test') {
            when {
                expression { params.RUN_TESTS }
            }
            steps {
                sh 'npm test'
            }
        }

        stage('Deploy') {
            when {
                expression { params.DEPLOY }
            }
            steps {
                sh "./deploy.sh ${params.ENVIRONMENT} ${params.VERSION}"
            }
        }
    }
}
```

## 六、Active Choices参数（动态参数）

```groovy
// 需要安装 Active Choices 插件
properties([
    parameters([
        [$class: 'ChoiceParameter',
            name: 'SERVICE',
            choiceType: 'PT_SINGLE_SELECT',
            script: [$class: 'GroovyScript',
                script: [$class: 'SecureGroovyScript',
                    script: 'return ["web", "api", "worker"]',
                    sandbox: true
                ]
            ]
        ],
        [$class: 'CascadeChoiceParameter',
            name: 'VERSION',
            choiceType: 'PT_SINGLE_SELECT',
            referencedParameters: 'SERVICE',
            script: [$class: 'GroovyScript',
                script: [$class: 'SecureGroovyScript',
                    script: '''
                        if (SERVICE == "web") return ["1.0.0", "1.1.0", "2.0.0"]
                        if (SERVICE == "api") return ["2.0.0", "2.1.0", "3.0.0"]
                        return ["1.0.0"]
                    ''',
                    sandbox: true
                ]
            ]
        ]
    ])
])
```

## 七、从URL触发参数化构建

```bash
# 通过URL触发构建（需要Build Token Root插件）
curl -X POST "http://jenkins-url/job/myjob/buildWithParameters" \
  --user admin:api-token \
  --data-urlencode "VERSION=1.2.3" \
  --data-urlencode "ENVIRONMENT=staging" \
  --data-urlencode "DEPLOY=true"

# 使用Jenkins CLI
java -jar jenkins-cli.jar \
  -s http://jenkins-url/ \
  -auth admin:api-token \
  build myjob \
  -p VERSION=1.2.3 \
  -p ENVIRONMENT=staging \
  -p DEPLOY=true \
  -s \
  -v
```
