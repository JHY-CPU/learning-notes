# 流水线阶段

## 一、阶段定义

```groovy
stages {
    stage('Build') {
        steps {
            sh 'mvn clean package'
        }
    }
    
    stage('Test') {
        parallel {
            stage('Unit Test') {
                steps { sh 'mvn test' }
            }
            stage('Integration Test') {
                steps { sh 'mvn verify' }
            }
        }
    }
    
    stage('Deploy') {
        when { branch 'main' }
        steps { sh './deploy.sh' }
    }
}
```

## 二、条件执行

```groovy
stage('Deploy') {
    when {
        branch 'main'
        expression { params.DEPLOY == 'true' }
        changeRequest()
    }
    steps { sh './deploy.sh' }
}
```

## 三、Post操作

```groovy
post {
    always { echo 'Always runs' }
    success { notifySlack('Success') }
    failure { notifySlack('Failed') }
    unstable { echo 'Tests failed' }
    cleanup { cleanWs() }
}
```

## 四、注意事项

1. **阶段命名**：使用描述性名称
2. **并行阶段**：独立测试并行运行
3. **条件逻辑**：使用when控制执行
4. **清理资源**：post中清理临时文件

## 五、when条件详解

```groovy
stage('Deploy') {
    when {
        branch 'main'                        // 主分支
        changeRequest()                      // PR/MR
        tag pattern: "v\\d+", comparator: "REGEXP"  // 版本标签
        expression { params.DEPLOY == 'true' }      // 参数条件
        environment name: 'DEPLOY_ENV', value: 'prod' // 环境变量
        allOf {                              // 全部满足
            branch 'main'
            expression { currentBuild.duration > 60 }
        }
        anyOf {                              // 任一满足
            branch 'main'
            branch 'develop'
        }
        not {                                // 否定
            branch 'hotfix/*'
        }
    }
    steps {
        sh './deploy.sh'
    }
}
```

## 六、try-catch错误处理

```groovy
stage('Deploy with Retry') {
    steps {
        script {
            retry(3) {
                sh './deploy.sh'
            }
        }
    }
}

stage('Deploy with Timeout') {
    steps {
        timeout(time: 10, unit: 'MINUTES') {
            sh './deploy.sh'
        }
    }
}

stage('Deploy with Fallback') {
    steps {
        script {
            try {
                sh './deploy-primary.sh'
            } catch (Exception e) {
                echo "Primary deploy failed: ${e.message}"
                sh './deploy-fallback.sh'
            }
        }
    }
}
```

## 七、共享工作区（stash/unstash）

```groovy
pipeline {
    agent none

    stages {
        stage('Build') {
            agent { label 'linux' }
            steps {
                sh 'npm run build'
                stash includes: 'dist/**', name: 'build-output'
            }
        }

        stage('Test on Linux') {
            agent { label 'linux' }
            steps {
                unstash 'build-output'
                sh 'npm test'
            }
        }

        stage('Test on Windows') {
            agent { label 'windows' }
            steps {
                unstash 'build-output'
                bat 'npm test'
            }
        }
    }
}
```
