# 继承与拷贝控制

## 一、概念说明

派生类的拷贝/移动操作（C++标准 §12.8）必须正确处理基类部分。通常派生类的拷贝构造函数调用基类的拷贝构造函数。如果基类的拷贝/移动操作被删除或不可访问，派生类对应的操作也会被隐式删除。

### 1.1 五法则在继承中的应用

| 操作 | 派生类如何处理基类 |
|------|-------------------|
| 拷贝构造 | 调用基类拷贝构造（非默认构造） |
| 拷贝赋值 | 调用基类拷贝赋值 |
| 移动构造 | 调用基类移动构造 |
| 移动赋值 | 调用基类移动赋值 |
| 析构 | 自动调用基类析构 |

## 二、具体用法

### 2.1 派生类拷贝控制

```cpp
#include <iostream>

class Base {
    int baseData;
public:
    Base(int d = 0) : baseData(d) {}
    Base(const Base& o) : baseData(o.baseData) {
        std::cout << "Base拷贝构造" << std::endl;
    }
    Base& operator=(const Base& o) {
        baseData = o.baseData;
        std::cout << "Base拷贝赋值" << std::endl;
        return *this;
    }
    Base(Base&& o) noexcept : baseData(o.baseData) {
        o.baseData = 0;
        std::cout << "Base移动构造" << std::endl;
    }
    Base& operator=(Base&& o) noexcept {
        baseData = o.baseData;
        o.baseData = 0;
        std::cout << "Base移动赋值" << std::endl;
        return *this;
    }
    virtual ~Base() = default;
    int get() const { return baseData; }
};

class Derived : public Base {
    int derivedData;
public:
    Derived(int b = 0, int d = 0) : Base(b), derivedData(d) {}

    // 派生类拷贝构造：必须调用基类拷贝构造
    Derived(const Derived& o) : Base(o), derivedData(o.derivedData) {
        std::cout << "Derived拷贝构造" << std::endl;
    }

    // 派生类拷贝赋值：必须调用基类拷贝赋值
    Derived& operator=(const Derived& o) {
        Base::operator=(o);  // 基类赋值
        derivedData = o.derivedData;
        std::cout << "Derived拷贝赋值" << std::endl;
        return *this;
    }

    // 移动构造
    Derived(Derived&& o) noexcept
        : Base(std::move(o)), derivedData(o.derivedData) {
        o.derivedData = 0;
        std::cout << "Derived移动构造" << std::endl;
    }

    // 移动赋值
    Derived& operator=(Derived&& o) noexcept {
        Base::operator=(std::move(o));
        derivedData = o.derivedData;
        o.derivedData = 0;
        std::cout << "Derived移动赋值" << std::endl;
        return *this;
    }

    int getDerived() const { return derivedData; }
};

int main() {
    Derived d1(10, 20);

    std::cout << "\n=== 拷贝构造 ===" << std::endl;
    Derived d2 = d1;

    std::cout << "\n=== 拷贝赋值 ===" << std::endl;
    Derived d3;
    d3 = d1;

    std::cout << "\n=== 移动构造 ===" << std::endl;
    Derived d4 = std::move(d1);

    std::cout << "\n=== 移动赋值 ===" << std::endl;
    Derived d5;
    d5 = std::move(d2);

    return 0;
}
```

### 2.2 忘记调用基类拷贝构造的错误

```cpp
#include <iostream>

class Base {
    int data;
public:
    Base(int d = 0) : data(d) {}
    Base(const Base& o) : data(o.data) {
        std::cout << "Base拷贝构造" << std::endl;
    }
    int get() const { return data; }
};

// 错误：没有显式调用基类拷贝构造
class BadDerived : public Base {
    int extra;
public:
    BadDerived(int d, int e) : Base(d), extra(e) {}
    BadDerived(const BadDerived& o) : /* 缺少 Base(o) */, extra(o.extra) {
        // 基类部分被默认构造（不是拷贝！），data是未初始化的垃圾值
        std::cout << "BadDerived拷贝构造（基类未正确拷贝）" << std::endl;
    }
};

// 正确：显式调用基类拷贝构造
class GoodDerived : public Base {
    int extra;
public:
    GoodDerived(int d, int e) : Base(d), extra(e) {}
    GoodDerived(const GoodDerived& o) : Base(o), extra(o.extra) {
        std::cout << "GoodDerived拷贝构造（正确）" << std::endl;
    }
};

int main() {
    BadDerived bad(42, 100);
    BadDerived bad2 = bad;
    std::cout << "bad2.get() = " << bad2.get() << " (应该是42，但实际是垃圾值)" << std::endl;

    GoodDerived good(42, 100);
    GoodDerived good2 = good;
    std::cout << "good2.get() = " << good2.get() << " (正确：42)" << std::endl;

    return 0;
}
```

### 2.3 基类禁用拷贝对派生类的影响

```cpp
#include <iostream>
#include <memory>

class NonCopyable {
protected:
    NonCopyable() = default;
    NonCopyable(const NonCopyable&) = delete;
    NonCopyable& operator=(const NonCopyable&) = delete;
    NonCopyable(NonCopyable&&) = default;
    NonCopyable& operator=(NonCopyable&&) = default;
};

class Resource : public NonCopyable {
    std::unique_ptr<int[]> data;
public:
    Resource(size_t n) : data(std::make_unique<int[]>(n)) {}

    // Resource的拷贝构造/赋值被隐式删除
    // 但移动操作可用（基类移动是default）
    Resource(Resource&&) = default;
    Resource& operator=(Resource&&) = default;
};

int main() {
    Resource r1(100);
    // Resource r2 = r1;    // 编译错误！拷贝构造被删除
    Resource r2 = std::move(r1);  // OK：移动

    return 0;
}
```

### 2.4 虚函数与拷贝的交互

```cpp
#include <iostream>
#include <memory>

class Base {
public:
    virtual std::unique_ptr<Base> clone() const {
        return std::make_unique<Base>(*this);
    }
    virtual void show() const { std::cout << "Base" << std::endl; }
    virtual ~Base() = default;
};

class Derived : public Base {
public:
    std::unique_ptr<Base> clone() const override {
        return std::make_unique<Derived>(*this);
    }
    void show() const override { std::cout << "Derived" << std::endl; }
};

int main() {
    std::unique_ptr<Base> p = std::make_unique<Derived>();
    auto copy = p->clone();  // 调用Derived::clone()，返回Derived的拷贝
    copy->show();  // "Derived"

    return 0;
}
```

## 三、注意事项与常见陷阱

1. **派生类拷贝构造必须显式调用基类拷贝构造**：否则调用基类默认构造，基类部分不是拷贝
2. **派生类拷贝赋值必须显式调用基类拷贝赋值**：`Base::operator=(o)`
3. **基类析构声明为virtual保证多态删除正确**：通过基类指针delete派生类对象
4. **定义了任何拷贝/移动操作会阻止编译器自动生成移动操作**：五法则
5. **基类禁用拷贝会隐式删除派生类的拷贝操作**：编译器自动处理
6. **使用clone模式实现多态拷贝**：`virtual unique_ptr<Base> clone() const`
