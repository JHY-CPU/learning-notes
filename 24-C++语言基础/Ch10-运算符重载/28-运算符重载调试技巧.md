# 运算符重载调试技巧

## 一、概念说明

调试运算符重载时，可以在运算符函数中添加日志来追踪调用链和参数值。利用`__PRETTY_FUNCTION__`（GCC/Clang）或`__FUNCSIG__`（MSVC）可以自动输出函数签名。

调试策略：
- 使用 `std::cerr` 输出调试信息（无缓冲，立即输出）
- 使用宏控制调试输出的开关
- 使用代理类区分调试和非调试模式

## 二、具体用法

### 2.1 调试日志包装

```cpp
#include <iostream>
#include <string>
using namespace std;

class DebugVec {
    int x, y;
public:
    DebugVec(int x = 0, int y = 0) : x(x), y(y) {}

    friend DebugVec operator+(const DebugVec& a, const DebugVec& b) {
        cerr << "[DEBUG] operator+: ("
             << a.x << "," << a.y << ") + ("
             << b.x << "," << b.y << ")" << endl;
        return DebugVec(a.x + b.x, a.y + b.y);
    }

    friend bool operator==(const DebugVec& a, const DebugVec& b) {
        cerr << "[DEBUG] operator==: ("
             << a.x << "," << a.y << ") == ("
             << b.x << "," << b.y << ")" << endl;
        return a.x == b.x && a.y == b.y;
    }

    friend ostream& operator<<(ostream& os, const DebugVec& v) {
        return os << "(" << v.x << "," << v.y << ")";
    }
};

int main() {
    DebugVec a(1, 2), b(3, 4);
    DebugVec c = a + b;
    cout << "c = " << c << endl;
    // [DEBUG] operator+: (1,2) + (3,4)
    // c = (4,6)

    cout << "a == b: " << (a == b) << endl;
    // [DEBUG] operator==: (1,2) == (3,4)
    // a == b: 0

    return 0;
}
```

### 2.2 使用宏控制调试

```cpp
#include <iostream>
using namespace std;

// 编译时开关：-DDEBUG_MODE 启用调试
#ifdef DEBUG_MODE
    #define DEBUG_LOG(msg) cerr << "[DEBUG] " << msg << endl
#else
    #define DEBUG_LOG(msg)
#endif

class SmartInt {
    int val;
public:
    SmartInt(int v = 0) : val(v) {}

    SmartInt operator+(const SmartInt& other) const {
        DEBUG_LOG("operator+: " << val << " + " << other.val);
        return SmartInt(val + other.val);
    }

    SmartInt& operator+=(const SmartInt& other) {
        DEBUG_LOG("operator+=: " << val << " += " << other.val);
        val += other.val;
        return *this;
    }

    friend ostream& operator<<(ostream& os, const SmartInt& si) {
        return os << si.val;
    }
};

int main() {
    SmartInt a(10), b(20);
    SmartInt c = a + b;
    cout << "c = " << c << endl;  // 30

    a += b;
    cout << "a = " << a << endl;  // 30

    return 0;
}
```

### 2.3 函数签名追踪

```cpp
#include <iostream>
using namespace std;

class Traced {
    int id;
    static int nextId;
public:
    Traced() : id(nextId++) {
        cerr << "[构造] Traced #" << id << endl;
    }

    Traced(const Traced& o) : id(nextId++) {
        cerr << "[拷贝构造] Traced #" << id << " from #" << o.id << endl;
    }

    Traced(Traced&& o) : id(nextId++) {
        cerr << "[移动构造] Traced #" << id << " from #" << o.id << endl;
    }

    ~Traced() {
        cerr << "[析构] Traced #" << id << endl;
    }

    Traced& operator=(const Traced& o) {
        cerr << "[拷贝赋值] #" << id << " = #" << o.id << endl;
        return *this;
    }

    friend Traced operator+(const Traced& a, const Traced& b) {
        cerr << "[operator+] #" << a.id << " + #" << b.id << endl;
        return Traced();
    }

    friend ostream& operator<<(ostream& os, const Traced& t) {
        return os << "Traced#" << t.id;
    }
};

int Traced::nextId = 1;

int main() {
    Traced a, b;
    Traced c = a + b;
    cout << c << endl;

    return 0;
}
```

## 三、注意事项与常见陷阱

1. **使用std::cerr输出调试信息**：cerr不缓冲，立即输出，不会被cout的缓冲影响
2. **可以通过宏`#ifdef DEBUG`控制调试输出**：发布版本不产生调试开销
3. **注意调试日志可能影响运算符的性能语义**：不要在性能关键路径上加日志
4. **考虑使用代理类在调试和非调试模式间切换**：编译时选择
5. **__PRETTY_FUNCTION__和__FUNCSIG__是编译器扩展**：非标准，但广泛支持
6. **不要在const成员函数中修改调试状态**：使用mutable或全局状态
