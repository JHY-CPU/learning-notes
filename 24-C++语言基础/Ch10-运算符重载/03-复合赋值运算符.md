# 复合赋值运算符

## 一、概念说明

复合赋值运算符（`+= -= *= /=`等，C++11标准 §13.5.3）修改左操作数并返回其引用。设计原则是**先实现复合赋值，再基于它实现二元算术运算符**，避免代码重复。

| 运算符 | 成员/非成员 | 返回类型 | 典型实现 |
|--------|-----------|---------|---------|
| `+=` | 成员 | `T&` | 修改并返回*this |
| `+` | 非成员/友元 | `T` | 基于+=实现 |

## 二、具体用法

### 2.1 完整的分数类实现

```cpp
#include <iostream>
#include <numeric>
using namespace std;

class Fraction {
    int num, den;
    void normalize() {
        int g = gcd(abs(num), abs(den));
        num /= g; den /= g;
        if (den < 0) { num = -num; den = -den; }
    }
public:
    Fraction(int n = 0, int d = 1) : num(n), den(d) { normalize(); }

    // 复合赋值：成员函数，返回引用
    Fraction& operator+=(const Fraction& rhs) {
        num = num * rhs.den + rhs.num * den;
        den *= rhs.den;
        normalize();
        return *this;
    }

    Fraction& operator-=(const Fraction& rhs) {
        num = num * rhs.den - rhs.num * den;
        den *= rhs.den;
        normalize();
        return *this;
    }

    Fraction& operator*=(const Fraction& rhs) {
        num *= rhs.num;
        den *= rhs.den;
        normalize();
        return *this;
    }

    Fraction& operator/=(const Fraction& rhs) {
        num *= rhs.den;
        den *= rhs.num;
        normalize();
        return *this;
    }

    // 二元算术：非成员，基于复合赋值实现
    friend Fraction operator+(Fraction lhs, const Fraction& rhs) { return lhs += rhs; }
    friend Fraction operator-(Fraction lhs, const Fraction& rhs) { return lhs -= rhs; }
    friend Fraction operator*(Fraction lhs, const Fraction& rhs) { return lhs *= rhs; }
    friend Fraction operator/(Fraction lhs, const Fraction& rhs) { return lhs /= rhs; }

    // 比较
    bool operator==(const Fraction& rhs) const {
        return num == rhs.num && den == rhs.den;
    }

    friend ostream& operator<<(ostream& os, const Fraction& f) {
        return os << f.num << "/" << f.den;
    }
};

int main() {
    Fraction a(1, 3), b(1, 6);
    cout << "a + b = " << a + b << endl;  // 1/2
    cout << "a - b = " << a - b << endl;  // 1/6
    cout << "a * b = " << a * b << endl;  // 1/18
    cout << "a / b = " << a / b << endl;  // 2/1

    // 链式赋值
    Fraction c;
    c += a;
    c += b;
    cout << "c = " << c << endl;  // 1/2

    return 0;
}
```

### 2.2 向量类的复合赋值

```cpp
#include <iostream>
using namespace std;

class Vec3 {
public:
    double x, y, z;
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3& operator+=(const Vec3& v) { x += v.x; y += v.y; z += v.z; return *this; }
    Vec3& operator-=(const Vec3& v) { x -= v.x; y -= v.y; z -= v.z; return *this; }
    Vec3& operator*=(double s) { x *= s; y *= s; z *= s; return *this; }

    friend Vec3 operator+(Vec3 a, const Vec3& b) { return a += b; }
    friend Vec3 operator-(Vec3 a, const Vec3& b) { return a -= b; }
    friend Vec3 operator*(Vec3 a, double s) { return a *= s; }
    friend Vec3 operator*(double s, Vec3 a) { return a *= s; }

    friend ostream& operator<<(ostream& os, const Vec3& v) {
        return os << "(" << v.x << "," << v.y << "," << v.z << ")";
    }
};

int main() {
    Vec3 a(1, 2, 3), b(4, 5, 6);
    cout << "a + b = " << a + b << endl;  // (5,7,9)
    cout << "a * 2 = " << a * 2 << endl;  // (2,4,6)
    cout << "2 * a = " << 2 * a << endl;  // (2,4,6)

    return 0;
}
```

## 三、注意事项与常见陷阱

1. **复合赋值运算符必须返回 `*this` 的非const引用**：支持链式赋值
2. **复合赋值是成员函数**：因为修改了左操作数
3. **基于复合赋值实现二元算术可避免代码重复**：`operator+` 基于 `operator+=`
4. **注意运算优先级和结合性保持不变**：重载不改变优先级
5. **除法赋值需检查除零**：在operator/=中添加检查
6. **C++20可用 `<=>` 自动生成比较运算符**：减少代码量
