# 位运算应用实例

## 一、概念说明

位运算在实际开发中有许多高效应用场景：权限系统、位图、状态标志、压缩存储等。掌握这些实例能帮助理解位运算的实际价值。

## 二、具体用法

### 2.1 权限系统

```cpp
#include <iostream>
using namespace std;

// 文件权限系统
enum Permission : unsigned int {
    READ    = 1 << 0,  // 0001 = 1
    WRITE   = 1 << 1,  // 0010 = 2
    EXECUTE = 1 << 2,  // 0100 = 4
    ADMIN   = 1 << 3   // 1000 = 8
};

void checkPermission(unsigned int userPerm, Permission perm, const char* name) {
    bool has = (userPerm & perm) != 0;
    cout << name << ": " << (has ? "允许" : "拒绝") << endl;
}

int main() {
    // 设置用户权限
    unsigned int user1 = READ | WRITE;           // 读写权限
    unsigned int user2 = READ | WRITE | EXECUTE | ADMIN;  // 完全权限

    cout << "用户1权限检查:" << endl;
    checkPermission(user1, READ, "读");
    checkPermission(user1, WRITE, "写");
    checkPermission(user1, EXECUTE, "执行");
    checkPermission(user1, ADMIN, "管理员");

    cout << "\n用户2权限检查:" << endl;
    checkPermission(user2, READ, "读");
    checkPermission(user2, EXECUTE, "执行");
    checkPermission(user2, ADMIN, "管理员");

    // 添加权限
    user1 |= EXECUTE;
    cout << "\n添加执行权限后:" << endl;
    checkPermission(user1, EXECUTE, "执行");

    // 移除权限
    user1 &= ~WRITE;
    cout << "移除写权限后:" << endl;
    checkPermission(user1, WRITE, "写");

    return 0;
}
```

输出：
```
用户1权限检查:
读: 允许
写: 允许
执行: 拒绝
管理员: 拒绝

用户2权限检查:
读: 允许
写: 允许
执行: 允许
管理员: 允许

添加执行权限后:
执行: 允许
移除写权限后:
写: 拒绝
```

### 2.2 位图（Bitmap）

```cpp
#include <iostream>
#include <bitset>
#include <vector>
using namespace std;

class Bitmap {
    vector<unsigned int> bits;
    int size;
public:
    Bitmap(int n) : size(n) {
        bits.resize((n + 31) / 32, 0);  // 每个int存32位
    }

    void set(int pos) {
        if (pos >= 0 && pos < size) {
            bits[pos / 32] |= (1u << (pos % 32));
        }
    }

    void clear(int pos) {
        if (pos >= 0 && pos < size) {
            bits[pos / 32] &= ~(1u << (pos % 32));
        }
    }

    bool test(int pos) const {
        if (pos >= 0 && pos < size) {
            return (bits[pos / 32] >> (pos % 32)) & 1;
        }
        return false;
    }

    int count() const {
        int total = 0;
        for (unsigned int b : bits) {
            // Kernighan算法：数1的个数
            unsigned int x = b;
            while (x) {
                x &= (x - 1);
                total++;
            }
        }
        return total;
    }
};

int main() {
    Bitmap bm(100);

    bm.set(5);
    bm.set(10);
    bm.set(42);
    bm.set(99);

    cout << "位5: " << (bm.test(5) ? "已设置" : "未设置") << endl;
    cout << "位6: " << (bm.test(6) ? "已设置" : "未设置") << endl;
    cout << "已设置位数: " << bm.count() << endl;

    bm.clear(10);
    cout << "清除位10后，已设置位数: " << bm.count() << endl;

    return 0;
}
```

输出：
```
位5: 已设置
位6: 未设置
已设置位数: 4
清除位10后，已设置位数: 3
```

### 2.3 状态标志

```cpp
#include <iostream>
using namespace std;

// 游戏角色状态
enum Status : unsigned int {
    ALIVE    = 1 << 0,
    VISIBLE  = 1 << 1,
    CAN_MOVE = 1 << 2,
    POISONED = 1 << 3,
    FROZEN   = 1 << 4,
    BURNING  = 1 << 5
};

class Character {
    unsigned int status;
public:
    Character() : status(ALIVE | VISIBLE | CAN_MOVE) {}

    void addStatus(Status s) { status |= s; }
    void removeStatus(Status s) { status &= ~s; }
    bool hasStatus(Status s) const { return (status & s) != 0; }
    bool isDead() const { return !hasStatus(ALIVE); }
    bool canAct() const { return hasStatus(ALIVE) && !hasStatus(FROZEN); }
};

int main() {
    Character hero;
    cout << "存活: " << hero.hasStatus(ALIVE) << endl;
    cout << "可以行动: " << hero.canAct() << endl;

    hero.addStatus(POISONED);
    hero.addStatus(FROZEN);
    cout << "中毒: " << hero.hasStatus(POISONED) << endl;
    cout << "冰冻: " << hero.hasStatus(FROZEN) << endl;
    cout << "可以行动: " << hero.canAct() << endl;

    hero.removeStatus(FROZEN);
    cout << "解冻后可以行动: " << hero.canAct() << endl;

    return 0;
}
```

输出：
```
存活: 1
可以行动: 1
中毒: 1
冰冻: 1
可以行动: 0
解冻后可以行动: 1
```

## 三、注意事项与常见陷阱

1. **枚举值必须是2的幂**：权限/状态位必须用`1 << n`定义，不能随意赋值
2. **无符号类型**：位运算应使用无符号类型，有符号右移行为不确定
3. **位操作组合**：`|`添加标志，`&~`移除标志，`&`检查标志，`^`翻转标志
4. **不要过度使用**：简单的布尔变量比位标志更易读，只有在需要大量标志时才用位运算
5. **线程安全**：位操作不是原子的，多线程需要额外同步
