# 信号处理

## 一、傅里叶变换（FFT）

### 1.1 基本 FFT

```matlab
% 生成测试信号
Fs = 1000;             % 采样率 1000 Hz
T = 1/Fs;              % 采样间隔
L = 1500;              % 信号长度
t = (0:L-1)*T;         % 时间向量

% 含两个正弦分量 + 噪声的信号
S = 0.7*sin(2*pi*50*t) + sin(2*pi*120*t);   % 50Hz + 120Hz
X = S + 2*randn(size(t));                     % 添加噪声

% FFT
Y = fft(X);

% 单边幅度谱
P2 = abs(Y/L);                  % 双边谱
P1 = P2(1:L/2+1);              % 单边谱
P1(2:end-1) = 2*P1(2:end-1);   % 补偿能量

f = Fs*(0:(L/2))/L;             % 频率轴

figure;
subplot(2,1,1);
plot(t(1:200), X(1:200));
title('含噪声时域信号');
xlabel('时间 (s)'); ylabel('幅值');

subplot(2,1,2);
plot(f, P1, 'b-', 'LineWidth', 1.5);
title('单边幅度谱');
xlabel('频率 (Hz)'); ylabel('|P1(f)|');
xlim([0 200]);
grid on;
```

### 1.2 频谱分析进阶

```matlab
% 补零提升频率分辨率
Fs = 1000;
t = 0:1/Fs:0.5;
x = cos(2*pi*100*t) + 0.5*cos(2*pi*105*t);

% 不补零
N1 = length(x);
Y1 = fft(x);
f1 = (0:N1-1)*Fs/N1;

% 补零到 4096 点
N2 = 4096;
Y2 = fft(x, N2);
f2 = (0:N2-1)*Fs/N2;

figure;
subplot(2,1,1);
plot(f1(1:N1/2), abs(Y1(1:N1/2))/N1, 'b-');
title('不补零'); xlabel('频率 (Hz)');

subplot(2,1,2);
plot(f2(1:N2/2), abs(Y2(1:N2/2))/N1, 'r-');
title('补零到 4096 点'); xlabel('频率 (Hz)');
xlim([80 130]);

% 功率谱密度 (PSD)
Fs = 1000;
t = 0:1/Fs:1-1/Fs;
x = cos(2*pi*50*t) + randn(size(t));

% 使用 pwelch 方法（推荐）
[pxx, f_psd] = pwelch(x, hamming(256), 128, 512, Fs);

figure;
plot(f_psd, 10*log10(pxx));
title('功率谱密度 (Welch 方法)');
xlabel('频率 (Hz)'); ylabel('PSD (dB/Hz)');
grid on;
```

## 二、数字滤波器

### 2.1 FIR 滤波器

```matlab
% 设计 FIR 低通滤波器
Fs = 1000;
N = 100;               % 滤波器阶数
Fc = 150;              % 截止频率

% 使用 fir1 设计
b = fir1(N, Fc/(Fs/2), 'low');  % 归一化频率 = 截止/(Fs/2)

% 频率响应
[H, f_freq] = freqz(b, 1, 1024, Fs);

figure;
subplot(2,1,1);
plot(f_freq, 20*log10(abs(H)), 'b-', 'LineWidth', 1.5);
xlabel('频率 (Hz)'); ylabel('幅值 (dB)');
title('FIR 低通滤波器频率响应');
ylim([-80 5]); grid on;

subplot(2,1,2);
plot(f_freq, unwrap(angle(H))*180/pi, 'r-', 'LineWidth', 1.5);
xlabel('频率 (Hz)'); ylabel('相位 (度)');
title('相位响应'); grid on;

% 应用滤波器
t = 0:1/Fs:1;
x = sin(2*pi*50*t) + sin(2*pi*300*t);  % 50Hz + 300Hz
y = filter(b, 1, x);                    % FIR 滤波

figure;
subplot(2,1,1);
plot(t(1:200), x(1:200), 'b-');
title('原始信号 (50Hz + 300Hz)');

subplot(2,1,2);
plot(t(1:200), y(1:200), 'r-');
title('FIR 低通滤波后 (保留50Hz)');
```

### 2.2 IIR 滤波器（巴特沃斯）

```matlab
% butter 设计巴特沃斯滤波器
Fs = 1000;
Fc_low = 50;
Fc_high = 200;

% 低通滤波器
order = 6;
[b_lp, a_lp] = butter(order, Fc_low/(Fs/2), 'low');

% 带通滤波器
[b_bp, a_bp] = butter(order/2, [Fc_low Fc_high]/(Fs/2), 'bandpass');

% 高通滤波器
[b_hp, a_hp] = butter(order, 100/(Fs/2), 'high');

% 频率响应对比
[H_lp, f] = freqz(b_lp, a_lp, 1024, Fs);
[H_bp, ~] = freqz(b_bp, a_bp, 1024, Fs);
[H_hp, ~] = freqz(b_hp, a_hp, 1024, Fs);

figure;
plot(f, 20*log10(abs(H_lp)), 'b-', 'LineWidth', 1.5); hold on;
plot(f, 20*log10(abs(H_bp)), 'r-', 'LineWidth', 1.5);
plot(f, 20*log10(abs(H_hp)), 'g-', 'LineWidth', 1.5);
legend('低通', '带通', '高通');
xlabel('频率 (Hz)'); ylabel('幅值 (dB)');
title('巴特沃斯滤波器频率响应');
ylim([-60 5]); grid on;
```

### 2.3 零相位滤波

```matlab
% filtfilt：零相位滤波（前后各滤一次，消除相位延迟）
Fs = 1000;
t = 0:1/Fs:1;
x = sin(2*pi*50*t) + 0.5*randn(size(t));

% 设计低通滤波器
[b, a] = butter(4, 100/(Fs/2), 'low');

% filter vs filtfilt
y_filter = filter(b, a, x);
y_filtfilt = filtfilt(b, a, x);

figure;
plot(t(1:200), x(1:200), 'b-', 'LineWidth', 0.5); hold on;
plot(t(1:200), y_filter(1:200), 'r-', 'LineWidth', 1.5);
plot(t(1:200), y_filtfilt(1:200), 'g-', 'LineWidth', 1.5);
legend('含噪声信号', 'filter（有延迟）', 'filtfilt（零延迟）');
title('filter vs filtfilt 对比');

% cheby1：切比雪夫 I 型滤波器
[b_cheb, a_cheb] = cheby1(4, 0.5, 100/(Fs/2), 'low');

% ellip：椭圆滤波器（最陡过渡带）
[b_ellip, a_ellip] = ellip(4, 0.5, 40, 100/(Fs/2), 'low');

% 频率响应对比
figure;
for {b_i, a_i, name} = {{b,a,'Butter'},{b_cheb,a_cheb,'Cheby1'},{b_ellip,a_ellip,'Ellip'}}
    [H, f] = freqz(b_i{1}, a_i{1}, 1024, Fs);
    plot(f, 20*log10(abs(H)), 'LineWidth', 1.5); hold on;
end
legend('Butterworth', 'Chebyshev I', 'Elliptic');
xlabel('频率 (Hz)'); ylabel('幅值 (dB)');
title('IIR 滤波器类型对比');
ylim([-80 5]); grid on;
```

## 三、短时傅里叶变换（Spectrogram）

```matlab
% 生成线性调频信号（chirp）
Fs = 2000;
t = 0:1/Fs:2;
x = chirp(t, 100, 2, 500) + 0.3*randn(size(t));  % 从100Hz扫到500Hz

% spectrogram 分析
window = 256;          % 窗口长度
noverlap = 200;        % 重叠长度
nfft = 512;            % FFT 点数

figure;
spectrogram(x, hamming(window), noverlap, nfft, Fs, 'yaxis');
title('线性调频信号时频图');
colorbar;

% 返回值形式
[S, F, T] = spectrogram(x, hamming(window), noverlap, nfft, Fs);

figure;
imagesc(T, F, 20*log10(abs(S)));
axis xy;
colorbar;
xlabel('时间 (s)'); ylabel('频率 (Hz)');
title('时频图 (手动绘制)');
colormap(jet);

% stft 函数（R2017b+）
% [S_stft, F_stft, T_stft] = stft(x, Fs, 'Window', hamming(256), ...
%                                   'OverlapLength', 200, 'FFTLength', 512);
```

## 四、卷积

```matlab
% 离散卷积
x = [1 2 3 4 5];
h = [1 0 -1];     % 高通差分核

% conv 函数
y = conv(x, h);
fprintf('卷积结果: %s\n', mat2str(y));

% 卷积长度
fprintf('x 长度: %d, h 长度: %d, y 长度: %d\n', length(x), length(h), length(y));
% 结果长度 = length(x) + length(h) - 1

% 不同卷积模式
y_full = conv(x, h, 'full');    % 完整卷积（默认）
y_same = conv(x, h, 'same');    % 与较长输入等长
y_valid = conv(x, h, 'valid');  % 仅重叠部分

% 卷积与 FFT 的关系（快速卷积）
N = 10000;
x_large = randn(1, N);
h_large = randn(1, 100);

% 时域卷积
tic;
y_time = conv(x_large, h_large);
t_time = toc;

% 频域快速卷积
tic;
n_fft = 2^nextpow2(N + length(h_large) - 1);
y_freq = ifft(fft(x_large, n_fft) .* fft(h_large, n_fft));
y_freq = real(y_freq(1:length(y_time)));
t_freq = toc;

fprintf('时域卷积: %.4f 秒\n', t_time);
fprintf('频域卷积: %.4f 秒\n', t_freq);
fprintf('误差: %.2e\n', norm(y_time - y_freq));

% 应用：移动平均（平滑）
t = linspace(0, 4*pi, 500);
signal = sin(t) + 0.5*randn(size(t));
kernel = ones(1, 20) / 20;  % 20 点移动平均
smoothed = conv(signal, kernel, 'same');

figure;
plot(t, signal, 'b-', 'LineWidth', 0.5); hold on;
plot(t, smoothed, 'r-', 'LineWidth', 2);
legend('含噪声信号', '移动平均平滑');
title('卷积实现信号平滑');
```

## 五、窗函数

```matlab
N = 64;  % 窗长度

% 常用窗函数
w_rect = rectwin(N);           % 矩形窗
w_hann = hann(N);              % 汉宁窗
w_hamm = hamming(N);           % 汉明窗
w_black = blackman(N);         % 布莱克曼窗
w_cheb = chebwin(N, 60);       % 切比雪夫窗（旁瓣衰减 60dB）
w_kaiser = kaiser(N, 8.6);     % 凯泽窗（beta=8.6）
w_tukey = tukeywin(N, 0.5);    % Tukey 窗（锥形比例 0.5）

% 绘制窗函数
figure;
plot(w_rect, 'LineWidth', 1.5); hold on;
plot(w_hann, 'LineWidth', 1.5);
plot(w_hamm, 'LineWidth', 1.5);
plot(w_black, 'LineWidth', 1.5);
plot(w_kaiser, 'LineWidth', 1.5);
legend('矩形窗', '汉宁窗', '汉明窗', '布莱克曼窗', '凯泽窗');
title('窗函数对比');
xlabel('样本'); ylabel('幅值');

% 窗函数的频谱特性
figure;
for {w, name} = {{rectwin, '矩形'}, {hann, '汉宁'}, {hamming, '汉明'}, {blackman, '布莱克曼'}}
    [H_w, f_w] = freqz(w{1}, 1, 1024, 'whole');
    plot(f_w/pi, 20*log10(abs(H_w)/max(abs(H_w))), 'LineWidth', 1.5);
    hold on;
end
legend('矩形', '汉宁', '汉明', '布莱克曼');
xlabel('归一化频率 (×\pi rad/sample)');
ylabel('归一化幅值 (dB)');
title('窗函数频谱特性');
ylim([-100 5]); grid on;
```

## 六、采样与重采样

```matlab
% 信号采样
Fs_original = 44100;    % 原始采样率
Fs_new = 8000;          % 目标采样率

t_orig = 0:1/Fs_original:0.01;
x_orig = sin(2*pi*1000*t_orig);

% 降采样
[P, Q] = rat(Fs_new / Fs_original);  % 计算有理数比
x_down = resample(x_orig, P, Q);

% 插值上采样
x_up = resample(x_down, Q, P);

figure;
subplot(3,1,1);
stem(t_orig(1:100), x_orig(1:100), 'b.');
title(sprintf('原始信号 (%d Hz)', Fs_original));

subplot(3,1,2);
t_down = (0:length(x_down)-1) / Fs_new;
stem(t_down(1:50), x_down(1:50), 'r.');
title(sprintf('降采样后 (%d Hz)', Fs_new));

subplot(3,1,3);
t_up = (0:length(x_up)-1) / Fs_original;
stem(t_up(1:100), x_up(1:100), 'g.');
title('重采样回原始采样率');

% 信号加窗截取
Fs = 1000;
t = 0:1/Fs:5;
x = chirp(t, 100, 5, 500);

% 截取指定时间段
idx = (t >= 1) & (t <= 2);
x_segment = x(idx);
t_segment = t(idx);
```
