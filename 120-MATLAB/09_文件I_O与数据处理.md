# 文件 I/O 与数据处理

## 一、MAT 文件读写

### 1.1 基本 load/save

```matlab
% 保存变量
A = rand(100);
B = magic(5);
description = '测试数据';

save('data.mat');              % 保存所有变量
save('data.mat', 'A', 'B');   % 只保存 A 和 B
save('data.mat', '-v7.3');    % 使用 v7.3 格式（支持 >2GB 数据）
save('data_partial.mat', 'A', '-append');  % 追加保存

% 加载变量
load('data.mat');              % 加载所有变量到工作区
load('data.mat', 'A');         % 只加载 A
S = load('data.mat');          % 加载为结构体
S = load('data.mat', 'A');     % 结构体中只含 A

% 检查 .mat 文件内容
whos('-file', 'data.mat');     % 查看文件中的变量信息
matfile_info = matfile('data.mat');  % 创建 matfile 对象（支持懒加载）

% 懒加载（大文件不全部读入内存）
mf = matfile('data.mat');
A_subset = mf.A(1:10, 1:10);  % 只读取部分数据
```

### 1.2 跨版本兼容

```matlab
% v5: 旧版 MATLAB（<7.3），文件 < 2GB
save('compat.mat', '-v5');

% v7: MATLAB 7.0+
save('compat.mat', '-v7');

% v7.3: HDF5 格式，支持大文件和部分加载
save('large_data.mat', '-v7.3');

% 检查版本
mf = matfile('data.mat');
info = h5info('data.mat');  % v7.3 格式可用 h5info
```

## 二、表格数据读写

### 2.1 readtable / writetable

```matlab
% 创建示例表格
T = table();
T.Name = {'Alice'; 'Bob'; 'Charlie'; 'Diana'};
T.Age = [25; 30; 35; 28];
T.Score = [85.5; 92.3; 78.9; 95.1];
T.Department = {'CS'; 'EE'; 'CS'; 'Math'};
disp(T);

% 写入 CSV
writetable(T, 'data.csv');

% 写入 Excel
writetable(T, 'data.xlsx');
writetable(T, 'data.xlsx', 'Sheet', 'Sheet1', 'Range', 'A1');

% 写入带格式的 Excel
writetable(T, 'formatted.xlsx', ...
    'WriteVariableNames', true, ...
    'WriteRowNames', false);

% 读取 CSV
T_read = readtable('data.csv');
disp(T_read);

% 读取 Excel
T_excel = readtable('data.xlsx');
T_sheet2 = readtable('data.xlsx', 'Sheet', 'Sheet2');
T_range = readtable('data.xlsx', 'Range', 'A1:C5');

% 读取选项
opts = detectImportOptions('data.csv');
opts.VariableTypes{3} = 'double';    % 指定数据类型
opts = setvaropts(opts, 'Name', 'TrimWhitespace', true);
T_custom = readtable('data.csv', opts);

% 表格操作
T_sorted = sortrows(T, 'Age');                    % 按年龄排序
T_filtered = T(T.Age > 28, :);                     % 筛选年龄 > 28
T_filtered2 = T(contains(T.Department, 'CS'), :);  % 按字符串筛选
T.Average = (T.Score + T.Age) / 2;                 % 添加新列
T(:, 'Average') = [];                               % 删除列

% 表格统计
mean_age = mean(T.Age);
grouped = groupsummary(T, 'Department', 'mean', 'Score');
disp(grouped);

% 表格合并
T2 = table({'Alice'; 'Bob'; 'Eve'}, [100; 200; 150], ...
    'VariableNames', {'Name', 'Salary'});
T_merged = outerjoin(T, T2, 'Keys', 'Name', 'MergeKeys', true);
```

### 2.2 数据类型转换

```matlab
% table <-> array
T = table([1;2;3], [4;5;6], 'VariableNames', {'A','B'});
A_array = table2array(T);     % 转为数值矩阵
T_back = array2table(A_array, 'VariableNames', {'X','Y'});

% table <-> struct
S = table2struct(T);          % 转为结构体数组
T_from_struct = struct2table(S);

% table <-> cell
C_cell = table2cell(T);
T_from_cell = cell2table({'Alice', 25; 'Bob', 30}, ...
    'VariableNames', {'Name', 'Age'});
```

## 三、文本文件读写

### 3.1 纯文本文件

```matlab
% 写入文本文件
data = rand(5, 3);
dlmwrite('data.txt', data, 'delimiter', '\t', 'precision', '%.6f');

% 带标题行
fid = fopen('data_with_header.txt', 'w');
fprintf(fid, 'Column1\tColumn2\tColumn3\n');
fprintf(fid, '%.4f\t%.4f\t%.4f\n', data');
fclose(fid);

% 读取纯文本文件
data_read = dlmread('data.txt', '\t');

% 更灵活的读取方式
M = readmatrix('data.txt');

% 读取为字符串
lines = readlines('data_with_header.txt');
text = fileread('data_with_header.txt');

% 逐行读取
fid = fopen('data_with_header.txt', 'r');
header = fgetl(fid);   % 读标题行
while ~feof(fid)
    line = fgetl(fid);
    values = str2double(strsplit(line, '\t'));
    fprintf('读取: %.4f, %.4f, %.4f\n', values(1), values(2), values(3));
end
fclose(fid);
```

### 3.2 日志文件写入

```matlab
% 日志记录
logFile = 'simulation.log';
fid = fopen(logFile, 'w');
fprintf(fid, '=== 仿真日志 ===\n');

    fprintf(fid, 'Trial %d: Result = %.4f\n', trial, result);
    fprintf(fid, '  状态: %s\n', result > 50 ? '通过' : '未通过');

fclose(fid);

% 追加模式
fid = fopen(logFile, 'a');
fprintf(fid, '--- 追加信息 ---\n');
fclose(fid);
```

## 五、图像文件读写

```matlab
% 读取图像
img = imread('photo.jpg');
imshow(img);
title('原始图像');

info = imfinfo('photo.jpg');
fprintf('尺寸: %d x %d\n', info.Width, info.Height);
fprintf('格式: %s\n', info.Format);
fprintf('色彩类型: %s\n', info.ColorType);

% 保存图像
imwrite(img, 'output.png');
imwrite(img, 'output.jpg', 'Quality', 95);

% 图像处理
gray = rgb2gray(img);                % 转灰度
bw = imbinarize(gray);               % 二值化
resized = imresize(img, 0.5);        % 缩小50%
rotated = imrotate(img, 45);         % 旋转45度
cropped = imcrop(img, [100 100 200 200]);  % 裁剪

% 显示多个图像
figure;
subplot(2, 2, 1); imshow(img); title('原始');
subplot(2, 2, 2); imshow(gray); title('灰度');
subplot(2, 2, 3); imshow(bw); title('二值化');
subplot(2, 2, 4); imshow(rotated); title('旋转45°');
```

## 六、JSON 数据处理

```matlab
% 创建结构体
data = struct();
data.name = 'MATLAB Notes';
data.version = 'R2024b';
data.topics = {'基础', '可视化', '信号处理', '符号计算'};
data.config = struct('precision', 'double', 'display', 'compact');

% 写入 JSON
jsonStr = jsonencode(data, 'PrettyPrint', true);
fid = fopen('config.json', 'w', 'n', 'UTF-8');
fwrite(fid, jsonStr);
fclose(fid);

% 读取 JSON
jsonText = fileread('config.json');
dataRead = jsondecode(jsonText);
fprintf('名称: %s\n', dataRead.name);
fprintf('主题: %s\n', strjoin(dataRead.topics, ', '));

% JSON <-> table
T = table({'A'; 'B'; 'C'}, [1; 2; 3], [10; 20; 30], ...
    'VariableNames', {'ID', 'X', 'Y'});
jsonTable = jsonencode(T);
T_back = jsondecode(jsonTable);
```

## 八、批量文件处理

```matlab
% 获取文件列表
files = dir('*.csv');
fprintf('找到 %d 个 CSV 文件\n', length(files));

% 批量读取并合并
allData = table();
for i = 1:length(files)
    filename = fullfile(files(i).folder, files(i).name);
    T = readtable(filename);
    T.SourceFile = repmat({files(i).name}, height(T), 1);
    allData = [allData; T];  %#ok<AGROW>
end
fprintf('总共 %d 行数据\n', height(allData));

% 批量处理图像
imgFiles = dir('*.jpg');
for i = 1:length(imgFiles)
    img = imread(fullfile(imgFiles(i).folder, imgFiles(i).name));
    grayImg = rgb2gray(img);
    [~, name, ~] = fileparts(imgFiles(i).name);
    imwrite(grayImg, fullfile('output', [name '_gray.png']));
end

% 使用 cellfun 批量处理
filenames = {'data1.csv', 'data2.csv', 'data3.csv'};
tables = cellfun(@(f) readtable(f), filenames, 'UniformOutput', false);
combined = vertcat(tables{:});
```
