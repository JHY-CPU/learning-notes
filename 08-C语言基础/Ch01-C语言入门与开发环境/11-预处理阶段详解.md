# 预处理阶段详解

## 一、预处理概述

预处理是C语言编译的第一个阶段，由预处理器（cpp）完成。预处理器在正式编译之前对源代码进行文本替换和处理，输出纯C代码。

```bash
# 查看预处理结果
gcc -E source.c -o source.i

# 预处理器也可以单独使用
cpp source.c -o source.i
```

## 二、头文件包含

### 2.1 #include指令

```c
/* 方式1：尖括号 - 从系统路径查找 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* 方式2：双引号 - 先从当前目录查找，再查找系统路径 */
#include "myheader.h"
#include "utils/utils.h"
```

### 2.2 搜索路径

**尖括号 `<>` 的搜索顺序**：
1. `-I` 指定的目录
2. `C_INCLUDE_PATH` 环境变量指定的目录
3. 系统默认目录（如 `/usr/include`）

**双引号 `""` 的搜索顺序**：
1. 当前源文件所在目录
2. `-I` 指定的目录
3. `C_INCLUDE_PATH` 环境变量指定的目录
4. 系统默认目录

```bash
# 查看GCC的默认头文件搜索路径
echo | gcc -E -Wp,-v - 2>&1 | grep "^ "

# 指定额外的头文件搜索路径
gcc -I./include -I../common/include source.c
```

### 2.3 头文件包含示例

```c
/* main.c */
#include <stdio.h>
#include "mylib.h"

int main(void) {
    printf("%d\n", my_function(42));
    return 0;
}
```

预处理后（简化）：
```c
/* 展开 stdio.h 的内容（数百行）... */
/* 展开 mylib.h 的内容... */

int my_function(int x);

int main(void) {
    printf("%d\n", my_function(42));
    return 0;
}
```

## 三、宏定义

### 3.1 对象式宏

```c
/* 简单的宏定义 */
#define PI          3.14159265
#define MAX_BUFFER  1024
#define PROGRAM_NAME "MyApplication"
#define TRUE        1
#define FALSE       0

/* 使用 */
double circumference = 2 * PI * radius;
char buffer[MAX_BUFFER];
```

### 3.2 函数式宏

```c
/* 函数式宏定义 */
#define SQUARE(x)       ((x) * (x))
#define MAX(a, b)       ((a) > (b) ? (a) : (b))
#define MIN(a, b)       ((a) < (b) ? (a) : (b))
#define ABS(x)          ((x) < 0 ? -(x) : (x))
#define SWAP(a, b)      do { typeof(a) _t = (a); (a) = (b); (b) = _t; } while(0)

/* 使用示例 */
int x = 5;
int sq = SQUARE(x);         /* 展开为: ((5) * (5)) = 25 */
int big = MAX(3, 7);        /* 展开为: ((3) > (7) ? (3) : (7)) = 7 */

/* 宏的陷阱 */
int y = 8;
int bad = SQUARE(y++);      /* 错误！展开为: ((y++) * (y++)) 未定义行为 */
```

### 3.3 宏定义的注意事项

```c
/* 宏参数需要加括号 */
#define BAD_SQUARE(x)   x * x        /* 错误的定义 */
#define GOOD_SQUARE(x)  ((x) * (x))  /* 正确的定义 */

/* 问题示例 */
int result = BAD_SQUARE(1 + 2);
/* 展开为: 1 + 2 * 1 + 2 = 5（错误！应该是9）*/

int correct = GOOD_SQUARE(1 + 2);
/* 展开为: ((1 + 2) * (1 + 2)) = 9（正确）*/
```

### 3.4 特殊宏运算符

```c
/* # 字符串化运算符 */
#define TO_STRING(x)    #x
printf("%s\n", TO_STRING(hello));   /* 输出: "hello" */

/* ## 连接运算符 */
#define CONCAT(a, b)    a##b
int xy = 100;
printf("%d\n", CONCAT(x, y));       /* 输出: 100 */

/* 可变参数宏 */
#define DEBUG_PRINT(fmt, ...) \
    printf("[DEBUG %s:%d] " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)

DEBUG_PRINT("value = %d", 42);
/* 展开为: printf("[DEBUG %s:%d] " "value = %d" "\n", "main.c", 10, 42); */
```

### 3.5 多行宏

```c
/* 使用反斜杠续行 */
#define PRINT_ARRAY(arr, n, fmt) \
    do { \
        printf("["); \
        for (int _i = 0; _i < (n); _i++) { \
            printf(fmt, (arr)[_i]); \
            if (_i < (n) - 1) printf(", "); \
        } \
        printf("]\n"); \
    } while(0)

/* 使用 */
int nums[] = {1, 2, 3, 4, 5};
PRINT_ARRAY(nums, 5, "%d");
/* 输出: [1, 2, 3, 4, 5] */
```

## 四、条件编译

### 4.1 #ifdef / #ifndef / #endif

```c
/* 防止头文件重复包含 */
#ifndef MY_HEADER_H
#define MY_HEADER_H

/* 头文件内容 */
void my_function(void);
int my_variable;

#endif /* MY_HEADER_H */
```

### 4.2 #if / #elif / #else / #endif

```c
/* 根据编译选项选择代码 */
#if defined(DEBUG)
    #define LOG(msg) printf("[DEBUG] %s\n", msg)
#elif defined(RELEASE_WITH_LOG)
    #define LOG(msg) syslog(LOG_INFO, "%s", msg)
#else
    #define LOG(msg) /* nothing */
#endif

/* 根据C标准版本选择代码 */
#if __STDC_VERSION__ >= 201112L
    /* C11 及以上 */
    #include <stdalign.h>
    _Alignas(16) int aligned_var;
#elif __STDC_VERSION__ >= 199901L
    /* C99 */
    for (int i = 0; i < 10; i++) { }
#else
    /* C89/C90 */
    int i;
    for (i = 0; i < 10; i++) { }
#endif
```

### 4.3 平台条件编译

```c
/* 根据操作系统选择代码 */
#ifdef _WIN32
    #include <windows.h>
    #define SLEEP(ms) Sleep(ms)
    #define PATH_SEP '\\'
    #define PLATFORM "Windows"
#elif defined(__linux__)
    #include <unistd.h>
    #define SLEEP(ms) usleep((ms) * 1000)
    #define PATH_SEP '/'
    #define PLATFORM "Linux"
#elif defined(__APPLE__)
    #include <unistd.h>
    #define SLEEP(ms) usleep((ms) * 1000)
    #define PATH_SEP '/'
    #define PLATFORM "macOS"
#else
    #error "Unsupported platform"
#endif
```

### 4.4 #error 和 #warning

```c
/* 强制编译错误 */
#if !defined(__STDC__)
    #error "This compiler does not support ANSI C"
#endif

#if sizeof(int) < 4
    #error "int must be at least 4 bytes"
#endif

/* 编译警告（GCC扩展）*/
#ifdef __GNUC__
    #warning "Using GCC-specific extensions"
#endif
```

### 4.5 #pragma指令

```c
/* #pragma 是编译器特定的指令 */

/* GCC: 忽略特定警告 */
#pragma GCC diagnostic push
#pragma GCC diagnostic ignored "-Wunused-variable"
    int unused_var;  /* 不会警告 */
#pragma GCC diagnostic pop

/* 一次性包含（非标准但广泛支持）*/
#pragma once
/* 等价于 #ifndef / #define / #endif 模式 */

/* 控制结构体打包 */
#pragma pack(push, 1)
struct PackedStruct {
    char a;
    int b;
    char c;
};
#pragma pack(pop)

/* GCC/GNU: 指定段 */
#pragma GCC poison malloc free  /* 禁止使用 malloc 和 free */
```

## 五、预定义宏

### 5.1 标准预定义宏

```c
#include <stdio.h>

int main(void) {
    printf("文件名: %s\n", __FILE__);
    printf("行号: %d\n", __LINE__);
    printf("编译日期: %s\n", __DATE__);
    printf("编译时间: %s\n", __TIME__);
    printf("函数名: %s\n", __func__);    /* C99 */

    /* C标准符合度 */
#if __STDC__
    printf("符合C标准: 是\n");
    printf("C标准版本: %ld\n", __STDC_VERSION__);
#endif

    return 0;
}
```

### 5.2 GCC预定义宏

```c
/* 编译器信息 */
#ifdef __GNUC__
    printf("编译器: GCC %d.%d.%d\n",
           __GNUC__, __GNUC_MINOR__, __GNUC_PATCHLEVEL__);
#elif defined(__clang__)
    printf("编译器: Clang %s\n", __clang_version__);
#elif defined(_MSC_VER)
    printf("编译器: MSVC %d\n", _MSC_VER);
#endif

/* 平台信息 */
#ifdef __linux__
    printf("平台: Linux\n");
#elif defined(_WIN32)
    printf("平台: Windows\n");
#elif defined(__APPLE__)
    printf("平台: macOS\n");
#endif

/* 架构信息 */
#ifdef __x86_64__
    printf("架构: x86_64\n");
#elif defined(__aarch64__)
    printf("架构: ARM64\n");
#endif
```

### 5.3 实用调试宏

```c
/* 调试打印宏 */
#ifdef DEBUG
    #define DBG_PRINT(fmt, ...) \
        fprintf(stderr, "[%s:%d:%s] " fmt "\n", \
                __FILE__, __LINE__, __func__, ##__VA_ARGS__)
#else
    #define DBG_PRINT(fmt, ...)
#endif

/* 断言宏 */
#define ASSERT(cond, msg) \
    do { \
        if (!(cond)) { \
            fprintf(stderr, "断言失败: %s\n文件: %s, 行: %d\n", \
                    msg, __FILE__, __LINE__); \
            abort(); \
        } \
    } while(0)

/* 使用 */
DBG_PRINT("x = %d, y = %d", x, y);
ASSERT(ptr != NULL, "指针不能为空");
```

## 八、预处理的实际应用

### 8.1 配置管理

```c
/* config.h - 编译配置 */
#ifndef CONFIG_H
#define CONFIG_H

/* 功能开关 */
#define FEATURE_LOGIN       1
#define FEATURE_PAYMENT     0
#define FEATURE_ANALYTICS   1

/* 配置参数 */
#define MAX_USERS           1000

```

### 8.2 日志系统

```c
/* logging.h */
#ifndef LOGGING_H
#define LOGGING_H

#include <stdio.h>
#include <time.h>

#define LOG_LEVEL_NONE   0
#define LOG_LEVEL_ERROR  1
#define LOG_LEVEL_WARN   2
#define LOG_LEVEL_INFO   3
#define LOG_LEVEL_DEBUG  4

#ifndef CURRENT_LOG_LEVEL
    #define CURRENT_LOG_LEVEL LOG_LEVEL_DEBUG
#endif

#define LOG_ERROR(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_ERROR) \
        fprintf(stderr, "[ERROR] " fmt "\n", ##__VA_ARGS__); } while(0)

#define LOG_WARN(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_WARN) \
        fprintf(stderr, "[WARN]  " fmt "\n", ##__VA_ARGS__); } while(0)

#define LOG_INFO(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_INFO) \
        fprintf(stdout, "[INFO]  " fmt "\n", ##__VA_ARGS__); } while(0)

#define LOG_DEBUG(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_DEBUG) \
        fprintf(stdout, "[DEBUG] " fmt "\n", ##__VA_ARGS__); } while(0)

#endif
```

### 8.3 泛型宏（C11 _Generic）

```c
/* 使用 _Generic 实现泛型宏（C11）*/
#include <stdio.h>
#include <math.h>

#define ABS(x) _Generic((x), \
    int: abs, \
    long: labs, \
    long long: llabs, \
    float: fabsf, \
    double: fabs, \
    long double: fabsl \
)(x)

#define PRINT_VAL(x) _Generic((x), \
    int: printf("%d\n", x), \
    double: printf("%f\n", x), \
    char*: printf("%s\n", x), \
    default: printf("未知类型\n") \
)

int main(void) {
    printf("abs(-5) = %d\n", ABS(-5));
    printf("abs(-3.14) = %f\n", ABS(-3.14));
    PRINT_VAL(42);
    PRINT_VAL(3.14);
    PRINT_VAL("hello");
    return 0;
}
```

## 九、查看预处理结果

```bash
# 查看预处理后的完整代码
gcc -E source.c -o source.i

# 保留注释
gcc -E -C source.c -o source.i

# 查看包含的头文件列表
gcc -E -H source.c

# 只看宏定义
gcc -dM -E source.c

# 查看系统预定义宏
gcc -dM -E - < /dev/null

# 指定头文件路径
gcc -E -I./include source.c -o source.i
```

## 十、关键要点

> **重要提示**：
> 1. 预处理是文本替换过程，不涉及语法分析
> 2. 函数式宏的参数必须加括号，整个表达式也要加括号
> 3. `#ifndef HEADER_H / #define HEADER_H / #endif` 防止头文件重复包含
> 4. `#` 运算符将参数转为字符串，`##` 运算符连接标记
> 5. 条件编译常用于跨平台、调试、功能开关
> 6. 使用 `do { ... } while(0)` 包裹多行宏是最佳实践
> 7. `#pragma once` 是非标准但广泛支持的防止重复包含方式
> 8. 使用 `gcc -E` 可以查看预处理结果，有助于调试宏问题
