# 跨平台编程基础

## 一、跨平台编程概述

C语言的一大优势是可移植性，但不同平台之间仍存在差异。编写跨平台C程序需要了解这些差异并采取相应措施。

```
主要平台差异：
┌───────────┬────────────────┬────────────────┬────────────────┐
│           │    Windows     │     Linux      │     macOS      │
├───────────┼────────────────┼────────────────┼────────────────┤
│ 文件格式  │ PE (.exe/.dll) │ ELF (无扩展/.so)│ Mach-O (.dylib)│
│ 行尾符    │ \r\n           │ \n             │ \n             │
│ 路径分隔  │ \              │ /              │ /              │
│ 动态库    │ .dll           │ .so            │ .dylib         │
│ 静态库    │ .lib           │ .a             │ .a             │
│ 编译器    │ MSVC/GCC       │ GCC/Clang      │ Clang          │
│ 线程API   │ Win32 Thread   │ POSIX pthread  │ pthread        │
└───────────┴────────────────┴────────────────┴────────────────┘
```

## 二、平台检测宏

### 2.1 操作系统检测

```c

#ifdef PLATFORM_WINDOWS

    #endif
#else
    #define API __attribute__((visibility("default")))
#endif

    #define PATH_SEP_STR "\\"
#else
    #define PATH_SEP '/'
    #define PATH_SEP_STR "/"
#endif

/* 行尾符 */
#ifdef PLATFORM_WINDOWS
    #define LINE_ENDING "\r\n"
#else
    #define LINE_ENDING "\n"
#endif

#endif /* PLATFORM_H */
```

## 四、文件系统差异

### 4.1 路径处理

```c
#include "platform.h"
#include <stdio.h>
#include <string.h>

/* 跨平台路径拼接 */
void path_join(char *dest, size_t size, const char *dir, const char *file) {
    size_t dir_len = strlen(dir);
    if (dir_len > 0 && dir[dir_len - 1] == PATH_SEP) {
        snprintf(dest, size, "%s%s", dir, file);
    } else {
        snprintf(dest, size, "%s%c%s", dir, PATH_SEP, file);
    }
}

/* 规范化路径（将所有分隔符统一）*/
void path_normalize(char *path) {
    for (; *path; path++) {
        #ifdef PLATFORM_WINDOWS
            if (*path == '/') *path = '\\';
        #else
            if (*path == '\\') *path = '/';
        #endif
    }
}
```

### 4.2 文件操作

```c
#include <stdio.h>

/* 跨平台文件操作 */
#ifdef PLATFORM_WINDOWS

    #define F_OK 0
#else
    #include <unistd.h>
#endif

/* 安全打开文件 */

    #else

    return fp;
}
```

## 五、线程差异

### 5.1 线程封装

```c
/* thread.h - 跨平台线程封装 */
#ifndef THREAD_H
#define THREAD_H

#ifdef PLATFORM_WINDOWS
    #include <windows.h>
    typedef HANDLE thread_t;
    typedef CRITICAL_SECTION mutex_t;
#else
    #include <pthread.h>
    typedef pthread_t thread_t;
    typedef pthread_mutex_t mutex_t;
#endif

typedef void *(*thread_func_t)(void *);

/* 创建线程 */
static inline int thread_create(thread_t *thread, thread_func_t func, void *arg) {
    #ifdef PLATFORM_WINDOWS
        *thread = CreateThread(NULL, 0, (LPTHREAD_START_ROUTINE)func, arg, 0, NULL);
        return (*thread == NULL) ? -1 : 0;
    #else
        return pthread_create(thread, NULL, func, arg);
    #endif
}

/* 等待线程结束 */
static inline int thread_join(thread_t thread) {
    #ifdef PLATFORM_WINDOWS
        WaitForSingleObject(thread, INFINITE);
        CloseHandle(thread);
        return 0;
    #else
        return pthread_join(thread, NULL);
    #endif
}

/* 互斥锁操作 */
static inline int mutex_init(mutex_t *mutex) {
    #ifdef PLATFORM_WINDOWS
        InitializeCriticalSection(mutex);
        return 0;
    #else
        return pthread_mutex_init(mutex, NULL);
    #endif
}

static inline int mutex_lock(mutex_t *mutex) {
    #ifdef PLATFORM_WINDOWS
        EnterCriticalSection(mutex);
        return 0;
    #else
        return pthread_mutex_lock(mutex);
    #endif
}

static inline int mutex_unlock(mutex_t *mutex) {
    #ifdef PLATFORM_WINDOWS
        LeaveCriticalSection(mutex);
        return 0;
    #else
        return pthread_mutex_unlock(mutex);
    #endif
}

static inline int mutex_destroy(mutex_t *mutex) {
    #ifdef PLATFORM_WINDOWS
        DeleteCriticalSection(mutex);
        return 0;
    #else
        return pthread_mutex_destroy(mutex);
    #endif
}

#endif /* THREAD_H */
```

## 六、时间差异

### 6.1 高精度计时

```c
#include <stdint.h>

#ifdef PLATFORM_WINDOWS
    #include <windows.h>
    typedef LARGE_INTEGER time_point_t;

    static inline void time_now(time_point_t *tp) {
        QueryPerformanceCounter(tp);
    }

    static inline double time_diff_seconds(time_point_t start, time_point_t end) {
        LARGE_INTEGER freq;
        QueryPerformanceFrequency(&freq);
        return (double)(end.QuadPart - start.QuadPart) / (double)freq.QuadPart;
    }
#else
    #include <time.h>
    typedef struct timespec time_point_t;

    static inline void time_now(time_point_t *tp) {
        clock_gettime(CLOCK_MONOTONIC, tp);
    }

    static inline double time_diff_seconds(time_point_t start, time_point_t end) {
        return (end.tv_sec - start.tv_sec) + (end.tv_nsec - start.tv_nsec) / 1e9;
    }
#endif

/* 使用示例 */
/*
time_point_t start, end;
time_now(&start);
// ... 要计时的代码 ...
time_now(&end);
printf("耗时: %.6f 秒\n", time_diff_seconds(start, end));
*/
```

```c
    #endif
```

## 八、字节序处理

### 8.1 字节序检测和转换

```c
#include <stdint.h>

/* 检测字节序 */
static inline int is_little_endian(void) {
    uint16_t x = 1;
    return *(uint8_t *)&x == 1;
}

/* 字节序转换 */
static inline uint16_t swap_endian16(uint16_t x) {
    return (x >> 8) | (x << 8);
}

static inline uint32_t swap_endian32(uint32_t x) {
    return ((x >> 24) & 0x000000FF) |
           ((x >> 8)  & 0x0000FF00) |
           ((x << 8)  & 0x00FF0000) |
           ((x << 24) & 0xFF000000);
}
```

## 九、条件编译实践

### 9.1 完整的跨平台模块

```c
/* cross_platform.c */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* ========== 平台检测 ========== */
#if defined(_WIN32) || defined(_WIN64)
    #define PLATFORM_WINDOWS
#elif defined(__linux__)
    #define PLATFORM_LINUX
#elif defined(__APPLE__)
    #define PLATFORM_MACOS
#endif

/* ========== 平台特定头文件 ========== */
#ifdef PLATFORM_WINDOWS
    #include <windows.h>
    #include <direct.h>
    #include <io.h>
#else
    #include <unistd.h>
    #include <sys/stat.h>
    #include <dirent.h>
#endif

/* ========== 统一接口 ========== */

/* 获取当前目录 */
int get_current_directory(char *buf, size_t size) {
    #ifdef PLATFORM_WINDOWS
        return _getcwd(buf, (int)size) != NULL ? 0 : -1;
    #else
        return getcwd(buf, size) != NULL ? 0 : -1;
    #endif
}

/* 检查文件是否存在 */
int file_exists(const char *path) {
    #ifdef PLATFORM_WINDOWS
        return _access(path, 0) == 0;
    #else
        return access(path, F_OK) == 0;
    #endif
}

/* 创建目录 */
int make_directory(const char *path) {
    #ifdef PLATFORM_WINDOWS
        return _mkdir(path);
    #else
        return mkdir(path, 0755);
    #endif
}

/* 睡眠毫秒 */
void sleep_milliseconds(int ms) {
    #ifdef PLATFORM_WINDOWS
        Sleep(ms);
    #else
        usleep(ms * 1000);
    #endif
}

/* 获取环境变量 */
const char *get_env(const char *name) {
    #ifdef PLATFORM_WINDOWS
        static char buf[1024];
        DWORD len = GetEnvironmentVariableA(name, buf, sizeof(buf));
        return (len > 0 && len < sizeof(buf)) ? buf : NULL;
    #else
        return getenv(name);
    #endif
}

/* ========== 测试 ========== */
int main(void) {
    char cwd[1024];

    #ifdef PLATFORM_WINDOWS
        printf("Platform: Windows\n");
    #elif defined(PLATFORM_LINUX)
        printf("Platform: Linux\n");
    #elif defined(PLATFORM_MACOS)
        printf("Platform: macOS\n");
    #endif

    if (get_current_directory(cwd, sizeof(cwd)) == 0) {
        printf("Current directory: %s\n", cwd);
    }

    printf("File exists: %s\n", file_exists(__FILE__) ? "yes" : "no");

    return 0;
}
```

## 十、关键要点

> **重要提示**：
> 1. 使用预处理宏检测平台：`_WIN32`、`__linux__`、`__APPLE__`
> 2. 路径分隔符：Windows用 `\`，Unix用 `/`
> 3. 行尾符：Windows用 `\r\n`，Unix用 `\n`
> 4. 动态库：Windows用 `.dll`，Linux用 `.so`，macOS用 `.dylib`
> 5. 线程：Windows用 Win32 API，Unix用 POSIX pthread
> 6. 时间：Windows用 `QueryPerformanceCounter`，Unix用 `clock_gettime`
> 7. 使用条件编译隔离平台差异，对外提供统一接口
> 8. 固定宽度整数类型（`stdint.h`）有助于保证可移植性
> 9. 避免依赖特定平台的行为（如文件路径格式）
> 10. 在多个平台上测试代码以确保真正的可移植性
