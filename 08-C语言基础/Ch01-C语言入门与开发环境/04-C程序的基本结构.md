# C程序的基本结构

## 一、C程序的组成要素

一个完整的C程序由以下几个基本部分组成：

```
┌─────────────────────────────────┐
│        预处理指令区域             │  ① #include, #define 等
├─────────────────────────────────┤
│        全局声明区域               │  ② 全局变量、函数声明
├─────────────────────────────────┤
│        main 函数                 │  ③ 程序入口
│   ┌───────────────────────────┐ │
│   │    局部变量声明            │ │  ④ 变量定义
│   │    执行语句               │ │  ⑤ 赋值、调用、控制流等
│   │    return 语句            │ │  ⑥ 返回值
│   └───────────────────────────┘ │
├─────────────────────────────────┤
│        其他函数定义               │  ⑦ 自定义函数
└─────────────────────────────────┘
```

## 二、完整示例程序

```c
/*
 * 文件名: calculator.c
 * 功能: 简单的计算器程序
 * 作者: 学习者
 * 日期: 2024-01-01
 */

/* ==================== 预处理指令 ==================== */
#include <stdio.h>    /* 标准输入输出 */
#include <stdlib.h>   /* 标准库函数 */
#include <math.h>     /* 数学函数 */

/* ==================== 宏定义 ==================== */
#define PI 3.14159265358979
#define MAX(a, b) ((a) > (b) ? (a) : (b))

/* ==================== 函数声明 ==================== */
double add(double a, double b);
double subtract(double a, double b);
double multiply(double a, double b);
double divide(double a, double b, int *success);

/* ==================== 全局变量 ==================== */
int operation_count = 0;  /* 记录操作次数 */

/* ==================== main 函数 ==================== */
int main(void) {
    /* 局部变量声明 */
    double num1, num2, result;
    char op;
    int success;

    printf("=== 简单计算器 ===\n");
    printf("请输入表达式（如 3 + 4）：");

    /* 读取输入 */
    scanf("%lf %c %lf", &num1, &op, &num2);

    /* 执行计算 */
    switch (op) {
        case '+':
            result = add(num1, num2);
            break;
        case '-':
            result = subtract(num1, num2);
            break;
        case '*':
            result = multiply(num1, num2);
            break;
        case '/':
            result = divide(num1, num2, &success);
            if (!success) {
                printf("错误：除数不能为零！\n");
                return 1;
            }
            break;
        default:
            printf("错误：未知运算符 '%c'\n", op);
            return 1;
    }

    /* 输出结果 */
    printf("结果: %.2f %c %.2f = %.2f\n", num1, op, num2, result);
    printf("总共执行了 %d 次操作\n", operation_count);

    return 0;
}

/* ==================== 函数定义 ==================== */
double add(double a, double b) {
    operation_count++;
    return a + b;
}

double subtract(double a, double b) {
    operation_count++;
    return a - b;
}

double multiply(double a, double b) {
    operation_count++;
    return a * b;
}

double divide(double a, double b, int *success) {
    if (b == 0) {
        *success = 0;
        return 0;
    }
    *success = 1;
    operation_count++;
    return a / b;
}
```

## 三、预处理指令

### 3.1 头文件包含

```c
/* 方式1：尖括号 - 从系统路径查找 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* 方式2：双引号 - 先从当前目录查找，再查找系统路径 */
#include "myheader.h"
#include "utils/utils.h"
```

### 3.2 宏定义

```c
/* 对象式宏 */
#define MAX_SIZE    1024
#define PI          3.14159265
#define DEBUG_MODE  1

/* 函数式宏 */
#define SQUARE(x)    ((x) * (x))
#define MAX(a, b)    ((a) > (b) ? (a) : (b))
#define MIN(a, b)    ((a) < (b) ? (a) : (b))

/* 多行宏（使用反斜杠续行）*/
#define PRINT_DEBUG(msg) \
    do { \
        if (DEBUG_MODE) { \
            printf("[DEBUG] %s (file: %s, line: %d)\n", \
                   msg, __FILE__, __LINE__); \
        } \
    } while(0)
```

### 3.3 条件编译

```c
/* 防止头文件重复包含 */
#ifndef MY_HEADER_H
#define MY_HEADER_H

/* 头文件内容 */
void my_function(void);

#endif /* MY_HEADER_H */

/* 平台条件编译 */
#ifdef _WIN32
    #include <windows.h>
    #define PLATFORM "Windows"
#elif defined(__linux__)
    #include <unistd.h>
    #define PLATFORM "Linux"
#elif defined(__APPLE__)
    #include <TargetConditionals.h>
    #define PLATFORM "macOS"
#else
    #define PLATFORM "Unknown"
#endif

/* 版本控制 */
#if C_VERSION >= 11
    _Static_assert(sizeof(int) >= 4, "int too small");
#endif
```

## 四、注释

### 4.1 单行注释（C99+）

```c
// 这是单行注释

```

### 4.2 多行注释（K&R风格）

```c
/* 这是多行注释
 * 可以跨越多行
 * 常用于函数说明
 */

/*
 * 函数名: calculate_average
 * 功能: 计算数组的平均值
 * 参数:
 *   arr - 整数数组
 *   n   - 数组长度
 * 返回值: 平均值
 */
double calculate_average(int arr[], int n) {
    /* 实现代码 */
    return 0.0;
}
```

### 4.3 文档注释风格

```c
/**
 * @brief  计算两个整数的最大公约数
 * @param  a  第一个整数
 * @param  b  第二个整数
 * @return 最大公约数
 * @note   使用欧几里得算法实现
 * @example
 *   int result = gcd(12, 8);  // result = 4
 */
int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}
```

## 五、语句与表达式

### 5.1 语句类型

```c
#include <stdio.h>

int main(void) {
    /* 1. 声明语句 */
    int x;
    double y = 3.14;
    char name[50];

    /* 2. 表达式语句 */
    x = 10;               /* 赋值语句 */
    x++;                   /* 自增语句 */
    printf("Hello\n");    /* 函数调用语句 */

    /* 3. 复合语句（代码块）*/
    {
        int temp = x;
        x = x * 2;
        printf("temp=%d, x=%d\n", temp, x);
    }

    /* 4. 控制流语句 */
    if (x > 0) {
        printf("x是正数\n");
    } else {
        printf("x不是正数\n");
    }

    /* 5. 空语句 */
    ;  /* 什么也不做 */

    return 0;  /* 6. 跳转语句 */
}
```

### 5.2 表达式

```c
/* 表达式是产生值的代码片段 */

int a = 5, b = 3;

/* 算术表达式 */
int sum = a + b;        /* 结果: 8 */
int diff = a - b;       /* 结果: 2 */
int prod = a * b;       /* 结果: 15 */

/* 关系表达式 */
int is_greater = (a > b);  /* 结果: 1 (真) */
int is_equal = (a == b);   /* 结果: 0 (假) */

/* 逻辑表达式 */
int logic_and = (a > 0 && b > 0);  /* 结果: 1 */
int logic_or = (a > 0 || b < 0);   /* 结果: 1 */

/* 条件表达式（三目运算符）*/
int max = (a > b) ? a : b;  /* 结果: 5 */

/* 逗号表达式（取最后一个表达式的值）*/
int result = (a++, b++, a + b);  /* 结果: 10 */
```

## 六、函数

### 6.1 函数的基本结构

```c
/*
 * 返回值类型 函数名(参数列表)
 * {
 *     局部变量声明;
 *     执行语句;
 *     return 返回值;
 * }
 */

/* 示例1：无参无返回值 */
void greet(void) {
    printf("你好！\n");
}

/* 示例2：有参有返回值 */
int square(int n) {
    return n * n;
}

/* 示例3：多参数 */
double average(double a, double b, double c) {
    return (a + b + c) / 3.0;
}

/* 示例4：指针参数（修改调用者的变量）*/
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

/* 示例5：数组参数 */
int sum_array(int arr[], int n) {
    int total = 0;
    for (int i = 0; i < n; i++) {
        total += arr[i];
    }
    return total;
}
```

### 6.2 函数声明与定义

```c
/* =============== 头文件区域 =============== */
/* 函数声明（原型）：告诉编译器函数的存在 */
int add(int a, int b);
void print_message(const char *msg);
double calculate_pi(void);

/* =============== main 函数 =============== */
int main(void) {
    int result = add(3, 4);
    print_message("Hello");
    return 0;
}

/* =============== 函数定义区域 =============== */
/* 在main之后定义函数，但需要在前面声明 */
int add(int a, int b) {
    return a + b;
}

void print_message(const char *msg) {
    printf("%s\n", msg);
}

double calculate_pi(void) {
    return 3.14159265358979;
}
```

## 七、变量的作用域

```c
#include <stdio.h>

/* 全局变量 - 文件作用域 */
int global_var = 100;

/* 静态全局变量 - 仅本文件可见 */
static int file_static_var = 200;

void demo_function(void) {
    /* 局部变量 - 函数作用域 */
    int local_var = 10;

    /* 静态局部变量 - 函数结束时不销毁 */
    static int call_count = 0;
    call_count++;

    printf("调用次数: %d\n", call_count);
    printf("local_var: %d\n", local_var);
    printf("global_var: %d\n", global_var);
}

int main(void) {
    /* 局部变量会屏蔽同名的全局变量 */
    int global_var = 42;  /* 这是一个新的局部变量 */

    printf("main中的 global_var: %d\n", global_var);       /* 42 */
    printf("文件级的 global_var: %d\n", ::global_var);      /* C语言不支持 :: 运算符 */

    /* 正确访问全局变量的方式是不要同名 */

    demo_function();  /* 调用次数: 1 */
    demo_function();  /* 调用次数: 2 */

    return 0;
}
```

## 八、程序结构模板

```c
/*
 * 文件名: project_template.c
 * 描述: C程序结构模板
 */

/* ========== 第一部分：预处理指令 ========== */
/* 标准库头文件 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* 自定义头文件 */
/* #include "myheader.h" */

/* 宏定义 */
#define VERSION "1.0.0"
#define MAX_BUFFER 256

/* 条件编译 */
#ifdef DEBUG
    #define DBG_PRINT(fmt, ...) printf("[DBG] " fmt "\n", ##__VA_ARGS__)
#else
    #define DBG_PRINT(fmt, ...)
#endif

/* ========== 第二部分：类型定义 ========== */
typedef struct {
    int x;
    int y;
} Point;

typedef enum {
    STATUS_OK,
    STATUS_ERROR,
    STATUS_TIMEOUT
} Status;

/* ========== 第三部分：全局变量 ========== */
static int g_initialized = 0;

/* ========== 第四部分：函数声明 ========== */
Status initialize(void);
void cleanup(void);
void process(void);

/* ========== 第五部分：main函数 ========== */
int main(int argc, char *argv[]) {
    /* 解析命令行参数 */
    if (argc > 1) {
        printf("参数: %s\n", argv[1]);
    }

    /* 初始化 */
    if (initialize() != STATUS_OK) {
        fprintf(stderr, "初始化失败\n");
        return EXIT_FAILURE;
    }

    /* 主处理逻辑 */
    process();

    /* 清理 */
    cleanup();

    return EXIT_SUCCESS;
}

/* ========== 第六部分：函数定义 ========== */
Status initialize(void) {
    DBG_PRINT("初始化开始");
    g_initialized = 1;
    DBG_PRINT("初始化完成");
    return STATUS_OK;
}

void cleanup(void) {
    DBG_PRINT("清理资源");
    g_initialized = 0;
}

void process(void) {
    DBG_PRINT("开始处理");
    printf("C程序模板 v%s\n", VERSION);
}
```

## 九、关键要点

> **重要提示**：
> 1. 预处理指令以 `#` 开头，在编译前处理
> 2. `main` 函数是程序的入口点
> 3. 函数声明可以放在 `main` 之前，定义放在之后
> 4. 变量有作用域：局部变量、全局变量、静态变量
> 5. 注释用于提高代码可读性，不会被编译
> 6. 良好的代码结构包括：预处理、声明、main、函数定义
> 7. 头文件防止重复包含使用 `#ifndef` / `#define` / `#endif`
> 8. 使用 `static` 关键字可以限制变量和函数的作用域
