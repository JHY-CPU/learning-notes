## 二、头文件设计

### 2.1 头文件原则

```c
/*
 * 头文件设计原则：
 * 1. 头文件只放声明，不放定义（内联函数例外）
 * 2. 使用 include guard 防止重复包含
 * 3. 头文件应自包含（不依赖使用者先包含其他头文件）
 * 4. 最小化头文件依赖
 * 5. 区分公共头文件和内部头文件
 */

/* common.h - 公共头文件示例 */
#ifndef MYPROJECT_COMMON_H
#define MYPROJECT_COMMON_H

#include <stddef.h>    /* size_t */
#include <stdint.h>    /* 固定宽度整数 */

/* 版本信息 */
#define MYPROJECT_VERSION_MAJOR 1
#define MYPROJECT_VERSION_MINOR 0
#define MYPROJECT_VERSION_PATCH 0

/* 返回值定义 */
typedef enum {
    MP_OK = 0,
    MP_ERROR = -1,
    MP_ERROR_INVALID_PARAM = -2,
    MP_ERROR_OUT_OF_MEMORY = -3,
    MP_ERROR_IO = -4,
} mp_status_t;

/* 常用类型定义 */
typedef struct mp_context mp_context_t;

/* 公共函数声明 */
mp_status_t mp_init(mp_context_t **ctx);
void mp_cleanup(mp_context_t *ctx);

#endif /* MYPROJECT_COMMON_H */
```

### 2.2 头文件依赖管理

```c
/* 错误：头文件不自包含 */
/* module.h */
#ifndef MODULE_H
#define MODULE_H
void process(Data *data);   /* Data 从哪来？*/
#endif

/* 正确：头文件自包含 */
/* module.h */
#ifndef MODULE_H
#define MODULE_H
#include "types.h"          /* 自己包含需要的头文件 */
void process(Data *data);
#endif

/* 最佳实践：使用前置声明减少依赖 */
/* module.h */
#ifndef MODULE_H
#define MODULE_H
typedef struct Data Data;   /* 前置声明 */
void process(Data *data);   /* 只需要指针，不需要完整定义 */
#endif
```

## 三、内存管理

### 3.1 内存分配和释放

```c
#include <stdlib.h>
#include <string.h>

/* 原则1：分配后立即检查 */
void *safe_malloc(size_t size) {
    void *ptr = malloc(size);
    if (ptr == NULL) {
        fprintf(stderr, "内存分配失败: %zu 字节\n", size);
        abort();
    }
    return ptr;
}

/* 原则2：使用配对的分配/释放函数 */
int *create_int_array(int n) {
    int *arr = (int *)calloc(n, sizeof(int));  /* calloc初始化为0 */
    return arr;
}

void destroy_int_array(int *arr) {
    free(arr);
    /* 不要在free后使用arr */
}

/* 原则3：释放后置空 */
#define SAFE_FREE(ptr) do { free(ptr); (ptr) = NULL; } while(0)

/* 原则4：统一的资源管理模式 */
typedef struct {
    char *buffer;
    FILE *file;
    int *temp_array;
} Resources;

Resources *resources_create(void) {
    Resources *res = (Resources *)calloc(1, sizeof(Resources));
    return res;
}

void resources_destroy(Resources *res) {
    if (res == NULL) return;
    free(res->buffer);
    if (res->file) fclose(res->file);
    free(res->temp_array);
    free(res);
}
```

### 3.2 避免内存问题

```c
/* 避免内存泄漏 */
char *read_line(FILE *fp) {
    char *line = NULL;
    size_t len = 0;
    ssize_t read;

    if ((read = getline(&line, &len, fp)) == -1) {
        free(line);    /* 失败时也要释放 */
        return NULL;
    }
    return line;       /* 调用者负责释放 */
}

/* 避免悬空指针 */
void example(void) {
    int *p = (int *)malloc(sizeof(int));
    *p = 42;
    /* ... 使用 p ... */
    free(p);
    p = NULL;   /* 防止悬空指针 */
}

/* 避免缓冲区溢出 */
void safe_strcpy(char *dest, size_t dest_size, const char *src) {
    strncpy(dest, src, dest_size - 1);
    dest[dest_size - 1] = '\0';   /* 确保以null结尾 */
}

/* 使用安全的字符串函数 */
/*
 * 推荐使用：
 * snprintf() 代替 sprintf()
 * strncpy() 代替 strcpy()（注意手动添加null结尾）
 * strncat() 代替 strcat()
 * fgets()    代替 gets()（gets已被C11移除）
 */
```

## 四、错误处理

### 4.1 错误处理策略

```c
/* 策略1：返回错误码 */
typedef enum {
    SUCCESS = 0,
    ERR_INVALID_PARAM,
    ERR_FILE_NOT_FOUND,
    ERR_OUT_OF_MEMORY,
    ERR_PERMISSION_DENIED,
    ERR_IO,
    ERR_TIMEOUT,
} ErrorCode;

ErrorCode read_config(const char *path, Config **cfg) {
    if (path == NULL || cfg == NULL) {
        return ERR_INVALID_PARAM;
    }

    FILE *fp = fopen(path, "r");
    if (fp == NULL) {
        return ERR_FILE_NOT_FOUND;
    }

    *cfg = (Config *)malloc(sizeof(Config));
    if (*cfg == NULL) {
        fclose(fp);
        return ERR_OUT_OF_MEMORY;
    }

    /* ... 解析配置 ... */

    fclose(fp);
    return SUCCESS;
}

/* 策略2：错误码 + 详细错误信息 */
static char g_error_msg[256] = {0};

const char *get_last_error(void) {
    return g_error_msg;
}

int open_database(const char *path) {
    if (path == NULL) {
        snprintf(g_error_msg, sizeof(g_error_msg), "路径不能为空");
        return -1;
    }
    /* ... */
    return 0;
}

/* 策略3：goto清理模式（Linux内核风格）*/
int process_file(const char *input, const char *output) {
    int ret = -1;
    FILE *fin = NULL;
    FILE *fout = NULL;
    char *buffer = NULL;

    fin = fopen(input, "r");
    if (fin == NULL) {
        fprintf(stderr, "无法打开输入文件: %s\n", input);
        goto cleanup;
    }

    fout = fopen(output, "w");
    if (fout == NULL) {
        fprintf(stderr, "无法打开输出文件: %s\n", output);
        goto cleanup;
    }

    buffer = (char *)malloc(4096);
    if (buffer == NULL) {
        fprintf(stderr, "内存分配失败\n");
        goto cleanup;
    }

    /* 处理文件 */
    while (fgets(buffer, 4096, fin) != NULL) {
        fputs(buffer, fout);
    }

    ret = 0;  /* 成功 */

cleanup:
    free(buffer);
    if (fout) fclose(fout);
    if (fin) fclose(fin);
    return ret;
}
```

## 五、模块化编程

### 5.1 模块接口设计

```c
/* stack.h - 栈模块接口 */
#ifndef STACK_H
#define STACK_H

#include <stddef.h>

/* 不透明类型（隐藏实现细节）*/
typedef struct Stack Stack;

/* 创建和销毁 */
Stack *stack_create(size_t element_size, size_t capacity);
void stack_destroy(Stack *stack);

/* 基本操作 */
int stack_push(Stack *stack, const void *element);
int stack_pop(Stack *stack, void *element);
int stack_peek(const Stack *stack, void *element);

/* 状态查询 */
size_t stack_size(const Stack *stack);
int stack_is_empty(const Stack *stack);

#endif
```

```c
/* stack.c - 栈模块实现 */
#include "stack.h"
#include <stdlib.h>
#include <string.h>

struct Stack {
    void *data;
    size_t element_size;
    size_t capacity;
    size_t size;
};

Stack *stack_create(size_t element_size, size_t capacity) {
    if (element_size == 0 || capacity == 0) return NULL;

    Stack *stack = (Stack *)malloc(sizeof(Stack));
    if (stack == NULL) return NULL;

    stack->data = malloc(element_size * capacity);
    if (stack->data == NULL) {
        free(stack);
        return NULL;
    }

    stack->element_size = element_size;
    stack->capacity = capacity;
    stack->size = 0;

    return stack;
}

void stack_destroy(Stack *stack) {
    if (stack == NULL) return;
    free(stack->data);
    free(stack);
}

int stack_push(Stack *stack, const void *element) {
    if (stack == NULL || element == NULL) return -1;
    if (stack->size >= stack->capacity) return -1;

    char *dest = (char *)stack->data + stack->size * stack->element_size;
    memcpy(dest, element, stack->element_size);
    stack->size++;

    return 0;
}

int stack_pop(Stack *stack, void *element) {
    if (stack == NULL || stack->size == 0) return -1;

    stack->size--;
    if (element != NULL) {
        char *src = (char *)stack->data + stack->size * stack->element_size;
        memcpy(element, src, stack->element_size);
    }

    return 0;
}

size_t stack_size(const Stack *stack) {
    return stack ? stack->size : 0;
}

int stack_is_empty(const Stack *stack) {
    return stack == NULL || stack->size == 0;
}
```

### 5.2 使用模块

```c
/* main.c - 使用栈模块 */
#include <stdio.h>
#include "stack.h"

int main(void) {
    Stack *stack = stack_create(sizeof(int), 100);
    if (stack == NULL) {
        fprintf(stderr, "创建栈失败\n");
        return 1;
    }

    /* 压入数据 */
    for (int i = 0; i < 10; i++) {
        stack_push(stack, &i);
    }

    printf("栈大小: %zu\n", stack_size(stack));

    /* 弹出数据 */
    int value;
    while (!stack_is_empty(stack)) {
        stack_pop(stack, &value);
        printf("%d ", value);
    }
    printf("\n");

    stack_destroy(stack);
    return 0;
}
```

### 6.2 const正确性

```c
/* 使用const保护不应修改的数据 */

/* 参数const：不修改输入 */
int calculate_sum(const int *arr, int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += arr[i];
        // arr[i] = 0;  /* 编译错误 */
    }
    return sum;
}

/* 指针const：指针本身不可修改 */
void process(int *const ptr) {
    *ptr = 42;    /* 可以修改指向的值 */
    // ptr = NULL; /* 编译错误：指针本身不能修改 */
}

/* 双重const：指针和指向的值都不能修改 */
void read_only(const int *const ptr) {
    // *ptr = 42;  /* 编译错误 */
    // ptr = NULL; /* 编译错误 */
    printf("%d\n", *ptr);
}

/* 返回const：防止调用者修改 */
const char *get_version(void) {
    return "1.0.0";  /* 返回的字符串不应被修改 */
}
```

## 七、测试

### 7.1 简单测试框架

```c
/* test_framework.h */
#ifndef TEST_FRAMEWORK_H
#define TEST_FRAMEWORK_H

#include <stdio.h>
#include <stdlib.h>

static int g_tests_run = 0;
static int g_tests_passed = 0;

#define TEST_START(name) \
    do { \
        g_tests_run++; \
        printf("[测试] %s ... ", name); \
    } while(0)

#define TEST_PASS() \
    do { \
        g_tests_passed++; \
        printf("通过\n"); \
    } while(0)

#define TEST_FAIL(msg) \
    do { \
        printf("失败: %s (文件: %s, 行: %d)\n", msg, __FILE__, __LINE__); \
    } while(0)

#define ASSERT_TRUE(cond) \
    do { \
        if (!(cond)) { \
            TEST_FAIL("条件为假: " #cond); \
            return -1; \
        } \
    } while(0)

#define ASSERT_EQ(a, b) \
    do { \
        if ((a) != (b)) { \
            TEST_FAIL("不相等: " #a " != " #b); \
            return -1; \
        } \
    } while(0)

#define ASSERT_NOT_NULL(ptr) \
    do { \
        if ((ptr) == NULL) { \
            TEST_FAIL("指针为空: " #ptr); \
            return -1; \
        } \
    } while(0)

#define TEST_SUMMARY() \
    do { \
        printf("\n测试完成: %d/%d 通过\n", g_tests_passed, g_tests_run); \
        return (g_tests_passed == g_tests_run) ? 0 : 1; \
    } while(0)

#endif
```

## 九、性能考虑

### 9.1 常见优化技巧

```c
/* 1. 使用内联函数替代简单宏 */
static inline int max(int a, int b) {
    return (a > b) ? a : b;
}

/* 2. 避免不必要的内存分配 */
/* 不好 */
void process(void) {
    char *buf = malloc(1024);
    /* ... */
    free(buf);
}

/* 好：使用栈缓冲区 */
void process(void) {
    char buf[1024];
    /* ... */
}

/* 3. 循环优化 */
/* 不好 */
for (int i = 0; i < strlen(s); i++) {  /* 每次循环都调用strlen */
    /* ... */
}

/* 好 */
size_t len = strlen(s);
for (size_t i = 0; i < len; i++) {
    /* ... */
}

/* 4. 使用restrict关键字告诉编译器无别名 */
void add_arrays(int *restrict dest,
                const int *restrict a,
                const int *restrict b,
                int n) {
    for (int i = 0; i < n; i++) {
        dest[i] = a[i] + b[i];
    }
}
```

## 十一、关键要点

> **重要提示**：
> 1. 项目结构应清晰分离：include、src、tests
> 2. 头文件只放声明，使用 include guard 保护
> 3. 内存管理遵循：分配-检查-使用-释放-置空 的流程
> 4. 使用错误码或goto清理模式进行错误处理
> 5. 模块化设计使用不透明指针隐藏实现细节
> 6. 充分使用const保护不应修改的数据
> 7. 避免全局状态，使用上下文结构体传递状态
> 8. 编写自动化测试，保证代码质量
> 9. 使用断言（assert）检查程序的前置/后置条件
> 10. 坚持代码审查，保持代码风格一致性
