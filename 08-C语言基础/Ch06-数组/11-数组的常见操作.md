# 数组的常见操作

## 1. 查找

### 1.1 线性查找

在数组中逐个比较，查找目标值。

```c
#include <stdio.h>

// 返回目标值的下标，未找到返回-1
int linear_search(const int arr[], int n, int target) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == target) {
            return i;
        }
    }
    return -1;
}

int main(void) {
    int arr[] = {23, 45, 12, 67, 34, 89, 56};
    int n = sizeof(arr) / sizeof(arr[0]);

    int idx = linear_search(arr, n, 67);
    if (idx != -1)
        printf("找到67，下标为%d\n", idx);
    else
        printf("未找到\n");

    return 0;
}
```

### 1.2 查找最大值和最小值

```c
int find_max(const int arr[], int n) {
    int max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) max = arr[i];
    }
    return max;
}

int find_min(const int arr[], int n) {
    int min = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] < min) min = arr[i];
    }
    return min;
}
```

## 2. 排序（冒泡排序）

```c
#include <stdio.h>

void bubble_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int swapped = 0;
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = 1;
            }
        }
        if (!swapped) break;  // 未发生交换，已有序
    }
}

int main(void) {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);

    bubble_sort(arr, n);

    printf("排序后: ");
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");

    return 0;
}
```

## 3. 插入元素

### 3.1 在指定位置插入

```c
#include <stdio.h>

#define MAX_SIZE 100

// 在arr的index位置插入value，*n为当前元素个数
int insert_at(int arr[], int *n, int index, int value) {
    if (*n >= MAX_SIZE || index < 0 || index > *n) {
        return 0;  // 插入失败
    }

    // 从末尾开始，逐个后移
    for (int i = *n; i > index; i--) {
        arr[i] = arr[i - 1];
    }

    arr[index] = value;
    (*n)++;
    return 1;  // 插入成功
}

int main(void) {
    int arr[MAX_SIZE] = {10, 20, 30, 40, 50};
    int n = 5;

    insert_at(arr, &n, 2, 25);  // 在下标2处插入25

    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    // 输出: 10 20 25 30 40 50

    return 0;
}
```

### 3.2 有序插入

```c
// 在已排序数组中插入，保持有序
void insert_sorted(int arr[], int *n, int value) {
    int i = *n - 1;
    while (i >= 0 && arr[i] > value) {
        arr[i + 1] = arr[i];
        i--;
    }
    arr[i + 1] = value;
    (*n)++;
}
```

## 4. 删除元素

### 4.1 按位置删除

```c
int remove_at(int arr[], int *n, int index) {
    if (index < 0 || index >= *n) {
        return 0;
    }

    // 从index开始，逐个前移
    for (int i = index; i < *n - 1; i++) {
        arr[i] = arr[i + 1];
    }

    (*n)--;
    return 1;
}
```

### 4.2 按值删除（删除第一个匹配项）

```c
int remove_value(int arr[], int *n, int value) {
    for (int i = 0; i < *n; i++) {
        if (arr[i] == value) {
            return remove_at(arr, &i, i) ? 1 : 0;
            // 注意：这里应该用 n 而不是 &i
        }
    }
    return 0;
}

// 正确写法
int remove_first(int arr[], int *n, int value) {
    for (int i = 0; i < *n; i++) {
        if (arr[i] == value) {
            for (int j = i; j < *n - 1; j++) {
                arr[j] = arr[j + 1];
            }
            (*n)--;
            return 1;
        }
    }
    return 0;
}
```

## 5. 反转数组

```c
void reverse(int arr[], int n) {
    for (int i = 0; i < n / 2; i++) {
        int temp = arr[i];
        arr[i] = arr[n - 1 - i];
        arr[n - 1 - i] = temp;
    }
}

// 使用双指针的写法
void reverse_v2(int arr[], int n) {
    int *left = arr;
    int *right = arr + n - 1;
    while (left < right) {
        int temp = *left;
        *left = *right;
        *right = temp;
        left++;
        right--;
    }
}
```

## 6. 数组去重（已排序数组）

```c
int remove_duplicates(int arr[], int n) {
    if (n == 0) return 0;

    int j = 0;  // 新数组的末尾下标
    for (int i = 1; i < n; i++) {
        if (arr[i] != arr[j]) {
            j++;
            arr[j] = arr[i];
        }
    }
    return j + 1;  // 返回去重后的长度
}
```

## 7. 数组合并

```c
void merge(const int a[], int na, const int b[], int nb, int result[]) {
    int i = 0, j = 0, k = 0;
    while (i < na && j < nb) {
        if (a[i] <= b[j])
            result[k++] = a[i++];
        else
            result[k++] = b[j++];
    }
    while (i < na) result[k++] = a[i++];
    while (j < nb) result[k++] = b[j++];
}
```

---

## 重点总结

- 插入/删除操作需要**移动元素**，时间复杂度O(n)
- 反转数组只需**交换一半**的元素对
- 排序是很多操作的前提（二分查找、去重等）
- 有序数组的合并利用**双指针**技巧，时间复杂度O(n+m)
- 保持数组操作的**边界检查**，避免越界
