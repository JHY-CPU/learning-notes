# 数组的应用：直方图

## 1. 用数组统计频率

直方图的本质是用数组的**下标**作为类别，用**元素值**作为计数。

```c
// 统计掷骰子各面出现的次数
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void) {
    int freq[6] = {0};  // freq[0]对应1点，freq[5]对应6点

    srand((unsigned)time(NULL));

    // 模拟掷骰子10000次
    for (int i = 0; i < 10000; i++) {
        int roll = rand() % 6;  // 0-5
        freq[roll]++;
    }

    // 输出频率
    for (int i = 0; i < 6; i++) {
        printf("面%d: %4d次 (%.1f%%)\n",
               i + 1, freq[i], freq[i] / 100.0);
    }

    return 0;
}
```

## 2. 字符频率统计

```c
#include <stdio.h>
#include <ctype.h>

int main(void) {
    int freq[26] = {0};  // 'a'-'z' 的频率
    char text[] = "Hello World Programming";

    for (int i = 0; text[i] != '\0'; i++) {
        char c = tolower(text[i]);
        if (c >= 'a' && c <= 'z') {
            freq[c - 'a']++;
        }
    }

    printf("字母频率:\n");
    for (int i = 0; i < 26; i++) {
        if (freq[i] > 0) {
            printf("'%c': %d\n", 'a' + i, freq[i]);
        }
    }

    return 0;
}
```

## 3. 绘制文本直方图

### 3.1 水平直方图

```c
#include <stdio.h>

void draw_horizontal_hist(const int data[], int n, const char *labels[]) {
    printf("\n水平直方图:\n");
    for (int i = 0; i < n; i++) {
        printf("%-8s | ", labels[i]);
        for (int j = 0; j < data[i]; j++) {
            printf("#");
        }
        printf(" (%d)\n", data[i]);
    }
}

int main(void) {
    int scores[] = {0, 0, 0, 0, 0};  // <60, 60-69, 70-79, 80-89, 90-100
    const char *labels[] = {"<60", "60-69", "70-79", "80-89", "90-100"};

    int grades[] = {85, 92, 78, 90, 55, 88, 76, 95, 82, 63, 79, 91, 67, 84};
    int n = sizeof(grades) / sizeof(grades[0]);

    // 统计各分数段
    for (int i = 0; i < n; i++) {
        if (grades[i] >= 90)      scores[4]++;
        else if (grades[i] >= 80) scores[3]++;
        else if (grades[i] >= 70) scores[2]++;
        else if (grades[i] >= 60) scores[1]++;
        else                      scores[0]++;
    }

    draw_horizontal_hist(scores, 5, labels);

    return 0;
}
```

输出：
```
水平直方图:
<60      | ## (2)
60-69    | ## (2)
70-79    | ### (3)
80-89    | #### (4)
90-100   | ### (3)
```

### 3.2 垂直直方图

```c
#include <stdio.h>

void draw_vertical_hist(const int data[], int n, const char *labels[]) {
    // 找最大值
    int max_val = 0;
    for (int i = 0; i < n; i++) {
        if (data[i] > max_val) max_val = data[i];
    }

    printf("\n垂直直方图:\n");

    // 从上到下逐行绘制
    for (int row = max_val; row >= 1; row--) {
        printf("%3d | ", row);
        for (int col = 0; col < n; col++) {
            if (data[col] >= row)
                printf(" ## ");
            else
                printf("    ");
        }
        printf("\n");
    }

    // 底部标签
        printf("    +");
    for (int i = 0; i < n; i++) printf("----");
    printf("\n      ");
    for (int i = 0; i < n; i++) printf("%-4s", labels[i]);
    printf("\n");
}
```

## 4. 温度分布直方图

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void) {
    // 温度范围: -10 到 40 度，共51个桶
    int temp_hist[51] = {0};

    srand((unsigned)time(NULL));

    // 模拟一年365天的最高温度
    for (int i = 0; i < 365; i++) {
        int temp = -10 + rand() % 51;
        temp_hist[temp + 10]++;  // 偏移10使下标从0开始
    }

    printf("温度分布:\n");
    for (int i = 0; i < 51; i++) {
        if (temp_hist[i] > 0) {
            printf("%3d°C: ", i - 10);
            for (int j = 0; j < temp_hist[i]; j++) {
                printf("*");
            }
            printf(" (%d天)\n", temp_hist[i]);
        }
    }

    return 0;
}
```

## 5. ASCII码频率统计

```c
void ascii_frequency(const char *text) {
    int freq[128] = {0};

    for (int i = 0; text[i] != '\0'; i++) {
        freq[(unsigned char)text[i]]++;
    }

    printf("字符频率:\n");
    for (int i = 32; i < 127; i++) {  // 可打印字符
        if (freq[i] > 0) {
            printf("'%c' (%3d): %d\n", i, i, freq[i]);
        }
    }
}
```

---

## 重点总结

- 直方图用数组下标做**类别**，元素值做**计数**
- 统计前必须初始化数组为全零 `= {0}`
- 注意数组下标的偏移（如温度-10到40需要偏移）
- 字符频率用 `char - 'a'` 映射到数组下标
- 绘制文本直方图时注意**最大值**的确定
