# 内存管理面试题

## 一、内存布局

```c
#include <stdio.h>
#include <stdlib.h>

int global_var = 10;            // 数据段（已初始化）
static int static_var = 20;     // 数据段（已初始化）
int uninit_var;                 // BSS段（未初始化）

void memory_layout_demo() {
    int local_var = 30;          // 栈
    static int local_static = 40; // 数据段
    int *heap_var = (int *)malloc(sizeof(int)); // 堆

    printf("代码段 (函数地址):  %p\n", (void *)memory_layout_demo);
    printf("数据段 (global):   %p\n", (void *)&global_var);
    printf("数据段 (static):   %p\n", (void *)&static_var);
    printf("BSS段 (uninit):    %p\n", (void *)&uninit_var);
    printf("堆区 (heap):       %p\n", (void *)heap_var);
    printf("栈区 (local):      %p\n", (void *)&local_var);

    free(heap_var);
}

/*
典型内存布局（从低到高）:
┌─────────────────┐  低地址
│    代码段 (.text)  │  只读，存放可执行指令
├─────────────────┤
│   数据段 (.data)  │  已初始化的全局/静态变量
├─────────────────┤
│   BSS段 (.bss)   │  未初始化的全局/静态变量
├─────────────────┤
│      堆 (heap)    │  向高地址增长
│        ↓          │
│                   │
│        ↑          │
│      栈 (stack)   │  向低地址增长
├─────────────────┤
│  内核空间          │  高地址
└─────────────────┘
*/
```

## 二、malloc/free详解

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void malloc_demo() {
    // malloc：分配未初始化的内存
    int *p1 = (int *)malloc(10 * sizeof(int));
    if (p1 == NULL) {
        printf("malloc失败\n");
        return;
    }
    // malloc不初始化，内容未定义
    printf("malloc后: p1[0] = %d (未定义)\n", p1[0]);

    // calloc：分配并清零
    int *p2 = (int *)calloc(10, sizeof(int));
    printf("calloc后: p2[0] = %d (已清零)\n", p2[0]);

    // realloc：调整大小
    p1 = (int *)realloc(p1, 20 * sizeof(int));
    if (p1 == NULL) {
        printf("realloc失败\n");
        free(p2);
        return;
    }

    free(p1);
    free(p2);
    // free后置空，防止悬空指针
    p1 = NULL;
    p2 = NULL;
}
```

## 三、经典面试题：内存泄漏

```c
// 错误示例1：丢失指针
void leak_example1() {
    int *p = (int *)malloc(sizeof(int));
    p = NULL;  // 原内存泄漏！无法free
}

// 错误示例2：提前返回
int leak_example2() {
    char *buf = (char *)malloc(1024);
    if (buf == NULL) return -1;

    // 某些操作可能导致提前返回
    int error = 1;
    if (error) {
        return -1;  // 忘记free(buf)，内存泄漏
    }

    free(buf);
    return 0;
}

// 正确写法：使用goto cleanup
int correct_example() {
    int ret = -1;
    char *buf1 = NULL;
    char *buf2 = NULL;

    buf1 = (char *)malloc(1024);
    if (!buf1) goto cleanup;

    buf2 = (char *)malloc(1024);
    if (!buf2) goto cleanup;

    // 正常处理
    ret = 0;

cleanup:
    free(buf1);
    free(buf2);
    return ret;
}
```

## 四、经典面试题：缓冲区溢出

```c
#include <stdio.h>
#include <string.h>

void buffer_overflow_demo() {
    char buf[10];
    // strcpy不检查长度，可能导致溢出
    // strcpy(buf, "This string is way too long!");  // 危险！

    // 安全做法：使用strncpy
    strncpy(buf, "Hello World!", sizeof(buf) - 1);
    buf[sizeof(buf) - 1] = '\0';
    printf("安全拷贝: %s\n", buf);

    // 或使用snprintf
    snprintf(buf, sizeof(buf), "Format: %d", 12345);
    printf("格式化: %s\n", buf);
}
```

## 五、内存对齐

```c
#include <stdio.h>
#include <stddef.h>

struct Example1 {
    char a;     // 1字节
    int b;      // 4字节
    char c;     // 1字节
};  // 大小：12字节（不是6字节）

struct Example2 {
    int b;      // 4字节
    char a;     // 1字节
    char c;     // 1字节
};  // 大小：8字节（排列顺序影响大小）

#pragma pack(1)  // 关闭对齐
struct Packed {
    char a;
    int b;
    char c;
};  // 大小：6字节
#pragma pack()   // 恢复默认对齐

void alignment_demo() {
    printf("Example1 大小: %zu\n", sizeof(struct Example1));  // 12
    printf("Example2 大小: %zu\n", sizeof(struct Example2));  // 8
    printf("Packed 大小: %zu\n", sizeof(struct Packed));      // 6
}
```

## 六、double free与use after free

```c
void double_free_demo() {
    int *p = (int *)malloc(sizeof(int));
    *p = 42;
    free(p);
    // free(p);  // 错误！double free，未定义行为

    // free后置空是好习惯
    p = NULL;
    // free(p);  // free(NULL)是安全的，但没有意义
}

void use_after_free_demo() {
    int *p = (int *)malloc(sizeof(int));
    *p = 42;
    free(p);
    // printf("%d\n", *p);  // 错误！use after free
    p = NULL;  // 防止误用
}
```

## 七、简易内存池

```c
#include <stdio.h>
#include <stddef.h>

#define POOL_SIZE 1024

static char pool[POOL_SIZE];
static size_t pool_offset = 0;

void *pool_alloc(size_t size) {
    // 对齐到8字节
    size = (size + 7) & ~7;
    if (pool_offset + size > POOL_SIZE) return NULL;
    void *ptr = pool + pool_offset;
    pool_offset += size;
    return ptr;
}

void pool_reset() {
    pool_offset = 0;
}

void memory_pool_demo() {
    int *a = (int *)pool_alloc(sizeof(int) * 10);
    char *b = (char *)pool_alloc(256);
    printf("池使用: %zu / %d 字节\n", pool_offset, POOL_SIZE);
    pool_reset();  // 一次性释放所有
    printf("重置后: %zu 字节\n", pool_offset);
}
```

## 八、面试常见问答

```text
Q: malloc和calloc的区别？
A: malloc只分配不初始化，calloc会将内存清零。
   calloc(n, size)接收两个参数，更容易检测溢出。

Q: free之后需要置NULL吗？
A: 不是必须的，但建议置NULL以防止悬空指针。

Q: 如何检测内存泄漏？
A: 使用valgrind、AddressSanitizer、mtrace等工具。

Q: 堆和栈的区别？
A: 栈自动管理、速度快、空间小；堆手动管理、灵活、空间大。

Q: 什么是内存碎片？
A: 频繁分配释放导致堆中出现大量不连续的小空闲块。
*/
```
