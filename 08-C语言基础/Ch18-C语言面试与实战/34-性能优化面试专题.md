# 性能优化面试专题

## 一、性能分析方法

```c
#include <stdio.h>
#include <time.h>
#include <sys/time.h>

/*
 * 性能分析工具：
 * - perf: Linux性能分析利器
 * - gprof: GNU性能分析工具
 * - valgrind/cachegrind: 缓存分析
 * - strace/ltrace: 系统调用/库调用跟踪
 * - vtune: Intel性能分析器
 */

// 简易计时器
typedef struct {
    struct timeval start, end;
} Timer;

void timer_start(Timer *t) {
    gettimeofday(&t->start, NULL);
}

double timer_stop(Timer *t) {
    gettimeofday(&t->end, NULL);
    return (t->end.tv_sec - t->start.tv_sec) * 1000.0 +
           (t->end.tv_usec - t->start.tv_usec) / 1000.0;
}

// RDTSC高精度计时（x86）
#ifdef __x86_64__
static inline unsigned long long rdtsc() {
    unsigned hi, lo;
    __asm__ volatile ("rdtsc" : "=a"(lo), "=d"(hi));
    return ((unsigned long long)lo) | ((unsigned long long)hi << 32);
}
#endif
```

## 二、CPU优化

```c
#include <stdint.h>

// 1. 减少分支预测失败
// 差
int count_positive_bad(int *arr, int n) {
    int count = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) count++;  // 分支可能预测失败
    }
    return count;
}

// 好：无分支版本
int count_positive_good(int *arr, int n) {
    int count = 0;
    for (int i = 0; i < n; i++) {
        count += (arr[i] > 0);  // 无分支
    }
    return count;
}

// 2. 循环展开
void sum_array_unrolled(int *arr, int n) {
    int sum = 0;
    int i;
    // 4路展开
    for (i = 0; i + 3 < n; i += 4) {
        sum += arr[i] + arr[i+1] + arr[i+2] + arr[i+3];
    }
    for (; i < n; i++) sum += arr[i];
}

// 3. SIMD友好（示例概念）
void add_arrays_simd(float *a, float *b, float *c, int n) {
    // 编译器自动向量化（需-O3 -march=native）
    #pragma GCC ivdep
    for (int i = 0; i < n; i++) {
        c[i] = a[i] + b[i];
    }
}

// 4. 查表代替计算
static const uint8_t popcount_table[256] = {
    0,1,1,2,1,2,2,3,1,2,2,3,2,3,3,4,
    // ... 填充256个值
};

int popcount_lookup(uint8_t x) {
    return popcount_table[x];
}
```

## 四、IO优化

```c
#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/mman.h>

// 1. 缓冲IO vs 无缓冲IO
void io_optimization() {
    // 缓冲IO（标准库）
    FILE *fp = fopen("test.txt", "w");
    setvbuf(fp, NULL, _IOFBF, 65536);  // 设置64K缓冲
    fprintf(fp, "buffered write\n");
    fclose(fp);

    // 无缓冲IO（系统调用）
    int fd = open("test.txt", O_WRONLY | O_DIRECT);  // 直接IO
    // 注意：直接IO需要对齐的缓冲区
    close(fd);
}

// 2. 批量操作
void batch_operations() {
    // 差：逐个写入
    // for (int i = 0; i < 10000; i++)
    //     write(fd, &data[i], sizeof(int));

    // 好：批量写入
    // write(fd, data, 10000 * sizeof(int));
}

// 3. mmap文件映射
void mmap_file_access(const char *filename) {
    int fd = open(filename, O_RDONLY);
    struct stat st;
    fstat(fd, &st);

    char *mapped = mmap(NULL, st.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
    if (mapped != MAP_FAILED) {
        // 直接访问，无需read系统调用
        for (off_t i = 0; i < st.st_size; i++) {
            // 处理 mapped[i]
        }
        munmap(mapped, st.st_size);
    }
    close(fd);
}
```

## 六、多线程优化

```c
#include <pthread.h>

// 1. 减少锁竞争
// 细粒度锁
#define NUM_BUCKETS 16
pthread_mutex_t locks[NUM_BUCKETS];

// 每个桶一个锁，减少竞争
void fine_grained_lock(int bucket_id) {
    pthread_mutex_lock(&locks[bucket_id % NUM_BUCKETS]);
    // 操作数据
    pthread_mutex_unlock(&locks[bucket_id % NUM_BUCKETS]);
}

// 2. 无锁数据结构（CAS操作）
int atomic_increment(int *counter) {
    int old_val;
    do {
        old_val = *counter;
    } while (!__sync_bool_compare_and_swap(counter, old_val, old_val + 1));
    return old_val + 1;
}

// 3. 避免false sharing
// 差：不同线程的数据在同一缓存行
struct BadShared {
    int counter1;  // 线程1使用
    int counter2;  // 线程2使用（同一缓存行！）
};

// 好：填充到不同缓存行
struct GoodShared {
    int counter1;
    char pad1[60];  // 填充到64字节
    int counter2;
    char pad2[60];
};

// 或用对齐属性
struct AlignedShared {
    int counter1 __attribute__((aligned(64)));
    int counter2 __attribute__((aligned(64)));
};
```

```c
int main(void) {
    Timer t;

    // 测试循环展开效果
    int arr[1000000];
    for (int i = 0; i < 1000000; i++) arr[i] = i;

    timer_start(&t);
    int sum = 0;
    for (int i = 0; i < 1000000; i++) sum += arr[i];
    printf("普通循环: %.3f ms, sum=%d\n", timer_stop(&t), sum);

    timer_start(&t);
    sum = 0;
    sum_array_unrolled(arr, 1000000);
    printf("展开循环: %.3f ms\n", timer_stop(&t));

    return 0;
}
```
