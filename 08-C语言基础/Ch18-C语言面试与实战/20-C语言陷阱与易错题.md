# C语言陷阱与易错题

## 一、运算符陷阱

```c
#include <stdio.h>

void trap1_assignment_in_condition() {
    // 陷阱1：= 和 == 混淆
    int x = 0;
    // if (x = 5)  // 赋值！x变为5，条件为真
    if (x == 5) printf("正确: x等于5\n");

    // 好习惯：常量放左边
    // if (5 == x)  // 写反会编译报错
}

void trap2_operator_precedence() {
    // 陷阱2：运算符优先级
    int a = 1, b = 2, c = 3;

    // *p++ 等价于 *(p++)，不是 (*p)++
    int arr[] = {10, 20, 30};
    int *p = arr;
    printf("*p++ = %d\n", *p++);  // 10，然后p后移
    printf("*p = %d\n", *p);      // 20

    // 位运算优先级低于比较运算符
    // if (flags & 0x01 == 0)  // 错误！==优先级高于&
    // 应当写成: if ((flags & 0x01) == 0)
}

void trap3_sizeof_pitfalls() {
    // 陷阱3：sizeof对指针和数组的区别
    int arr[10];
    int *p = arr;
    printf("sizeof(arr) = %zu\n", sizeof(arr));  // 40
    printf("sizeof(p)   = %zu\n", sizeof(p));    // 8 (指针大小)

    // 在函数参数中数组退化为指针
    // void func(int arr[10]) { sizeof(arr) == sizeof(int*) }
}
```

## 二、字符串陷阱

```c
#include <string.h>

void trap4_string_literal() {
    // 陷阱4：字符串字面量不可修改
    char *s1 = "hello";
    // s1[0] = 'H';  // 未定义行为！崩溃

    char s2[] = "hello";
    s2[0] = 'H';  // OK，数组副本可以修改
    printf("s2 = %s\n", s2);

    // sizeof vs strlen
    char s3[] = "hello";
    printf("sizeof(s3) = %zu\n", sizeof(s3));  // 6 (含'\0')
    printf("strlen(s3) = %zu\n", strlen(s3));  // 5
}

void trap5_strcpy_overflow() {
    // 陷阱5：strcpy不检查缓冲区大小
    char buf[5];
    // strcpy(buf, "Hello World!");  // 缓冲区溢出！

    // 安全写法
    strncpy(buf, "Hello World!", sizeof(buf) - 1);
    buf[sizeof(buf) - 1] = '\0';
    printf("安全拷贝: %s\n", buf);
}
```

## 三、指针陷阱

```c
#include <stdlib.h>

void trap6_uninitialized_pointer() {
    // 陷阱6：未初始化指针
    int *p;       // 野指针，指向随机地址
    // *p = 42;   // 崩溃！

    // 正确做法
    int *q = NULL;

void trap7_dangling_pointer() {
    // 陷阱7：悬空指针
    int *p = (int *)malloc(sizeof(int));
    *p = 42;

    // printf("%d\n", *p);  // 未定义行为

    p = NULL;  // 释放后置空
}

    // }

    // 正确：返回动态分配的内存或静态变量
```

## 四、循环陷阱

```c
void trap9_off_by_one() {
    // 陷阱9：差一错误
    int arr[10];
    // for (int i = 0; i <= 10; i++)  // 越界！应该< 10
    for (int i = 0; i < 10; i++) arr[i] = i;
}

void trap10_signed_unsigned() {
    // 陷阱10：有符号/无符号比较
    unsigned int u = 1;
    int i = -1;
    // if (i < u)  // -1被转为无符号数！变成很大的正数
    //     printf("这行不会执行\n");
    // else
    //     printf("-1 > 1? 因为无符号转换\n");

    // 正确做法
    if (i < 0 || (unsigned int)i < u)
        printf("正确比较\n");
}
```

## 五、宏陷阱

```c
#define SQUARE(x) x*x
#define SAFE_SQUARE(x) ((x)*(x))

void trap11_macro_side_effect() {
    // 陷阱11：宏参数副作用
    int a = 5;
    printf("SQUARE(a++) = %d\n", SQUARE(a++));         // 25，a变为7
    a = 5;
    printf("SAFE_SQUARE(a++) = %d\n", SAFE_SQUARE(a++)); // 展开为((a++)*(a++))，a自增两次

    // 正确：避免在宏中使用有副作用的表达式
    a = 5;
    int sq = a * a;
    a++;
}
```

## 六、数组陷阱

```c
void trap12_array_decay() {
    // 陷阱12：数组作为参数退化为指针
    int arr[5] = {1, 2, 3, 4, 5};

    // 在函数中无法得知数组大小
    // 解决方案：额外传递大小参数
    printf("arr大小: %zu\n", sizeof(arr));  // 20
}

void trap13_no_bounds_check() {
    // 陷阱13：C不检查数组越界
    int arr[5] = {0};
    arr[5] = 100;     // 越界！不报错但结果不可预测
    arr[-1] = 200;    // 越界！
    printf("越界写入可能导致段错误或数据损坏\n");
}
```

## 七、内存陷阱

```c
void trap14_memory_leak() {
    // 陷阱14：内存泄漏
    int *p = (int *)malloc(100 * sizeof(int));
    p = (int *)malloc(200 * sizeof(int));  // 第一次分配泄漏
    free(p);

    // 正确：先free再重新分配
    p = (int *)malloc(100 * sizeof(int));
    free(p);
    p = (int *)malloc(200 * sizeof(int));
    free(p);
}

void trap15_double_free() {
    // 陷阱15：重复释放
    int *p = (int *)malloc(sizeof(int));
    free(p);
    // free(p);  // double free！未定义行为
    p = NULL;   // 好习惯
}
```

## 八、整数溢出

```c
#include <limits.h>

void trap16_integer_overflow() {
    // 陷阱16：整数溢出
    int a = INT_MAX;
    printf("INT_MAX = %d\n", a);
    a++;  // 溢出！未定义行为
    printf("INT_MAX+1 = %d\n", a);  // INT_MIN

    // 安全：溢出检查
    int b = INT_MAX;
    if (b > INT_MAX - 1) {
        printf("加法会溢出！\n");
    }
}

void trap17_strlen_returns_size_t() {
    // 陷阱17：strlen返回size_t（无符号）
    char *s = "";
    // if (strlen(s) - 1 < 0)  // 永远为假！size_t无符号
    //     printf("不会执行\n");

    if (strlen(s) == 0) {
        printf("字符串为空\n");
    }
}
```

## 九、类型转换陷阱

```c
void trap18_float_precision() {
    // 陷阱18：浮点精度
    float f = 0.1f;
    double d = 0.1;
    // f == d 可能为假！精度不同

    // 正确：使用容差比较
    if (fabs(f - d) < 1e-6) {
        printf("近似相等\n");
    }
}

void trap19_char_sign() {
    // 陷阱19：char可能是有符号或无符号（取决于编译器）
    char c = 200;  // 如果是有符号char，这是负数！
    printf("c = %d\n", (int)c);

    // 明确使用 signed char 或 unsigned char
    unsigned char uc = 200;
    printf("uc = %d\n", (int)uc);  // 始终是200
}
```

## 十、综合易错题

```c
void trap20_comma_operator() {
    // 陷阱20：逗号运算符
    int a = (1, 2, 3);  // a = 3
    printf("逗号表达式: %d\n", a);

    int b = 0;
    b = 1, 2, 3;  // b = 1（赋值优先级高于逗号）
    printf("b = %d\n", b);

    b = (1, 2, 3);  // b = 3
    printf("b = %d\n", b);
}

int main(void) {
    trap4_string_literal();
    trap10_signed_unsigned();
    trap16_integer_overflow();
    trap20_comma_operator();
    printf("所有陷阱题执行完毕\n");
    return 0;
}
```
