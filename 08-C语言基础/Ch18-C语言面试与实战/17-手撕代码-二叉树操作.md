# 手撕代码 - 二叉树操作

## 基础结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left, *right;
} TreeNode;

TreeNode *new_node(int val) {
    TreeNode *n = (TreeNode *)malloc(sizeof(TreeNode));
    n->val = val;
    n->left = n->right = NULL;
    return n;
}
```

## 一、四种遍历（递归+迭代）

```c
// 前序递归
void preorder(TreeNode *root) {
    if (!root) return;
    printf("%d ", root->val);
    preorder(root->left);
    preorder(root->right);
}

// 前序迭代
void preorder_iter(TreeNode *root) {
    TreeNode *stack[100]; int top = -1;
    if (root) stack[++top] = root;
    while (top >= 0) {
        TreeNode *node = stack[top--];
        printf("%d ", node->val);
        if (node->right) stack[++top] = node->right;
        if (node->left) stack[++top] = node->left;
    }
}

// 中序递归
void inorder(TreeNode *root) {
    if (!root) return;
    inorder(root->left);
    printf("%d ", root->val);
    inorder(root->right);
}

// 中序迭代
void inorder_iter(TreeNode *root) {
    TreeNode *stack[100]; int top = -1;
    TreeNode *curr = root;
    while (curr || top >= 0) {
        while (curr) { stack[++top] = curr; curr = curr->left; }
        curr = stack[top--];
        printf("%d ", curr->val);
        curr = curr->right;
    }
}

// 后序迭代（双栈法）
void postorder_iter(TreeNode *root) {
    if (!root) return;
    TreeNode *s1[100], *s2[100];
    int top1 = -1, top2 = -1;
    s1[++top1] = root;
    while (top1 >= 0) {
        TreeNode *node = s1[top1--];
        s2[++top2] = node;
        if (node->left) s1[++top1] = node->left;
        if (node->right) s1[++top1] = node->right;
    }
    while (top2 >= 0) printf("%d ", s2[top2--]->val);
}

// 层序遍历
void level_order(TreeNode *root) {
    if (!root) return;
    TreeNode *q[200]; int front = 0, rear = 0;
    q[rear++] = root;
    while (front < rear) {
        int size = rear - front;
        for (int i = 0; i < size; i++) {
            TreeNode *node = q[front++];
            printf("%d ", node->val);
            if (node->left) q[rear++] = node->left;
            if (node->right) q[rear++] = node->right;
        }
        printf("\n");
    }
}
```

## 二、树的属性

```c
int max_depth(TreeNode *root) {
    if (!root) return 0;
    int l = max_depth(root->left);
    int r = max_depth(root->right);
    return (l > r ? l : r) + 1;
}

int min_depth(TreeNode *root) {
    if (!root) return 0;
    if (!root->left) return min_depth(root->right) + 1;
    if (!root->right) return min_depth(root->left) + 1;
    int l = min_depth(root->left);
    int r = min_depth(root->right);
    return (l < r ? l : r) + 1;
}

int count_nodes(TreeNode *root) {
    return root ? 1 + count_nodes(root->left) + count_nodes(root->right) : 0;
}

int tree_sum(TreeNode *root) {
    return root ? root->val + tree_sum(root->left) + tree_sum(root->right) : 0;
}

bool is_balanced(TreeNode *root) {
    if (!root) return true;
    int l = max_depth(root->left);
    int r = max_depth(root->right);
    return abs(l - r) <= 1 && is_balanced(root->left) && is_balanced(root->right);
}

bool is_symmetric(TreeNode *root) {
    bool check(TreeNode *l, TreeNode *r) {
        if (!l && !r) return true;
        if (!l || !r) return false;
        return l->val == r->val && check(l->left, r->right) && check(l->right, r->left);
    }
    return !root || check(root->left, root->right);
}
```

## 三、BST操作

```c
TreeNode *bst_insert(TreeNode *root, int val) {
    if (!root) return new_node(val);
    if (val < root->val) root->left = bst_insert(root->left, val);
    else if (val > root->val) root->right = bst_insert(root->right, val);
    return root;
}

TreeNode *bst_search(TreeNode *root, int val) {
    while (root) {
        if (val < root->val) root = root->left;
        else if (val > root->val) root = root->right;
        else return root;
    }
    return NULL;
}

TreeNode *bst_min(TreeNode *root) {
    while (root->left) root = root->left;
    return root;
}

TreeNode *bst_delete(TreeNode *root, int val) {
    if (!root) return NULL;
    if (val < root->val) root->left = bst_delete(root->left, val);
    else if (val > root->val) root->right = bst_delete(root->right, val);
    else {
        if (!root->left) { TreeNode *r = root->right; free(root); return r; }
        if (!root->right) { TreeNode *l = root->left; free(root); return l; }
        TreeNode *min = bst_min(root->right);
        root->val = min->val;
        root->right = bst_delete(root->right, min->val);
    }
    return root;
}

bool is_valid_bst(TreeNode *root) {
    bool helper(TreeNode *node, long lo, long hi) {
        if (!node) return true;
        if (node->val <= lo || node->val >= hi) return false;
        return helper(node->left, lo, node->val) &&
               helper(node->right, node->val, hi);
    }
    return helper(root, LONG_MIN, LONG_MAX);
}
```

## 四、LCA最近公共祖先

```c
TreeNode *lca(TreeNode *root, TreeNode *p, TreeNode *q) {
    if (!root || root == p || root == q) return root;
    TreeNode *left = lca(root->left, p, q);
    TreeNode *right = lca(root->right, p, q);
    if (left && right) return root;
    return left ? left : right;
}
```

## 五、路径问题

```c
bool has_path_sum(TreeNode *root, int target) {
    if (!root) return false;
    if (!root->left && !root->right) return root->val == target;
    return has_path_sum(root->left, target - root->val) ||
           has_path_sum(root->right, target - root->val);
}

// 所有路径
void all_paths(TreeNode *root, int *path, int len) {
    if (!root) return;
    path[len++] = root->val;
    if (!root->left && !root->right) {
        for (int i = 0; i < len; i++) printf("%d%s", path[i], i < len-1 ? "->" : "\n");
        return;
    }
    all_paths(root->left, path, len);
    all_paths(root->right, path, len);
}
```

## 六、序列化与反序列化

```c
void serialize(TreeNode *root, char *buf, int *pos) {
    if (!root) { *pos += sprintf(buf + *pos, "#,"); return; }
    *pos += sprintf(buf + *pos, "%d,", root->val);
    serialize(root->left, buf, pos);
    serialize(root->right, buf, pos);
}

TreeNode *deserialize_helper(const char **data) {
    if (**data == '#') { *data += 2; return NULL; }
    int val = 0, neg = 0;
    if (**data == '-') { neg = 1; (*data)++; }
    while (**data != ',') { val = val * 10 + (**data - '0'); (*data)++; }
    (*data)++;
    TreeNode *node = new_node(neg ? -val : val);
    node->left = deserialize_helper(data);
    node->right = deserialize_helper(data);
    return node;
}

TreeNode *deserialize(const char *data) { return deserialize_helper(&data); }
```

## 七、二叉树翻转

```c
TreeNode *invert_tree(TreeNode *root) {
    if (!root) return NULL;
    TreeNode *tmp = root->left;
    root->left = invert_tree(root->right);
    root->right = invert_tree(tmp);
    return root;
}
```
