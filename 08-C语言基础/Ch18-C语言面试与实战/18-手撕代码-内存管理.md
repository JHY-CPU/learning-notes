# 手撕代码 - 内存管理

## 一、简易malloc/free实现

```c
#include <stdio.h>
#include <stddef.h>
#include <string.h>
#include <unistd.h>

/*
 * 使用sbrk系统调用手动实现堆内存管理
 * 核心思想：维护一个空闲块链表，分配时查找合适块
 */

typedef struct BlockHeader {
    size_t size;               // 数据区大小
    int free;                  // 是否空闲
    struct BlockHeader *next;  // 下一个块
} BlockHeader;

static BlockHeader *head = NULL;  // 链表头

// 在空闲链表中查找合适大小的块
BlockHeader *find_free_block(BlockHeader **last, size_t size) {
    BlockHeader *curr = head;
    while (curr) {
        if (curr->free && curr->size >= size) {
            return curr;
        }
        *last = curr;
        curr = curr->next;
    }
    return NULL;
}

// 扩展堆空间
BlockHeader *request_space(BlockHeader *last, size_t size) {
    BlockHeader *block = (BlockHeader *)sbrk(
        sizeof(BlockHeader) + size);
    if (block == (void *)-1) return NULL;

    block->size = size;
    block->free = 0;
    block->next = NULL;

    if (last) last->next = block;
    return block;
}

// 简易malloc
void *my_malloc(size_t size) {
    if (size == 0) return NULL;

    BlockHeader *block;
    if (!head) {
        // 第一次调用
        block = request_space(NULL, size);
        if (!block) return NULL;
        head = block;
    } else {
        BlockHeader *last = head;
        block = find_free_block(&last, size);
        if (block) {
            // 找到合适的空闲块
            block->free = 0;
        } else {
            // 需要扩展堆
            block = request_space(last, size);
            if (!block) return NULL;
        }
    }
    return (void *)(block + 1);  // 返回数据区指针
}

// 简易free
void my_free(void *ptr) {
    if (!ptr) return;
    BlockHeader *block = (BlockHeader *)ptr - 1;
    block->free = 1;
    // 可选：合并相邻空闲块（碎片整理）
    // 可选：如果最后块空闲，用sbrk回收
}

// 获取块信息
void *get_block_ptr(void *ptr) {
    return (BlockHeader *)ptr - 1;
}
```

## 二、合并空闲块（碎片整理）

```c
void merge_free_blocks() {
    BlockHeader *curr = head;
    while (curr && curr->next) {
        if (curr->free && curr->next->free) {
            curr->size += sizeof(BlockHeader) + curr->next->size;
            curr->next = curr->next->next;
            // 不要移动curr，可能连续多个空闲块
        } else {
            curr = curr->next;
        }
    }
}
```

## 三、带碎片整理的free

```c
void my_free_merge(void *ptr) {
    if (!ptr) return;
    BlockHeader *block = (BlockHeader *)ptr - 1;
    block->free = 1;
    merge_free_blocks();
}
```

## 四、内存池实现

```c
/*
 * 内存池：一次性分配大块内存，内部管理小块
 * 优点：分配快、无碎片、释放方便
 */

#define POOL_BLOCK_SIZE 64
#define POOL_BLOCK_COUNT 1024

typedef struct PoolBlock {
    struct PoolBlock *next;
} PoolBlock;

typedef struct {
    char *memory;          // 大块内存
    PoolBlock *free_list;  // 空闲链表
    size_t block_size;
    size_t block_count;
    size_t allocated;
} MemoryPool;

MemoryPool *pool_create(size_t block_size, size_t block_count) {
    MemoryPool *pool = (MemoryPool *)malloc(sizeof(MemoryPool));
    if (!pool) return NULL;

    pool->block_size = (block_size < sizeof(PoolBlock))
                       ? sizeof(PoolBlock) : block_size;
    pool->block_count = block_count;
    pool->allocated = 0;

    // 分配大块内存
    pool->memory = (char *)malloc(pool->block_size * block_count);
    if (!pool->memory) { free(pool); return NULL; }

    // 初始化空闲链表
    pool->free_list = NULL;
    for (size_t i = 0; i < block_count; i++) {
        PoolBlock *block = (PoolBlock *)(pool->memory + i * pool->block_size);
        block->next = pool->free_list;
        pool->free_list = block;
    }
    return pool;
}

void *pool_alloc(MemoryPool *pool) {
    if (!pool || !pool->free_list) return NULL;
    PoolBlock *block = pool->free_list;
    pool->free_list = block->next;
    pool->allocated++;
    return (void *)block;
}

void pool_free(MemoryPool *pool, void *ptr) {
    if (!pool || !ptr) return;
    PoolBlock *block = (PoolBlock *)ptr;
    block->next = pool->free_list;
    pool->free_list = block;
    pool->allocated--;
}

void pool_reset(MemoryPool *pool) {
    pool->free_list = NULL;
    for (size_t i = 0; i < pool->block_count; i++) {
        PoolBlock *block = (PoolBlock *)(pool->memory + i * pool->block_size);
        block->next = pool->free_list;
        pool->free_list = block;
    }
    pool->allocated = 0;
}

void pool_destroy(MemoryPool *pool) {
    if (pool) {
        free(pool->memory);
        free(pool);
    }
}

size_t pool_available(MemoryPool *pool) {
    return pool->block_count - pool->allocated;
}
```

## 五、对齐分配器

```c
// 分配指定对齐的内存
void *aligned_alloc_custom(size_t alignment, size_t size) {
    if (alignment & (alignment - 1)) return NULL;  // 必须是2的幂
    size_t total = size + alignment + sizeof(void *);
    void *raw = malloc(total);
    if (!raw) return NULL;

    // 计算对齐地址
    uintptr_t addr = (uintptr_t)raw + sizeof(void *);
    uintptr_t aligned = (addr + alignment - 1) & ~(alignment - 1);

    // 存储原始指针（在对齐地址之前）
    void **store = (void **)(aligned - sizeof(void *));
    *store = raw;
    return (void *)aligned;
}

void aligned_free_custom(void *ptr) {
    if (!ptr) return;
    void *raw = *((void **)((uintptr_t)ptr - sizeof(void *)));
    free(raw);
}
```

## 六、引用计数

```c
typedef struct {
    int ref_count;
    size_t size;
    char data[];  // 柔性数组
} RefCounted;

void *rc_alloc(size_t size) {
    RefCounted *rc = (RefCounted *)malloc(sizeof(RefCounted) + size);
    if (!rc) return NULL;
    rc->ref_count = 1;
    rc->size = size;
    return rc->data;
}

void rc_retain(void *ptr) {
    RefCounted *rc = (RefCounted *)((char *)ptr - sizeof(RefCounted));
    rc->ref_count++;
}

void rc_release(void *ptr) {
    if (!ptr) return;
    RefCounted *rc = (RefCounted *)((char *)ptr - sizeof(RefCounted));
    rc->ref_count--;
    if (rc->ref_count == 0) {
        free(rc);
    }
}

int rc_count(void *ptr) {
    RefCounted *rc = (RefCounted *)((char *)ptr - sizeof(RefCounted));
    return rc->ref_count;
}
```

## 测试

```c
int main(void) {
    // 内存池测试
    MemoryPool *pool = pool_create(64, 100);
    printf("池可用: %zu / %zu\n", pool_available(pool), pool->block_count);

    void *blocks[10];
    for (int i = 0; i < 10; i++) {
        blocks[i] = pool_alloc(pool);
    }
    printf("分配10块后: 可用 %zu\n", pool_available(pool));

    for (int i = 0; i < 5; i++) {
        pool_free(pool, blocks[i]);
    }
    printf("释放5块后: 可用 %zu\n", pool_available(pool));

    pool_reset(pool);
    printf("重置后: 可用 %zu\n", pool_available(pool));
    pool_destroy(pool);

    // 引用计数测试
    int *data = (int *)rc_alloc(sizeof(int) * 10);
    *data = 42;
    printf("引用计数: %d\n", rc_count(data));
    rc_retain(data);
    printf("引用计数: %d\n", rc_count(data));
    rc_release(data);
    printf("引用计数: %d\n", rc_count(data));
    rc_release(data);  // 释放

    return 0;
}
```
