# 位运算面试题

## 一、基础位操作

```c
#include <stdio.h>
#include <stdint.h>

void basic_bit_ops() {
    unsigned int a = 0b1010;  // 10
    unsigned int b = 0b1100;  // 12

    printf("a & b  = %u (AND)\n", a & b);    // 8  (1000)
    printf("a | b  = %u (OR)\n",  a | b);    // 14 (1110)
    printf("a ^ b  = %u (XOR)\n", a ^ b);    // 6  (0110)
    printf("~a     = %u (NOT)\n", ~a);        // 很大的数
    printf("a << 2 = %u (左移)\n", a << 2);   // 40
    printf("a >> 1 = %u (右移)\n", a >> 1);   // 5
}
```

## 二、常用技巧

```c
// 判断奇偶
int is_odd(int n) { return n & 1; }
int is_even(int n) { return !(n & 1); }

// 交换两个数（不用临时变量）
void xor_swap(int *a, int *b) {
    if (a != b) {  // 相同地址会出错
        *a ^= *b;
        *b ^= *a;
        *a ^= *b;
    }
}

// 取绝对值（不用if）
int abs_bit(int n) {
    int mask = n >> 31;  // 全0或全1
    return (n ^ mask) - mask;
}

// 判断是否为2的幂
int is_power_of_2(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}

// 计算二进制中1的个数（Hamming Weight）
int count_bits(int n) {
    int count = 0;
    while (n) {
        n &= (n - 1);  // 清除最低位的1
        count++;
    }
    return count;
}

// Brian Kernighan算法变体
int count_bits_v2(unsigned int n) {
    n = n - ((n >> 1) & 0x55555555);
    n = (n & 0x33333333) + ((n >> 2) & 0x33333333);
    n = (n + (n >> 4)) & 0x0F0F0F0F;
    return (n * 0x01010101) >> 24;
}

void tricks_demo() {
    printf("\n=== 位运算技巧 ===\n");
    printf("5是奇数: %d\n", is_odd(5));
    printf("16是2的幂: %d\n", is_power_of_2(16));
    printf("15是2的幂: %d\n", is_power_of_2(15));
    printf("abs(-42) = %d\n", abs_bit(-42));
    printf("count_bits(0b10110110) = %d\n", count_bits(0b10110110));
}
```

## 三、位掩码操作

```c
#define READ    0x01  // 0001
#define WRITE   0x02  // 0010
#define EXECUTE 0x04  // 0100
#define ADMIN   0x08  // 1000

void bitmask_demo() {
    unsigned int perm = READ | WRITE;  // 0011

    // 设置权限
    perm |= EXECUTE;    // 添加执行权限
    printf("权限: %d (rwx)\n", perm);

    // 检查权限
    if (perm & READ) printf("有读权限\n");
    if (perm & WRITE) printf("有写权限\n");
    if (perm & EXECUTE) printf("有执行权限\n");
    if (!(perm & ADMIN)) printf("无管理员权限\n");

    // 清除权限
    perm &= ~WRITE;     // 移除写权限
    printf("移除写后: %d\n", perm);

    // 切换权限
    perm ^= EXECUTE;    // 切换执行权限
    printf("切换执行后: %d\n", perm);
}
```

## 四、状态压缩

```c
// 用整数位表示集合
// 例：5个任务的状态用5位表示

#define TASK_COUNT 5
#define ALL_DONE ((1 << TASK_COUNT) - 1)  // 0b11111

void state_compression() {
    unsigned int state = 0;

    // 完成任务i
    void complete_task(int i) {
        state |= (1 << i);
    }

    // 检查任务i是否完成
    int is_done(int i) {
        return (state >> i) & 1;
    }

    // 所有任务是否完成
    int all_done() {
        return state == ALL_DONE;
    }

    complete_task(0);
    complete_task(2);
    complete_task(4);
    printf("状态: %05b\n", state);
    printf("任务1完成: %d\n", is_done(1));
    printf("任务2完成: %d\n", is_done(2));
    printf("全部完成: %d\n", all_done());
}
```

## 五、位运算高级技巧

```c
// 找出只出现一次的元素（其余出现两次）
int find_single(int arr[], int n) {
    int result = 0;
    for (int i = 0; i < n; i++) result ^= arr[i];
    return result;
}

// 找出两个只出现一次的元素
void find_two_singles(int arr[], int n, int *a, int *b) {
    int xor_all = 0;
    for (int i = 0; i < n; i++) xor_all ^= arr[i];

    // 找到最右边的1
    int diff_bit = xor_all & (-xor_all);

    *a = 0; *b = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] & diff_bit) *a ^= arr[i];
        else *b ^= arr[i];
    }
}

// 不用+号实现加法
int add_without_plus(int a, int b) {
    while (b != 0) {
        int carry = a & b;    // 进位
        a = a ^ b;            // 无进位加法
        b = carry << 1;       // 进位左移
    }
    return a;
}

// 取最近的更大的2的幂
uint32_t next_power_of_2(uint32_t n) {
    n--;
    n |= n >> 1;
    n |= n >> 2;
    n |= n >> 4;
    n |= n >> 8;
    n |= n >> 16;
    return n + 1;
}

void advanced_demo() {
    printf("\n=== 高级位运算 ===\n");

    int arr[] = {2, 3, 7, 3, 2};
    printf("单一元素: %d\n", find_single(arr, 5));  // 7

    int arr2[] = {1, 2, 1, 3, 2, 5};
    int x, y;
    find_two_singles(arr2, 6, &x, &y);
    printf("两个单一元素: %d, %d\n", x, y);  // 3, 5

    printf("不用+号: 15+27=%d\n", add_without_plus(15, 27));

    printf("next_pow2(5)=%u\n", next_power_of_2(5));
    printf("next_pow2(17)=%u\n", next_power_of_2(17));
}
```

## 六、子集枚举

```c
void enumerate_subsets(int n) {
    printf("\n集合{0,1,2}的所有子集:\n");
    for (int mask = 0; mask < (1 << n); mask++) {
        printf("{ ");
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) printf("%d ", i);
        }
        printf("}\n");
    }
}
```

## 七、面试真题

```c
void interview_questions() {
    printf("\n=== 面试真题 ===\n");

    // Q1: 不用判断语句实现取最大值
    // 如果a>b则a-b的符号位为0
    int a = 5, b = 3;
    int diff = a - b;
    int sign = (diff >> 31) & 1;  // 0或1
    int max_val = a - diff * sign;
    printf("max(%d,%d)=%d\n", a, b, max_val);

    // Q2: 位反转
    unsigned int x = 0b10110010;
    unsigned int rev = 0;
    for (int i = 0; i < 8; i++) {
        rev = (rev << 1) | (x & 1);
        x >>= 1;
    }
    printf("反转: %08b\n", rev);

    // Q3: 异或的性质
    // a ^ a = 0
    // a ^ 0 = a
    // a ^ b = b ^ a
    // (a ^ b) ^ c = a ^ (b ^ c)
    printf("5^5=%d, 5^0=%d\n", 5 ^ 5, 5 ^ 0);
}
```
