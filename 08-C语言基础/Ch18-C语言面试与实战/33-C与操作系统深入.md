# C与操作系统深入

## 一、系统调用

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/syscall.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <errno.h>
#include <string.h>

/*
 * 系统调用是用户态程序与内核交互的唯一接口。
 * 从用户态进入内核态的流程：
 * 1. 用户程序调用库函数（如read）
 * 2. 库函数触发软中断/syscall指令
 * 3. CPU从用户态切换到内核态
 * 4. 内核根据系统调用号执行对应函数
 * 5. 返回用户态
 */

// 直接使用syscall（绕过glibc）
void direct_syscall_demo() {
    const char *msg = "Hello via syscall\n";
    // write系统调用号在x86_64上是1
    syscall(SYS_write, 1, msg, strlen(msg));

    // 或者用更清晰的方式
    syscall(SYS_getpid);  // 获取进程ID
}

// 文件系统调用
void file_syscall_demo() {
    // open系统调用
    int fd = open("/tmp/test.txt", O_CREAT | O_WRONLY, 0644);
    if (fd < 0) {
        perror("open");
        return;
    }

    // write系统调用
    const char *data = "Hello, OS!\n";
    ssize_t written = write(fd, data, strlen(data));
    printf("写入 %zd 字节\n", written);

    // 获取文件信息
    struct stat st;
    fstat(fd, &st);
    printf("文件大小: %ld\n", (long)st.st_size);
    printf("Inode: %lu\n", (unsigned long)st.st_ino);

    close(fd);
    unlink("/tmp/test.txt");
}
```

 * 上下文切换过程：
 *
 * 1. 保存当前进程/线程的CPU状态
 *    - 通用寄存器（RAX, RBX, RCX...）

 *    - 刷新TLB（或使用ASID避免刷新）
 *
 * 4. 加载新进程的CPU状态
 *
 * 开销：通常1-10微秒

## 三、进程调度

```c
/*
 * 常见调度算法：
 *
 * 1. 先来先服务（FCFS）：简单但可能导致饥饿
 * 2. 最短作业优先（SJF）：最优平均等待时间
 * 3. 时间片轮转（RR）：公平，适合交互式系统
 * 4. 优先级调度：可以是静态或动态
 * 5. 多级反馈队列（MLFQ）：Linux使用的变体
 *
 * Linux CFS（完全公平调度器）：
 * - 使用红黑树管理所有可运行进程
 * - 虚拟运行时间（vruntime）最小的进程先运行
 * - 保证每个进程获得公平的CPU份额
 */

// 简化的调度器模拟
#include <stdlib.h>

typedef struct {
    int pid;
    int burst_time;
    int remaining;
    int priority;
    int arrival_time;
} Process;

// FCFS调度
void fcfs_schedule(Process procs[], int n) {
    int total_wait = 0, current_time = 0;
    printf("FCFS调度顺序: ");
    for (int i = 0; i < n; i++) {
        int wait = current_time - procs[i].arrival_time;
        if (wait < 0) { current_time = procs[i].arrival_time; wait = 0; }
        total_wait += wait;
        printf("P%d ", procs[i].pid);
        current_time += procs[i].burst_time;
    }
    printf("\n平均等待时间: %.2f\n", (float)total_wait / n);
}

// 时间片轮转
void rr_schedule(Process procs[], int n, int quantum) {
    int *remaining = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) remaining[i] = procs[i].burst_time;

    int time = 0, done = 0;
    printf("RR调度 (量子=%d): ", quantum);
    while (done < n) {
        for (int i = 0; i < n; i++) {
            if (remaining[i] > 0) {
                int exec = remaining[i] < quantum ? remaining[i] : quantum;
                remaining[i] -= exec;
                time += exec;
                printf("P%d(%d) ", procs[i].pid, exec);
                if (remaining[i] == 0) done++;
            }
        }
    }
    printf("\n完成时间: %d\n", time);
    free(remaining);
}
```

## 四、虚拟内存管理

```c
/*
 * 虚拟内存核心机制：
 *
 * 虚拟地址 -> 物理地址 转换流程：
 * 1. CPU发出虚拟地址
 * 2. MMU查询TLB（快表）
 * 3. TLB未命中则查询页表
 * 4. 页表给出物理页框号
 * 5. 组合物理地址访问内存
 *
 * 页表结构（以x86_64四级页表为例）：
 * PML4 -> PDPT -> PD -> PT -> 物理页
 * 每级索引9位，共36位虚拟地址空间
 */

// 模拟页表查询
#define PAGE_SIZE 4096
#define VPN(addr) ((addr) / PAGE_SIZE)
#define OFFSET(addr) ((addr) % PAGE_SIZE)

typedef struct {
    unsigned int frame_number;
    unsigned int valid : 1;
    unsigned int dirty : 1;
    unsigned int referenced : 1;
} PageTableEntry;

// 简化的地址翻译
unsigned int translate(PageTableEntry *pt, unsigned int vaddr) {
    unsigned int vpn = VPN(vaddr);
    unsigned int offset = OFFSET(vaddr);

    if (!pt[vpn].valid) {
        printf("缺页异常: VPN=%u\n", vpn);
        // 触发页面错误处理
        return 0xFFFFFFFF;  // 错误
    }

    pt[vpn].referenced = 1;
    return (pt[vpn].frame_number * PAGE_SIZE) + offset;
}
```

## 五、中断与异常

```c
/*
 * 中断分类：
 *
 * 1. 外部中断（硬件中断）
 *    - 可屏蔽中断（INTR）
 *    - 不可屏蔽中断（NMI）
 *
 * 2. 内部中断（异常）
 *    - 故障（Fault）：可恢复，返回当前指令
 *      如：缺页异常、除零错误
 *    - 陷阱（Trap）：有意触发，返回下一条指令
 *      如：断点、系统调用
 *    - 终止（Abort）：不可恢复
 *      如：硬件错误
 *
 * 中断处理流程：
 * 1. CPU保存当前上下文到栈
 * 2. 根据中断号查IDT（中断描述符表）
 * 3. 跳转到中断处理程序
 * 4. 处理完成后iret返回
 */

// 设置信号处理（模拟中断处理）
#include <signal.h>

void sigsegv_handler(int sig) {
    printf("捕获段错误信号 %d\n", sig);
    // 生产环境应该退出，这里仅演示
    _exit(1);
}

void sigfpe_handler(int sig) {
    printf("捕获浮点异常信号 %d (可能是除零)\n", sig);
    _exit(1);
}

void interrupt_demo() {
    signal(SIGSEGV, sigsegv_handler);
    signal(SIGFPE, sigfpe_handler);
    // int *p = NULL; *p = 1;  // 触发SIGSEGV
    // int a = 1 / 0;          // 触发SIGFPE
}
```

## 六、内存映射与mmap

```c
#include <sys/mman.h>

void mmap_demo() {
    // 匿名映射（不关联文件）
    void *addr = mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                      MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    if (addr != MAP_FAILED) {
        strcpy((char *)addr, "mmap数据");
        printf("mmap: %s\n", (char *)addr);
        munmap(addr, 4096);
    }

    // 文件映射
    int fd = open("/tmp/mmap_test", O_RDWR | O_CREAT, 0644);
    ftruncate(fd, 4096);

    char *mapped = (char *)mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                                MAP_SHARED, fd, 0);
    if (mapped != MAP_FAILED) {
        strcpy(mapped, "通过mmap写入文件");
        msync(mapped, 4096, MS_SYNC);  // 强制同步到磁盘
        munmap(mapped, 4096);
    }
    close(fd);
    unlink("/tmp/mmap_test");
}
```
