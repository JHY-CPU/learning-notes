### 题目1：手写单例模式（C语言实现）

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

// 懒汉式单例（线程安全）
typedef struct {
    int config_value;
    char name[64];
} Config;

static Config *instance = NULL;
static pthread_once_t once = PTHREAD_ONCE_INIT;

static void init_instance() {
    instance = (Config *)malloc(sizeof(Config));
    if (instance) {
        instance->config_value = 42;
        strcpy(instance->name, "default");
    }
}

Config *get_instance() {
    pthread_once(&once, init_instance);
    return instance;
}
```

### 题目2：实现一个简单的HashMap

```c
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

#define MAP_SIZE 64

typedef struct Entry {
    char *key;
    int value;
    struct Entry *next;
} Entry;

typedef struct {
    Entry *buckets[MAP_SIZE];
} HashMap;

unsigned int hash(const char *key) {
    unsigned int h = 5381;
    while (*key) h = ((h << 5) + h) + (unsigned char)*key++;
    return h % MAP_SIZE;
}

HashMap *map_create() {
    return (HashMap *)calloc(1, sizeof(HashMap));
}

void map_put(HashMap *m, const char *key, int val) {
    unsigned int idx = hash(key);
    Entry *e = m->buckets[idx];
    while (e) {
        if (strcmp(e->key, key) == 0) { e->value = val; return; }
        e = e->next;
    }
    Entry *new_e = (Entry *)malloc(sizeof(Entry));
    new_e->key = strdup(key);
    new_e->value = val;
    new_e->next = m->buckets[idx];
    m->buckets[idx] = new_e;
}

bool map_get(HashMap *m, const char *key, int *val) {
    Entry *e = m->buckets[hash(key)];
    while (e) {
        if (strcmp(e->key, key) == 0) { *val = e->value; return true; }
        e = e->next;
    }
    return false;
}
```

### 题目3：手写线程池（简化版）

```c
#include <pthread.h>
#include <stdlib.h>

#define MAX_THREADS 4
#define MAX_QUEUE 100

typedef void (*TaskFunc)(void *);

typedef struct {
    TaskFunc func;
    void *arg;
} Task;

typedef struct {
    pthread_t threads[MAX_THREADS];
    Task queue[MAX_QUEUE];
    int front, rear, size;
    pthread_mutex_t lock;
    pthread_cond_t cond;
    int shutdown;
} ThreadPool;

void *worker(void *arg) {
    ThreadPool *pool = (ThreadPool *)arg;
    while (1) {
        pthread_mutex_lock(&pool->lock);
        while (pool->size == 0 && !pool->shutdown)
            pthread_cond_wait(&pool->cond, &pool->lock);
        if (pool->shutdown) {
            pthread_mutex_unlock(&pool->lock);
            break;
        }
        Task task = pool->queue[pool->front];
        pool->front = (pool->front + 1) % MAX_QUEUE;
        pool->size--;
        pthread_mutex_unlock(&pool->lock);
        task.func(task.arg);
    }
    return NULL;
}

ThreadPool *pool_create() {
    ThreadPool *pool = (ThreadPool *)calloc(1, sizeof(ThreadPool));
    pthread_mutex_init(&pool->lock, NULL);
    pthread_cond_init(&pool->cond, NULL);
    for (int i = 0; i < MAX_THREADS; i++)
        pthread_create(&pool->threads[i], NULL, worker, pool);
    return pool;
}
```

### 题目4：实现一个阻塞队列

```c
#include <pthread.h>

#define BQ_CAPACITY 100

typedef struct {
    int data[BQ_CAPACITY];
    int front, rear, size;
    pthread_mutex_t mutex;
    pthread_cond_t not_empty;
    pthread_cond_t not_full;
} BlockingQueue;

void bq_init(BlockingQueue *q) {
    q->front = q->rear = q->size = 0;
    pthread_mutex_init(&q->mutex, NULL);
    pthread_cond_init(&q->not_empty, NULL);
    pthread_cond_init(&q->not_full, NULL);
}

void bq_put(BlockingQueue *q, int item) {
    pthread_mutex_lock(&q->mutex);
    while (q->size == BQ_CAPACITY)
        pthread_cond_wait(&q->not_full, &q->mutex);
    q->data[q->rear] = item;
    q->rear = (q->rear + 1) % BQ_CAPACITY;
    q->size++;
    pthread_cond_signal(&q->not_empty);
    pthread_mutex_unlock(&q->mutex);
}

int bq_take(BlockingQueue *q) {
    pthread_mutex_lock(&q->mutex);
    while (q->size == 0)
        pthread_cond_wait(&q->not_empty, &q->mutex);
    int item = q->data[q->front];
    q->front = (q->front + 1) % BQ_CAPACITY;
    q->size--;
    pthread_cond_signal(&q->not_full);
    pthread_mutex_unlock(&q->mutex);
    return item;
}
```

### 题目5：TopK问题

```c
#include <stdlib.h>

// 最小堆实现TopK
void sift_down(int *heap, int n, int i) {
    int smallest = i, l = 2*i+1, r = 2*i+2;
    if (l < n && heap[l] < heap[smallest]) smallest = l;
    if (r < n && heap[r] < heap[smallest]) smallest = r;
    if (smallest != i) {
        int t = heap[i]; heap[i] = heap[smallest]; heap[smallest] = t;
        sift_down(heap, n, smallest);
    }
}

void top_k(int *arr, int n, int k) {
    // 用前k个元素建最小堆
    int *heap = (int *)malloc(k * sizeof(int));
    for (int i = 0; i < k; i++) heap[i] = arr[i];
    for (int i = k/2 - 1; i >= 0; i--) sift_down(heap, k, i);

    // 遍历剩余元素
    for (int i = k; i < n; i++) {
        if (arr[i] > heap[0]) {
            heap[0] = arr[i];
            sift_down(heap, k, 0);
        }
    }

    printf("Top %d: ", k);
    for (int i = 0; i < k; i++) printf("%d ", heap[i]);
    printf("\n");
    free(heap);
}
```

### 题目6：实现strcpy的安全版本

```c
#include <stddef.h>

size_t safe_strcpy(char *dest, const char *src, size_t dest_size) {
    if (!dest || !src || dest_size == 0) return 0;
    size_t i;
    for (i = 0; i < dest_size - 1 && src[i]; i++)
        dest[i] = src[i];
    dest[i] = '\0';
    // 返回需要的长度（不含'\0'）
    while (src[i]) i++;
    return i;  // 如果>dest_size-1说明发生了截断
}
```

### 题目7：C语言实现简单对象系统

```c
// 基于函数指针模拟虚函数
typedef struct Animal Animal;

typedef struct {
    void (*speak)(Animal *self);
    void (*destroy)(Animal *self);
} AnimalVTable;

struct Animal {
    AnimalVTable *vtable;
    char name[32];
};

void animal_speak(Animal *a) { a->vtable->speak(a); }
void animal_destroy(Animal *a) { a->vtable->destroy(a); }

// Dog实现
void dog_speak(Animal *self) { printf("%s: 汪汪!\n", self->name); }
void dog_destroy(Animal *self) { printf("释放Dog: %s\n", self->name); free(self); }

AnimalVTable dog_vtable = {dog_speak, dog_destroy};

Animal *dog_create(const char *name) {
    Animal *d = (Animal *)malloc(sizeof(Animal));
    d->vtable = &dog_vtable;
    strncpy(d->name, name, 31);
    return d;
}
```
