# 手撕代码 - 字符串操作

## 一、手写strlen

```c
#include <stddef.h>
#include <stdio.h>

// 方法1：计数器
size_t my_strlen_v1(const char *s) {
    size_t len = 0;
    while (*s++) len++;
    return len;
}

// 方法2：指针差
size_t my_strlen_v2(const char *s) {
    const char *p = s;
    while (*p) p++;
    return (size_t)(p - s);
}

// 方法3：递归
size_t my_strlen_v3(const char *s) {
    return *s ? 1 + my_strlen_v3(s + 1) : 0;
}

// 方法4：四字节优化
size_t my_strlen_v4(const char *s) {
    const char *p = s;
    // 先按字节对齐
    while ((size_t)p & 3) {
        if (!*p) return (size_t)(p - s);
        p++;
    }
    // 按4字节检查
    const unsigned int *wp = (const unsigned int *)p;
    while (1) {
        unsigned int w = *wp++;
        // 检查是否有0字节
        if (((w - 0x01010101) & ~w & 0x80808080) != 0) {
            // 找到0字节，回退到字节级检查
            const char *cp = (const char *)(wp - 1);
            for (int i = 0; i < 4; i++) {
                if (!cp[i]) return (size_t)(cp - s) + i;
            }
        }
    }
}
```

## 二、手写strcpy

```c
// 基础版本
char *my_strcpy(char *dest, const char *src) {
    char *ret = dest;
    while ((*dest++ = *src++));
    return ret;
}

// 安全版本：带边界检查
char *my_strcpy_safe(char *dest, const char *src, size_t dest_size) {
    if (!dest || !src || dest_size == 0) return NULL;
    char *ret = dest;
    while (--dest_size && (*dest++ = *src++));
    *dest = '\0';
    return ret;
}
```

## 三、手写strncpy

```c
char *my_strncpy(char *dest, const char *src, size_t n) {
    char *ret = dest;
    while (n-- && (*dest++ = *src++));
    while (n-- > 0) *dest++ = '\0';  // 填充剩余字节
    return ret;
}
```

## 四、手写strcat

```c
char *my_strcat(char *dest, const char *src) {
    char *ret = dest;
    while (*dest) dest++;       // 找到末尾
    while ((*dest++ = *src++)); // 拷贝
    return ret;
}

// 安全版本
char *my_strncat(char *dest, const char *src,
                 size_t dest_size, size_t n) {
    size_t dest_len = 0;
    while (dest[dest_len]) dest_len++;

    size_t i = 0;
    while (i < n && src[i] && dest_len + i < dest_size - 1) {
        dest[dest_len + i] = src[i];
        i++;
    }
    dest[dest_len + i] = '\0';
    return dest;
}
```

## 五、手写strcmp

```c
int my_strcmp(const char *s1, const char *s2) {
    while (*s1 && *s1 == *s2) {
        s1++;
        s2++;
    }
    return (unsigned char)*s1 - (unsigned char)*s2;
}

int my_strncmp(const char *s1, const char *s2, size_t n) {
    while (n-- && *s1 && *s1 == *s2) {
        s1++;
        s2++;
    }
    if (n == (size_t)-1) return 0;
    return (unsigned char)*s1 - (unsigned char)*s2;
}
```

```c
char *my_strdup(const char *s) {
    size_t len = strlen(s) + 1;
    char *dup = (char *)malloc(len);
    if (dup) memcpy(dup, s, len);
```

## 七、手写strstr

```c
// 暴力法 O(m*n)
char *my_strstr(const char *haystack, const char *needle) {
    if (!*needle) return (char *)haystack;
    while (*haystack) {
        const char *h = haystack;
        const char *n = needle;
        while (*h && *n && *h == *n) { h++; n++; }
        if (!*n) return (char *)haystack;
        haystack++;
    }
    return NULL;
}
```

## 八、手写strtok

```c
char *my_strtok(char *str, const char *delim) {
    static char *saved = NULL;
    if (str) saved = str;
    if (!saved || !*saved) return NULL;

    // 跳过前导分隔符
    while (*saved && strchr(delim, *saved)) saved++;
    if (!*saved) return NULL;

    char *token_start = saved;
    while (*saved && !strchr(delim, *saved)) saved++;

    if (*saved) {
        *saved = '\0';
        saved++;
    } else {
        saved = NULL;
    }
    return token_start;
}
```

## 九、手写memcpy

```c
void *my_memcpy(void *dest, const void *src, size_t n) {
    char *d = (char *)dest;
    const char *s = (const char *)src;
    while (n--) *d++ = *s++;
    return dest;
}
```

## 十、手写memmove

```c
void *my_memmove(void *dest, const void *src, size_t n) {
    char *d = (char *)dest;
    const char *s = (const char *)src;
    if (d < s) {
        while (n--) *d++ = *s++;
    } else {
        d += n;
        s += n;
        while (n--) *--d = *--s;
    }
    return dest;
}
```

## 十一、手写memset

```c
void *my_memset(void *s, int c, size_t n) {
    unsigned char *p = (unsigned char *)s;
    while (n--) *p++ = (unsigned char)c;
    return s;
}
```

## 十二、手写sprintf (简化版)

```c
#include <stdarg.h>
#include <stdio.h>

int my_itoa(int num, char *buf) {
    int len = 0;
    int is_neg = 0;
    if (num < 0) { is_neg = 1; num = -num; }
    char tmp[12];
    do {
        tmp[len++] = '0' + (num % 10);
        num /= 10;
    } while (num);
    if (is_neg) tmp[len++] = '-';
    for (int i = 0; i < len; i++) {
        buf[i] = tmp[len - 1 - i];
    }
    buf[len] = '\0';
    return len;
}
```

## 十三、手写atoi/itoa

```c
int my_atoi(const char *s) {
    while (*s == ' ') s++;
    int sign = (*s == '-') ? -1 : 1;
    if (*s == '+' || *s == '-') s++;
    long result = 0;
    while (*s >= '0' && *s <= '9') {
        result = result * 10 + (*s - '0');
        s++;
    }
    return (int)(sign * result);
}

void my_itoa(int n, char *buf) {
    int i = 0, neg = 0;
    if (n < 0) { neg = 1; n = -n; }
    if (n == 0) { buf[i++] = '0'; }
    else {
        char tmp[12];
        while (n) { tmp[i++] = '0' + n % 10; n /= 10; }
        if (neg) tmp[i++] = '-';
        for (int j = 0; j < i; j++) buf[j] = tmp[i-1-j];
    }
    buf[i] = '\0';
}
```
