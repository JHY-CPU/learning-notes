# 预处理器面试题

## 一、宏定义基础

```c
#include <stdio.h>

// 对象宏
#define PI 3.14159
#define MAX_SIZE 1024
#define TRUE 1
#define FALSE 0

// 函数宏（注意参数要加括号）
#define SQUARE(x) ((x) * (x))
#define MAX(a, b) ((a) > (b) ? (a) : (b))
#define MIN(a, b) ((a) < (b) ? (a) : (b))

// 经典陷阱：宏参数不加括号
#define BAD_SQUARE(x) x * x     // 错误示范
#define GOOD_SQUARE(x) ((x) * (x))  // 正确

void basic_macro_demo() {
    printf("PI = %f\n", PI);
    printf("SQUARE(3+1) = %d\n", SQUARE(3 + 1));  // 16 (正确)

    // BAD_SQUARE(3+1) 展开为 3+1*3+1 = 7
    printf("BAD_SQUARE(3+1) = %d\n", BAD_SQUARE(3 + 1));   // 7 (错误)
    printf("GOOD_SQUARE(3+1) = %d\n", GOOD_SQUARE(3 + 1)); // 16 (正确)
}
```

## 二、字符串化与连接

```c
// # 字符串化运算符
#define TO_STRING(x) #x
#define PRINT_VAR(var) printf(#var " = %d\n", var)

// ## 连接运算符
#define CONCAT(a, b) a##b
#define MAKE_FUNC(name) void func_##name() { printf(#name " called\n"); }

// 生成函数
MAKE_FUNC(init)
MAKE_FUNC(cleanup)

void string_concat_demo() {
    printf("%s\n", TO_STRING(Hello World));  // "Hello World"

    int score = 95;
    PRINT_VAR(score);  // "score = 95"

    int xy = 100;
    printf("CONCAT(x,y) = %d\n", CONCAT(x, y));  // 100

    func_init();
    func_cleanup();
}
```

## 四、可变参数宏

```c
#include <stdio.h>

// __VA_ARGS__ 可变参数
#define LOG(fmt, ...) printf("[LOG] " fmt "\n", ##__VA_ARGS__)
#define ERROR(fmt, ...) printf("[ERROR:%s:%d] " fmt "\n", \
    __FILE__, __LINE__, ##__VA_ARGS__)

// ##__VA_ARGS__ 在GCC/Clang中，当没有可变参数时会去掉前面的逗号
#define TRACE(...) printf(__VA_ARGS__)

void variadic_macro_demo() {
    LOG("程序启动");
    LOG("处理数据: %d, %d", 10, 20);
    ERROR("内存分配失败");
    TRACE("简单输出\n");
}
```

## 五、常见陷阱题

```c
// 陷阱1：宏的副作用
#define MAX_INC(a, b) ((a) > (b) ? (a) : (b))
void trap1() {
    int x = 5, y = 8;
    // MAX_INC(x++, y++) 会展开为：
    // ((x++) > (y++) ? (x++) : (y++))
    // x被自增了两次！
    int z = MAX_INC(x++, y++);
    printf("x=%d, y=%d, z=%d\n", x, y, z);  // x=6,y=9,z=9 (意外！)
}

// 陷阱2：宏与类型无关但不安全
#define SWAP(a, b) do { \
    typeof(a) _tmp = a; \
    a = b; \
    b = _tmp; \
} while(0)

// 陷阱3：宏定义中的分号
// 错误：if/else使用时会出现问题
#define BAD_IF(cond, action) if (cond) action;
// 正确：使用do-while(0)包裹
#define GOOD_IF(cond, action) do { if (cond) { action; } } while(0)

// 陷阱4：宏不能递归
// #define FACT(n) ((n) <= 1 ? 1 : (n) * FACT(n-1))  // 无效！

void trap_demo() {
    trap1();
}
```

## 六、预定义宏

```c
#include <stdio.h>

void predefined_macros_demo() {
    printf("文件: %s\n", __FILE__);
    printf("行号: %d\n", __LINE__);
    printf("日期: %s\n", __DATE__);
    printf("时间: %s\n", __TIME__);
    printf("函数: %s\n", __func__);
    printf("标准: ");
    #if __STDC_VERSION__ >= 201112L
        printf("C11\n");
    #elif __STDC_VERSION__ >= 199901L
        printf("C99\n");
    #else
        printf("C89/C90\n");
    #endif
}
```

## 七、宏实现泛型（C11之前）

```c
// 用宏实现类型无关的比较
#define DEFINE_COMP_FUNC(type) \
    type max_##type(type a, type b) { \
        return a > b ? a : b; \
    }

DEFINE_COMP_FUNC(int)
DEFINE_COMP_FUNC(double)

// typeof/typeof GNU扩展实现泛型交换
#define GENERIC_SWAP(a, b) do { \
    __typeof__(a) _t = (a); \
    (a) = (b); \
    (b) = _t; \
} while(0)

// C11 _Generic关键字
#define type_name(x) _Generic((x), \
    int: "int", \
    double: "double", \
    char*: "char*", \
    default: "unknown")

void generic_macro_demo() {
    printf("max_int(3,5) = %d\n", max_int(3, 5));
    printf("max_double(3.14,2.71) = %f\n", max_double(3.14, 2.71));

    int a = 10, b = 20;
    GENERIC_SWAP(a, b);
    printf("交换后: a=%d, b=%d\n", a, b);

    printf("type_name(a) = %s\n", type_name(a));
    printf("type_name(3.14) = %s\n", type_name(3.14));
}
```

## 八、#pragma指令

```c
// #pragma once 代替头文件保护（非标准但广泛支持）
// #pragma pack(n) 控制对齐（见结构体章节）
// #pragma message("编译时消息")
// #pragma warning 控制警告

#ifdef __GNUC__
    #pragma GCC diagnostic push
    #pragma GCC diagnostic ignored "-Wunused-variable"
    int unused_var;  // 不会产生警告
    #pragma GCC diagnostic pop
#endif

void pragma_demo() {
    #pragma message("正在编译pragma_demo函数")
    printf("#pragma指令用于向编译器传递额外信息\n");
}
```
