# 系统编程面试专题

## 一、进程管理

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <sys/types.h>

// fork返回值：子进程返回0，父进程返回子进程PID
void fork_demo() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("fork失败");
        exit(1);
    } else if (pid == 0) {
        // 子进程
        printf("子进程 PID=%d, PPID=%d\n", getpid(), getppid());
        execlp("ls", "ls", "-l", NULL);
        perror("exec失败");
        exit(1);
    } else {
        // 父进程
        printf("父进程 PID=%d, 子进程PID=%d\n", getpid(), pid);
        int status;
        waitpid(pid, &status, 0);
        if (WIFEXITED(status)) {
            printf("子进程正常退出: %d\n", WEXITSTATUS(status));
        } else if (WIFSIGNALED(status)) {
            printf("子进程被信号终止: %d\n", WTERMSIG(status));
        }
    }
}

// 僵尸进程：子进程退出后父进程未wait
// 孤儿进程：父进程先退出，子进程被init收养
```

## 二、进程间通信（IPC）

```c
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>
#include <sys/msg.h>
#include <string.h>

// 1. 管道
void pipe_demo() {
    int fd[2];
    pipe(fd);

    if (fork() == 0) {
        close(fd[0]);  // 关闭读端
        write(fd[1], "hello", 6);
        close(fd[1]);
        exit(0);
    } else {
        close(fd[1]);  // 关闭写端
        char buf[64];
        read(fd[0], buf, sizeof(buf));
        printf("管道收到: %s\n", buf);
        close(fd[0]);
        wait(NULL);
    }
}

// 2. 共享内存
void shm_demo() {
    key_t key = ftok("/tmp", 'A');
    int shmid = shmget(key, 1024, IPC_CREAT | 0666);

    if (fork() == 0) {
        // 子进程写
        char *shm = (char *)shmat(shmid, NULL, 0);
        strcpy(shm, "共享内存数据");
        shmdt(shm);
        exit(0);
    } else {
        // 父进程读
        wait(NULL);
        char *shm = (char *)shmat(shmid, NULL, 0);
        printf("共享内存: %s\n", shm);
        shmdt(shm);
        shmctl(shmid, IPC_RMID, NULL);  // 删除
    }
}

// 3. 消息队列
typedef struct {
    long mtype;
    char mtext[256];
} Message;

void msg_queue_demo() {
    key_t key = ftok("/tmp", 'B');
    int msqid = msgget(key, IPC_CREAT | 0666);

    if (fork() == 0) {
        Message msg = {1, "队列消息"};
        msgsnd(msqid, &msg, sizeof(msg.mtext), 0);
        exit(0);
    } else {
        wait(NULL);
        Message msg;
        msgrcv(msqid, &msg, sizeof(msg.mtext), 1, 0);
        printf("消息队列: %s\n", msg.mtext);
        msgctl(msqid, IPC_RMID, NULL);
    }
}
```

## 三、多线程同步

```c
#include <pthread.h>
#include <semaphore.h>

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_rwlock_t rwlock = PTHREAD_rwlock_INITIALIZER;
sem_t semaphore;
pthread_cond_t cond = PTHREAD_COND_INITIALIZER;

// 互斥锁
void mutex_demo() {
    int counter = 0;

    void *worker(void *arg) {
        for (int i = 0; i < 100000; i++) {
            pthread_mutex_lock(&mutex);
            counter++;
            pthread_mutex_unlock(&mutex);
        }
        return NULL;
    }

    pthread_t t1, t2;
    pthread_create(&t1, NULL, worker, NULL);
    pthread_create(&t2, NULL, worker, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    printf("counter = %d\n", counter);  // 200000
    pthread_mutex_destroy(&mutex);
}

// 读写锁
void rwlock_demo() {
    int shared_data = 0;

    void *reader(void *arg) {
        pthread_rwlock_rdlock(&rwlock);
        printf("读取: %d\n", shared_data);
        pthread_rwlock_unlock(&rwlock);
        return NULL;
    }

    void *writer(void *arg) {
        pthread_rwlock_wrlock(&rwlock);
        shared_data++;
        printf("写入: %d\n", shared_data);
        pthread_rwlock_unlock(&rwlock);
        return NULL;
    }
}

// 条件变量（生产者消费者）
void condvar_demo() {
    int ready = 0;

    void *producer(void *arg) {
        pthread_mutex_lock(&mutex);
        ready = 1;
        printf("生产完成\n");
        pthread_cond_signal(&cond);
        pthread_mutex_unlock(&mutex);
        return NULL;
    }

    void *consumer(void *arg) {
        pthread_mutex_lock(&mutex);
        while (!ready)
            pthread_cond_wait(&cond, &mutex);
        printf("消费完成\n");
        pthread_mutex_unlock(&mutex);
        return NULL;
    }

    pthread_t p, c;
    pthread_create(&p, NULL, producer, NULL);
    pthread_create(&c, NULL, consumer, NULL);
    pthread_join(p, NULL);
    pthread_join(c, NULL);
    pthread_mutex_destroy(&mutex);
    pthread_cond_destroy(&cond);
}
```

## 四、IO模型

```c
#include <sys/select.h>
#include <poll.h>
#include <sys/epoll.h>
#include <fcntl.h>

/*
五种IO模型：
1. 阻塞IO：默认，调用后一直等待
2. 非阻塞IO：O_NONBLOCK，立即返回
3. IO多路复用：select/poll/epoll
4. 信号驱动IO：SIGIO
5. 异步IO：aio_read/aio_write
*/

// 设置非阻塞
void set_nonblocking(int fd) {
    int flags = fcntl(fd, F_GETFL, 0);
    fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

// select示例
void select_example(int listen_fd) {
    fd_set read_fds;
    int max_fd = listen_fd;

    while (1) {
        FD_ZERO(&read_fds);
        FD_SET(listen_fd, &read_fds);

        struct timeval tv = {5, 0};  // 5秒超时
        int ready = select(max_fd + 1, &read_fds, NULL, NULL, &tv);
        if (ready > 0 && FD_ISSET(listen_fd, &read_fds)) {
            // 有新连接
            printf("新连接到达\n");
        }
    }
}

// epoll示例
void epoll_example(int listen_fd) {
    int epfd = epoll_create1(0);
    struct epoll_event ev, events[100];

    ev.events = EPOLLIN;
    ev.data.fd = listen_fd;
    epoll_ctl(epfd, EPOLL_CTL_ADD, listen_fd, &ev);

    while (1) {
        int nfds = epoll_wait(epfd, events, 100, -1);
        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == listen_fd) {
                // 接受新连接
            } else {
                // 处理数据
            }
        }
    }
}
```

## 五、信号处理

```c
#include <signal.h>

volatile sig_atomic_t running = 1;

void signal_handler(int sig) {
    if (sig == SIGINT || sig == SIGTERM) {
        running = 0;
    }
}

void signal_demo() {
    struct sigaction sa;
    sa.sa_handler = signal_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = 0;

    sigaction(SIGINT, &sa, NULL);
    sigaction(SIGTERM, &sa, NULL);

    printf("按Ctrl+C退出...\n");
    while (running) {
        sleep(1);
    }
    printf("收到退出信号\n");
}

// 忽略信号
// signal(SIGPIPE, SIG_IGN);
```

## 六、文件操作

```c
#include <fcntl.h>
#include <sys/stat.h>

void file_io_demo() {
    // 低级IO
    int fd = open("test.txt", O_CREAT | O_WRONLY, 0644);
    write(fd, "Hello\n", 6);
    close(fd);

    // 文件锁
    struct flock lock;
    lock.l_type = F_WRLCK;
    lock.l_whence = SEEK_SET;
    lock.l_start = 0;
    lock.l_len = 0;  // 锁整个文件

    fd = open("test.txt", O_RDWR);
    if (fcntl(fd, F_SETLK, &lock) == -1) {
        printf("文件已被锁定\n");
    }

    // mmap内存映射
    struct stat st;
    fstat(fd, &st);
    char *mapped = mmap(NULL, st.st_size, PROT_READ,
                        MAP_PRIVATE, fd, 0);
    if (mapped != MAP_FAILED) {
        printf("mmap内容: %.10s...\n", mapped);
        munmap(mapped, st.st_size);
    }
    close(fd);
}
```

## 面试常问总结

```text
Q: select/poll/epoll区别？
A: select: 1024限制，遍历O(n)，每次需重置
   poll: 无数量限制，遍历O(n)
   epoll: 无数量限制，事件驱动O(1)，支持ET/LT

Q: 进程与线程区别？
A: 进程: 独立地址空间，开销大，通信复杂
   线程: 共享地址空间，开销小，同步复杂

Q: 死锁条件及预防？
A: 四条件: 互斥、持有等待、不可剥夺、循环等待
   预防: 破坏任一条件即可

Q: 什么是惊群效应？
A: 多个进程/线程同时等待同一事件，事件发生时全部被唤醒，
   但只有一个能处理。解决方案: EPOLLEXCLUSIVE, SO_REUSEPORT
*/
```
