## 华为面试题

### 题目1：字符串反转（不使用额外空间）

```c
#include <stdio.h>
#include <string.h>

void reverse_string(char *str) {
    int len = strlen(str);
    for (int i = 0; i < len / 2; i++) {
        char temp = str[i];
        str[i] = str[len - 1 - i];
        str[len - 1 - i] = temp;
    }
}

// 华为变体：反转句子中的单词
// "hello world" -> "world hello"
void reverse_words(char *str) {
    int len = strlen(str);
    // 先整体反转
    reverse_string(str);
    // 再逐个单词反转
    int start = 0;
    for (int i = 0; i <= len; i++) {
        if (str[i] == ' ' || str[i] == '\0') {
            for (int j = start; j < (start + i) / 2; j++) {
                char temp = str[j];
                str[j] = str[start + i - 1 - j];
                str[start + i - 1 - j] = temp;
            }
            start = i + 1;
        }
    }
}
```

### 题目2：查找字符串中第一个只出现一次的字符

```c
char first_unique(const char *str) {
    int freq[256] = {0};
    for (int i = 0; str[i]; i++)
        freq[(unsigned char)str[i]]++;
    for (int i = 0; str[i]; i++)
        if (freq[(unsigned char)str[i]] == 1)
            return str[i];
    return '\0';
}
```

### 题单3：实现memcpy

```c
void *my_memcpy(void *dest, const void *src, size_t n) {
    char *d = (char *)dest;
    const char *s = (const char *)src;
    // 处理重叠：从后向前
    if (d > s && d < s + n) {
        d += n; s += n;
        while (n--) *--d = *--s;
    } else {
        while (n--) *d++ = *s++;
    }
    return dest;
}
```

### 题目4：实现LRU缓存

```c
#include <stdlib.h>

#define CACHE_SIZE 100

typedef struct LRUNode {
    int key, value;
    struct LRUNode *prev, *next;
} LRUNode;

typedef struct {
    int capacity, size;
    LRUNode *head, *tail;
    LRUNode *map[CACHE_SIZE];  // 简化hash
} LRUCache;

LRUCache *lru_create(int cap) {
    LRUCache *c = (LRUCache *)calloc(1, sizeof(LRUCache));
    c->capacity = cap;
    c->head = (LRUNode *)calloc(1, sizeof(LRUNode));
    c->tail = (LRUNode *)calloc(1, sizeof(LRUNode));
    c->head->next = c->tail;
    c->tail->prev = c->head;
    return c;
}

void lru_remove(LRUNode *node) {
    node->prev->next = node->next;
    node->next->prev = node->prev;
}

void lru_add_to_head(LRUCache *c, LRUNode *node) {
    node->next = c->head->next;
    node->prev = c->head;
    c->head->next->prev = node;
    c->head->next = node;
}

int lru_get(LRUCache *c, int key) {
    if (key < CACHE_SIZE && c->map[key]) {
        LRUNode *node = c->map[key];
        lru_remove(node);
        lru_add_to_head(c, node);
        return node->value;
    }
    return -1;
}

void lru_put(LRUCache *c, int key, int value) {
    if (key < CACHE_SIZE && c->map[key]) {
        c->map[key]->value = value;
        lru_remove(c->map[key]);
        lru_add_to_head(c, c->map[key]);
    } else {
        LRUNode *node = (LRUNode *)malloc(sizeof(LRUNode));
        node->key = key; node->value = value;
        if (c->size >= c->capacity) {
            LRUNode *last = c->tail->prev;
            lru_remove(last);
            c->map[last->key] = NULL;
            free(last);
            c->size--;
        }
        lru_add_to_head(c, node);
        if (key < CACHE_SIZE) c->map[key] = node;
        c->size++;
    }
}
```

### 题目5：大数加法

```c
#include <string.h>

void big_number_add(const char *a, const char *b, char *result) {
    int len_a = strlen(a), len_b = strlen(b);
    int max_len = (len_a > len_b ? len_a : len_b) + 1;
    int carry = 0, k = max_len;

    result[k] = '\0';
    int i = len_a - 1, j = len_b - 1;

    while (i >= 0 || j >= 0 || carry) {
        int sum = carry;
        if (i >= 0) sum += a[i--] - '0';
        if (j >= 0) sum += b[j--] - '0';
        carry = sum / 10;
        result[--k] = '0' + (sum % 10);
    }

    if (k > 0) memmove(result, result + k, max_len - k + 1);
}
```

char *my_strncpy(char *dest, const char *src, size_t n) {
    char *ret = dest;
    while (n-- && (*dest++ = *src++));

### 题目7：判断链表是否有环并找到入口

```c
typedef struct Node {
    int val;
    struct Node *next;
} Node;

Node *detect_cycle(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) {
            Node *entry = head;
            while (entry != slow) {
                entry = entry->next;
                slow = slow->next;
            }
            return entry;
        }
    }
    return NULL;
}
```

### 题目8：C语言中struct和union的区别

```c
/*
struct：所有成员分别占用独立空间，总大小 >= 各成员大小之和
union：所有成员共享同一块内存，总大小 = 最大成员大小（考虑对齐）

应用场景：
- struct：同时需要多个不同类型的数据
- union：同一时刻只需要存储一种类型的数据（节省内存）
- union常见用途：变体类型、类型转换、嵌入式寄存器映射
*/

void struct_union_demo() {
    struct { int i; float f; char c; } s;
    union  { int i; float f; char c; } u;

    printf("struct大小: %zu\n", sizeof(s));  // 12 (考虑对齐)
    printf("union大小: %zu\n", sizeof(u));   // 4 (最大成员)
}
```
