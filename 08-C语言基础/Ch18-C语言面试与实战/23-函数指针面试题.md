# 函数指针面试题

## 一、函数指针基础

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }
int divide(int a, int b) { return b ? a / b : 0; }

void basic_function_pointer() {
    // 声明函数指针
    int (*op)(int, int);

    op = add;
    printf("add(3,4) = %d\n", op(3, 4));

    op = sub;
    printf("sub(3,4) = %d\n", op(3, 4));

    // 函数名即函数地址
    printf("add == &add: %d\n", add == &add);  // 1
}
```

## 二、typedef简化

## 三、函数指针数组（跳转表）

```c
// 跳转表：用数组索引代替switch
typedef enum { OP_ADD, OP_SUB, OP_MUL, OP_DIV, OP_COUNT } OpCode;

BinaryOp jump_table[OP_COUNT] = {add, sub, mul, divide};

int execute(OpCode op, int a, int b) {
    if (op >= 0 && op < OP_COUNT) {
        return jump_table[op](a, b);
    }
    return 0;
}

void jump_table_demo() {
    printf("跳转表:\n");
    for (int i = 0; i < OP_COUNT; i++) {
        printf("  op[%d](8,2) = %d\n", i, execute(i, 8, 2));
    }
}
```

## 四、回调函数

```c
// 回调函数作为参数
void for_each(int *arr, int n, void (*callback)(int)) {
    for (int i = 0; i < n; i++) {
        callback(arr[i]);
    }
}

void print_value(int x) { printf("%d ", x); }
void print_square(int x) { printf("%d ", x * x); }

// 比较回调
typedef int (*CompareFn)(const void *, const void *);

int compare_int_asc(const void *a, const void *b) {
    return *(const int *)a - *(const int *)b;
}

int compare_int_desc(const void *a, const void *b) {
    return *(const int *)b - *(const int *)a;
}

void callback_demo() {
    int arr[] = {3, 1, 4, 1, 5, 9};
    printf("值: "); for_each(arr, 6, print_value); printf("\n");
    printf("平方: "); for_each(arr, 6, print_square); printf("\n");

    qsort(arr, 6, sizeof(int), compare_int_asc);
    printf("升序: "); for_each(arr, 6, print_value); printf("\n");

    qsort(arr, 6, sizeof(int), compare_int_desc);
    printf("降序: "); for_each(arr, 6, print_value); printf("\n");
}
```

## 五、返回函数指针的函数

```c
// 根据操作符返回对应的函数
BinaryOp get_operation(char op) {
    switch (op) {
        case '+': return add;
        case '-': return sub;
        case '*': return mul;
        case '/': return divide;
        default:  return NULL;
    }
}

void get_op_demo() {
    BinaryOp op = get_operation('+');
    if (op) printf("get_operation('+')(3,4) = %d\n", op(3, 4));

    op = get_operation('*');
    if (op) printf("get_operation('*')(3,4) = %d\n", op(3, 4));
}
```

## 六、复杂声明解读

void complex_declarations() {
    /*
     * int (*f)(void)：f是函数指针，无参，返回int
     * int (*f[])(int)：f是数组，元素是函数指针，参数int，返回int
     * int *(*f)(int *)：f是函数指针，参数int*，返回int*
     * void (*signal(int, void(*)(int)))(int)：

## 七、计算器实现

```c
#include <math.h>

double calc_add(double a, double b) { return a + b; }
double calc_sub(double a, double b) { return a - b; }
double calc_mul(double a, double b) { return a * b; }
double calc_pow(double a, double b) { return pow(a, b); }

typedef double (*CalcFunc)(double, double);

typedef struct {
    char name;
    CalcFunc func;
} Calculator;

Calculator calcs[] = {
    {'+', calc_add},
    {'-', calc_sub},
    {'*', calc_mul},
    {'^', calc_pow},
};

double calculator(char op, double a, double b) {
    for (int i = 0; i < 4; i++) {
        if (calcs[i].name == op) return calcs[i].func(a, b);
    }
    return 0;
}

void calculator_demo() {
    printf("计算器: 3+4=%.0f\n", calculator('+', 3, 4));
    printf("计算器: 2^10=%.0f\n", calculator('^', 2, 10));
}
```
