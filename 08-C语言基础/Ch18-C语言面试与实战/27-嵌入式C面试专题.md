# 嵌入式C面试专题

## 一、volatile关键字

```c
#include <stdio.h>

/*
 * volatile告诉编译器：此变量可能被意外修改，不要优化。
 *
 * 使用场景：
 * 1. 硬件寄存器访问
 * 2. 中断服务程序中的共享变量
 * 3. 多线程共享变量（但不能替代锁）
 */

// 模拟硬件寄存器
#define STATUS_REG  (*(volatile unsigned int *)0x40021000)
#define DATA_REG    (*(volatile unsigned int *)0x40021004)

// 没有volatile，编译器可能优化成只读一次
void wait_for_ready_bad() {
    // 编译器可能优化为死循环！
    // while ((STATUS_REG & 0x01) == 0);
}

// 正确：使用volatile
void wait_for_ready_good() {
    volatile unsigned int *status = (volatile unsigned int *)0x40021000;
    while ((*status & 0x01) == 0) {
        // 每次都会从内存读取
    }
}

// 中断共享变量
volatile int interrupt_flag = 0;

void ISR_handler() {
    interrupt_flag = 1;  // 中断中设置
}

void main_loop() {
    while (!interrupt_flag) {
        // 等待中断
    }
    printf("中断已触发\n");
    interrupt_flag = 0;
}
```

## 二、const与嵌入式

```c
// const放在不同位置的含义
const int *p1;          // 指向常量的指针（不能通过p1修改）
int const *p2;          // 同上
int *const p3 = NULL;   // 常量指针（p3不能指向别处）
const int *const p4 = NULL;  // 都不可变

// 嵌入式中的const：存放在Flash而非RAM
const int lookup_table[256] = {
    0, 1, 2, 3, /* ... */
    // 这些数据存在Flash中，节省RAM
};

// 嵌入式中常用：只读硬件寄存器
const volatile unsigned int *DEVICE_ID =
    (const volatile unsigned int *)0x40022000;
```

## 三、位操作与寄存器

```c
#include <stdint.h>

// 寄存器操作宏
#define REG(addr)           (*(volatile uint32_t *)(addr))
#define SET_BIT(reg, bit)   ((reg) |= (1U << (bit)))
#define CLR_BIT(reg, bit)   ((reg) &= ~(1U << (bit)))
#define GET_BIT(reg, bit)   (((reg) >> (bit)) & 1U)
#define TOG_BIT(reg, bit)   ((reg) ^= (1U << (bit)))

// 多位操作
#define SET_BITS(reg, mask, val) \
    ((reg) = ((reg) & ~(mask)) | ((val) & (mask)))

// GPIO示例
#define GPIOA_BASE    0x40020000
#define GPIOA_MODER   REG(GPIOA_BASE + 0x00)
#define GPIOA_OTYPER  REG(GPIOA_BASE + 0x04)
#define GPIOA_ODR     REG(GPIOA_BASE + 0x14)
#define GPIOA_IDR     REG(GPIOA_BASE + 0x10)

void gpio_init() {
    // 设置PA5为输出模式
    SET_BITS(GPIOA_MODER, 0x3 << (5 * 2), 0x1 << (5 * 2));

    // PA5输出高电平
    SET_BIT(GPIOA_ODR, 5);

    // PA5输出低电平
    CLR_BIT(GPIOA_ODR, 5);

    // 读取PA5输入
    int pin_state = GET_BIT(GPIOA_IDR, 5);
}
```

## 四、中断与ISR

```c
/*
 * 中断服务程序（ISR）注意事项：
 * 1. 尽量短小精悍
 * 2. 不能调用不可重入函数（如malloc, printf）
 * 3. 共享变量用volatile
 * 4. 使用临界区保护共享数据
 */

// 临界区保护（ARM Cortex-M示例）
static inline uint32_t enter_critical() {
    uint32_t primask;
    __asm volatile ("mrs %0, primask" : "=r"(primask));
    __asm volatile ("cpsid i");
    return primask;
}

static inline void exit_critical(uint32_t primask) {
    if (!(primask & 1)) {
        __asm volatile ("cpsie i");
    }
}

// 共享变量保护示例
volatile uint32_t tick_count = 0;

void SysTick_Handler() {  // ISR
    tick_count++;
}

uint32_t get_tick() {
    uint32_t primask = enter_critical();
    uint32_t ticks = tick_count;
    exit_critical(primask);
    return ticks;
}
```

## 六、端序处理

```c
#include <arpa/inet.h>

// 大小端转换
uint16_t swap16(uint16_t x) {
    return (x >> 8) | (x << 8);
}

uint32_t swap32(uint32_t x) {
    return ((x >> 24) & 0xFF) |
           ((x >> 8)  & 0xFF00) |
           ((x << 8)  & 0xFF0000) |
           ((x << 24) & 0xFF000000);
}

// 网络字节序（大端）转主机字节序
// 使用 ntohs(), ntohl(), htons(), htonl()

int is_little_endian() {
    union { uint32_t i; uint8_t c[4]; } test = {0x01020304};
    return test.c[0] == 0x04;
}
```

## 八、低功耗与优化

```c
// 嵌入式优化技巧
// 1. 查表代替计算
static const uint8_t bit_reverse_table[256] = {
    0x00, 0x80, 0x40, 0xC0, /* ... */
};

// 2. 位操作代替乘除
// x * 4  -> x << 2
// x / 8  -> x >> 3
// x % 8  -> x & 7

// 3. 使用register关键字（编译器通常忽略）
register int counter __asm__("r4");

// 4. 静态分配代替动态分配
// 嵌入式中避免malloc，使用静态数组或内存池

// 5. 减少函数调用开销
static inline int max(int a, int b) {
    return a > b ? a : b;
}
```
