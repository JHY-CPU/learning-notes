# 手撕代码 - 常用数据结构

## 一、动态数组（Vector）

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

typedef struct {
    int *data;
    size_t size;
    size_t capacity;
} Vector;

Vector *vec_create(size_t init_cap) {
    Vector *v = (Vector *)malloc(sizeof(Vector));
    v->capacity = init_cap > 0 ? init_cap : 4;
    v->size = 0;
    v->data = (int *)malloc(v->capacity * sizeof(int));
    return v;
}

void vec_push(Vector *v, int val) {
    if (v->size >= v->capacity) {
        v->capacity *= 2;
        v->data = (int *)realloc(v->data, v->capacity * sizeof(int));
    }
    v->data[v->size++] = val;
}

bool vec_pop(Vector *v, int *val) {
    if (v->size == 0) return false;
    *val = v->data[--v->size];
    return true;
}

int vec_get(Vector *v, size_t index) {
    return v->data[index];
}

void vec_set(Vector *v, size_t index, int val) {
    v->data[index] = val;
}

void vec_insert(Vector *v, size_t index, int val) {
    vec_push(v, 0);  // 确保有空间
    memmove(&v->data[index + 1], &v->data[index],
            (v->size - index - 1) * sizeof(int));
    v->data[index] = val;
}

void vec_remove(Vector *v, size_t index) {
    memmove(&v->data[index], &v->data[index + 1],
            (v->size - index - 1) * sizeof(int));
    v->size--;
}

void vec_free(Vector *v) {
    free(v->data);
    free(v);
}
```

## 二、双端队列（Deque）

```c
#define DEQUE_INIT_CAP 8

typedef struct {
    int *data;
    int capacity;
    int front;  // 指向第一个元素
    int size;
} Deque;

Deque *deque_create() {
    Deque *d = (Deque *)malloc(sizeof(Deque));
    d->capacity = DEQUE_INIT_CAP;
    d->data = (int *)malloc(d->capacity * sizeof(int));
    d->front = 0;
    d->size = 0;
    return d;
}

void deque_push_front(Deque *d, int val) {
    if (d->size >= d->capacity) {
        // 扩容并重新排列
        int new_cap = d->capacity * 2;
        int *new_data = (int *)malloc(new_cap * sizeof(int));
        for (int i = 0; i < d->size; i++) {
            new_data[i] = d->data[(d->front + i) % d->capacity];
        }
        free(d->data);
        d->data = new_data;
        d->capacity = new_cap;
        d->front = 0;
    }
    d->front = (d->front - 1 + d->capacity) % d->capacity;
    d->data[d->front] = val;
    d->size++;
}

void deque_push_back(Deque *d, int val) {
    if (d->size >= d->capacity) {
        int new_cap = d->capacity * 2;
        int *new_data = (int *)malloc(new_cap * sizeof(int));
        for (int i = 0; i < d->size; i++) {
            new_data[i] = d->data[(d->front + i) % d->capacity];
        }
        free(d->data);
        d->data = new_data;
        d->capacity = new_cap;
        d->front = 0;
    }
    int back = (d->front + d->size) % d->capacity;
    d->data[back] = val;
    d->size++;
}

bool deque_pop_front(Deque *d, int *val) {
    if (d->size == 0) return false;
    *val = d->data[d->front];
    d->front = (d->front + 1) % d->capacity;
    d->size--;
    return true;
}

bool deque_pop_back(Deque *d, int *val) {
    if (d->size == 0) return false;
    int back = (d->front + d->size - 1) % d->capacity;
    *val = d->data[back];
    d->size--;
    return true;
}

void deque_free(Deque *d) { free(d->data); free(d); }
```

## 三、有序集合（基于BST）

```c
typedef struct SetNode {
    int val;
    struct SetNode *left, *right;
} SetNode;

typedef struct {
    SetNode *root;
    int size;
} OrderedSet;

SetNode *set_new_node(int val) {
    SetNode *n = (SetNode *)malloc(sizeof(SetNode));
    n->val = val;
    n->left = n->right = NULL;
    return n;
}

bool set_insert(OrderedSet *s, int val) {
    if (!s->root) {
        s->root = set_new_node(val);
        s->size++;
        return true;
    }
    SetNode *curr = s->root, *parent = NULL;
    while (curr) {
        parent = curr;
        if (val < curr->val) curr = curr->left;
        else if (val > curr->val) curr = curr->right;
        else return false;  // 已存在
    }
    if (val < parent->val) parent->left = set_new_node(val);
    else parent->right = set_new_node(val);
    s->size++;
    return true;
}

bool set_contains(OrderedSet *s, int val) {
    SetNode *curr = s->root;
    while (curr) {
        if (val < curr->val) curr = curr->left;
        else if (val > curr->val) curr = curr->right;
        else return true;
    }
    return false;
}

void set_inorder(SetNode *node) {
    if (!node) return;
    set_inorder(node->left);
    printf("%d ", node->val);
    set_inorder(node->right);
}
```

## 四、双向链表

```c
typedef struct DNode {
    int data;
    struct DNode *prev, *next;
} DNode;

typedef struct {
    DNode *head, *tail;
    int size;
} DoublyList;

DoublyList *dlist_create() {
    DoublyList *list = (DoublyList *)malloc(sizeof(DoublyList));
    list->head = list->tail = NULL;
    list->size = 0;
    return list;
}

void dlist_push_back(DoublyList *list, int data) {
    DNode *node = (DNode *)malloc(sizeof(DNode));
    node->data = data;
    node->next = NULL;
    node->prev = list->tail;
    if (list->tail) list->tail->next = node;
    else list->head = node;
    list->tail = node;
    list->size++;
}

void dlist_push_front(DoublyList *list, int data) {
    DNode *node = (DNode *)malloc(sizeof(DNode));
    node->data = data;
    node->prev = NULL;
    node->next = list->head;
    if (list->head) list->head->prev = node;
    else list->tail = node;
    list->head = node;
    list->size++;
}

void dlist_print(DoublyList *list) {
    DNode *curr = list->head;
    while (curr) {
        printf("%d <-> ", curr->data);
        curr = curr->next;
    }
    printf("NULL\n");
}
```

## 五、并查集（Union-Find）

```c
#define MAXN 1000

int parent[MAXN], rank_[MAXN];

void uf_init(int n) {
    for (int i = 0; i < n; i++) {
        parent[i] = i;
        rank_[i] = 0;
    }
}

int uf_find(int x) {
    if (parent[x] != x)
        parent[x] = uf_find(parent[x]);  // 路径压缩
    return parent[x];
}

void uf_union(int x, int y) {
    int px = uf_find(x), py = uf_find(y);
    if (px == py) return;
    if (rank_[px] < rank_[py]) parent[px] = py;
    else if (rank_[px] > rank_[py]) parent[py] = px;
    else { parent[py] = px; rank_[px]++; }
}

bool uf_connected(int x, int y) {
    return uf_find(x) == uf_find(y);
}
```

## 六、字典树（Trie）

```c
#define ALPHA_SIZE 26

typedef struct TrieNode {
    struct TrieNode *children[ALPHA_SIZE];
    bool is_end;
} TrieNode;

TrieNode *trie_create() {
    TrieNode *node = (TrieNode *)calloc(1, sizeof(TrieNode));
    return node;
}

void trie_insert(TrieNode *root, const char *word) {
    TrieNode *curr = root;
    while (*word) {
        int idx = *word - 'a';
        if (!curr->children[idx])
            curr->children[idx] = trie_create();
        curr = curr->children[idx];
        word++;
    }
    curr->is_end = true;
}

bool trie_search(TrieNode *root, const char *word) {
    TrieNode *curr = root;
    while (*word) {
        int idx = *word - 'a';
        if (!curr->children[idx]) return false;
        curr = curr->children[idx];
        word++;
    }
    return curr->is_end;
}

bool trie_starts_with(TrieNode *root, const char *prefix) {
    TrieNode *curr = root;
    while (*prefix) {
        int idx = *prefix - 'a';
        if (!curr->children[idx]) return false;
        curr = curr->children[idx];
        prefix++;
    }
    return true;
}
```

## 测试

```c
int main(void) {
    // 动态数组
    Vector *v = vec_create(4);
    for (int i = 0; i < 10; i++) vec_push(v, i * 10);
    printf("Vector大小: %zu\n", v->size);
    for (size_t i = 0; i < v->size; i++) printf("%d ", vec_get(v, i));
    printf("\n");
    vec_free(v);

    // 并查集
    uf_init(10);
    uf_union(1, 2);
    uf_union(3, 4);
    uf_union(2, 4);
    printf("1和3连通: %d\n", uf_connected(1, 3));

    // Trie
    TrieNode *trie = trie_create();
    trie_insert(trie, "apple");
    trie_insert(trie, "app");
    printf("搜索apple: %d\n", trie_search(trie, "apple"));
    printf("搜索app: %d\n", trie_search(trie, "app"));
    printf("前缀ap: %d\n", trie_starts_with(trie, "ap"));

    return 0;
}
```
