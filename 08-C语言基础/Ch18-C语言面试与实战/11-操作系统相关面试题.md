# 操作系统相关面试题

## 一、进程与线程

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>

// 进程创建示例
void process_demo() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("fork失败");
        exit(1);
    } else if (pid == 0) {
        // 子进程
        printf("子进程: PID=%d, 父进程PID=%d\n", getpid(), getppid());
        exit(42);
    } else {
        // 父进程
        printf("父进程: PID=%d, 子进程PID=%d\n", getpid(), pid);
        int status;
        waitpid(pid, &status, 0);
        if (WIFEXITED(status)) {
            printf("子进程退出码: %d\n", WEXITSTATUS(status));
        }
    }
}
```

## 二、线程编程（POSIX）

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

#define NUM_THREADS 5

// 共享资源（需要同步）
int shared_counter = 0;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

void *thread_func(void *arg) {
    int thread_id = *(int *)arg;

    // 加锁保护共享资源
    pthread_mutex_lock(&mutex);
    shared_counter++;
    printf("线程 %d: counter = %d\n", thread_id, shared_counter);
    pthread_mutex_unlock(&mutex);

    return NULL;
}

void thread_demo() {
    pthread_t threads[NUM_THREADS];
    int thread_ids[NUM_THREADS];

    for (int i = 0; i < NUM_THREADS; i++) {
        thread_ids[i] = i;
        pthread_create(&threads[i], NULL, thread_func, &thread_ids[i]);
    }

    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    printf("最终counter = %d\n", shared_counter);
    pthread_mutex_destroy(&mutex);
}
```

## 三、进程间通信

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

// 管道通信
void pipe_demo() {
    int pipefd[2];
    if (pipe(pipefd) == -1) {
        perror("pipe");
        exit(1);
    }

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程读
        close(pipefd[1]);
        char buf[256];
        ssize_t n = read(pipefd[0], buf, sizeof(buf));
        if (n > 0) {
            buf[n] = '\0';
            printf("子进程收到: %s\n", buf);
        }
        close(pipefd[0]);
        exit(0);
    } else {
        // 父进程写
        close(pipefd[0]);
        const char *msg = "来自父进程的消息";
        write(pipefd[1], msg, strlen(msg));
        close(pipefd[1]);
        wait(NULL);
    }
}
```

## 四、信号量与互斥锁

```c
#include <pthread.h>
#include <semaphore.h>

// 信号量实现生产者消费者
#define BUFFER_SIZE 5
int buffer[BUFFER_SIZE];
int in = 0, out = 0;

sem_t empty;  // 空槽位计数
sem_t full;   // 满槽位计数
pthread_mutex_t buf_mutex = PTHREAD_MUTEX_INITIALIZER;

void *producer(void *arg) {
    for (int i = 0; i < 10; i++) {
        sem_wait(&empty);
        pthread_mutex_lock(&buf_mutex);

        buffer[in] = i;
        printf("生产: %d (位置%d)\n", i, in);
        in = (in + 1) % BUFFER_SIZE;

        pthread_mutex_unlock(&buf_mutex);
        sem_post(&full);
    }
    return NULL;
}

void *consumer(void *arg) {
    for (int i = 0; i < 10; i++) {
        sem_wait(&full);
        pthread_mutex_lock(&buf_mutex);

        int item = buffer[out];
        printf("消费: %d (位置%d)\n", item, out);
        out = (out + 1) % BUFFER_SIZE;

        pthread_mutex_unlock(&buf_mutex);
        sem_post(&empty);
    }
    return NULL;
}
```

## 五、死锁

```c
/*
死锁四个必要条件：
1. 互斥条件：资源一次只能被一个进程使用
2. 持有并等待：持有资源的同时等待其他资源
3. 不可剥夺：资源不能被强制剥夺
4. 循环等待：存在资源的循环等待链

预防策略：
- 破坏条件1：使用共享锁
- 破坏条件2：一次性申请所有资源
- 破坏条件3：允许资源被剥夺
- 破坏条件4：按序申请资源
*/

pthread_mutex_t lock_a = PTHREAD_MUTEX_INITIALIZER;
pthread_mutex_t lock_b = PTHREAD_MUTEX_INITIALIZER;

// 死锁示例（不要这样写！）
void *deadlock_thread1(void *arg) {
    pthread_mutex_lock(&lock_a);
    sleep(1);  // 等待另一个线程获取lock_b
    pthread_mutex_lock(&lock_b);  // 死锁！
    printf("Thread1 获取了两个锁\n");
    pthread_mutex_unlock(&lock_b);
    pthread_mutex_unlock(&lock_a);
    return NULL;
}

// 正确做法：按固定顺序获取锁
void *safe_thread(void *arg) {
    pthread_mutex_lock(&lock_a);
    pthread_mutex_lock(&lock_b);
    // 安全操作
    pthread_mutex_unlock(&lock_b);
    pthread_mutex_unlock(&lock_a);
    return NULL;
}
```

## 六、虚拟内存与页面置换

```c
/*
虚拟内存核心概念：
- 虚拟地址 -> 物理地址（通过页表映射）
- 页面置换算法：LRU、FIFO、Clock

LRU简化实现（用数组模拟）
*/
#include <stdio.h>

#define FRAME_COUNT 3

int is_in_frames(int frames[], int n, int page) {
    for (int i = 0; i < n; i++) {
        if (frames[i] == page) return i;
    }
    return -1;
}

void lru_page_replace(int pages[], int n) {
    int frames[FRAME_COUNT];
    int last_used[FRAME_COUNT];
    int faults = 0;
    int filled = 0;

    for (int i = 0; i < FRAME_COUNT; i++) {
        frames[i] = -1;
        last_used[i] = -1;
    }

    for (int i = 0; i < n; i++) {
        int idx = is_in_frames(frames, FRAME_COUNT, pages[i]);
        if (idx >= 0) {
            // 命中，更新使用时间
            last_used[idx] = i;
            printf("页面 %d: 命中\n", pages[i]);
        } else {
            // 缺页
            faults++;
            if (filled < FRAME_COUNT) {
                frames[filled] = pages[i];
                last_used[filled] = i;
                filled++;
            } else {
                // 找LRU页面替换
                int lru = 0;
                for (int j = 1; j < FRAME_COUNT; j++) {
                    if (last_used[j] < last_used[lru]) lru = j;
                }
                frames[lru] = pages[i];
                last_used[lru] = i;
            }
            printf("页面 %d: 缺页! [", pages[i]);
            for (int j = 0; j < FRAME_COUNT; j++)
                printf(" %d", frames[j]);
            printf(" ]\n");
        }
    }
    printf("总缺页次数: %d / %d\n", faults, n);
}
```

## 七、面试常问问题总结

```text
Q: 进程和线程的区别？
A: 进程是资源分配的基本单位，有独立地址空间；
   线程是调度的基本单位，共享进程地址空间。
   线程切换开销比进程小。

Q: 什么是上下文切换？
A: CPU从一个进程/线程切换到另一个的过程，
   需要保存和恢复寄存器、程序计数器、栈指针等。

Q: 什么是僵尸进程？如何处理？
A: 子进程退出后父进程未调用wait()，子进程变成僵尸进程。
   处理方法：父进程调用wait/waitpid，或捕获SIGCHLD信号。

Q: 进程间通信方式有哪些？
A: 管道、消息队列、共享内存、信号量、信号、socket

Q: 什么是协程？
A: 用户态的轻量级线程，由程序自身调度，切换开销更小。
*/
```
