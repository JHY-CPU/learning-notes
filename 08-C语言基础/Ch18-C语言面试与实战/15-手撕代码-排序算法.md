# 手撕代码 - 排序算法

## 一、快速排序

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void swap(int *a, int *b) {
    int t = *a; *a = *b; *b = t;
}

// 基础快排
int partition_basic(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (arr[j] <= pivot) {
            swap(&arr[++i], &arr[j]);
        }
    }
    swap(&arr[i + 1], &arr[high]);
    return i + 1;
}

void quick_sort_basic(int arr[], int low, int high) {
    if (low < high) {
        int pi = partition_basic(arr, low, high);
        quick_sort_basic(arr, low, pi - 1);
        quick_sort_basic(arr, pi + 1, high);
    }
}

// 三数取中优化
int partition_median(int arr[], int low, int high) {
    int mid = low + (high - low) / 2;
    if (arr[low] > arr[mid]) swap(&arr[low], &arr[mid]);
    if (arr[low] > arr[high]) swap(&arr[low], &arr[high]);
    if (arr[mid] > arr[high]) swap(&arr[mid], &arr[high]);
    swap(&arr[mid], &arr[high]);  // 中值作pivot
    return partition_basic(arr, low, high);
}

// 三路快排（处理大量重复元素）
void quick_sort_3way(int arr[], int low, int high) {
    if (low >= high) return;
    int lt = low, gt = high, i = low + 1;
    int pivot = arr[low];
    while (i <= gt) {
        if (arr[i] < pivot) swap(&arr[lt++], &arr[i++]);
        else if (arr[i] > pivot) swap(&arr[i], &arr[gt--]);
        else i++;
    }
    quick_sort_3way(arr, low, lt - 1);
    quick_sort_3way(arr, gt + 1, high);
}

// 非递归快排（用栈模拟）
void quick_sort_iterative(int arr[], int n) {
    int stack[1000];
    int top = -1;
    stack[++top] = 0;
    stack[++top] = n - 1;

    while (top >= 0) {
        int high = stack[top--];
        int low = stack[top--];
        int pi = partition_basic(arr, low, high);
        if (pi - 1 > low) {
            stack[++top] = low;
            stack[++top] = pi - 1;
        }
        if (pi + 1 < high) {
            stack[++top] = pi + 1;
            stack[++top] = high;
        }
    }
}
```

## 二、归并排序

```c
// 自顶向下递归
void merge(int arr[], int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    int L[n1], R[n2];
    for (int i = 0; i < n1; i++) L[i] = arr[l + i];
    for (int j = 0; j < n2; j++) R[j] = arr[m + 1 + j];

    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2)
        arr[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void merge_sort(int arr[], int l, int r) {
    if (l < r) {
        int m = l + (r - l) / 2;
        merge_sort(arr, l, m);
        merge_sort(arr, m + 1, r);
        merge(arr, l, m, r);
    }
}

// 链表归并排序
typedef struct ListNode {
    int val;
    struct ListNode *next;
} ListNode;

ListNode *find_middle(ListNode *head) {
    ListNode *slow = head, *fast = head->next;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;
}

ListNode *merge_list(ListNode *l1, ListNode *l2) {
    ListNode dummy = {0, NULL};
    ListNode *tail = &dummy;
    while (l1 && l2) {
        if (l1->val <= l2->val) {
            tail->next = l1; l1 = l1->next;
        } else {
            tail->next = l2; l2 = l2->next;
        }
        tail = tail->next;
    }
    tail->next = l1 ? l1 : l2;
    return dummy.next;
}

ListNode *merge_sort_list(ListNode *head) {
    if (!head || !head->next) return head;
    ListNode *mid = find_middle(head);
    ListNode *right = mid->next;
    mid->next = NULL;
    return merge_list(merge_sort_list(head), merge_sort_list(right));
}
```

## 三、堆排序

```c
void heapify(int arr[], int n, int i) {
    int largest = i, l = 2*i+1, r = 2*i+2;
    if (l < n && arr[l] > arr[largest]) largest = l;
    if (r < n && arr[r] > arr[largest]) largest = r;
    if (largest != i) {
        swap(&arr[i], &arr[largest]);
        heapify(arr, n, largest);
    }
}

void heap_sort(int arr[], int n) {
    for (int i = n/2 - 1; i >= 0; i--)
        heapify(arr, n, i);
    for (int i = n - 1; i > 0; i--) {
        swap(&arr[0], &arr[i]);
        heapify(arr, i, 0);
    }
}
```

## 四、插入排序

```c
void insertion_sort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i], j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j+1] = arr[j];
            j--;
        }
        arr[j+1] = key;
    }
}

// 希尔排序（插入排序的改进）
void shell_sort(int arr[], int n) {
    for (int gap = n/2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i++) {
            int key = arr[i], j = i;
            while (j >= gap && arr[j-gap] > key) {
                arr[j] = arr[j-gap];
                j -= gap;
            }
            arr[j] = key;
        }
    }
}
```

## 五、冒泡排序

```c
void bubble_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int swapped = 0;
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j+1]) {
                swap(&arr[j], &arr[j+1]);
                swapped = 1;
            }
        }
        if (!swapped) break;  // 优化：无交换则已有序
    }
}
```

## 六、选择排序

```c
void selection_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) min_idx = j;
        }
        if (min_idx != i) swap(&arr[i], &arr[min_idx]);
    }
}
```

## 七、计数排序

```c
void counting_sort(int arr[], int n, int max_val) {
    int *count = (int *)calloc(max_val + 1, sizeof(int));
    int *output = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) count[arr[i]]++;
    for (int i = 1; i <= max_val; i++) count[i] += count[i-1];
    for (int i = n - 1; i >= 0; i--) {
        output[count[arr[i]] - 1] = arr[i];
        count[arr[i]]--;
    }
    memcpy(arr, output, n * sizeof(int));
    free(count);
    free(output);
}
```

## 八、桶排序

```c
void bucket_sort(int arr[], int n, int max_val) {
    int num_buckets = 10;
    int bucket_size = (max_val + 1) / num_buckets + 1;
    int *buckets[10];
    int bucket_count[10] = {0};

    for (int i = 0; i < num_buckets; i++)
        buckets[i] = (int *)malloc(n * sizeof(int));

    for (int i = 0; i < n; i++) {
        int idx = arr[i] / bucket_size;
        buckets[idx][bucket_count[idx]++] = arr[i];
    }

    int k = 0;
    for (int i = 0; i < num_buckets; i++) {
        insertion_sort(buckets[i], bucket_count[i]);
        for (int j = 0; j < bucket_count[i]; j++)
            arr[k++] = buckets[i][j];
        free(buckets[i]);
    }
}
```

void print_arr(const char *name, int arr[], int n) {
    printf("%s: ", name);
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");
}

int main(void) {
    int arr[] = {64, 34, 25, 12, 22, 11, 90, 1};
    int n = 8;

    int tmp[8];
    #define TEST(name, fn) \
        memcpy(tmp, arr, sizeof(arr)); fn; print_arr(name, tmp, n);

    TEST("快排", quick_sort_basic(tmp, 0, n-1))
    TEST("归并", merge_sort(tmp, 0, n-1))
    TEST("堆排", heap_sort(tmp, n))
    TEST("插入", insertion_sort(tmp, n))
    TEST("希尔", shell_sort(tmp, n))
    TEST("冒泡", bubble_sort(tmp, n))
    TEST("选择", selection_sort(tmp, n))
    return 0;
}
