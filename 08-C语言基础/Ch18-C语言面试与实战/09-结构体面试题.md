# 结构体面试题

## 一、结构体大小计算

```c
#include <stdio.h>
#include <stddef.h>

// 规则1：每个成员的偏移量必须是其自身大小的整数倍
struct S1 { char a; int b; char c; };
// 布局: a[1] + pad[3] + b[4] + c[1] + pad[3] = 12

struct S2 { int b; char a; char c; };
// 布局: b[4] + a[1] + c[1] + pad[2] = 8

struct S3 { double d; char c; int i; };
// 布局: d[8] + c[1] + pad[3] + i[4] = 16

// 规则2：整个结构体的大小必须是最大成员大小的整数倍
struct S4 { char a; double b; };
// 布局: a[1] + pad[7] + b[8] = 16

void struct_size_demo() {
    printf("S1: %zu (预期12)\n", sizeof(struct S1));
    printf("S2: %zu (预期8)\n", sizeof(struct S2));
    printf("S3: %zu (预期16)\n", sizeof(struct S3));
    printf("S4: %zu (预期16)\n", sizeof(struct S4));
}
```

## 二、结构体嵌套

```c
struct Inner {
    char c;
    int x;
};  // 大小8

struct Outer {
    char a;
    struct Inner inner;  // 嵌套结构体按最大成员对齐
    char b;
};  // a[1]+pad[3]+inner[8]+b[1]+pad[3] = 16

void nested_struct_demo() {
    printf("Inner: %zu\n", sizeof(struct Inner));   // 8
    printf("Outer: %zu\n", sizeof(struct Outer));   // 16
}
```

## 三、位域（Bit Fields）

```c
#include <stdio.h>

// 位域：精确控制每个字段占用的位数
struct Flags {
    unsigned int is_active  : 1;  // 1位，0或1
    unsigned int is_admin   : 1;
    unsigned int role       : 3;  // 3位，0-7
    unsigned int level      : 4;  // 4位，0-15
    unsigned int reserved   : 7;  // 保留位
};  // 共16位 = 2字节，但考虑对齐可能是4字节

// TCP标志位的典型应用
struct TCPFlags {
    unsigned int fin : 1;
    unsigned int syn : 1;
    unsigned int rst : 1;
    unsigned int psh : 1;
    unsigned int ack : 1;
    unsigned int urg : 1;
    unsigned int ece : 1;
    unsigned int cwr : 1;
};

void bitfield_demo() {
    struct Flags f = {0};
    f.is_active = 1;
    f.is_admin = 0;
    f.role = 5;
    f.level = 10;

    printf("Flags大小: %zu\n", sizeof(struct Flags));
    printf("active=%u, admin=%u, role=%u, level=%u\n",
           f.is_active, f.is_admin, f.role, f.level);

    struct TCPFlags tcp = {0};
    tcp.syn = 1;
    tcp.ack = 1;
    printf("TCP Flags大小: %zu\n", sizeof(struct TCPFlags));
    printf("SYN=%u, ACK=%u\n", tcp.syn, tcp.ack);
}
```

## 四、结构体对齐控制

```c
#include <stddef.h>

#pragma pack(push, 1)  // 1字节对齐
struct Packed1 {
    char a;
    int b;
    char c;
};  // 大小6
#pragma pack(pop)

#pragma pack(push, 2)  // 2字节对齐
struct Packed2 {
    char a;
    int b;
    char c;
};  // 大小8
#pragma pack(pop)

// 使用alignas (C11)
struct Aligned {
    alignas(16) int x;  // x地址对齐到16字节边界
    char c;
};

void alignment_control_demo() {
    printf("pack(1): %zu\n", sizeof(struct Packed1));  // 6
    printf("pack(2): %zu\n", sizeof(struct Packed2));  // 8
    printf("默认: %zu\n", sizeof(struct S1));           // 12
}
```

## 五、结构体指针与动态分配

```c
#include <stdlib.h>
#include <string.h>

typedef struct {
    int id;
    char *name;  // 动态字符串
    double score;
} Student;

Student *create_student(int id, const char *name, double score) {
    Student *s = (Student *)malloc(sizeof(Student));
    if (!s) return NULL;
    s->id = id;
    s->name = (char *)malloc(strlen(name) + 1);
    if (s->name) strcpy(s->name, name);
    s->score = score;
    return s;
}

void free_student(Student *s) {
    if (s) {
        free(s->name);
        free(s);
    }
}
```

## 六、结构体数组排序

```c
#include <stdlib.h>
#include <string.h>

typedef struct {
    char name[32];
    int age;
    double salary;
} Employee;

int compare_by_salary(const void *a, const void *b) {
    double diff = ((Employee *)a)->salary - ((Employee *)b)->salary;
    return (diff > 0) - (diff < 0);
}

int compare_by_name(const void *a, const void *b) {
    return strcmp(((Employee *)a)->name, ((Employee *)b)->name);
}

void struct_sort_demo() {
    Employee emps[] = {
        {"张三", 30, 8000},
        {"李四", 25, 12000},
        {"王五", 35, 9500}
    };
    int n = 3;

    qsort(emps, n, sizeof(Employee), compare_by_salary);
    printf("按薪资排序:\n");
    for (int i = 0; i < n; i++) {
        printf("  %s: %.0f\n", emps[i].name, emps[i].salary);
    }
}
```

## 七、联合体（Union）

```c
#include <stdio.h>

// 所有成员共享同一块内存
union Data {
    int i;
    float f;
    char str[20];
};

// 联合体常用于变体类型
typedef struct {
    int type;  // 0=int, 1=float, 2=string
    union {
        int int_val;
        float float_val;
        char str_val[64];
    } data;
} Variant;

void union_demo() {
    union Data d;
    d.i = 10;
    printf("i = %d\n", d.i);
    d.f = 3.14f;
    printf("f = %f\n", d.f);
    printf("此时i = %d (无意义)\n", d.i);
    printf("Union大小: %zu\n", sizeof(union Data));

    // 使用联合体判断大小端
    union {
        int i;
        char c[4];
    } endian_test = {0x01020304};
    printf("字节序: %s\n",
           endian_test.c[0] == 0x04 ? "小端" : "大端");
}
```

## 八、offsetof宏

```c
#include <stddef.h>

struct Record {
    char type;
    int id;
    double value;
    char name[16];
};

void offset_demo() {
    printf("offsetof(type)  = %zu\n", offsetof(struct Record, type));
    printf("offsetof(id)    = %zu\n", offsetof(struct Record, id));
    printf("offsetof(value) = %zu\n", offsetof(struct Record, value));
    printf("offsetof(name)  = %zu\n", offsetof(struct Record, name));
    printf("结构体总大小      = %zu\n", sizeof(struct Record));
}
```

## 九、结构体作为函数参数

```c
// 值传递：会拷贝整个结构体，大型结构体效率低
void print_by_value(Student s) {
    printf("ID: %d, Score: %.1f\n", s.id, s.score);
}

// 指针传递：推荐，只传递8字节指针
void print_by_ptr(const Student *s) {
    printf("ID: %d, Score: %.1f\n", s->id, s->score);
}

// 返回结构体指针
Student *find_student(Student arr[], int n, int id) {
    for (int i = 0; i < n; i++) {
        if (arr[i].id == id) return &arr[i];
    }
    return NULL;
}
```
