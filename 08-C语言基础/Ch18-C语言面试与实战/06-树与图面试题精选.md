## 基础结构定义

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left, *right;
} TreeNode;

TreeNode *new_node(int val) {
    TreeNode *node = (TreeNode *)malloc(sizeof(TreeNode));
    node->val = val;
    node->left = node->right = NULL;
    return node;
}
```

## 题目1：二叉树前中后序遍历

```c
// 前序遍历（根-左-右）
void preorder(TreeNode *root) {
    if (!root) return;
    printf("%d ", root->val);
    preorder(root->left);
    preorder(root->right);
}

// 中序遍历（左-根-右）
void inorder(TreeNode *root) {
    if (!root) return;
    inorder(root->left);
    printf("%d ", root->val);
    inorder(root->right);
}

// 后序遍历（左-右-根）
void postorder(TreeNode *root) {
    if (!root) return;
    postorder(root->left);
    postorder(root->right);
    printf("%d ", root->val);
}
```

void inorder_iterative(TreeNode *root) {
    TreeNode *stack[100];
    int top = -1;
    TreeNode *curr = root;

    while (curr || top >= 0) {
        while (curr) {
            stack[++top] = curr;
            curr = curr->left;
        }
        curr = stack[top--];
        printf("%d ", curr->val);
        curr = curr->right;
    }
}

## 题目3：层序遍历（BFS）

```c
void level_order(TreeNode *root) {
    if (!root) return;
    TreeNode *queue[200];
    int front = 0, rear = 0;
    queue[rear++] = root;

    while (front < rear) {
        int level_size = rear - front;
        for (int i = 0; i < level_size; i++) {
            TreeNode *node = queue[front++];
            printf("%d ", node->val);
            if (node->left) queue[rear++] = node->left;
            if (node->right) queue[rear++] = node->right;
        }
        printf("\n");  // 每层一行
    }
}
```

## 题目4：二叉树最大深度

```c
int max_depth(TreeNode *root) {
    if (!root) return 0;
    int left = max_depth(root->left);
    int right = max_depth(root->right);
    return (left > right ? left : right) + 1;
}
```

## 题目5：判断对称二叉树

```c
bool is_mirror(TreeNode *t1, TreeNode *t2) {
    if (!t1 && !t2) return true;
    if (!t1 || !t2) return false;
    return (t1->val == t2->val)
        && is_mirror(t1->left, t2->right)
        && is_mirror(t1->right, t2->left);
}

bool is_symmetric(TreeNode *root) {
    return is_mirror(root, root);
}
```

## 题目7：最近公共祖先（LCA）

```c
// 普通二叉树的LCA
TreeNode *lowest_common_ancestor(TreeNode *root, TreeNode *p, TreeNode *q) {
    if (!root || root == p || root == q) return root;
    TreeNode *left = lowest_common_ancestor(root->left, p, q);
    TreeNode *right = lowest_common_ancestor(root->right, p, q);
    if (left && right) return root;
    return left ? left : right;
}

// BST的LCA（利用有序性质）
TreeNode *lca_bst(TreeNode *root, TreeNode *p, TreeNode *q) {
    while (root) {
        if (p->val < root->val && q->val < root->val)
            root = root->left;
        else if (p->val > root->val && q->val > root->val)
            root = root->right;
        else
            return root;
    }
    return NULL;
}
```

## 题目8：二叉树的序列化与反序列化

```c
void serialize_helper(TreeNode *root, char *buf, int *pos) {
    if (!root) {
        *pos += sprintf(buf + *pos, "#,");
        return;
    }
    *pos += sprintf(buf + *pos, "%d,", root->val);
    serialize_helper(root->left, buf, pos);
    serialize_helper(root->right, buf, pos);
}

char *serialize(TreeNode *root) {
    char *buf = (char *)malloc(4096);
    int pos = 0;
    serialize_helper(root, buf, &pos);
    return buf;
}

TreeNode *deserialize_helper(const char **data) {
    if (**data == '#') {
        (*data) += 2;  // skip "#,"
        return NULL;
    }
    int val = 0;
    while (**data != ',') {
        val = val * 10 + (**data - '0');
        (*data)++;
    }
    (*data)++;  // skip ','
    TreeNode *node = new_node(val);
    node->left = deserialize_helper(data);
    node->right = deserialize_helper(data);
    return node;
}

TreeNode *deserialize(const char *data) {
    return deserialize_helper(&data);
}
```

## 题目9：路径总和

```c
bool has_path_sum(TreeNode *root, int target) {
    if (!root) return false;
    if (!root->left && !root->right) {
        return root->val == target;
    }
    int remain = target - root->val;
    return has_path_sum(root->left, remain)
        || has_path_sum(root->right, remain);
}
```

## 题目11：DFS图遍历

```c
#define MAX_V 100

typedef struct {
    int adj[MAX_V][MAX_V];
    int V;
} Graph;

bool visited[MAX_V];

void dfs(Graph *g, int v) {
    visited[v] = true;
    printf("%d ", v);
    for (int i = 0; i < g->V; i++) {
        if (g->adj[v][i] && !visited[i]) {
            dfs(g, i);
        }
    }
}
```

## 题目12：BFS图遍历

```c
void bfs(Graph *g, int start) {
    bool visited[MAX_V] = {false};
    int queue[MAX_V], front = 0, rear = 0;

    visited[start] = true;
    queue[rear++] = start;

    while (front < rear) {
        int v = queue[front++];
        printf("%d ", v);
        for (int i = 0; i < g->V; i++) {
            if (g->adj[v][i] && !visited[i]) {
                visited[i] = true;
                queue[rear++] = i;
            }
        }
    }
}
```
