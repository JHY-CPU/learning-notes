# 数据结构基础题

## 一、栈的实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

#define MAX_STACK 1000

// 顺序栈
typedef struct {
    int data[MAX_STACK];
    int top;
} Stack;

void stack_init(Stack *s) { s->top = -1; }
bool stack_empty(Stack *s) { return s->top == -1; }
bool stack_full(Stack *s) { return s->top == MAX_STACK - 1; }

bool stack_push(Stack *s, int val) {
    if (stack_full(s)) return false;
    s->data[++s->top] = val;
    return true;
}

bool stack_pop(Stack *s, int *val) {
    if (stack_empty(s)) return false;
    *val = s->data[s->top--];
    return true;
}

bool stack_peek(Stack *s, int *val) {
    if (stack_empty(s)) return false;
    *val = s->data[s->top];
    return true;
}

// 链栈
typedef struct StackNode {
    int data;
    struct StackNode *next;
} StackNode;

typedef struct {
    StackNode *top;
    int size;
} LinkedStack;

void linked_stack_push(LinkedStack *s, int val) {
    StackNode *node = (StackNode *)malloc(sizeof(StackNode));
    node->data = val;
    node->next = s->top;
    s->top = node;
    s->size++;
}

bool linked_stack_pop(LinkedStack *s, int *val) {
    if (!s->top) return false;
    StackNode *temp = s->top;
    *val = temp->data;
    s->top = temp->next;
    free(temp);
    s->size--;
    return true;
}
```

## 二、队列的实现

```c
// 循环队列
#define MAX_QUEUE 1000

typedef struct {
    int data[MAX_QUEUE];
    int front, rear, size;
} Queue;

void queue_init(Queue *q) {
    q->front = q->rear = q->size = 0;
}

bool queue_empty(Queue *q) { return q->size == 0; }
bool queue_full(Queue *q) { return q->size == MAX_QUEUE; }

bool queue_enqueue(Queue *q, int val) {
    if (queue_full(q)) return false;
    q->data[q->rear] = val;
    q->rear = (q->rear + 1) % MAX_QUEUE;
    q->size++;
    return true;
}

bool queue_dequeue(Queue *q, int *val) {
    if (queue_empty(q)) return false;
    *val = q->data[q->front];
    q->front = (q->front + 1) % MAX_QUEUE;
    q->size--;
    return true;
}

// 用两个栈实现队列
typedef struct {
    Stack s1;  // 入队栈
    Stack s2;  // 出队栈
} QueueFromStack;

void qfs_enqueue(QueueFromStack *q, int val) {
    stack_push(&q->s1, val);
}

bool qfs_dequeue(QueueFromStack *q, int *val) {
    if (stack_empty(&q->s2)) {
        while (!stack_empty(&q->s1)) {
            int v;
            stack_pop(&q->s1, &v);
            stack_push(&q->s2, v);
        }
    }
    return stack_pop(&q->s2, val);
}
```

## 三、哈希表

```c
#include <string.h>

#define HASH_SIZE 97

typedef struct HashNode {
    char *key;
    int value;
    struct HashNode *next;
} HashNode;

typedef struct {
    HashNode *buckets[HASH_SIZE];
} HashMap;

unsigned int hash_func(const char *key) {
    unsigned int hash = 5381;
    while (*key) {
        hash = ((hash << 5) + hash) + (unsigned char)*key++;
    }
    return hash % HASH_SIZE;
}

void hashmap_put(HashMap *map, const char *key, int value) {
    unsigned int idx = hash_func(key);
    // 查找是否已存在
    HashNode *node = map->buckets[idx];
    while (node) {
        if (strcmp(node->key, key) == 0) {
            node->value = value;
            return;
        }
        node = node->next;
    }
    // 新建节点插入头部
    HashNode *new_node = (HashNode *)malloc(sizeof(HashNode));
    new_node->key = strdup(key);
    new_node->value = value;
    new_node->next = map->buckets[idx];
    map->buckets[idx] = new_node;
}

bool hashmap_get(HashMap *map, const char *key, int *value) {
    unsigned int idx = hash_func(key);
    HashNode *node = map->buckets[idx];
    while (node) {
        if (strcmp(node->key, key) == 0) {
            *value = node->value;
            return true;
        }
        node = node->next;
    }
    return false;
}

void hashmap_remove(HashMap *map, const char *key) {
    unsigned int idx = hash_func(key);
    HashNode *node = map->buckets[idx], *prev = NULL;
    while (node) {
        if (strcmp(node->key, key) == 0) {
            if (prev) prev->next = node->next;
            else map->buckets[idx] = node->next;
            free(node->key);
            free(node);
            return;
        }
        prev = node;
        node = node->next;
    }
}
```

## 四、堆（优先队列）

```c
// 最小堆
typedef struct {
    int *data;
    int size;
    int capacity;
} MinHeap;

MinHeap *heap_create(int capacity) {
    MinHeap *h = (MinHeap *)malloc(sizeof(MinHeap));
    h->data = (int *)malloc(capacity * sizeof(int));
    h->size = 0;
    h->capacity = capacity;
    return h;
}

void heap_swap(int *a, int *b) {
    int t = *a; *a = *b; *b = t;
}

void heap_sift_up(MinHeap *h, int i) {
    while (i > 0) {
        int parent = (i - 1) / 2;
        if (h->data[parent] <= h->data[i]) break;
        heap_swap(&h->data[parent], &h->data[i]);
        i = parent;
    }
}

void heap_sift_down(MinHeap *h, int i) {
    while (2 * i + 1 < h->size) {
        int child = 2 * i + 1;
        if (child + 1 < h->size &&
            h->data[child + 1] < h->data[child]) child++;
        if (h->data[i] <= h->data[child]) break;
        heap_swap(&h->data[i], &h->data[child]);
        i = child;
    }
}

void heap_push(MinHeap *h, int val) {
    if (h->size >= h->capacity) return;
    h->data[h->size] = val;
    heap_sift_up(h, h->size);
    h->size++;
}

bool heap_pop(MinHeap *h, int *val) {
    if (h->size == 0) return false;
    *val = h->data[0];
    h->data[0] = h->data[--h->size];
    heap_sift_down(h, 0);
    return true;
}

int heap_peek(MinHeap *h) { return h->data[0]; }
```

## 五、用栈实现括号匹配

```c
bool is_valid_parentheses(const char *s) {
    char stack[1000];
    int top = -1;

    for (int i = 0; s[i]; i++) {
        char c = s[i];
        if (c == '(' || c == '[' || c == '{') {
            stack[++top] = c;
        } else {
            if (top < 0) return false;
            char open = stack[top--];
            if ((c == ')' && open != '(') ||
                (c == ']' && open != '[') ||
                (c == '}' && open != '{')) {
                return false;
            }
        }
    }
    return top == -1;
}
```

## 六、单调栈

```c
// 找每个元素右边第一个比它大的元素
void next_greater(int arr[], int n, int result[]) {
    int stack[1000], top = -1;

    for (int i = n - 1; i >= 0; i--) {
        while (top >= 0 && arr[stack[top]] <= arr[i]) top--;
        result[i] = (top >= 0) ? arr[stack[top]] : -1;
        stack[++top] = i;
    }
}
```

## 七、前缀和

```c
// 一维前缀和
typedef struct {
    int *prefix;
    int size;
} PrefixSum;

PrefixSum *prefix_sum_create(int arr[], int n) {
    PrefixSum *ps = (PrefixSum *)malloc(sizeof(PrefixSum));
    ps->prefix = (int *)malloc((n + 1) * sizeof(int));
    ps->size = n;
    ps->prefix[0] = 0;
    for (int i = 0; i < n; i++) {
        ps->prefix[i + 1] = ps->prefix[i] + arr[i];
    }
    return ps;
}

// 查询区间 [left, right] 的和
int prefix_sum_query(PrefixSum *ps, int left, int right) {
    return ps->prefix[right + 1] - ps->prefix[left];
}
```
