## 三、双指针技巧

```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>

// 快慢指针：链表环检测、找中点
// 对撞指针：有序数组两数之和、回文判断
// 滑动窗口：子串/子数组问题

// 例1：有序数组两数之和
bool two_sum(int *arr, int n, int target, int *i, int *j) {
    int left = 0, right = n - 1;
    while (left < right) {
        int sum = arr[left] + arr[right];
        if (sum == target) { *i = left; *j = right; return true; }
        else if (sum < target) left++;
        else right--;
    }
    return false;
}

// 例2：最长无重复子串
int length_of_longest_substring(const char *s) {
    int last[256] = {0};
    int max_len = 0, start = 0;
    for (int i = 0; s[i]; i++) {
        if (last[(unsigned char)s[i]] > start)
            start = last[(unsigned char)s[i]];
        last[(unsigned char)s[i]] = i + 1;
        int len = i - start + 1;
        if (len > max_len) max_len = len;
    }
    return max_len;
}

// 例3：接雨水
int trap(int *height, int n) {
    int left = 0, right = n - 1;
    int left_max = 0, right_max = 0, water = 0;
    while (left < right) {
        if (height[left] < height[right]) {
            if (height[left] >= left_max) left_max = height[left];
            else water += left_max - height[left];
            left++;
        } else {
            if (height[right] >= right_max) right_max = height[right];
            else water += right_max - height[right];
            right--;
        }
    }
    return water;
}
```

## 四、滑动窗口

```c
// 最小覆盖子串
#include <string.h>

void min_window(const char *s, const char *t, char *result) {
    int need[256] = {0}, window[256] = {0};
    int t_len = strlen(t);
    for (int i = 0; t[i]; i++) need[(unsigned char)t[i]]++;

    int left = 0, count = 0, min_len = strlen(s) + 1, start = 0;
    for (int right = 0; s[right]; right++) {
        unsigned char rc = s[right];
        window[rc]++;
        if (need[rc] > 0 && window[rc] <= need[rc]) count++;

        while (count == t_len) {
            if (right - left + 1 < min_len) {
                min_len = right - left + 1;
                start = left;
            }
            unsigned char lc = s[left++];
            window[lc]--;
            if (need[lc] > 0 && window[lc] < need[lc]) count--;
        }
    }

    if (min_len <= (int)strlen(s)) {
        strncpy(result, s + start, min_len);
        result[min_len] = '\0';
    } else {
        result[0] = '\0';
    }
}
```

## 五、动态规划

```c
// 例1：最长递增子序列
int lis(int *arr, int n) {
    int dp[n];
    for (int i = 0; i < n; i++) dp[i] = 1;
    int max_len = 1;
    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (arr[j] < arr[i] && dp[j] + 1 > dp[i])
                dp[i] = dp[j] + 1;
        }
        if (dp[i] > max_len) max_len = dp[i];
    }
    return max_len;
}

// 例2：背包问题
int knapsack(int *weights, int *values, int n, int capacity) {
    int dp[capacity + 1];
    memset(dp, 0, sizeof(dp));
    for (int i = 0; i < n; i++) {
        for (int w = capacity; w >= weights[i]; w--) {
            if (dp[w - weights[i]] + values[i] > dp[w])
                dp[w] = dp[w - weights[i]] + values[i];
        }
    }
    return dp[capacity];
}

// 例3：编辑距离
int edit_distance(const char *s1, const char *s2) {
    int m = strlen(s1), n = strlen(s2);
    int dp[m + 1][n + 1];
    for (int i = 0; i <= m; i++) dp[i][0] = i;
    for (int j = 0; j <= n; j++) dp[0][j] = j;
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (s1[i-1] == s2[j-1])
                dp[i][j] = dp[i-1][j-1];
            else
                dp[i][j] = 1 + (dp[i-1][j] < dp[i][j-1] ?
                    (dp[i-1][j] < dp[i-1][j-1] ? dp[i-1][j] : dp[i-1][j-1]) :
                    (dp[i][j-1] < dp[i-1][j-1] ? dp[i][j-1] : dp[i-1][j-1]));
        }
    }
    return dp[m][n];
}
```

## 六、回溯法

```c
// 全排列
void permute(int *arr, int n, int depth, int *used, int *current) {
    if (depth == n) {
        for (int i = 0; i < n; i++) printf("%d ", current[i]);
        printf("\n");
        return;
    }
    for (int i = 0; i < n; i++) {
        if (!used[i]) {
            used[i] = 1;
            current[depth] = arr[i];
            permute(arr, n, depth + 1, used, current);
            used[i] = 0;
        }
    }
}

// N皇后
bool is_safe(int *queens, int row, int col) {
    for (int i = 0; i < row; i++) {
        if (queens[i] == col || abs(queens[i] - col) == abs(i - row))
            return false;
    }
    return true;
}

void solve_n_queens(int *queens, int n, int row, int *count) {
    if (row == n) { (*count)++; return; }
    for (int col = 0; col < n; col++) {
        if (is_safe(queens, row, col)) {
            queens[row] = col;
            solve_n_queens(queens, n, row + 1, count);
        }
    }
}

// 旋转数组查找
int search_rotated(int *arr, int n, int target) {
    int lo = 0, hi = n - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        if (arr[mid] == target) return mid;
        if (arr[lo] <= arr[mid]) {
            if (target >= arr[lo] && target < arr[mid]) hi = mid - 1;
            else lo = mid + 1;
        } else {
            if (target > arr[mid] && target <= arr[hi]) lo = mid + 1;
            else hi = mid - 1;
        }
    }
    return -1;
}
```
