## 一、命名规范

```c
#include <stdio.h>
```

## 二、函数设计原则

```c
// 原则1：单一职责
// 差：一个函数做太多事情
void process_data_bad(int *data, int n) {
    // 验证 + 排序 + 打印 + 保存
}

// 好：拆分成小函数
bool validate_data(const int *data, int n) {
    if (!data || n <= 0) return false;
    for (int i = 0; i < n; i++) {
        if (data[i] < 0) return false;
    }
    return true;
}

void sort_data(int *data, int n) {
    qsort(data, n, sizeof(int),
          (int (*)(const void *, const void *))(
              // 比较函数
              (int (*)(const int *, const int *))0
          ));
}

void print_data(const int *data, int n) {
    for (int i = 0; i < n; i++) printf("%d ", data[i]);
    printf("\n");
}

// 原则2：参数检查
int safe_divide(int a, int b, int *result) {
    if (!result) return -1;
    if (b == 0) return -2;
    *result = a / b;
    return 0;  // 成功
}

// 原则3：不要超过3-4个参数
// 参数太多时用结构体
typedef struct {
    const char *host;
    int port;
    int timeout;
    bool use_ssl;
} ConnectConfig;

int connect_server(const ConnectConfig *config);
```

## 三、错误处理模式

```c
#include <errno.h>

// 模式1：返回错误码
typedef enum {
    SUCCESS = 0,
    ERR_NULL_PTR = -1,
    ERR_NO_MEMORY = -2,
    ERR_INVALID_ARG = -3,
    ERR_IO = -4,
} ErrorCode;

ErrorCode process(const char *input, char **output) {
    if (!input || !output) return ERR_NULL_PTR;

    *output = (char *)malloc(strlen(input) + 1);
    if (!*output) return ERR_NO_MEMORY;

    strcpy(*output, input);
    return SUCCESS;
}

// 模式2：goto清理（Linux内核风格）
ErrorCode read_and_process(const char *filename, char **data) {
    ErrorCode ret = SUCCESS;
    FILE *fp = NULL;
    char *buf = NULL;

    if (!filename || !data) {
        ret = ERR_NULL_PTR;
        goto cleanup;
    }

    fp = fopen(filename, "r");
    if (!fp) {
        ret = ERR_IO;
        goto cleanup;
    }

    buf = (char *)malloc(4096);
    if (!buf) {
        ret = ERR_NO_MEMORY;
        goto cleanup;
    }

    if (!fgets(buf, 4096, fp)) {
        ret = ERR_IO;
        goto cleanup;
    }

    *data = buf;
    buf = NULL;  // 防止free

cleanup:
    free(buf);
    if (fp) fclose(fp);
    return ret;
}
```

## 五、代码重构示例

```c
// 重构前：面条代码
int process_old(int *arr, int n) {
    int sum = 0, count = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) {
            sum += arr[i];
            count++;
        }
    }
    if (count == 0) return 0;
    int avg = sum / count;
    int max_val = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max_val) max_val = arr[i];
    }
    printf("avg=%d, max=%d\n", avg, max_val);
    return avg;
}

// 重构后：模块化
typedef struct {
    int sum;
    int count;
    int max_val;
    int min_val;
} Stats;

Stats compute_stats(const int *arr, int n) {
    Stats s = {0, 0, arr[0], arr[0]};
    for (int i = 0; i < n; i++) {
        s.sum += arr[i];
        s.count++;
        if (arr[i] > s.max_val) s.max_val = arr[i];
        if (arr[i] < s.min_val) s.min_val = arr[i];
    }
    return s;
}

void print_stats(const Stats *s) {
    double avg = s->count > 0 ? (double)s->sum / s->count : 0;
    printf("avg=%.2f, max=%d, min=%d\n", avg, s->max_val, s->min_val);
}
```

## 六、宏的规范用法

```c
// 宏定义安全规范
// 1. 参数加括号
#define MAX(a, b) ((a) > (b) ? (a) : (b))

// 2. 用do-while(0)包裹多语句宏
#define SWAP(a, b) do { \
    typeof(a) _tmp = (a); \
    (a) = (b); \
    (b) = _tmp; \
} while (0)

// 3. 避免有副作用的参数
// 好：int result = MAX(x, y);
// 差：int result = MAX(x++, y++);

// 4. 条件编译保护头文件
#ifndef MY_MODULE_H
#define MY_MODULE_H
// ...
#endif
```
