# 手撕代码 - 链表操作

## 基础结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

Node *create_node(int data) {
    Node *node = (Node *)malloc(sizeof(Node));
    node->data = data;
    node->next = NULL;
    return node;
}

void print_list(Node *head) {
    while (head) {
        printf("%d -> ", head->data);
        head = head->next;
    }
    printf("NULL\n");
}

void free_list(Node *head) {
    while (head) {
        Node *tmp = head;
        head = head->next;
        free(tmp);
    }
}
```

## 一、链表创建与插入

```c
// 尾插法创建
Node *create_from_array(int arr[], int n) {
    Node dummy = {0, NULL};
    Node *tail = &dummy;
    for (int i = 0; i < n; i++) {
        tail->next = create_node(arr[i]);
        tail = tail->next;
    }
    return dummy.next;
}

// 在指定位置插入（0-indexed）
Node *insert_at(Node *head, int pos, int data) {
    Node *new_node = create_node(data);
    if (pos == 0) {
        new_node->next = head;
        return new_node;
    }
    Node *curr = head;
    for (int i = 0; i < pos - 1 && curr; i++) curr = curr->next;
    if (!curr) { free(new_node); return head; }
    new_node->next = curr->next;
    curr->next = new_node;
    return head;
}

// 有序插入
Node *insert_sorted(Node *head, int data) {
    Node *new_node = create_node(data);
    if (!head || data <= head->data) {
        new_node->next = head;
        return new_node;
    }
    Node *curr = head;
    while (curr->next && curr->next->data < data)
        curr = curr->next;
    new_node->next = curr->next;
    curr->next = new_node;
    return head;
}
```

## 二、链表删除

```c
// 删除第一个值为val的节点
Node *delete_value(Node *head, int val) {
    if (!head) return NULL;
    if (head->data == val) {
        Node *tmp = head->next;
        free(head);
        return tmp;
    }
    Node *curr = head;
    while (curr->next && curr->next->data != val)
        curr = curr->next;
    if (curr->next) {
        Node *tmp = curr->next;
        curr->next = tmp->next;
        free(tmp);
    }
    return head;
}

// 删除所有值为val的节点
Node *delete_all_value(Node *head, int val) {
    Node dummy = {0, head};
    Node *curr = &dummy;
    while (curr->next) {
        if (curr->next->data == val) {
            Node *tmp = curr->next;
            curr->next = tmp->next;
            free(tmp);
        } else {
            curr = curr->next;
        }
    }
    return dummy.next;
}

// 删除第pos个节点（0-indexed）
Node *delete_at(Node *head, int pos) {
    if (!head) return NULL;
    if (pos == 0) {
        Node *tmp = head->next;
        free(head);
        return tmp;
    }
    Node *curr = head;
    for (int i = 0; i < pos - 1 && curr->next; i++) curr = curr->next;
    if (curr->next) {
        Node *tmp = curr->next;
        curr->next = tmp->next;
        free(tmp);
    }
    return head;
}
```

## 三、链表反转

```c
// 迭代反转
Node *reverse_iterative(Node *head) {
    Node *prev = NULL, *curr = head, *next;
    while (curr) {
        next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
    }
    return prev;
}

// 递归反转
Node *reverse_recursive(Node *head) {
    if (!head || !head->next) return head;
    Node *new_head = reverse_recursive(head->next);
    head->next->next = head;
    head->next = NULL;
    return new_head;
}

// 反转第m到第n个节点（1-indexed）
Node *reverse_between(Node *head, int m, int n) {
    Node dummy = {0, head};
    Node *pre = &dummy;
    for (int i = 1; i < m; i++) pre = pre->next;
    Node *curr = pre->next;
    for (int i = 0; i < n - m; i++) {
        Node *tmp = curr->next;
        curr->next = tmp->next;
        tmp->next = pre->next;
        pre->next = tmp;
    }
    return dummy.next;
}
```

## 四、链表合并

```c
// 合并两个有序链表（迭代）
Node *merge_sorted(Node *l1, Node *l2) {
    Node dummy = {0, NULL};
    Node *tail = &dummy;
    while (l1 && l2) {
        if (l1->data <= l2->data) {
            tail->next = l1; l1 = l1->next;
        } else {
            tail->next = l2; l2 = l2->next;
        }
        tail = tail->next;
    }
    tail->next = l1 ? l1 : l2;
    return dummy.next;
}
```

## 五、环检测与处理

```c
// 快慢指针检测环
bool has_cycle(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) return true;
    }
    return false;
}

// 找环入口
Node *find_cycle_entry(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) {
            Node *entry = head;
            while (entry != slow) {
                entry = entry->next;
                slow = slow->next;
            }
            return entry;
        }
    }
    return NULL;
}

// 求环长度
int cycle_length(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) {
            int len = 1;
            Node *p = slow->next;
            while (p != slow) { p = p->next; len++; }
            return len;
        }
    }
    return 0;
}
```

```c
// 归并排序（链表）
Node *sort_list(Node *head) {
    if (!head || !head->next) return head;
    Node *slow = head, *fast = head->next;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    Node *mid = slow->next;
    slow->next = NULL;
    return merge_sorted(sort_list(head), sort_list(mid));
}
```

```c
Node *find_intersection(Node *a, Node *b) {
    Node *p1 = a, *p2 = b;
    while (p1 != p2) {
        p1 = p1 ? p1->next : b;
        p2 = p2 ? p2->next : a;
    }
    return p1;
}
```

## 八、回文链表判断

```c
bool is_palindrome_list(Node *head) {
    if (!head || !head->next) return true;
    // 找中点
    Node *slow = head, *fast = head;
    while (fast->next && fast->next->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    // 反转后半部分
    Node *prev = NULL, *curr = slow->next;
    while (curr) {
        Node *next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
    }
    // 比较
    Node *p1 = head, *p2 = prev;
    while (p2) {
        if (p1->data != p2->data) return false;
        p1 = p1->next;
        p2 = p2->next;
    }
    return true;
}
```

```c
Node *add_two_numbers(Node *l1, Node *l2) {
    Node dummy = {0, NULL};
    Node *curr = &dummy;
    int carry = 0;
    while (l1 || l2 || carry) {
        int sum = carry;
        if (l1) { sum += l1->data; l1 = l1->next; }
        if (l2) { sum += l2->data; l2 = l2->next; }
        carry = sum / 10;
        curr->next = create_node(sum % 10);
        curr = curr->next;
    }
    return dummy.next;
}
```

## 十、K个一组反转

```c
Node *reverse_k_group(Node *head, int k) {
    // 检查是否有k个节点
    Node *check = head;
    for (int i = 0; i < k; i++) {
        if (!check) return head;
        check = check->next;
    }
    // 反转前k个
    Node *prev = NULL, *curr = head, *next;
    for (int i = 0; i < k; i++) {
        next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
    }
    head->next = reverse_k_group(curr, k);
    return prev;
}
```
