# 网络编程面试题

## 一、TCP三次握手与四次挥手

```c
/*
三次握手（建立连接）：
客户端 -> SYN=1, seq=x           -> 服务器  (第一次握手)
客户端 <- SYN=1, ACK=1, seq=y, ack=x+1 <- 服务器  (第二次握手)
客户端 -> ACK=1, seq=x+1, ack=y+1 -> 服务器  (第三次握手)

为什么需要三次握手？
防止已失效的连接请求到达服务端，造成资源浪费。
两次握手无法确认客户端的接收能力。

四次挥手（断开连接）：
客户端 -> FIN=1, seq=u           -> 服务器  (第一次挥手)
客户端 <- ACK=1, ack=u+1         <- 服务器  (第二次挥手)
客户端 <- FIN=1, seq=w           <- 服务器  (第三次挥手)
客户端 -> ACK=1, ack=w+1         -> 服务器  (第四次挥手)

为什么四次挥手？
TCP是全双工的，每个方向需要单独关闭。
服务器可能还有数据要发送，所以ACK和FIN分开。
*/
```

## 二、TCP与UDP对比

```c
/*
| 特性        | TCP              | UDP              |
|-----------|------------------|------------------|
| 连接        | 面向连接          | 无连接            |
| 可靠性      | 可靠传输          | 尽最大努力交付     |

*/
```

## 三、TCP服务器代码

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

void tcp_server_demo() {
    int server_fd, client_fd;
    struct sockaddr_in server_addr, client_addr;
    socklen_t client_len = sizeof(client_addr);
    char buffer[1024];

    // 1. 创建socket
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) {
        perror("socket");
        exit(1);
    }

    // 设置SO_REUSEADDR
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // 2. 绑定地址
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(8080);

    if (bind(server_fd, (struct sockaddr *)&server_addr,
             sizeof(server_addr)) < 0) {
        perror("bind");
        exit(1);
    }

    // 3. 监听
    if (listen(server_fd, 5) < 0) {
        perror("listen");
        exit(1);
    }
    printf("服务器监听端口 8080...\n");

    // 4. 接受连接
    client_fd = accept(server_fd, (struct sockaddr *)&client_addr,
                       &client_len);
    if (client_fd < 0) {
        perror("accept");
        exit(1);
    }
    printf("客户端已连接: %s:%d\n",
           inet_ntoa(client_addr.sin_addr), ntohs(client_addr.sin_port));

    // 5. 收发数据
    ssize_t n = recv(client_fd, buffer, sizeof(buffer) - 1, 0);
    if (n > 0) {
        buffer[n] = '\0';
        printf("收到: %s\n", buffer);
        send(client_fd, "Hello from server", 17, 0);
    }

    close(client_fd);
    close(server_fd);
}
```

## 四、TCP客户端代码

```c
void tcp_client_demo() {
    int sock_fd;
    struct sockaddr_in server_addr;
    char buffer[1024];

    // 1. 创建socket
    sock_fd = socket(AF_INET, SOCK_STREAM, 0);

    // 2. 连接服务器
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(8080);
    inet_pton(AF_INET, "127.0.0.1", &server_addr.sin_addr);

    if (connect(sock_fd, (struct sockaddr *)&server_addr,
                sizeof(server_addr)) < 0) {
        perror("connect");
        close(sock_fd);
        return;
    }

    // 3. 收发数据
    send(sock_fd, "Hello from client", 17, 0);
    ssize_t n = recv(sock_fd, buffer, sizeof(buffer) - 1, 0);
    if (n > 0) {
        buffer[n] = '\0';
        printf("收到: %s\n", buffer);
    }

    close(sock_fd);
}
```

## 五、UDP通信

```c
void udp_server_demo() {
    int sock_fd;
    struct sockaddr_in server_addr, client_addr;
    char buffer[1024];
    socklen_t addr_len = sizeof(client_addr);

    sock_fd = socket(AF_INET, SOCK_DGRAM, 0);

    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(9090);

    bind(sock_fd, (struct sockaddr *)&server_addr, sizeof(server_addr));

    printf("UDP服务器监听端口 9090...\n");
    ssize_t n = recvfrom(sock_fd, buffer, sizeof(buffer) - 1, 0,
                         (struct sockaddr *)&client_addr, &addr_len);
    if (n > 0) {
        buffer[n] = '\0';
        printf("UDP收到: %s\n", buffer);
        sendto(sock_fd, "UDP reply", 9, 0,
               (struct sockaddr *)&client_addr, addr_len);
    }
    close(sock_fd);
}
```

## 六、IO多路复用

```c
#include <sys/select.h>

void select_demo(int listen_fd) {
    fd_set read_fds, master;
    int max_fd = listen_fd;
    int client_fds[FD_SETSIZE];
    int num_clients = 0;

    FD_ZERO(&master);
    FD_SET(listen_fd, &master);

    for (int i = 0; i < FD_SETSIZE; i++) client_fds[i] = -1;

    while (1) {
        read_fds = master;
        int ready = select(max_fd + 1, &read_fds, NULL, NULL, NULL);

        if (FD_ISSET(listen_fd, &read_fds)) {
            // 新连接
            struct sockaddr_in client_addr;
            socklen_t len = sizeof(client_addr);
            int client_fd = accept(listen_fd,
                                   (struct sockaddr *)&client_addr, &len);
            if (client_fd >= 0) {
                FD_SET(client_fd, &master);
                if (client_fd > max_fd) max_fd = client_fd;
                printf("新连接: fd=%d\n", client_fd);
            }
        }

        // 检查已连接客户端
        for (int fd = 0; fd <= max_fd; fd++) {
            if (fd != listen_fd && FD_ISSET(fd, &read_fds)) {
                char buf[1024];
                ssize_t n = recv(fd, buf, sizeof(buf), 0);
                if (n <= 0) {
                    close(fd);
                    FD_CLR(fd, &master);
                    printf("连接关闭: fd=%d\n", fd);
                } else {
                    send(fd, buf, n, 0);  // echo
                }
            }
        }
    }
}
```

## 七、HTTP请求解析

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    char method[8];
    char path[256];
    char version[16];
} HTTPRequest;

int parse_http_request(const char *raw, HTTPRequest *req) {
    // 解析 "GET /path HTTP/1.1"
    if (sscanf(raw, "%7s %255s %15s",
               req->method, req->path, req->version) != 3) {
        return -1;
    }
    return 0;
}

void send_http_response(int fd, int status,
                        const char *content_type, const char *body) {
    char header[512];
    int len = snprintf(header, sizeof(header),
        "HTTP/1.1 %d OK\r\n"
        "Content-Type: %s\r\n"
        "Content-Length: %zu\r\n"
        "Connection: close\r\n"
        "\r\n",
        status, content_type, strlen(body));
    send(fd, header, len, 0);
    send(fd, body, strlen(body), 0);
}
```

## 八、面试常问问题

```text
Q: 为什么TIME_WAIT要等待2MSL？
A: 1) 确保最后的ACK能到达对方
   2) 确保旧连接的延迟报文不会干扰新连接

Q: 什么是粘包问题？如何解决？
A: TCP是流式协议，没有消息边界。
   解决方法：1) 固定长度 2) 分隔符 3) 长度前缀

Q: select/poll/epoll区别？
A: select: 有FD_SETSIZE限制（1024），每次需重新设置fd_set
   poll: 无数量限制，但需遍历所有fd
   epoll: 无数量限制，事件驱动，O(1)复杂度

Q: 什么是Nagle算法？
A: 将小数据包合并后再发送，减少网络中的小包数量。
   可通过TCP_NODELAY禁用。

Q: 如何处理高并发？
A: 多进程/多线程/IO多路复用/异步IO/协程
*/
```
