## 题目1：实现strlen

```c
#include <stdio.h>
#include <stddef.h>

// 方法1：计数器法
size_t my_strlen1(const char *str) {
    size_t count = 0;
    while (*str++ != '\0') {
        count++;
    }
    return count;
}

// 方法2：指针相减法
size_t my_strlen2(const char *str) {
    const char *p = str;
    while (*p != '\0') {
        p++;
    }
    return (size_t)(p - str);
}

// 方法3：递归法（面试加分）
size_t my_strlen3(const char *str) {
    if (*str == '\0') return 0;
    return 1 + my_strlen3(str + 1);
}

int main(void) {
    const char *test = "Hello, World!";
    printf("strlen1: %zu\n", my_strlen1(test));
    printf("strlen2: %zu\n", my_strlen2(test));
    printf("strlen3: %zu\n", my_strlen3(test));
    return 0;
}
```

## 题目2：实现strcpy

```c
#include <stdio.h>

// 标准strcpy实现
char *my_strcpy(char *dest, const char *src) {
    if (dest == NULL || src == NULL) return NULL;
    char *ret = dest;
    while ((*dest++ = *src++) != '\0');
    return ret;
}

// 带长度检查的安全版本
char *my_strncpy(char *dest, const char *src, size_t n) {
    if (dest == NULL || src == NULL || n == 0) return NULL;
    char *ret = dest;
    while (n-- > 0 && (*dest++ = *src++) != '\0');
    // 填充剩余空间为'\0'
    while (n-- > 0) *dest++ = '\0';
    return ret;
}

int main(void) {
    char dest[50];
    my_strcpy(dest, "Hello");
    printf("strcpy: %s\n", dest);

    char dest2[10];
    my_strncpy(dest2, "Hello World", sizeof(dest2) - 1);
    dest2[sizeof(dest2) - 1] = '\0';
    printf("strncpy: %s\n", dest2);
    return 0;
}
```

## 题目3：实现strcat

```c
#include <stdio.h>

char *my_strcat(char *dest, const char *src) {
    if (dest == NULL || src == NULL) return NULL;
    char *ret = dest;
    // 找到dest的末尾
    while (*dest != '\0') dest++;
    // 拷贝src
    while ((*dest++ = *src++) != '\0');
    return ret;
}

// 安全的strncat
char *my_strncat(char *dest, const char *src,
                 size_t dest_size, size_t n) {
    size_t dest_len = 0;
    while (dest[dest_len] != '\0') dest_len++;
    if (dest_len >= dest_size - 1) return dest;

    size_t i = 0;
    while (i < n && src[i] != '\0' && dest_len + i < dest_size - 1) {
        dest[dest_len + i] = src[i];
        i++;
    }
    dest[dest_len + i] = '\0';
    return dest;
}

int main(void) {
    char buf[50] = "Hello";
    my_strcat(buf, ", World!");
    printf("strcat: %s\n", buf);

    char buf2[20] = "Hello";
    my_strncat(buf2, ", World!!", sizeof(buf2), 5);
    printf("strncat: %s\n", buf2);
    return 0;
}
```

## 题目4：实现strcmp

```c
#include <stdio.h>

int my_strcmp(const char *s1, const char *s2) {
    while (*s1 && (*s1 == *s2)) {
        s1++;
        s2++;
    }
    return (unsigned char)*s1 - (unsigned char)*s2;
}

// 带长度限制的比较
int my_strncmp(const char *s1, const char *s2, size_t n) {
    while (n-- > 0) {
        if (*s1 != *s2) {
            return (unsigned char)*s1 - (unsigned char)*s2;
        }
        if (*s1 == '\0') return 0;
        s1++;
        s2++;
    }
    return 0;
}

int main(void) {
    printf("strcmp(\"abc\",\"abc\") = %d\n", my_strcmp("abc", "abc"));   // 0
    printf("strcmp(\"abc\",\"abd\") = %d\n", my_strcmp("abc", "abd"));   // -1
    printf("strcmp(\"abd\",\"abc\") = %d\n", my_strcmp("abd", "abc"));   // 1
    printf("strncmp(\"abcdef\",\"abcxyz\",3) = %d\n",
           my_strncmp("abcdef", "abcxyz", 3));  // 0
    return 0;
}
```

## 题目5：字符串反转

```c
#include <stdio.h>
#include <string.h>

void reverse_string(char *str) {
    if (str == NULL) return;
    int len = strlen(str);
    for (int i = 0; i < len / 2; i++) {
        char temp = str[i];
        str[i] = str[len - 1 - i];
        str[len - 1 - i] = temp;
    }
}

// 双指针法
void reverse_string2(char *str) {
    if (str == NULL) return;
    char *left = str;
    char *right = str + strlen(str) - 1;
    while (left < right) {
        char temp = *left;
        *left++ = *right;
        *right-- = temp;
    }
}

int main(void) {
    char s1[] = "Hello World";
    reverse_string(s1);
    printf("反转1: %s\n", s1);  // dlroW olleH

    char s2[] = "abcdef";
    reverse_string2(s2);
    printf("反转2: %s\n", s2);  // fedcba
    return 0;
}
```

## 题目6：判断回文串

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

int is_palindrome(const char *str) {
    int left = 0;
    int right = strlen(str) - 1;
    while (left < right) {
        if (str[left] != str[right]) return 0;
        left++;
        right--;
    }
    return 1;
}

// 忽略大小写和非字母字符
int is_palindrome_alnum(const char *str) {
    int left = 0;
    int right = strlen(str) - 1;
    while (left < right) {
        while (left < right && !isalnum(str[left])) left++;
        while (left < right && !isalnum(str[right])) right--;
        if (tolower(str[left]) != tolower(str[right])) return 0;
        left++;
        right--;
    }
    return 1;
}

int main(void) {
    printf("racecar: %d\n", is_palindrome("racecar"));       // 1
    printf("hello: %d\n", is_palindrome("hello"));           // 0
    printf("A man, a plan, Panama: %d\n",
           is_palindrome_alnum("A man, a plan, Panama"));     // 回文检查
    return 0;
}
```

## 题目7：字符串中第一个不重复字符

```c
#include <stdio.h>
#include <string.h>

char first_unique_char(const char *str) {
    int freq[256] = {0};
    int len = strlen(str);

    // 统计频率
    for (int i = 0; i < len; i++) {
        freq[(unsigned char)str[i]]++;
    }

    // 找第一个频率为1的字符
    for (int i = 0; i < len; i++) {
        if (freq[(unsigned char)str[i]] == 1) {
            return str[i];
        }
    }
    return '\0';
}

int main(void) {
    const char *tests[] = {"aabbc", "abcabc", "xxyz", "hello"};
    for (int i = 0; i < 4; i++) {
        char c = first_unique_char(tests[i]);
        printf("\"%s\" -> '%c' (index: %ld)\n",
               tests[i], c ? c : '0',
               c ? strchr(tests[i], c) - tests[i] : -1);
    }
    return 0;
}
```

## 题目8：实现strstr

```c
#include <stdio.h>
#include <string.h>

// 暴力匹配 O(m*n)
char *my_strstr(const char *haystack, const char *needle) {
    if (*needle == '\0') return (char *)haystack;

    while (*haystack) {
        const char *h = haystack;
        const char *n = needle;
        while (*h && *n && *h == *n) {
            h++;
            n++;
        }
        if (*n == '\0') return (char *)haystack;
        haystack++;
    }
    return NULL;
}

int main(void) {
    const char *text = "Hello World Programming";
    const char *pattern = "World";

    char *result = my_strstr(text, pattern);
    if (result) {
        printf("找到子串: %s\n", result);
        printf("位置: %ld\n", result - text);
    }

    result = my_strstr(text, "xyz");
    printf("找不到: %s\n", result ? "找到" : "NULL");
    return 0;
}
```

## 题目9：字符串分割

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// 自定义字符串分割
int split_string(const char *str, char delimiter,
                 char tokens[][64], int max_tokens) {
    int count = 0;
    const char *start = str;
    const char *p = str;

    while (*p && count < max_tokens) {
        if (*p == delimiter) {
            int len = p - start;
            if (len > 0) {
                strncpy(tokens[count], start, len);
                tokens[count][len] = '\0';
                count++;
            }
            start = p + 1;
        }
        p++;
    }
    // 处理最后一个token
    if (*start && count < max_tokens) {
        strcpy(tokens[count], start);
        count++;
    }
    return count;
}

int main(void) {
    char input[] = "one,two,three,four,five";
    char tokens[10][64];

    int n = split_string(input, ',', tokens, 10);
    printf("分割结果(%d个):\n", n);
    for (int i = 0; i < n; i++) {
        printf("  [%d] \"%s\"\n", i, tokens[i]);
    }
    return 0;
}
```

## 题目10：字符串转整数（atoi实现）

```c
#include <stdio.h>
#include <ctype.h>
#include <limits.h>

int my_atoi(const char *str) {
    // 跳过前导空白
    while (isspace(*str)) str++;

    // 处理正负号
    int sign = 1;
    if (*str == '+' || *str == '-') {
        sign = (*str == '-') ? -1 : 1;
        str++;
    }

    // 转换数字，注意溢出
    long result = 0;
    while (isdigit(*str)) {
        result = result * 10 + (*str - '0');
        // 溢出检查
        if (sign == 1 && result > INT_MAX) return INT_MAX;
        if (sign == -1 && -result < INT_MIN) return INT_MIN;
        str++;
    }
    return (int)(sign * result);
}

int main(void) {
    printf("atoi(\"42\") = %d\n", my_atoi("42"));
    printf("atoi(\"-42\") = %d\n", my_atoi("-42"));
    printf("atoi(\"   123abc\") = %d\n", my_atoi("   123abc"));
    printf("atoi(\"2147483647\") = %d\n", my_atoi("2147483647"));
    printf("atoi(\"-2147483648\") = %d\n", my_atoi("-2147483648"));
    return 0;
}
```

## 题目11：去除字符串首尾空格

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

char *trim(char *str) {
    if (str == NULL) return NULL;

    // 去除尾部空格
    char *end = str + strlen(str) - 1;
    while (end >= str && isspace(*end)) {
        *end = '\0';
        end--;
    }

    // 去除前导空格
    char *start = str;
    while (*start && isspace(*start)) start++;

    // 移动到开头
    if (start != str) {
        memmove(str, start, strlen(start) + 1);
    }
    return str;
}

int main(void) {
    char s1[] = "  Hello World  ";
    printf("原: \"%s\"\n", s1);
    printf("去: \"%s\"\n", trim(s1));

    char s2[] = "\t tab \n";
    printf("原: \"%s\"\n", s2);
    printf("去: \"%s\"\n", trim(s2));
    return 0;
}
```

## 题目12：判断两个字符串是否为变位词

```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>

bool is_anagram(const char *s1, const char *s2) {
    int freq[256] = {0};

    // 统计s1中每个字符
    while (*s1) freq[(unsigned char)*s1++]++;
    // 减去s2中每个字符
    while (*s2) freq[(unsigned char)*s2++]--;

    // 检查是否全部归零
    for (int i = 0; i < 256; i++) {
        if (freq[i] != 0) return false;
    }
    return true;
}

int main(void) {
    printf("listen/silent: %s\n",
           is_anagram("listen", "silent") ? "是变位词" : "不是");
    printf("hello/world: %s\n",
           is_anagram("hello", "world") ? "是变位词" : "不是");
    printf("anagram/nagaram: %s\n",
           is_anagram("anagram", "nagaram") ? "是变位词" : "不是");
    return 0;
}
```

## 题目13：最长公共前缀

```c
#include <stdio.h>
#include <string.h>

char *longest_common_prefix(char *strs[], int n, char *result) {
    if (n <= 0) {
        result[0] = '\0';
        return result;
    }

    strcpy(result, strs[0]);
    for (int i = 1; i < n; i++) {
        int j = 0;
        while (result[j] && strs[i][j] && result[j] == strs[i][j]) {
            j++;
        }
        result[j] = '\0';
        if (result[0] == '\0') break;
    }
    return result;
}

int main(void) {
    char *strs1[] = {"flower", "flow", "flight"};
    char prefix[256];
    longest_common_prefix(strs1, 3, prefix);
    printf("公共前缀: \"%s\"\n", prefix);  // "fl"

    char *strs2[] = {"dog", "racecar", "car"};
    longest_common_prefix(strs2, 3, prefix);
    printf("公共前缀: \"%s\"\n", prefix);  // ""
    return 0;
}
```

## 题目14：字符串压缩

```c
#include <stdio.h>
#include <string.h>

// "aabcccccaaa" -> "a2b1c5a3"
int compress_string(char *str) {
    int len = strlen(str);
    if (len <= 1) return len;

    char buf[256];
    int idx = 0;
    int count = 1;

    for (int i = 1; i <= len; i++) {
        if (i < len && str[i] == str[i - 1]) {
            count++;
        } else {
            buf[idx++] = str[i - 1];
            idx += sprintf(buf + idx, "%d", count);
            count = 1;
        }
    }
    buf[idx] = '\0';

    if (idx < len) {
        strcpy(str, buf);
    }
    return (idx < len) ? idx : len;
}

int main(void) {
    char s1[] = "aabcccccaaa";
    int new_len = compress_string(s1);
    printf("压缩后: \"%s\" (长度: %d)\n", s1, new_len);

    char s2[] = "abc";
    new_len = compress_string(s2);
    printf("不压缩: \"%s\" (长度: %d)\n", s2, new_len);
    return 0;
}
```

## 题目15：实现memcpy与memmove

```c
#include <stdio.h>
#include <string.h>

// memcpy不处理内存重叠
void *my_memcpy(void *dest, const void *src, size_t n) {
    char *d = (char *)dest;
    const char *s = (const char *)src;
    while (n--) {
        *d++ = *s++;
    }
    return dest;
}

// memmove处理内存重叠（从后往前拷贝）
void *my_memmove(void *dest, const void *src, size_t n) {
    char *d = (char *)dest;
    const char *s = (const char *)src;
    if (d < s) {
        // 源在前，从前向后拷贝
        while (n--) *d++ = *s++;
    } else {
        // 目标在前（重叠），从后向前拷贝
        d += n;
        s += n;
        while (n--) *--d = *--s;
    }
    return dest;
}

int main(void) {
    // memcpy示例
    int src[] = {1, 2, 3, 4, 5};
    int dst[5];
    my_memcpy(dst, src, sizeof(src));
    printf("memcpy: ");
    for (int i = 0; i < 5; i++) printf("%d ", dst[i]);
    printf("\n");

    // memmove示例（重叠情况）
    char buf[] = "Hello World";
    my_memmove(buf, buf + 6, 5);  // "World World"
    buf[5] = '\0';
    printf("memmove: %s\n", buf);
    return 0;
}
```

## 题目16：字符串匹配KMP算法

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

void compute_lps(const char *pattern, int *lps, int m) {
    int len = 0;
    lps[0] = 0;
    int i = 1;
    while (i < m) {
        if (pattern[i] == pattern[len]) {
            lps[i++] = ++len;
        } else if (len) {
            len = lps[len - 1];
        } else {
            lps[i++] = 0;
        }
    }
}

int kmp_search(const char *text, const char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    if (m == 0) return 0;

    int *lps = (int *)malloc(m * sizeof(int));
    compute_lps(pattern, lps, m);

    int i = 0, j = 0;
    while (i < n) {
        if (text[i] == pattern[j]) {
            i++; j++;
        }
        if (j == m) {
            free(lps);
            return i - j;  // 找到匹配
        } else if (i < n && text[i] != pattern[j]) {
            j ? (j = lps[j - 1]) : i++;
        }
    }
    free(lps);
    return -1;  // 未找到
}

int main(void) {
    const char *text = "ABABDABACDABABCABAB";
    const char *pattern = "ABABCABAB";
    int pos = kmp_search(text, pattern);
    printf("KMP匹配位置: %d\n", pos);
    return 0;
}
```

## 题目17：字符串全排列

```c
#include <stdio.h>
#include <string.h>

void swap(char *a, char *b) {
    char t = *a; *a = *b; *b = t;
}

void permute(char *str, int left, int right) {
    if (left == right) {
        printf("%s\n", str);
        return;
    }
    for (int i = left; i <= right; i++) {
        swap(&str[left], &str[i]);
        permute(str, left + 1, right);
        swap(&str[left], &str[i]);  // 回溯
    }
}

int main(void) {
    char str[] = "ABC";
    printf("字符串 \"%s\" 的全排列:\n", str);
    permute(str, 0, strlen(str) - 1);
    return 0;
}
```

## 题目18：整数转字符串

```c
#include <stdio.h>
#include <stdlib.h>

void int_to_string(int num, char *str) {
    int i = 0;
    int is_negative = 0;

    if (num == 0) {
        str[0] = '0';
        str[1] = '\0';
        return;
    }

    if (num < 0) {
        is_negative = 1;
        num = -num;
    }

    // 提取每一位（逆序）
    while (num > 0) {
        str[i++] = '0' + (num % 10);
        num /= 10;
    }

    if (is_negative) str[i++] = '-';
    str[i] = '\0';

    // 反转
    for (int j = 0; j < i / 2; j++) {
        char temp = str[j];
        str[j] = str[i - 1 - j];
        str[i - 1 - j] = temp;
    }
}

int main(void) {
    char buf[32];
    int_to_string(12345, buf);
    printf("itoa(12345) = \"%s\"\n", buf);
    int_to_string(-9876, buf);
    printf("itoa(-9876) = \"%s\"\n", buf);
    int_to_string(0, buf);
    printf("itoa(0) = \"%s\"\n", buf);
    return 0;
}
```
