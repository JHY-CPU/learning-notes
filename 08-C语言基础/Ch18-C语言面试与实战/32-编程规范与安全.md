## 一、安全编码原则


## 二、缓冲区溢出防护

```c
// 危险函数（不要使用）
// strcpy, strcat, sprintf, gets, scanf("%s", buf)

// 安全替代
void buffer_overflow_protection() {
    char dest[64];
    const char *src = "Hello World";

    // 1. 使用带长度限制的函数
    strncpy(dest, src, sizeof(dest) - 1);
    dest[sizeof(dest) - 1] = '\0';

    // 2. 使用snprintf
    snprintf(dest, sizeof(dest), "Value: %d", 42);

    // 3. 自定义安全拷贝
    // safe_strcpy(dest, src, sizeof(dest));

    // 4. 动态分配时检查长度
    size_t src_len = strlen(src);
    if (src_len < SIZE_MAX) {
        char *dynamic = (char *)malloc(src_len + 1);
        if (dynamic) {
            memcpy(dynamic, src, src_len + 1);
            free(dynamic);
        }
    }
}
```

## 三、整数溢出防护

```c
#include <stdint.h>

// 安全加法
bool safe_add_uint32(uint32_t a, uint32_t b, uint32_t *result) {
    if (a > UINT32_MAX - b) return false;
    *result = a + b;
    return true;
}

bool safe_mul_uint32(uint32_t a, uint32_t b, uint32_t *result) {
    if (b != 0 && a > UINT32_MAX / b) return false;
    *result = a * b;
    return true;
}

// 安全分配：检查乘法溢出
void *safe_malloc_array(size_t count, size_t size) {
    if (size != 0 && count > SIZE_MAX / size) return NULL;
    return malloc(count * size);
}

void integer_overflow_demo() {
    uint32_t result;
    if (safe_add_uint32(UINT32_MAX, 1, &result))
        printf("结果: %u\n", result);
    else
        printf("加法溢出！\n");

    void *arr = safe_malloc_array(100, sizeof(int));
    if (arr) {
        printf("安全分配成功\n");
        free(arr);
    }
}
```

## 五、输入验证

```c
#include <ctype.h>
#include <limits.h>

// 验证整数输入
bool validate_int(const char *str, int *out) {
    if (!str || !*str) return false;

    char *end;
    errno = 0;
    long val = strtol(str, &end, 10);

    if (*end != '\0') return false;  // 有非数字字符
    if (errno == ERANGE) return false;  // 溢出
    if (val < INT_MIN || val > INT_MAX) return false;

    *out = (int)val;
    return true;
}

// 验证IP地址
bool validate_ip(const char *ip) {
    int a, b, c, d;
    if (sscanf(ip, "%d.%d.%d.%d", &a, &b, &c, &d) != 4) return false;
    return a >= 0 && a <= 255 && b >= 0 && b <= 255 &&
           c >= 0 && c <= 255 && d >= 0 && d <= 255;
}

// 验证SQL输入（防注入，简化版）
bool validate_sql_input(const char *input) {
    // 检查危险字符
    const char *dangerous = "';\"--/";
    for (int i = 0; input[i]; i++) {
        if (strchr(dangerous, input[i])) return false;
    }
    return true;
}

// 参数边界检查
int process_index(int index, int array_size) {
    if (index < 0 || index >= array_size) {
        fprintf(stderr, "索引越界: %d (大小: %d)\n", index, array_size);
        return -1;
    }
    return index;
}
```

## 六、内存安全

```c
// 使用后置空
#define SAFE_FREE(ptr) do { free(ptr); (ptr) = NULL; } while(0)

// 内存清零（敏感数据）
void secure_memzero(void *ptr, size_t len) {
    volatile unsigned char *p = (volatile unsigned char *)ptr;
    while (len--) *p++ = 0;
}

// 防止double free
typedef struct {
    void *ptr;
    bool freed;
} SafePointer;

void safe_free(SafePointer *sp) {
    if (sp && !sp->freed) {
        secure_memzero(sp->ptr, 0);  // 清零
        free(sp->ptr);
        sp->ptr = NULL;
        sp->freed = true;
    }
}
```

## 七、文件操作安全

```c
#include <unistd.h>
#include <sys/stat.h>

// 安全的临时文件
FILE *create_safe_temp(const char *prefix) {
    char template_[256];
    snprintf(template_, sizeof(template_), "/tmp/%s_XXXXXX", prefix);
    int fd = mkstemp(template_);
    if (fd == -1) return NULL;
    return fdopen(fd, "w+");
}

// 检查文件权限
bool check_file_permissions(const char *path) {
    struct stat st;
    if (stat(path, &st) != 0) return false;

    // 检查是否是符号链接（防止TOCTOU攻击）
    if (S_ISLNK(st.st_mode)) return false;

    // 检查权限是否过于宽松
    if (st.st_mode & S_IWOTH) return false;  // 其他用户可写

    return true;
}

// 安全的文件路径拼接
int safe_path_join(char *dest, size_t dest_size,
                   const char *dir, const char *file) {
    // 检查目录遍历攻击
    if (strstr(file, "..")) return -1;
    if (file[0] == '/') return -1;

    int written = snprintf(dest, dest_size, "%s/%s", dir, file);
    return (written > 0 && (size_t)written < dest_size) ? 0 : -1;
}
```

## 八、密码学安全

```c
// 使用cryptographically secure random
#include <fcntl.h>
#include <unistd.h>

int secure_random(void *buf, size_t len) {
    int fd = open("/dev/urandom", O_RDONLY);
    if (fd < 0) return -1;
    ssize_t n = read(fd, buf, len);
    close(fd);
    return (n == (ssize_t)len) ? 0 : -1;
}

// 时间安全的比较（防止时序攻击）
bool secure_compare(const void *a, const void *b, size_t len) {
    const unsigned char *pa = (const unsigned char *)a;
    const unsigned char *pb = (const unsigned char *)b;
    unsigned char diff = 0;
    for (size_t i = 0; i < len; i++) diff |= pa[i] ^ pb[i];
    return diff == 0;
}
```
