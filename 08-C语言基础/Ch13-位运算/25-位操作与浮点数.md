# 25 - 位操作与浮点数

## IEEE 754 浮点数格式

虽然位运算不能直接对浮点数执行，但可以通过类型转换来查看和操作浮点数的位级表示。

### 单精度浮点数（32位）

```
  1位符号    8位指数         23位尾数
  ┌───┬──────────┬──────────────────────────────┐
  │ S │ EEEEEEEE │ MMMMMMMMMMMMMMMMMMMMMMMMM    │
  └───┴──────────┴──────────────────────────────┘
  位31      30-23              22-0

值 = (-1)^S × 1.M × 2^(E-127)
```

### 双精度浮点数（64位）

```
  1位符号   11位指数                    52位尾数
  ┌───┬─────────────┬─────────────────────────────────────────────┐
  │ S │ EEEEEEEEEEE │ MMMMMMM...MMMMMMM                           │
  └───┴─────────────┴─────────────────────────────────────────────┘
  位63      62-52                    51-0

值 = (-1)^S × 1.M × 2^(E-1023)
```

## 查看浮点数的位表示

```c
#include <stdio.h>
#include <stdint.h>
#include <string.h>

// 使用 union 查看浮点数的位表示
typedef union {
    float f;
    uint32_t u;
} FloatUnion;

void print_float_bits(float value) {
    FloatUnion fu;
    fu.f = value;

    uint32_t bits = fu.u;
    uint32_t sign     = (bits >> 31) & 1;
    uint32_t exponent = (bits >> 23) & 0xFF;
    uint32_t mantissa = bits & 0x7FFFFF;

    printf("值: %f\n", value);
    printf("符号: %u\n", sign);
    printf("指数: %u (实际: %d)\n", exponent, (int)exponent - 127);
    printf("尾数: 0x%06X\n", mantissa);
    printf("完整: 0x%08X\n\n", bits);
}

int main(void) {
    print_float_bits(1.0f);     // S=0, E=127, M=0
    print_float_bits(-1.0f);    // S=1, E=127, M=0
    print_float_bits(3.14f);    // 3.14 的位表示
    print_float_bits(0.0f);     // 零
    print_float_bits(1.0f/0.0f); // 无穷大

    return 0;
}
```

## 浮点数特殊值检测

```c
#include <stdio.h>
#include <stdint.h>
#include <math.h>

typedef union {
    float f;
    uint32_t u;
} FloatUnion;

// 检测 NaN
int is_nan(float x) {
    FloatUnion fu;
    fu.f = x;
    uint32_t exp = (fu.u >> 23) & 0xFF;
    uint32_t man = fu.u & 0x7FFFFF;
    return (exp == 0xFF) && (man != 0);  // 指数全1，尾数非0
}

// 检测无穷大
int is_inf(float x) {
    FloatUnion fu;
    fu.f = x;
    uint32_t exp = (fu.u >> 23) & 0xFF;
    uint32_t man = fu.u & 0x7FFFFF;
    return (exp == 0xFF) && (man == 0);  // 指数全1，尾数为0
}

// 检测零
int is_zero(float x) {
    FloatUnion fu;
    fu.f = x;
    return (fu.u & 0x7FFFFFFF) == 0;  // 指数和尾数全0
}

// 检测非规格化数（Denormalized）
int is_denormal(float x) {
    FloatUnion fu;
    fu.f = x;
    uint32_t exp = (fu.u >> 23) & 0xFF;
    uint32_t man = fu.u & 0x7FFFFF;
    return (exp == 0) && (man != 0);
}

int main(void) {
    float values[] = {
        1.0f, 0.0f, -0.0f,
        INFINITY, -INFINITY,
        NAN, 1e-40f  // 非规格化数
    };
    const char *names[] = {
        "1.0", "0.0", "-0.0",
        "INF", "-INF", "NaN", "1e-40"
    };

    for (int i = 0; i < 7; i++) {
        printf("%-6s: NaN=%d Inf=%d Zero=%d Denorm=%d\n",
               names[i],
               is_nan(values[i]),
               is_inf(values[i]),
               is_zero(values[i]),
               is_denormal(values[i]));
    }

    return 0;
}
```

## 位操作实现快速浮点近似

### 快速平方根倒数（经典 Quake III 算法）

```c
#include <stdio.h>

// 经典的快速平方根倒数算法（Quake III Arena 源码中的魔法数）
float fast_inverse_sqrt(float x) {
    float x2 = x * 0.5f;
    FloatUnion fu;
    fu.f = x;

    // 魔法步骤：利用浮点数位表示的整数近似
    fu.u = 0x5F3759DF - (fu.u >> 1);  // 神奇的初始近似

    float result = fu.f;

    // 一次牛顿迭代提高精度
    result = result * (1.5f - (x2 * result * result));

    return result;
}

int main(void) {
    float x = 4.0f;
    float inv_sqrt = fast_inverse_sqrt(x);
    float exact = 1.0f / 1.0f;  // sqrt(4) = 2, 1/2 = 0.5

    printf("1/sqrt(%f) ≈ %f (快速)\n", x, inv_sqrt);
    printf("1/sqrt(%f) = %f (精确)\n", x, 1.0f / __builtin_sqrtf(x));

    return 0;
}
```

## 浮点数位操作的注意事项

```c
#include <stdio.h>
#include <stdint.h>
#include <math.h>

int main(void) {
    // 1. 不能直接对浮点数使用位运算
    // float f = 3.14;
    // f << 1;  // 编译错误！

    // 2. 通过 union 或 memcpy 进行位操作
    typedef union { float f; uint32_t u; } FU;

    // 3. 浮点数的位比较
    FU a, b;
    a.f = 0.1f + 0.2f;
    b.f = 0.3f;

    // 位级比较：它们可能不相等！
    printf("0.1 + 0.2 == 0.3? %s\n",
           a.u == b.u ? "位级相等" : "位级不等");

    // 4. 浮点数取绝对值（位操作）
    FU x;
    x.f = -3.14f;
    x.u &= 0x7FFFFFFF;  // 清除符号位
    printf("|-3.14| = %f\n", x.f);

    return 0;
}
```

## 关键要点

- IEEE 754 浮点数由符号位、指数位、尾数位组成
- 使用 union 或 memcpy 将浮点数的位表示解释为整数
- NaN：指数全1且尾数非0
- 无穷大：指数全1且尾数为0
- 非规格化数：指数全0且尾数非0
- 通过清除符号位（`& 0x7FFFFFFF`）可以快速取浮点数绝对值
- 位级操作在快速浮点近似算法中有重要应用
