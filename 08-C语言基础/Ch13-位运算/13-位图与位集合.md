# 13 - 位图与位集合

## 概念

位图（Bitmap）或位集合（Bitset）是一种用**位数组**来表示集合的数据结构。每个位代表一个元素是否存在：1 表示存在，0 表示不存在。

### 优点

- **极高的空间效率**：1 个字节可以存储 8 个元素的存在信息
- **高效的集合操作**：并集、交集、差集只需一条位运算指令
- **快速查找/插入/删除**：O(1) 时间复杂度

### 缺点

- 只能存储整数类型（或可映射为整数的元素）
- 元素范围较大时仍需较多内存
- 不适合稀疏集合（元素分布分散）

## 基本实现

```c
#include <stdio.h>
#include <string.h>

#define BITMAP_SIZE 1024  // 支持 0-1023 的整数

typedef struct {
    unsigned int bits[BITMAP_SIZE / 32];  // 每个 unsigned int 存 32 位
} Bitmap;

// 初始化位图
void bitmap_init(Bitmap *bm) {
    memset(bm->bits, 0, sizeof(bm->bits));
}

// 设置元素（将第 n 位置 1）
void bitmap_set(Bitmap *bm, int n) {
    bm->bits[n / 32] |= (1U << (n % 32));
}

// 清除元素（将第 n 位清零）
void bitmap_clear(Bitmap *bm, int n) {
    bm->bits[n / 32] &= ~(1U << (n % 32));
}

// 检查元素是否存在
int bitmap_test(Bitmap *bm, int n) {
    return (bm->bits[n / 32] >> (n % 32)) & 1;
}

// 翻转元素
void bitmap_flip(Bitmap *bm, int n) {
    bm->bits[n / 32] ^= (1U << (n % 32));
}

int main(void) {
    Bitmap bm;
    bitmap_init(&bm);

    bitmap_set(&bm, 5);
    bitmap_set(&bm, 100);
    bitmap_set(&bm, 500);

    printf("5 存在: %s\n", bitmap_test(&bm, 5) ? "是" : "否");
    printf("6 存在: %s\n", bitmap_test(&bm, 6) ? "是" : "否");
    printf("100 存在: %s\n", bitmap_test(&bm, 100) ? "是" : "否");

    bitmap_clear(&bm, 100);
    printf("清除100后: %s\n", bitmap_test(&bm, 100) ? "存在" : "不存在");

    return 0;
}
```

## 集合运算

位图的一个巨大优势是可以用**一条位运算**完成集合操作。

```c
#include <stdio.h>
#include <string.h>

#define BITMAP_WORDS 4  // 4 * 32 = 128 位

typedef struct {
    unsigned int bits[BITMAP_WORDS];
} Bitmap;

void bitmap_init(Bitmap *bm) {
    memset(bm->bits, 0, sizeof(bm->bits));
}

void bitmap_set(Bitmap *bm, int n) {
    bm->bits[n / 32] |= (1U << (n % 32));
}

int bitmap_test(Bitmap *bm, int n) {
    return (bm->bits[n / 32] >> (n % 32)) & 1;
}

// 并集：A ∪ B
Bitmap bitmap_union(Bitmap *a, Bitmap *b) {
    Bitmap result;
    for (int i = 0; i < BITMAP_WORDS; i++)
        result.bits[i] = a->bits[i] | b->bits[i];
    return result;
}

// 交集：A ∩ B
Bitmap bitmap_intersection(Bitmap *a, Bitmap *b) {
    Bitmap result;
    for (int i = 0; i < BITMAP_WORDS; i++)
        result.bits[i] = a->bits[i] & b->bits[i];
    return result;
}

// 差集：A - B（在A中但不在B中）
Bitmap bitmap_difference(Bitmap *a, Bitmap *b) {
    Bitmap result;
    for (int i = 0; i < BITMAP_WORDS; i++)
        result.bits[i] = a->bits[i] & ~b->bits[i];
    return result;
}

// 对称差集：A ⊕ B（在A或B中，但不同时在两者中）
Bitmap bitmap_symmetric_diff(Bitmap *a, Bitmap *b) {
    Bitmap result;
    for (int i = 0; i < BITMAP_WORDS; i++)
        result.bits[i] = a->bits[i] ^ b->bits[i];
    return result;
}

// 打印集合中的元素
void bitmap_print(Bitmap *bm) {
    printf("{ ");
    for (int i = 0; i < BITMAP_WORDS * 32; i++) {
        if (bitmap_test(bm, i))
            printf("%d ", i);
    }
    printf("}\n");
}

int main(void) {
    Bitmap a, b;
    bitmap_init(&a);
    bitmap_init(&b);

    // 集合 A = {1, 3, 5, 7, 9}
    bitmap_set(&a, 1); bitmap_set(&a, 3);
    bitmap_set(&a, 5); bitmap_set(&a, 7); bitmap_set(&a, 9);

    // 集合 B = {2, 3, 5, 8, 9}
    bitmap_set(&b, 2); bitmap_set(&b, 3);
    bitmap_set(&b, 5); bitmap_set(&b, 8); bitmap_set(&b, 9);

    printf("A = "); bitmap_print(&a);
    printf("B = "); bitmap_print(&b);

    Bitmap u = bitmap_union(&a, &b);
    printf("A ∪ B = "); bitmap_print(&u);

    Bitmap inter = bitmap_intersection(&a, &b);
    printf("A ∩ B = "); bitmap_print(&inter);

    Bitmap diff = bitmap_difference(&a, &b);
    printf("A - B = "); bitmap_print(&diff);

    Bitmap sym = bitmap_symmetric_diff(&a, &b);
    printf("A ⊕ B = "); bitmap_print(&sym);

    return 0;
}
```

## 简化版：使用单个整数作为位集合

对于小范围的集合（0-31），可以直接用一个 `unsigned int`：

```c
#include <stdio.h>

typedef unsigned int Bitset;

void bs_add(Bitset *s, int n)    { *s |= (1U << n); }
void bs_remove(Bitset *s, int n) { *s &= ~(1U << n); }
int  bs_contains(Bitset s, int n){ return (s >> n) & 1; }
void bs_toggle(Bitset *s, int n) { *s ^= (1U << n); }

// 集合操作
Bitset bs_union(Bitset a, Bitset b)        { return a | b; }
Bitset bs_intersection(Bitset a, Bitset b)  { return a & b; }
Bitset bs_difference(Bitset a, Bitset b)    { return a & ~b; }

void bs_print(Bitset s) {
    printf("{ ");
    for (int i = 0; i < 32; i++) {
        if (bs_contains(s, i))
            printf("%d ", i);
    }
    printf("}\n");
}

int main(void) {
    Bitset a = 0, b = 0;

    bs_add(&a, 1); bs_add(&a, 3); bs_add(&a, 5);
    bs_add(&b, 2); bs_add(&b, 3); bs_add(&b, 5);

    printf("A = "); bs_print(a);
    printf("B = "); bs_print(b);
    printf("A ∪ B = "); bs_print(bs_union(a, b));
    printf("A ∩ B = "); bs_print(bs_intersection(a, b));

    return 0;
}
```

## 应用：筛法求素数

```c
#include <stdio.h>
#include <string.h>

#define MAX_N 1000000

unsigned int sieve[(MAX_N + 31) / 32];

void set_bit(int n)  { sieve[n / 32] |= (1U << (n % 32)); }
void clear_bit(int n){ sieve[n / 32] &= ~(1U << (n % 32)); }
int test_bit(int n)  { return (sieve[n / 32] >> (n % 32)) & 1; }

int main(void) {
    // 0 表示素数，1 表示合数
    memset(sieve, 0, sizeof(sieve));
    set_bit(0); set_bit(1);  // 0 和 1 不是素数

    for (int i = 2; i * i <= MAX_N; i++) {
        if (!test_bit(i)) {  // i 是素数
            for (int j = i * i; j <= MAX_N; j += i)
                set_bit(j);  // 标记合数
        }
    }

    // 统计并打印前20个素数
    int count = 0;
    printf("素数: ");
    for (int i = 2; i <= MAX_N && count < 20; i++) {
        if (!test_bit(i)) {
            printf("%d ", i);
            count++;
        }
    }
    printf("\n");

    // 统计总数
    int total = 0;
    for (int i = 2; i <= MAX_N; i++)
        if (!test_bit(i)) total++;
    printf("2到%d之间有%d个素数\n", MAX_N, total);

    return 0;
}
```

## 关键要点

- 位图用每个位表示一个元素是否存在，空间效率极高
- 集合操作（并、交、差）只需一条位运算
- `n / 32` 定位数组下标，`n % 32` 定位位位置
- 位图适合元素范围有限且密集的场景
- 经典应用：素数筛、数据去重、集合运算、布隆过滤器基础
