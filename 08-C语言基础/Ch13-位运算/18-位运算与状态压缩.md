# 18 - 位运算与状态压缩

## 概念

状态压缩（State Compression）是算法竞赛中的经典技巧：用一个整数的二进制位来表示一个**集合**或**状态**，从而用位运算高效地处理组合问题。

## 用位表示集合

一个 n 位的整数可以表示 n 个元素的集合：第 i 位为 1 表示元素 i 在集合中。

```c
// 集合 {0, 2, 3} 表示为：0b1101 = 13
//   位: 3  2  1  0
//       1  1  0  1
```

## 基本集合操作

```c
#include <stdio.h>

int main(void) {
    int S = 0b1101;  // 集合 {0, 2, 3}

    // 检查元素 i 是否在集合中
    int i = 2;
    if (S & (1 << i))
        printf("%d 在集合中\n", i);

    // 添加元素
    S |= (1 << 1);  // 添加元素1 → {0, 1, 2, 3}
    printf("添加1后: 0b%d%d%d%d\n",
           (S>>3)&1, (S>>2)&1, (S>>1)&1, (S>>0)&1);

    // 删除元素
    S &= ~(1 << 2);  // 删除元素2 → {0, 1, 3}
    printf("删除2后: 0b%d%d%d%d\n",
           (S>>3)&1, (S>>2)&1, (S>>1)&1, (S>>0)&1);

    // 最小元素
    int lowest = S & (-S);  // 取最低位的1
    printf("最低位: %d\n", lowest);

    // 删除最小元素
    S &= S - 1;

    return 0;
}
```

## 集合枚举

枚举 n 个元素的所有子集：

```c
#include <stdio.h>

void enumerate_subsets(int n) {
    // 共有 2^n 个子集
    int total = 1 << n;

    for (int mask = 0; mask < total; mask++) {
        printf("{ ");
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i))
                printf("%d ", i);
        }
        printf("}\n");
    }
}

int main(void) {
    enumerate_subsets(3);
    return 0;
}
```

## 旅行商问题（TSP）状态压缩 DP

经典的状态压缩动态规划问题。

```c
#include <stdio.h>
#include <limits.h>

#define MAX_N 16
#define INF INT_MAX

int dp[1 << MAX_N][MAX_N];  // dp[S][i]：经过集合S中的城市，当前在城市i的最短距离
int dist[MAX_N][MAX_N];      // 城市间距离

int tsp(int n) {
    // 初始化
    for (int i = 0; i < (1 << n); i++)
        for (int j = 0; j < n; j++)
            dp[i][j] = INF;

    // 从城市0出发
    dp[1][0] = 0;  // 集合{0}，在城市0，距离0

    // 遍历所有状态
    for (int S = 1; S < (1 << n); S++) {
        for (int u = 0; u < n; u++) {
            if (dp[S][u] == INF) continue;
            if (!(S & (1 << u))) continue;  // u不在集合S中

            // 尝试前往未访问的城市
            for (int v = 0; v < n; v++) {
                if (S & (1 << v)) continue;      // v已在集合中，跳过
                if (dist[u][v] == INF) continue;  // 无路径

                int new_S = S | (1 << v);  // 将v加入集合
                int new_dist = dp[S][u] + dist[u][v];

                if (new_dist < dp[new_S][v])
                    dp[new_S][v] = new_dist;
            }
        }
    }

    // 找到回到城市0的最短路径
    int full = (1 << n) - 1;  // 所有城市都访问过的状态
    int result = INF;
    for (int i = 0; i < n; i++) {
        if (dp[full][i] != INF && dist[i][0] != INF) {
            int total = dp[full][i] + dist[i][0];
            if (total < result) result = total;
        }
    }

    return result;
}

int main(void) {
    int n = 4;

    // 距离矩阵
    int d[4][4] = {
        {0,  10, 15, 20},
        {10, 0,  35, 25},
        {15, 35, 0,  30},
        {20, 25, 30, 0}
    };

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            dist[i][j] = d[i][j];

    printf("TSP最短路径: %d\n", tsp(n));  // 80

    return 0;
}
```

## 集合的固定大小子集枚举

```c
#include <stdio.h>

// 枚举 n 个元素中恰好选 k 个的所有方案
void enumerate_fixed_size(int n, int k) {
    int S = (1 << k) - 1;  // 最小的 k 个1: 0b00...011...1

    while (S < (1 << n)) {
        // 输出当前子集
        printf("{ ");
        for (int i = 0; i < n; i++) {
            if (S & (1 << i))
                printf("%d ", i);
        }
        printf("}\n");

        // Gosper's Hack：找下一个有k个1的数
        int c = S & (-S);
        int r = S + c;
        S = (((r ^ S) >> 2) / c) | r;
    }
}

int main(void) {
    printf("从4个元素中选2个:\n");
    enumerate_fixed_size(4, 2);
    return 0;
}
```

## 状态压缩优化技巧

```c
#include <stdio.h>

// 快速统计集合中的元素个数
int popcount(int S) {
    int count = 0;
    while (S) {
        S &= S - 1;
        count++;
    }
    return count;
}

// 只遍历 S 的所有子集（不含 S 本身）
void enumerate_proper_subsets(int S) {
    int sub = S;
    while (sub) {
        sub = (sub - 1) & S;  // 上一个子集
        if (sub) {
            printf("{ ");
            for (int i = 0; i < 32; i++)
                if (sub & (1 << i)) printf("%d ", i);
            printf("}\n");
        }
    }
}

int main(void) {
    int S = 0b1011;  // {0, 1, 3}
    printf("集合 {0,1,3} 的真子集:\n");
    enumerate_proper_subsets(S);
    return 0;
}
```

## 关键要点

- 状态压缩用一个整数的位表示集合/状态
- `1 << i` 表示只含元素 i 的集合
- `S | (1 << i)` 添加元素，`S & ~(1 << i)` 删除元素
- `S & (-S)` 取最低位元素，`S & (S-1)` 删除最低位元素
- 枚举所有子集：`for (S = 0; S < (1 << n); S++)`
- 枚举 S 的所有子集：`sub = (sub - 1) & S`
- 状态压缩 DP 适合 n <= 20 左右的组合优化问题
