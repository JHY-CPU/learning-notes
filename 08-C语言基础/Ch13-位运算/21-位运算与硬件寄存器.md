# 21 - 位运算与硬件寄存器

## 概述

在嵌入式开发中，硬件寄存器的每一位都有特定的含义。位运算是操作这些寄存器的唯一方式。

## 寄存器的基本操作

### 典型场景

```c
#include <stdio.h>
#include <stdint.h>

// 模拟一个 GPIO 寄存器
volatile uint32_t GPIO_PORT = 0;

// 定义引脚
#define PIN_LED    (1U << 5)
#define PIN_BUTTON (1U << 3)
#define PIN_BUZZER (1U << 7)

int main(void) {
    // 点亮 LED（设置第5位）
    GPIO_PORT |= PIN_LED;
    printf("LED ON: 0x%08X\n", GPIO_PORT);

    // 熄灭 LED（清除第5位）
    GPIO_PORT &= ~PIN_LED;
    printf("LED OFF: 0x%08X\n", GPIO_PORT);

    // 读取按钮状态
    if (GPIO_PORT & PIN_BUTTON)
        printf("按钮按下\n");
    else
        printf("按钮未按下\n");

    // 翻转蜂鸣器
    GPIO_PORT ^= PIN_BUZZER;

    return 0;
}
```

## volatile 关键字

`volatile` 告诉编译器：这个变量的值可能在编译器不知道的情况下被改变（如硬件修改），不要对其进行优化。

```c
#include <stdio.h>
#include <stdint_t.h>

// 不使用 volatile：编译器可能优化掉重复读取
int read_status_wrong(uint32_t *status_reg) {
    while (*status_reg & 0x01 == 0)  // 等待就绪
        ;
    return *status_reg;
}

// 使用 volatile：每次都会实际读取硬件
int read_status_correct(volatile uint32_t *status_reg) {
    while (*status_reg & 0x01 == 0)  // 编译器不会优化这个循环
        ;
    return *status_reg;
}
```

## 寄存器操作模式

### 1. 读-修改-写（Read-Modify-Write）

```c
#include <stdio.h>
#include <stdint.h>

volatile uint32_t REG = 0x000000FF;

// 设置特定位
void reg_set_bits(volatile uint32_t *reg, uint32_t mask) {
    *reg |= mask;
}

// 清除特定位
void reg_clear_bits(volatile uint32_t *reg, uint32_t mask) {
    *reg &= ~mask;
}

// 修改特定位字段
void reg_modify_field(volatile uint32_t *reg,
                      uint32_t mask, int shift, uint32_t value) {
    *reg = (*reg & ~mask) | ((value << shift) & mask);
}

// 读取特定位字段
uint32_t reg_read_field(volatile uint32_t *reg,
                        uint32_t mask, int shift) {
    return (*reg & mask) >> shift;
}

int main(void) {
    printf("初始值: 0x%08X\n", REG);

    reg_set_bits(&REG, (1U << 16));
    printf("设置第16位: 0x%08X\n", REG);

    reg_clear_bits(&REG, 0xFF);
    printf("清除低8位: 0x%08X\n", REG);

    return 0;
}
```

### 2. 写 1 清除（Write-1-to-Clear）

某些寄存器（如中断状态寄存器）需要写 1 来清除标志。

```c
#include <stdio.h>
#include <stdint.h>

volatile uint32_t INT_STATUS = 0;  // 中断状态寄存器

// 定义中断源
#define INT_TIMER   (1U << 0)
#define INT_UART    (1U << 1)
#define INT_GPIO    (1U << 2)

// 清除中断（写1清除）
void clear_interrupt(volatile uint32_t *status, uint32_t int_mask) {
    *status = int_mask;  // 写1清除
}

int main(void) {
    // 模拟中断发生
    INT_STATUS = INT_TIMER | INT_GPIO;
    printf("中断状态: 0x%02X\n", INT_STATUS);

    // 处理定时器中断
    if (INT_STATUS & INT_TIMER) {
        printf("处理定时器中断\n");
        clear_interrupt(&INT_STATUS, INT_TIMER);
    }

    printf("清除后状态: 0x%02X\n", INT_STATUS);

    return 0;
}
```

## 位带操作（Bit-Banding）

ARM Cortex-M 微控制器提供位带功能，将每个位映射到一个独立的地址，可以直接访问而无需读-修改-写。

```c
#include <stdio.h>
#include <stdint.h>

// 模拟位带操作
#define BITBAND_SRAM_BASE   0x20000000
#define BITBAND_SRAM_ALIAS  0x22000000

// 计算位带别名地址
// alias_addr = alias_base + (byte_offset * 32) + (bit_number * 4)
uint32_t *bitband_address(uint32_t *addr, int bit) {
    uint32_t offset = (uint32_t)addr - BITBAND_SRAM_BASE;
    uint32_t alias = BITBAND_SRAM_ALIAS + (offset * 32) + (bit * 4);
    return (uint32_t *)alias;
}

int main(void) {
    printf("位带操作概念演示\n");
    printf("地址 0x%08X 的第 5 位\n", 0x20000000);
    printf("位带别名地址: 0x%08X\n",
           BITBAND_SRAM_ALIAS + (0 * 32) + (5 * 4));
    // 0x22000000 + 0 + 20 = 0x22000014

    return 0;
}
```

## 嵌入式 GPIO 封装

```c
#include <stdio.h>
#include <stdint.h>

// 模拟 STM32 GPIO 寄存器组
typedef struct {
    volatile uint32_t MODER;    // 模式寄存器
    volatile uint32_t OTYPER;   // 输出类型寄存器
    volatile uint32_t OSPEEDR;  // 输出速度寄存器
    volatile uint32_t PUPDR;    // 上拉/下拉寄存器
    volatile uint32_t IDR;      // 输入数据寄存器
    volatile uint32_t ODR;      // 输出数据寄存器
    volatile uint32_t BSRR;     // 位设置/复位寄存器
} GPIO_TypeDef;

// 模拟 GPIO 端口
GPIO_TypeDef GPIOA_sim = {0};

// 引脚操作宏
#define GPIO_PIN_5  (1U << 5)
#define GPIO_PIN_13 (1U << 13)

// 设置引脚输出高电平（使用BSRR寄存器，原子操作）
void gpio_set(GPIO_TypeDef *gpio, uint32_t pin) {
    gpio->BSRR = pin;  // 写1到低16位 = 设置
}

// 设置引脚输出低电平
void gpio_reset(GPIO_TypeDef *gpio, uint32_t pin) {
    gpio->BSRR = pin << 16;  // 写1到高16位 = 复位
}

// 读取引脚状态
int gpio_read(GPIO_TypeDef *gpio, uint32_t pin) {
    return (gpio->IDR & pin) ? 1 : 0;
}

int main(void) {
    GPIO_TypeDef *GPIOA = &GPIOA_sim;

    // 点亮 LED
    gpio_set(GPIOA, GPIO_PIN_5);
    printf("ODR: 0x%08X\n", GPIOA->ODR);

    // 熄灭 LED
    gpio_reset(GPIOA, GPIO_PIN_5);
    printf("ODR: 0x%08X\n", GPIOA->ODR);

    return 0;
}
```

## 关键要点

- 硬件寄存器的每一位都有特定功能，必须用位运算操作
- `volatile` 防止编译器优化对硬件寄存器的访问
- 读-修改-写模式：先读取整个寄存器，修改特定位，再写回
- BSRR 类寄存器支持原子操作，避免竞态条件
- 位带操作将单个位映射到独立地址，简化位访问
- 使用宏定义寄存器地址和位掩码，提高代码可读性
