# 17 - 位运算与编码

## 格雷码（Gray Code）

格雷码是一种二进制编码方式，**相邻两个数的格雷码只有一位不同**。这在数字电路中可以减少切换时的错误。

### 二进制转格雷码

```
格雷码 = 二进制 ^ (二进制 >> 1)
```

```c
#include <stdio.h>

// 二进制转格雷码
unsigned int binary_to_gray(unsigned int n) {
    return n ^ (n >> 1);
}

// 格雷码转二进制
unsigned int gray_to_binary(unsigned int n) {
    unsigned int mask = n;
    while (mask >>= 1) {
        n ^= mask;
    }
    return n;
}

int main(void) {
    printf("二进制 -> 格雷码:\n");
    for (int i = 0; i < 16; i++) {
        unsigned int gray = binary_to_gray(i);
        printf("  %2d (", i);
        // 打印4位二进制
        for (int b = 3; b >= 0; b--) printf("%d", (i >> b) & 1);
        printf(") -> ");
        for (int b = 3; b >= 0; b--) printf("%d", (gray >> b) & 1);
        printf(" (%2u)\n", gray);
    }

    printf("\n格雷码 -> 二进制（验证）:\n");
    for (int i = 0; i < 16; i++) {
        unsigned int gray = binary_to_gray(i);
        unsigned int binary = gray_to_binary(gray);
        printf("  格雷码 %2u -> 二进制 %2u\n", gray, binary);
    }

    return 0;
}
```

### 格雷码的特点

- 任意相邻两个数只有一位不同（汉明距离为1）
- 循环性：最后一个数和第一个数也只有一位不同
- 应用：旋转编码器、Karnaugh 图、错误最小化

## 奇偶校验（Parity Check）

奇偶校验是最简单的错误检测方法，通过计算数据中 1 的个数来添加校验位。

```c
#include <stdio.h>

// 计算奇偶位（偶校验：使1的个数为偶数）
unsigned char even_parity(unsigned char data) {
    unsigned char parity = 0;
    unsigned char temp = data;

    while (temp) {
        parity ^= 1;   // 每遇到一个1就翻转
        temp &= temp - 1;  // 清除最低位的1（Brian Kernighan）
    }

    // parity=0 表示1的个数是偶数，不需要添加
    // parity=1 表示1的个数是奇数，需要添加
    return parity;
}

// 快速奇偶计算（分治法）
unsigned char fast_parity(unsigned int n) {
    n ^= n >> 16;
    n ^= n >> 8;
    n ^= n >> 4;
    n ^= n >> 2;
    n ^= n >> 1;
    return n & 1;
}

int main(void) {
    unsigned char data[] = {0b10110011, 0b10101010, 0b11111111, 0b00000000};

    for (int i = 0; i < 4; i++) {
        printf("数据: ");
        for (int b = 7; b >= 0; b--) printf("%d", (data[i] >> b) & 1);
        printf(" -> 奇偶位: %u\n", even_parity(data[i]));
    }

    return 0;
}
```

## 汉明码思想

汉明码是一种纠错码，可以检测并纠正**单比特错误**。核心思想是利用多个校验位覆盖不同的数据位。

```c
#include <stdio.h>

// 简化的汉明码编码（7,4）：4位数据 + 3位校验 = 7位
unsigned char hamming_encode_74(unsigned char data) {
    // data 的低4位是数据位
    unsigned char d1 = (data >> 0) & 1;
    unsigned char d2 = (data >> 1) & 1;
    unsigned char d3 = (data >> 2) & 1;
    unsigned char d4 = (data >> 3) & 1;

    // 计算校验位
    unsigned char p1 = d1 ^ d2 ^ d4;  // 校验位1：覆盖位置1,3,5,7
    unsigned char p2 = d1 ^ d3 ^ d4;  // 校验位2：覆盖位置2,3,6,7
    unsigned char p3 = d2 ^ d3 ^ d4;  // 校验位3：覆盖位置4,5,6,7

    // 组装：位置 1=p1, 2=p2, 3=d1, 4=p3, 5=d2, 6=d3, 7=d4
    unsigned char encoded = 0;
    encoded |= p1 << 0;
    encoded |= p2 << 1;
    encoded |= d1 << 2;
    encoded |= p3 << 3;
    encoded |= d2 << 4;
    encoded |= d3 << 5;
    encoded |= d4 << 6;

    return encoded;
}

int main(void) {
    unsigned char data = 0b1011;  // 4位数据

    unsigned char encoded = hamming_encode_74(data);
    printf("数据: ");
    for (int b = 3; b >= 0; b--) printf("%d", (data >> b) & 1);
    printf("\n编码: ");
    for (int b = 6; b >= 0; b--) printf("%d", (encoded >> b) & 1);
    printf("\n");

    return 0;
}
```

## CRC 校验（循环冗余校验）

CRC 是通信中常用的错误检测方法，基于多项式除法（用位运算实现）。

```c
#include <stdio.h>
#include <stdint.h>

// CRC-8 简单实现
uint8_t crc8(const uint8_t *data, int len) {
    uint8_t crc = 0x00;

    for (int i = 0; i < len; i++) {
        crc ^= data[i];
        for (int j = 0; j < 8; j++) {
            if (crc & 0x80)
                crc = (crc << 1) ^ 0x07;  // 多项式 x^8+x^2+x+1
            else
                crc <<= 1;
        }
    }
    return crc;
}

int main(void) {
    uint8_t message[] = {0x48, 0x65, 0x6C, 0x6C, 0x6F};  // "Hello"

    uint8_t checksum = crc8(message, 5);
    printf("CRC-8 校验值: 0x%02X\n", checksum);

    // 添加校验值后验证
    uint8_t with_check[6];
    for (int i = 0; i < 5; i++) with_check[i] = message[i];
    with_check[5] = checksum;

    if (crc8(with_check, 6) == 0)
        printf("数据校验通过\n");
    else
        printf("数据校验失败\n");

    return 0;
}
```

## 关键要点

- 格雷码：相邻数只有一位不同，`gray = n ^ (n >> 1)`
- 奇偶校验：最简单的错误检测，统计 1 的个数
- 汉明码：可以检测和纠正单比特错误
- CRC：基于多项式除法，广泛用于通信协议
- 这些编码方法都大量依赖位运算来实现
