# 07 - 右移运算

## 概念

右移运算符 `>>` 将一个数的所有二进制位向右移动指定的位数。右移 n 位相当于除以 2 的 n 次方（向下取整）。

右移运算比左移复杂，因为**有符号数和无符号数的右移行为不同**。

## 逻辑右移 vs 算术右移

### 逻辑右移（Logical Right Shift）

用于**无符号数**：左侧始终用 0 填充。

```
1011 0100 >> 2（逻辑右移）
= 0010 1101
  ^^ 用0填充
```

### 算术右移（Arithmetic Right Shift）

用于**有符号数**：左侧用**符号位**填充（正数补0，负数补1），保持符号不变。

```
1011 0100 >> 2（算术右移，最高位是符号位1）
= 1110 1101
  ^^ 用符号位1填充（保持负数）
```

```
0011 0100 >> 2（算术右移，最高位是符号位0）
= 0000 1101
  ^^ 用符号位0填充（保持正数）
```

## 基本用法

```c
#include <stdio.h>

void print_binary_8(unsigned char n) {
    for (int i = 7; i >= 0; i--)
        printf("%d", (n >> i) & 1);
}

int main(void) {
    // 无符号右移（逻辑右移）
    unsigned char a = 180;  // 10110100
    printf("无符号: ");
    print_binary_8(a);
    printf(" >> 2 = ");
    print_binary_8(a >> 2);
    printf(" (%u)\n", a >> 2);  // 45 (00101101)

    // 有符号右移（算术右移，实现定义的行为）
    signed char b = -76;  // 10110100（补码）
    printf("有符号: %d >> 2 = %d\n", b, b >> 2);  // -19

    // 正数的右移
    signed char c = 100;  // 01100100
    printf("正数: %d >> 2 = %d\n", c, c >> 2);  // 25

    return 0;
}
```

## 右移与除法的关系

```c
#include <stdio.h>

int main(void) {
    int n = 100;

    printf("%d / 2  = %d,  %d >> 1 = %d\n", n, n / 2, n, n >> 1);
    printf("%d / 4  = %d,  %d >> 2 = %d\n", n, n / 4, n, n >> 2);
    printf("%d / 8  = %d,  %d >> 3 = %d\n", n, n / 8, n, n >> 3);
    printf("%d / 16 = %d,  %d >> 4 = %d\n", n, n / 16, n, n >> 4);

    // 注意：对于负数，右移和除法可能结果不同！
    int m = -100;
    printf("\n负数 -100:\n");
    printf("除法: %d / 4 = %d\n", m, m / 4);     // -25
    printf("右移: %d >> 2 = %d\n", m, m >> 2);    // -25（算术右移）

    // 但对于 -5 / 2：
    int p = -5;
    printf("\n负数 -5:\n");
    printf("除法: %d / 2 = %d\n", p, p / 2);     // -2（向0取整）
    printf("右移: %d >> 1 = %d\n", p, p >> 1);    // -3（向下取整）

    // 除法向0取整，算术右移向下取整
    // -5 / 2 = -2（向0取整）
    // -5 >> 1 = -3（向下取整，即 floor(-5/2)）

    return 0;
}
```

## 实际应用

### 1. 二分查找

```c
#include <stdio.h>

int binary_search(int arr[], int n, int target) {
    int left = 0, right = n - 1;

    while (left <= right) {
        int mid = left + ((right - left) >> 1);  // 等价于 (left + right) / 2
        // 使用 >> 1 替代 / 2

        if (arr[mid] == target)
            return mid;
        else if (arr[mid] < target)
            left = mid + 1;
        else
            right = mid - 1;
    }
    return -1;
}

int main(void) {
    int arr[] = {1, 3, 5, 7, 9, 11, 13, 15};
    int n = sizeof(arr) / sizeof(arr[0]);

    int index = binary_search(arr, n, 7);
    printf("找到7在索引: %d\n", index);  // 3

    return 0;
}
```

### 2. 提取特定位段

```c
#include <stdio.h>

int main(void) {
    unsigned int value = 0xABCD1234;

    // 提取最高字节
    unsigned char high_byte = value >> 24;
    printf("最高字节: 0x%02X\n", high_byte);  // 0xAB

    // 提取第 16-23 位
    unsigned char byte2 = (value >> 16) & 0xFF;
    printf("第2字节: 0x%02X\n", byte2);  // 0xCD

    // 提取第 4-7 位
    unsigned int nibble = (value >> 4) & 0x0F;
    printf("第4-7位: 0x%X\n", nibble);  // 0x3

    return 0;
}
```

### 3. 逐位处理

```c
#include <stdio.h>

// 统计一个数中 1 的个数
int count_ones(unsigned int n) {
    int count = 0;
    while (n) {
        count += (n & 1);  // 检查最低位
        n >>= 1;           // 右移一位
    }
    return count;
}

int main(void) {
    printf("5 的二进制有 %d 个1\n", count_ones(5));    // 2 (0101)
    printf("15 的二进制有 %d 个1\n", count_ones(15));   // 4 (1111)
    printf("1023 的二进制有 %d 个1\n", count_ones(1023)); // 10

    return 0;
}
```

### 4. 快速计算平均值

```c
#include <stdio.h>

int main(void) {
    int a = 17, b = 25;

    // 快速计算两个非负整数的平均值（避免溢出）
    // (a + b) / 2 在 a+b 溢出时会出错
    // (a >> 1) + (b >> 1) + (a & b & 1) 是安全的做法
    int avg = (a >> 1) + (b >> 1) + (a & b & 1);
    printf("(%d + %d) / 2 = %d\n", a, b, avg);  // 21

    return 0;
}
```

## 有符号右移的行为

C 标准规定：**有符号数的右移行为是实现定义的**（implementation-defined）。

- 大多数现代平台（x86、ARM）使用**算术右移**（保留符号位）
- 某些平台可能使用逻辑右移（补0）
- 编译器可以自由选择

```c
#include <stdio.h>

int main(void) {
    signed int x = -8;

    // 在大多数平台上，这是算术右移
    // -8 的补码: 11111111 11111111 11111111 11111000
    // >> 1:      11111111 11111111 11111111 11111100 = -4
    printf("-8 >> 1 = %d\n", x >> 1);  // -4（在大多数平台上）

    // 建议：对有符号数使用右移时，先转换为无符号类型
    unsigned int ux = (unsigned int)x;
    unsigned int shifted = ux >> 1;
    printf("无符号右移: %u\n", shifted);

    return 0;
}
```

## 移位量的安全范围

```c
#include <stdio.h>

int main(void) {
    // 对于 N 位类型，移位量必须满足：0 <= shift < N
    // int 是 32 位：有效范围 0-31

    unsigned int a = 1;

    printf("1 << 0  = %u\n", a << 0);
    printf("1 << 31 = %u\n", a << 31);
    // printf("1 << 32 = %u\n", a << 32);  // 未定义行为！

    // 安全的做法
    int shift_amount = 10;
    if (shift_amount >= 0 && shift_amount < (int)sizeof(unsigned int) * 8) {
        unsigned int result = a << shift_amount;
        printf("安全左移: %u\n", result);  // 1024
    }

    return 0;
}
```

## 关键要点

- `>>` 右移：所有位向右移动 n 位
- **逻辑右移**（无符号数）：左侧补 0
- **算术右移**（有符号数）：左侧补符号位，保持正负号
- 右移 n 位近似等于除以 2^n（但负数的取整方式不同）
- 有符号右移的行为是实现定义的，建议使用无符号类型
- 右移常用于二分查找、位段提取、逐位处理等场景
