# 14 - 位字段结构体（Bit Fields）

## 概念

位字段（Bit Fields）是 C 语言结构体中的一种特殊成员，允许我们**按位**指定每个成员占用的比特数。这在需要精确控制内存布局的场景中非常有用。

## 基本语法

```c
struct 结构体名 {
    类型 成员名 : 位宽;
};
```

## 基本示例

```c
#include <stdio.h>

struct Flags {
    unsigned int is_active  : 1;   // 1位：是否激活
    unsigned int is_visible : 1;   // 1位：是否可见
    unsigned int priority   : 3;   // 3位：优先级（0-7）
    unsigned int color      : 3;   // 3位：颜色（0-7）
    unsigned int reserved   : 24;  // 24位：保留
};

int main(void) {
    struct Flags f = {0};

    f.is_active = 1;
    f.is_visible = 1;
    f.priority = 5;
    f.color = 3;

    printf("大小: %zu 字节\n", sizeof(f));  // 4 字节（32位）
    printf("活跃: %u\n", f.is_active);
    printf("可见: %u\n", f.is_visible);
    printf("优先级: %u\n", f.priority);
    printf("颜色: %u\n", f.color);

    return 0;
}
```

## 网络协议报头示例

### IPv4 报头

```c
#include <stdio.h>

// IPv4 报头（简化版）
struct IPv4Header {
    unsigned int version     : 4;   // 版本号（4）
    unsigned int ihl         : 4;   // 首部长度
    unsigned int dscp        : 6;   // 服务类型
    unsigned int ecn         : 2;   // 显式拥塞通知
    unsigned int total_len   : 16;  // 总长度
    unsigned int identification : 16; // 标识
    unsigned int flags       : 3;   // 标志
    unsigned int frag_offset : 13;  // 片偏移
    unsigned int ttl         : 8;   // 生存时间
    unsigned int protocol    : 8;   // 协议
    unsigned int checksum    : 16;  // 首部校验和
    unsigned int src_addr;          // 源地址（32位）
    unsigned int dst_addr;          // 目标地址（32位）
};

int main(void) {
    struct IPv4Header header = {0};

    header.version = 4;
    header.ihl = 5;           // 5 * 4 = 20 字节
    header.ttl = 64;
    header.protocol = 6;      // TCP
    header.total_len = 40;

    printf("版本: %u\n", header.version);
    printf("首部长度: %u 字节\n", header.ihl * 4);
    printf("TTL: %u\n", header.ttl);
    printf("协议: %u (TCP)\n", header.protocol);
    printf("总长度: %u\n", header.total_len);
    printf("报头大小: %zu 字节\n", sizeof(header));

    return 0;
}
```

### TCP 标志位

```c
#include <stdio.h>

struct TCPFlags {
    unsigned int fin : 1;  // 结束
    unsigned int syn : 1;  // 同步
    unsigned int rst : 1;  // 重置
    unsigned int psh : 1;  // 推送
    unsigned int ack : 1;  // 确认
    unsigned int urg : 1;  // 紧急
    unsigned int ece : 1;  // ECN-Echo
    unsigned int cwr : 1;  // 拥塞窗口减少
};

int main(void) {
    struct TCPFlags flags = {0};

    // 建立连接：SYN 包
    flags.syn = 1;
    printf("SYN 包: FIN=%u SYN=%u ACK=%u\n", flags.fin, flags.syn, flags.ack);

    // 确认连接：SYN-ACK 包
    flags.ack = 1;
    printf("SYN-ACK 包: FIN=%u SYN=%u ACK=%u\n", flags.fin, flags.syn, flags.ack);

    return 0;
}
```

## 硬件寄存器示例

```c
#include <stdio.h>

// UART 状态寄存器
struct UARTStatus {
    unsigned int rx_ready    : 1;  // 接收就绪
    unsigned int tx_ready    : 1;  // 发送就绪
    unsigned int parity_err  : 1;  // 奇偶校验错误
    unsigned int frame_err   : 1;  // 帧错误
    unsigned int overrun_err : 1;  // 溢出错误
    unsigned int reserved    : 3;  // 保留
};

int main(void) {
    struct UARTStatus status = {0};
    status.rx_ready = 1;
    status.tx_ready = 1;

    if (status.rx_ready)
        printf("数据已就绪，可以读取\n");

    if (status.frame_err)
        printf("检测到帧错误\n");

    return 0;
}
```

## 位字段的注意事项

### 1. 存储单元与对齐

```c
#include <stdio.h>

struct Example1 {
    unsigned int a : 3;
    unsigned int b : 5;   // 8位，一个字节内
};

struct Example2 {
    unsigned int a : 3;
    unsigned int b : 5;
    unsigned int c : 10;  // 总共18位
    unsigned int d : 20;  // 总共38位，跨越多个存储单元
};

int main(void) {
    printf("Example1 大小: %zu 字节\n", sizeof(struct Example1));  // 4
    printf("Example2 大小: %zu 字节\n", sizeof(struct Example2));  // 8（取决于编译器）
    return 0;
}
```

### 2. 不可取地址

位字段成员不能使用 `&` 取地址，因为它们可能不按字节对齐。

```c
struct Flags {
    unsigned int x : 1;
    unsigned int y : 1;
};

int main(void) {
    struct Flags f = {1, 0};
    // unsigned int *p = &f.x;  // 错误！不能对位字段取地址
    return 0;
}
```

### 3. 不可组成数组

```c
struct Flags {
    unsigned int x : 1;
    // unsigned int arr[3];  // 普通成员和位字段可以混用
};
// struct Flags f_arr[10];  // 可以，但每个数组元素是独立的结构体
```

### 4. 跨越存储单元的实现定义行为

```c
struct Cross {
    unsigned int a : 10;
    unsigned int b : 10;
    unsigned int c : 10;
    unsigned int d : 10;  // 总共40位，跨越存储单元边界
    // 具体如何存储是实现定义的
};
```

## 位字段 vs 手动位操作

| 对比 | 位字段 | 手动位操作 |
|------|--------|-----------|
| 可读性 | 高 | 需要注释 |
| 可移植性 | 实现定义 | 完全可控 |
| 性能 | 编译器优化 | 通常更快 |
| 灵活性 | 受限于语法 | 完全灵活 |

```c
// 位字段方式
struct Fields {
    unsigned int mode : 3;
    unsigned int enable : 1;
};

// 手动位操作方式
unsigned int reg;
#define MODE_MASK    0x07
#define MODE_SHIFT   0
#define ENABLE_BIT   (1 << 3)

// 设置
reg = (reg & ~MODE_MASK) | (mode << MODE_SHIFT);
reg |= ENABLE_BIT;

// 读取
int mode = (reg & MODE_MASK) >> MODE_SHIFT;
int enable = (reg >> 3) & 1;
```

## 关键要点

- 位字段允许精确控制结构体成员的位宽
- 常用于网络协议报头、硬件寄存器、嵌入式系统
- 位字段成员不能取地址，不能组成数组
- 位字段的存储布局是实现定义的（编译器相关）
- 对于需要精确控制位布局的场景，手动位操作更可靠
- 位字段的对齐和填充遵循编译器的默认行为
