# 16 - 位运算与加密

## 概述

位运算是加密算法的基础操作之一。从最简单的异或加密到复杂的分组密码（如 AES），位运算无处不在。

## 1. 异或加密

异或加密是最简单的对称加密：用相同的密钥加密和解密。

```c
#include <stdio.h>
#include <string.h>

// 基本异或加密
void xor_encrypt(char *data, int data_len, char key) {
    for (int i = 0; i < data_len; i++) {
        data[i] ^= key;
    }
}

int main(void) {
    char message[] = "Hello, World!";
    char key = 0x5A;

    printf("原始: %s\n", message);

    xor_encrypt(message, strlen(message), key);
    printf("加密: ");
    for (int i = 0; i < (int)strlen(message); i++)
        printf("%02X ", (unsigned char)message[i]);
    printf("\n");

    xor_encrypt(message, strlen(message), key);  // 再次异或 = 解密
    printf("解密: %s\n", message);

    return 0;
}
```

## 2. 多字节密钥异或加密

```c
#include <stdio.h>
#include <string.h>

void xor_encrypt_key(char *data, int data_len,
                     const char *key, int key_len) {
    for (int i = 0; i < data_len; i++) {
        data[i] ^= key[i % key_len];  // 循环使用密钥
    }
}

int main(void) {
    char message[] = "C语言位运算很有趣";
    char key[] = "MySecretKey";

    printf("原始: %s\n", message);

    xor_encrypt_key(message, strlen(message), key, strlen(key));
    printf("加密后（十六进制）: ");
    for (int i = 0; i < (int)strlen(message); i++)
        printf("%02X", (unsigned char)message[i]);
    printf("\n");

    xor_encrypt_key(message, strlen(message), key, strlen(key));
    printf("解密: %s\n", message);

    return 0;
}
```

## 3. 流密码思想

```c
#include <stdio.h>
#include <string.h>

// 简单的伪随机密钥流生成器
unsigned int lfsr = 0xACE1u;  // 线性反馈移位寄存器种子

unsigned char next_key_byte(void) {
    // 16位LFSR，多项式 x^16 + x^14 + x^13 + x^11 + 1
    unsigned lsb = lfsr & 1;
    lfsr >>= 1;
    if (lsb)
        lfsr ^= 0xB400u;
    return (unsigned char)(lfsr & 0xFF);
}

void stream_encrypt(char *data, int len) {
    for (int i = 0; i < len; i++) {
        data[i] ^= next_key_byte();
    }
}

int main(void) {
    char message[] = "Stream cipher example";

    // 重置种子
    unsigned int seed = lfsr;

    printf("原始: %s\n", message);
    stream_encrypt(message, strlen(message));

    printf("加密: ");
    for (int i = 0; i < (int)strlen(message); i++)
        printf("%02X ", (unsigned char)message[i]);
    printf("\n");

    // 解密：重新初始化种子
    lfsr = seed;
    stream_encrypt(message, strlen(message));
    printf("解密: %s\n", message);

    return 0;
}
```

## 4. 位运算实现简单的 Feistel 网络

Feistel 网络是 DES 等分组密码的基础结构。

```c
#include <stdio.h>
#include <stdint.h>

// 简单的轮函数
static uint32_t round_function(uint32_t half_block, uint32_t subkey) {
    uint32_t result = half_block ^ subkey;
    // 简单的非线性变换
    result = ((result << 3) | (result >> 29)) ^ 0xA5A5A5A5u;
    return result;
}

// 简化的4轮Feistel加密
uint64_t feistel_encrypt(uint64_t block, uint32_t keys[4]) {
    uint32_t left = (uint32_t)(block >> 32);
    uint32_t right = (uint32_t)(block & 0xFFFFFFFF);

    for (int i = 0; i < 4; i++) {
        uint32_t temp = right;
        right = left ^ round_function(right, keys[i]);
        left = temp;
    }

    return ((uint64_t)right << 32) | left;
}

// Feistel 解密（密钥顺序反转）
uint64_t feistel_decrypt(uint64_t block, uint32_t keys[4]) {
    uint32_t left = (uint32_t)(block >> 32);
    uint32_t right = (uint32_t)(block & 0xFFFFFFFF);

    for (int i = 3; i >= 0; i--) {
        uint32_t temp = left;
        left = right ^ round_function(left, keys[i]);
        right = temp;
    }

    return ((uint64_t)left << 32) | right;
}

int main(void) {
    uint32_t keys[4] = {0x12345678, 0x9ABCDEF0, 0xFEDCBA98, 0x76543210};
    uint64_t plaintext = 0x48656C6C6F210000ULL;  // "Hello!"

    printf("明文: 0x%016llX\n", (unsigned long long)plaintext);

    uint64_t ciphertext = feistel_encrypt(plaintext, keys);
    printf("密文: 0x%016llX\n", (unsigned long long)ciphertext);

    uint64_t decrypted = feistel_decrypt(ciphertext, keys);
    printf("解密: 0x%016llX\n", (unsigned long long)decrypted);

    return 0;
}
```

## 5. 位运算混淆

加密算法常使用位运算来增加混淆性：

```c
#include <stdio.h>
#include <stdint.h>

// 简单的位混淆函数
uint32_t bit_mix(uint32_t x) {
    x ^= x >> 16;
    x *= 0x85EBCA6B;
    x ^= x >> 13;
    x *= 0xC2B2AE35;
    x ^= x >> 16;
    return x;
}

int main(void) {
    uint32_t input = 42;
    uint32_t mixed = bit_mix(input);

    printf("输入: %u\n", input);
    printf("混淆后: %u (0x%08X)\n", mixed, mixed);

    return 0;
}
```

## 安全警告

> **注意**：以上示例仅用于教学目的，展示位运算在加密中的应用原理。实际生产环境应使用经过验证的加密库（如 OpenSSL、libsodium），不要自行设计加密算法。

## 关键要点

- 异或加密是最简单的对称加密，但安全性极低
- 流密码通过生成伪随机密钥流与明文异或
- Feistel 网络是分组密码的基础结构（如 DES）
- 现代加密算法（AES、ChaCha20）大量使用位运算
- 实际项目中应使用成熟的加密库，不要自己实现
