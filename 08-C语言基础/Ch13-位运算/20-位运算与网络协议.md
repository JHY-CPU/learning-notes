# 20 - 位运算与网络协议

## IP 地址与子网掩码

IP 地址和子网掩码是位运算在网络编程中最典型的应用。

### 子网掩码运算

```c
#include <stdio.h>
#include <stdint.h>

// 将IP地址转为字符串
void ip_to_str(uint32_t ip, char *buf) {
    sprintf(buf, "%u.%u.%u.%u",
            (ip >> 24) & 0xFF,
            (ip >> 16) & 0xFF,
            (ip >> 8) & 0xFF,
            ip & 0xFF);
}

int main(void) {
    uint32_t ip = 0xC0A80101;   // 192.168.1.1
    uint32_t mask = 0xFFFFFF00;  // 255.255.255.0 (/24)

    char buf[16];

    // 网络地址 = IP & 子网掩码
    uint32_t network = ip & mask;
    ip_to_str(network, buf);
    printf("网络地址: %s\n", buf);  // 192.168.1.0

    // 广播地址 = 网络地址 | ~子网掩码
    uint32_t broadcast = network | ~mask;
    ip_to_str(broadcast, buf);
    printf("广播地址: %s\n", buf);  // 192.168.1.255

    // 主机部分 = IP & ~子网掩码
    uint32_t host_part = ip & ~mask;
    printf("主机号: %u\n", host_part);  // 1

    // 判断两个IP是否在同一子网
    uint32_t ip2 = 0xC0A80102;  // 192.168.1.2
    if ((ip & mask) == (ip2 & mask))
        printf("两个IP在同一子网\n");

    return 0;
}
```

### CIDR 表示法

```c
#include <stdio.h>
#include <stdint.h>

// 从 CIDR 前缀长度生成子网掩码
uint32_t cidr_to_mask(int prefix_len) {
    if (prefix_len == 0) return 0;
    return (~0U) << (32 - prefix_len);
}

// 从子网掩码计算前缀长度
int mask_to_cidr(uint32_t mask) {
    int count = 0;
    while (mask & (1U << (31 - count)))
        count++;
    return count;
}

int main(void) {
    // /24 掩码
    uint32_t mask24 = cidr_to_mask(24);
    printf("/24 掩码: 0x%08X\n", mask24);  // 0xFFFFFF00

    // /16 掩码
    uint32_t mask16 = cidr_to_mask(16);
    printf("/16 掩码: 0x%08X\n", mask16);  // 0xFFFF0000

    // 反向计算
    printf("0xFFFFFF00 = /%d\n", mask_to_cidr(0xFFFFFF00));  // 24

    return 0;
}
```

## 协议字段提取

### 从网络数据包中提取字段

```c
#include <stdio.h>
#include <stdint.h>

// 模拟从网络数据包中解析 IPv4 头部
struct PacketInfo {
    uint8_t  version;
    uint8_t  ihl;
    uint8_t  dscp;
    uint8_t  ttl;
    uint8_t  protocol;
    uint16_t src_port;
    uint16_t dst_port;
};

void parse_ipv4_header(const uint8_t *packet, struct PacketInfo *info) {
    // 版本和首部长度（第1字节）
    info->version = (packet[0] >> 4) & 0x0F;
    info->ihl     = packet[0] & 0x0F;

    // 服务类型（第2字节）
    info->dscp = (packet[1] >> 2) & 0x3F;

    // TTL（第9字节）
    info->ttl = packet[8];

    // 协议（第10字节）
    info->protocol = packet[9];

    // 如果是TCP/UDP，解析端口（假设无IP选项，从第21字节开始）
    if (info->protocol == 6 || info->protocol == 17) {
        int offset = info->ihl * 4;
        info->src_port = (packet[offset] << 8) | packet[offset + 1];
        info->dst_port = (packet[offset + 2] << 8) | packet[offset + 3];
    }
}

const char *protocol_name(uint8_t proto) {
    switch (proto) {
        case 1:  return "ICMP";
        case 6:  return "TCP";
        case 17: return "UDP";
        default: return "Unknown";
    }
}

int main(void) {
    // 模拟一个 TCP 数据包的头部
    uint8_t packet[] = {
        0x45, 0x00, 0x00, 0x3C,  // 版本4, IHL=5, DSCP=0
        0x1C, 0x46, 0x40, 0x00,  // 标识和标志
        0x40, 0x06, 0x00, 0x00,  // TTL=64, 协议=TCP
        0xC0, 0xA8, 0x01, 0x01,  // 源IP: 192.168.1.1
        0xC0, 0xA8, 0x01, 0x02,  // 目标IP: 192.168.1.2
        0x00, 0x50, 0x1F, 0x90,  // 源端口80, 目标端口8080
    };

    struct PacketInfo info;
    parse_ipv4_header(packet, &info);

    printf("版本: %u\n", info.version);
    printf("首部长度: %u 字节\n", info.ihl * 4);
    printf("TTL: %u\n", info.ttl);
    printf("协议: %s\n", protocol_name(info.protocol));
    printf("源端口: %u\n", info.src_port);
    printf("目标端口: %u\n", info.dst_port);

    return 0;
}
```

## 字节序处理

```c
#include <stdio.h>
#include <stdint.h>

// 手动实现字节序转换
uint32_t htonl_manual(uint32_t host) {
    return ((host & 0xFF000000) >> 24) |
           ((host & 0x00FF0000) >> 8)  |
           ((host & 0x0000FF00) << 8)  |
           ((host & 0x000000FF) << 24);
}

uint16_t htons_manual(uint16_t host) {
    return ((host & 0xFF00) >> 8) |
           ((host & 0x00FF) << 8);
}

int main(void) {
    uint32_t ip_host = 0xC0A80101;  // 192.168.1.1
    uint32_t ip_net = htonl_manual(ip_host);

    printf("主机序: 0x%08X\n", ip_host);
    printf("网络序: 0x%08X\n", ip_net);
    printf("转回:   0x%08X\n", htonl_manual(ip_net));

    uint16_t port = 8080;
    printf("\n端口 %u -> 网络序: %u\n", port, htons_manual(port));

    return 0;
}
```

## 关键要点

- IP 地址和子网掩码的核心操作：`&` 取网络地址，`| ~` 取广播地址
- CIDR 前缀长度可以转换为掩码：`~0U << (32 - prefix)`
- 网络协议报头的字段提取依赖位移和掩码操作
- 网络字节序（大端）和主机字节序的转换用位运算实现
- `packet[0] >> 4` 提取高4位，`packet[0] & 0x0F` 提取低4位
