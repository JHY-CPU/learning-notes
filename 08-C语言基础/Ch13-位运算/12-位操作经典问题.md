# 12 - 位操作经典问题

## 1. 判断一个数是否为 2 的幂

### 原理

2 的幂的二进制表示中**只有一个 1**。利用 `n & (n-1)` 消除最低位的 1 后，结果为 0。

```c
#include <stdio.h>

// 判断是否为2的幂
int is_power_of_two(unsigned int n) {
    return (n > 0) && ((n & (n - 1)) == 0);
}

int main(void) {
    for (unsigned int i = 0; i <= 33; i++) {
        if (is_power_of_two(i))
            printf("%u 是2的幂\n", i);
    }
    // 输出: 1, 2, 4, 8, 16, 32
    return 0;
}
```

## 2. 找到大于等于 n 的最小的 2 的幂

### 原理

将 n 的最高位 1 之后的所有位都置为 1，然后加 1。

```c
#include <stdio.h>

// 找到 >= n 的最小的2的幂
unsigned int next_power_of_two(unsigned int n) {
    if (n == 0) return 1;
    n--;                // 处理 n 本身就是2的幂的情况
    n |= n >> 1;        // 将最高位1右边的所有位置1
    n |= n >> 2;
    n |= n >> 4;
    n |= n >> 8;
    n |= n >> 16;
    return n + 1;
}

int main(void) {
    printf("next_pow2(%u) = %u\n", 0, next_power_of_two(0));    // 1
    printf("next_pow2(%u) = %u\n", 5, next_power_of_two(5));    // 8
    printf("next_pow2(%u) = %u\n", 8, next_power_of_two(8));    // 8
    printf("next_pow2(%u) = %u\n", 9, next_power_of_two(9));    // 16
    printf("next_pow2(%u) = %u\n", 1000, next_power_of_two(1000)); // 1024
    return 0;
}
```

## 3. 反转整数的二进制位

### 问题

将一个整数的二进制位顺序反转。例如：`0b10110000` 反转后为 `0b00001101`。

```c
#include <stdio.h>

// 反转32位整数的二进制位
unsigned int reverse_bits(unsigned int n) {
    n = ((n & 0x55555555U) << 1)  | ((n >> 1)  & 0x55555555U);  // 相邻1位交换
    n = ((n & 0x33333333U) << 2)  | ((n >> 2)  & 0x33333333U);  // 相邻2位交换
    n = ((n & 0x0F0F0F0FU) << 4)  | ((n >> 4)  & 0x0F0F0F0FU);  // 相邻4位交换
    n = ((n & 0x00FF00FFU) << 8)  | ((n >> 8)  & 0x00FF00FFU);  // 相邻8位交换
    n = (n << 16) | (n >> 16);                                   // 相邻16位交换
    return n;
}

// 简单方法（逐位反转）
unsigned int reverse_bits_simple(unsigned int n) {
    unsigned int result = 0;
    for (int i = 0; i < 32; i++) {
        result = (result << 1) | (n & 1);
        n >>= 1;
    }
    return result;
}

int main(void) {
    unsigned int n = 0b10110000;
    printf("原始: 0x%02X\n", n);
    printf("反转: 0x%08X\n", reverse_bits(n));
    printf("反转: 0x%08X\n", reverse_bits_simple(n));
    return 0;
}
```

## 4. 不使用额外变量交换两个数

```c
#include <stdio.h>

int main(void) {
    int a = 42, b = 99;
    printf("交换前: a = %d, b = %d\n", a, b);

    a ^= b;
    b ^= a;
    a ^= b;

    printf("交换后: a = %d, b = %d\n", a, b);
    return 0;
}
```

## 5. 找出缺失的数

### 问题

数组包含 0 到 n 中的 n 个数（缺一个），找出缺失的那个。

```c
#include <stdio.h>

int missing_number(int arr[], int n) {
    // n 是数组长度，完整序列是 0..n
    int xor_all = 0;
    int xor_arr = 0;

    for (int i = 0; i <= n; i++)
        xor_all ^= i;         // 0^1^2^...^n

    for (int i = 0; i < n; i++)
        xor_arr ^= arr[i];    // 数组中所有元素异或

    return xor_all ^ xor_arr; // 缺失的数
}

int main(void) {
    int arr[] = {0, 1, 3, 4, 5};  // 缺失2
    int n = sizeof(arr) / sizeof(arr[0]);
    printf("缺失的数: %d\n", missing_number(arr, n));  // 2
    return 0;
}
```

## 6. 只出现一次的数（其余出现三次）

### 问题

数组中除了一个数出现一次，其余都出现三次，找出出现一次的数。

```c
#include <stdio.h>

int single_number(int arr[], int n) {
    int ones = 0, twos = 0;

    for (int i = 0; i < n; i++) {
        ones = (ones ^ arr[i]) & ~twos;
        twos = (twos ^ arr[i]) & ~ones;
    }

    return ones;
}

int main(void) {
    int arr[] = {2, 2, 3, 2, 5, 5, 5, 7, 7, 7};
    int n = sizeof(arr) / sizeof(arr[0]);
    printf("只出现一次的数: %d\n", single_number(arr, n));  // 3
    return 0;
}
```

## 7. 计算绝对值（不使用条件判断）

```c
#include <stdio.h>

int abs_bitwise(int n) {
    int mask = n >> 31;  // 全0（正数）或全1（负数）
    return (n ^ mask) - mask;
    // 正数: (n ^ 0) - 0 = n
    // 负数: (n ^ 0xFFFFFFFF) - 0xFFFFFFFF = ~n + 1 = -n
}

int main(void) {
    printf("|42| = %d\n", abs_bitwise(42));    // 42
    printf("|-42| = %d\n", abs_bitwise(-42));  // 42
    printf("|0| = %d\n", abs_bitwise(0));      // 0
    return 0;
}
```

## 8. 不使用比较运算找出两个数中的较大值

```c
#include <stdio.h>

int max_bitwise(int a, int b) {
    int diff = a - b;
    int sign = (diff >> 31) & 1;  // 0 表示 a >= b，1 表示 a < b
    return a - (diff & (-sign));
    // a >= b 时：sign=0，返回 a - 0 = a
    // a < b 时：sign=1，返回 a - (a-b) = b
}

int main(void) {
    printf("max(3, 7) = %d\n", max_bitwise(3, 7));    // 7
    printf("max(10, 5) = %d\n", max_bitwise(10, 5));  // 10
    return 0;
}
```

## 9. 判断一个数是否为 4 的幂

```c
#include <stdio.h>

int is_power_of_four(unsigned int n) {
    // 1. 必须是2的幂
    // 2. 1必须在奇数位上（0, 2, 4, ...）
    //    0x55555555 = 01010101...
    return (n > 0) &&
           ((n & (n - 1)) == 0) &&       // 是2的幂
           ((n & 0x55555555U) != 0);      // 1在奇数位置
}

int main(void) {
    for (unsigned int i = 1; i <= 65; i++) {
        if (is_power_of_four(i))
            printf("%u 是4的幂\n", i);
    }
    // 输出: 1, 4, 16, 64
    return 0;
}
```

## 10. 子集生成

```c
#include <stdio.h>

// 用位运算生成集合 {0,1,2,...,n-1} 的所有子集
void generate_subsets(int arr[], int n) {
    int total = 1 << n;  // 共 2^n 个子集

    for (int mask = 0; mask < total; mask++) {
        printf("{ ");
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i))
                printf("%d ", arr[i]);
        }
        printf("}\n");
    }
}

int main(void) {
    int arr[] = {1, 2, 3};
    generate_subsets(arr, 3);
    return 0;
}
```

## 关键要点

- `n & (n-1)` 消除最低位 1，是许多位操作技巧的基础
- 2 的幂只有一个 1，判断条件：`n > 0 && (n & (n-1)) == 0`
- 位反转利用分治交换法，逐级合并
- 异或可以找出数组中出现奇数次的元素
- 位运算可以实现无分支的绝对值、最大值计算
- 子集生成利用 2^n 个整数来枚举所有可能的子集
