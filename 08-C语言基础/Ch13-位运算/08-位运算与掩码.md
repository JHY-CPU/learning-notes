# 08 - 位运算与掩码

## 什么是掩码

掩码（Mask）是一个用于**选择性操作**某些位的二进制模式。通过掩码，可以提取、设置、清除或翻转整数中的特定位。

掩码的核心思想：**掩码中为 1 的位是我们关心的位，为 0 的位是我们不关心的位**。

## 创建掩码

### 1. 单个位的掩码

```c
// 第 n 位为 1，其余为 0
unsigned int mask_bit_n = 1U << n;

// 示例
unsigned int bit0 = 1U << 0;  // 00000001
unsigned int bit3 = 1U << 3;  // 00001000
unsigned int bit7 = 1U << 7;  // 10000000
```

### 2. 连续位的掩码

```c
// 低 n 位全为 1
unsigned int mask_low_n = (1U << n) - 1;

// 示例
unsigned int low4 = (1U << 4) - 1;  // 00001111
unsigned int low8 = (1U << 8) - 1;  // 11111111
```

### 3. 从特定位开始的连续位掩码

```c
// 从第 start 位开始，宽度为 width 的掩码
unsigned int mask = ((1U << width) - 1) << start;

// 示例：从第 3 位开始，宽度为 4
unsigned int field = ((1U << 4) - 1) << 3;  // 01111000
```

### 4. 常用预定义掩码

```c
#define MASK_LOW_BYTE   0x000000FFU   // 低8位
#define MASK_HIGH_BYTE  0xFF000000U   // 高8位
#define MASK_LOW_16     0x0000FFFFU   // 低16位
#define MASK_EVEN_BITS  0x55555555U   // 偶数位（0,2,4,...）
#define MASK_ODD_BITS   0xAAAAAAAAU   // 奇数位（1,3,5,...）
#define MASK_LOW_NIBBLE 0x0FU         // 低4位
```

## 掩码的四大操作

### 1. 提取位（Read / Extract）

使用 `value & mask` 提取掩码指定的位。

```c
#include <stdio.h>

int main(void) {
    unsigned int value = 0b11010110;

    // 提取第3位
    int bit3 = (value >> 3) & 1;
    printf("第3位: %d\n", bit3);  // 0

    // 提取低4位
    unsigned int low_nibble = value & 0x0F;
    printf("低4位: 0x%X\n", low_nibble);  // 0x6

    // 提取第4-7位（高4位）
    unsigned int high_nibble = (value >> 4) & 0x0F;
    printf("高4位: 0x%X\n", high_nibble);  // 0xD

    // 提取中间4位（第2-5位）
    unsigned int middle = (value >> 2) & 0x0F;
    printf("中间4位: 0x%X\n", middle);  // 0x5

    return 0;
}
```

### 2. 设置位（Set）

使用 `value | mask` 将掩码指定的位置 1。

```c
#include <stdio.h>

int main(void) {
    unsigned int value = 0b10100000;

    // 设置第0位
    value |= (1U << 0);
    printf("设置第0位: 0x%02X\n", value);  // 0xA1

    // 同时设置第2位和第4位
    value |= (1U << 2) | (1U << 4);
    printf("设置第2、4位: 0x%02X\n", value);  // 0xB5

    return 0;
}
```

### 3. 清除位（Clear）

使用 `value & ~mask` 将掩码指定的位清零。

```c
#include <stdio.h>

int main(void) {
    unsigned int value = 0xFF;

    // 清除第3位
    value &= ~(1U << 3);
    printf("清除第3位: 0x%02X\n", value);  // 0xF7

    // 清除低4位
    value = 0xFF;
    value &= ~0x0F;
    printf("清除低4位: 0x%02X\n", value);  // 0xF0

    return 0;
}
```

### 4. 翻转位（Toggle / Flip）

使用 `value ^ mask` 将掩码指定的位翻转。

```c
#include <stdio.h>

int main(void) {
    unsigned int value = 0b10100101;

    // 翻转第0位
    value ^= (1U << 0);
    printf("翻转第0位: %u\n", value);  // 164 (10100100)

    // 翻转低4位
    value ^= 0x0F;
    printf("翻转低4位: %u\n", value);  // 171 (10101011)

    return 0;
}
```

## 综合示例：位字段操作

```c
#include <stdio.h>

// 一个32位的配置寄存器定义
// 位 0-2:   模式选择（3位）
// 位 3-7:   时钟分频（5位）
// 位 8-15:  数据值（8位）
// 位 16:    使能标志
// 位 17-31: 保留

#define MODE_MASK     0x07U         // 0000 0111
#define CLOCK_MASK    0xF8U         // 1111 1000
#define CLOCK_SHIFT   3
#define DATA_MASK     0xFF00U       // 位8-15
#define DATA_SHIFT    8
#define ENABLE_BIT    (1U << 16)

void print_binary32(unsigned int n) {
    for (int i = 31; i >= 0; i--) {
        printf("%d", (n >> i) & 1);
        if (i % 8 == 0 && i != 0) printf(" ");
    }
}

int main(void) {
    unsigned int config = 0;

    // 设置模式为 5
    config |= (5 & MODE_MASK);
    printf("设置模式后: 0x%08X\n", config);

    // 设置时钟分频为 12
    config |= (12 << CLOCK_SHIFT) & CLOCK_MASK;
    printf("设置时钟后: 0x%08X\n", config);

    // 设置数据值为 0xAB
    config |= (0xAB << DATA_SHIFT) & DATA_MASK;
    printf("设置数据后: 0x%08X\n", config);

    // 使能
    config |= ENABLE_BIT;
    printf("使能后:     0x%08X\n", config);

    printf("\n完整二进制: ");
    print_binary32(config);
    printf("\n");

    // 读取各个字段
    printf("\n读取字段:\n");
    printf("模式: %u\n", config & MODE_MASK);
    printf("时钟: %u\n", (config & CLOCK_MASK) >> CLOCK_SHIFT);
    printf("数据: 0x%02X\n", (config & DATA_MASK) >> DATA_SHIFT);
    printf("使能: %s\n", (config & ENABLE_BIT) ? "是" : "否");

    // 修改模式（先清除再设置）
    config = (config & ~MODE_MASK) | 3;
    printf("\n修改模式为3: %u\n", config & MODE_MASK);

    return 0;
}
```

## 动态掩码创建函数

```c
#include <stdio.h>

// 创建从 bit_start 开始，bit_width 位宽的掩码
unsigned int create_mask(int bit_start, int bit_width) {
    if (bit_width <= 0 || bit_width > 32) return 0;
    unsigned int mask = ((1U << bit_width) - 1) << bit_start;
    return mask;
}

// 从 value 中提取从 bit_start 开始的 bit_width 位
unsigned int extract_field(unsigned int value, int bit_start, int bit_width) {
    unsigned int mask = (1U << bit_width) - 1;
    return (value >> bit_start) & mask;
}

// 将 field_value 写入 value 的指定位置
unsigned int set_field(unsigned int value, int bit_start, int bit_width,
                       unsigned int field_value) {
    unsigned int mask = (1U << bit_width) - 1;
    value &= ~(mask << bit_start);                    // 清除原值
    value |= (field_value & mask) << bit_start;       // 写入新值
    return value;
}

int main(void) {
    unsigned int reg = 0xFFFFFFFF;

    // 提取第 8-15 位
    unsigned int field = extract_field(reg, 8, 8);
    printf("第8-15位: 0x%02X\n", field);  // 0xFF

    // 修改第 8-15 位为 0x42
    reg = set_field(reg, 8, 8, 0x42);
    printf("修改后: 0x%08X\n", reg);

    return 0;
}
```

## 关键要点

- 掩码是一个用于选择性操作某些位的二进制模式
- 四大基本操作：提取（`&`）、设置（`|`）、清除（`& ~`）、翻转（`^`）
- `1U << n` 创建单个位的掩码
- `(1U << n) - 1` 创建低 n 位全 1 的掩码
- 对有符号数进行位操作时使用无符号类型避免意外
- 掩码操作在硬件编程、协议解析、配置管理中极为常见
