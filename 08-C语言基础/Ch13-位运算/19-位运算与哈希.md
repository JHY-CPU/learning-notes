# 19 - 位运算与哈希

## 概述

哈希函数是将任意大小的数据映射到固定大小值的函数。位运算在哈希函数中扮演关键角色：用于**混合**输入数据的各个位，确保每一位都能影响最终结果。

## 基本哈希混合技巧

### 整数哈希

```c
#include <stdio.h>
#include <stdint_t.h>

// 经典的整数哈希函数（来自 MurmurHash 思想）
uint32_t hash_int(uint32_t x) {
    x ^= x >> 16;
    x *= 0x85EBCA6B;
    x ^= x >> 13;
    x *= 0xC2B2AE35;
    x ^= x >> 16;
    return x;
}

// Thomas Wang 的整数哈希
uint32_t hash_int_wang(uint32_t x) {
    x = ~x + (x << 15);  // x = (x << 15) - x - 1
    x ^= (x >> 12);
    x += (x << 2);
    x ^= (x >> 4);
    x += (x << 3) + (x << 11);
    x ^= (x >> 16);
    return x;
}

// Bob Jenkins 的 One-at-a-Time 哈希
uint32_t hash_jenkins(const char *key) {
    uint32_t hash = 0;
    while (*key) {
        hash += (unsigned char)*key++;
        hash += (hash << 10);
        hash ^= (hash >> 6);
    }
    hash += (hash << 3);
    hash ^= (hash >> 11);
    hash += (hash << 15);
    return hash;
}

int main(void) {
    // 测试整数哈希
    for (uint32_t i = 0; i < 10; i++) {
        printf("hash(%u) = %u\n", i, hash_int(i));
    }

    // 测试字符串哈希
    printf("\nhash(\"hello\") = %u\n", hash_jenkins("hello"));
    printf("hash(\"world\") = %u\n", hash_jenkins("world"));
    printf("hash(\"hellp\") = %u\n", hash_jenkins("hellp"));

    return 0;
}
```

## 位运算在哈希中的作用

### 1. 混淆（Avalanche Effect）

好的哈希函数需要**雪崩效应**：输入的任何一位变化都应该影响输出的大约一半位。

```c
#include <stdio.h>
#include <stdint.h>

// 演示雪崩效应
uint32_t avalanche_demo(uint32_t x) {
    // 每一步都混合所有位
    x ^= x >> 16;   // 高16位和低16位混合
    x *= 0x45D9F3B;  // 乘法让每一位影响更多位
    x ^= x >> 16;   // 再次混合
    return x;
}

int count_different_bits(uint32_t a, uint32_t b) {
    uint32_t diff = a ^ b;
    int count = 0;
    while (diff) {
        diff &= diff - 1;
        count++;
    }
    return count;
}

int main(void) {
    // 仅相差1位的输入
    uint32_t a = 100;
    uint32_t b = 101;  // 只差最后一位

    uint32_t ha = avalanche_demo(a);
    uint32_t hb = avalanche_demo(b);

    printf("输入: %u, %u (差%d位)\n", a, b,
           count_different_bits(a, b));
    printf("哈希: %u, %u (差%d位)\n", ha, hb,
           count_different_bits(ha, hb));

    return 0;
}
```

### 2. 乘法哈希

利用大质数的乘法来分散哈希值。

```c
#include <stdio.h>
#include <stdint.h>

// 使用位运算的乘法哈希
uint32_t multiply_hash(uint32_t key, uint32_t table_size) {
    // 使用黄金比例的近似值
    uint32_t hash = key * 2654435761u;  // 2^32 / φ
    // 取高位（比取低位更均匀）
    return hash >> (32 - __builtin_ctz(table_size));
}

int main(void) {
    uint32_t table_size = 16;  // 必须是2的幂
    for (uint32_t i = 0; i < 20; i++) {
        printf("hash(%u) mod %u = %u\n", i, table_size,
               multiply_hash(i, table_size));
    }
    return 0;
}
```

## 字符串哈希函数

### DJB2 哈希

```c
#include <stdio.h>
#include <stdint.h>

// DJB2 哈希（Daniel J. Bernstein）
uint32_t djb2_hash(const char *str) {
    uint32_t hash = 5381;
    int c;
    while ((c = *str++)) {
        // hash * 33 + c
        hash = ((hash << 5) + hash) + c;
    }
    return hash;
}

int main(void) {
    printf("djb2(\"hello\") = %u\n", djb2_hash("hello"));
    printf("djb2(\"world\") = %u\n", djb2_hash("world"));
    printf("djb2(\"Hello\") = %u\n", djb2_hash("Hello"));  // 注意大小写
    return 0;
}
```

### FNV-1a 哈希

```c
#include <stdio.h>
#include <stdint.h>

// FNV-1a 32位哈希
uint32_t fnv1a_hash(const char *str) {
    uint32_t hash = 0x811C9DC5;  // FNV offset basis
    while (*str) {
        hash ^= (unsigned char)*str++;  // 先异或
        hash *= 0x01000193;             // 再乘以 FNV prime
    }
    return hash;
}

int main(void) {
    printf("fnv1a(\"hello\") = %u\n", fnv1a_hash("hello"));
    printf("fnv1a(\"world\") = %u\n", fnv1a_hash("world"));
    return 0;
}
```

## 布隆过滤器（简化的位哈希）

```c
#include <stdio.h>
#include <string.h>
#include <stdint.h>

#define BLOOM_SIZE 1024

static uint32_t bloom_filter[BLOOM_SIZE / 32];

// 多个哈希函数
uint32_t hash1(const char *str) { return fnv1a_hash(str); }
uint32_t hash2(const char *str) { return djb2_hash(str); }

void bloom_add(const char *str) {
    uint32_t h1 = hash1(str) % BLOOM_SIZE;
    uint32_t h2 = hash2(str) % BLOOM_SIZE;
    bloom_filter[h1 / 32] |= (1U << (h1 % 32));
    bloom_filter[h2 / 32] |= (1U << (h2 % 32));
}

int bloom_check(const char *str) {
    uint32_t h1 = hash1(str) % BLOOM_SIZE;
    uint32_t h2 = hash2(str) % BLOOM_SIZE;
    return (bloom_filter[h1 / 32] & (1U << (h1 % 32))) &&
           (bloom_filter[h2 / 32] & (1U << (h2 % 32)));
}

int main(void) {
    memset(bloom_filter, 0, sizeof(bloom_filter));

    bloom_add("apple");
    bloom_add("banana");
    bloom_add("cherry");

    printf("apple: %s\n", bloom_check("apple") ? "可能在集合中" : "不在");
    printf("grape: %s\n", bloom_check("grape") ? "可能在集合中" : "不在");

    return 0;
}
```

## 关键要点

- 哈希函数用位运算（异或、移位、乘法）来混合输入位
- 雪崩效应：好的哈希函数让每一位输入都影响大量输出位
- `x ^= x >> n` 是最常见的混合操作
- 乘以大质数可以有效地分散位的分布
- 布隆过滤器是位图和多个哈希函数的结合
