# 24 - 位扫描操作

## 概念

位扫描（Bit Scan）操作用于快速定位整数中**值为 1 的位的位置**。这些操作在很多底层算法中至关重要。

## 常见位扫描函数

| 函数 | 含义 | 示例（0b1010） |
|------|------|----------------|
| `ffs` (Find First Set) | 找最低位的 1 的位置（从1开始） | ffs(0b1010) = 2 |
| `fls` (Find Last Set) | 找最高位的 1 的位置（从1开始） | fls(0b1010) = 4 |
| `ctz` (Count Trailing Zeros) | 从最低位开始数0的个数 | ctz(0b1010) = 1 |
| `clz` (Count Leading Zeros) | 从最高位开始数0的个数 | clz(0b1010) = 28（32位） |
| `popcount` | 统计1的个数 | popcount(0b1010) = 2 |

## 自己实现位扫描

### 找最低位的 1

```c
#include <stdio.h>

// 方法1：逐位检查
int find_first_set_v1(unsigned int n) {
    if (n == 0) return 0;
    int pos = 1;
    while (!(n & 1)) {
        n >>= 1;
        pos++;
    }
    return pos;
}

// 方法2：利用 n & (-n) 得到最低位的1
int find_first_set_v2(unsigned int n) {
    if (n == 0) return 0;
    // n & (-n) 得到只含最低位1的值
    unsigned int lowest_bit = n & (-n);

    // 计算这个值的位置
    int pos = 1;
    while (lowest_bit > 1) {
        lowest_bit >>= 1;
        pos++;
    }
    return pos;
}

// 方法3：ctz（Count Trailing Zeros）
int count_trailing_zeros(unsigned int n) {
    if (n == 0) return 32;  // 或者返回 -1 表示无1
    int count = 0;
    while ((n & 1) == 0) {
        count++;
        n >>= 1;
    }
    return count;
}

int main(void) {
    unsigned int n = 0b10100100;  // 164

    printf("ffs(%u) = %d\n", n, find_first_set_v1(n));  // 3
    printf("ctz(%u) = %d\n", n, count_trailing_zeros(n)); // 2

    return 0;
}
```

### 找最高位的 1

```c
#include <stdio.h>

// 方法1：逐位检查
int find_last_set(unsigned int n) {
    if (n == 0) return 0;
    int pos = 0;
    while (n) {
        pos++;
        n >>= 1;
    }
    return pos;
}

// 方法2：clz（Count Leading Zeros）
int count_leading_zeros(unsigned int n) {
    if (n == 0) return 32;
    int count = 0;
    if ((n & 0xFFFF0000) == 0) { count += 16; n <<= 16; }
    if ((n & 0xFF000000) == 0) { count += 8;  n <<= 8;  }
    if ((n & 0xF0000000) == 0) { count += 4;  n <<= 4;  }
    if ((n & 0xC0000000) == 0) { count += 2;  n <<= 2;  }
    if ((n & 0x80000000) == 0) { count += 1; }
    return count;
}

int main(void) {
    unsigned int n = 0b00101100;  // 44

    printf("fls(%u) = %d\n", n, find_last_set(n));    // 6
    printf("clz(%u) = %d\n", n, count_leading_zeros(n)); // 26

    return 0;
}
```

## 使用 GCC/Clang 内置函数

```c
#include <stdio.h>

int main(void) {
    unsigned int n = 0b10100100;  // 164

    // GCC/Clang 内置函数
    printf("ffs(%u) = %d\n", n, __builtin_ffs(n));     // 3
    printf("ctz(%u) = %d\n", n, __builtin_ctz(n));     // 2
    printf("clz(%u) = %d\n", n, __builtin_clz(n));     // 25
    printf("popcount(%u) = %d\n", n, __builtin_popcount(n)); // 3

    // 64位版本
    unsigned long long big = 0x8000000000000000ULL;
    printf("clzll(0x8000...0) = %d\n", __builtin_clzll(big));  // 0

    return 0;
}
```

## 实际应用

### 1. 位集合的迭代

```c
#include <stdio.h>

// 快速迭代集合中的所有元素
void iterate_set(unsigned int set) {
    while (set) {
        int pos = __builtin_ctz(set);  // 最低位元素
        printf("%d ", pos);
        set &= set - 1;  // 删除最低位
    }
    printf("\n");
}

int main(void) {
    unsigned int set = 0b10101100;  // 元素 {2, 3, 5, 7}
    printf("集合中的元素: ");
    iterate_set(set);
    return 0;
}
```

### 2. 最近的 2 的幂

```c
#include <stdio.h>

// 找 <= n 的最大2的幂
unsigned int floor_power_of_two(unsigned int n) {
    if (n == 0) return 0;
    return 1U << (31 - __builtin_clz(n));
}

// 找 >= n 的最小2的幂
unsigned int ceil_power_of_two(unsigned int n) {
    if (n <= 1) return 1;
    return 1U << (32 - __builtin_clz(n - 1));
}

int main(void) {
    printf("floor_pow2(100) = %u\n", floor_power_of_two(100));  // 64
    printf("ceil_pow2(100) = %u\n", ceil_power_of_two(100));    // 128
    printf("floor_pow2(64) = %u\n", floor_power_of_two(64));    // 64
    printf("ceil_pow2(64) = %u\n", ceil_power_of_two(64));      // 64
    return 0;
}
```

### 3. 高效的哈希表探查

```c
#include <stdio.h>

// 使用位扫描找到空闲的桶
int find_free_bucket(unsigned int occupied_mask, int max_buckets) {
    if (occupied_mask == (unsigned int)((1U << max_buckets) - 1))
        return -1;  // 所有桶都满了
    // 找到第一个为0的位
    return __builtin_ctz(~occupied_mask);
}

int main(void) {
    unsigned int occupied = 0b1011;  // 桶0、1、3被占用
    int free_bucket = find_free_bucket(occupied, 4);
    printf("空闲桶: %d\n", free_bucket);  // 2
    return 0;
}
```

## C23 标准函数

C23 标准在 `<stdbit.h>` 中引入了标准位扫描函数：

```c
// C23 <stdbit.h>
// stdc_first_trailing_one(x)  — 对应 ffs
// stdc_first_trailing_zero(x) — 对应 ctz 取反
// stdc_first_leading_one(x)   — 对应 fls
// stdc_first_leading_zero(x)  — 对应 clz 取反
// stdc_count_ones(x)          — 对应 popcount
// stdc_count_zeros(x)         — 对应 位宽 - popcount
```

## 关键要点

- `ffs` 找最低位1的位置，`fls` 找最高位1的位置
- `ctz` 数最低位开始的0的个数，`clz` 数最高位开始的0的个数
- `n & (-n)` 提取最低位的 1
- `n & (n-1)` 清除最低位的 1
- GCC/Clang 提供高效的内置函数，通常编译为单条 CPU 指令（如 x86 的 BSF/BSR/LZCNT/TZCNT）
- 位扫描在位集合迭代、哈希表、内存分配器中有广泛应用
