# 23 - 位运算优化技巧

## 用位运算替代条件判断

位运算可以消除分支，提高 CPU 流水线效率。

### 1. 无分支取最小值

```c
#include <stdio.h>

// 无分支取两个非负整数的最小值
int min_no_branch(int a, int b) {
    return b ^ ((a ^ b) & -(a < b));
    // a < b 时，-(a<b) = -1 (全1)，结果为 a
    // a >= b 时，-(a<b) = 0，结果为 b
}

int main(void) {
    printf("min(3, 7) = %d\n", min_no_branch(3, 7));  // 3
    printf("min(10, 5) = %d\n", min_no_branch(10, 5));  // 5
    return 0;
}
```

### 2. 无分支绝对值

```c
#include <stdio.h>

int abs_no_branch(int n) {
    int mask = n >> 31;  // 全0或全1
    return (n ^ mask) - mask;
    // 正数: mask=0, (n^0)-0 = n
    // 负数: mask=-1, (n^-1)-(-1) = ~n+1 = -n
}

int main(void) {
    printf("|42| = %d\n", abs_no_branch(42));
    printf("|-42| = %d\n", abs_no_branch(-42));
    return 0;
}
```

### 3. 无分支符号函数

```c
#include <stdio.h>

// sign(x): 返回 -1, 0, 或 1
int sign_no_branch(int n) {
    return (n > 0) - (n < 0);
    // 或者用位运算：
    // return (n >> 31) | ((-n) >> 31);
}

int main(void) {
    printf("sign(5) = %d\n", sign_no_branch(5));    // 1
    printf("sign(-3) = %d\n", sign_no_branch(-3));   // -1
    printf("sign(0) = %d\n", sign_no_branch(0));     // 0
    return 0;
}
```

## 分支消除

### 4. 替代 if-else 的位运算

```c
#include <stdio.h>

int main(void) {
    // 传统写法
    int x = 5;
    int result;

    if (x & 1)
        result = x * 3 + 1;
    else
        result = x / 2;

    // 位运算写法（消除分支）
    int is_odd = x & 1;
    int if_odd = x * 3 + 1;
    int if_even = x >> 1;
    result = if_even ^ ((if_odd ^ if_even) & (-is_odd));

    printf("result = %d\n", result);

    return 0;
}
```

## 快速数学运算

### 5. 检查是否为 2 的幂

```c
// 传统方法：循环
int is_pow2_loop(unsigned int n) {
    if (n == 0) return 0;
    while (n > 1) {
        if (n & 1) return 0;
        n >>= 1;
    }
    return 1;
}

// 位运算方法：O(1)
int is_pow2(unsigned int n) {
    return n && !(n & (n - 1));
}
```

### 6. 向上取整到 2 的幂

```c
#include <stdio.h>

// 将 n 向上取整到最近的 2 的幂
unsigned int ceil_pow2(unsigned int n) {
    if (n == 0) return 1;
    n--;
    n |= n >> 1;
    n |= n >> 2;
    n |= n >> 4;
    n |= n >> 8;
    n |= n >> 16;
    return n + 1;
}

int main(void) {
    printf("ceil_pow2(5) = %u\n", ceil_pow2(5));    // 8
    printf("ceil_pow2(8) = %u\n", ceil_pow2(8));    // 8
    printf("ceil_pow2(9) = %u\n", ceil_pow2(9));    // 16
    return 0;
}
```

### 7. 快速计算 log2（向下取整）

```c
#include <stdio.h>

int fast_log2(unsigned int n) {
    int result = 0;
    if (n >= 1 << 16) { result += 16; n >>= 16; }
    if (n >= 1 << 8)  { result += 8;  n >>= 8;  }
    if (n >= 1 << 4)  { result += 4;  n >>= 4;  }
    if (n >= 1 << 2)  { result += 2;  n >>= 2;  }
    if (n >= 1 << 1)  { result += 1; }
    return result;
}

int main(void) {
    printf("log2(1) = %d\n", fast_log2(1));    // 0
    printf("log2(7) = %d\n", fast_log2(7));    // 2
    printf("log2(8) = %d\n", fast_log2(8));    // 3
    printf("log2(1023) = %d\n", fast_log2(1023)); // 9
    return 0;
}
```

## 位运算循环技巧

### 8. 循环左移和右移

```c
#include <stdio.h>
#include <stdint.h>

// 循环左移
uint32_t rotate_left(uint32_t x, int n) {
    n &= 31;  // 确保 n < 32
    return (x << n) | (x >> (32 - n));
}

// 循环右移
uint32_t rotate_right(uint32_t x, int n) {
    n &= 31;
    return (x >> n) | (x << (32 - n));
}

int main(void) {
    uint32_t x = 0x12345678;

    printf("原始: 0x%08X\n", x);
    printf("循环左移4位: 0x%08X\n", rotate_left(x, 4));   // 0x23456781
    printf("循环右移4位: 0x%08X\n", rotate_right(x, 4));  // 0x81234567

    return 0;
}
```

### 9. 交替位交错

```c
#include <stdio.h>
#include <stdint.h>

// 将两个16位数交错合并为32位
// a 的位放在偶数位置，b 的位放在奇数位置
uint32_t interleave(uint16_t a, uint16_t b) {
    uint32_t x = a;
    uint32_t y = b;

    x = (x | (x << 8)) & 0x00FF00FF;
    x = (x | (x << 4)) & 0x0F0F0F0F;
    x = (x | (x << 2)) & 0x33333333;
    x = (x | (x << 1)) & 0x55555555;

    y = (y | (y << 8)) & 0x00FF00FF;
    y = (y | (y << 4)) & 0x0F0F0F0F;
    y = (y | (y << 2)) & 0x33333333;
    y = (y | (y << 1)) & 0x55555555;

    return x | (y << 1);
}

int main(void) {
    uint16_t a = 0xAAAA, b = 0x5555;
    printf("交错结果: 0x%08X\n", interleave(a, b));  // 0xFFFFFFFF
    return 0;
}
```

## 性能对比

```c
#include <stdio.h>
#include <time.h>
#include <stdint.h>

int main(void) {
    const int N = 100000000;
    volatile int result = 0;
    clock_t start, end;

    // 测试 if-else
    start = clock();
    for (int i = 0; i < N; i++) {
        if (i & 1) result = i * 3 + 1;
        else result = i / 2;
    }
    end = clock();
    printf("if-else: %.3f 秒\n", (double)(end - start) / CLOCKS_PER_SEC);

    // 测试位运算
    start = clock();
    for (int i = 0; i < N; i++) {
        int odd = i & 1;
        result = (i >> 1) ^ (((i * 3 + 1) ^ (i >> 1)) & (-odd));
    }
    end = clock();
    printf("位运算: %.3f 秒\n", (double)(end - start) / CLOCKS_PER_SEC);

    return 0;
}
```

## 关键要点

- 位运算可以消除分支，减少 CPU 分支预测失败的开销
- `mask & (a ^ b) ^ b` 模式可以无分支选择
- `x & (x-1)` 清除最低位的 1，`x & (-x)` 取最低位的 1
- 循环移位用 `(x << n) | (x >> (32-n))` 实现
- 位运算优化在高频循环中效果最明显
- 现代编译器已经能自动进行部分位运算优化
