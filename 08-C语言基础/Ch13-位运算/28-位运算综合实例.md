## 实例1：简单的内存分配器

使用位图实现一个简单的内存分配器，跟踪固定大小的内存块。

```c
#include <stdio.h>
#include <string.h>
#include <stdint.h>

#define POOL_SIZE 256         // 内存池大小
#define BLOCK_SIZE 16         // 每个块大小
#define NUM_BLOCKS (POOL_SIZE / BLOCK_SIZE)  // 16个块

static uint8_t memory_pool[POOL_SIZE];
static uint32_t bitmap = 0;  // 16个块用16位表示

// 分配 n 个连续的块
int alloc_blocks(int n) {
    if (n <= 0 || n > NUM_BLOCKS) return -1;

    unsigned int mask = (1U << n) - 1;  // n 个连续的 1

    for (int i = 0; i <= NUM_BLOCKS - n; i++) {
        // 检查从位置 i 开始的 n 个块是否都空闲
        if ((bitmap & (mask << i)) == 0) {
            bitmap |= (mask << i);  // 标记为已分配
            return i;  // 返回起始块号
        }
    }
    return -1;  // 无足够连续空间
}

// 释放 n 个块
void free_blocks(int start, int n) {
    unsigned int mask = ((1U << n) - 1) << start;
    bitmap &= ~mask;
}

// 获取块的内存地址
void *get_block_addr(int block_num) {
    return &memory_pool[block_num * BLOCK_SIZE];
}

// 打印内存状态
void print_memory_status(void) {
    printf("内存状态: [");
    for (int i = NUM_BLOCKS - 1; i >= 0; i--) {
        printf("%c", (bitmap & (1U << i)) ? '#' : '.');
    }
    printf("] 已用: %d/%d\n", __builtin_popcount(bitmap), NUM_BLOCKS);
}

int main(void) {
    memset(memory_pool, 0, sizeof(memory_pool));

    printf("=== 简单内存分配器 ===\n\n");
    print_memory_status();

    int b1 = alloc_blocks(3);
    printf("分配3块 -> 起始块: %d\n", b1);
    print_memory_status();

    int b2 = alloc_blocks(4);
    printf("分配4块 -> 起始块: %d\n", b2);
    print_memory_status();

    int b3 = alloc_blocks(2);
    printf("分配2块 -> 起始块: %d\n", b3);
    print_memory_status();

    free_blocks(b1, 3);
    printf("释放块%d开始的3块\n", b1);
    print_memory_status();

    int b4 = alloc_blocks(5);
    printf("分配5块 -> 起始块: %d\n", b4);
    print_memory_status();

    return 0;
}
```

## 实例2：权限控制系统

一个完整的多用户、多资源权限控制系统。

```c
#include <stdio.h>
#include <string.h>

// 权限定义
#define PERM_READ    (1U << 0)
#define PERM_WRITE   (1U << 1)
#define PERM_EXECUTE (1U << 2)
#define PERM_DELETE  (1U << 3)
#define PERM_ADMIN   (1U << 4)

#define MAX_USERS  10
#define MAX_RESOURCES 5

typedef struct {
    char name[32];
    unsigned int permissions[MAX_RESOURCES];  // 每个资源的权限
} User;

User users[MAX_USERS];
int user_count = 0;

// 添加用户
int add_user(const char *name) {
    if (user_count >= MAX_USERS) return -1;
    strncpy(users[user_count].name, name, 31);
    memset(users[user_count].permissions, 0, sizeof(users[user_count].permissions));
    return user_count++;
}

// 授予权限
void grant(int user_id, int resource, unsigned int perm) {
    if (user_id >= 0 && user_id < user_count)
        users[user_id].permissions[resource] |= perm;
}

// 撤销权限
void revoke(int user_id, int resource, unsigned int perm) {
    if (user_id >= 0 && user_id < user_count)
        users[user_id].permissions[resource] &= ~perm;
}

// 检查权限
int check(int user_id, int resource, unsigned int perm) {
    if (user_id >= 0 && user_id < user_count)
        return (users[user_id].permissions[resource] & perm) == perm;
    return 0;
}

// 打印权限
void print_permissions(int user_id, int resource) {
    if (user_id < 0 || user_id >= user_count) return;
    unsigned int perm = users[user_id].permissions[resource];
    printf("%s 在资源%d: ", users[user_id].name, resource);
    if (perm & PERM_READ)    printf("R");
    if (perm & PERM_WRITE)   printf("W");
    if (perm & PERM_EXECUTE) printf("X");
    if (perm & PERM_DELETE)  printf("D");
    if (perm & PERM_ADMIN)   printf("A");
    if (perm == 0)           printf("(无权限)");
    printf("\n");
}

int main(void) {
    printf("=== 权限控制系统 ===\n\n");

    int alice = add_user("Alice");
    int bob = add_user("Bob");

    grant(alice, 0, PERM_READ | PERM_WRITE);
    grant(alice, 1, PERM_READ);
    grant(bob, 0, PERM_READ);
    grant(bob, 1, PERM_READ | PERM_WRITE | PERM_EXECUTE);

    print_permissions(alice, 0);
    print_permissions(alice, 1);
    print_permissions(bob, 0);
    print_permissions(bob, 1);

    printf("\n检查: Alice 读资源0: %s\n",
           check(alice, 0, PERM_READ) ? "允许" : "拒绝");
    printf("检查: Alice 执行资源0: %s\n",
           check(alice, 0, PERM_EXECUTE) ? "允许" : "拒绝");

    grant(alice, 0, PERM_EXECUTE);
    printf("\n授予权限后: ");
    print_permissions(alice, 0);

    revoke(bob, 1, PERM_EXECUTE);
    printf("撤销权限后: ");
    print_permissions(bob, 1);

    return 0;
}
```

## 实例3：有限集合的位运算计算

使用位运算求解组合数学问题。

```c
#include <stdio.h>

// 用位运算生成 n 个元素的所有 k 元子集
void combinations(int n, int k) {
    int comb = (1 << k) - 1;  // 最小的 k 个 1
    int count = 0;

    while (comb < (1 << n)) {
        printf("{ ");
        for (int i = 0; i < n; i++) {
            if (comb & (1 << i))
                printf("%c ", 'A' + i);
        }
        printf("}\n");
        count++;

        // Gosper's Hack
        int c = comb & (-comb);
        int r = comb + c;
        comb = (((r ^ comb) >> 2) / c) | r;
    }
    printf("共 %d 种组合\n", count);
}

int main(void) {
    printf("从 {A,B,C,D} 中选 2 个的所有组合:\n");
    combinations(4, 2);
    return 0;
}
```

## 实例4：简单的状态机

使用位运算实现有限状态机。

```c
#include <stdio.h>
#include <stdint.h>

// 定义状态（用位表示）
#define STATE_IDLE      (1U << 0)
#define STATE_RUNNING   (1U << 1)
#define STATE_PAUSED    (1U << 2)
#define STATE_STOPPED   (1U << 3)
#define STATE_ERROR     (1U << 4)

// 定义事件
typedef enum {
    EVT_START,
    EVT_PAUSE,
    EVT_RESUME,
    EVT_STOP,
    EVT_ERROR,
    EVT_RESET
} Event;

// 状态转换表
typedef struct {
    unsigned int current_state;
    Event event;
    unsigned int next_state;
} Transition;

Transition transitions[] = {
    {STATE_IDLE,    EVT_START,   STATE_RUNNING},
    {STATE_RUNNING, EVT_PAUSE,   STATE_PAUSED},
    {STATE_RUNNING, EVT_STOP,    STATE_STOPPED},
    {STATE_RUNNING, EVT_ERROR,   STATE_ERROR},
    {STATE_PAUSED,  EVT_RESUME,  STATE_RUNNING},
    {STATE_PAUSED,  EVT_STOP,    STATE_STOPPED},
    {STATE_ERROR,   EVT_RESET,   STATE_IDLE},
    {STATE_STOPPED, EVT_RESET,   STATE_IDLE},
};

const char *state_name(unsigned int state) {
    switch (state) {
        case STATE_IDLE:    return "空闲";
        case STATE_RUNNING: return "运行中";
        case STATE_PAUSED:  return "暂停";
        case STATE_STOPPED: return "停止";
        case STATE_ERROR:   return "错误";
        default:            return "未知";
    }
}

const char *event_name(Event evt) {
    switch (evt) {
        case EVT_START:  return "启动";
        case EVT_PAUSE:  return "暂停";
        case EVT_RESUME: return "恢复";
        case EVT_STOP:   return "停止";
        case EVT_ERROR:  return "出错";
        case EVT_RESET:  return "重置";
        default:         return "未知";
    }
}

// 处理事件
unsigned int process_event(unsigned int current, Event evt) {
    int n = sizeof(transitions) / sizeof(transitions[0]);
    for (int i = 0; i < n; i++) {
        if (transitions[i].current_state == current &&
            transitions[i].event == evt) {
            return transitions[i].next_state;
        }
    }
    return current;  // 无匹配转换，保持当前状态
}

int main(void) {
    unsigned int state = STATE_IDLE;

    Event events[] = {EVT_START, EVT_PAUSE, EVT_RESUME, EVT_STOP, EVT_RESET};
    int n = sizeof(events) / sizeof(events[0]);

    printf("=== 状态机演示 ===\n");
    printf("当前状态: %s\n", state_name(state));

    for (int i = 0; i < n; i++) {
        printf("事件: %s -> ", event_name(events[i]));
        state = process_event(state, events[i]);
        printf("状态: %s\n", state_name(state));
    }

    return 0;
}
```

## 实例5：简易布隆过滤器

```c
#include <stdio.h>
#include <string.h>
#include <stdint.h>

#define BLOOM_BITS 1024
#define BLOOM_WORDS (BLOOM_BITS / 32)

static uint32_t bloom[BLOOM_WORDS];

// 两个简单的哈希函数
uint32_t hash1(const char *s) {
    uint32_t h = 5381;
    while (*s) h = ((h << 5) + h) + (unsigned char)*s++;
    return h;
}

uint32_t hash2(const char *s) {
    uint32_t h = 0x811C9DC5;
    while (*s) { h ^= (unsigned char)*s++; h *= 0x01000193; }
    return h;
}

void bloom_add(const char *s) {
    uint32_t h1 = hash1(s) % BLOOM_BITS;
    uint32_t h2 = hash2(s) % BLOOM_BITS;
    bloom[h1 / 32] |= (1U << (h1 % 32));
    bloom[h2 / 32] |= (1U << (h2 % 32));
}

int bloom_contains(const char *s) {
    uint32_t h1 = hash1(s) % BLOOM_BITS;
    uint32_t h2 = hash2(s) % BLOOM_BITS;
    return (bloom[h1 / 32] & (1U << (h1 % 32))) &&
           (bloom[h2 / 32] & (1U << (h2 % 32)));
}

int main(void) {
    memset(bloom, 0, sizeof(bloom));

    const char *words[] = {"apple", "banana", "cherry", "date", "elderberry"};
    for (int i = 0; i < 5; i++) bloom_add(words[i]);

    printf("apple: %s\n", bloom_contains("apple") ? "存在" : "不存在");
    printf("grape: %s\n", bloom_contains("grape") ? "存在(误判)" : "不存在");
    printf("cherry: %s\n", bloom_contains("cherry") ? "存在" : "不存在");

    return 0;
}
```

## 关键要点

- 位运算是实现高效内存管理的基础工具
- 权限控制系统利用位的独立性实现紧凑的权限存储
- 组合枚举可以通过位集合和 Gosper's Hack 高效实现
- 状态机可以用位掩码来表示和操作状态
- 布隆过滤器结合了位图和多个哈希函数，是高效的概率数据结构
- 这些实例展示了位运算在系统编程、算法、数据结构中的广泛应用
