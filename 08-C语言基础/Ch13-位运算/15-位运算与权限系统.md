# 15 - 位运算与权限系统

## Linux 文件权限概述

Linux 文件权限系统是位运算最经典的应用之一。每个文件有三组权限（所有者、所属组、其他用户），每组包含读（r）、写（w）、执行（x）三种权限。

```
rwx rwx rwx
│││ │││ │││
│││ │││ └── 其他用户的权限
│││ └── 所属组的权限
└── 所有者的权限

每种权限对应一个位：
r = 4 (100)
w = 2 (010)
x = 1 (001)
```

## 基本权限操作

```c
#include <stdio.h>

// 定义权限常量
#define OWNER_R   0400  // 八进制: 100 000 000
#define OWNER_W   0200
#define OWNER_X   0100
#define GROUP_R   0040
#define GROUP_W   0020
#define GROUP_X   0010
#define OTHER_R   0004
#define OTHER_W   0002
#define OTHER_X   0001

// 常用组合
#define OWNER_RW  (OWNER_R | OWNER_W)           // 0600
#define OWNER_ALL (OWNER_R | OWNER_W | OWNER_X) // 0700
#define ALL_R     (OWNER_R | GROUP_R | OTHER_R)  // 0444
#define ALL_RW    (0666)
#define ALL_RWX   (0777)

// 检查权限
void check_permissions(int mode) {
    printf("权限: %04o\n", mode);
    printf("  所有者: %c%c%c\n",
           (mode & OWNER_R) ? 'r' : '-',
           (mode & OWNER_W) ? 'w' : '-',
           (mode & OWNER_X) ? 'x' : '-');
    printf("  组:     %c%c%c\n",
           (mode & GROUP_R) ? 'r' : '-',
           (mode & GROUP_W) ? 'w' : '-',
           (mode & GROUP_X) ? 'x' : '-');
    printf("  其他:   %c%c%c\n",
           (mode & OTHER_R) ? 'r' : '-',
           (mode & OTHER_W) ? 'w' : '-',
           (mode & OTHER_X) ? 'x' : '-');
}

int main(void) {
    int mode = OWNER_RW | GROUP_R | OTHER_R;  // 0644
    check_permissions(mode);
    return 0;
}
```

## 权限的增删改查

```c
#include <stdio.h>

// 授予权限
int grant_permission(int mode, int perm) {
    return mode | perm;
}

// 撤销权限
int revoke_permission(int mode, int perm) {
    return mode & ~perm;
}

// 切换权限
int toggle_permission(int mode, int perm) {
    return mode ^ perm;
}

// 检查是否拥有权限
int has_permission(int mode, int perm) {
    return (mode & perm) == perm;
}

int main(void) {
    int mode = 0644;  // rw-r--r--

    printf("初始权限: %04o\n", mode);

    // 授予所有者执行权限
    mode = grant_permission(mode, 0100);
    printf("添加执行: %04o\n", mode);  // 0744

    // 检查权限
    if (has_permission(mode, 0400))
        printf("拥有所有者读权限\n");

    // 授予组写权限
    mode = grant_permission(mode, 0020);
    printf("组可写: %04o\n", mode);  // 0764

    // 撤销其他用户读权限
    mode = revoke_permission(mode, 0004);
    printf("撤销其他读: %04o\n", mode);  // 0760

    return 0;
}
```

## 使用符号表示法

```c
#include <stdio.h>

// 将数字权限转换为符号字符串
void mode_to_string(int mode, char *str) {
    str[0] = (mode & 0400) ? 'r' : '-';
    str[1] = (mode & 0200) ? 'w' : '-';
    str[2] = (mode & 0100) ? 'x' : '-';
    str[3] = (mode & 0040) ? 'r' : '-';
    str[4] = (mode & 0020) ? 'w' : '-';
    str[5] = (mode & 0010) ? 'x' : '-';
    str[6] = (mode & 0004) ? 'r' : '-';
    str[7] = (mode & 0002) ? 'w' : '-';
    str[8] = (mode & 0001) ? 'x' : '-';
    str[9] = '\0';
}

int main(void) {
    char perm_str[10];

    mode_to_string(0755, perm_str);
    printf("0755 = %s\n", perm_str);  // rwxr-xr-x

    mode_to_string(0644, perm_str);
    printf("0644 = %s\n", perm_str);  // rw-r--r--

    mode_to_string(0600, perm_str);
    printf("0600 = %s\n", perm_str);  // rw-------

    return 0;
}
```

## 多用户权限系统

```c
#include <stdio.h>

// 定义多种权限类型
#define PERM_READ    (1 << 0)
#define PERM_WRITE   (1 << 1)
#define PERM_EXECUTE (1 << 2)
#define PERM_DELETE  (1 << 3)
#define PERM_ADMIN   (1 << 4)

// 角色定义
#define ROLE_GUEST   PERM_READ
#define ROLE_USER    (PERM_READ | PERM_WRITE)
#define ROLE_EDITOR  (PERM_READ | PERM_WRITE | PERM_DELETE)
#define ROLE_ADMIN   (PERM_READ | PERM_WRITE | PERM_EXECUTE | PERM_DELETE | PERM_ADMIN)

const char *perm_name(int perm) {
    switch (perm) {
        case PERM_READ:    return "读取";
        case PERM_WRITE:   return "写入";
        case PERM_EXECUTE: return "执行";
        case PERM_DELETE:  return "删除";
        case PERM_ADMIN:   return "管理";
        default:           return "未知";
    }
}

void print_permissions(int role) {
    printf("权限列表: ");
    for (int i = 0; i < 5; i++) {
        if (role & (1 << i))
            printf("[%s] ", perm_name(1 << i));
    }
    printf("\n");
}

int main(void) {
    printf("游客: "); print_permissions(ROLE_GUEST);
    printf("用户: "); print_permissions(ROLE_USER);
    printf("编辑: "); print_permissions(ROLE_EDITOR);
    printf("管理员: "); print_permissions(ROLE_ADMIN);

    // 检查是否有写权限
    if (ROLE_USER & PERM_WRITE)
        printf("\n用户角色有写权限\n");

    // 临时提升权限
    int user_role = ROLE_USER;
    printf("\n当前: "); print_permissions(user_role);

    user_role |= PERM_EXECUTE;  // 临时添加执行权限
    printf("临时提升: "); print_permissions(user_role);

    user_role &= ~PERM_EXECUTE;  // 恢复
    printf("恢复后: "); print_permissions(user_role);

    return 0;
}
```

## 关键要点

- Linux 权限用 9 位（3 组 x 3 种权限）表示，常用八进制表示
- `|=` 授予权限，`&= ~` 撤销权限，`&` 检查权限
- 多用户权限系统中，用不同位代表不同操作权限
- 位运算权限系统高效、紧凑，适合网络传输和持久化存储
