# 22 - 端序与字节序

## 概念

字节序（Endianness）是指多字节数据在内存中的**存储顺序**。这是一个与位运算密切相关的话题。

## 大端 vs 小端

以 32 位整数 `0x12345678` 为例：

```
地址:   0x00  0x01  0x02  0x03

大端(Big-Endian):    12    34    56    78
                     ↑高位字节在低地址

小端(Little-Endian): 78    56    34    12
                     ↑低位字节在低地址
```

## 检测当前系统的字节序

```c
#include <stdio.h>

int is_little_endian(void) {
    unsigned int x = 1;
    return *(unsigned char *)&x == 1;
    // 如果最低地址的字节是1，就是小端
}

int main(void) {
    if (is_little_endian())
        printf("当前系统是小端（Little-Endian）\n");
    else
        printf("当前系统是大端（Big-Endian）\n");

    return 0;
}
```

## 字节序转换

### 使用位运算手动实现

```c
#include <stdio.h>
#include <stdint.h>

// 16位字节序转换
uint16_t swap16(uint16_t x) {
    return (x >> 8) | (x << 8);
}

// 32位字节序转换
uint32_t swap32(uint32_t x) {
    return ((x & 0x000000FF) << 24) |
           ((x & 0x0000FF00) << 8)  |
           ((x & 0x00FF0000) >> 8)  |
           ((x & 0xFF000000) >> 24);
}

// 64位字节序转换
uint64_t swap64(uint64_t x) {
    return ((x & 0x00000000000000FFULL) << 56) |
           ((x & 0x000000000000FF00ULL) << 40) |
           ((x & 0x0000000000FF0000ULL) << 24) |
           ((x & 0x00000000FF000000ULL) << 8)  |
           ((x & 0x000000FF00000000ULL) >> 8)  |
           ((x & 0x0000FF0000000000ULL) >> 24) |
           ((x & 0x00FF000000000000ULL) >> 40) |
           ((x & 0xFF00000000000000ULL) >> 56);
}

int main(void) {
    uint32_t original = 0x12345678;
    uint32_t swapped = swap32(original);

    printf("原始: 0x%08X\n", original);
    printf("转换: 0x%08X\n", swapped);

    uint16_t port = 8080;
    printf("\n端口 %u -> 网络序: %u\n", port, swap16(port));

    return 0;
}
```

### 使用编译器内置函数

```c
#include <stdio.h>
#include <stdint.h>

// GCC / Clang 内置函数
uint32_t bswap32_builtin(uint32_t x) {
    return __builtin_bswap32(x);
}

uint64_t bswap64_builtin(uint64_t x) {
    return __builtin_bswap64(x);
}

int main(void) {
    uint32_t x = 0x12345678;
    printf("0x%08X -> 0x%08X\n", x, bswap32_builtin(x));
    return 0;
}
```

## 网络字节序

网络协议使用**大端字节序**（网络字节序）。发送和接收数据时需要转换。

```c
#include <stdio.h>
#include <stdint.h>

// 检测字节序
static inline int is_le(void) {
    uint16_t x = 1;
    return *(uint8_t *)&x;
}

// 主机序 -> 网络序
uint32_t my_htonl(uint32_t host) {
    if (is_le())
        return swap32(host);
    return host;
}

// 网络序 -> 主机序
uint32_t my_ntohl(uint32_t net) {
    if (is_le())
        return swap32(net);
    return net;
}

// 16位版本
uint16_t my_htons(uint16_t host) {
    if (is_le())
        return swap16(host);
    return host;
}

int main(void) {
    uint32_t ip = 0xC0A80101;  // 192.168.1.1

    printf("主机序 IP: 0x%08X\n", ip);
    printf("网络序 IP: 0x%08X\n", my_htonl(ip));
    printf("转回主机序: 0x%08X\n", my_ntohl(my_htonl(ip)));

    uint16_t port = 8080;
    printf("\n端口 主机序: %u, 网络序: %u\n", port, my_htons(port));

    return 0;
}
```

## 内存中查看字节序

```c
#include <stdio.h>
#include <stdint.h>

void print_bytes(const void *ptr, int size) {
    const unsigned char *p = (const unsigned char *)ptr;
    for (int i = 0; i < size; i++) {
        printf("地址 %p: 0x%02X\n", (void *)(p + i), p[i]);
    }
}

int main(void) {
    uint32_t value = 0x12345678;

    printf("值: 0x%08X\n内存布局:\n", value);
    print_bytes(&value, sizeof(value));

    return 0;
}
```

小端系统上的输出：
```
值: 0x12345678
内存布局:
地址 0x7fff: 0x78   (最低地址存最低字节)
地址 0x8000: 0x56
地址 0x8001: 0x34
地址 0x8002: 0x12   (最高地址存最高字节)
```

## 实际应用：解析二进制文件格式

```c
#include <stdio.h>
#include <stdint.h>

// 从字节流中解析大端32位整数
uint32_t read_be32(const uint8_t *data) {
    return ((uint32_t)data[0] << 24) |
           ((uint32_t)data[1] << 16) |
           ((uint32_t)data[2] << 8)  |
           ((uint32_t)data[3]);
}

// 从字节流中解析小端32位整数
uint32_t read_le32(const uint8_t *data) {
    return ((uint32_t)data[3] << 24) |
           ((uint32_t)data[2] << 16) |
           ((uint32_t)data[1] << 8)  |
           ((uint32_t)data[0]);
}

int main(void) {
    uint8_t data[] = {0x12, 0x34, 0x56, 0x78};

    printf("大端解析: 0x%08X\n", read_be32(data));  // 0x12345678
    printf("小端解析: 0x%08X\n", read_le32(data));  // 0x78563412

    return 0;
}
```

## 关键要点

- 大端：高位字节在低地址（网络字节序）
- 小端：低位字节在低地址（x86/ARM 默认）
- 网络编程必须使用网络字节序（大端），通过 `htonl`/`ntohl` 转换
- 字节序转换的本质就是位运算（移位和掩码）
- 解析二进制格式时，必须根据格式规范选择正确的字节序
