# 09 - 标志位管理

## 概念

标志位管理是位运算最常见的应用之一：用一个整数的**不同位**来表示多个**布尔状态**（开/关、真/假），而不是为每个状态分配一个独立的变量。

这种方式的优点：
- **节省内存**：32 个布尔标志只需 4 字节（一个 `int`）
- **高效操作**：一次位运算可以同时处理多个标志
- **方便传输**：可以作为一个整数传递给函数或写入文件

## 基本方法：使用宏定义标志

```c
#include <stdio.h>

// 定义标志位
#define FLAG_A    (1U << 0)   // 0000 0001
#define FLAG_B    (1U << 1)   // 0000 0010
#define FLAG_C    (1U << 2)   // 0000 0100
#define FLAG_D    (1U << 3)   // 0000 1000
#define FLAG_E    (1U << 4)   // 0001 0000
#define FLAG_F    (1U << 5)   // 0010 0000

int main(void) {
    unsigned int flags = 0;  // 所有标志初始为关闭

    // 设置标志
    flags |= FLAG_A;         // 开启 A
    flags |= FLAG_C;         // 开启 C
    flags |= FLAG_F;         // 开启 F

    // 检查标志
    if (flags & FLAG_A) printf("FLAG_A 已开启\n");
    if (flags & FLAG_B) printf("FLAG_B 已开启\n");  // 不会输出
    if (flags & FLAG_C) printf("FLAG_C 已开启\n");

    // 清除标志
    flags &= ~FLAG_C;        // 关闭 C
    printf("关闭 C 后，C %s\n", (flags & FLAG_C) ? "开启" : "关闭");

    // 翻转标志
    flags ^= FLAG_B;         // 开启 B（原来关闭）
    flags ^= FLAG_A;         // 关闭 A（原来开启）

    // 打印当前所有标志
    printf("\n当前标志: 0x%02X\n", flags);
    printf("  FLAG_A: %s\n", (flags & FLAG_A) ? "ON" : "OFF");
    printf("  FLAG_B: %s\n", (flags & FLAG_B) ? "ON" : "OFF");
    printf("  FLAG_C: %s\n", (flags & FLAG_C) ? "ON" : "OFF");
    printf("  FLAG_D: %s\n", (flags & FLAG_D) ? "ON" : "OFF");
    printf("  FLAG_E: %s\n", (flags & FLAG_E) ? "ON" : "OFF");
    printf("  FLAG_F: %s\n", (flags & FLAG_F) ? "ON" : "OFF");

    return 0;
}
```

## 使用枚举定义标志

当标志位较多时，使用枚举配合移位使代码更清晰。

```c
#include <stdio.h>

// 使用枚举定义游戏中的道具状态
typedef enum {
    ITEM_SWORD    = 1 << 0,   // 剑
    ITEM_SHIELD   = 1 << 1,   // 盾
    ITEM_POTION   = 1 << 2,   // 药水
    ITEM_KEY      = 1 << 3,   // 钥匙
    ITEM_MAP      = 1 << 4,   // 地图
    ITEM_BOOTS    = 1 << 5,   // 靴子
    ITEM_RING     = 1 << 6,   // 戒指
    ITEM_AMULET   = 1 << 7,   // 护身符
} ItemFlag;

const char *item_name(ItemFlag item) {
    switch (item) {
        case ITEM_SWORD:  return "剑";
        case ITEM_SHIELD: return "盾";
        case ITEM_POTION: return "药水";
        case ITEM_KEY:    return "钥匙";
        case ITEM_MAP:    return "地图";
        case ITEM_BOOTS:  return "靴子";
        case ITEM_RING:   return "戒指";
        case ITEM_AMULET: return "护身符";
        default:          return "未知";
    }
}

int main(void) {
    unsigned int inventory = 0;

    // 获得道具
    inventory |= ITEM_SWORD | ITEM_SHIELD;
    printf("获得了剑和盾\n");

    inventory |= ITEM_POTION;
    printf("获得了药水\n");

    // 检查是否拥有某道具
    if (inventory & ITEM_SWORD) printf("拥有剑\n");
    if (!(inventory & ITEM_KEY)) printf("没有钥匙\n");

    // 使用钥匙开门后丢弃
    inventory |= ITEM_KEY;
    printf("找到了钥匙\n");
    inventory &= ~ITEM_KEY;
    printf("使用了钥匙（丢弃）\n");

    // 列出所有道具
    printf("\n当前背包: ");
    for (int i = 0; i < 8; i++) {
        if (inventory & (1 << i)) {
            printf("%s ", item_name(1 << i));
        }
    }
    printf("\n");

    return 0;
}
```

## 组合标志

```c
#include <stdio.h>

// 定义一组相关的标志
#define READ_PERM     (1U << 0)
#define WRITE_PERM    (1U << 1)
#define EXECUTE_PERM  (1U << 2)

// 组合标志
#define RW_PERM       (READ_PERM | WRITE_PERM)
#define ALL_PERM      (READ_PERM | WRITE_PERM | EXECUTE_PERM)

int main(void) {
    // 检查是否拥有全部权限
    unsigned int user_perm = READ_PERM | WRITE_PERM | EXECUTE_PERM;

    if ((user_perm & ALL_PERM) == ALL_PERM)
        printf("拥有全部权限\n");

    // 检查是否至少拥有读写权限
    if ((user_perm & RW_PERM) == RW_PERM)
        printf("拥有读写权限\n");

    // 检查是否拥有任何权限
    if (user_perm & ALL_PERM)
        printf("至少拥有一种权限\n");

    // 添加权限
    unsigned int new_perm = 0;
    new_perm |= RW_PERM;   // 一次性添加读写权限
    printf("新权限: %u\n", new_perm);  // 3

    return 0;
}
```

## 检查标志组合的技巧

```c
#include <stdio.h>

int main(void) {
    unsigned int flags = 0b1011;  // 有 FLAG_A, FLAG_B, FLAG_D

    // 1. 检查某个标志是否开启
    if (flags & (1 << 0)) printf("第0位开启\n");

    // 2. 检查是否所有指定标志都开启
    unsigned int required = 0b0011;  // 需要第0和第1位
    if ((flags & required) == required)
        printf("所有需要的标志都已开启\n");

    // 3. 检查是否有任何一个指定标志开启
    unsigned int any_of = 0b0100;  // 检查第2位
    if (flags & any_of)
        printf("至少有一个指定标志开启\n");

    // 4. 检查是否有不期望的标志开启
    unsigned int unwanted = 0b1000;
    if (flags & unwanted)
        printf("有不期望的标志开启\n");

    // 5. 清除所有指定标志
    flags &= ~0b0011;  // 清除第0和第1位
    printf("清除后: 0x%X\n", flags);

    return 0;
}
```

## 函数参数中的标志传递

```c
#include <stdio.h>

// 文件打开模式标志
#define MODE_READ    (1U << 0)
#define MODE_WRITE   (1U << 1)
#define MODE_APPEND  (1U << 2)
#define MODE_BINARY  (1U << 3)
#define MODE_CREATE  (1U << 4)

void open_file(const char *filename, unsigned int mode) {
    printf("打开文件 '%s'，模式: ", filename);
    if (mode & MODE_READ)   printf("读 ");
    if (mode & MODE_WRITE)  printf("写 ");
    if (mode & MODE_APPEND) printf("追加 ");
    if (mode & MODE_BINARY) printf("二进制 ");
    if (mode & MODE_CREATE) printf("创建 ");
    printf("\n");
}

int main(void) {
    open_file("data.txt", MODE_READ);
    open_file("log.txt", MODE_WRITE | MODE_APPEND);
    open_file("image.bin", MODE_READ | MODE_BINARY);

    return 0;
}
```

## 统计开启的标志数

```c
#include <stdio.h>

int count_set_bits(unsigned int n) {
    int count = 0;
    while (n) {
        count += n & 1;
        n >>= 1;
    }
    return count;
}

int main(void) {
    unsigned int flags = 0b10110101;
    printf("开启了 %d 个标志\n", count_set_bits(flags));  // 5

    return 0;
}
```

## 关键要点

- 用一个整数的不同位来管理多个布尔状态，节省内存且高效
- 使用宏或枚举为每个位赋予有意义的名字
- `|=` 设置标志，`&= ~` 清除标志，`^=` 翻转标志，`&` 检查标志
- 可以预定义组合标志（如 `RW_PERM`）简化常用操作
- 检查"所有标志都开启"用 `(flags & required) == required`
- 检查"至少一个标志开启"用 `flags & required`
