# 线程基础

## 1. 线程概述

```c
#include <stdio.h>
#include <pthread.h>

/*
 * 线程 vs 进程:
 *
 * 进程:
 *   - 独立的地址空间
 *   - 创建/切换开销大
 *   - 通过 IPC 通信
 *   - 安全性好（地址空间隔离）
 *
 * 线程:
 *   - 共享进程的地址空间
 *   - 创建/切换开销小
 *   - 直接读写共享内存
 *   - 需要同步机制保护共享数据
 *
 * 线程共享的资源:
 *   - 代码段、数据段、堆
 *   - 文件描述符表
 *   - 信号处理函数
 *   - 进程 ID、用户 ID、组 ID
 *
 * 线程私有的资源:
 *   - 线程 ID
 *   - 栈（局部变量、函数调用链）
 *   - 寄存器上下文
 *   - errno 变量
 *   - 信号掩码
 *   - 线程特定数据 (TSD)
 */
```

## 2. 创建线程

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

/*
 * int pthread_create(pthread_t *thread, const pthread_attr_t *attr,
 *                    void *(*start_routine)(void *), void *arg);
 *
 * 参数:
 *   thread: 返回线程 ID
 *   attr:   线程属性（NULL 表示默认）
 *   start_routine: 线程入口函数
 *   arg:    传递给入口函数的参数
 *
 * 返回: 0 成功，非零错误码
 *
 * 注意：编译时需要 -pthread 选项
 */

// 基本线程创建
void *thread_func(void *arg)
{
    int thread_num = *(int *)arg;
    printf("Thread %d: running (TID=%lu)\n",
           thread_num, (unsigned long)pthread_self());

    // 执行工作
    for (int i = 0; i < 3; i++) {
        printf("Thread %d: working %d\n", thread_num, i);
        sleep(1);
    }

    printf("Thread %d: done\n", thread_num);
    return NULL;
}

void basic_thread_create(void)
{
    pthread_t tid;
    int thread_num = 1;
    int err;

    err = pthread_create(&tid, NULL, thread_func, &thread_num);
    if (err != 0) {
        fprintf(stderr, "pthread_create: %s\n", strerror(err));
        return;
    }

    printf("Main: created thread %lu\n", (unsigned long)tid);

    // 等待线程结束
    pthread_join(tid, NULL);
    printf("Main: thread finished\n");
}

// 传递结构体参数
typedef struct {
    int id;
    char name[64];
    double value;
} ThreadArg;

void *worker_thread(void *arg)
{
    ThreadArg *ta = (ThreadArg *)arg;
    printf("Thread: id=%d, name=%s, value=%.2f\n",
           ta->id, ta->name, ta->value);

    // 返回结果
    double *result = malloc(sizeof(double));
    *result = ta->value * 2;
    return result;
}

void thread_with_struct_arg(void)
{
    pthread_t tid;
    ThreadArg arg = {1, "Worker", 3.14};

    pthread_create(&tid, NULL, worker_thread, &arg);

    double *result;
    pthread_join(tid, (void **)&result);
    printf("Result: %.2f\n", *result);
    free(result);
}

// 创建多个线程
void *worker(void *arg)
{
    int id = *(int *)arg;
    printf("Worker %d started\n", id);
    sleep(id);
    printf("Worker %d finished\n", id);
    return NULL;
}

void create_multiple_threads(void)
{
    pthread_t threads[5];
    int ids[5];

    for (int i = 0; i < 5; i++) {
        ids[i] = i;
        pthread_create(&threads[i], NULL, worker, &ids[i]);
    }

    for (int i = 0; i < 5; i++) {
        pthread_join(threads[i], NULL);
    }

    printf("All threads finished\n");
}
```

## 3. 线程终止

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>

/*
 * 线程终止方式:
 * 1. 线程函数 return（推荐）
 * 2. pthread_exit()
 * 3. 被其他线程 pthread_cancel()
 * 4. 进程终止（exit/_exit/信号）
 *
 * void pthread_exit(void *retval);
 * int pthread_cancel(pthread_t thread);
 * int pthread_join(pthread_t thread, void **retval);
 */

// 正常退出
void *normal_exit(void *arg)
{
    (void)arg;

    // 方式1：return
    return (void *)42;

    // 方式2：pthread_exit
    // pthread_exit((void *)42);
    // 不会执行到这里
}

void thread_exit_example(void)
{
    pthread_t tid;
    void *retval;

    pthread_create(&tid, NULL, normal_exit, NULL);
    pthread_join(tid, &retval);

    printf("Thread returned: %ld\n", (long)retval);
}

// 分离线程：不需要 join
void *detached_thread(void *arg)
{
    (void)arg;
    printf("Detached thread running\n");
    sleep(2);
    printf("Detached thread finished\n");
    // 自动释放资源，不需要 join
    return NULL;
}

void detached_thread_example(void)
{
    pthread_t tid;
    pthread_attr_t attr;

    pthread_attr_init(&attr);
    pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);

    pthread_create(&tid, &attr, detached_thread, NULL);
    pthread_attr_destroy(&attr);

    // 不能对分离线程调用 join
    // pthread_join(tid, NULL);  // 会返回 EINVAL

    printf("Main continues without waiting\n");
    sleep(3);
}
```

## 4. 线程 ID 与比较

```c
#include <pthread.h>
#include <stdio.h>

/*
 * pthread_t pthread_self(void);     - 获取自己的线程 ID
 * int pthread_equal(pthread_t t1, pthread_t t2);  - 比较线程 ID
 *
 * 注意：pthread_t 可能是结构体或整数
 * 不能直接用 == 比较，必须用 pthread_equal
 */

void *show_tid(void *arg)
{
    (void)arg;
    pthread_t self = pthread_self();
    printf("Thread ID: %lu\n", (unsigned long)self);
    return NULL;
}

void thread_id_example(void)
{
    pthread_t main_tid = pthread_self();
    printf("Main thread ID: %lu\n", (unsigned long)main_tid);

    pthread_t tid;
    pthread_create(&tid, NULL, show_tid, NULL);
    pthread_join(tid, NULL);

    // 比较线程 ID
    if (pthread_equal(main_tid, tid)) {
        printf("Same thread\n");
    } else {
        printf("Different threads\n");
    }
}
```

## 5. 线程属性

```c
#include <pthread.h>
#include <stdio.h>
#include <string.h>

/*
 * int pthread_attr_init(pthread_attr_t *attr);
 * int pthread_attr_destroy(pthread_attr_t *attr);
 *
 * 常用属性:
 *   detachstate   - 分离状态
 *   stackaddr     - 栈地址（已废弃）
 *   stacksize     - 栈大小
 *   guardsize     - 栈保护区大小
 *   schedpolicy   - 调度策略
 *   schedparam    - 调度参数
 *   inheritsched  - 继承调度属性的方式
 *   scope         - 竞争范围
 */

void thread_attr_example(void)
{
    pthread_attr_t attr;
    pthread_t tid;

    pthread_attr_init(&attr);

    // 设置分离状态
    pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);

    // 设置栈大小
    size_t stack_size;
    pthread_attr_getstacksize(&attr, &stack_size);
    printf("Default stack size: %zu bytes\n", stack_size);

    // 设置更大的栈（2MB）
    pthread_attr_setstacksize(&attr, 2 * 1024 * 1024);

    // 获取栈保护区大小
    size_t guard_size;
    pthread_attr_getguardsize(&attr, &guard_size);
    printf("Guard size: %zu bytes\n", guard_size);

    // 创建线程
    pthread_create(&tid, &attr, worker, NULL);

    pthread_attr_destroy(&attr);
}

// 获取和设置线程栈
void stack_attributes(void)
{
    pthread_attr_t attr;
    void *stack_addr;
    size_t stack_size;

    pthread_attr_init(&attr);

    // 获取当前线程的栈信息
    pthread_getattr_np(pthread_self(), &attr);
    pthread_attr_getstack(&attr, &stack_addr, &stack_size);

    printf("Stack address: %p\n", stack_addr);
    printf("Stack size: %zu bytes\n", stack_size);

    pthread_attr_destroy(&attr);
}
```

## 6. 线程与进程的信号处理

```c
#include <pthread.h>
#include <signal.h>
#include <stdio.h>

/*
 * 信号处理在多线程环境中的注意事项:
 * 1. 信号处理函数是进程级别的（所有线程共享）
 * 2. 信号发送给进程，内核选择一个不阻塞该信号的线程处理
 * 3. 可以用 pthread_sigmask 控制每个线程的信号掩码
 * 4. 建议：在一个专用线程中用 sigwait 同步处理信号
 */

// 专用信号处理线程
void *signal_thread(void *arg)
{
    (void)arg;
    sigset_t set;
    int sig;

    sigemptyset(&set);
    sigaddset(&set, SIGINT);
    sigaddset(&set, SIGTERM);

    while (1) {
        sigwait(&set, &sig);
        printf("Signal thread: received signal %d\n", sig);

        if (sig == SIGTERM) break;
    }

    return NULL;
}

void signal_thread_example(void)
{
    sigset_t set;

    // 所有线程阻塞 SIGINT 和 SIGTERM
    sigemptyset(&set);
    sigaddset(&set, SIGINT);
    sigaddset(&set, SIGTERM);
    pthread_sigmask(SIG_BLOCK, &set, NULL);

    // 创建信号处理线程
    pthread_t sig_tid;
    pthread_create(&sig_tid, NULL, signal_thread, NULL);

    // 其他工作线程继承信号掩码
    // ...

    pthread_join(sig_tid, NULL);
}
```
