# 内存映射

## 1. mmap 基础

```c
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

/*
 * void *mmap(void *addr, size_t length, int prot,
 *            int flags, int fd, off_t offset);
 *
 * int munmap(void *addr, size_t length);
 * int mprotect(void *addr, size_t len, int prot);
 * int msync(void *addr, size_t length, int flags);
 * void *mremap(void *old_addr, size_t old_size,
 *              size_t new_size, int flags, ...);
 *
 * prot (保护标志):
 *   PROT_READ  - 可读
 *   PROT_WRITE - 可写
 *   PROT_EXEC  - 可执行
 *   PROT_NONE  - 不可访问
 *
 * flags:
 *   MAP_SHARED  - 共享映射（修改写回文件）
 *   MAP_PRIVATE - 私有映射（写时复制）
 *   MAP_FIXED   - 固定地址
 *   MAP_ANONYMOUS - 匿名映射（不关联文件）
 */

// 文件映射示例
void file_mapping_example(void)
{
    int fd = open("test.dat", O_RDWR | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return; }

    // 扩展文件到 4KB
    size_t size = 4096;
    ftruncate(fd, size);

    // 将文件映射到内存
    char *mapped = mmap(NULL, size, PROT_READ | PROT_WRITE,
                        MAP_SHARED, fd, 0);
    if (mapped == MAP_FAILED) { perror("mmap"); close(fd); return; }

    close(fd);  // 映射后可以关闭 fd

    // 通过指针操作文件内容
    strcpy(mapped, "Hello, mmap!");
    mapped[100] = 'X';

    // 强制同步到磁盘
    msync(mapped, size, MS_SYNC);

    printf("Content: %s\n", mapped);

    // 解除映射
    munmap(mapped, size);
}

// 只读文件映射
void readonly_mapping(const char *path)
{
    int fd = open(path, O_RDONLY);
    if (fd == -1) { perror("open"); return; }

    struct stat st;
    fstat(fd, &st);

    const char *mapped = mmap(NULL, st.st_size, PROT_READ,
                               MAP_PRIVATE, fd, 0);
    if (mapped == MAP_FAILED) { perror("mmap"); close(fd); return; }
    close(fd);

    // 可以像读内存一样读文件
    for (off_t i = 0; i < st.st_size; i++) {
        putchar(mapped[i]);
    }

    munmap((void *)mapped, st.st_size);
}
```

## 2. 匿名映射

```c
#include <sys/mman.h>
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * 匿名映射不关联任何文件
 * 常用于：
 * 1. 分配大块内存（代替 malloc）
 * 2. 父子进程共享内存（fork 后 MAP_SHARED 仍共享）
 */

void anonymous_mapping_example(void)
{
    // 匿名映射（Linux 方式）
    size_t size = 4096;
    char *area = mmap(NULL, size, PROT_READ | PROT_WRITE,
                      MAP_SHARED | MAP_ANONYMOUS, -1, 0);
    if (area == MAP_FAILED) { perror("mmap"); return; }

    strcpy(area, "Shared data");

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程可以读写共享区域
        printf("Child reads: %s\n", area);
        strcpy(area, "Modified by child");
        _exit(0);
    } else {
        wait(NULL);
        printf("Parent reads: %s\n", area);
    }

    munmap(area, size);
}

// 使用 /dev/zero 方式（更可移植）
void dev_zero_mapping(void)
{
    int fd = open("/dev/zero", O_RDWR);
    if (fd == -1) { perror("open"); return; }

    int *shared = mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                       MAP_SHARED, fd, 0);
    close(fd);

    if (shared == MAP_FAILED) { perror("mmap"); return; }

    shared[0] = 42;

    pid_t pid = fork();
    if (pid == 0) {
        printf("Child: %d\n", shared[0]);
        shared[0] = 100;
        _exit(0);
    } else {
        wait(NULL);
        printf("Parent: %d\n", shared[0]);
    }

    munmap(shared, 4096);
}
```

## 3. mprotect - 修改内存保护

```c
#include <sys/mman.h>
#include <stdio.h>
#include <signal.h>
#include <setjmp.h>

/*
 * int mprotect(void *addr, size_t len, int prot);
 *
 * 修改已映射区域的保护属性
 * addr 必须页对齐（sysconf(_SC_PAGESIZE)）
 */

static sigjmp_buf jmp_env;

void segv_handler(int sig)
{
    (void)sig;
    siglongjmp(jmp_env, 1);
}

void mprotect_example(void)
{
    size_t pagesize = sysconf(_SC_PAGESIZE);

    // 分配一页内存
    char *page = mmap(NULL, pagesize, PROT_READ | PROT_WRITE,
                       MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    if (page == MAP_FAILED) { perror("mmap"); return; }

    // 写入数据
    strcpy(page, "Protected data");
    printf("Before protect: %s\n", page);

    // 设为只读
    mprotect(page, pagesize, PROT_READ);
    printf("Page is now read-only\n");

    // 尝试写入（会触发 SIGSEGV）
    signal(SIGSEGV, segv_handler);

    if (sigsetjmp(jmp_env, 1) == 0) {
        page[0] = 'X';  // 这会触发段错误
    } else {
        printf("Caught SIGSEGV - write protection works!\n");
    }

    // 恢复读写
    mprotect(page, pagesize, PROT_READ | PROT_WRITE);
    page[0] = 'Y';
    printf("After restoring: %s\n", page);

    munmap(page, pagesize);
}
```

## 4. mremap - 调整映射大小

```c
#include <sys/mman.h>
#include <stdio.h>
#include <string.h>

/*
 * void *mremap(void *old_address, size_t old_size,
 *              size_t new_size, int flags, ...);
 *
 * flags:
 *   MREMAP_MAYMOVE - 允许移动映射位置
 *
 * Linux 特有，比 munmap+mmap 更高效
 */

void mremap_example(void)
{
    size_t size = 4096;

    // 初始映射
    char *area = mmap(NULL, size, PROT_READ | PROT_WRITE,
                       MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    if (area == MAP_FAILED) { perror("mmap"); return; }

    strcpy(area, "Initial data");
    printf("Original at %p: %s\n", (void *)area, area);

    // 扩大到 8KB
    size_t new_size = 8192;
    char *new_area = mremap(area, size, new_size, MREMAP_MAYMOVE);
    if (new_area == MAP_FAILED) {
        perror("mremap");
        munmap(area, size);
        return;
    }

    printf("Resized to %p: %s\n", (void *)new_area, new_area);

    // 写入更多数据
    strcpy(new_area + 4096, "Additional data");
    printf("New content at offset 4096: %s\n", new_area + 4096);

    munmap(new_area, new_size);
}
```

## 5. 内存映射文件拷贝

```c
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

/*
 * 使用 mmap 实现高效文件拷贝
 * 适用于大文件，避免多次 read/write 的系统调用开销
 */

int mmap_copy(const char *src, const char *dst)
{
    int fd_src = open(src, O_RDONLY);
    if (fd_src == -1) { perror("open src"); return -1; }

    struct stat st;
    fstat(fd_src, &st);
    size_t size = st.st_size;

    // 映射源文件（只读）
    const char *src_map = mmap(NULL, size, PROT_READ, MAP_PRIVATE, fd_src, 0);
    if (src_map == MAP_FAILED) { perror("mmap src"); close(fd_src); return -1; }

    // 创建目标文件
    int fd_dst = open(dst, O_RDWR | O_CREAT | O_TRUNC, 0644);
    if (fd_dst == -1) { perror("open dst"); munmap((void *)src_map, size); close(fd_src); return -1; }
    ftruncate(fd_dst, size);

    // 映射目标文件（可写）
    char *dst_map = mmap(NULL, size, PROT_READ | PROT_WRITE,
                          MAP_SHARED, fd_dst, 0);
    if (dst_map == MAP_FAILED) { perror("mmap dst"); /* cleanup */ return -1; }

    // 一次内存拷贝完成文件复制
    memcpy(dst_map, src_map, size);

    // 同步到磁盘
    msync(dst_map, size, MS_SYNC);

    // 清理
    munmap((void *)src_map, size);
    munmap(dst_map, size);
    close(fd_src);
    close(fd_dst);

    return 0;
}
```

## 6. madvise - 内存使用建议

```c
#include <sys/mman.h>
#include <stdio.h>

/*
 * int madvise(void *addr, size_t length, int advice);
 *
 * advice:
 *   MADV_NORMAL     - 默认行为
 *   MADV_RANDOM     - 随机访问（禁用预读）
 *   MADV_SEQUENTIAL - 顺序访问（增大预读）
 *   MADV_WILLNEED   - 即将使用（预读）
 *   MADV_DONTNEED   - 不再使用（可释放物理页）
 *   MADV_HUGEPAGE   - 使用大页（Linux 4.4+）
 *   MADV_NOHUGEPAGE - 不使用大页
 */

void madvise_example(void)
{
    size_t size = 1024 * 1024;  // 1MB

    char *area = mmap(NULL, size, PROT_READ | PROT_WRITE,
                       MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);

    // 告诉内核我们将顺序访问
    madvise(area, size, MADV_SEQUENTIAL);

    // 顺序读取
    for (size_t i = 0; i < size; i += 4096) {
        area[i] = 1;
    }

    // 不再需要这些页面
    madvise(area, size, MADV_DONTNEED);

    munmap(area, size);
}
```
