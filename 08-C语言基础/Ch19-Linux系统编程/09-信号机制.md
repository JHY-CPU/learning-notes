# 信号机制

## 1. 信号概述

```c
#include <signal.h>
#include <stdio.h>

/*
 * 信号是一种软件中断，用于通知进程发生了某个事件
 *
 * 常见信号:
 *   SIGHUP  (1)  - 终端挂起/控制进程终止
 *   SIGINT  (2)  - 键盘中断 (Ctrl+C)
 *   SIGQUIT (3)  - 键盘退出 (Ctrl+\)
 *   SIGILL  (4)  - 非法指令
 *   SIGABRT (6)  - 异常终止 (abort)
 *   SIGFPE  (8)  - 浮点异常
 *   SIGKILL (9)  - 强制终止（不可捕获/忽略）
 *   SIGSEGV (11) - 段错误
 *   SIGPIPE (13) - 管道破裂
 *   SIGALRM (14) - 定时器报警
 *   SIGTERM (15) - 终止信号（默认终止）
 *   SIGUSR1 (10) - 用户自定义信号 1
 *   SIGUSR2 (12) - 用户自定义信号 2
 *   SIGCHLD (17) - 子进程状态改变
 *   SIGCONT (18) - 继续执行
 *   SIGSTOP (19) - 暂停（不可捕获/忽略）
 *   SIGTSTP (20) - 终端暂停 (Ctrl+Z)
 */

// 打印所有信号的默认行为
void print_signal_info(void)
{
    int sigs[] = {SIGHUP, SIGINT, SIGQUIT, SIGILL, SIGABRT,
                  SIGFPE, SIGKILL, SIGSEGV, SIGPIPE, SIGALRM,
                  SIGTERM, SIGUSR1, SIGUSR2, SIGCHLD};
    const char *names[] = {"SIGHUP", "SIGINT", "SIGQUIT", "SIGILL",
                           "SIGABRT", "SIGFPE", "SIGKILL", "SIGSEGV",
                           "SIGPIPE", "SIGALRM", "SIGTERM", "SIGUSR1",
                           "SIGUSR2", "SIGCHLD"};

    for (int i = 0; i < 14; i++) {
        struct sigaction sa;
        sigaction(sigs[i], NULL, &sa);
        printf("%-10s (%2d): ", names[i], sigs[i]);

        if (sa.sa_handler == SIG_DFL) {
            // 检查默认行为
            if (sigs[i] == SIGCHLD || sigs[i] == SIGCONT ||
                sigs[i] == SIGURG || sigs[i] == SIGWINCH)
                printf("Ignore\n");
            else
                printf("Terminate\n");
        } else if (sa.sa_handler == SIG_IGN) {
            printf("Ignore (explicit)\n");
        } else {
            printf("Caught (handler=%p)\n", (void *)sa.sa_handler);
        }
    }
}
```

## 2. signal 函数

```c
#include <signal.h>
#include <stdio.h>
#include <unistd.h>

/*
 * sighandler_t signal(int signum, sighandler_t handler);
 *
 * handler:
 *   SIG_DFL  - 默认处理
 *   SIG_IGN  - 忽略信号
 *   自定义函数 - 捕获信号
 *
 * 注意：signal 是旧的信号处理接口
 * 不同系统实现可能不同，推荐使用 sigaction
 */

// 信号处理函数
void sigint_handler(int sig)
{
    // 注意：只能调用异步信号安全的函数
    const char msg[] = "\nCaught SIGINT, press Ctrl+C again to quit\n";
    write(STDERR_FILENO, msg, sizeof(msg) - 1);
}

void signal_example(void)
{
    // 注册信号处理函数
    signal(SIGINT, sigint_handler);

    printf("Press Ctrl+C to test signal handler...\n");
    printf("Press Ctrl+C twice to quit\n");

    while (1) {
        pause();  // 等待信号
    }
}

// 忽略信号
void ignore_signals(void)
{
    signal(SIGINT, SIG_IGN);   // 忽略 Ctrl+C
    signal(SIGQUIT, SIG_IGN);  // 忽释 Ctrl+\

    printf("SIGINT and SIGQUIT are now ignored\n");
    printf("Press Ctrl+C or Ctrl+\\ - nothing will happen\n");

    sleep(10);

    // 恢复默认行为
    signal(SIGINT, SIG_DFL);
    signal(SIGQUIT, SIG_DFL);
}
```

## 3. sigaction 函数

```c
#include <signal.h>
#include <stdio.h>
#include <unistd.h>
#include <string.h>

/*
 * int sigaction(int signum, const struct sigaction *act,
 *               struct sigaction *oldact);
 *
 * struct sigaction {
 *     void     (*sa_handler)(int);      // 简单处理函数
 *     void     (*sa_sigaction)(int, siginfo_t *, void *); // 详细处理函数
 *     sigset_t   sa_mask;               // 处理期间阻塞的信号
 *     int        sa_flags;              // 标志
 *     void     (*sa_restorer)(void);    // 内部使用
 * };
 *
 * sa_flags:
 *   SA_NOCLDSTOP  - 子进程停止时不发送 SIGCHLD
 *   SA_NOCLDWAIT  - 子进程终止时不创建僵尸
 *   SA_NODEFER    - 不自动阻塞正在处理的信号
 *   SA_ONSTACK    - 使用替代信号栈
 *   SA_RESETHAND  - 处理后恢复默认行为
 *   SA_RESTART    - 自动重启被中断的系统调用
 *   SA_SIGINFO    - 使用 sa_sigaction 而非 sa_handler
 */

// 基本 sigaction 使用
void sigaction_basic(void)
{
    struct sigaction sa;

    memset(&sa, 0, sizeof(sa));
    sa.sa_handler = sigint_handler;
    sigemptyset(&sa.sa_mask);   // 处理时不阻塞其他信号
    sa.sa_flags = SA_RESTART;   // 自动重启被中断的系统调用

    if (sigaction(SIGINT, &sa, NULL) == -1) {
        perror("sigaction");
    }

    printf("SIGINT handler installed\n");
}

// 使用 SA_SIGINFO 获取详细信号信息
void siginfo_handler(int sig, siginfo_t *info, void *ucontext)
{
    (void)ucontext;
    char buf[256];
    int len;

    len = snprintf(buf, sizeof(buf),
        "Signal %d received:\n"
        "  Sender PID: %d\n"
        "  Sender UID: %d\n"
        "  Signal code: %d\n",
        sig,
        info->si_pid,
        info->si_uid,
        info->si_code);

    write(STDERR_FILENO, buf, len);
}

void sigaction_siginfo(void)
{
    struct sigaction sa;

    memset(&sa, 0, sizeof(sa));
    sa.sa_sigaction = siginfo_handler;
    sa.sa_flags = SA_SIGINFO;
    sigemptyset(&sa.sa_mask);

    sigaction(SIGUSR1, &sa, NULL);
    sigaction(SIGUSR2, &sa, NULL);

    printf("PID: %d - Send me signals: kill -USR1 %d\n",
           getpid(), getpid());

    while (1) pause();
}

// 处理期间阻塞其他信号
void sigaction_block(void)
{
    struct sigaction sa;

    memset(&sa, 0, sizeof(sa));
    sa.sa_handler = sigint_handler;

    // 处理 SIGINT 期间阻塞 SIGQUIT
    sigemptyset(&sa.sa_mask);
    sigaddset(&sa.sa_mask, SIGQUIT);

    sa.sa_flags = SA_RESTART;
    sigaction(SIGINT, &sa, NULL);
}
```

## 5. 信号集操作

```c
#include <signal.h>
#include <stdio.h>

/*
 * int sigemptyset(sigset_t *set);       - 清空信号集
 * int sigfillset(sigset_t *set);        - 填充所有信号
 * int sigaddset(sigset_t *set, int signum);  - 添加信号
 * int sigdelset(sigset_t *set, int signum);  - 删除信号
 * int sigismember(const sigset_t *set, int signum); - 检查信号
 *
 * int sigprocmask(int how, const sigset_t *set, sigset_t *oldset);
 *   how: SIG_BLOCK, SIG_UNBLOCK, SIG_SETMASK
 *
 * int sigpending(sigset_t *set);        - 获取未决信号集
 * int sigsuspend(const sigset_t *mask); - 临时替换掩码并等待
 */

void signal_set_examples(void)
{
    sigset_t set;

    // 初始化信号集
    sigemptyset(&set);

    // 添加信号
    sigaddset(&set, SIGINT);
    sigaddset(&set, SIGQUIT);

    // 检查信号
    if (sigismember(&set, SIGINT)) {
        printf("SIGINT is in the set\n");
    }

    // 阻塞 SIGINT 和 SIGQUIT
    sigset_t oldset;
    sigprocmask(SIG_BLOCK, &set, &oldset);

    printf("SIGINT and SIGQUIT are now blocked\n");
    printf("Press Ctrl+C - signal will be pending\n");
    sleep(5);

    // 检查未决信号
    sigset_t pending;
    sigpending(&pending);
    if (sigismember(&pending, SIGINT)) {
        printf("SIGINT is pending\n");
    }

    // 解除阻塞（此时未决信号会被递送）
    sigprocmask(SIG_SETMASK, &oldset, NULL);
    printf("Signals unblocked - pending signals delivered now\n");
}
```

## 6. 可重入函数与信号安全

```c
#include <signal.h>
#include <unistd.h>
#include <stdio.h>
#include <errno.h>

/*
 * 信号处理函数中只能调用异步信号安全的函数
 *
 * 异步信号安全函数（可以在信号处理中安全调用）:
 *   write, read, close, open, lseek
 *   _exit, exec, wait, waitpid
 *   kill, signal, sigaction
 *   getpid, alarm, pause, sleep
 *
 * 非信号安全函数（不应在信号处理中调用）:
 *   printf, fprintf, sprintf  (使用了全局静态缓冲区)
 *   malloc, free              (可能持有锁)
 *   localtime, gethostbyname  (使用了静态数据)
 */

// 错误示例：在信号处理中使用了非安全函数
void bad_handler(int sig)
{
    // 错误：printf 不是信号安全的！
    // printf("Received signal %d\n", sig);

    // 可能的问题：
    // 1. 如果在 printf 中间被中断，输出可能混乱
    // 2. 如果在 malloc 中被中断，可能死锁
}

// 正确示例：只使用信号安全的函数
void good_handler(int sig)
{
    // 使用 write 而非 printf
    const char msg[] = "Signal received\n";
    int saved_errno = errno;  // 保存 errno
    write(STDERR_FILENO, msg, sizeof(msg) - 1);
    errno = saved_errno;      // 恢复 errno
}

// 信号安全的错误输出
void safe_error(const char *msg)
{
    if (msg) {
        size_t len = 0;
        while (msg[len]) len++;
        write(STDERR_FILENO, msg, len);
    }
}

// 使用 self-pipe 技巧避免信号安全问题
// 在信号处理中只写管道，主循环中读管道处理
static int signal_pipe[2];

void pipe_handler(int sig)
{
    int saved_errno = errno;
    write(signal_pipe[1], &sig, sizeof(sig));
    errno = saved_errno;
}

void setup_signal_pipe(void)
{
    pipe(signal_pipe);
    signal(SIGINT, pipe_handler);
    signal(SIGTERM, pipe_handler);

    // 主循环中用 select/poll 监听 signal_pipe[0]
}
```

## 7. 综合示例：优雅的信号处理

```c
#include <signal.h>
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/wait.h>

static volatile sig_atomic_t got_sigchld = 0;
static volatile sig_atomic_t got_sigterm = 0;
static volatile sig_atomic_t got_sigusr1 = 0;

void handler(int sig)
{
    int saved_errno = errno;

    switch (sig) {
    case SIGCHLD: got_sigchld = 1; break;
    case SIGTERM: got_sigterm = 1; break;
    case SIGUSR1: got_sigusr1 = 1; break;
    }

    errno = saved_errno;
}

void setup_handlers(void)
{
    struct sigaction sa;
    memset(&sa, 0, sizeof(sa));
    sa.sa_handler = handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = SA_RESTART;

    sigaction(SIGCHLD, &sa, NULL);
    sigaction(SIGTERM, &sa, NULL);
    sigaction(SIGUSR1, &sa, NULL);

    // 忽略 SIGPIPE
    sa.sa_handler = SIG_IGN;
    sigaction(SIGPIPE, &sa, NULL);
}

int main(void)
{
    setup_handlers();

    printf("PID: %d\n", getpid());
    printf("Send SIGUSR1 to print status\n");
    printf("Send SIGTERM to exit\n");

    while (!got_sigterm) {
        if (got_sigchld) {
            got_sigchld = 0;
            // 回收所有已退出的子进程
            int status;
            pid_t pid;
            while ((pid = waitpid(-1, &status, WNOHANG)) > 0) {
                printf("Child %d exited\n", pid);
            }
        }

        if (got_sigusr1) {
            got_sigusr1 = 0;
            printf("Status: Running (PID=%d)\n", getpid());
        }

        // 使用 pause 或 nanosleep 而非 busy loop
        struct timespec ts = {1, 0};
        nanosleep(&ts, NULL);
    }

    printf("Shutting down gracefully\n");
    return 0;
}
```
