# 进程间通信 - 信号量

## 1. System V 信号量

```c
#include <sys/ipc.h>
#include <sys/sem.h>
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * System V 信号量 API:
 *   int semget(key_t key, int nsems, int semflg);  - 创建/获取
 *   int semop(int semid, struct sembuf *sops, size_t nsops);  - 操作
 *   int semctl(int semid, int semnum, int cmd, ...);  - 控制
 *
 * struct sembuf {
 *     unsigned short sem_num;  - 信号量编号
 *     short          sem_op;   - 操作值（P=-1, V=+1）
 *     short          sem_flg;  - 标志（SEM_UNDO 等）
 * };
 *
 * semctl cmd:
 *   SETVAL  - 设置值
 *   GETVAL  - 获取值
 *   IPC_RMID - 删除
 *   GETALL  - 获取所有值
 *   SETALL  - 设置所有值
 */

union semun {
    int val;
    struct semid_ds *buf;
    unsigned short *array;
};

// P 操作（等待）
int sem_p(int semid, int semnum)
{
    struct sembuf op = {semnum, -1, SEM_UNDO};
    return semop(semid, &op, 1);
}

// V 操作（发信号）
int sem_v(int semid, int semnum)
{
    struct sembuf op = {semnum, 1, SEM_UNDO};
    return semop(semid, &op, 1);
}

// 获取信号量值
int sem_get(int semid, int semnum)
{
    return semctl(semid, semnum, GETVAL);
}

void sysv_semaphore_example(void)
{
    key_t key = ftok("/tmp", 'S');

    // 创建包含 1 个信号量的集合
    int semid = semget(key, 1, IPC_CREAT | 0666);
    if (semid == -1) { perror("semget"); return; }

    // 初始化信号量值为 1（用作互斥锁）
    union semun arg;
    arg.val = 1;
    semctl(semid, 0, SETVAL, arg);

    int counter = 0;
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程
        for (int i = 0; i < 100000; i++) {
            sem_p(semid, 0);   // 加锁
            counter++;
            sem_v(semid, 0);   // 解锁
        }
        printf("Child counter: %d\n", counter);
        _exit(0);
    } else {
        // 父进程
        for (int i = 0; i < 100000; i++) {
            sem_p(semid, 0);
            counter++;
            sem_v(semid, 0);
        }
        wait(NULL);
        printf("Parent counter: %d\n", counter);
        printf("Shared value: %d\n", sem_get(semid, 0));

        // 清理
        semctl(semid, 0, IPC_RMID);
    }
}

// 信号量集合（多个信号量）
void semaphore_set_example(void)
{
    key_t key = ftok("/tmp", 'M');
    int num_sems = 3;

    int semid = semget(key, num_sems, IPC_CREAT | 0666);

    // 初始化所有信号量
    unsigned short values[] = {1, 0, 5};  // 不同初始值
    union semun arg;
    arg.array = values;
    semctl(semid, 0, SETALL, arg);

    // 打印所有值
    for (int i = 0; i < num_sems; i++) {
        printf("Semaphore %d: %d\n", i, sem_get(semid, i));
    }

    // 批量操作：原子地操作多个信号量
    struct sembuf ops[2];
    ops[0].sem_num = 0;
    ops[0].sem_op  = -1;
    ops[0].sem_flg = 0;
    ops[1].sem_num = 1;
    ops[1].sem_op  = 1;
    ops[1].sem_flg = 0;

    semop(semid, ops, 2);  // 原子地减信号量0，加信号量1

    printf("After atomical operation:\n");
    for (int i = 0; i < num_sems; i++) {
        printf("Semaphore %d: %d\n", i, sem_get(semid, i));
    }

    semctl(semid, 0, IPC_RMID);
}
```

## 2. POSIX 有名信号量

```c
#include <semaphore.h>
#include <fcntl.h>
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * POSIX 有名信号量:
 *   sem_t *sem_open(const char *name, int oflag, ...);
 *   int sem_close(sem_t *sem);
 *   int sem_unlink(const char *name);
 *   int sem_wait(sem_t *sem);
 *   int sem_trywait(sem_t *sem);
 *   int sem_timedwait(sem_t *sem, const struct timespec *abs_timeout);
 *   int sem_post(sem_t *sem);
 *   int sem_getvalue(sem_t *sem, int *sval);
 *
 * 名称格式：以 '/' 开头，如 "/my_sem"
 * 可用于无亲缘关系的进程间同步
 */

void named_semaphore_example(void)
{
    const char *sem_name = "/my_named_sem";

    // 创建/打开有名信号量
    sem_t *sem = sem_open(sem_name, O_CREAT, 0666, 1);
    if (sem == SEM_FAILED) { perror("sem_open"); return; }

    pid_t pid = fork();

    if (pid == 0) {
        // 子进程
        sem_t *child_sem = sem_open(sem_name, 0);

        sem_wait(child_sem);
        printf("Child: acquired semaphore\n");
        sleep(2);
        printf("Child: releasing semaphore\n");
        sem_post(child_sem);

        sem_close(child_sem);
        _exit(0);
    } else {
        // 父进程
        sleep(1);  // 让子进程先获取信号量

        sem_wait(sem);
        printf("Parent: acquired semaphore\n");
        sem_post(sem);
        printf("Parent: released semaphore\n");

        wait(NULL);
        sem_close(sem);
        sem_unlink(sem_name);
    }
}
```

## 3. POSIX 无名信号量

```c
#include <semaphore.h>
#include <pthread.h>
#include <stdio.h>
#include <sys/mman.h>

/*
 * POSIX 无名信号量:
 *   int sem_init(sem_t *sem, int pshared, unsigned int value);
 *   int sem_destroy(sem_t *sem);
 *
 * pshared:
 *   0  - 线程间共享
 *   非0 - 进程间共享（必须在共享内存中）
 */

// 线程间使用
static sem_t thread_sem;

void *thread_worker(void *arg)
{
    int id = *(int *)arg;

    sem_wait(&thread_sem);
    printf("Thread %d: in critical section\n", id);
    sleep(1);
    sem_post(&thread_sem);

    return NULL;
}

void unnamed_sem_threads(void)
{
    pthread_t t1, t2;
    int id1 = 1, id2 = 2;

    sem_init(&thread_sem, 0, 1);  // 线程间，初始值1

    pthread_create(&t1, NULL, thread_worker, &id1);
    pthread_create(&t2, NULL, thread_worker, &id2);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    sem_destroy(&thread_sem);
}

// 进程间使用（需要共享内存）
typedef struct {
    sem_t mutex;
    int shared_counter;
} ProcessShared;

void unnamed_sem_processes(void)
{
    // 使用 mmap 创建共享内存
    ProcessShared *shared = mmap(NULL, sizeof(ProcessShared),
                                  PROT_READ | PROT_WRITE,
                                  MAP_SHARED | MAP_ANONYMOUS,
                                  -1, 0);

    // 初始化信号量（pshared=1 表示进程间共享）
    sem_init(&shared->mutex, 1, 1);
    shared->shared_counter = 0;

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程
        for (int i = 0; i < 10000; i++) {
            sem_wait(&shared->mutex);
            shared->shared_counter++;
            sem_post(&shared->mutex);
        }
        _exit(0);
    } else {
        // 父进程
        for (int i = 0; i < 10000; i++) {
            sem_wait(&shared->mutex);
            shared->shared_counter++;
            sem_post(&shared->mutex);
        }
        wait(NULL);
        printf("Final counter: %d (expected 20000)\n",
               shared->shared_counter);

        sem_destroy(&shared->mutex);
        munmap(shared, sizeof(ProcessShared));
    }
}
```
