# 资源限制

## 1. getrlimit / setrlimit

```c
#include <sys/resource.h>
#include <stdio.h>
#include <stdlib.h>

/*
 * int getrlimit(int resource, struct rlimit *rlim);
 * int setrlimit(int resource, const struct rlimit *rlim);
 * int prlimit(pid_t pid, int resource,
 *             const struct rlimit *new_limit,
 *             struct rlimit *old_limit);
 *
 * struct rlimit {
 *     rlim_t rlim_cur;  - 软限制（当前限制）
 *     rlim_t rlim_max;  - 硬限制（最大值）
 * };
 *
 * 资源类型:
 *   RLIMIT_CPU      - CPU 时间（秒）
 *   RLIMIT_FSIZE    - 文件大小（字节）
 *   RLIMIT_DATA     - 数据段大小
 *   RLIMIT_STACK    - 栈大小
 *   RLIMIT_CORE     - core dump 文件大小
 *   RLIMIT_NOFILE   - 打开文件数
 *   RLIMIT_AS       - 虚拟内存大小
 *   RLIMIT_NPROC    - 子进程数
 *   RLIMIT_MEMLOCK  - 锁定的内存
 *   RLIMIT_NICE     - nice 值
 *   RLIMIT_RTPRIO   - 实时优先级
 */

const char *rlim_str(rlim_t val)
{
    static char buf[32];
    if (val == RLIM_INFINITY)
        return "unlimited";
    snprintf(buf, sizeof(buf), "%lu", (unsigned long)val);
    return buf;
}

void print_resource_limit(int resource, const char *name)
{
    struct rlimit rl;
    getrlimit(resource, &rl);
    printf("%-20s soft=%-12s hard=%s\n",
           name, rlim_str(rl.rlim_cur), rlim_str(rl.rlim_max));
}

void show_all_limits(void)
{
    printf("Resource Limits:\n");
    printf("%-20s %-14s %s\n", "Resource", "Soft", "Hard");
    printf("-------------------------------------------\n");

    print_resource_limit(RLIMIT_CPU, "CPU (sec)");
    print_resource_limit(RLIMIT_FSIZE, "FSIZE (bytes)");
    print_resource_limit(RLIMIT_DATA, "DATA (bytes)");
    print_resource_limit(RLIMIT_STACK, "STACK (bytes)");
    print_resource_limit(RLIMIT_CORE, "CORE (bytes)");
    print_resource_limit(RLIMIT_NOFILE, "NOFILE");
    print_resource_limit(RLIMIT_AS, "AS (bytes)");
    print_resource_limit(RLIMIT_NPROC, "NPROC");
    print_resource_limit(RLIMIT_MEMLOCK, "MEMLOCK (bytes)");
}

// 修改资源限制
void modify_limits(void)
{
    struct rlimit rl;

    // 增加文件描述符限制
    getrlimit(RLIMIT_NOFILE, &rl);
    printf("Current NOFILE: soft=%lu, hard=%lu\n",
           (unsigned long)rl.rlim_cur, (unsigned long)rl.rlim_max);

    rl.rlim_cur = 4096;
    if (rl.rlim_cur > rl.rlim_max) {
        rl.rlim_cur = rl.rlim_max;
    }

    if (setrlimit(RLIMIT_NOFILE, &rl) == -1) {
        perror("setrlimit NOFILE");
    }

    // 限制核心转储大小
    rl.rlim_cur = 0;
    rl.rlim_max = 0;
    setrlimit(RLIMIT_CORE, &rl);
    printf("Core dumps disabled\n");

    // 限制文件大小（防止程序创建超大文件）
    rl.rlim_cur = 100 * 1024 * 1024;  // 100MB
    rl.rlim_max = 100 * 1024 * 1024;
    setrlimit(RLIMIT_FSIZE, &rl);
}
```

## 2. ulimit 与 shell 资源控制

```c
#include <stdio.h>
#include <sys/resource.h>
#include <unistd.h>

/*
 * ulimit 命令等价的系统调用:
 *
 * ulimit -c  -> RLIMIT_CORE
 * ulimit -f  -> RLIMIT_FSIZE
 * ulimit -n  -> RLIMIT_NOFILE
 * ulimit -s  -> RLIMIT_STACK
 * ulimit -u  -> RLIMIT_NPROC
 * ulimit -v  -> RLIMIT_AS
 * ulimit -t  -> RLIMIT_CPU
 */

// 在程序中实现简单的 ulimit 功能
int my_ulimit(const char *opt, long value)
{
    struct rlimit rl;
    int resource;

    switch (opt[0]) {
    case 'c': resource = RLIMIT_CORE; break;
    case 'f': resource = RLIMIT_FSIZE; break;
    case 'n': resource = RLIMIT_NOFILE; break;
    case 's': resource = RLIMIT_STACK; break;
    case 'u': resource = RLIMIT_NPROC; break;
    case 'v': resource = RLIMIT_AS; break;
    case 't': resource = RLIMIT_CPU; break;
    default:
        fprintf(stderr, "Unknown option: %s\n", opt);
        return -1;
    }

    if (value < 0) {
        // 查询
        getrlimit(resource, &rl);
        printf("soft=%lu, hard=%lu\n",
               (unsigned long)rl.rlim_cur,
               (unsigned long)rl.rlim_max);
    } else {
        // 设置
        getrlimit(resource, &rl);
        rl.rlim_cur = value;
        if (setrlimit(resource, &rl) == -1) {
            perror("setrlimit");
            return -1;
        }
    }

    return 0;
}
```

## 3. cgroup 基础

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

/*
 * cgroup (Control Group) 是 Linux 内核的资源管理机制
 * 通过文件系统接口进行配置
 *
 * 常用控制器:
 *   cpu     - CPU 时间限制
 *   memory  - 内存使用限制
 *   blkio   - 块设备 IO 限制
 *   pids    - 进程数限制
 *   cpuset  - CPU 亲和性
 *
 * 典型路径: /sys/fs/cgroup/<controller>/
 */

// 将进程加入 cgroup
int add_to_cgroup(const char *cgroup_path, pid_t pid)
{
    char path[512];
    snprintf(path, sizeof(path), "%s/tasks", cgroup_path);

    FILE *fp = fopen(path, "w");
    if (!fp) {
        perror("fopen cgroup tasks");
        return -1;
    }

    fprintf(fp, "%d\n", pid);
    fclose(fp);
    return 0;
}

// 设置内存限制
int set_memory_limit(const char *cgroup_path, size_t limit_mb)
{
    char path[512];
    snprintf(path, sizeof(path), "%s/memory.limit_in_bytes", cgroup_path);

    FILE *fp = fopen(path, "w");
    if (!fp) {
        perror("fopen memory.limit_in_bytes");
        return -1;
    }

    fprintf(fp, "%zu\n", limit_mb * 1024 * 1024);
    fclose(fp);
    return 0;
}

// 设置 CPU 配额
int set_cpu_quota(const char *cgroup_path, int quota_us, int period_us)
{
    char path[512];

    // 设置周期
    snprintf(path, sizeof(path), "%s/cpu.cfs_period_us", cgroup_path);
    FILE *fp = fopen(path, "w");
    if (!fp) return -1;
    fprintf(fp, "%d\n", period_us);
    fclose(fp);

    // 设置配额
    snprintf(path, sizeof(path), "%s/cpu.cfs_quota_us", cgroup_path);
    fp = fopen(path, "w");
    if (!fp) return -1;
    fprintf(fp, "%d\n", quota_us);
    fclose(fp);

    return 0;
}

void cgroup_demo(void)
{
    const char *cg = "/sys/fs/cgroup/memory/my_app";

    // 创建 cgroup（需要 root 权限）
    mkdir(cg, 0755);

    // 设置内存限制为 512MB
    set_memory_limit(cg, 512);

    // 将当前进程加入
    add_to_cgroup(cg, getpid());

    printf("Process %d in cgroup with 512MB memory limit\n", getpid());
}
```
