# 数据库编程

## 1. SQLite C API 基础

```c
#include <sqlite3.h>
#include <stdio.h>
#include <string.h>

/*
 * SQLite 是嵌入式数据库，无需服务器
 * 编译: gcc -o db_demo db_demo.c -lsqlite3
 *
 * 核心 API:
 *   int sqlite3_open(const char *filename, sqlite3 **ppDb);
 *   int sqlite3_close(sqlite3 *db);
 *   int sqlite3_exec(sqlite3 *db, const char *sql,
 *                     int (*callback)(void*,int,char**,char**),
 *                     void *arg, char **errmsg);
 *   int sqlite3_prepare_v2(sqlite3 *db, const char *zSql,
 *                           int nByte, sqlite3_stmt **ppStmt,
 *                           const char **pzTail);
 *   int sqlite3_step(sqlite3_stmt *pStmt);
 *   int sqlite3_finalize(sqlite3_stmt *pStmt);
 */

// 回调函数
int callback(void *data, int argc, char **argv, char **col_name)
{
    (void)data;
    for (int i = 0; i < argc; i++) {
        printf("%s = %s\n", col_name[i], argv[i] ? argv[i] : "NULL");
    }
    printf("---\n");
    return 0;
}

void sqlite_basic_example(void)
{
    sqlite3 *db;
    char *err_msg = NULL;

    // 打开/创建数据库
    int rc = sqlite3_open("test.db", &db);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "Cannot open DB: %s\n", sqlite3_errmsg(db));
        return;
    }

    // 创建表
    const char *sql =
        "CREATE TABLE IF NOT EXISTS users("
        "id INTEGER PRIMARY KEY AUTOINCREMENT,"
        "name TEXT NOT NULL,"
        "email TEXT,"
        "age INTEGER);";

    rc = sqlite3_exec(db, sql, 0, 0, &err_msg);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "SQL error: %s\n", err_msg);
        sqlite3_free(err_msg);
    }

    // 插入数据（使用 exec）
    sql = "INSERT INTO users (name, email, age) VALUES "
          "('Alice', 'alice@example.com', 25),"
          "('Bob', 'bob@example.com', 30),"
          "('Charlie', 'charlie@example.com', 35);";

    rc = sqlite3_exec(db, sql, 0, 0, &err_msg);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "Insert error: %s\n", err_msg);
        sqlite3_free(err_msg);
    }

    // 查询数据
    sql = "SELECT * FROM users;";
    sqlite3_exec(db, sql, callback, NULL, &err_msg);

    sqlite3_close(db);
}
```

## 2. 预编译语句

```c
#include <sqlite3.h>
#include <stdio.h>

void prepared_statement_example(void)
{
    sqlite3 *db;
    sqlite3_stmt *stmt;

    sqlite3_open("test.db", &db);

    // 准备插入语句
    const char *insert_sql =
        "INSERT INTO users (name, email, age) VALUES (?, ?, ?);";

    sqlite3_prepare_v2(db, insert_sql, -1, &stmt, NULL);

    // 绑定参数并执行
    const char *names[] = {"David", "Eve", "Frank"};
    const char *emails[] = {"david@test.com", "eve@test.com", "frank@test.com"};
    int ages[] = {28, 22, 40};

    for (int i = 0; i < 3; i++) {
        sqlite3_bind_text(stmt, 1, names[i], -1, SQLITE_STATIC);
        sqlite3_bind_text(stmt, 2, emails[i], -1, SQLITE_STATIC);
        sqlite3_bind_int(stmt, 3, ages[i]);

        sqlite3_step(stmt);
        sqlite3_reset(stmt);  // 重置以便下次使用
    }

    sqlite3_finalize(stmt);

    // 准备查询语句
    const char *select_sql =
        "SELECT id, name, age FROM users WHERE age > ?;";

    sqlite3_prepare_v2(db, select_sql, -1, &stmt, NULL);
    sqlite3_bind_int(stmt, 1, 25);

    printf("Users older than 25:\n");
    while (sqlite3_step(stmt) == SQLITE_ROW) {
        int id = sqlite3_column_int(stmt, 0);
        const char *name = (const char *)sqlite3_column_text(stmt, 1);
        int age = sqlite3_column_int(stmt, 2);
        printf("  ID=%d, Name=%s, Age=%d\n", id, name, age);
    }

    sqlite3_finalize(stmt);
    sqlite3_close(db);
}
```

## 3. 事务处理

```c
#include <sqlite3.h>
#include <stdio.h>

void transaction_example(void)
{
    sqlite3 *db;
    char *err_msg = NULL;

    sqlite3_open("test.db", &db);

    // 开始事务
    sqlite3_exec(db, "BEGIN TRANSACTION;", 0, 0, &err_msg);

    // 批量插入（事务内更快）
    sqlite3_exec(db, "PRAGMA synchronous=OFF;", 0, 0, NULL);

    const char *sql =
        "INSERT INTO users (name, email, age) VALUES "
        "('Grace', 'grace@test.com', 27),"
        "('Henry', 'henry@test.com', 33);";

    int rc = sqlite3_exec(db, sql, 0, 0, &err_msg);

    if (rc == SQLITE_OK) {
        sqlite3_exec(db, "COMMIT;", 0, 0, &err_msg);
        printf("Transaction committed\n");
    } else {
        sqlite3_exec(db, "ROLLBACK;", 0, 0, &err_msg);
        fprintf(stderr, "Transaction rolled back: %s\n", err_msg);
        sqlite3_free(err_msg);
    }

    sqlite3_close(db);
}
```

## 4. BDB（Berkeley DB）简介

```c
#include <db.h>
#include <stdio.h>
#include <string.h>

/*
 * BDB 是高性能嵌入式键值数据库
 * 编译: gcc -o bdb_demo bdb_demo.c -ldb
 *
 * 类型:
 *   DB_BTREE  - B+树
 *   DB_HASH   - 哈希表
 *   DB_QUEUE  - 定长记录队列
 *   DB_RECNO  - 变长记录队列
 */

void bdb_example(void)
{
    DB *dbp;
    DBT key, data;
    int ret;

    // 创建数据库句柄
    db_create(&dbp, NULL, 0);

    // 打开数据库
    ret = dbp->open(dbp, NULL, "mydb.db", NULL, DB_BTREE,
                    DB_CREATE, 0664);
    if (ret != 0) {
        fprintf(stderr, "DB open failed: %s\n", db_strerror(ret));
        return;
    }

    // 写入数据
    memset(&key, 0, sizeof(key));
    memset(&data, 0, sizeof(data));

    key.data = "user:1";
    key.size = strlen("user:1");

    const char *value = "Alice|25|alice@example.com";
    data.data = (void *)value;
    data.size = strlen(value) + 1;

    ret = dbp->put(dbp, NULL, &key, &data, 0);
    if (ret == 0) printf("Put: %s -> %s\n", (char *)key.data, value);

    // 读取数据
    memset(&data, 0, sizeof(data));
    ret = dbp->get(dbp, NULL, &key, &data, 0);
    if (ret == 0) {
        printf("Get: %s -> %s\n", (char *)key.data, (char *)data.data);
    }

    // 删除数据
    ret = dbp->del(dbp, NULL, &key, 0);
    if (ret == 0) printf("Deleted: %s\n", (char *)key.data);

    // 关闭数据库
    dbp->close(dbp, 0);
}
```
