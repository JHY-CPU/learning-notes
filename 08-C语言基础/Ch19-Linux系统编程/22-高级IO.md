# 高级 IO

## 1. 分散聚集 IO（Scatter/Gather）

```c
#include <sys/uio.h>
#include <unistd.h>
#include <fcntl.h>
#include <stdio.h>
#include <string.h>

/*
 * ssize_t readv(int fd, const struct iovec *iov, int iovcnt);
 * ssize_t writev(int fd, const struct iovec *iov, int iovcnt);
 *
 * struct iovec {
 *     void  *iov_base;
 *     size_t iov_len;
 * };
 *
 * 优势：一次系统调用操作多个缓冲区
 * 应用：HTTP 响应头 + 文件内容一次性写入
 */

void writev_http_response(int conn_fd, const char *body, size_t body_len)
{
    // HTTP 响应头
    char header[256];
    int header_len = snprintf(header, sizeof(header),
        "HTTP/1.1 200 OK\r\n"
        "Content-Length: %zu\r\n"
        "Content-Type: text/html\r\n"
        "Connection: close\r\n"
        "\r\n", body_len);

    struct iovec iov[2];
    iov[0].iov_base = header;
    iov[0].iov_len  = header_len;
    iov[1].iov_base = (void *)body;
    iov[1].iov_len  = body_len;

    // 一次系统调用写入头和体
    writev(conn_fd, iov, 2);
}

void readv_example(void)
{
    int fd = open("data.txt", O_RDONLY);
    if (fd < 0) { perror("open"); return; }

    char buf1[64], buf2[128], buf3[256];
    struct iovec iov[3];

    iov[0].iov_base = buf1;
    iov[0].iov_len  = sizeof(buf1);
    iov[1].iov_base = buf2;
    iov[1].iov_len  = sizeof(buf2);
    iov[2].iov_base = buf3;
    iov[2].iov_len  = sizeof(buf3);

    ssize_t n = readv(fd, iov, 3);
    printf("Read %zd bytes total\n", n);

    close(fd);
}
```

## 2. pread / pwrite

```c
#include <unistd.h>
#include <fcntl.h>
#include <stdio.h>

/*
 * ssize_t pread(int fd, void *buf, size_t count, off_t offset);
 * ssize_t pwrite(int fd, const void *buf, size_t count, off_t offset);
 *
 * 原子性操作：不改变文件偏移量
 * 适用于多线程共享同一文件描述符的场景
 */

void pread_pwrite_demo(void)
{
    int fd = open("data.bin", O_RDWR | O_CREAT, 0644);
    if (fd < 0) { perror("open"); return; }

    // 写入初始数据
    char data[] = "0123456789ABCDEFGHIJ";
    write(fd, data, strlen(data));

    // 在不同偏移量处原子读写
    char buf[11];
    ssize_t n;

    // 从偏移 5 读取 10 字节（不改变文件偏移量）
    n = pread(fd, buf, 10, 5);
    buf[n] = '\0';
    printf("Read from offset 5: %s\n", buf);

    // 在偏移 0 写入（不影响其他线程的偏移量）
    pwrite(fd, "XXXX", 4, 0);

    // 从偏移 10 读取
    n = pread(fd, buf, 10, 10);
    buf[n] = '\0';
    printf("Read from offset 10: %s\n", buf);

    close(fd);
}
```

## 3. sendfile - 零拷贝传输

```c
#include <sys/sendfile.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <stdio.h>
#include <unistd.h>

/*
 * ssize_t sendfile(int out_fd, int in_fd, off_t *offset, size_t count);
 *
 * 在内核空间直接传输数据，避免用户空间拷贝
 * in_fd  必须支持 mmap（通常是普通文件）
 * out_fd 必须是 socket
 *
 * 优势：减少数据拷贝次数，提高性能
 * 应用：Web 服务器发送静态文件
 */

void sendfile_example(const char *filename, int client_fd)
{
    int file_fd = open(filename, O_RDONLY);
    if (file_fd < 0) { perror("open"); return; }

    struct stat st;
    fstat(file_fd, &st);

    // 先发送 HTTP 头
    char header[256];
    int hlen = snprintf(header, sizeof(header),
        "HTTP/1.1 200 OK\r\n"
        "Content-Length: %ld\r\n"
        "Content-Type: application/octet-stream\r\n"
        "\r\n", (long)st.st_size);
    write(client_fd, header, hlen);

    // 零拷贝发送文件内容
    off_t offset = 0;
    size_t remaining = st.st_size;
    while (remaining > 0) {
        ssize_t n = sendfile(client_fd, file_fd, &offset, remaining);
        if (n <= 0) break;
        remaining -= n;
    }

    close(file_fd);
}
```

## 4. splice - 管道零拷贝

```c
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

/*
 * ssize_t splice(int fd_in, loff_t *off_in,
 *                int fd_out, loff_t *off_out,
 *                size_t len, unsigned int flags);
 *
 * 在两个文件描述符之间移动数据，零拷贝
 * 至少有一端必须是管道
 *
 * flags:
 *   SPLICE_F_MOVE     - 移动页面
 *   SPLICE_F_NONBLOCK - 非阻塞
 *   SPLICE_F_MORE     - 还有更多数据
 */

void splice_echo(int in_fd, int out_fd)
{
    int pipefd[2];
    pipe(pipefd);

    while (1) {
        // 从 in_fd 移动到管道（零拷贝）
        ssize_t n = splice(in_fd, NULL, pipefd[1], NULL,
                           65536, SPLICE_F_NONBLOCK);
        if (n <= 0) break;

        // 从管道移动到 out_fd（零拷贝）
        splice(pipefd[0], NULL, out_fd, NULL,
               n, SPLICE_F_NONBLOCK);
    }

    close(pipefd[0]);
    close(pipefd[1]);
}

// 使用 splice 的回显服务器
void splice_server(int port)
{
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 5);

    while (1) {
        int conn_fd = accept(listen_fd, NULL, NULL);
        if (conn_fd < 0) continue;

        splice_echo(conn_fd, conn_fd);
        close(conn_fd);
    }
}
```

## 5. tee - 管道复制

```c
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

/*
 * ssize_t tee(int fd_in, int fd_out, size_t len, unsigned int flags);
 *
 * 在两个管道之间复制数据（不消耗源数据）
 * 类似于 tee 命令：同时输出到两个目的地
 */

void tee_example(void)
{
    int pipe1[2], pipe2[2];
    pipe(pipe1);
    pipe(pipe2);

    // 写入一些数据到 pipe1
    write(pipe1[1], "Hello, tee!", 11);
    close(pipe1[1]);

    // tee 复制数据到 pipe2（不消耗 pipe1 的数据）
    ssize_t n = tee(pipe1[0], pipe2[1], 65536, 0);
    printf("tee copied %zd bytes\n", n);

    // 从两个管道读取
    char buf1[64], buf2[64];

    n = read(pipe1[0], buf1, sizeof(buf1));
    buf1[n] = '\0';
    printf("pipe1: %s\n", buf1);

    n = read(pipe2[0], buf2, sizeof(buf2));
    buf2[n] = '\0';
    printf("pipe2: %s\n", buf2);

    close(pipe1[0]);
    close(pipe2[0]);
    close(pipe2[1]);
}
```

## 6. 异步 IO（AIO）

```c
#include <aio.h>
#include <fcntl.h>
#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <errno.h>

/*
 * POSIX AIO:
 *   int aio_read(struct aiocb *aiocbp);
 *   int aio_write(struct aiocb *aiocbp);
 *   int aio_error(const struct aiocb *aiocbp);
 *   ssize_t aio_return(struct aiocb *aiocbp);
 *   int aio_suspend(const struct aiocb *const list[],
 *                   int nitems, const struct timespec *timeout);
 *   int aio_cancel(int fd, struct aiocb *aiocbp);
 *   int lio_listio(int mode, struct aiocb *const list[],
 *                  int nitems, struct sigevent *sig);
 */

// 异步读取示例
void aio_read_example(const char *filename)
{
    int fd = open(filename, O_RDONLY);
    if (fd < 0) { perror("open"); return; }

    struct stat st;
    fstat(fd, &st);

    char *buf = malloc(st.st_size + 1);

    struct aiocb cb;
    memset(&cb, 0, sizeof(cb));
    cb.aio_fildes = fd;
    cb.aio_buf    = buf;
    cb.aio_nbytes = st.st_size;
    cb.aio_offset = 0;

    // 发起异步读取
    if (aio_read(&cb) == -1) {
        perror("aio_read");
        free(buf);
        close(fd);
        return;
    }

    // 可以做其他工作...
    printf("Doing other work while reading...\n");

    // 等待完成
    while (aio_error(&cb) == EINPROGRESS) {
        usleep(1000);
    }

    ssize_t n = aio_return(&cb);
    if (n > 0) {
        buf[n] = '\0';
        printf("Read %zd bytes\n", n);
    }

    free(buf);
    close(fd);
}

// 使用信号通知完成
static volatile sig_atomic_t aio_done = 0;

void aio_sig_handler(int sig, siginfo_t *info, void *ucontext)
{
    (void)sig;
    (void)ucontext;
    struct aiocb *cb = (struct aiocb *)info->si_value.sival_ptr;
    printf("AIO completed: %zd bytes\n", aio_return(cb));
    aio_done = 1;
}

void aio_with_signal(void)
{
    int fd = open("data.txt", O_RDONLY);
    char buf[4096];

    struct aiocb cb;
    memset(&cb, 0, sizeof(cb));
    cb.aio_fildes = fd;
    cb.aio_buf    = buf;
    cb.aio_nbytes = sizeof(buf);

    // 设置信号通知
    cb.aio_sigevent.sigev_notify = SIGEV_SIGNAL;
    cb.aio_sigevent.sigev_signo  = SIGUSR1;
    cb.aio_sigevent.sigev_value.sival_ptr = &cb;

    signal(SIGUSR1, (void(*)(int))aio_sig_handler);

    aio_read(&cb);

    while (!aio_done) {
        sleep(1);
    }

    close(fd);
}
```
