# 线程同步

## 1. 互斥锁（Mutex）

```c
#include <pthread.h>
#include <stdio.h>
#include <string.h>

/*
 * int pthread_mutex_init(pthread_mutex_t *mutex,
 *                        const pthread_mutexattr_t *attr);
 * int pthread_mutex_destroy(pthread_mutex_t *mutex);
 * int pthread_mutex_lock(pthread_mutex_t *mutex);
 * int pthread_mutex_unlock(pthread_mutex_t *mutex);
 * int pthread_mutex_trylock(pthread_mutex_t *mutex);
 * int pthread_mutex_timedlock(pthread_mutex_t *mutex,
 *                             const struct timespec *abs_timeout);
 *
 * 互斥锁属性:
 *   PTHREAD_MUTEX_NORMAL     - 普通锁（不检测死锁）
 *   PTHREAD_MUTEX_ERRORCHECK - 错误检查锁
 *   PTHREAD_MUTEX_RECURSIVE  - 递归锁（可重入）
 *   PTHREAD_MUTEX_DEFAULT    - 默认锁
 */

// 全局共享变量
static int shared_counter = 0;
static pthread_mutex_t counter_mutex = PTHREAD_MUTEX_INITIALIZER;

// 有竞争问题的代码
void *bad_increment(void *arg)
{
    int n = *(int *)arg;
    for (int i = 0; i < n; i++) {
        shared_counter++;  // 非原子操作，有数据竞争
    }
    return NULL;
}

// 正确使用互斥锁
void *good_increment(void *arg)
{
    int n = *(int *)arg;
    for (int i = 0; i < n; i++) {
        pthread_mutex_lock(&counter_mutex);
        shared_counter++;
        pthread_mutex_unlock(&counter_mutex);
    }
    return NULL;
}

void mutex_basic_example(void)
{
    pthread_t t1, t2;
    int count = 1000000;

    shared_counter = 0;
    pthread_create(&t1, NULL, good_increment, &count);
    pthread_create(&t2, NULL, good_increment, &count);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    printf("Counter: %d (expected %d)\n", shared_counter, count * 2);
}

// 动态初始化互斥锁
void mutex_dynamic_example(void)
{
    pthread_mutex_t mutex;
    pthread_mutexattr_t attr;

    pthread_mutexattr_init(&attr);
    pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_ERRORCHECK);

    pthread_mutex_init(&mutex, &attr);

    // 使用互斥锁
    pthread_mutex_lock(&mutex);
    // ... 临界区 ...
    pthread_mutex_unlock(&mutex);

    pthread_mutex_destroy(&mutex);
    pthread_mutexattr_destroy(&attr);
}

// trylock 示例
void *worker_with_trylock(void *arg)
{
    pthread_mutex_t *mutex = (pthread_mutex_t *)arg;

    if (pthread_mutex_trylock(mutex) == 0) {
        printf("Lock acquired, working...\n");
        sleep(1);
        pthread_mutex_unlock(mutex);
    } else {
        printf("Lock is busy, doing something else...\n");
    }
    return NULL;
}
```

## 2. 读写锁（Read-Write Lock）

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

/*
 * int pthread_rwlock_init(pthread_rwlock_t *rwlock,
 *                         const pthread_rwlockattr_t *attr);
 * int pthread_rwlock_destroy(pthread_rwlock_t *rwlock);
 * int pthread_rwlock_rdlock(pthread_rwlock_t *rwlock);   - 读锁
 * int pthread_rwlock_wrlock(pthread_rwlock_t *rwlock);   - 写锁
 * int pthread_rwlock_unlock(pthread_rwlock_t *rwlock);
 * int pthread_rwlock_tryrdlock(pthread_rwlock_t *rwlock);
 * int pthread_rwlock_trywrlock(pthread_rwlock_t *rwlock);
 *
 * 读写锁特点:
 *   - 多个线程可以同时持有读锁（共享）
 *   - 只有一个线程可以持有写锁（独占）
 *   - 写锁优先级高于读锁
 */

static pthread_rwlock_t rwlock = PTHREAD_RWLOCK_INITIALIZER;
static int shared_data = 0;

// 读线程
void *reader(void *arg)
{
    int id = *(int *)arg;

    for (int i = 0; i < 5; i++) {
        pthread_rwlock_rdlock(&rwlock);
        printf("Reader %d: data = %d\n", id, shared_data);
        usleep(100000);  // 模拟读取操作
        pthread_rwlock_unlock(&rwlock);
        usleep(50000);
    }
    return NULL;
}

// 写线程
void *writer(void *arg)
{
    int id = *(int *)arg;

    for (int i = 0; i < 3; i++) {
        pthread_rwlock_wrlock(&rwlock);
        shared_data++;
        printf("Writer %d: set data = %d\n", id, shared_data);
        usleep(200000);  // 模拟写入操作
        pthread_rwlock_unlock(&rwlock);
        usleep(100000);
    }
    return NULL;
}

void rwlock_example(void)
{
    pthread_t readers[3], writers[1];
    int rids[3] = {1, 2, 3};
    int wid = 1;

    pthread_create(&writers[0], NULL, writer, &wid);
    for (int i = 0; i < 3; i++) {
        pthread_create(&readers[i], NULL, reader, &rids[i]);
    }

    pthread_join(writers[0], NULL);
    for (int i = 0; i < 3; i++) {
        pthread_join(readers[i], NULL);
    }
}
```

## 3. 条件变量（Condition Variable）

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

/*
 * int pthread_cond_init(pthread_cond_t *cond,
 *                       const pthread_condattr_t *attr);
 * int pthread_cond_destroy(pthread_cond_t *cond);
 * int pthread_cond_wait(pthread_cond_t *cond,
 *                       pthread_mutex_t *mutex);
 * int pthread_cond_timedwait(pthread_cond_t *cond,
 *                            pthread_mutex_t *mutex,
 *                            const struct timespec *abstime);
 * int pthread_cond_signal(pthread_cond_t *cond);       - 唤醒一个
 * int pthread_cond_broadcast(pthread_cond_t *cond);    - 唤醒所有
 *
 * 使用模式:
 *   等待方:
 *     pthread_mutex_lock(&mutex);
 *     while (!condition) {
 *         pthread_cond_wait(&cond, &mutex);
 *     }
 *     // 处理条件
 *     pthread_mutex_unlock(&mutex);
 *
 *   通知方:
 *     pthread_mutex_lock(&mutex);
 *     condition = 1;
 *     pthread_cond_signal(&cond);
 *     pthread_mutex_unlock(&mutex);
 */

// 生产者-消费者模型
#define QUEUE_SIZE 10

typedef struct {
    int data[QUEUE_SIZE];
    int count;
    int head;
    int tail;
    pthread_mutex_t mutex;
    pthread_cond_t not_empty;  // 队列非空
    pthread_cond_t not_full;   // 队列非满
} BoundedQueue;

void queue_init(BoundedQueue *q)
{
    q->count = 0;
    q->head = 0;
    q->tail = 0;
    pthread_mutex_init(&q->mutex, NULL);
    pthread_cond_init(&q->not_empty, NULL);
    pthread_cond_init(&q->not_full, NULL);
}

void queue_push(BoundedQueue *q, int item)
{
    pthread_mutex_lock(&q->mutex);

    // 等待队列有空间
    while (q->count == QUEUE_SIZE) {
        pthread_cond_wait(&q->not_full, &q->mutex);
    }

    q->data[q->tail] = item;
    q->tail = (q->tail + 1) % QUEUE_SIZE;
    q->count++;

    pthread_cond_signal(&q->not_empty);
    pthread_mutex_unlock(&q->mutex);
}

int queue_pop(BoundedQueue *q)
{
    pthread_mutex_lock(&q->mutex);

    // 等待队列有数据
    while (q->count == 0) {
        pthread_cond_wait(&q->not_empty, &q->mutex);
    }

    int item = q->data[q->head];
    q->head = (q->head + 1) % QUEUE_SIZE;
    q->count--;

    pthread_cond_signal(&q->not_full);
    pthread_mutex_unlock(&q->mutex);

    return item;
}

BoundedQueue g_queue;

void *producer(void *arg)
{
    int id = *(int *)arg;
    for (int i = 0; i < 20; i++) {
        int item = id * 100 + i;
        queue_push(&g_queue, item);
        printf("Producer %d: produced %d\n", id, item);
        usleep(rand() % 100000);
    }
    return NULL;
}

void *consumer(void *arg)
{
    int id = *(int *)arg;
    for (int i = 0; i < 20; i++) {
        int item = queue_pop(&g_queue);
        printf("Consumer %d: consumed %d\n", id, item);
        usleep(rand() % 150000);
    }
    return NULL;
}

void condvar_example(void)
{
    pthread_t prod, cons;
    int pid = 1, cid = 1;

    queue_init(&g_queue);

    pthread_create(&prod, NULL, producer, &pid);
    pthread_create(&cons, NULL, consumer, &cid);

    pthread_join(prod, NULL);
    pthread_join(cons, NULL);
}
```

## 4. 屏障（Barrier）

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

/*
 * int pthread_barrier_init(pthread_barrier_t *barrier,
 *                          const pthread_barrierattr_t *attr,
 *                          unsigned count);
 * int pthread_barrier_destroy(pthread_barrier_t *barrier);
 * int pthread_barrier_wait(pthread_barrier_t *barrier);
 *
 * 屏障用于让多个线程在某个点同步
 * 所有线程都到达屏障点后才继续执行
 */

#define NUM_WORKERS 4

static pthread_barrier_t barrier;

void *phase_worker(void *arg)
{
    int id = *(int *)arg;

    // 阶段1：计算
    printf("Worker %d: Phase 1\n", id);
    sleep(id + 1);

    // 等待所有线程完成阶段1
    printf("Worker %d: Waiting at barrier\n", id);
    pthread_barrier_wait(&barrier);

    // 阶段2：所有线程同时开始
    printf("Worker %d: Phase 2 (all synchronized)\n", id);

    return NULL;
}

void barrier_example(void)
{
    pthread_t workers[NUM_WORKERS];
    int ids[NUM_WORKERS];

    pthread_barrier_init(&barrier, NULL, NUM_WORKERS);

    for (int i = 0; i < NUM_WORKERS; i++) {
        ids[i] = i;
        pthread_create(&workers[i], NULL, phase_worker, &ids[i]);
    }

    for (int i = 0; i < NUM_WORKERS; i++) {
        pthread_join(workers[i], NULL);
    }

    pthread_barrier_destroy(&barrier);
}
```

## 5. 信号量（Semaphore）

```c
#include <semaphore.h>
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

/*
 * POSIX 信号量:
 *   int sem_init(sem_t *sem, int pshared, unsigned int value);
 *   int sem_destroy(sem_t *sem);
 *   int sem_wait(sem_t *sem);      - P 操作（减1，阻塞）
 *   int sem_trywait(sem_t *sem);   - 非阻塞 P
 *   int sem_timedwait(sem_t *sem, const struct timespec *abs_timeout);
 *   int sem_post(sem_t *sem);      - V 操作（加1）
 *   int sem_getvalue(sem_t *sem, int *sval);
 *
 * pshared:
 *   0 - 线程间共享
 *   非0 - 进程间共享（需要放在共享内存中）
 */

// 使用信号量实现互斥
static sem_t sem;

void *sem_worker(void *arg)
{
    int id = *(int *)arg;

    for (int i = 0; i < 5; i++) {
        sem_wait(&sem);  // P 操作

        // 临界区
        printf("Thread %d: in critical section\n", id);
        usleep(100000);

        sem_post(&sem);  // V 操作
        usleep(50000);
    }
    return NULL;
}

void semaphore_mutex_example(void)
{
    pthread_t threads[3];
    int ids[3] = {1, 2, 3};

    sem_init(&sem, 0, 1);  // 初始值为1，等价于互斥锁

    for (int i = 0; i < 3; i++) {
        pthread_create(&threads[i], NULL, sem_worker, &ids[i]);
    }

    for (int i = 0; i < 3; i++) {
        pthread_join(threads[i], NULL);
    }

    sem_destroy(&sem);
}

// 有界缓冲区使用两个信号量
#define BUF_SIZE 5
static int buffer[BUF_SIZE];
static int buf_count = 0;
static sem_t empty_slots;  // 空槽数量
static sem_t full_slots;   // 已用槽数量
static pthread_mutex_t buf_mutex = PTHREAD_MUTEX_INITIALIZER;

void *sem_producer(void *arg)
{
    (void)arg;
    for (int i = 0; i < 10; i++) {
        sem_wait(&empty_slots);          // 等待有空槽
        pthread_mutex_lock(&buf_mutex);

        buffer[buf_count++] = i;
        printf("Produced: %d (count=%d)\n", i, buf_count);

        pthread_mutex_unlock(&buf_mutex);
        sem_post(&full_slots);           // 增加已用槽

        usleep(rand() % 100000);
    }
    return NULL;
}

void *sem_consumer(void *arg)
{
    (void)arg;
    for (int i = 0; i < 10; i++) {
        sem_wait(&full_slots);           // 等待有数据
        pthread_mutex_lock(&buf_mutex);

        int item = buffer[--buf_count];
        printf("Consumed: %d (count=%d)\n", item, buf_count);

        pthread_mutex_unlock(&buf_mutex);
        sem_post(&empty_slots);          // 增加空槽

        usleep(rand() % 150000);
    }
    return NULL;
}

void semaphore_buffer_example(void)
{
    pthread_t prod, cons;

    sem_init(&empty_slots, 0, BUF_SIZE);
    sem_init(&full_slots, 0, 0);

    pthread_create(&prod, NULL, sem_producer, NULL);
    pthread_create(&cons, NULL, sem_consumer, NULL);

    pthread_join(prod, NULL);
    pthread_join(cons, NULL);

    sem_destroy(&empty_slots);
    sem_destroy(&full_slots);
}
```
