# 进程关系

## 1. 进程组

```c
#include <unistd.h>
#include <sys/types.h>
#include <stdio.h>

/*
 * 每个进程属于一个进程组
 * 进程组 ID (PGID) = 组长进程的 PID
 * 组长进程：PGID == PID
 *
 * pid_t getpgrp(void);               - 获取当前进程组
 * pid_t getpgid(pid_t pid);          - 获取指定进程的进程组
 * int setpgid(pid_t pid, pid_t pgid); - 设置进程组
 */

void show_process_group(void)
{
    printf("PID: %d\n", getpid());
    printf("PGID: %d\n", getpgrp());

    // 查看父进程的进程组
    printf("Parent PGID: %d\n", getpgid(getppid()));
}

void process_group_example(void)
{
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程创建新的进程组
        setpgid(0, 0);  // 自己成为组长
        printf("Child: PID=%d, PGID=%d\n", getpid(), getpgrp());
        sleep(5);
        _exit(0);
    } else {
        sleep(1);
        printf("Parent: PID=%d, PGID=%d\n", getpid(), getpgrp());
        printf("Child PID=%d, PGID=%d\n", pid, getpgid(pid));
        wait(NULL);
    }
}
```

## 2. 会话（Session）

```c
#include <unistd.h>
#include <stdio.h>

/*
 * pid_t setsid(void);
 * pid_t getsid(pid_t pid);
 *
 * setsid 创建新会话:
 *   1. 调用进程成为新会话的会话首进程 (session leader)
 *   2. 调用进程成为新进程组的组长
 *   3. 调用进程没有控制终端
 *
 * 限制：进程组组长调用 setsid 会失败
 */

void session_example(void)
{
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程创建新会话
        pid_t sid = setsid();
        if (sid == -1) {
            perror("setsid");
            _exit(1);
        }
        printf("New session: SID=%d, PID=%d, PGID=%d\n",
               sid, getpid(), getpgrp());

        // 此进程没有控制终端
        _exit(0);
    } else {
        wait(NULL);
    }
}

// 检查是否是会话首进程
void check_session_leader(void)
{
    pid_t sid = getsid(0);
    if (sid == getpid()) {
        printf("I am the session leader\n");
    } else {
        printf("Session leader is PID %d\n", sid);
    }
}
```

## 3. 控制终端

```c
#include <unistd.h>
#include <stdio.h>
#include <fcntl.h>

/*
 * 每个会话最多有一个控制终端
 * 会话首进程自动获得控制终端
 *
 * 前台进程组：接收终端输入和信号
 * 后台进程组：不接收终端输入，但仍可接收信号
 *
 * 终端产生的信号：
 *   SIGINT  - Ctrl+C
 *   SIGQUIT - Ctrl+\
 *   SIGTSTP - Ctrl+Z
 *   SIGHUP  - 终端断开
 */

void show_terminal_info(void)
{
    // 获取控制终端的文件名
    char *tty = ctermid(NULL);
    printf("Controlling terminal: %s\n", tty);

    // 获取终端文件描述符
    int fd = open(ctermid(NULL), O_RDONLY);
    if (fd != -1) {
        printf("Terminal fd: %d\n", fd);
        close(fd);
    }

    // tcgetpgrp - 获取前台进程组
    pid_t fg_pgid = tcgetpgrp(STDIN_FILENO);
    printf("Foreground process group: %d\n", fg_pgid);

    // tcsetpgrp - 设置前台进程组
    // tcsetpgrp(STDIN_FILENO, some_pgid);
}
```

## 4. 守护进程基础概念

```c
#include <unistd.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>

/*
 * 守护进程（Daemon）特点:
 *   1. 生命周期长（通常随系统启动到关闭）
 *   2. 在后台运行，没有控制终端
 *   3. 父进程是 init（PID 1）
 *   4. 不受用户登录/注销影响
 *
 * 常见守护进程:
 *   httpd - Web 服务器
 *   sshd  - SSH 服务器
 *   crond - 定时任务
 *   syslogd - 系统日志
 */

// 守护进程化的基本步骤
int become_daemon(void)
{
    pid_t pid = fork();
    if (pid == -1) return -1;
    if (pid > 0) _exit(0);  // 父进程退出

    // 1. 创建新会话
    if (setsid() == -1) return -1;

    // 2. 第二次 fork（可选但推荐）
    //    防止会话首进程重新获取控制终端
    pid = fork();
    if (pid == -1) return -1;
    if (pid > 0) _exit(0);

    // 3. 清除文件创建掩码
    umask(0);

    // 4. 改变工作目录
    if (chdir("/") == -1) return -1;

    // 5. 关闭不需要的文件描述符
    close(STDIN_FILENO);
    close(STDOUT_FILENO);
    close(STDERR_FILENO);

    // 6. 重定向标准IO到 /dev/null
    open("/dev/null", O_RDWR);  // fd 0 - stdin
    dup(0);                      // fd 1 - stdout
    dup(0);                      // fd 2 - stderr

    return 0;
}

// 使用示例
int daemon_example(void)
{
    if (become_daemon() == -1) {
        // 注意：此时 stderr 已关闭，无法打印错误
        _exit(1);
    }

    // 守护进程主循环
    while (1) {
        // ... 守护进程的工作 ...
        sleep(10);
    }

    return 0;
}
```

## 5. 孤儿进程组

```c
#include <unistd.h>
#include <stdio.h>
#include <signal.h>
#include <sys/wait.h>

/*
 * 孤儿进程组（Orphaned Process Group）:
 *   进程组中的每个进程的父进程
 *   要么是该组的成员，要么是其他会话的成员
 *
 * 当进程组成为孤儿时，如果组中任何进程停止（stopped），
 * 系统会向所有成员发送 SIGHUP 和 SIGCONT
 */

void orphaned_process_group(void)
{
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程
        printf("Child: PID=%d, PGID=%d\n", getpid(), getpgrp());

        // 子进程设置自己的进程组
        setpgid(0, 0);

        // 发送 SIGTSTP 给自己（暂停）
        // 注意：当父进程退出，此进程组变成孤儿进程组
        // 系统会自动发送 SIGHUP 和 SIGCONT
        kill(getpid(), SIGTSTP);

        printf("Child continues after orphan handling\n");
        _exit(0);
    } else {
        // 父进程立即退出，子进程成为孤儿
        printf("Parent exiting, child becomes orphan\n");
        // 不调用 wait，让 init 接管
    }
}
```
