# IO 多路复用

## 1. select

```c
#include <sys/select.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>

/*
 * int select(int nfds, fd_set *readfds, fd_set *writefds,
 *            fd_set *exceptfds, struct timeval *timeout);
 *
 * void FD_ZERO(fd_set *set);
 * void FD_SET(int fd, fd_set *set);
 * void FD_CLR(int fd, fd_set *set);
 * int  FD_ISSET(int fd, fd_set *set);
 *
 * nfds: 最大文件描述符 + 1
 * timeout: NULL 永久等待，{0,0} 非阻塞
 *
 * 缺陷:
 *   1. FD_SETSIZE 限制（通常 1024）
 *   2. 每次需要重新设置 fd_set
 *   3. 需要遍历所有 fd 检查就绪状态
 */

#define MAX_CLIENTS FD_SETSIZE

void select_server(int port)
{
    int listen_fd, client_fds[MAX_CLIENTS];
    int max_fd, nready;
    fd_set readfds;
    char buf[1024];

    listen_fd = socket(AF_INET, SOCK_STREAM, 0);

    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 10);

    // 初始化客户端数组
    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_fds[i] = -1;
    }

    printf("Select server on port %d\n", port);

    while (1) {
        // 设置 fd_set
        FD_ZERO(&readfds);
        FD_SET(listen_fd, &readfds);
        max_fd = listen_fd;

        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (client_fds[i] != -1) {
                FD_SET(client_fds[i], &readfds);
                if (client_fds[i] > max_fd) {
                    max_fd = client_fds[i];
                }
            }
        }

        // 等待事件
        nready = select(max_fd + 1, &readfds, NULL, NULL, NULL);
        if (nready < 0) { perror("select"); continue; }

        // 检查新连接
        if (FD_ISSET(listen_fd, &readfds)) {
            int conn_fd = accept(listen_fd, NULL, NULL);
            printf("New connection: fd=%d\n", conn_fd);

            // 添加到客户端数组
            int added = 0;
            for (int i = 0; i < MAX_CLIENTS; i++) {
                if (client_fds[i] == -1) {
                    client_fds[i] = conn_fd;
                    added = 1;
                    break;
                }
            }
            if (!added) {
                printf("Too many clients\n");
                close(conn_fd);
            }
            if (--nready <= 0) continue;
        }

        // 检查客户端数据
        for (int i = 0; i < MAX_CLIENTS && nready > 0; i++) {
            if (client_fds[i] == -1) continue;
            if (!FD_ISSET(client_fds[i], &readfds)) continue;

            ssize_t n = read(client_fds[i], buf, sizeof(buf) - 1);
            if (n <= 0) {
                printf("Client fd=%d disconnected\n", client_fds[i]);
                close(client_fds[i]);
                client_fds[i] = -1;
            } else {
                buf[n] = '\0';
                write(client_fds[i], buf, n);  // echo
            }
            nready--;
        }
    }

    close(listen_fd);
}
```

## 2. poll

```c
#include <poll.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>

/*
 * int poll(struct pollfd *fds, nfds_t nfds, int timeout);
 *
 * struct pollfd {
 *     int   fd;       - 文件描述符
 *     short events;   - 关注的事件
 *     short revents;  - 实际发生的事件
 * };
 *
 * 事件:
 *   POLLIN   - 可读
 *   POLLOUT  - 可写
 *   POLLERR  - 错误
 *   POLLHUP  - 挂起
 *   POLLNVAL - 无效描述符
 *
 * 优势：没有 fd 数量限制
 */

#define MAX_FDS 1024

void poll_server(int port)
{
    struct pollfd fds[MAX_FDS];
    int nfds = 0;
    char buf[1024];

    // 创建监听 socket
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 10);

    // 初始化 pollfd 数组
    fds[0].fd = listen_fd;
    fds[0].events = POLLIN;
    nfds = 1;

    printf("Poll server on port %d\n", port);

    while (1) {
        int nready = poll(fds, nfds, -1);
        if (nready < 0) { perror("poll"); continue; }

        // 检查监听 socket
        if (fds[0].revents & POLLIN) {
            int conn_fd = accept(listen_fd, NULL, NULL);
            if (nfds < MAX_FDS) {
                fds[nfds].fd = conn_fd;
                fds[nfds].events = POLLIN;
                nfds++;
                printf("New client: fd=%d\n", conn_fd);
            } else {
                close(conn_fd);
            }
            if (--nready <= 0) continue;
        }

        // 检查客户端
        for (int i = 1; i < nfds && nready > 0; i++) {
            if (fds[i].revents & (POLLIN | POLLERR)) {
                ssize_t n = read(fds[i].fd, buf, sizeof(buf) - 1);
                if (n <= 0) {
                    printf("Client fd=%d closed\n", fds[i].fd);
                    close(fds[i].fd);
                    // 移除（将最后一个移到当前位置）
                    fds[i] = fds[nfds - 1];
                    nfds--;
                    i--;
                } else {
                    buf[n] = '\0';
                    write(fds[i].fd, buf, n);
                }
                nready--;
            }
        }
    }

    close(listen_fd);
}
```

## 3. epoll

```c
#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <fcntl.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <errno.h>

/*
 * epoll 是 Linux 特有的高性能 IO 多路复用机制
 *
 * int epoll_create(int size);          - 创建 epoll 实例
 * int epoll_create1(int flags);        - 创建（推荐）
 * int epoll_ctl(int epfd, int op, int fd, struct epoll_event *event);  - 控制
 * int epoll_wait(int epfd, struct epoll_event *events, int maxevents, int timeout);  - 等待
 *
 * epoll_ctl op:
 *   EPOLL_CTL_ADD - 添加
 *   EPOLL_CTL_MOD - 修改
 *   EPOLL_CTL_DEL - 删除
 *
 * 事件:
 *   EPOLLIN  - 可读
 *   EPOLLOUT - 可写
 *   EPOLLET  - 边缘触发模式
 *   EPOLLRDHUP - 对端关闭
 *   EPOLLONESHOT - 一次性触发
 */

#define MAX_EVENTS 64
#define BUF_SIZE 4096

int set_nonblocking(int fd)
{
    int flags = fcntl(fd, F_GETFL, 0);
    return fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

void epoll_server(int port)
{
    int listen_fd, epoll_fd;
    struct epoll_event ev, events[MAX_EVENTS];
    char buf[BUF_SIZE];

    // 创建监听 socket
    listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    set_nonblocking(listen_fd);

    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 128);

    // 创建 epoll 实例
    epoll_fd = epoll_create1(0);
    if (epoll_fd == -1) { perror("epoll_create1"); return; }

    // 添加监听 socket 到 epoll
    ev.events = EPOLLIN;
    ev.data.fd = listen_fd;
    epoll_ctl(epoll_fd, EPOLL_CTL_ADD, listen_fd, &ev);

    printf("Epoll server (LT mode) on port %d\n", port);

    while (1) {
        int nfds = epoll_wait(epoll_fd, events, MAX_EVENTS, -1);

        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == listen_fd) {
                // 新连接
                while (1) {
                    struct sockaddr_in client_addr;
                    socklen_t client_len = sizeof(client_addr);
                    int conn_fd = accept(listen_fd,
                                         (struct sockaddr *)&client_addr,
                                         &client_len);
                    if (conn_fd == -1) {
                        if (errno == EAGAIN || errno == EWOULDBLOCK)
                            break;
                        perror("accept");
                        break;
                    }

                    set_nonblocking(conn_fd);
                    ev.events = EPOLLIN | EPOLLRDHUP;
                    ev.data.fd = conn_fd;
                    epoll_ctl(epoll_fd, EPOLL_CTL_ADD, conn_fd, &ev);
                    printf("Client connected: fd=%d\n", conn_fd);
                }
            } else if (events[i].events & EPOLLIN) {
                // 可读事件
                int fd = events[i].data.fd;
                ssize_t n = read(fd, buf, sizeof(buf));
                if (n <= 0) {
                    printf("Client fd=%d closed\n", fd);
                    epoll_ctl(epoll_fd, EPOLL_CTL_DEL, fd, NULL);
                    close(fd);
                } else {
                    write(fd, buf, n);  // echo
                }
            }
        }
    }

    close(listen_fd);
    close(epoll_fd);
}
```

## 4. epoll ET 模式（边缘触发）

```c
#include <sys/epoll.h>
#include <sys/socket.h>
#include <fcntl.h>
#include <stdio.h>
#include <unistd.h>
#include <errno.h>

/*
 * LT (Level Triggered) vs ET (Edge Triggered):
 *
 * LT（水平触发）：
 *   - 默认模式
 *   - 只要缓冲区有数据，每次 epoll_wait 都返回
 *   - 编程简单
 *
 * ET（边缘触发）：
 *   - EPOLLET 标志
 *   - 只在状态变化时通知一次
 *   - 必须一次读完所有数据（非阻塞 + 循环读到 EAGAIN）
 *   - 性能更高，减少了 epoll_wait 调用次数
 */

void et_read_handler(int fd)
{
    char buf[4096];
    ssize_t n;

    // ET 模式必须循环读取直到 EAGAIN
    while (1) {
        n = read(fd, buf, sizeof(buf));
        if (n > 0) {
            write(fd, buf, n);  // echo
        } else if (n == 0) {
            // 客户端关闭
            close(fd);
            break;
        } else {
            if (errno == EAGAIN || errno == EWOULDBLOCK) {
                break;  // 数据读完了
            }
            perror("read");
            close(fd);
            break;
        }
    }
}

void epoll_et_server(int port)
{
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    set_nonblocking(listen_fd);

    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 128);

    int epoll_fd = epoll_create1(0);

    struct epoll_event ev;
    ev.events = EPOLLIN | EPOLLET;  // 边缘触发
    ev.data.fd = listen_fd;
    epoll_ctl(epoll_fd, EPOLL_CTL_ADD, listen_fd, &ev);

    struct epoll_event events[64];

    printf("Epoll server (ET mode) on port %d\n", port);

    while (1) {
        int nfds = epoll_wait(epoll_fd, events, 64, -1);

        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == listen_fd) {
                // ET 模式：必须 accept 到 EAGAIN
                while (1) {
                    int conn_fd = accept(listen_fd, NULL, NULL);
                    if (conn_fd == -1) break;

                    set_nonblocking(conn_fd);
                    ev.events = EPOLLIN | EPOLLET;
                    ev.data.fd = conn_fd;
                    epoll_ctl(epoll_fd, EPOLL_CTL_ADD, conn_fd, &ev);
                }
            } else {
                et_read_handler(events[i].data.fd);
            }
        }
    }

    close(listen_fd);
    close(epoll_fd);
}
```

## 5. epoll vs select/poll 对比

```
特性对比:
┌──────────────┬──────────┬──────────┬──────────┐
│    特性       │  select  │   poll   │  epoll   │
├──────────────┼──────────┼──────────┼──────────┤
│ fd 数量限制   │ 1024     │ 无限制   │ 无限制    │
│ fd 集合传递  │ 每次传递  │ 每次传递  │ epoll_ctl │
│ 就绪检测    │ 遍历      │ 遍历      │ 回调通知  │
│ 时间复杂度  │ O(n)     │ O(n)     │ O(1)     │
│ 触发模式    │ LT       │ LT       │ LT/ET    │
│ 平台       │ 跨平台    │ 跨平台    │ Linux    │
│ 内核数据结构 │ bitmap   │ 链表     │ 红黑树+链表│
└──────────────┴──────────┴──────────┴──────────┘
```
