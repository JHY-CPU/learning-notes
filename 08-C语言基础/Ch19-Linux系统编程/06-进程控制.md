# 进程控制

## 1. 进程标识

```c
#include <unistd.h>
#include <sys/types.h>
#include <stdio.h>

/*
 * 每个进程有唯一的 PID（正整数，最大值由 /proc/sys/kernel/pid_max 决定）
 * 典型最大值：32768（旧系统）或 4194304（新系统）
 */

void show_process_ids(void)
{
    pid_t pid = getpid();       // 当前进程 PID
    pid_t ppid = getppid();     // 父进程 PID
    uid_t uid = getuid();       // 实际用户 ID
    uid_t euid = geteuid();     // 有效用户 ID
    gid_t gid = getgid();       // 实际组 ID
    gid_t egid = getegid();     // 有效组 ID

    printf("PID:  %d\n", pid);
    printf("PPID: %d\n", ppid);
    printf("UID:  %d, EUID: %d\n", uid, euid);
    printf("GID:  %d, EGID: %d\n", gid, egid);

    // 进程组和会话
    pid_t pgid = getpgid(0);
    pid_t sid = getsid(0);
    printf("PGID: %d\n", pgid);
    printf("SID:  %d\n", sid);
}
```

## 2. fork 系统调用

```c
#include <unistd.h>
#include <sys/types.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/wait.h>

/*
 * pid_t fork(void);
 *
 * 返回值:
 *   在父进程中返回子进程的 PID（>0）
 *   在子进程中返回 0
 *   失败返回 -1
 *
 * fork 之后父子进程：
 *   - 共享代码段
 *   - 各自拥有独立的数据段、堆、栈
 *   - 文件描述符表被复制（共享文件表项和偏移量）
 *   - 子进程继承父进程的信号处理设置
 */

void fork_example(void)
{
    printf("Before fork (PID=%d)\n", getpid());

    pid_t pid = fork();

    if (pid == -1) {
        perror("fork");
        return;
    } else if (pid == 0) {
        // 子进程
        printf("Child: PID=%d, PPID=%d\n", getpid(), getppid());
        // 子进程可以执行不同的代码
        exit(42);
    } else {
        // 父进程
        printf("Parent: PID=%d, Child PID=%d\n", getpid(), pid);

        int status;
        waitpid(pid, &status, 0);
        if (WIFEXITED(status)) {
            printf("Child exited with status %d\n", WEXITSTATUS(status));
        }
    }
}

// fork 后的文件描述符行为
void fork_fd_example(void)
{
    int fd = open("test.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程写入
        write(fd, "Child writes\n", 13);
        exit(0);
    } else {
        // 父进程等待子进程完成
        wait(NULL);
        // 父进程写入（注意偏移量是共享的！）
        write(fd, "Parent writes\n", 14);
    }

    close(fd);
    // 文件内容：Child writes\nParent writes\n
}
```

## 3. vfork 系统调用

```c
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>

/*
 * pid_t vfork(void);
 *
 * 与 fork 的区别：
 * 1. vfork 保证子进程先运行，父进程被挂起
 * 2. 子进程共享父进程的地址空间（不是复制）
 * 3. 子进程必须调用 exec 或 _exit，不能 return
 * 4. 更高效但更危险
 *
 * 现代 Linux 中 fork 使用写时复制(Copy-on-Write)，
 * vfork 的性能优势已经很小
 */

void vfork_example(void)
{
    printf("Before vfork\n");

    pid_t pid = vfork();

    if (pid == 0) {
        // 子进程 - 必须调用 exec 或 _exit
        printf("Child (vfork): PID=%d\n", getpid());
        _exit(0);  // 不能 return！
    } else {
        // 父进程 - 子进程 _exit 后才恢复执行
        printf("Parent: child PID=%d\n", pid);
    }
}
```

## 4. exec 系列函数

```c
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/wait.h>

/*
 * exec 系列函数替换当前进程映像为新的程序
 *
 * int execl(const char *path, const char *arg, ...);
 * int execlp(const char *file, const char *arg, ...);
 * int execle(const char *path, const char *arg, ..., char *const envp[]);
 * int execv(const char *path, char *const argv[]);
 * int execvp(const char *file, char *const argv[]);
 * int execvpe(const char *file, char *const argv[], char *const envp[]);
 *
 * 命名规则:
 *   l - 参数列表形式
 *   v - 参数数组形式
 *   p - 在 PATH 中搜索
 *   e - 指定环境变量
 *
 * 成功时不返回，失败返回 -1
 */

void exec_examples(void)
{
    pid_t pid = fork();

    if (pid == 0) {
        // execl - 列出参数，指定完整路径
        execl("/bin/ls", "ls", "-la", "/tmp", NULL);
        perror("execl");  // 只有 exec 失败才会执行到这里
        _exit(1);
    }
    wait(NULL);

    pid = fork();
    if (pid == 0) {
        // execlp - 列出参数，在 PATH 中搜索
        execlp("echo", "echo", "Hello from exec", NULL);
        perror("execlp");
        _exit(1);
    }
    wait(NULL);

    pid = fork();
    if (pid == 0) {
        // execvp - 数组参数
        char *args[] = {"ls", "-la", "--color=auto", NULL};
        execvp("ls", args);
        perror("execvp");
        _exit(1);
    }
    wait(NULL);

    pid = fork();
    if (pid == 0) {
        // execle - 自定义环境变量
        char *env[] = {"PATH=/bin:/usr/bin", "HOME=/tmp", NULL};
        execle("/bin/echo", "echo", "Custom env", NULL, env);
        perror("execle");
        _exit(1);
    }
    wait(NULL);
}

// 实现简易 shell 的核心逻辑
int execute_command(const char *cmd)
{
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程执行命令
        char *args[] = {"sh", "-c", (char *)cmd, NULL};
        execvp("sh", args);
        perror("execvp");
        _exit(127);
    } else if (pid > 0) {
        int status;
        waitpid(pid, &status, 0);
        return WIFEXITED(status) ? WEXITSTATUS(status) : -1;
    }

    return -1;
}
```

## 5. wait 和 waitpid

```c
#include <sys/wait.h>
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>

/*
 * pid_t wait(int *status);
 * pid_t waitpid(pid_t pid, int *status, int options);
 *
 * wait: 等待任意子进程终止
 * waitpid:
 *   pid > 0  - 等待指定 PID 的子进程
 *   pid == 0 - 等待同进程组的任意子进程
 *   pid == -1 - 等待任意子进程
 *   pid < -1 - 等待进程组 |pid| 中的子进程
 *
 * options:
 *   WNOHANG   - 非阻塞
 *   WUNTRACED - 报告已停止的子进程
 *   WCONTINUED - 报告已继续的子进程
 *
 * 状态检查宏:
 *   WIFEXITED(status)    - 正常终止
 *   WEXITSTATUS(status)  - 退出码
 *   WIFSIGNALED(status)  - 被信号终止
 *   WTERMSIG(status)     - 终止信号编号
 *   WCOREDUMP(status)    - 产生核心转储
 *   WIFSTOPPED(status)   - 被信号暂停
 *   WSTOPSIG(status)     - 暂停信号编号
 */

void wait_example(void)
{
    pid_t children[3];
    int num_children = 3;

    for (int i = 0; i < num_children; i++) {
        pid_t pid = fork();
        if (pid == 0) {
            printf("Child %d (PID=%d) starting\n", i, getpid());
            sleep(i + 1);
            printf("Child %d exiting\n", i);
            exit(i);
        }
        children[i] = pid;
    }

    // 等待所有子进程
    for (int i = 0; i < num_children; i++) {
        int status;
        pid_t child_pid = wait(&status);
        // wait 返回哪个子进程是不确定的

        if (WIFEXITED(status)) {
            printf("Child PID %d exited with status %d\n",
                   child_pid, WEXITSTATUS(status));
        }
    }
}

// waitpid 非阻塞示例
void waitpid_nonblocking(void)
{
    pid_t pids[5];

    for (int i = 0; i < 5; i++) {
        pids[i] = fork();
        if (pids[i] == 0) {
            sleep(i + 1);
            exit(i);
        }
    }

    // 非阻塞等待
    int remaining = 5;
    while (remaining > 0) {
        int status;
        pid_t pid = waitpid(-1, &status, WNOHANG);
        if (pid > 0) {
            printf("Child %d finished\n", pid);
            remaining--;
        } else if (pid == 0) {
            // 还没有子进程退出
            printf("Doing other work...\n");
            usleep(500000);
        } else {
            perror("waitpid");
            break;
        }
    }
}

// 检查子进程退出状态的详细方法
void check_exit_status(int status)
{
    if (WIFEXITED(status)) {
        printf("Normal exit, status = %d\n", WEXITSTATUS(status));
    }
    if (WIFSIGNALED(status)) {
        printf("Killed by signal %d\n", WTERMSIG(status));
        if (WCOREDUMP(status)) {
            printf("Core dumped\n");
        }
    }
    if (WIFSTOPPED(status)) {
        printf("Stopped by signal %d\n", WSTOPSIG(status));
    }
    if (WIFCONTINUED(status)) {
        printf("Continued\n");
    }
}
```

## 6. 僵尸进程与孤儿进程

```c
#include <unistd.h>
#include <sys/wait.h>
#include <stdio.h>
#include <stdlib.h>

/*
 * 僵尸进程 (Zombie):
 *   子进程退出后，父进程未调用 wait/waitpid 收集状态
 *   子进程成为僵尸，占用进程表条目
 *
 * 孤儿进程 (Orphan):
 *   父进程先于子进程退出
 *   子进程被 init（PID 1）收养
 *   孤儿进程退出时 init 会自动收集状态
 */

void zombie_example(void)
{
    pid_t pid = fork();

    if (pid == 0) {
        printf("Child %d exiting\n", getpid());
        exit(0);  // 子进程退出
    }

    // 父进程不调用 wait，子进程成为僵尸
    printf("Parent %d sleeping (child %d is zombie now)\n",
           getpid(), pid);
    sleep(10);  // 用 ps aux | grep Z 查看僵尸进程

    // 最终收集子进程
    wait(NULL);
    printf("Child collected\n");
}

// 避免僵尸进程的方法1：忽略 SIGCHLD
void avoid_zombie_sigchld(void)
{
    signal(SIGCHLD, SIG_IGN);  // 忽略 SIGCHLD
    // 内核自动清理子进程，不需要 wait

    pid_t pid = fork();
    if (pid == 0) {
        printf("Child %d\n", getpid());
        exit(0);
    }

    printf("Parent continues, no wait needed\n");
    sleep(2);
}

// 避免僵尸进程的方法2：SIGCHLD 处理函数
void sigchld_handler(int sig)
{
    (void)sig;
    int saved_errno = errno;
    // 循环回收所有已退出的子进程
    while (waitpid(-1, NULL, WNOHANG) > 0)
        ;
    errno = saved_errno;
}

void avoid_zombie_handler(void)
{
    struct sigaction sa;
    sa.sa_handler = sigchld_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = SA_RESTART | SA_NOCLDSTOP;
    sigaction(SIGCHLD, &sa, NULL);

    for (int i = 0; i < 5; i++) {
        if (fork() == 0) {
            printf("Child %d\n", getpid());
            sleep(1);
            exit(0);
        }
    }

    // 父进程做其他工作
    sleep(5);
}
```

## 7. 进程间竞争条件

```c
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/wait.h>

/*
 * fork 之后父子进程的执行顺序是不确定的
 * 如果程序依赖特定执行顺序，需要使用同步机制
 */

void race_condition_example(void)
{
    pid_t pid = fork();

    // 无法确定哪个进程先执行到这一行
    printf("Hello from PID %d\n", getpid());

    if (pid > 0) {
        // 简单的同步：父进程等待子进程
        wait(NULL);
    }
}
```
