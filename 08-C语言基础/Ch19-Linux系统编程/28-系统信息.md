# 系统信息

## 1. uname 系统信息

```c
#include <sys/utsname.h>
#include <stdio.h>
#include <unistd.h>

/*
 * int uname(struct utsname *buf);
 *
 * struct utsname {
 *     char sysname[];    - 操作系统名称
 *     char nodename[];   - 网络节点名称
 *     char release[];    - 内核发行版本
 *     char version[];    - 内核版本
 *     char machine[];    - 硬件架构
 *     char domainname[]; - NIS/YP 域名
 * };
 */

void show_uname(void)
{
    struct utsname uts;

    if (uname(&uts) == -1) {
        perror("uname");
        return;
    }

    printf("System:  %s\n", uts.sysname);
    printf("Node:    %s\n", uts.nodename);
    printf("Release: %s\n", uts.release);
    printf("Version: %s\n", uts.version);
    printf("Machine: %s\n", uts.machine);

    // 其他系统信息
    printf("Hostname: ");
    char hostname[256];
    gethostname(hostname, sizeof(hostname));
    printf("%s\n", hostname);

    // 系统配置信息
    printf("Page size: %ld\n", sysconf(_SC_PAGESIZE));
    printf("CPU count: %ld\n", sysconf(_SC_NPROCESSORS_ONLN));
    printf("Max open files: %ld\n", sysconf(_SC_OPEN_MAX));
    printf("Clock ticks: %ld\n", sysconf(_SC_CLK_TCK));
}
```

## 2. /proc 文件系统

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/*
 * /proc 是虚拟文件系统，提供内核和进程信息
 *
 * 常用文件:
 *   /proc/cpuinfo      - CPU 信息
 *   /proc/meminfo      - 内存信息
 *   /proc/version      - 内核版本
 *   /proc/uptime       - 系统运行时间
 *   /proc/loadavg      - 系统负载
 *   /proc/stat         - 系统统计
 *   /proc/self/        - 当前进程信息
 *   /proc/PID/         - 指定进程信息
 */

// 读取 /proc 文件内容
char *read_proc_file(const char *path)
{
    FILE *fp = fopen(path, "r");
    if (!fp) return NULL;

    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    rewind(fp);

    char *buf = malloc(size + 1);
    if (!buf) { fclose(fp); return NULL; }

    fread(buf, 1, size, fp);
    buf[size] = '\0';
    fclose(fp);

    return buf;
}

void show_cpuinfo(void)
{
    printf("=== CPU Info ===\n");
    FILE *fp = fopen("/proc/cpuinfo", "r");
    if (!fp) { perror("fopen"); return; }

    char line[256];
    int count = 0;
    while (fgets(line, sizeof(line), fp)) {
        if (strncmp(line, "model name", 10) == 0 ||
            strncmp(line, "cpu MHz", 7) == 0 ||
            strncmp(line, "cpu cores", 9) == 0 ||
            strncmp(line, "cache size", 10) == 0) {
            printf("%s", line);
            count++;
        }
        if (count >= 4) break;
    }
    fclose(fp);
}

void show_meminfo(void)
{
    printf("\n=== Memory Info ===\n");
    FILE *fp = fopen("/proc/meminfo", "r");
    if (!fp) { perror("fopen"); return; }

    char line[256];
    for (int i = 0; i < 10 && fgets(line, sizeof(line), fp); i++) {
        printf("%s", line);
    }
    fclose(fp);
}

void show_loadavg(void)
{
    printf("\n=== Load Average ===\n");
    FILE *fp = fopen("/proc/loadavg", "r");
    if (!fp) { perror("fopen"); return; }

    char line[256];
    if (fgets(line, sizeof(line), fp)) {
        printf("%s", line);
    }
    fclose(fp);
}

void show_uptime(void)
{
    printf("\n=== Uptime ===\n");
    FILE *fp = fopen("/proc/uptime", "r");
    if (!fp) { perror("fopen"); return; }

    double uptime, idle;
    if (fscanf(fp, "%lf %lf", &uptime, &idle) == 2) {
        printf("Uptime: %.0f seconds (%.1f days)\n",
               uptime, uptime / 86400.0);
        printf("Idle:   %.0f seconds (%.1f days)\n",
               idle, idle / 86400.0);
    }
    fclose(fp);
}

// 进程信息
void show_proc_status(pid_t pid)
{
    char path[64];
    snprintf(path, sizeof(path), "/proc/%d/status", pid);

    FILE *fp = fopen(path, "r");
    if (!fp) { perror("fopen"); return; }

    printf("\n=== Process %d Status ===\n", pid);
    char line[256];
    while (fgets(line, sizeof(line), fp)) {
        // 只显示关键信息
        if (strncmp(line, "Name:", 5) == 0 ||
            strncmp(line, "State:", 6) == 0 ||
            strncmp(line, "Pid:", 4) == 0 ||
            strncmp(line, "PPid:", 5) == 0 ||
            strncmp(line, "VmSize:", 7) == 0 ||
            strncmp(line, "VmRSS:", 6) == 0 ||
            strncmp(line, "Threads:", 8) == 0) {
            printf("%s", line);
        }
    }
    fclose(fp);
}
```

## 3. 系统资源

```c
#include <sys/resource.h>
#include <sys/sysinfo.h>
#include <stdio.h>

/*
 * int getrusage(int who, struct rusage *usage);
 * int getloadavg(double *loadavg, int nelem);
 * int sysinfo(struct sysinfo *info);
 */

void show_rusage(void)
{
    struct rusage usage;

    getrusage(RUSAGE_SELF, &usage);
    printf("=== Process Resource Usage ===\n");
    printf("User CPU:   %ld.%06ld sec\n",
           usage.ru_utime.tv_sec, usage.ru_utime.tv_usec);
    printf("System CPU: %ld.%06ld sec\n",
           usage.ru_stime.tv_sec, usage.ru_stime.tv_usec);
    printf("Max RSS:    %ld KB\n", usage.ru_maxrss);
    printf("Page faults (no I/O): %ld\n", usage.ru_minflt);
    printf("Page faults (I/O):    %ld\n", usage.ru_majflt);
    printf("Voluntary context switches: %ld\n", usage.ru_nvcsw);
    printf("Involuntary context switches: %ld\n", usage.ru_nivcsw);
}

void show_sysinfo(void)
{
    struct sysinfo info;
    sysinfo(&info);

    printf("\n=== System Info ===\n");
    printf("Uptime: %ld days, %ld:%02ld:%02ld\n",
           info.uptime / 86400,
           (info.uptime % 86400) / 3600,
           (info.uptime % 3600) / 60,
           info.uptime % 60);
    printf("Total RAM: %lu MB\n",
           (unsigned long)(info.totalram * info.mem_unit / 1024 / 1024));
    printf("Free RAM:  %lu MB\n",
           (unsigned long)(info.freeram * info.mem_unit / 1024 / 1024));
    printf("Total swap: %lu MB\n",
           (unsigned long)(info.totalswap * info.mem_unit / 1024 / 1024));
    printf("Free swap:  %lu MB\n",
           (unsigned long)(info.freeswap * info.mem_unit / 1024 / 1024));
    printf("Processes: %u\n", info.procs);
}

void show_load_average(void)
{
    double load[3];
    getloadavg(load, 3);
    printf("Load average: %.2f, %.2f, %.2f\n",
           load[0], load[1], load[2]);
}
```
