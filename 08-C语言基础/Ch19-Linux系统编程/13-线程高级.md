# 线程高级主题

## 1. 线程池

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

/*
 * 线程池优势:
 * 1. 避免频繁创建/销毁线程的开销
 * 2. 控制并发线程数量
 * 3. 任务队列管理和负载均衡
 */

// 任务结构
typedef struct Task {
    void (*function)(void *);
    void *arg;
    struct Task *next;
} Task;

// 线程池结构
typedef struct {
    pthread_t *threads;
    int thread_count;
    Task *task_queue_head;
    Task *task_queue_tail;
    int queue_size;
    int shutdown;
    pthread_mutex_t lock;
    pthread_cond_t notify;
} ThreadPool;

// 工作线程函数
void *thread_pool_worker(void *arg)
{
    ThreadPool *pool = (ThreadPool *)arg;

    while (1) {
        pthread_mutex_lock(&pool->lock);

        // 等待任务或关闭信号
        while (pool->queue_size == 0 && !pool->shutdown) {
            pthread_cond_wait(&pool->notify, &pool->lock);
        }

        if (pool->shutdown && pool->queue_size == 0) {
            pthread_mutex_unlock(&pool->lock);
            break;
        }

        // 取出任务
        Task *task = pool->task_queue_head;
        if (task) {
            pool->task_queue_head = task->next;
            if (pool->task_queue_head == NULL) {
                pool->task_queue_tail = NULL;
            }
            pool->queue_size--;
        }

        pthread_mutex_unlock(&pool->lock);

        // 执行任务
        if (task) {
            task->function(task->arg);
            free(task);
        }
    }

    return NULL;
}

// 创建线程池
ThreadPool *thread_pool_create(int num_threads)
{
    ThreadPool *pool = calloc(1, sizeof(ThreadPool));
    if (!pool) return NULL;

    pool->threads = malloc(num_threads * sizeof(pthread_t));
    pool->thread_count = num_threads;
    pool->shutdown = 0;
    pool->queue_size = 0;
    pool->task_queue_head = NULL;
    pool->task_queue_tail = NULL;

    pthread_mutex_init(&pool->lock, NULL);
    pthread_cond_init(&pool->notify, NULL);

    for (int i = 0; i < num_threads; i++) {
        pthread_create(&pool->threads[i], NULL, thread_pool_worker, pool);
    }

    return pool;
}

// 添加任务
int thread_pool_add(ThreadPool *pool, void (*func)(void *), void *arg)
{
    Task *task = malloc(sizeof(Task));
    if (!task) return -1;

    task->function = func;
    task->arg = arg;
    task->next = NULL;

    pthread_mutex_lock(&pool->lock);

    if (pool->task_queue_tail) {
        pool->task_queue_tail->next = task;
    } else {
        pool->task_queue_head = task;
    }
    pool->task_queue_tail = task;
    pool->queue_size++;

    pthread_cond_signal(&pool->notify);
    pthread_mutex_unlock(&pool->lock);

    return 0;
}

// 销毁线程池
void thread_pool_destroy(ThreadPool *pool)
{
    pthread_mutex_lock(&pool->lock);
    pool->shutdown = 1;
    pthread_mutex_unlock(&pool->lock);

    // 唤醒所有线程
    pthread_cond_broadcast(&pool->notify);

    for (int i = 0; i < pool->thread_count; i++) {
        pthread_join(pool->threads[i], NULL);
    }

    free(pool->threads);

    // 清理剩余任务
    Task *task = pool->task_queue_head;
    while (task) {
        Task *next = task->next;
        free(task);
        task = next;
    }

    pthread_mutex_destroy(&pool->lock);
    pthread_cond_destroy(&pool->notify);
    free(pool);
}

// 使用示例
void my_task(void *arg)
{
    int id = *(int *)arg;
    printf("Task %d running in thread %lu\n",
           id, (unsigned long)pthread_self());
    usleep(100000);
}

void thread_pool_demo(void)
{
    ThreadPool *pool = thread_pool_create(4);

    int task_ids[20];
    for (int i = 0; i < 20; i++) {
        task_ids[i] = i;
        thread_pool_add(pool, my_task, &task_ids[i]);
    }

    sleep(2);
    thread_pool_destroy(pool);
    printf("All tasks completed\n");
}
```

## 2. 线程特定数据（TSD / TLS）

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/*
 * 线程特定数据 (Thread-Specific Data):
 *   每个线程拥有该数据的独立副本
 *
 * int pthread_key_create(pthread_key_t *key,
 *                        void (*destructor)(void*));
 * int pthread_key_delete(pthread_key_t key);
 * void *pthread_getspecific(pthread_key_t key);
 * int pthread_setspecific(pthread_key_t key, const void *value);
 *
 * 或使用 __thread 关键字（GCC 扩展，更高效）
 */

// 方法1：使用 pthread_key
static pthread_key_t log_key;
static pthread_once_t log_once = PTHREAD_ONCE_INIT;

typedef struct {
    char prefix[32];
    FILE *stream;
} ThreadLog;

static void log_destructor(void *data)
{
    ThreadLog *log = (ThreadLog *)data;
    if (log->stream) fclose(log->stream);
    free(log);
}

static void log_key_init(void)
{
    pthread_key_create(&log_key, log_destructor);
}

// 为每个线程获取独立的日志上下文
ThreadLog *get_thread_log(void)
{
    pthread_once(&log_once, log_key_init);

    ThreadLog *log = pthread_getspecific(log_key);
    if (!log) {
        log = calloc(1, sizeof(ThreadLog));
        snprintf(log->prefix, sizeof(log->prefix),
                 "[Thread %lu]", (unsigned long)pthread_self());
        log->stream = stderr;
        pthread_setspecific(log_key, log);
    }
    return log;
}

void thread_log(const char *fmt, ...)
{
    ThreadLog *log = get_thread_log();
    fprintf(log->stream, "%s ", log->prefix);
    va_list args;
    va_start(args, fmt);
    vfprintf(log->stream, fmt, args);
    va_end(args);
    fprintf(log->stream, "\n");
}

void *tsd_worker(void *arg)
{
    (void)arg;
    ThreadLog *log = get_thread_log();
    snprintf(log->prefix, sizeof(log->prefix),
             "[Worker %lu]", (unsigned long)pthread_self());

    thread_log("Starting work");
    sleep(1);
    thread_log("Work completed");

    return NULL;
}

void tsd_example(void)
{
    pthread_t t1, t2;
    pthread_create(&t1, NULL, tsd_worker, NULL);
    pthread_create(&t2, NULL, tsd_worker, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
}

// 方法2：使用 __thread 关键字（推荐）
__thread int thread_local_counter = 0;

void *tls_worker(void *arg)
{
    int count = *(int *)arg;
    for (int i = 0; i < count; i++) {
        thread_local_counter++;
    }
    printf("Thread %lu: counter = %d\n",
           (unsigned long)pthread_self(), thread_local_counter);
    return NULL;
}

void tls_example(void)
{
    pthread_t t1, t2;
    int n = 100;
    pthread_create(&t1, NULL, tls_worker, &n);
    pthread_create(&t2, NULL, tls_worker, &n);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    // 每个线程的 counter 都是 100
}
```

## 3. 线程取消

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

/*
 * int pthread_cancel(pthread_t thread);
 * int pthread_setcancelstate(int state, int *oldstate);
 *   PTHREAD_CANCEL_ENABLE  - 允许取消（默认）
 *   PTHREAD_CANCEL_DISABLE - 禁止取消
 * int pthread_setcanceltype(int type, int *oldtype);
 *   PTHREAD_CANCEL_DEFERRED  - 延迟取消（在取消点取消，默认）
 *   PTHREAD_CANCEL_ASYNCHRONOUS - 异步取消（随时取消）
 * void pthread_testcancel(void);  - 手动取消点
 *
 * 取消点函数:
 *   pthread_join, pthread_cond_wait, pthread_testcancel
 *   read, write, open, close
 *   sleep, nanosleep, select, poll
 *   printf, malloc, free
 */

void *cancellable_worker(void *arg)
{
    (void)arg;
    int old_state, old_type;

    printf("Worker started\n");

    // 禁止取消
    pthread_setcancelstate(PTHREAD_CANCEL_DISABLE, &old_state);
    // ... 不可取消的初始化工作 ...
    printf("Doing non-cancellable setup...\n");
    sleep(1);

    // 重新允许取消
    pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, &old_state);

    // 使用延迟取消（在取消点才会被取消）
    pthread_setcanceltype(PTHREAD_CANCEL_DEFERRED, &old_type);

    while (1) {
        printf("Worker running...\n");

        // sleep 是一个取消点
        sleep(1);

        // 也可以手动设置取消点
        // pthread_testcancel();
    }

    return NULL;
}

// 使用清理处理函数
void cleanup_handler(void *arg)
{
    printf("Cleanup: %s\n", (char *)arg);
}

void *cancellable_with_cleanup(void *arg)
{
    (void)arg;

    pthread_cleanup_push(cleanup_handler, "first");
    pthread_cleanup_push(cleanup_handler, "second");

    while (1) {
        sleep(1);
        // 注意：pthread_cleanup_push/pop 必须成对出现
        // 不能在中间使用 break/return/goto
    }

    pthread_cleanup_pop(0);  // 参数 0 = 不执行
    pthread_cleanup_pop(0);

    return NULL;
}

void cancel_example(void)
{
    pthread_t tid;
    void *retval;

    pthread_create(&tid, NULL, cancellable_worker, NULL);

    sleep(3);
    printf("Cancelling worker...\n");
    pthread_cancel(tid);

    pthread_join(tid, &retval);
    if (retval == PTHREAD_CANCELED) {
        printf("Worker was cancelled\n");
    }
}
```

## 4. 自旋锁

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

/*
 * int pthread_spin_init(pthread_spinlock_t *lock, int pshared);
 * int pthread_spin_destroy(pthread_spinlock_t *lock);
 * int pthread_spin_lock(pthread_spinlock_t *lock);
 * int pthread_spin_unlock(pthread_spinlock_t *lock);
 * int pthread_spin_trylock(pthread_spinlock_t *lock);
 *
 * pshared:
 *   PTHREAD_PROCESS_PRIVATE - 进程内使用（默认）
 *   PTHREAD_PROCESS_SHARED  - 进程间共享
 *
 * 自旋锁特点:
 *   - 忙等待（不会让出 CPU）
 *   - 适用于极短的临界区
 *   - 避免了上下文切换开销
 *   - 不要在单核系统上使用
 */

static pthread_spinlock_t spinlock;
static int spin_counter = 0;

void *spin_worker(void *arg)
{
    int n = *(int *)arg;
    for (int i = 0; i < n; i++) {
        pthread_spin_lock(&spinlock);
        spin_counter++;
        pthread_spin_unlock(&spinlock);
    }
    return NULL;
}

void spinlock_example(void)
{
    pthread_t t1, t2;
    int count = 100000;

    pthread_spin_init(&spinlock, PTHREAD_PROCESS_PRIVATE);

    pthread_create(&t1, NULL, spin_worker, &count);
    pthread_create(&t2, NULL, spin_worker, &count);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    printf("Spin counter: %d (expected %d)\n",
           spin_counter, count * 2);

    pthread_spin_destroy(&spinlock);
}
```
