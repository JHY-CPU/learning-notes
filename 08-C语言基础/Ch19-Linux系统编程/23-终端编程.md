# 终端编程

## 1. termios 概述

```c
#include <termios.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

/*
 * struct termios {
 *     tcflag_t c_iflag;  - 输入模式
 *     tcflag_t c_oflag;  - 输出模式
 *     tcflag_t c_cflag;  - 控制模式
 *     tcflag_t c_lflag;  - 本地模式
 *     cc_t     c_cc[NCCS]; - 特殊字符
 *     speed_t  c_ispeed; - 输入波特率
 *     speed_t  c_ospeed; - 输出波特率
 * };
 *
 * int tcgetattr(int fd, struct termios *termios_p);
 * int tcsetattr(int fd, int optional_actions,
 *               const struct termios *termios_p);
 * int cfsetispeed(struct termios *termios_p, speed_t speed);
 * int cfsetospeed(struct termios *termios_p, speed_t speed);
 */

void show_termios_info(void)
{
    struct termios t;

    if (tcgetattr(STDIN_FILENO, &t) == -1) {
        perror("tcgetattr");
        return;
    }

    printf("Input flags:  0x%08x\n", t.c_iflag);
    printf("Output flags: 0x%08x\n", t.c_oflag);
    printf("Control flags:0x%08x\n", t.c_cflag);
    printf("Local flags:  0x%08x\n", t.c_lflag);

    printf("\nSpecial characters:\n");
    printf("  VINTR (Ctrl+C): %d\n", t.c_cc[VINTR]);
    printf("  VQUIT (Ctrl+\\): %d\n", t.c_cc[VQUIT]);
    printf("  VERASE (Backspace): %d\n", t.c_cc[VERASE]);
    printf("  VEOL: %d\n", t.c_cc[VEOL]);
    printf("  VTIME: %d\n", t.c_cc[VTIME]);
    printf("  VMIN: %d\n", t.c_cc[VMIN]);
}
```

## 2. 终端模式

```c
#include <termios.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

/*
 * 规范模式 (Canonical mode):
 *   - 按行处理，输入在回车后才可用
 *   - 支持行编辑（退格、Ctrl+U 等）
 *   - c_lflag 中的 ICANON 标志
 *
 * 非规范模式 (Non-canonical mode):
 *   - 字符可用立即返回
 *   - 通过 VMIN 和 VTIME 控制
 *   - 适用于实时交互程序
 */

// 保存和恢复终端设置
static struct termios orig_termios;

void save_terminal(void)
{
    tcgetattr(STDIN_FILENO, &orig_termios);
}

void restore_terminal(void)
{
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);
}

// 设置非规范模式
void set_raw_mode(void)
{
    struct termios raw;

    tcgetattr(STDIN_FILENO, &raw);

    // 关闭回显和规范模式
    raw.c_lflag &= ~(ECHO | ICANON);

    // 最小输入：1 字节，无超时
    raw.c_cc[VMIN]  = 1;
    raw.c_cc[VTIME] = 0;

    tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
}

// 读取单个按键
int read_key(void)
{
    char c;
    ssize_t n = read(STDIN_FILENO, &c, 1);
    if (n == 1) return c;
    return -1;
}

void raw_mode_demo(void)
{
    save_terminal();
    set_raw_mode();

    printf("Press keys (ESC to quit):\n");

    while (1) {
        int c = read_key();
        if (c == 27) {  // ESC
            break;
        }
        printf("Key: %d ('%c')\n", c, c >= 32 ? c : '?');
    }

    restore_terminal();
    printf("\nTerminal restored\n");
}
```

## 3. 密码输入

```c
#include <termios.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

// 关闭回显读取密码
void read_password(char *buf, size_t size)
{
    struct termios oldt, newt;

    printf("Password: ");
    fflush(stdout);

    // 获取当前设置
    tcgetattr(STDIN_FILENO, &oldt);
    newt = oldt;

    // 关闭回显
    newt.c_lflag &= ~ECHO;
    tcsetattr(STDIN_FILENO, TCSANOW, &newt);

    // 读取密码
    if (fgets(buf, size, stdin) != NULL) {
        size_t len = strlen(buf);
        if (len > 0 && buf[len - 1] == '\n') {
            buf[len - 1] = '\0';
        }
    }

    // 恢复设置
    tcsetattr(STDIN_FILENO, TCSANOW, &oldt);

    printf("\n");
}

void password_demo(void)
{
    char password[128];
    read_password(password, sizeof(password));
    printf("You entered: %s\n", password);
}
```

## 4. 终端控制

```c
#include <termios.h>
#include <sys/ioctl.h>
#include <unistd.h>
#include <stdio.h>

/*
 * int ioctl(int fd, unsigned long request, ...);
 *   TIOCGWINSZ - 获取窗口大小
 *   TIOCSWINSZ - 设置窗口大小
 *   TIOCSPGRP  - 设置前台进程组
 *   TIOCGPGRP  - 获取前台进程组
 */

void show_window_size(void)
{
    struct winsize ws;

    if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == -1) {
        perror("ioctl");
        return;
    }

    printf("Terminal: %d rows x %d columns\n", ws.ws_row, ws.ws_col);
    printf("Pixel size: %d x %d\n", ws.ws_xpixel, ws.ws_pixel);
}

// 终端控制序列
void terminal_control(void)
{
    // 清屏
    printf("\033[2J");

    // 移动光标到 (1, 1)
    printf("\033[1;1H");

    // 设置颜色
    printf("\033[31m");  // 红色
    printf("Red text");
    printf("\033[0m");   // 重置

    // 隐藏光标
    printf("\033[?25l");

    // 显示光标
    printf("\033[?25h");
}
```

## 5. 伪终端（PTY）

```c
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/*
 * 伪终端（Pseudo Terminal）:
 *   一对虚拟设备：主设备（PTY Master）和从设备（PTY Slave）
 *   从设备对用户程序表现为终端
 *   主设备用于控制程序读写
 *
 * 应用：ssh、script、expect、终端模拟器
 *
 * int posix_openpt(int flags);
 * int grantpt(int fd);
 * int unlockpt(int fd);
 * char *ptsname(int fd);
 */

int pty_fork_exec(const char *program)
{
    // 打开 PTY 主设备
    int master = posix_openpt(O_RDWR | O_NOCTTY);
    if (master < 0) { perror("posix_openpt"); return -1; }

    // 授予从设备访问权限
    if (grantpt(master) < 0) { perror("grantpt"); close(master); return -1; }

    // 解锁从设备
    if (unlockpt(master) < 0) { perror("unlockpt"); close(master); return -1; }

    // 获取从设备名称
    char *slave_name = ptsname(master);
    if (!slave_name) { perror("ptsname"); close(master); return -1; }

    printf("PTY: master=%d, slave=%s\n", master, slave_name);

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程：打开从设备作为终端
        close(master);

        int slave = open(slave_name, O_RDWR);
        if (slave < 0) { perror("open slave"); _exit(1); }

        // 建立新的会话
        setsid();

        // 设置控制终端
        ioctl(slave, TIOCSCTTY, 0);

        // 重定向标准 IO
        dup2(slave, STDIN_FILENO);
        dup2(slave, STDOUT_FILENO);
        dup2(slave, STDERR_FILENO);
        if (slave > 2) close(slave);

        // 执行程序
        execlp(program, program, NULL);
        perror("execlp");
        _exit(1);
    }

    return master;
}

// 简易终端模拟器
void simple_terminal(const char *program)
{
    int master = pty_fork_exec(program);
    if (master < 0) return;

    fd_set fds;
    char buf[4096];

    while (1) {
        FD_ZERO(&fds);
        FD_SET(STDIN_FILENO, &fds);
        FD_SET(master, &fds);
        int maxfd = (STDIN_FILENO > master) ? STDIN_FILENO : master;

        if (select(maxfd + 1, &fds, NULL, NULL, NULL) < 0) break;

        // 用户输入 -> PTY
        if (FD_ISSET(STDIN_FILENO, &fds)) {
            ssize_t n = read(STDIN_FILENO, buf, sizeof(buf));
            if (n <= 0) break;
            write(master, buf, n);
        }

        // PTY 输出 -> 屏幕
        if (FD_ISSET(master, &fds)) {
            ssize_t n = read(master, buf, sizeof(buf));
            if (n <= 0) break;
            write(STDOUT_FILENO, buf, n);
        }
    }

    close(master);
}
```
