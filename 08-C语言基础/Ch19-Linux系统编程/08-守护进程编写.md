# 守护进程编写

## 1. 完整的守护进程创建

```c
#include <unistd.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <stdlib.h>
#include <fcntl.h>
#include <signal.h>
#include <stdio.h>
#include <sys/resource.h>
#include <syslog.h>

/*
 * 创建守护进程的标准步骤（UNIX 编程经典方法）:
 * 1. 调用 umask(0)
 * 2. 调用 fork()，父进程退出
 * 3. 调用 setsid() 创建新会话
 * 4. 再次 fork()，父进程退出（防止获取控制终端）
 * 5. 改变工作目录到 /
 * 6. 关闭所有不需要的文件描述符
 * 7. 将 stdin/stdout/stderr 重定向到 /dev/null
 */

int daemonize(int nochdir, int noclose)
{
    pid_t pid;

    // 第一次 fork
    pid = fork();
    if (pid < 0) return -1;
    if (pid > 0) _exit(EXIT_SUCCESS);

    // 创建新会话
    if (setsid() < 0) return -1;

    // 第二次 fork
    pid = fork();
    if (pid < 0) return -1;
    if (pid > 0) _exit(EXIT_SUCCESS);

    // 清除文件创建掩码
    umask(0);

    // 改变工作目录
    if (!nochdir) {
        if (chdir("/") < 0) return -1;
    }

    // 关闭所有文件描述符
    if (!noclose) {
        struct rlimit rl;
        if (getrlimit(RLIMIT_NOFILE, &rl) < 0)
            return -1;

        int maxfd = (rl.rlim_max == RLIM_INFINITY) ? 1024 : rl.rlim_max;
        for (int fd = 0; fd < maxfd; fd++) {
            close(fd);
        }

        // 重定向标准IO到 /dev/null
        int nullfd = open("/dev/null", O_RDWR);
        if (nullfd != 0) return -1;
        if (dup2(nullfd, STDOUT_FILENO) < 0) return -1;
        if (dup2(nullfd, STDERR_FILENO) < 0) return -1;
        if (nullfd > 2) close(nullfd);
    }

    return 0;
}
```

## 2. 守护进程使用 daemon() 函数

```c
#include <unistd.h>
#include <stdio.h>

/*
 * int daemon(int nochdir, int noclose);  // BSD 风格，glibc 提供
 *
 * nochdir: 0 = chdir("/")，1 = 保持当前目录
 * noclose: 0 = 重定向到 /dev/null，1 = 不重定向
 *
 * 注意：这不是 POSIX 标准函数，但 Linux 支持
 */

void use_daemon_function(void)
{
    // 简单方式创建守护进程
    if (daemon(0, 0) == -1) {
        perror("daemon");
        exit(EXIT_FAILURE);
    }

    // 此时已经是守护进程
    // 主循环
    while (1) {
        sleep(10);
    }
}
```

## 3. 单实例守护进程（文件锁）

```c
#include <unistd.h>
#include <fcntl.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>

/*
 * 使用文件锁确保只有一个实例运行
 * 常见的锁文件目录: /var/run/<daemon>.pid
 */

static const char *LOCKFILE = "/var/run/mydaemon.pid";
static int lockfd = -1;

int acquire_lock(const char *path)
{
    int fd = open(path, O_WRONLY | O_CREAT, 0644);
    if (fd < 0) return -1;

    struct flock fl;
    fl.l_type   = F_WRLCK;
    fl.l_whence = SEEK_SET;
    fl.l_start  = 0;
    fl.l_len    = 0;

    if (fcntl(fd, F_SETLK, &fl) < 0) {
        if (errno == EACCES || errno == EAGAIN) {
            close(fd);
            return 1;  // 已有实例运行
        }
        close(fd);
        return -1;
    }

    // 截断并写入 PID
    ftruncate(fd, 0);
    char pid_str[32];
    snprintf(pid_str, sizeof(pid_str), "%ld\n", (long)getpid());
    write(fd, pid_str, strlen(pid_str));

    return fd;
}

int ensure_single_instance(void)
{
    lockfd = acquire_lock(LOCKFILE);
    if (lockfd == 1) {
        fprintf(stderr, "Daemon already running\n");
        return -1;
    }
    if (lockfd < 0) {
        perror("acquire_lock");
        return -1;
    }
    return 0;
}

// 程序退出时自动释放锁
void cleanup_lock(void)
{
    if (lockfd >= 0) {
        close(lockfd);  // 锁自动释放
        unlink(LOCKFILE);
    }
}
```

## 4. syslog 日志系统

```c
#include <syslog.h>
#include <stdio.h>

/*
 * void openlog(const char *ident, int option, int facility);
 * void syslog(int priority, const char *format, ...);
 * void closelog(void);
 * int setlogmask(int mask);
 *
 * 优先级（从高到低）:
 *   LOG_EMERG   - 系统不可用
 *   LOG_ALERT   - 必须立即行动
 *   LOG_CRIT    - 临界条件
 *   LOG_ERR     - 错误条件
 *   LOG_WARNING - 警告条件
 *   LOG_NOTICE  - 正常但重要的条件
 *   LOG_INFO    - 信息性消息
 *   LOG_DEBUG   - 调试信息
 *
 * 设施（facility）:
 *   LOG_DAEMON - 守护进程（默认）
 *   LOG_USER   - 用户级消息
 *   LOG_LOCAL0-7 - 本地使用
 *
 * option:
 *   LOG_PID    - 每条消息包含 PID
 *   LOG_CONS   - 发送到控制台
 *   LOG_NDELAY - 立即打开连接
 */

void syslog_examples(void)
{
    // 打开 syslog 连接
    openlog("mydaemon", LOG_PID | LOG_CONS, LOG_DAEMON);

    // 写入不同级别的日志
    syslog(LOG_INFO, "Daemon started, PID=%d", getpid());
    syslog(LOG_WARNING, "High memory usage: %d%%", 85);
    syslog(LOG_ERR, "Failed to open config file: %s", strerror(errno));
    syslog(LOG_DEBUG, "Processing request %d", request_id);

    // 只记录 INFO 及以上级别的消息
    setlogmask(LOG_UPTO(LOG_INFO));
    syslog(LOG_DEBUG, "This will NOT be logged");

    // 关闭 syslog
    closelog();
}

// 封装日志函数
void log_msg(int priority, const char *fmt, ...)
{
    va_list args;
    va_start(args, fmt);
    vsyslog(priority, fmt, args);
    va_end(args);
}

// 使用示例
void log_examples(void)
{
    openlog("myapp", LOG_PID, LOG_DAEMON);

    log_msg(LOG_INFO, "Application started");
    log_msg(LOG_ERR, "Connection failed: %s", "timeout");
    log_msg(LOG_DEBUG, "Variable x = %d", 42);

    closelog();
}
```

## 5. 重读配置文件

```c
#include <signal.h>
#include <stdio.h>

/*
 * 守护进程通常通过 SIGHUP 信号重新读取配置文件
 * 使用方法: kill -HUP <daemon_pid>
 */

static volatile sig_atomic_t reload_config = 0;

void sighup_handler(int sig)
{
    (void)sig;
    reload_config = 1;
}

void setup_reload_handler(void)
{
    struct sigaction sa;
    sa.sa_handler = sighup_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = 0;
    sigaction(SIGHUP, &sa, NULL);
}

void daemon_main_loop(void)
{
    setup_reload_handler();

    while (1) {
        if (reload_config) {
            reload_config = 0;
            syslog(LOG_INFO, "Reloading configuration...");

            // 重新读取配置

            // reinit_connections();

            syslog(LOG_INFO, "Configuration reloaded");
        }

        // 正常的工作逻辑
        // process_requests();
        sleep(1);
    }
}
```

## 6. 信号处理与优雅退出

```c
#include <signal.h>
#include <unistd.h>
#include <stdio.h>

static volatile sig_atomic_t running = 1;

void signal_handler(int sig)
{
    if (sig == SIGTERM || sig == SIGINT) {
        running = 0;
    }
}

void setup_signal_handlers(void)
{
    struct sigaction sa;
    sa.sa_flags = 0;
    sigemptyset(&sa.sa_mask);

    // 终止信号
    sa.sa_handler = signal_handler;
    sigaction(SIGTERM, &sa, NULL);
    sigaction(SIGINT, &sa, NULL);

    // 忽略管道信号
    sa.sa_handler = SIG_IGN;
    sigaction(SIGPIPE, &sa, NULL);

    // 忽略子进程信号（避免僵尸）
    sa.sa_handler = SIG_IGN;
    sigaction(SIGCHLD, &sa, NULL);
}

void daemon_loop(void)
{
    setup_signal_handlers();

    openlog("mydaemon", LOG_PID, LOG_DAEMON);
    syslog(LOG_INFO, "Daemon starting");

    while (running) {
        // 主工作循环
        syslog(LOG_DEBUG, "Processing...");
        sleep(5);
    }

    // 清理资源
    syslog(LOG_INFO, "Daemon shutting down");
    // cleanup_resources();
    closelog();
}
```

## 7. 完整守护进程示例

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <signal.h>
#include <sys/stat.h>
#include <syslog.h>
#include <string.h>
#include <errno.h>

/*
 * 完整的守护进程模板
 * 功能：监视指定目录，记录文件变化
 */

#define PIDFILE "/var/run/dirmon.pid"
#define WATCHDIR "/tmp/watch"

static volatile sig_atomic_t keep_running = 1;

static void handle_signal(int sig)
{
    if (sig == SIGTERM || sig == SIGINT) keep_running = 0;
    if (sig == SIGHUP) {
        syslog(LOG_INFO, "Received SIGHUP, would reload config");
    }
}

static int write_pidfile(void)
{
    int fd = open(PIDFILE, O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd < 0) return -1;

    char buf[32];
    snprintf(buf, sizeof(buf), "%ld\n", (long)getpid());
    write(fd, buf, strlen(buf));
    close(fd);
    return 0;
}

static void remove_pidfile(void)
{
    unlink(PIDFILE);
}

int main(int argc, char *argv[])
{
    // 解析参数
    int foreground = 0;
    if (argc > 1 && strcmp(argv[1], "-f") == 0) {
        foreground = 1;
    }

    // 创建守护进程
    if (!foreground) {
        if (daemon(0, 0) < 0) {
            perror("daemon");
            exit(EXIT_FAILURE);
        }
    }

    // 设置信号处理
    struct sigaction sa = {0};
    sa.sa_handler = handle_signal;
    sigaction(SIGTERM, &sa, NULL);
    sigaction(SIGINT, &sa, NULL);
    sigaction(SIGHUP, &sa, NULL);
    signal(SIGCHLD, SIG_IGN);
    signal(SIGPIPE, SIG_IGN);

    // 写入 PID 文件
    if (write_pidfile() < 0) {
        syslog(LOG_ERR, "Cannot write PID file");
        exit(EXIT_FAILURE);
    }

    // 打开日志
    openlog("dirmon", LOG_PID, LOG_DAEMON);
    syslog(LOG_INFO, "Daemon started (PID=%d)", getpid());

    // 主循环
    while (keep_running) {
        // 检查监视目录是否存在
        struct stat st;
        if (stat(WATCHDIR, &st) < 0) {
            syslog(LOG_WARNING, "Watch directory not found: %s", WATCHDIR);
            sleep(10);
            continue;
        }

        syslog(LOG_DEBUG, "Watching %s", WATCHDIR);
        sleep(30);
    }

    // 清理
    syslog(LOG_INFO, "Daemon shutting down");
    remove_pidfile();
    closelog();

    return EXIT_SUCCESS;
}
```
