# 进程间通信 - 消息队列

## 1. System V 消息队列

```c
#include <sys/ipc.h>
#include <sys/msg.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>

/*
 * System V 消息队列 API:
 *   int msgget(key_t key, int msgflg);          - 创建/获取队列
 *   int msgsnd(int msqid, const void *msgp, size_t msgsz, int msgflg);  - 发送
 *   ssize_t msgrcv(int msqid, void *msgp, size_t msgsz, long msgtyp, int msgflg);  - 接收
 *   int msgctl(int msqid, int cmd, struct msqid_ds *buf);  - 控制
 *
 * 消息结构:
 *   struct msgbuf {
 *       long mtype;      - 消息类型（>0）
 *       char mtext[1];   - 消息内容（可变长度）
 *   };
 *
 * msgflg:
 *   IPC_CREAT - 不存在则创建
 *   IPC_EXCL  - 已存在则报错
 *   IPC_NOWAIT - 非阻塞
 */

// 定义消息结构
#define MAX_TEXT 256

typedef struct {
    long mtype;       // 消息类型
    char mtext[MAX_TEXT];  // 消息内容
} Message;

// 创建消息队列
int create_msg_queue(key_t key)
{
    int msqid = msgget(key, IPC_CREAT | 0666);
    if (msqid == -1) {
        perror("msgget");
    }
    return msqid;
}

// 发送消息
int send_message(int msqid, long type, const char *text)
{
    Message msg;
    msg.mtype = type;
    strncpy(msg.mtext, text, MAX_TEXT - 1);
    msg.mtext[MAX_TEXT - 1] = '\0';

    if (msgsnd(msqid, &msg, strlen(msg.mtext) + 1, 0) == -1) {
        perror("msgsnd");
        return -1;
    }

    printf("Sent [type=%ld]: %s\n", type, text);
    return 0;
}

// 接收消息
int receive_message(int msqid, long type, char *buf, size_t bufsize)
{
    Message msg;

    ssize_t n = msgrcv(msqid, &msg, MAX_TEXT, type, 0);
    if (n == -1) {
        perror("msgrcv");
        return -1;
    }

    strncpy(buf, msg.mtext, bufsize - 1);
    buf[bufsize - 1] = '\0';
    printf("Received [type=%ld]: %s\n", msg.mtype, buf);
    return 0;
}

// 删除消息队列
int remove_msg_queue(int msqid)
{
    if (msgctl(msqid, IPC_RMID, NULL) == -1) {
        perror("msgctl IPC_RMID");
        return -1;
    }
    return 0;
}

// 获取队列信息
void show_queue_info(int msqid)
{
    struct msqid_ds buf;
    if (msgctl(msqid, IPC_STAT, &buf) == -1) {
        perror("msgctl IPC_STAT");
        return;
    }

    printf("Queue info:\n");
    printf("  Messages: %lu\n", (unsigned long)buf.msg_qnum);
    printf("  Bytes: %lu\n", (unsigned long)buf.msg_cbytes);
    printf("  Max bytes: %lu\n", (unsigned long)buf.msg_qbytes);
    printf("  Last send PID: %d\n", buf.msg_lspid);
    printf("  Last recv PID: %d\n", buf.msg_lrpid);
}

void msg_queue_example(void)
{
    key_t key = ftok("/tmp", 'M');
    if (key == -1) { perror("ftok"); return; }

    int msqid = create_msg_queue(key);
    if (msqid == -1) return;

    // 发送消息
    send_message(msqid, 1, "Hello");
    send_message(msqid, 2, "World");
    send_message(msqid, 1, "Again");

    show_queue_info(msqid);

    // 按类型接收（类型 1）
    char buf[MAX_TEXT];
    receive_message(msqid, 1, buf, sizeof(buf));
    receive_message(msqid, 1, buf, sizeof(buf));

    // 删除队列
    remove_msg_queue(msqid);
}
```

## 2. 消息队列：客户端-服务器模型

```c
#include <sys/ipc.h>
#include <sys/msg.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * 使用消息队列实现客户端-服务器通信
 * 服务器消息类型 = 客户端 PID
 * 客户端发送请求（type=1），服务器回复（type=client_pid）
 */

#define SERVER_TYPE 1
#define MAX_TEXT 256

typedef struct {
    long mtype;
    pid_t client_pid;
    char mtext[MAX_TEXT];
} RequestMsg;

typedef struct {
    long mtype;
    char mtext[MAX_TEXT];
} ResponseMsg;

// 服务器端
void server_process(int msqid)
{
    RequestMsg req;
    ResponseMsg resp;

    printf("Server: waiting for requests...\n");

    while (1) {
        // 接收客户端请求（type=SERVER_TYPE）
        ssize_t n = msgrcv(msqid, &req, sizeof(req) - sizeof(long),
                           SERVER_TYPE, 0);
        if (n == -1) {
            perror("msgrcv");
            break;
        }

        printf("Server: request from PID %d: %s\n",
               req.client_pid, req.mtext);

        // 检查退出命令
        if (strcmp(req.mtext, "quit") == 0) {
            printf("Server: shutting down\n");
            break;
        }

        // 处理请求（这里简单地将字符串转为大写）
        resp.mtype = req.client_pid;
        for (int i = 0; req.mtext[i]; i++) {
            resp.mtext[i] = (req.mtext[i] >= 'a' && req.mtext[i] <= 'z')
                            ? req.mtext[i] - 32 : req.mtext[i];
        }
        resp.mtext[strlen(req.mtext)] = '\0';

        // 回复客户端
        if (msgsnd(msqid, &resp, strlen(resp.mtext) + 1, 0) == -1) {
            perror("msgsnd reply");
        }
    }
}

// 客户端端
void client_process(int msqid, const char *message)
{
    RequestMsg req;
    ResponseMsg resp;

    req.mtype = SERVER_TYPE;
    req.client_pid = getpid();
    strncpy(req.mtext, message, MAX_TEXT - 1);

    // 发送请求
    if (msgsnd(msqid, &req, sizeof(req) - sizeof(long), 0) == -1) {
        perror("msgsnd request");
        return;
    }

    // 接收回复（type = 自己的 PID）
    ssize_t n = msgrcv(msqid, &resp, MAX_TEXT, getpid(), 0);
    if (n > 0) {
        printf("Client %d: server replied: %s\n", getpid(), resp.mtext);
    }
}

void client_server_example(void)
{
    key_t key = ftok("/tmp", 'S');
    int msqid = msgget(key, IPC_CREAT | 0666);

    pid_t pid = fork();
    if (pid == 0) {
        // 客户端
        sleep(1);  // 等待服务器启动
        client_process(msqid, "hello server");
        client_process(msqid, "test message");
        _exit(0);
    } else {
        // 服务器
        server_process(msqid);
        wait(NULL);
        msgctl(msqid, IPC_RMID, NULL);
    }
}
```

## 3. POSIX 消息队列

```c
#include <mqueue.h>
#include <stdio.h>
#include <string.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <unistd.h>

/*
 * POSIX 消息队列 API:
 *   mqd_t mq_open(const char *name, int oflag, ...);
 *   int mq_close(mqd_t mqdes);
 *   int mq_unlink(const char *name);
 *   int mq_send(mqd_t mqdes, const char *msg_ptr,
 *               size_t msg_len, unsigned msg_prio);
 *   ssize_t mq_receive(mqd_t mqdes, char *msg_ptr,
 *                      size_t msg_len, unsigned *msg_prio);
 *   int mq_getattr(mqd_t mqdes, struct mq_attr *attr);
 *   int mq_setattr(mqd_t mqdes, const struct mq_attr *newattr,
 *                  struct mq_attr *oldattr);
 *   int mq_notify(mqd_t mqdes, const struct sigevent *notification);
 *
 * 队列名称必须以 '/' 开头
 */

// 创建 POSIX 消息队列
mqd_t create_posix_queue(const char *name, int maxmsg, int msgsize)
{
    struct mq_attr attr;
    attr.mq_flags   = 0;
    attr.mq_maxmsg  = maxmsg;     // 最大消息数
    attr.mq_msgsize = msgsize;    // 每条消息最大字节数
    attr.mq_curmsgs = 0;

    mqd_t mq = mq_open(name, O_CREAT | O_RDWR, 0666, &attr);
    if (mq == (mqd_t)-1) {
        perror("mq_open");
    }
    return mq;
}

void posix_mq_example(void)
{
    const char *queue_name = "/my_test_queue";

    // 创建队列
    mqd_t mq = create_posix_queue(queue_name, 10, 256);
    if (mq == (mqd_t)-1) return;

    // 发送消息（带优先级）
    const char *messages[] = {
        "Low priority message",
        "High priority message",
        "Normal message",
        NULL
    };
    unsigned int priorities[] = {1, 10, 5};

    for (int i = 0; messages[i]; i++) {
        if (mq_send(mq, messages[i], strlen(messages[i]) + 1,
                    priorities[i]) == -1) {
            perror("mq_send");
        }
        printf("Sent (priority=%u): %s\n", priorities[i], messages[i]);
    }

    // 接收消息（按优先级排序）
    char buf[256];
    unsigned int prio;

    struct mq_attr attr;
    mq_getattr(mq, &attr);

    printf("\nQueue has %ld messages\n", attr.mq_curmsgs);

    while (attr.mq_curmsgs > 0) {
        ssize_t n = mq_receive(mq, buf, sizeof(buf), &prio);
        if (n > 0) {
            printf("Received (priority=%u): %s\n", prio, buf);
        }
        mq_getattr(mq, &attr);
    }

    // 清理
    mq_close(mq);
    mq_unlink(queue_name);
}

// 异步通知
#include <signal.h>

static volatile sig_atomic_t msg_ready = 0;

void mq_notify_handler(int sig)
{
    (void)sig;
    msg_ready = 1;
}

void posix_mq_notify_example(void)
{
    const char *queue_name = "/notify_queue";

    mqd_t mq = create_posix_queue(queue_name, 10, 256);

    // 设置通知
    struct sigevent sev;
    sev.sigev_notify = SIGEV_SIGNAL;
    sev.sigev_signo = SIGUSR1;
    signal(SIGUSR1, mq_notify_handler);

    mq_notify(mq, &sev);

    printf("Waiting for message notification...\n");

    while (!msg_ready) {
        sleep(1);
    }

    if (msg_ready) {
        char buf[256];
        unsigned int prio;
        ssize_t n = mq_receive(mq, buf, sizeof(buf), &prio);
        if (n > 0) {
            printf("Got message: %s\n", buf);
        }
    }

    mq_close(mq);
    mq_unlink(queue_name);
}
```
