# 信号深入

## 1. 可靠信号与不可靠信号

```c
#include <signal.h>
#include <stdio.h>

/*
 * 不可靠信号 (1-31):
 *   - 可能丢失：多个相同信号合并为一个
 *   - 执行期间自动恢复默认行为
 *   - 不能阻塞，只有 pending 状态
 *
 * 可靠信号 (SIGRTMIN - SIGRTMAX, 34-64):
 *   - 支持排队：相同信号不会丢失
 *   - 可以携带额外数据 (sigqueue)
 *   - 执行期间自动阻塞（SA_NODEFER 可覆盖）
 *
 * Linux 中 SIGRTMIN=34, SIGRTMAX=64
 * 实际可用的实时信号：32个（SIGRTMIN 到 SIGRTMAX）
 */

void show_realtime_signals(void)
{
    printf("SIGRTMIN = %d\n", SIGRTMIN);
    printf("SIGRTMAX = %d\n", SIGRTMAX);
    printf("Available RT signals: %d\n", SIGRTMAX - SIGRTMIN + 1);

    // 信号编号 1-31 是不可靠信号
    // 信号编号 34-64 是可靠信号（实时信号）
    // 信号编号 32, 33 保留给 NPTL 线程库

    // 实时信号的特点：
    // 1. 多个相同信号排队（不丢失）
    // 2. 传递顺序保证：编号小的先递送
    // 3. 可以附带数据
}
```

## 2. sigqueue - 发送带数据的信号

```c
#include <signal.h>
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>

/*
 * int sigqueue(pid_t pid, int sig, const union sigval value);
 *
 * union sigval {
 *     int   sival_int;     - 整数值
 *     void *sival_ptr;     - 指针值
 * };
 *
 * 发送信号并附带一个整数或指针数据
 * 接收方需要使用 SA_SIGINFO 标志的 sigaction
 */

// 发送端
void send_sigqueue(pid_t target_pid)
{
    union sigval value;
    value.sival_int = 42;

    if (sigqueue(target_pid, SIGUSR1, value) == -1) {
        perror("sigqueue");
    }

    // 发送实时信号
    value.sival_int = 100;
    if (sigqueue(target_pid, SIGRTMIN, value) == -1) {
        perror("sigqueue RT");
    }
}

// 接收端
void siginfo_handler(int sig, siginfo_t *info, void *ucontext)
{
    (void)ucontext;

    if (sig == SIGUSR1 || sig == SIGRTMIN) {
        printf("Signal %d from PID %d, value: %d, code: %d\n",
               sig, info->si_pid, info->si_value.sival_int,
               info->si_code);
    }
}

void setup_sigqueue_receiver(void)
{
    struct sigaction sa;
    sa.sa_sigaction = siginfo_handler;
    sa.sa_flags = SA_SIGINFO | SA_RESTART;
    sigemptyset(&sa.sa_mask);

    sigaction(SIGUSR1, &sa, NULL);
    sigaction(SIGRTMIN, &sa, NULL);

    printf("Receiver PID: %d\n", getpid());
}
```

## 3. sigsetjmp / siglongjmp

```c
#include <setjmp.h>
#include <signal.h>
#include <stdio.h>
#include <unistd.h>

/*
 * int sigsetjmp(sigjmp_buf env, int savesigs);
 * void siglongjmp(sigjmp_buf env, int val);
 *
 * 与 setjmp/longjmp 的区别：
 * sigsetjmp 可以选择是否保存信号掩码
 * savesigs 非零时保存当前信号掩码
 *
 * 搭配 siglongjmp 使用可正确恢复信号状态
 */

static sigjmp_buf jmpbuf;

void handler(int sig)
{
    (void)sig;
    printf("In signal handler, jumping back...\n");
    siglongjmp(jmpbuf, 1);
}

void sigsetjmp_example(void)
{
    signal(SIGINT, handler);

    // savesigs=1 保存当前信号掩码
    if (sigsetjmp(jmpbuf, 1) == 0) {
        printf("Initial call, waiting for SIGINT...\n");
        pause();
    } else {
        printf("Returned from siglongjmp\n");
        printf("Signal mask has been restored\n");
    }
}

// 实际应用：从信号处理中恢复
static sigjmp_buf recover_jmp;

void segv_handler(int sig)
{
    (void)sig;
    // 从段错误中恢复
    siglongjmp(recover_jmp, 1);
}

void safe_memory_access(void *addr)
{
    signal(SIGSEGV, segv_handler);

    if (sigsetjmp(recover_jmp, 1) == 0) {
        // 尝试访问内存
        int *p = (int *)addr;
        printf("Value at %p: %d\n", addr, *p);
    } else {
        printf("Caught SIGSEGV - bad memory access at %p\n", addr);
    }

    signal(SIGSEGV, SIG_DFL);  // 恢复默认处理
}
```

## 4. sigwait / sigwaitinfo

```c
#include <signal.h>
#include <stdio.h>
#include <unistd.h>
#include <pthread.h>

/*
 * int sigwait(const sigset_t *set, int *sig);
 * int sigwaitinfo(const sigset_t *set, siginfo_t *info);
 * int sigtimedwait(const sigset_t *set, siginfo_t *info,
 *                  const struct timespec *timeout);
 *
 * 同步等待信号：将信号从"异步"变为"同步"处理
 * 需要先阻塞要等待的信号
 *
 * 优势：
 * 1. 避免信号处理函数的复杂性
 * 2. 代码流程更清晰
 * 3. 可以在普通函数上下文中处理信号
 */

void sigwait_example(void)
{
    sigset_t set;
    int sig;

    // 阻塞 SIGINT 和 SIGTERM
    sigemptyset(&set);
    sigaddset(&set, SIGINT);
    sigaddset(&set, SIGTERM);
    sigprocmask(SIG_BLOCK, &set, NULL);

    printf("Waiting for SIGINT or SIGTERM...\n");

    // 同步等待信号
    while (1) {
        int ret = sigwait(&set, &sig);
        if (ret != 0) {
            perror("sigwait");
            break;
        }

        printf("Received signal %d\n", sig);

        if (sig == SIGTERM) {
            printf("Exiting...\n");
            break;
        }
        if (sig == SIGINT) {
            printf("SIGINT received, ignoring\n");
        }
    }
}

// sigwaitinfo 获取详细信息
void sigwaitinfo_example(void)
{
    sigset_t set;
    siginfo_t info;

    sigemptyset(&set);
    sigaddset(&set, SIGUSR1);
    sigprocmask(SIG_BLOCK, &set, NULL);

    printf("Waiting for SIGUSR1...\n");

    if (sigwaitinfo(&set, &info) > 0) {
        printf("Signal: %d\n", info.si_signo);
        printf("Sender PID: %d\n", info.si_pid);
        printf("Sender UID: %d\n", info.si_uid);
        printf("Data: %d\n", info.si_value.sival_int);
    }
}

// sigtimedwait 带超时等待
void sigtimedwait_example(void)
{
    sigset_t set;
    siginfo_t info;
    struct timespec timeout = {5, 0};  // 5 秒超时

    sigemptyset(&set);
    sigaddset(&set, SIGUSR1);
    sigprocmask(SIG_BLOCK, &set, NULL);

    printf("Waiting for SIGUSR1 (5s timeout)...\n");

    int ret = sigtimedwait(&set, &info, &timeout);
    if (ret > 0) {
        printf("Received signal %d from PID %d\n",
               info.si_signo, info.si_pid);
    } else {
        printf("Timeout - no signal received\n");
    }
}
```

## 5. abort 函数分析

```c
#include <stdlib.h>
#include <signal.h>
#include <stdio.h>
#include <unistd.h>

/*
 * void abort(void);
 *
 * abort 的行为:
 * 1. 尝试刷新 stdio 流
 * 2. 发送 SIGABRT 给自己
 * 3. 如果 SIGABRT 被捕获且处理函数返回：
 *    - abort 重新设置处理函数为 SIG_DFL
 *    - 再次发送 SIGABRT
 * 4. 即使 SIGABRT 被忽略，abort 仍会终止进程
 */

// 分析 abort 的行为
void abort_analysis(void)
{
    // abort() 的等价实现（简化版）：
    // 1. raise(SIGABRT)
    // 2. 如果 SIGABRT 被捕获且处理函数返回，
    //    重新 raise(SIGABRT) 并确保终止

    // 自定义 SIGABRT 处理
    void abrt_handler(int sig) {
        (void)sig;
        write(STDERR_FILENO, "SIGABRT caught, but will still die\n", 36);
        // 即使从这里返回，abort 会再次发送 SIGABRT
    }

    signal(SIGABRT, abrt_handler);
    abort();  // 进程一定会终止
}
```

## 6. 信号与系统调用

```c
#include <signal.h>
#include <unistd.h>
#include <stdio.h>
#include <errno.h>

/*
 * 慢速系统调用（可能永远阻塞的调用）:
 *   read (终端/管道/socket), write (管道/socket)
 *   open (FIFO), pause, wait, ioctl
 *   select, poll, accept, connect
 *
 * 当信号中断慢速系统调用时:
 *   - 没有 SA_RESTART: 返回 -1, errno = EINTR
 *   - 有 SA_RESTART: 自动重启
 *
 * 永远不会被信号重启的调用:
 *   poll, epoll_wait, select, nanosleep
 */

void eintr_example(void)
{
    char buf[256];
    ssize_t n;

    // 方式1：手动处理 EINTR
    while (1) {
        n = read(STDIN_FILENO, buf, sizeof(buf));
        if (n == -1) {
            if (errno == EINTR) {
                printf("read was interrupted, retrying...\n");
                continue;
            }
            perror("read");
            break;
        }
        break;
    }

    // 方式2：使用 SA_RESTART 标志
    struct sigaction sa;
    sa.sa_handler = handler;
    sa.sa_flags = SA_RESTART;  // 自动重启被中断的系统调用
    sigemptyset(&sa.sa_mask);
    sigaction(SIGINT, &sa, NULL);
}

// 使用 pselect 实现精确的信号+超时控制
#include <sys/select.h>

void pselect_example(void)
{
    sigset_t mask, oldmask;
    fd_set readfds;
    struct timespec timeout = {5, 0};

    // 阻塞 SIGINT
    sigemptyset(&mask);
    sigaddset(&mask, SIGINT);
    sigprocmask(SIG_BLOCK, &mask, &oldmask);

    // pselect 原子地：临时解除阻塞 + 等待
    FD_ZERO(&readfds);
    FD_SET(STDIN_FILENO, &readfds);

    int ret = pselect(STDIN_FILENO + 1, &readfds, NULL, NULL,
                      &timeout, &oldmask);
    // pselect 在等待期间使用 oldmask 作为信号掩码
    // 返回后信号掩码恢复为调用前的值

    if (ret > 0) {
        char buf[256];
        read(STDIN_FILENO, buf, sizeof(buf));
    } else if (ret == 0) {
        printf("Timeout\n");
    }
}
```
