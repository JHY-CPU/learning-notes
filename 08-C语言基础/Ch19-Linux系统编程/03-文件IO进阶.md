# 文件IO进阶

## 1. ioctl 系统调用

```c
#include <sys/ioctl.h>

/*
 * int ioctl(int fd, unsigned long request, ...);
 *
 * 用于设备控制，是各种杂项操作的"垃圾桶"接口
 * 终端、网络接口、磁盘设备等都通过 ioctl 控制
 */

#include <stdio.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <net/if.h>
#include <string.h>

// 获取终端窗口大小
void get_terminal_size(void)
{
    struct winsize ws;
    if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0) {
        printf("Terminal: %d rows x %d columns\n", ws.ws_row, ws.ws_col);
    }
}

// 获取网络接口信息
void get_network_interfaces(void)
{
    struct ifreq ifr;
    int sock = socket(AF_INET, SOCK_DGRAM, 0);
    if (sock == -1) { perror("socket"); return; }

    strncpy(ifr.ifr_name, "eth0", IFNAMSIZ - 1);
    if (ioctl(sock, SIOCGIFADDR, &ifr) == 0) {
        struct sockaddr_in *addr = (struct sockaddr_in *)&ifr.ifr_addr;
        printf("eth0 IP: %s\n", inet_ntoa(addr->sin_addr));
    }

    close(sock);
}
```

## 2. /dev/fd 目录

```c
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

/*
 * /dev/fd/n  对应当前进程的文件描述符 n
 * /dev/stdin  = /dev/fd/0
 * /dev/stdout = /dev/fd/1
 * /dev/stderr = /dev/fd/2
 */

void dev_fd_example(void)
{
    // 打开 /dev/fd/1 相当于 dup(1)
    int fd = open("/dev/fd/1", O_WRONLY);
    if (fd != -1) {
        write(fd, "Written via /dev/fd/1\n", 22);
        close(fd);
    }
}
```

## 3. 文件状态与 fstat

```c
#include <sys/stat.h>
#include <unistd.h>
#include <stdio.h>
#include <time.h>

/*
 * int stat(const char *pathname, struct stat *statbuf);
 * int fstat(int fd, struct stat *statbuf);
 * int lstat(const char *pathname, struct stat *statbuf);
 *
 * stat：获取文件状态（跟随符号链接）
 * fstat：通过文件描述符获取
 * lstat：获取文件状态（不跟随符号链接）
 */

void stat_example(const char *path)
{
    struct stat st;

    if (lstat(path, &st) == -1) {
        perror("lstat");
        return;
    }

    // 文件类型
    printf("File type: ");
    if (S_ISREG(st.st_mode))       printf("Regular file\n");
    else if (S_ISDIR(st.st_mode))  printf("Directory\n");
    else if (S_ISLNK(st.st_mode))  printf("Symbolic link\n");
    else if (S_ISCHR(st.st_mode))  printf("Character device\n");
    else if (S_ISBLK(st.st_mode))  printf("Block device\n");
    else if (S_ISFIFO(st.st_mode)) printf("FIFO/Pipe\n");
    else if (S_ISSOCK(st.st_mode)) printf("Socket\n");

    // 文件属性
    printf("Inode: %lu\n", (unsigned long)st.st_ino);
    printf("Size: %ld bytes\n", (long)st.st_size);
    printf("Blocks: %ld (512B blocks)\n", (long)st.st_blocks);
    printf("IO Block: %ld bytes\n", (long)st.st_blksize);
    printf("Links: %lu\n", (unsigned long)st.st_nlink);
    printf("UID: %u\n", st.st_uid);
    printf("GID: %u\n", st.st_gid);

    // 权限位
    printf("Permissions: %o\n", st.st_mode & 0777);

    // 时间信息
    printf("Last access:       %s", ctime(&st.st_atime));
    printf("Last modification: %s", ctime(&st.st_mtime));
    printf("Last status change:%s", ctime(&st.st_ctime));

    // 设备号
    if (S_ISCHR(st.st_mode) || S_ISBLK(st.st_mode)) {
        printf("Device: major=%lu, minor=%lu\n",
               (unsigned long)major(st.st_rdev),
               (unsigned long)minor(st.st_rdev));
    }
}
```

## 4. 文件权限管理

```c
#include <sys/stat.h>
#include <unistd.h>
#include <stdio.h>

/*
 * int chmod(const char *pathname, mode_t mode);
 * int fchmod(int fd, mode_t mode);
 *
 * int access(const char *pathname, int mode);
 *   mode: R_OK, W_OK, X_OK, F_OK
 */

void permission_examples(void)
{
    // 检查文件权限
    if (access("test.txt", F_OK) == 0) {
        printf("File exists\n");
    }
    if (access("test.txt", R_OK) == 0) {
        printf("File is readable\n");
    }
    if (access("test.txt", W_OK) == 0) {
        printf("File is writable\n");
    }

    // 修改文件权限
    chmod("test.txt", 0644);           // rw-r--r--
    chmod("script.sh", 0755);          // rwxr-xr-x

    // 添加组写权限
    struct stat st;
    stat("test.txt", &st);
    chmod("test.txt", st.st_mode | S_IWGRP);

    // umask：设置创建文件时的默认权限掩码
    mode_t old_mask = umask(0022);     // 新文件权限 = 0666 & ~0022 = 0644
    int fd = open("newfile.txt", O_CREAT | O_WRONLY, 0666);
    close(fd);
    umask(old_mask);                   // 恢复原来的 umask
}
```

## 5. 文件锁

```c
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

/*
 * 文件锁用于协调多个进程对同一文件的访问
 *
 * struct flock {
 *     short l_type;    - F_RDLCK, F_WRLCK, F_UNLCK
 *     short l_whence;  - SEEK_SET, SEEK_CUR, SEEK_END
 *     off_t l_start;   - 起始偏移
 *     off_t l_len;     - 长度（0表示到文件末尾）
 *     pid_t l_pid;     - 持有锁的进程ID
 * };
 */

// 设置文件锁（非阻塞）
int set_lock(int fd, int type, off_t start, off_t len)
{
    struct flock lock;
    lock.l_type   = type;
    lock.l_whence = SEEK_SET;
    lock.l_start  = start;
    lock.l_len    = len;

    return fcntl(fd, F_SETLK, &lock);
}

// 设置文件锁（阻塞，等待获取锁）
int set_lock_wait(int fd, int type, off_t start, off_t len)
{
    struct flock lock;
    lock.l_type   = type;
    lock.l_whence = SEEK_SET;
    lock.l_start  = start;
    lock.l_len    = len;

    return fcntl(fd, F_SETLKW, &lock);
}

// 获取当前锁信息
int get_lock_info(int fd, off_t start, off_t len)
{
    struct flock lock;
    lock.l_type   = F_WRLCK;
    lock.l_whence = SEEK_SET;
    lock.l_start  = start;
    lock.l_len    = len;

    if (fcntl(fd, F_GETLK, &lock) == -1) {
        perror("fcntl F_GETLK");
        return -1;
    }

    if (lock.l_type == F_UNLCK) {
        printf("Region is not locked\n");
    } else {
        printf("Region locked by PID %d, type: %s\n",
               lock.l_pid,
               lock.l_type == F_RDLCK ? "read" : "write");
    }

    return 0;
}

// 文件锁使用示例
void file_lock_example(void)
{
    int fd = open("shared.txt", O_RDWR | O_CREAT, 0644);
    if (fd == -1) { perror("open"); return; }

    // 获取写锁
    if (set_lock(fd, F_WRLCK, 0, 0) == -1) {
        if (errno == EAGAIN || errno == EACCES) {
            printf("File is locked by another process\n");
        } else {
            perror("set_lock");
        }
        close(fd);
        return;
    }

    printf("Lock acquired, writing data...\n");

    // 写入数据（独占访问）
    const char *data = "Critical data\n";
    write(fd, data, strlen(data));
    fsync(fd);

    // 释放锁
    set_lock(fd, F_UNLCK, 0, 0);
    printf("Lock released\n");

    close(fd);
}

// 写锁定整个文件的封装
int lock_file(int fd)
{
    return set_lock_wait(fd, F_WRLCK, 0, 0);
}

int unlock_file(int fd)
{
    return set_lock(fd, F_UNLCK, 0, 0);
}
```

## 6. 非阻塞 IO

```c
#include <fcntl.h>
#include <unistd.h>
#include <errno.h>
#include <stdio.h>

/*
 * 设置非阻塞的方式:
 * 1. open 时使用 O_NONBLOCK 标志
 * 2. 使用 fcntl 设置 O_NONBLOCK
 */

void nonblocking_example(void)
{
    // 方式1：打开时设置非阻塞
    int fd = open("fifo_test", O_RDONLY | O_NONBLOCK);

    // 方式2：通过 fcntl 设置
    // int flags = fcntl(fd, F_GETFL);
    // fcntl(fd, F_SETFL, flags | O_NONBLOCK);

    if (fd == -1) { perror("open"); return; }

    char buf[256];
    ssize_t n;

    while (1) {
        n = read(fd, buf, sizeof(buf));
        if (n > 0) {
            write(STDOUT_FILENO, buf, n);
        } else if (n == 0) {
            printf("EOF\n");
            break;
        } else {
            if (errno == EAGAIN || errno == EWOULDBLOCK) {
                // 非阻塞IO：暂时没有数据可读
                printf("No data available, doing other work...\n");
                usleep(100000);  // 模拟其他工作
                continue;
            } else {
                perror("read");
                break;
            }
        }
    }

    close(fd);
}

// 非阻塞 connect 示例（超时处理）
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

int nonblocking_connect(const char *ip, int port, int timeout_sec)
{
    int sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock == -1) { perror("socket"); return -1; }

    // 设置非阻塞
    int flags = fcntl(sock, F_GETFL, 0);
    fcntl(sock, F_SETFL, flags | O_NONBLOCK);

    struct sockaddr_in addr;
    addr.sin_family = AF_INET;
    addr.sin_port = htons(port);
    inet_pton(AF_INET, ip, &addr.sin_addr);

    int ret = connect(sock, (struct sockaddr *)&addr, sizeof(addr));
    if (ret == -1) {
        if (errno == EINPROGRESS) {
            // 连接正在进行，使用 select/poll 等待
            fd_set wfds;
            FD_ZERO(&wfds);
            FD_SET(sock, &wfds);

            struct timeval tv = {timeout_sec, 0};
            ret = select(sock + 1, NULL, &wfds, NULL, &tv);
            if (ret > 0) {
                int err;
                socklen_t len = sizeof(err);
                getsockopt(sock, SOL_SOCKET, SO_ERROR, &err, &len);
                if (err != 0) {
                    fprintf(stderr, "Connect failed: %s\n", strerror(err));
                    close(sock);
                    return -1;
                }
                printf("Connected!\n");
            } else {
                fprintf(stderr, "Connect timed out\n");
                close(sock);
                return -1;
            }
        } else {
            perror("connect");
            close(sock);
            return -1;
        }
    }

    // 恢复阻塞模式
    fcntl(sock, F_SETFL, flags);
    return sock;
}
```

## 7. scatter/gather IO（分散聚集读写）

```c
#include <sys/uio.h>
#include <unistd.h>
#include <string.h>
#include <stdio.h>

/*
 * ssize_t readv(int fd, const struct iovec *iov, int iovcnt);
 * ssize_t writev(int fd, const struct iovec *iov, int iovcnt);
 *
 * struct iovec {
 *     void  *iov_base;  - 缓冲区地址
 *     size_t iov_len;   - 缓冲区长度
 * };
 *
 * 优势：一次系统调用读写多个缓冲区
 */

void writev_example(void)
{
    int fd = open("scatter.dat", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return; }

    // 准备多个缓冲区
    struct iovec iov[3];
    char *header = "HEADER: ";
    char *body   = "This is the body content. ";
    char *footer = "END.\n";

    iov[0].iov_base = header;
    iov[0].iov_len  = strlen(header);
    iov[1].iov_base = body;
    iov[1].iov_len  = strlen(body);
    iov[2].iov_base = footer;
    iov[2].iov_len  = strlen(footer);

    // 一次系统调用写入所有缓冲区
    ssize_t nwritten = writev(fd, iov, 3);
    printf("Wrote %zd bytes\n", nwritten);

    close(fd);
}

void readv_example(void)
{
    int fd = open("scatter.dat", O_RDONLY);
    if (fd == -1) { perror("open"); return; }

    char header[8], body[64], footer[8];
    struct iovec iov[3];

    iov[0].iov_base = header;
    iov[0].iov_len  = sizeof(header);
    iov[1].iov_base = body;
    iov[1].iov_len  = sizeof(body);
    iov[2].iov_base = footer;
    iov[2].iov_len  = sizeof(footer);

    ssize_t nread = readv(fd, iov, 3);
    printf("Read %zd bytes\n", nread);

    // 打印结果（注意：不以 \0 结尾）
    printf("Header: '%.*s'\n", (int)iov[0].iov_len, header);
    printf("Body: '%.*s'\n", (int)iov[1].iov_len, body);
    printf("Footer: '%.*s'\n", (int)iov[2].iov_len, footer);

    close(fd);
}
```

## 8. 截断文件

```c
#include <unistd.h>
#include <sys/stat.h>
#include <stdio.h>

/*
 * int truncate(const char *path, off_t length);
 * int ftruncate(int fd, off_t length);
 *
 * 将文件截断到指定长度
 * 如果 length < 当前大小：截断，多余数据丢失
 * 如果 length > 当前大小：扩展，用零填充
 */

void truncate_examples(void)
{
    // 创建一个 100 字节的文件
    int fd = open("test_trunc.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    char buf[100];
    memset(buf, 'A', sizeof(buf));
    write(fd, buf, sizeof(buf));
    close(fd);

    struct stat st;

    // 截断到 50 字节
    truncate("test_trunc.txt", 50);
    stat("test_trunc.txt", &st);
    printf("After truncate to 50: %ld bytes\n", (long)st.st_size);

    // 扩展到 200 字节
    truncate("test_trunc.txt", 200);
    stat("test_trunc.txt", &st);
    printf("After truncate to 200: %ld bytes\n", (long)st.st_size);

    unlink("test_trunc.txt");
}
```

## 9. 综合示例：文件记录锁应用

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <string.h>
#include <errno.h>

#define RECORD_SIZE 128
#define NUM_RECORDS 10

/*
 * 使用文件锁实现简单的数据库记录锁
 * 演示读锁和写锁的使用
 */

typedef struct {
    int id;
    char name[64];
    double value;
    int deleted;
} Record;

// 写入记录（加写锁）
int write_record(int fd, int rec_num, Record *rec)
{
    if (rec_num < 0 || rec_num >= NUM_RECORDS) return -1;

    // 锁定记录区域
    struct flock lock;
    lock.l_type   = F_WRLCK;
    lock.l_whence = SEEK_SET;
    lock.l_start  = rec_num * RECORD_SIZE;
    lock.l_len    = RECORD_SIZE;

    if (fcntl(fd, F_SETLKW, &lock) == -1) {
        perror("write lock");
        return -1;
    }

    // 定位并写入
    lseek(fd, rec_num * RECORD_SIZE, SEEK_SET);
    write(fd, rec, sizeof(Record));

    // 解锁
    lock.l_type = F_UNLCK;
    fcntl(fd, F_SETLK, &lock);

    return 0;
}

// 读取记录（加读锁）
int read_record(int fd, int rec_num, Record *rec)
{
    if (rec_num < 0 || rec_num >= NUM_RECORDS) return -1;

    struct flock lock;
    lock.l_type   = F_RDLCK;
    lock.l_whence = SEEK_SET;
    lock.l_start  = rec_num * RECORD_SIZE;
    lock.l_len    = RECORD_SIZE;

    if (fcntl(fd, F_SETLKW, &lock) == -1) {
        perror("read lock");
        return -1;
    }

    lseek(fd, rec_num * RECORD_SIZE, SEEK_SET);
    ssize_t n = read(fd, rec, sizeof(Record));

    lock.l_type = F_UNLCK;
    fcntl(fd, F_SETLK, &lock);

    return (n == sizeof(Record)) ? 0 : -1;
}

int main(void)
{
    int fd = open("records.db", O_RDWR | O_CREAT, 0644);
    if (fd == -1) { perror("open"); return 1; }

    // 写入测试记录
    Record rec;
    for (int i = 0; i < 5; i++) {
        rec.id = i;
        snprintf(rec.name, sizeof(rec.name), "Record_%d", i);
        rec.value = i * 1.5;
        rec.deleted = 0;
        write_record(fd, i, &rec);
    }

    // 读取测试记录
    for (int i = 0; i < 5; i++) {
        if (read_record(fd, i, &rec) == 0) {
            printf("Record %d: id=%d, name=%s, value=%.1f\n",
                   i, rec.id, rec.name, rec.value);
        }
    }

    close(fd);
    return 0;
}
```
