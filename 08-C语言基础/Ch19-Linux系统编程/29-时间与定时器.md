# 时间与定时器

## 1. 时间获取

```c
#include <time.h>
#include <sys/time.h>
#include <stdio.h>

/*
 * time_t time(time_t *tloc);                    - 秒级时间戳
 * int gettimeofday(struct timeval *tv, ...);     - 微秒级
 * int clock_gettime(clockid_t clk_id, ...);      - 纳秒级
 *
 * clock_id:
 *   CLOCK_REALTIME   - 系统实时时钟
 *   CLOCK_MONOTONIC  - 单调时钟（不受时间调整影响）
 *   CLOCK_PROCESS_CPUTIME_ID - 进程 CPU 时间
 *   CLOCK_THREAD_CPUTIME_ID  - 线程 CPU 时间
 */

void time_examples(void)
{
    // time() - 秒级
    time_t t = time(NULL);
    printf("time(): %ld\n", (long)t);

    // gettimeofday() - 微秒级
    struct timeval tv;
    gettimeofday(&tv, NULL);
    printf("gettimeofday(): %ld.%06ld\n",
           (long)tv.tv_sec, (long)tv.tv_usec);

    // clock_gettime() - 纳秒级（推荐）
    struct timespec ts;
    clock_gettime(CLOCK_REALTIME, &ts);
    printf("CLOCK_REALTIME:  %ld.%09ld\n",
           (long)ts.tv_sec, (long)ts.tv_nsec);

    clock_gettime(CLOCK_MONOTONIC, &ts);
    printf("CLOCK_MONOTONIC: %ld.%09ld\n",
           (long)ts.tv_sec, (long)ts.tv_nsec);

    // 精确计时
    struct timespec start, end;
    clock_gettime(CLOCK_MONOTONIC, &start);

    // ... 执行代码 ...
    volatile long sum = 0;
    for (long i = 0; i < 1000000; i++) sum += i;

    clock_gettime(CLOCK_MONOTONIC, &end);

    double elapsed = (end.tv_sec - start.tv_sec) +
                     (end.tv_nsec - start.tv_nsec) / 1e9;
    printf("Elapsed: %.6f seconds\n", elapsed);
}
```

## 2. 时间格式化

```c
#include <time.h>
#include <stdio.h>
#include <string.h>

void time_format_examples(void)
{
    time_t t = time(NULL);

    // 转换为本地时间
    struct tm *local = localtime(&t);
    printf("Year: %d, Month: %d, Day: %d\n",
           local->tm_year + 1900, local->tm_mon + 1, local->tm_mday);
    printf("Hour: %d, Min: %d, Sec: %d\n",
           local->tm_hour, local->tm_min, local->tm_sec);

    // 转换为 UTC
    struct tm *utc = gmtime(&t);

    // strftime 格式化
    char buf[256];
    strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", local);
    printf("Formatted: %s\n", buf);

    strftime(buf, sizeof(buf), "%A, %B %d, %Y", local);
    printf("Full: %s\n", buf);

    // strptime 解析时间字符串
    struct tm parsed;
    memset(&parsed, 0, sizeof(parsed));
    strptime("2024-06-15 14:30:00", "%Y-%m-%d %H:%M:%S", &parsed);
    time_t parsed_t = mktime(&parsed);
    printf("Parsed timestamp: %ld\n", (long)parsed_t);

    // ctime 和 asctime
    printf("ctime: %s", ctime(&t));
    printf("asctime: %s", asctime(local));

    // timegm - UTC 版 mktime
    time_t utc_t = timegm(utc);
    printf("UTC timestamp: %ld\n", (long)utc_t);
}
```

## 3. POSIX 定时器

```c
#include <time.h>
#include <signal.h>
#include <stdio.h>
#include <unistd.h>

/*
 * POSIX 定时器 API:
 *   int timer_create(clockid_t clockid, struct sigevent *sevp,
 *                     timer_t *timerid);
 *   int timer_settime(timer_t timerid, int flags,
 *                     const struct itimerspec *new_value,
 *                     struct itimerspec *old_value);
 *   int timer_gettime(timer_t timerid, struct itimerspec *curr_value);
 *   int timer_delete(timer_t timerid);
 *   int timer_getoverrun(timer_t timerid);
 *
 * struct itimerspec {
 *     struct timespec it_interval;  - 周期间隔
 *     struct timespec it_value;     - 首次到期时间
 * };
 */

static volatile sig_atomic_t timer_tick = 0;

void timer_handler(int sig, siginfo_t *si, void *uc)
{
    (void)sig;
    (void)si;
    (void)uc;
    timer_tick++;
}

void posix_timer_example(void)
{
    // 设置信号处理
    struct sigaction sa;
    sa.sa_flags = SA_SIGINFO;
    sa.sa_sigaction = timer_handler;
    sigemptyset(&sa.sa_mask);
    sigaction(SIGALRM, &sa, NULL);

    // 创建定时器
    struct sigevent sev;
    sev.sigev_notify = SIGEV_SIGNAL;
    sev.sigev_signo = SIGALRM;
    sev.sigev_value.sival_ptr = NULL;

    timer_t timerid;
    timer_create(CLOCK_REALTIME, &sev, &timerid);

    // 设置定时器：1秒后触发，每秒重复
    struct itimerspec its;
    its.it_value.tv_sec = 1;     // 首次 1 秒后
    its.it_value.tv_nsec = 0;
    its.it_interval.tv_sec = 1;  // 每秒重复
    its.it_interval.tv_nsec = 0;

    timer_settime(timerid, 0, &its, NULL);

    printf("Timer started, ticks every second...\n");

    for (int i = 0; i < 5; i++) {
        pause();  // 等待信号
        printf("Tick %d\n", (int)timer_tick);
    }

    // 删除定时器
    timer_delete(timerid);
}

// 高精度定时器
void high_res_timer(void)
{
    struct sigevent sev;
    timer_t timerid;

    sev.sigev_notify = SIGEV_THREAD;
    sev.sigev_notify_function = (void (*)(sigval_t))timer_handler;
    sev.sigev_value.sival_ptr = NULL;
    sev.sigev_notify_attributes = NULL;

    timer_create(CLOCK_MONOTONIC, &sev, &timerid);

    struct itimerspec its;
    its.it_value.tv_sec = 0;
    its.it_value.tv_nsec = 1000000;  // 1 毫秒
    its.it_interval.tv_sec = 0;
    its.it_interval.tv_nsec = 1000000;

    timer_settime(timerid, 0, &its, NULL);

    sleep(1);
    printf("Timer ticks: %d\n", (int)timer_tick);

    timer_delete(timerid);
}
```

## 4. nanosleep 精确休眠

```c
#include <time.h>
#include <stdio.h>
#include <errno.h>

/*
 * int nanosleep(const struct timespec *req, struct timespec *rem);
 *
 * 比 sleep() 更精确
 * 被信号中断时返回 -1，errno = EINTR
 * 剩余时间保存在 *rem 中
 */

// 可靠的休眠（处理信号中断）
int reliable_nanosleep(double seconds)
{
    struct timespec req, rem;
    req.tv_sec = (time_t)seconds;
    req.tv_nsec = (long)((seconds - req.tv_sec) * 1e9);

    while (nanosleep(&req, &rem) == -1) {
        if (errno == EINTR) {
            req = rem;  // 继续休眠剩余时间
        } else {
            return -1;
        }
    }
    return 0;
}

// 毫秒级休眠
void msleep(int ms)
{
    struct timespec ts;
    ts.tv_sec = ms / 1000;
    ts.tv_nsec = (ms % 1000) * 1000000L;
    nanosleep(&ts, NULL);
}

// 微秒级休眠
void usleep_alt(int us)
{
    struct timespec ts;
    ts.tv_sec = us / 1000000;
    ts.tv_nsec = (us % 1000000) * 1000L;
    nanosleep(&ts, NULL);
}
```
