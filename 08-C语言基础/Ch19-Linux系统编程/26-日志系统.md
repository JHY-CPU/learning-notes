# 日志系统

## 1. syslog 系统

```c
#include <syslog.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>
#include <unistd.h>

/*
 * syslog API:
 *   void openlog(const char *ident, int option, int facility);
 *   void syslog(int priority, const char *format, ...);
 *   void closelog(void);
 *   int setlogmask(int mask);
 *
 * 优先级（从高到低）:
 *   LOG_EMERG   = 0  - 系统不可用
 *   LOG_ALERT   = 1  - 必须立即行动
 *   LOG_CRIT    = 2  - 临界条件
 *   LOG_ERR     = 3  - 错误条件
 *   LOG_WARNING = 4  - 警告条件
 *   LOG_NOTICE  = 5  - 正常但重要
 *   LOG_INFO    = 6  - 信息性
 *   LOG_DEBUG   = 7  - 调试信息
 *
 * 设施（facility）:
 *   LOG_DAEMON - 守护进程
 *   LOG_USER   - 用户进程
 *   LOG_LOCAL0-7 - 本地使用
 */

void syslog_examples(void)
{
    // 打开日志连接
    openlog("myapp", LOG_PID | LOG_CONS, LOG_DAEMON);

    // 记录不同级别的日志
    syslog(LOG_INFO, "Application started by UID %d", getuid());
    syslog(LOG_WARNING, "Low disk space: %d%% free", 15);
    syslog(LOG_ERR, "Failed to open config: %s", strerror(errno));
    syslog(LOG_DEBUG, "Processing request %d", 42);

    // 设置日志掩码：只记录 INFO 及以上
    setlogmask(LOG_UPTO(LOG_INFO));
    syslog(LOG_DEBUG, "This debug message will NOT be logged");

    closelog();
}

// 完整的日志封装
typedef enum {
    LOG_LEVEL_DEBUG,
    LOG_LEVEL_INFO,
    LOG_LEVEL_WARN,
    LOG_LEVEL_ERROR,
    LOG_LEVEL_FATAL
} LogLevel;

static LogLevel current_level = LOG_LEVEL_DEBUG;

void log_init(const char *name, LogLevel level)
{
    current_level = level;
    openlog(name, LOG_PID | LOG_NDELAY, LOG_DAEMON);
}

void log_msg(LogLevel level, const char *file, int line,
             const char *fmt, ...)
{
    if (level < current_level) return;

    // 将自定义级别映射到 syslog 优先级
    int priorities[] = {
        LOG_DEBUG, LOG_INFO, LOG_WARNING, LOG_ERR, LOG_CRIT
    };

    char buf[1024];
    va_list args;
    va_start(args, fmt);
    vsnprintf(buf, sizeof(buf), fmt, args);
    va_end(args);

    syslog(priorities[level], "[%s:%d] %s", file, line, buf);
}

#define LOG_DEBUG(fmt, ...) log_msg(LOG_LEVEL_DEBUG, __FILE__, __LINE__, fmt, ##__VA_ARGS__)
#define LOG_INFO(fmt, ...)  log_msg(LOG_INFO, __FILE__, __LINE__, fmt, ##__VA_ARGS__)
#define LOG_ERROR(fmt, ...) log_msg(LOG_LEVEL_ERROR, __FILE__, __LINE__, fmt, ##__VA_ARGS__)
```

## 2. 自定义日志库

```c
#include <stdio.h>
#include <stdarg.h>
#include <time.h>
#include <string.h>
#include <pthread.h>
#include <unistd.h>
#include <sys/stat.h>

/*
 * 自定义日志系统：
 * 支持多级别、文件输出、日志轮转、线程安全
 */

typedef enum {
    LVL_TRACE, LVL_DEBUG, LVL_INFO,
    LVL_WARN, LVL_ERROR, LVL_FATAL
} LogLevel;

typedef struct {
    FILE *fp;
    LogLevel level;
    char filename[256];
    size_t max_size;
    int max_backups;
    pthread_mutex_t lock;
} Logger;

static Logger g_logger;

static const char *level_names[] = {
    "TRACE", "DEBUG", "INFO", "WARN", "ERROR", "FATAL"
};

// 初始化日志
int logger_init(const char *filename, LogLevel level,
                size_t max_size, int max_backups)
{
    memset(&g_logger, 0, sizeof(g_logger));
    g_logger.level = level;
    g_logger.max_size = max_size;
    g_logger.max_backups = max_backups;
    strncpy(g_logger.filename, filename, sizeof(g_logger.filename) - 1);

    g_logger.fp = fopen(filename, "a");
    if (!g_logger.fp) {
        perror("fopen log");
        return -1;
    }

    pthread_mutex_init(&g_logger.lock, NULL);
    return 0;
}

// 日志轮转
static void rotate_log(void)
{
    if (!g_logger.fp) return;

    // 检查文件大小
    struct stat st;
    if (fstat(fileno(g_logger.fp), &st) == 0) {
        if ((size_t)st.st_size < g_logger.max_size) return;
    }

    fclose(g_logger.fp);

    // 轮转备份文件
    char old_path[512], new_path[512];

    // 删除最旧的备份
    snprintf(old_path, sizeof(old_path), "%s.%d",
             g_logger.filename, g_logger.max_backups);
    unlink(old_path);

    // 依次重命名
    for (int i = g_logger.max_backups - 1; i >= 1; i--) {
        snprintf(old_path, sizeof(old_path), "%s.%d",
                 g_logger.filename, i);
        snprintf(new_path, sizeof(new_path), "%s.%d",
                 g_logger.filename, i + 1);
        rename(old_path, new_path);
    }

    // 当前日志 -> .1
    snprintf(new_path, sizeof(new_path), "%s.1", g_logger.filename);
    rename(g_logger.filename, new_path);

    // 重新打开
    g_logger.fp = fopen(g_logger.filename, "a");
}

// 写日志
void logger_log(LogLevel level, const char *file, int line,
                const char *fmt, ...)
{
    if (level < g_logger.level) return;

    pthread_mutex_lock(&g_logger.lock);

    // 检查是否需要轮转
    rotate_log();

    // 时间戳
    time_t now = time(NULL);
    struct tm tm;
    localtime_r(&now, &tm);

    char time_buf[64];
    strftime(time_buf, sizeof(time_buf), "%Y-%m-%d %H:%M:%S", &tm);

    // 写入日志头
    fprintf(g_logger.fp, "[%s] [%s] [%s:%d] [%lu] ",
            time_buf, level_names[level], file, line,
            (unsigned long)getpid());

    // 写入消息
    va_list args;
    va_start(args, fmt);
    vfprintf(g_logger.fp, fmt, args);
    va_end(args);

    fprintf(g_logger.fp, "\n");
    fflush(g_logger.fp);

    pthread_mutex_unlock(&g_logger.lock);
}

void logger_close(void)
{
    pthread_mutex_lock(&g_logger.lock);
    if (g_logger.fp) {
        fclose(g_logger.fp);
        g_logger.fp = NULL;
    }
    pthread_mutex_unlock(&g_logger.lock);
    pthread_mutex_destroy(&g_logger.lock);
}

// 使用示例
#define LOG(level, fmt, ...) \
    logger_log(level, __FILE__, __LINE__, fmt, ##__VA_ARGS__)

void logger_demo(void)
{
    logger_init("/tmp/myapp.log", LVL_DEBUG, 1024 * 1024, 5);

    LOG(LVL_DEBUG, "Debug message");
    LOG(LVL_INFO, "Server started on port %d", 8080);
    LOG(LVL_WARN, "Memory usage: %d MB", 512);
    LOG(LVL_ERROR, "Connection failed: %s", "timeout");

    logger_close();
}
```
