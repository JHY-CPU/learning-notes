# 调试技术

## 1. GDB 基础

```c
#include <stdio.h>
#include <stdlib.h>

/*
 * 编译时开启调试信息:
 *   gcc -g -O0 -Wall -o program program.c
 *
 * GDB 常用命令:
 *   gdb ./program           - 启动调试
 *   gdb ./program core      - 分析 core dump
 *   gdb -p <pid>            - 附加到运行中的进程
 *
 * 运行控制:
 *   run [args]   - 运行程序
 *   break main   - 设置断点
 *   break file.c:42 - 在文件行号设置断点
 *   next (n)     - 单步（跳过函数）
 *   step (s)     - 单步（进入函数）
 *   continue (c) - 继续运行
 *   finish       - 运行到当前函数返回
 *   until        - 运行到当前循环结束
 *
 * 查看信息:
 *   print var    - 打印变量
 *   print/x var  - 十六进制
 *   display var  - 每次停止时自动打印
 *   info locals  - 所有局部变量
 *   info args    - 函数参数
 *   bt / where   - 调用栈
 *   frame N      - 切换到第 N 帧
 *   list         - 显示源代码
 *
 * 内存:
 *   x/10x addr  - 查看内存
 *   x/s addr    - 查看字符串
 *   watch var   - 观察点（值变化时中断）
 */

// 生成 core dump 的示例
void crash_example(void)
{
    int *p = NULL;
    *p = 42;  // 段错误，生成 core dump
}

// 可以用 GDB 调试的函数
int factorial(int n)
{
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

void gdb_debug_example(void)
{
    int result = factorial(5);
    printf("5! = %d\n", result);
}
```

## 2. Core Dump 分析

```c
#include <stdio.h>
#include <stdlib.h>
#include <sys/resource.h>

/*
 * Core Dump 配置:
 *   ulimit -c unlimited    - 允许生成 core dump
 *   /proc/sys/kernel/core_pattern - core 文件路径模式
 *
 * 分析 core dump:
 *   gdb ./program core
 *   (gdb) bt           - 查看调用栈
 *   (gdb) frame 0      - 切换到崩溃帧
 *   (gdb) print var    - 查看变量
 *   (gdb) info registers - 查看寄存器
 */

void enable_core_dump(void)
{
    struct rlimit rl;
    rl.rlim_cur = RLIM_INFINITY;
    rl.rlim_max = RLIM_INFINITY;
    setrlimit(RLIMIT_CORE, &rl);
}

// 常见崩溃类型

// 1. 空指针解引用
void null_deref(void)
{
    char *p = NULL;
    *p = 'a';  // SIGSEGV
}

// 2. 缓冲区溢出
void buffer_overflow(void)
{
    char buf[10];
    memset(buf, 'A', 100);  // 越界写入
}

// 3. 栈溢出
int stack_overflow(int n)
{
    char big[10000];  // 每次递归分配大量栈空间
    (void)big;
    return stack_overflow(n + 1);  // 无限递归
}

// 4. 双重释放
void double_free(void)
{
    int *p = malloc(sizeof(int));
    free(p);
    free(p);  // SIGABRT 或内存损坏
}

// 5. 使用已释放内存
void use_after_free(void)
{
    int *p = malloc(sizeof(int));
    *p = 42;
    free(p);
    printf("%d\n", *p);  // 未定义行为
}
```

## 3. strace - 系统调用跟踪

```bash
# strace 基本使用

# 跟踪程序的所有系统调用
strace ./program

# 跟踪特定系统调用
strace -e trace=open,read,write ./program

# 跟踪文件操作
strace -e trace=file ./program

# 跟踪网络操作
strace -e trace=network ./program

# 跟踪进程操作
strace -e trace=process ./program

# 附加到运行中的进程
strace -p <pid>

# 显示调用时间
strace -t ./program

# 显示系统调用耗时
strace -T ./program

# 统计系统调用
strace -c ./program

# 跟踪子进程
strace -f ./program

# 将输出写入文件
strace -o trace.log ./program
```

```c
// 用 strace 分析程序问题

// 问题：程序启动慢
// strace -T ./program
// 可能发现：connect() 调用超时

// 问题：文件找不到
// strace -e trace=open ./program
// 可能发现：程序在错误路径查找文件

// 问题：权限问题
// strace -e trace=open,openat ./program
// 可能发现：EACCES 错误
```

## 5. Valgrind - 内存检查

```bash
# Valgrind 内存检查工具

# 内存错误检查
valgrind --leak-check=full ./program

# 简洁输出
valgrind --leak-check=summary ./program

# 显示所有泄漏的详细信息
valgrind --leak-check=full --show-leak-kinds=all ./program

# 检查未初始化的内存
valgrind --track-origins=yes ./program

# 缓存分析
valgrind --tool=cachegrind ./program

# 堆分析
valgrind --tool=massif ./program
```

```c
// Valgrind 会检测的常见问题:

// 1. 内存泄漏
void memory_leak(void)
{
    char *p = malloc(100);
    // 忘记 free(p)
}

// 2. 使用未初始化的内存
void uninitialized_read(void)
{
    int *p = malloc(sizeof(int));
    if (*p > 0)  // 读取未初始化的值
        printf("positive\n");
    free(p);
}

// 3. 越界访问
void out_of_bounds(void)
{
    int *arr = malloc(10 * sizeof(int));
    arr[10] = 42;  // 越界
    free(arr);
}

// 4. 重复释放
void double_free_valgrind(void)
{
    int *p = malloc(sizeof(int));
    free(p);
    free(p);  // Valgrind 报错
}

// 5. 无效的 free
void invalid_free(void)
{
    int x;
    free(&x);  // 不是 malloc 返回的地址
}
```

## 6. AddressSanitizer

```bash
# 编译时开启 AddressSanitizer
gcc -fsanitize=address -g -o program program.c

# 也支持 UBSan (Undefined Behavior Sanitizer)
gcc -fsanitize=undefined -g -o program program.c

# 同时开启
gcc -fsanitize=address,undefined -g -o program program.c
```

```c
// ASan 会在运行时检测以下问题:
// - 堆/栈/全局变量的越界访问
// - 使用释放后的内存
// - 内存泄漏
// - double-free

void asan_example(void)
{
    int arr[5];
    arr[5] = 1;  // 栈缓冲区溢出 - ASan 立即报错
}
```

## 7. 调试宏

```c
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

// 调试打印宏
#ifdef DEBUG
#define DBG(fmt, ...) \
    fprintf(stderr, "[DEBUG] %s:%d: " fmt "\n", \
            __FILE__, __LINE__, ##__VA_ARGS__)
#else
#define DBG(fmt, ...)
#endif

// 错误打印宏
#define ERR(fmt, ...) \
    fprintf(stderr, "[ERROR] %s:%d: " fmt "\n", \
            __FILE__, __LINE__, ##__VA_ARGS__)

// 断言增强
#define ASSERT_MSG(cond, fmt, ...) \
    do { \
        if (!(cond)) { \
            fprintf(stderr, "ASSERT FAILED: %s:%d: %s: " fmt "\n", \
                    __FILE__, __LINE__, #cond, ##__VA_ARGS__); \
            abort(); \
        } \
    } while (0)

// 性能计时宏
#define TIME_IT(label, code) \
    do { \
        struct timespec _start, _end; \
        clock_gettime(CLOCK_MONOTONIC, &_start); \
        code; \
        clock_gettime(CLOCK_MONOTONIC, &_end); \
        double _elapsed = (_end.tv_sec - _start.tv_sec) + \
                          (_end.tv_nsec - _start.tv_nsec) / 1e9; \
        fprintf(stderr, "[TIME] %s: %.6f sec\n", label, _elapsed); \
    } while (0)

void debug_demo(void)
{
    DBG("Starting calculation");

    int x = 42;
    DBG("x = %d", x);

    ASSERT_MSG(x > 0, "x must be positive, got %d", x);

    TIME_IT("loop", {
        long sum = 0;
        for (long i = 0; i < 1000000; i++) sum += i;
        DBG("sum = %ld", sum);
    });
}
```
