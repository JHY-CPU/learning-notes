# 进程环境

## 1. 进程的启动与终止

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

/*
 * 进程启动:
 *   内核通过 exec 系统调用加载程序
 *   C 运行时设置好栈后调用 main()
 *
 * main 函数原型:
 *   int main(int argc, char *argv[], char *envp[]);
 *   argc: 命令行参数个数（包括程序名）
 *   argv: 参数字符串数组，argv[argc] == NULL
 *   envp: 环境变量数组（非 POSIX 标准，但 Linux 支持）
 */

// 处理命令行参数
int main(int argc, char *argv[])
{
    printf("Program name: %s\n", argv[0]);
    printf("Number of arguments: %d\n", argc - 1);

    for (int i = 1; i < argc; i++) {
        printf("argv[%d] = '%s'\n", i, argv[i]);
    }

    // 使用 getopt 解析选项
    int opt;
    int verbose = 0;
    char *output = NULL;

    while ((opt = getopt(argc, argv, "vo:h")) != -1) {
        switch (opt) {
        case 'v': verbose = 1; break;
        case 'o': output = optarg; break;
        case 'h':
            printf("Usage: %s [-v] [-o output] [file...]\n", argv[0]);
            return 0;
        default:
            fprintf(stderr, "Usage: %s [-v] [-o output] [file...]\n", argv[0]);
            return 1;
        }
    }

    // optind 指向第一个非选项参数
    for (int i = optind; i < argc; i++) {
        printf("File argument: %s\n", argv[i]);
    }

    return 0;
}
```

## 2. 进程终止

```c
#include <stdlib.h>
#include <stdio.h>

/*
 * 正常终止:
 *   1. main 函数 return
 *   2. 调用 exit()
 *   3. 调用 _exit() 或 _Exit()（不执行清理）
 *   4. 最后一个线程返回
 *   5. 最后一个线程调用 pthread_exit()
 *
 * 异常终止:
 *   1. 调用 abort()
 *   2. 接收到终止信号
 *   3. 最后一个线程响应取消请求
 */

void termination_examples(void)
{
    // exit() vs _exit()
    // exit()  - 执行 atexit 注册的函数，刷新并关闭 stdio 流
    // _exit() - 立即进入内核，不执行任何清理

    // 注册退出处理函数
    atexit(handler1);
    atexit(handler2);  // LIFO 顺序：先注册的后执行

    printf("main function\n");
    exit(0);  // 调用 handler2 -> handler1 -> 刷新流
}

void handler1(void) { printf("Cleanup handler 1\n"); }
void handler2(void) { printf("Cleanup handler 2\n"); }
```

## 3. 环境变量

```c
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

/*
 * char *getenv(const char *name);
 * int putenv(char *string);
 * int setenv(const char *name, const char *value, int overwrite);
 * int unsetenv(const char *name);
 */

void env_examples(void)
{
    // 读取环境变量
    char *home = getenv("HOME");
    char *path = getenv("PATH");
    char *lang = getenv("LANG");

    if (home) printf("HOME: %s\n", home);
    if (path) printf("PATH: %s\n", path);
    if (lang) printf("LANG: %s\n", lang);

    // 设置环境变量
    setenv("MY_VAR", "hello", 1);   // 第3个参数为1表示覆盖
    printf("MY_VAR: %s\n", getenv("MY_VAR"));

    // 修改 PATH
    char new_path[4096];
    snprintf(new_path, sizeof(new_path), "%s:/usr/local/bin", path);
    setenv("PATH", new_path, 1);

    // 删除环境变量
    unsetenv("MY_VAR");

    // putenv (不推荐，可能有内存问题)
    // putenv("MY_VAR=world");
}

// 打印所有环境变量
void print_all_env(char *envp[])
{
    for (int i = 0; envp[i] != NULL; i++) {
        printf("%s\n", envp[i]);
    }
}

// 通过 extern 访问全局 environ
extern char **environ;

void print_environ(void)
{
    for (char **env = environ; *env != NULL; env++) {
        printf("%s\n", *env);
    }
}
```

## 4. C 程序的内存布局

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

/*
 * C 程序的典型内存布局:
 *
 * 高地址
 * +------------------+
 * |     命令行参数     |
 * +------------------+
 * |     环境变量       |
 * +------------------+
 * |       栈         |  <- 自动变量、函数参数（向下增长）
 * +------------------+
 * |       |          |
 * |     空闲区        |
 * |       |          |
 * +------------------+
 * |       堆         |  <- malloc/calloc/realloc（向上增长）
 * +------------------+
 * |  未初始化数据(BSS) |  <- 全局/静态未初始化变量
 * +------------------+
 * |  已初始化数据      |  <- 全局/静态已初始化变量
 * +------------------+
 * |     正文段        |  <- 机器指令（只读）
 * +------------------+
 * 低地址
 */

// 全局已初始化变量 - 存放在 data 段
int global_init = 42;

// 全局未初始化变量 - 存放在 BSS 段
int global_uninit;

// 静态变量 - 存放在 data/BSS 段
static int static_var = 100;
static int static_uninit;

void show_memory_layout(void)
{
    // 局部变量 - 存放在栈上
    int local_var = 10;
    int local_arr[100];

    // 堆上分配
    char *heap_ptr = malloc(1024);

    // 静态局部变量 - 存放在 data 段
    static int local_static = 200;

    printf("===== Memory Layout =====\n");
    printf("Code (function):     %p\n", (void *)show_memory_layout);
    printf("Data (global_init):  %p\n", (void *)&global_init);
    printf("Data (static_var):   %p\n", (void *)&static_var);
    printf("Data (local_static): %p\n", (void *)&local_static);
    printf("BSS (global_uninit): %p\n", (void *)&global_uninit);
    printf("BSS (static_uninit): %p\n", (void *)&static_uninit);
    printf("Heap:                %p\n", (void *)heap_ptr);
    printf("Stack (local_var):   %p\n", (void *)&local_var);
    printf("Stack (local_arr):   %p\n", (void *)local_arr);
    printf("========================\n");

    free(heap_ptr);
}

// 通过 /proc/self/maps 查看进程内存映射
void show_proc_maps(void)
{
    FILE *fp = fopen("/proc/self/maps", "r");
    if (!fp) {
        perror("fopen /proc/self/maps");
        return;
    }

    char line[256];
    printf("\n=== /proc/self/maps ===\n");
    while (fgets(line, sizeof(line), fp)) {
        printf("%s", line);
    }
    fclose(fp);
}
```

## 5. 内存分配

```c
#include <stdlib.h>
#include <string.h>
#include <stdio.h>

/*
 * void *malloc(size_t size);
 * void *calloc(size_t nmemb, size_t size);   - 分配并清零
 * void *realloc(void *ptr, size_t size);     - 调整大小
 * void free(void *ptr);
 *
 * void *alloca(size_t size);  - 在栈上分配（自动释放，不可移植）
 */

void memory_allocation_examples(void)
{
    // malloc
    int *arr = malloc(100 * sizeof(int));
    if (!arr) { perror("malloc"); return; }
    // malloc 不初始化内容
    free(arr);

    // calloc - 分配并清零
    int *arr2 = calloc(100, sizeof(int));
    if (!arr2) { perror("calloc"); return; }
    // calloc 初始化所有字节为 0
    free(arr2);

    // realloc - 调整大小
    char *buf = malloc(64);
    strcpy(buf, "Hello");
    buf = realloc(buf, 128);  // 扩大到 128 字节
    if (!buf) { perror("realloc"); return; }
    strcat(buf, ", World!");
    printf("Buffer: %s\n", buf);
    free(buf);

    // realloc(NULL, size) 等价于 malloc(size)
    int *p = NULL;
    p = realloc(p, 100 * sizeof(int));

    // realloc(ptr, 0) 等价于 free(ptr)（某些实现）
    // 不推荐依赖此行为
    free(p);
}
```

## 6. atexit 与 on_exit

```c
#include <stdlib.h>
#include <stdio.h>

/*
 * int atexit(void (*function)(void));
 * int on_exit(void (*function)(int, void *), void *arg);  // Linux 特有
 *
 * 注册的函数在 exit() 或 main 返回时被调用
 * 调用顺序：LIFO（后注册先执行）
 * 最多可注册的函数数量：atexit 最多 32 个
 */

static void cleanup_temp_files(void)
{
    printf("Cleaning up temporary files...\n");
    unlink("/tmp/myapp.tmp");
}

static void flush_logs(void)
{
    printf("Flushing log buffers...\n");
}

static void save_state(void)
{
    printf("Saving application state...\n");
}

void atexit_demo(void)
{
    // 注册退出处理函数
    atexit(cleanup_temp_files);
    atexit(flush_logs);
    atexit(save_state);

    printf("Application running...\n");

    // 退出时执行: save_state -> flush_logs -> cleanup_temp_files
    exit(0);
}
```

## 7. setjmp / longjmp（非局部跳转）

```c
#include <setjmp.h>
#include <stdio.h>

/*
 * int setjmp(jmp_buf env);
 * void longjmp(jmp_buf env, int val);
 *
 * setjmp 保存当前执行环境（寄存器、栈指针等）到 env
 * longjmp 恢复 env 中的环境，从对应的 setjmp 返回
 * setjmp 返回 0（直接调用）或 val（从 longjmp 返回）
 *
 * 注意：
 * - longjmp 可以跳过多个函数调用
 * - 可能导致资源泄漏（打开的文件、分配的内存）
 * - 信号处理中应该使用 sigsetjmp/siglongjmp
 */

static jmp_buf jump_buffer;

void deep_function(int level)
{
    printf("Entering level %d\n", level);
    if (level == 3) {
        printf("Jumping back!\n");
        longjmp(jump_buffer, 42);  // 跳回 setjmp 处
    }
    deep_function(level + 1);
}

void setjmp_demo(void)
{
    int val = setjmp(jump_buffer);

    if (val == 0) {
        // 第一次调用 setjmp 返回 0
        printf("setjmp returned 0, calling deep function\n");
        deep_function(1);
    } else {
        // longjmp 跳回这里，返回 longjmp 的第二个参数
        printf("Returned from longjmp with value: %d\n", val);
    }
}

// 实际应用：错误处理
static jmp_buf error_jmp;

int risky_operation(void)
{
    FILE *fp = NULL;
    int *data = NULL;

    if (setjmp(error_jmp) != 0) {
        // 错误处理：清理资源
        printf("Error occurred, cleaning up...\n");
        if (fp) fclose(fp);
        if (data) free(data);
        return -1;
    }

    // 执行可能出错的操作
    fp = fopen("data.bin", "rb");
    if (!fp) longjmp(error_jmp, 1);

    data = malloc(1024);
    if (!data) longjmp(error_jmp, 2);

    // 正常执行...
    printf("Operation completed successfully\n");
    fclose(fp);
    free(data);
    return 0;
}
```

## 8. 资源限制查询

```c
#include <sys/resource.h>
#include <stdio.h>

/*
 * int getrlimit(int resource, struct rlimit *rlim);
 * int setrlimit(int resource, const struct rlimit *rlim);
 * int getrusage(int who, struct rusage *usage);
 */

void show_resource_limits(void)
{
    struct rlimit rl;

    // 打开文件数限制
    getrlimit(RLIMIT_NOFILE, &rl);
    printf("Open files: soft=%ld, hard=%ld\n",
           (long)rl.rlim_cur, (long)rl.rlim_max);

    // 栈大小限制
    getrlimit(RLIMIT_STACK, &rl);
    printf("Stack size: soft=%ld, hard=%ld\n",
           (long)rl.rlim_cur, (long)rl.rlim_max);

    // 文件大小限制
    getrlimit(RLIMIT_FSIZE, &rl);
    printf("File size: soft=%ld, hard=%ld\n",
           (long)rl.rlim_cur, (long)rl.rlim_max);

    // 核心转储大小
    getrlimit(RLIMIT_CORE, &rl);
    printf("Core dump: soft=%ld, hard=%ld\n",
           (long)rl.rlim_cur, (long)rl.rlim_max);

    // 进程资源使用情况
    struct rusage usage;
    getrusage(RUSAGE_SELF, &usage);
    printf("User CPU time: %ld.%06ld sec\n",
           usage.ru_utime.tv_sec, usage.ru_utime.tv_usec);
    printf("System CPU time: %ld.%06ld sec\n",
           usage.ru_stime.tv_sec, usage.ru_stime.tv_usec);
    printf("Max RSS: %ld KB\n", usage.ru_maxrss);
}
```
