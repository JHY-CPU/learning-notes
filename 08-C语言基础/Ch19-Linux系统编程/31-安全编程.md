# 安全编程

## 1. 用户与权限管理

```c
#include <unistd.h>
#include <sys/types.h>
#include <pwd.h>
#include <grp.h>
#include <stdio.h>

/*
 * 用户标识:
 *   uid_t getuid(void);    - 实际用户 ID
 *   uid_t geteuid(void);   - 有效用户 ID
 *   gid_t getgid(void);    - 实际组 ID
 *   gid_t getegid(void);   - 有效组 ID
 *
 * 权限操作:
 *   int setuid(uid_t uid);
 *   int seteuid(uid_t euid);
 *   int setgid(gid_t gid);
 *   int setegid(gid_t egid);
 *   int setreuid(uid_t ruid, uid_t euid);
 *   int setregid(gid_t rgid, gid_t egid);
 */

void show_user_info(void)
{
    printf("UID:  %d\n", getuid());
    printf("EUID: %d\n", geteuid());
    printf("GID:  %d\n", getgid());
    printf("EGID: %d\n", getegid());

    // 获取用户名
    struct passwd *pw = getpwuid(getuid());
    if (pw) {
        printf("User: %s\n", pw->pw_name);
        printf("Home: %s\n", pw->pw_dir);
        printf("Shell: %s\n", pw->pw_shell);
    }

    // 获取组信息
    struct group *gr = getgrgid(getgid());
    if (gr) {
        printf("Group: %s\n", gr->gr_name);
    }
}

// 安全地放弃权限
int drop_privileges(uid_t target_uid, gid_t target_gid)
{
    // 先设置组 ID
    if (setgid(target_gid) == -1) {
        perror("setgid");
        return -1;
    }

    // 再设置用户 ID
    if (setuid(target_uid) == -1) {
        perror("setuid");
        return -1;
    }

    // 验证
    if (setuid(0) != -1) {
        fprintf(stderr, "ERROR: Still able to gain root!\n");
        return -1;
    }

    return 0;
}

// SUID 程序的安全执行
void safe_suid_exec(void)
{
    // 保存特权 UID
    uid_t saved_uid = geteuid();

    // 临时放弃特权
    if (seteuid(getuid()) == -1) {
        perror("seteuid");
        return;
    }

    // 需要特权时恢复
    if (seteuid(saved_uid) == -1) {
        perror("seteuid restore");
        return;
    }

    // 执行特权操作...

    // 完成后再次放弃
    seteuid(getuid());
}
```

## 2. chroot

```c
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/stat.h>

/*
 * int chroot(const char *path);
 * int chdir(const char *path);
 *
 * chroot 将指定目录作为新的根目录
 * 用于隔离进程的文件系统视图
 *
 * 注意：
 * 1. 需要 root 权限
 * 2. 必须在 chroot 后 chdir("/")
 * 3. /dev, /proc 等可能需要预先挂载
 */

int setup_chroot_jail(const char *jail_path)
{
    // 1. 创建必要的目录和设备
    char path[512];

    snprintf(path, sizeof(path), "%s/dev", jail_path);
    mkdir(path, 0755);

    snprintf(path, sizeof(path), "%s/tmp", jail_path);
    mkdir(path, 1777);

    // 2. chroot
    if (chroot(jail_path) == -1) {
        perror("chroot");
        return -1;
    }

    // 3. 切换到新根目录
    if (chdir("/") == -1) {
        perror("chdir");
        return -1;
    }

    printf("Chroot jail active at %s\n", jail_path);
    return 0;
}

// 更安全的 chroot 方案
int secure_chroot(const char *path)
{
    if (chroot(path) == -1) return -1;
    if (chdir("/") == -1) return -1;

    // 防止 chroot 逃逸
    // 确保不能通过 ".." 访问原始根
    // 常见做法：chroot 后进入一个深层子目录

    return 0;
}
```

## 3. Linux Capabilities

```c
#include <sys/capability.h>
#include <stdio.h>

/*
 * Capabilities 将 root 权限细分为多个独立的能力
 * 常用 capabilities:
 *   CAP_NET_BIND_SERVICE - 绑定 < 1024 端口
 *   CAP_NET_RAW          - 原始套接字
 *   CAP_SYS_ADMIN        - 系统管理操作
 *   CAP_DAC_OVERRIDE     - 绕过文件权限检查
 *   CAP_SETUID           - 改变 UID
 *   CAP_CHOWN            - 改变文件所有者
 *
 * 编译: gcc -o cap_demo cap_demo.c -lcap
 */

void show_capabilities(void)
{
    cap_t caps = cap_get_proc();
    if (!caps) {
        perror("cap_get_proc");
        return;
    }

    char *caps_str = cap_to_text(caps, NULL);
    printf("Capabilities: %s\n", caps_str);
    cap_free(caps_str);
    cap_free(caps);
}

// 只保留需要的 capabilities
int keep_only_bind_port(void)
{
    cap_t caps = cap_get_proc();

    // 清空所有 capabilities
    cap_clear(caps);

    // 只保留绑定低端口的权限
    cap_value_t cap_list[] = {CAP_NET_BIND_SERVICE};
    cap_set_flag(caps, CAP_EFFECTIVE, 1, cap_list, CAP_SET);
    cap_set_flag(caps, CAP_PERMITTED, 1, cap_list, CAP_SET);

    if (cap_set_proc(cups) == -1) {
        perror("cap_set_proc");
        cap_free(caps);
        return -1;
    }

    cap_free(caps);
    return 0;
}
```

## 4. 安全编程最佳实践

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <signal.h>
#include <sys/resource.h>

/*
 * 安全编程清单:
 * 1. 检查所有返回值
 * 2. 最小权限原则
 * 3. 不信任输入
 * 4. 安全处理字符串
 * 5. 正确处理临时文件
 * 6. 防止竞态条件
 */

// 1. 安全的字符串操作
void safe_string_ops(void)
{
    char buf[64];

    // 使用 snprintf 而非 sprintf
    snprintf(buf, sizeof(buf), "Hello %s", "World");

    // 使用 strncpy 而非 strcpy
    strncpy(buf, "test", sizeof(buf) - 1);
    buf[sizeof(buf) - 1] = '\0';

    // 使用 strncat
    strncat(buf, " suffix", sizeof(buf) - strlen(buf) - 1);
}

// 2. 安全的输入处理
void safe_input(void)
{
    char buf[256];

    // 不要使用 gets()
    // 错误: gets(buf);

    // 使用 fgets
    if (fgets(buf, sizeof(buf), stdin) != NULL) {
        // 移除换行符
        size_t len = strlen(buf);
        if (len > 0 && buf[len - 1] == '\n') {
            buf[len - 1] = '\0';
        }
    }
}

// 3. 安全的权限设置
void secure_file_creation(const char *path)
{
    // 使用 umask 确保安全的默认权限
    mode_t old_mask = umask(0077);  // 只有所有者可访问

    int fd = open(path, O_WRONLY | O_CREAT | O_EXCL, 0600);
    if (fd == -1) {
        perror("open");
        umask(old_mask);
        return;
    }

    umask(old_mask);  // 恢复 umask
    // ... 使用 fd ...
    close(fd);
}

// 4. 安全的临时文件
void secure_temp(void)
{
    char template[] = "/tmp/myapp_XXXXXX";
    int fd = mkstemp(template);
    if (fd == -1) {
        perror("mkstemp");
        return;
    }

    unlink(template);  // 立即删除，但 fd 仍可用
    // ... 使用 fd ...
    close(fd);
}

// 5. 清理敏感数据
void secure_cleanup(void *buf, size_t len)
{
    volatile char *p = buf;
    while (len--) *p++ = 0;
}

// 6. 防止 Core Dump 泄露敏感信息
void disable_core_dump(void)
{
    struct rlimit rl = {0, 0};
    setrlimit(RLIMIT_CORE, &rl);
}

// 7. 安全的环境变量处理
const char *safe_getenv(const char *name)
{
    const char *val = getenv(name);
    if (!val) {
        fprintf(stderr, "Environment variable %s not set\n", name);
        return NULL;
    }
    return val;
}
```
