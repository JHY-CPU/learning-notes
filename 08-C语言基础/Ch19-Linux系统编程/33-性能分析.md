# 性能分析

## 1. gprof - GNU Profiler

```bash
# 使用 gprof 进行性能分析

# 1. 编译时开启 profiling
gcc -pg -g -o program program.c

# 2. 运行程序，生成 gmon.out
./program

# 3. 分析结果
gprof ./program gmon.out

# 平面分析（flat profile）：每个函数的耗时
# 调用图（call graph）：函数调用关系和耗时

# 常用选项
gprof -b ./program gmon.out        # 简洁输出
gprof -p ./program gmon.out        # 只显示平面分析
gprof -q ./program gmon.out        # 只显示调用图
```

## 2. perf - Linux 性能计数器

```bash
# perf 基本使用

# 统计程序性能事件
perf stat ./program

# 详细的事件统计
perf stat -e cycles,instructions,cache-misses ./program

# 记录性能数据
perf record -g ./program

# 查看报告
perf report

# 火焰图生成
perf record -g ./program
perf script > out.perf
# 使用 FlameGraph 工具
# git clone https://github.com/brendangregg/FlameGraph
# ./stackcollapse-perf.pl out.perf | ./flamegraph.pl > flame.svg

# 实时监控
perf top
```

```c
// perf 可以分析的内容:
//
// CPU 周期 (cycles)
// 指令数 (instructions)
// 缓存命中/未命中 (cache-references, cache-misses)
// 分支预测 (branches, branch-misses)
// TLB 未命中
// 页面错误 (page-faults)
// 上下文切换 (context-switches)
```

## 3. C 程序内性能测量

```c
#include <stdio.h>
#include <time.h>
#include <sys/time.h>
#include <string.h>
#include <stdlib.h>

// 高精度计时器
typedef struct {
    struct timespec start;
    struct timespec end;
} Timer;

void timer_start(Timer *t)
{
    clock_gettime(CLOCK_MONOTONIC, &t->start);
}

double timer_stop(Timer *t)
{
    clock_gettime(CLOCK_MONOTONIC, &t->end);
    return (t->end.tv_sec - t->start.tv_sec) +
           (t->end.tv_nsec - t->start.tv_nsec) / 1e9;
}

// 性能测试框架
#define BENCHMARK(name, iterations, code) \
    do { \
        Timer _t; \
        timer_start(&_t); \
        for (int _i = 0; _i < (iterations); _i++) { \
            code; \
        } \
        double _elapsed = timer_stop(&_t); \
        printf("[BENCH] %s: %.6f sec (%d iterations, %.2f ns/op)\n", \
               name, _elapsed, (iterations), \
               _elapsed / (iterations) * 1e9); \
    } while (0)

// 内存带宽测试
void benchmark_memory(void)
{
    size_t size = 64 * 1024 * 1024;  // 64MB
    char *buf = malloc(size);

    // 顺序写入
    BENCHMARK("memset 64MB", 1, {
        memset(buf, 0, size);
    });

    // 顺序读取
    volatile char sink;
    BENCHMARK("read 64MB", 1, {
        for (size_t i = 0; i < size; i += 64) {
            sink = buf[i];
        }
    });

    free(buf);
}

// 缓存性能测试
void benchmark_cache(void)
{
    size_t sizes[] = {4096, 65536, 1048576, 16777216, 268435456};
    const char *labels[] = {"L1", "L2", "L3", "RAM", "Large"};

    for (int i = 0; i < 5; i++) {
        char *buf = malloc(sizes[i]);
        memset(buf, 1, sizes[i]);

        BENCHMARK(labels[i], 10, {
            volatile long sum = 0;
            for (size_t j = 0; j < sizes[i]; j += 64) {
                sum += buf[j];
            }
        });

        free(buf);
    }
}

// 函数调用开销测试
static inline int add_inline(int a, int b) { return a + b; }
int add_normal(int a, int b) { return a + b; }

void benchmark_call_overhead(void)
{
    volatile int result;

    BENCHMARK("inline call", 10000000, {
        result = add_inline(1, 2);
    });

    BENCHMARK("normal call", 10000000, {
        result = add_normal(1, 2);
    });
}
```

## 4. 系统性能监控

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/resource.h>

/*
 * 通过 /proc 和系统调用监控系统性能
 */

// 读取 /proc/stat 获取 CPU 使用
void show_cpu_usage(void)
{
    FILE *fp = fopen("/proc/stat", "r");
    if (!fp) return;

    char line[256];
    if (fgets(line, sizeof(line), fp)) {
        unsigned long user, nice, system, idle, iowait, irq, softirq;
        sscanf(line, "cpu %lu %lu %lu %lu %lu %lu %lu",
               &user, &nice, &system, &idle, &iowait, &irq, &softirq);

        unsigned long total = user + nice + system + idle + iowait + irq + softirq;
        unsigned long used = total - idle;

        printf("CPU Usage: %.1f%%\n", 100.0 * used / total);
    }
    fclose(fp);
}

// 读取 /proc/meminfo
void show_memory_usage(void)
{
    FILE *fp = fopen("/proc/meminfo", "r");
    if (!fp) return;

    char line[256];
    unsigned long total = 0, free = 0, available = 0;

    while (fgets(line, sizeof(line), fp)) {
        if (strncmp(line, "MemTotal:", 9) == 0)
            sscanf(line, "MemTotal: %lu kB", &total);
        else if (strncmp(line, "MemFree:", 8) == 0)
            sscanf(line, "MemFree: %lu kB", &free);
        else if (strncmp(line, "MemAvailable:", 13) == 0)
            sscanf(line, "MemAvailable: %lu kB", &available);
    }
    fclose(fp);

    printf("Memory: Total=%lu MB, Free=%lu MB, Available=%lu MB\n",
           total / 1024, free / 1024, available / 1024);
}

// 进程资源使用
void show_process_resources(void)
{
    struct rusage usage;
    getrusage(RUSAGE_SELF, &usage);

    printf("Process CPU: user=%ld.%06ld, sys=%ld.%06ld\n",
           usage.ru_utime.tv_sec, usage.ru_utime.tv_usec,
           usage.ru_stime.tv_sec, usage.ru_stime.tv_usec);
    printf("Max RSS: %ld KB\n", usage.ru_maxrss);
    printf("Context switches: voluntary=%ld, involuntary=%ld\n",
           usage.ru_nvcsw, usage.ru_nivcsw);
}
```
