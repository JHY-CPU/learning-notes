# 文件与目录操作

## 1. 文件元数据：stat 系列函数

```c
#include <sys/stat.h>
#include <stdio.h>
#include <time.h>

/*
 * stat/fstat/lstat - 获取文件元信息
 * stat  跟随符号链接
 * lstat 不跟随符号链接
 * fstat 通过文件描述符
 */

void print_file_info(const char *path)
{
    struct stat sb;

    if (lstat(path, &sb) == -1) {
        perror("lstat");
        return;
    }

    // 打印文件类型和权限
    printf("File: %s\n", path);

    printf("Type: ");
    switch (sb.st_mode & S_IFMT) {
    case S_IFREG:  printf("regular file\n"); break;
    case S_IFDIR:  printf("directory\n"); break;
    case S_IFLNK:  printf("symbolic link\n"); break;
    case S_IFCHR:  printf("character device\n"); break;
    case S_IFBLK:  printf("block device\n"); break;
    case S_IFIFO:  printf("FIFO/pipe\n"); break;
    case S_IFSOCK: printf("socket\n"); break;
    default:       printf("unknown\n"); break;
    }

    printf("Size: %lld bytes\n", (long long)sb.st_size);
    printf("Blocks: %lld\n", (long long)sb.st_blocks);
    printf("IO Block: %ld\n", (long)sb.st_blksize);
    printf("Inode: %lu\n", (unsigned long)sb.st_ino);
    printf("Links: %lu\n", (unsigned long)sb.st_nlink);
    printf("UID: %u, GID: %u\n", sb.st_uid, sb.st_gid);

    // 权限位
    printf("Access: %04o\n", sb.st_mode & 07777);

    // 时间
    printf("Access: %s", ctime(&sb.st_atime));
    printf("Modify: %s", ctime(&sb.st_mtime));
    printf("Change: %s", ctime(&sb.st_ctime));
}
```

## 2. 目录创建与删除

```c
#include <sys/stat.h>
#include <unistd.h>
#include <stdio.h>
#include <errno.h>
#include <string.h>

/*
 * int mkdir(const char *pathname, mode_t mode);
 * int rmdir(const char *pathname);
 *
 * mkdir：创建目录（mode 受 umask 影响）
 * rmdir：删除空目录
 */

void mkdir_examples(void)
{
    // 创建目录
    if (mkdir("mydir", 0755) == -1) {
        if (errno == EEXIST) {
            printf("Directory already exists\n");
        } else {
            perror("mkdir");
        }
    }

    // 递归创建目录（类似 mkdir -p）
    mkdir_p("/tmp/a/b/c", 0755);
}

// 递归创建目录
int mkdir_p(const char *path, mode_t mode)
{
    char tmp[512];
    char *p = NULL;

    snprintf(tmp, sizeof(tmp), "%s", path);

    for (p = tmp + 1; *p; p++) {
        if (*p == '/') {
            *p = '\0';
            if (mkdir(tmp, mode) == -1 && errno != EEXIST) {
                return -1;
            }
            *p = '/';
        }
    }

    if (mkdir(tmp, mode) == -1 && errno != EEXIST) {
        return -1;
    }

    return 0;
}
```

## 3. 链接操作

```c
#include <unistd.h>
#include <stdio.h>

/*
 * 硬链接：
 *   int link(const char *oldpath, const char *newpath);
 *   int unlink(const char *pathname);
 *
 * 符号链接：
 *   int symlink(const char *target, const char *linkpath);
 *   ssize_t readlink(const char *pathname, char *buf, size_t bufsiz);
 *
 * 重命名：
 *   int rename(const char *oldpath, const char *newpath);
 */

void link_examples(void)
{
    // 创建硬链接
    if (link("original.txt", "hardlink.txt") == -1) {
        perror("link");
    }

    // 创建符号链接
    if (symlink("original.txt", "symlink.txt") == -1) {
        perror("symlink");
    }

    // 读取符号链接目标
    char target[256];
    ssize_t len = readlink("symlink.txt", target, sizeof(target) - 1);
    if (len != -1) {
        target[len] = '\0';
        printf("symlink.txt -> %s\n", target);
    }

    // 删除文件（减少链接计数）
    if (unlink("hardlink.txt") == -1) {
        perror("unlink");
    }

    // 重命名文件
    if (rename("old_name.txt", "new_name.txt") == -1) {
        perror("rename");
    }
}
```

## 4. 目录遍历

```c
#include <dirent.h>
#include <stdio.h>
#include <string.h>
#include <sys/stat.h>

/*
 * DIR *opendir(const char *name);
 * struct dirent *readdir(DIR *dirp);
 * int closedir(DIR *dirp);
 * void rewinddir(DIR *dirp);
 * int dirfd(DIR *dirp);
 *
 * struct dirent {
 *     ino_t          d_ino;
 *     off_t          d_off;
 *     unsigned short d_reclen;
 *     unsigned char  d_type;   - DT_REG, DT_DIR, DT_LNK, ...
 *     char           d_name[]; - 文件名
 * };
 */

void list_directory(const char *path)
{
    DIR *dir;
    struct dirent *entry;
    struct stat st;

    dir = opendir(path);
    if (dir == NULL) {
        perror("opendir");
        return;
    }

    printf("Contents of %s:\n", path);
    printf("%-20s %10s %s\n", "Name", "Size", "Type");
    printf("---- ---- ---- ---- ----\n");

    while ((entry = readdir(dir)) != NULL) {
        char fullpath[512];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        if (lstat(fullpath, &st) == -1) {
            perror("lstat");
            continue;
        }

        // 跳过 . 和 ..
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) {
            continue;
        }

        printf("%-20s %10ld ", entry->d_name, (long)st.st_size);

        if (S_ISDIR(st.st_mode))       printf("[DIR]\n");
        else if (S_ISREG(st.st_mode))  printf("[FILE]\n");
        else if (S_ISLNK(st.st_mode))  printf("[LINK]\n");
        else                           printf("[OTHER]\n");
    }

    closedir(dir);
}

// 递归遍历目录树
void walk_directory(const char *path, int depth)
{
    DIR *dir;
    struct dirent *entry;
    struct stat st;

    dir = opendir(path);
    if (!dir) return;

    while ((entry = readdir(dir)) != NULL) {
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) {
            continue;
        }

        char fullpath[512];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        if (lstat(fullpath, &st) == -1) continue;

        // 缩进打印
        for (int i = 0; i < depth; i++) printf("  ");
        printf("%s\n", entry->d_name);

        if (S_ISDIR(st.st_mode)) {
            walk_directory(fullpath, depth + 1);  // 递归子目录
        }
    }

    closedir(dir);
}
```

## 5. nftw - 文件树遍历

```c
#include <ftw.h>
#include <stdio.h>

/*
 * int nftw(const char *dirpath,
 *          int (*fn)(const char *fpath, const struct stat *sb,
 *                    int typeflag, struct FTW *ftwbuf),
 *          int nopenfd, int flags);
 *
 * typeflag: FTW_F(文件), FTW_D(目录), FTW_SL(符号链接), ...
 * flags: FTW_PHYS(不跟随符号链接), FTW_DEPTH(后序遍历)
 */

// 打印所有文件的大小
int print_file_info(const char *fpath, const struct stat *sb,
                    int typeflag, struct FTW *ftwbuf)
{
    // 缩进显示目录层级
    for (int i = 0; i < ftwbuf->level; i++) printf("  ");

    if (typeflag == FTW_F) {
        printf("%s (%ld bytes)\n", fpath, (long)sb->st_size);
    } else if (typeflag == FTW_D) {
        printf("%s/\n", fpath);
    } else if (typeflag == FTW_SL) {
        printf("%s ->\n", fpath);
    }

    return 0;  // 返回非零值可终止遍历
}

void nftw_example(void)
{
    // 遍历当前目录，最多打开 20 个文件描述符
    nftw(".", print_file_info, 20, FTW_PHYS);
}
```

## 6. 临时文件

```c
#include <stdlib.h>
#include <stdio.h>
#include <unistd.h>
#include <string.h>

/*
 * char *mkdtemp(char *template);   - 创建临时目录
 * int mkstemp(char *template);      - 创建临时文件（返回 fd）
 * char *tmpnam(char *s);            - 不安全，不要使用
 * FILE *tmpfile(void);              - 创建临时文件（自动删除）
 */

void temp_file_examples(void)
{
    // 方式1：tmpfile - 创建匿名临时文件
    FILE *fp = tmpfile();
    if (fp) {
        fprintf(fp, "Temporary data\n");
        rewind(fp);
        char buf[256];
        if (fgets(buf, sizeof(buf), fp)) {
            printf("Read from tmpfile: %s", buf);
        }
        fclose(fp);  // 关闭时自动删除
    }

    // 方式2：mkstemp - 安全创建临时文件
    char template[] = "/tmp/myapp_XXXXXX";
    int fd = mkstemp(template);
    if (fd == -1) {
        perror("mkstemp");
        return;
    }
    printf("Temp file: %s\n", template);  // XXXXXX 被替换为唯一后缀

    write(fd, "Temp data\n", 10);
    close(fd);

    // 用完后手动删除
    unlink(template);

    // 方式3：mkdtemp - 创建临时目录
    char dir_template[] = "/tmp/myapp_dir_XXXXXX";
    char *tmpdir = mkdtemp(dir_template);
    if (tmpdir) {
        printf("Temp directory: %s\n", tmpdir);
        // 使用完毕后
        rmdir(tmpdir);
    }
}
```

## 7. 综合示例：目录空间统计

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <dirent.h>
#include <sys/stat.h>
#include <unistd.h>

/*
 * 统计指定目录的总大小和文件数量
 * 演示目录遍历、文件信息获取
 */

typedef struct {
    unsigned long long total_size;
    unsigned long num_files;
    unsigned long num_dirs;
    unsigned long num_symlinks;
} DirStats;

int scan_directory(const char *path, DirStats *stats)
{
    DIR *dir;
    struct dirent *entry;
    struct stat st;
    char fullpath[1024];

    dir = opendir(path);
    if (!dir) {
        perror(path);
        return -1;
    }

    while ((entry = readdir(dir)) != NULL) {
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) {
            continue;
        }

        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        if (lstat(fullpath, &st) == -1) continue;

        if (S_ISDIR(st.st_mode)) {
            stats->num_dirs++;
            scan_directory(fullpath, stats);  // 递归
        } else if (S_ISREG(st.st_mode)) {
            stats->num_files++;
            stats->total_size += st.st_size;
        } else if (S_ISLNK(st.st_mode)) {
            stats->num_symlinks++;
        }
    }

    closedir(dir);
    return 0;
}

int main(int argc, char *argv[])
{
    const char *path = (argc > 1) ? argv[1] : ".";

    DirStats stats = {0};
    scan_directory(path, &stats);

    printf("Directory: %s\n", path);
    printf("Total size: %llu bytes (%.2f MB)\n",
           stats.total_size, stats.total_size / 1024.0 / 1024.0);
    printf("Files: %lu\n", stats.num_files);
    printf("Directories: %lu\n", stats.num_dirs);
    printf("Symlinks: %lu\n", stats.num_symlinks);

    return 0;
}
```
