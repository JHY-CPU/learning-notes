# 进程间通信 - 管道

## 1. 管道基础

```c
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <sys/wait.h>

/*
 * int pipe(int pipefd[2]);
 *
 * pipefd[0] - 读端
 * pipefd[1] - 写端
 *
 * 管道特点:
 *   - 半双工（只能单向通信）
 *   - 只能在有亲缘关系的进程间使用
 *   - 读写是阻塞的
 *   - 写满管道时 write 阻塞
 *   - 读空管道时 read 阻塞
 *   - 所有写端关闭后 read 返回 0
 *   - 所有读端关闭后 write 产生 SIGPIPE
 */

// 基本管道示例：父子进程通信
void basic_pipe_example(void)
{
    int pipefd[2];
    pid_t pid;
    char buf[256];

    if (pipe(pipefd) == -1) {
        perror("pipe");
        return;
    }

    pid = fork();
    if (pid == 0) {
        // 子进程：关闭读端，通过写端发送数据
        close(pipefd[0]);

        const char *msg = "Hello from child!\n";
        write(pipefd[1], msg, strlen(msg));
        close(pipefd[1]);
        _exit(0);
    } else {
        // 父进程：关闭写端，通过读端接收数据
        close(pipefd[1]);

        ssize_t n = read(pipefd[0], buf, sizeof(buf) - 1);
        if (n > 0) {
            buf[n] = '\0';
            printf("Parent received: %s", buf);
        }
        close(pipefd[0]);
        wait(NULL);
    }
}

// 双向通信：使用两个管道
void bidirectional_pipe(void)
{
    int pipe1[2], pipe2[2];  // 父->子, 子->父
    pid_t pid;
    char buf[256];

    pipe(pipe1);
    pipe(pipe2);

    pid = fork();
    if (pid == 0) {
        // 子进程
        close(pipe1[1]);  // 关闭 pipe1 写端
        close(pipe2[0]);  // 关闭 pipe2 读端

        // 从父进程读取
        ssize_t n = read(pipe1[0], buf, sizeof(buf) - 1);
        if (n > 0) {
            buf[n] = '\0';
            printf("Child received: %s", buf);
        }

        // 向父进程回复
        const char *reply = "Hello from child!";
        write(pipe2[1], reply, strlen(reply));

        close(pipe1[0]);
        close(pipe2[1]);
        _exit(0);
    } else {
        // 父进程
        close(pipe1[0]);  // 关闭 pipe1 读端
        close(pipe2[1]);  // 关闭 pipe2 写端

        // 向子进程发送
        const char *msg = "Hello from parent!";
        write(pipe1[1], msg, strlen(msg));
        close(pipe1[1]);  // 关闭写端，子进程 read 返回 0

        // 从子进程读取回复
        ssize_t n = read(pipe2[0], buf, sizeof(buf) - 1);
        if (n > 0) {
            buf[n] = '\0';
            printf("Parent received: %s\n", buf);
        }

        close(pipe2[0]);
        wait(NULL);
    }
}
```

## 2. 管道的使用场景

```c
#include <unistd.h>
#include <stdio.h>
#include <sys/wait.h>

// 重定向子进程的 stdout 到管道
void redirect_stdout_to_pipe(void)
{
    int pipefd[2];
    pid_t pid;
    char buf[4096];

    pipe(pipefd);
    pid = fork();

    if (pid == 0) {
        // 子进程
        close(pipefd[0]);

        // 将 stdout 重定向到管道写端
        dup2(pipefd[1], STDOUT_FILENO);
        close(pipefd[1]);

        // 执行 ls 命令，输出通过管道传递给父进程
        execlp("ls", "ls", "-la", NULL);
        perror("execlp");
        _exit(1);
    } else {
        // 父进程
        close(pipefd[1]);

        // 读取子进程的输出
        ssize_t n;
        while ((n = read(pipefd[0], buf, sizeof(buf))) > 0) {
            write(STDOUT_FILENO, buf, n);
        }

        close(pipefd[0]);
        wait(NULL);
    }
}

// 管道连接多个进程（类似 shell 管道）
// 实现 ls | sort | uniq -c
void pipeline_example(void)
{
    int pipe1[2], pipe2[2];
    pid_t pid1, pid2, pid3;

    pipe(pipe1);  // ls -> sort
    pipe(pipe2);  // sort -> uniq

    // 进程1: ls -la
    pid1 = fork();
    if (pid1 == 0) {
        close(pipe1[0]);
        close(pipe2[0]);
        close(pipe2[1]);

        dup2(pipe1[1], STDOUT_FILENO);
        close(pipe1[1]);

        execlp("ls", "ls", "-la", NULL);
        _exit(1);
    }

    // 进程2: sort
    pid2 = fork();
    if (pid2 == 0) {
        close(pipe1[1]);
        close(pipe2[0]);

        dup2(pipe1[0], STDIN_FILENO);
        dup2(pipe2[1], STDOUT_FILENO);
        close(pipe1[0]);
        close(pipe2[1]);

        execlp("sort", "sort", NULL);
        _exit(1);
    }

    // 进程3: uniq -c
    pid3 = fork();
    if (pid3 == 0) {
        close(pipe1[0]);
        close(pipe1[1]);
        close(pipe2[1]);

        dup2(pipe2[0], STDIN_FILENO);
        close(pipe2[0]);

        execlp("uniq", "uniq", "-c", NULL);
        _exit(1);
    }

    // 父进程：关闭所有管道端
    close(pipe1[0]);
    close(pipe1[1]);
    close(pipe2[0]);
    close(pipe2[1]);

    waitpid(pid1, NULL, 0);
    waitpid(pid2, NULL, 0);
    waitpid(pid3, NULL, 0);
}
```

## 3. FIFO（命名管道）

```c
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <errno.h>

/*
 * int mkfifo(const char *pathname, mode_t mode);
 *
 * FIFO 与普通管道的区别:
 *   - 有文件名，无亲缘关系的进程也可使用
 *   - 数据仍然先进先出
 *   - 需要手动删除（unlink）
 */

// 创建 FIFO
int create_fifo(const char *path)
{
    if (mkfifo(path, 0666) == -1) {
        if (errno != EEXIST) {
            perror("mkfifo");
            return -1;
        }
    }
    return 0;
}

// FIFO 写入端
void fifo_writer(const char *fifo_path)
{
    int fd;
    const char *messages[] = {
        "Message 1\n",
        "Message 2\n",
        "Message 3\n",
        NULL
    };

    printf("Writer: opening FIFO (blocking until reader connects)...\n");
    fd = open(fifo_path, O_WRONLY);
    if (fd == -1) { perror("open writer"); return; }

    printf("Writer: connected!\n");

    for (int i = 0; messages[i]; i++) {
        write(fd, messages[i], strlen(messages[i]));
        printf("Writer: sent %s", messages[i]);
        sleep(1);
    }

    close(fd);
    printf("Writer: done\n");
}

// FIFO 读取端
void fifo_reader(const char *fifo_path)
{
    int fd;
    char buf[256];
    ssize_t n;

    printf("Reader: opening FIFO (blocking until writer connects)...\n");
    fd = open(fifo_path, O_RDONLY);
    if (fd == -1) { perror("open reader"); return; }

    printf("Reader: connected!\n");

    while ((n = read(fd, buf, sizeof(buf) - 1)) > 0) {
        buf[n] = '\0';
        printf("Reader: received %s", buf);
    }

    close(fd);
    printf("Reader: done\n");
}

// FIFO 双向通信示例
void fifo_bidirectional(void)
{
    const char *fifo1 = "/tmp/fifo_a2b";
    const char *fifo2 = "/tmp/fifo_b2a";

    mkfifo(fifo1, 0666);
    mkfifo(fifo2, 0666);

    pid_t pid = fork();
    if (pid == 0) {
        // 进程 B：从 fifo1 读，向 fifo2 写
        int fd_r = open(fifo1, O_RDONLY);
        int fd_w = open(fifo2, O_WRONLY);

        char buf[256];
        ssize_t n = read(fd_r, buf, sizeof(buf));
        if (n > 0) {
            buf[n] = '\0';
            printf("Process B received: %s", buf);
        }

        const char *reply = "Hello from B\n";
        write(fd_w, reply, strlen(reply));

        close(fd_r);
        close(fd_w);
        _exit(0);
    } else {
        // 进程 A：向 fifo1 写，从 fifo2 读
        int fd_w = open(fifo1, O_WRONLY);
        int fd_r = open(fifo2, O_RDONLY);

        const char *msg = "Hello from A\n";
        write(fd_w, msg, strlen(msg));
        close(fd_w);

        char buf[256];
        ssize_t n = read(fd_r, buf, sizeof(buf));
        if (n > 0) {
            buf[n] = '\0';
            printf("Process A received: %s", buf);
        }

        close(fd_r);
        wait(NULL);

        unlink(fifo1);
        unlink(fifo2);
    }
}
```

## 4. popen / pclose

```c
#include <stdio.h>
#include <string.h>

/*
 * FILE *popen(const char *command, const char *mode);
 * int pclose(FILE *stream);
 *
 * mode:
 *   "r" - 读取命令的输出
 *   "w" - 向命令写入
 *
 * popen 内部创建管道，fork 子进程执行 shell 命令
 */

// 读取命令输出
void popen_read_example(void)
{
    FILE *fp;
    char buf[256];

    fp = popen("ls -la /tmp", "r");
    if (!fp) { perror("popen"); return; }

    while (fgets(buf, sizeof(buf), fp) != NULL) {
        printf("%s", buf);
    }

    int status = pclose(fp);
    printf("Command exit status: %d\n", WEXITSTATUS(status));
}

// 向命令写入
void popen_write_example(void)
{
    FILE *fp;

    fp = popen("sort", "w");
    if (!fp) { perror("popen"); return; }

    fprintf(fp, "banana\n");
    fprintf(fp, "apple\n");
    fprintf(fp, "cherry\n");

    pclose(fp);  // 关闭管道，sort 输出排序结果
}

// 获取命令输出到字符串
char *capture_command(const char *cmd)
{
    FILE *fp = popen(cmd, "r");
    if (!fp) return NULL;

    char *result = NULL;
    size_t total = 0;
    char buf[1024];
    size_t n;

    while ((n = fread(buf, 1, sizeof(buf), fp)) > 0) {
        result = realloc(result, total + n + 1);
        memcpy(result + total, buf, n);
        total += n;
    }

    if (result) {
        result[total] = '\0';
    }

    pclose(fp);
    return result;  // 调用者负责 free
}

void capture_example(void)
{
    char *output = capture_command("uname -a");
    if (output) {
        printf("System: %s", output);
        free(output);
    }
}
```
