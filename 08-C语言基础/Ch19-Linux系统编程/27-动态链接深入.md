# 动态链接深入

## 1. dlopen/dlsym 动态加载

```c
#include <dlfcn.h>
#include <stdio.h>

/*
 * 动态加载共享库 API:
 *   void *dlopen(const char *filename, int flags);
 *   void *dlsym(void *handle, const char *symbol);
 *   int dlclose(void *handle);
 *   char *dlerror(void);
 *
 * flags:
 *   RTLD_LAZY   - 延迟绑定（默认）
 *   RTLD_NOW    - 立即绑定
 *   RTLD_GLOBAL - 符号全局可见
 *   RTLD_LOCAL  - 符号不可被其他库解析（默认）
 *
 * 编译: gcc -o demo demo.c -ldl
 */

void dlopen_example(void)
{
    // 打开共享库
    void *handle = dlopen("libm.so.6", RTLD_LAZY);
    if (!handle) {
        fprintf(stderr, "dlopen: %s\n", dlerror());
        return;
    }

    // 获取函数指针
    double (*cosine)(double) = dlsym(handle, "cos");
    char *error = dlerror();
    if (error) {
        fprintf(stderr, "dlsym: %s\n", error);
        dlclose(handle);
        return;
    }

    // 调用函数
    printf("cos(2.0) = %f\n", cosine(2.0));

    dlclose(handle);
}

// 通用动态加载封装
typedef struct {
    void *handle;
} DLib;

DLib *dlib_open(const char *path)
{
    DLib *lib = malloc(sizeof(DLib));
    if (!lib) return NULL;

    lib->handle = dlopen(path, RTLD_LAZY);
    if (!lib->handle) {
        fprintf(stderr, "dlopen(%s): %s\n", path, dlerror());
        free(lib);
        return NULL;
    }
    return lib;
}

void *dlib_sym(DLib *lib, const char *name)
{
    dlerror();  // 清除之前的错误
    void *sym = dlsym(lib->handle, name);
    char *err = dlerror();
    if (err) {
        fprintf(stderr, "dlsym(%s): %s\n", name, err);
        return NULL;
    }
    return sym;
}

void dlib_close(DLib *lib)
{
    if (lib) {
        dlclose(lib->handle);
        free(lib);
    }
}
```

## 2. 插件架构

```c
#include <dlfcn.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/*
 * 使用 dlopen 实现插件架构
 * 插件实现统一接口，主程序动态加载
 */

// 插件接口
typedef struct {
    const char *name;
    int (*init)(void);
    int (*process)(const char *input, char *output, size_t size);
    void (*cleanup)(void);
} Plugin;

// 插件管理器
#define MAX_PLUGINS 32

typedef struct {
    Plugin *plugins[MAX_PLUGINS];
    void *handles[MAX_PLUGINS];
    int count;
} PluginManager;

PluginManager *plugin_manager_new(void)
{
    return calloc(1, sizeof(PluginManager));
}

// 加载插件
int plugin_load(PluginManager *mgr, const char *path)
{
    if (mgr->count >= MAX_PLUGINS) {
        fprintf(stderr, "Too many plugins\n");
        return -1;
    }

    void *handle = dlopen(path, RTLD_LAZY);
    if (!handle) {
        fprintf(stderr, "Cannot load %s: %s\n", path, dlerror());
        return -1;
    }

    // 查找插件接口符号
    Plugin *plugin = dlsym(handle, "plugin_interface");
    if (!plugin) {
        fprintf(stderr, "No plugin_interface in %s: %s\n", path, dlerror());
        dlclose(handle);
        return -1;
    }

    // 初始化插件
    if (plugin->init && plugin->init() != 0) {
        fprintf(stderr, "Plugin %s init failed\n", plugin->name);
        dlclose(handle);
        return -1;
    }

    mgr->plugins[mgr->count] = plugin;
    mgr->handles[mgr->count] = handle;
    mgr->count++;

    printf("Loaded plugin: %s\n", plugin->name);
    return 0;
}

// 卸载所有插件
void plugin_manager_free(PluginManager *mgr)
{
    for (int i = 0; i < mgr->count; i++) {
        if (mgr->plugins[i]->cleanup) {
            mgr->plugins[i]->cleanup();
        }
        dlclose(mgr->handles[i]);
    }
    free(mgr);
}

// 处理请求
int plugin_process(PluginManager *mgr, const char *name,
                   const char *input, char *output, size_t size)
{
    for (int i = 0; i < mgr->count; i++) {
        if (strcmp(mgr->plugins[i]->name, name) == 0) {
            return mgr->plugins[i]->process(input, output, size);
        }
    }
    return -1;
}
```

## 3. LD_PRELOAD 与符号拦截

```c
/*
 * LD_PRELOAD 机制：
 * 在程序加载时预加载指定的共享库
 * 可以拦截和替换标准库函数
 *
 * 使用方法:
 *   LD_PRELOAD=./myhook.so ./program
 */

// hook.c - 拦截 malloc/free
#define _GNU_SOURCE
#include <stdio.h>
#include <stdlib.h>
#include <dlfcn.h>

// 原始函数指针
static void *(*real_malloc)(size_t) = NULL;
static void (*real_free)(void *) = NULL;

// 确保初始化
static void init(void)
{
    if (!real_malloc) {
        real_malloc = dlsym(RTLD_NEXT, "malloc");
        real_free = dlsym(RTLD_NEXT, "free");
    }
}

// 拦截 malloc
void *malloc(size_t size)
{
    init();
    void *ptr = real_malloc(size);
    fprintf(stderr, "malloc(%zu) = %p\n", size, ptr);
    return ptr;
}

// 拦截 free
void free(void *ptr)
{
    init();
    fprintf(stderr, "free(%p)\n", ptr);
    real_free(ptr);
}

// 编译:
// gcc -shared -fPIC -o hook.so hook.c -ldl
// LD_PRELOAD=./hook.so ls

// 拦截 open 的例子
#include <fcntl.h>
#include <stdarg.h>

static int (*real_open)(const char *, int, ...) = NULL;

int open(const char *pathname, int flags, ...)
{
    if (!real_open) {
        real_open = dlsym(RTLD_NEXT, "open");
    }

    fprintf(stderr, "open(%s, %d)\n", pathname, flags);

    if (flags & O_CREAT) {
        va_list args;
        va_start(args, flags);
        mode_t mode = va_arg(args, mode_t);
        va_end(args);
        return real_open(pathname, flags, mode);
    }

    return real_open(pathname, flags);
}
```

## 4. 查看动态链接信息

```c
#include <link.h>
#include <stdio.h>

/*
 * 查看程序加载的共享库
 * 使用 dl_iterate_phdr 遍历所有加载的共享对象
 */

int callback(struct dl_phdr_info *info, size_t size, void *data)
{
    (void)size;
    (void)data;

    printf("Library: %s\n", info->dlpi_name);
    printf("  Base address: %p\n", (void *)info->dlpi_addr);

    for (int i = 0; i < info->dlpi_phnum; i++) {
        if (info->dlpi_phdr[i].p_type == PT_LOAD) {
            printf("  Segment %d: vaddr=%p, memsz=%zu\n",
                   i,
                   (void *)(info->dlpi_addr + info->dlpi_phdr[i].p_vaddr),
                   (size_t)info->dlpi_phdr[i].p_memsz);
        }
    }

    return 0;
}

void show_loaded_libraries(void)
{
    dl_iterate_phdr(callback, NULL);
}
```
