# 高级进程间通信

## 1. Unix 域 Socket 传递文件描述符

```c
#include <sys/socket.h>
#include <sys/un.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/wait.h>

/*
 * 使用 Unix 域 Socket 在进程间传递文件描述符
 * 使用 sendmsg/recvmsg + SCM_RIGHTS 辅助数据
 *
 * struct msghdr {
 *     void         *msg_name;
 *     socklen_t     msg_namelen;
 *     struct iovec *msg_iov;
 *     size_t        msg_iovlen;
 *     void         *msg_control;      - 辅助数据缓冲区
 *     size_t        msg_controllen;   - 辅助数据长度
 *     int           msg_flags;
 * };
 */

// 发送文件描述符
int send_fd(int sock_fd, int fd_to_send)
{
    struct msghdr msg;
    struct iovec iov;
    char buf[1];  // 需要至少发送 1 字节数据

    buf[0] = 0;
    iov.iov_base = buf;
    iov.iov_len  = 1;

    msg.msg_name    = NULL;
    msg.msg_namelen = 0;
    msg.msg_iov     = &iov;
    msg.msg_iovlen  = 1;

    // 辅助数据：包含要传递的 fd
    char control[CMSG_SPACE(sizeof(int))];
    msg.msg_control    = control;
    msg.msg_controllen = sizeof(control);

    struct cmsghdr *cmsg = CMSG_FIRSTHDR(&msg);
    cmsg->cmsg_level = SOL_SOCKET;
    cmsg->cmsg_type  = SCM_RIGHTS;
    cmsg->cmsg_len   = CMSG_LEN(sizeof(int));
    memcpy(CMSG_DATA(cmsg), &fd_to_send, sizeof(int));

    return sendmsg(sock_fd, &msg, 0);
}

// 接收文件描述符
int recv_fd(int sock_fd)
{
    struct msghdr msg;
    struct iovec iov;
    char buf[1];

    iov.iov_base = buf;
    iov.iov_len  = 1;

    msg.msg_name    = NULL;
    msg.msg_namelen = 0;
    msg.msg_iov     = &iov;
    msg.msg_iovlen  = 1;

    char control[CMSG_SPACE(sizeof(int))];
    msg.msg_control    = control;
    msg.msg_controllen = sizeof(control);

    ssize_t n = recvmsg(sock_fd, &msg, 0);
    if (n <= 0) return -1;

    struct cmsghdr *cmsg = CMSG_FIRSTHDR(&msg);
    if (cmsg && cmsg->cmsg_type == SCM_RIGHTS) {
        int fd;
        memcpy(&fd, CMSG_DATA(cmsg), sizeof(int));
        return fd;
    }

    return -1;
}

// 示例：父进程打开文件，传递给子进程
void pass_fd_example(void)
{
    int sv[2];  // Unix 域 socket 对
    socketpair(AF_UNIX, SOCK_STREAM, 0, sv);

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程：接收文件描述符
        close(sv[0]);

        int fd = recv_fd(sv[1]);
        if (fd >= 0) {
            // 使用传来的文件描述符
            char buf[256];
            ssize_t n = read(fd, buf, sizeof(buf) - 1);
            if (n > 0) {
                buf[n] = '\0';
                printf("Child read from passed fd: %s", buf);
            }
            close(fd);
        }

        close(sv[1]);
        _exit(0);
    } else {
        // 父进程：打开文件并传递
        close(sv[1]);

        int fd = open("/etc/hostname", O_RDONLY);
        if (fd >= 0) {
            send_fd(sv[0], fd);
            close(fd);  // 子进程有自己的 fd 副本
        }

        close(sv[0]);
        wait(NULL);
    }
}
```

## 2. 传递凭证

```c
#include <sys/socket.h>
#include <sys/un.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * 使用 SCM_CREDENTIALS 在 Unix 域 Socket 上传递进程凭证
 * 需要 SO_PASSCRED socket 选项
 */

struct ucred_msg {
    struct cmsghdr cmsg;
    struct ucred cred;
};

int send_credentials(int sock_fd)
{
    struct msghdr msg;
    struct iovec iov;
    char buf[1] = {0};

    iov.iov_base = buf;
    iov.iov_len  = 1;

    msg.msg_name    = NULL;
    msg.msg_namelen = 0;
    msg.msg_iov     = &iov;
    msg.msg_iovlen  = 1;

    struct ucred_msg ucm;
    msg.msg_control    = &ucm;
    msg.msg_controllen = sizeof(ucm);

    ucm.cmsg.cmsg_level = SOL_SOCKET;
    ucm.cmsg.cmsg_type  = SCM_CREDENTIALS;
    ucm.cmsg.cmsg_len   = CMSG_LEN(sizeof(struct ucred));

    ucm.cred.pid = getpid();
    ucm.cred.uid = getuid();
    ucm.cred.gid = getgid();

    return sendmsg(sock_fd, &msg, 0);
}

int recv_credentials(int sock_fd, struct ucred *cred)
{
    struct msghdr msg;
    struct iovec iov;
    char buf[1];

    iov.iov_base = buf;
    iov.iov_len  = 1;

    msg.msg_name    = NULL;
    msg.msg_namelen = 0;
    msg.msg_iov     = &iov;
    msg.msg_iovlen  = 1;

    struct ucred_msg ucm;
    msg.msg_control    = &ucm;
    msg.msg_controllen = sizeof(ucm);

    ssize_t n = recvmsg(sock_fd, &msg, 0);
    if (n <= 0) return -1;

    struct cmsghdr *cmsg = CMSG_FIRSTHDR(&msg);
    if (cmsg && cmsg->cmsg_type == SCM_CREDENTIALS) {
        memcpy(cred, CMSG_DATA(cmsg), sizeof(struct ucred));
        return 0;
    }

    return -1;
}
```

## 3. socketpair

```c
#include <sys/socket.h>
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * int socketpair(int domain, int type, int protocol, int sv[2]);
 *
 * 创建一对已连接的 socket
 * 类似于 pipe，但支持双向通信
 */

void socketpair_example(void)
{
    int sv[2];

    if (socketpair(AF_UNIX, SOCK_STREAM, 0, sv) == -1) {
        perror("socketpair");
        return;
    }

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程
        close(sv[0]);

        const char *msg = "Hello from child";
        write(sv[1], msg, strlen(msg));

        char buf[256];
        ssize_t n = read(sv[1], buf, sizeof(buf));
        if (n > 0) {
            buf[n] = '\0';
            printf("Child got: %s\n", buf);
        }

        close(sv[1]);
        _exit(0);
    } else {
        // 父进程
        close(sv[1]);

        char buf[256];
        ssize_t n = read(sv[0], buf, sizeof(buf));
        if (n > 0) {
            buf[n] = '\0';
            printf("Parent got: %s\n", buf);
        }

        const char *reply = "Hello from parent";
        write(sv[0], reply, strlen(reply));

        close(sv[0]);
        wait(NULL);
    }
}
```
