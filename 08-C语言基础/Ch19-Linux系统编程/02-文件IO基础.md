# 文件IO基础

## 1. 文件描述符与文件表

内核为每个进程维护一张**文件描述符表**，每个条目指向系统级的**文件表项**，文件表项再指向**inode表项**。

```
进程文件描述符表        系统文件表          inode表
[0] stdin  ----->  文件表项1 -----> inode A
[1] stdout ----->  文件表项2 -----> inode A
[2] stderr ----->  文件表项3 -----> inode B
[3]         ----->  文件表项4 -----> inode C
```

```c
#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>

// 每个进程最多可打开的文件数由 sysconf(_SC_OPEN_MAX) 决定
// 通常默认为 1024，可通过 ulimit -n 或 setrlimit 调整
```

## 2. open 系统调用

```c
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>

/*
 * int open(const char *pathname, int flags);
 * int open(const char *pathname, int flags, mode_t mode);
 *
 * flags 必选其一:
 *   O_RDONLY   - 只读
 *   O_WRONLY   - 只写
 *   O_RDWR     - 读写
 *
 * flags 可选组合（按位或）:
 *   O_CREAT    - 文件不存在时创建
 *   O_EXCL     - 与 O_CREAT 配合，文件存在则报错
 *   O_TRUNC    - 打开时截断文件为0
 *   O_APPEND   - 追加模式
 *   O_NONBLOCK - 非阻塞模式
 *   O_DIRECTORY - 必须是目录
 *   O_NOFOLLOW  - 不跟随符号链接
 *   O_CLOEXEC   - 执行 exec 时自动关闭
 *
 * mode (创建文件时的权限):
 *   S_IRUSR, S_IWUSR, S_IXUSR
 *   S_IRGRP, S_IWGRP, S_IXGRP
 *   S_IROTH, S_IWOTH, S_IXOTH
 */

void open_examples(void)
{
    int fd;

    // 1. 只读打开
    fd = open("data.txt", O_RDONLY);
    if (fd == -1) { perror("open"); return; }
    close(fd);

    // 2. 只写打开，不存在则创建，存在则截断
    fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return; }
    close(fd);

    // 3. 读写打开，不存在则创建，存在则报错
    fd = open("newfile.txt", O_RDWR | O_CREAT | O_EXCL, 0644);
    if (fd == -1) {
        if (errno == EEXIST) {
            printf("File already exists!\n");
        } else {
            perror("open");
        }
        return;
    }
    close(fd);

    // 4. 追加模式打开
    fd = open("log.txt", O_WRONLY | O_CREAT | O_APPEND, 0644);
    if (fd == -1) { perror("open"); return; }
    close(fd);

    // 5. 打开目录
    fd = open(".", O_RDONLY | O_DIRECTORY);
    if (fd == -1) { perror("open"); return; }
    close(fd);
}
```

## 3. creat 系统调用

## 4. read 系统调用

```c
#include <unistd.h>

/*
 * ssize_t read(int fd, void *buf, size_t count);
 *
 * 返回值:
 *   > 0 - 实际读取的字节数
 *   = 0 - 到达文件末尾 (EOF)
 *   = -1 - 出错，errno 被设置
 *
 * 注意：read 可能返回少于请求的字节数（短读取）
 */

void read_examples(void)
{
    int fd = open("data.txt", O_RDONLY);
    if (fd == -1) { perror("open"); return; }

    char buffer[1024];
    ssize_t n;

    // 方式1：一次性读取
    n = read(fd, buffer, sizeof(buffer) - 1);
    if (n > 0) {
        buffer[n] = '\0';
        printf("Read %zd bytes: %s\n", n, buffer);
    }

    close(fd);
}

// 循环读取完整文件
ssize_t read_all(int fd, char *buf, size_t size)
{
    ssize_t total = 0;
    ssize_t n;

    while (total < (ssize_t)size) {
        n = read(fd, buf + total, size - total);
        if (n == -1) {
            if (errno == EINTR) continue;  // 被信号中断，重试
            return -1;                      // 其他错误
        }
        if (n == 0) break;  // EOF
        total += n;
    }

    return total;
}
```

## 5. write 系统调用

```c
#include <unistd.h>

/*
 * ssize_t write(int fd, const void *buf, size_t count);
 *
 * 返回值:
 *   > 0 - 实际写入的字节数
 *   = -1 - 出错
 *
 * 注意：write 也可能写入少于请求的字节数（短写入）
 * 常见原因：磁盘已满、文件大小限制（RLIMIT_FSIZE）、
 *          管道缓冲区已满、信号中断
 */

void write_examples(void)
{
    int fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return; }

    const char *msg = "Hello, Linux IO!\n";
    ssize_t n = write(fd, msg, strlen(msg));
    if (n == -1) {
        perror("write");
    } else {
        printf("Wrote %zd bytes\n", n);
    }

    close(fd);
}

// 可靠的写入：处理短写入和信号中断
ssize_t write_all(int fd, const char *buf, size_t size)
{
    ssize_t total = 0;
    ssize_t n;

    while (total < (ssize_t)size) {
        n = write(fd, buf + total, size - total);
        if (n == -1) {
            if (errno == EINTR) continue;  // 被信号中断
            return -1;
        }
        total += n;
    }

    return total;
}
```

## 6. close 系统调用

```c
#include <unistd.h>

/*
 * int close(int fd);
 *
 * 关闭文件描述符，释放内核资源
 * 注意：close 也可能失败！常见原因：
 *   - NFS 写入错误（数据在 close 时才实际写入）
 *   - 文件系统已满
 * 重要：关闭后应将 fd 置为 -1，避免误用
 */

void close_example(void)
{
    int fd = open("data.txt", O_RDONLY);
    if (fd == -1) { perror("open"); return; }

    // ... 使用 fd ...

    if (close(fd) == -1) {
        perror("close");
    }
    fd = -1;  // 防止悬空引用
}
```

## 7. lseek 系统调用

```c
#include <unistd.h>
#include <sys/types.h>

/*
 * off_t lseek(int fd, off_t offset, int whence);
 *
 * whence:
 *   SEEK_SET - 从文件开头
 *   SEEK_CUR - 从当前位置
 *   SEEK_END - 从文件末尾
 *
 * 返回值：新的文件偏移量，失败返回 -1
 */

void lseek_examples(void)
{
    int fd = open("data.txt", O_RDWR);
    if (fd == -1) { perror("open"); return; }

    // 获取文件大小
    off_t size = lseek(fd, 0, SEEK_END);
    printf("File size: %ld bytes\n", (long)size);

    // 回到文件开头
    lseek(fd, 0, SEEK_SET);

    // 跳到第 100 字节
    lseek(fd, 100, SEEK_SET);

    // 相对当前位置前进 50 字节
    lseek(fd, 50, SEEK_CUR);

    // 回退 10 字节
    lseek(fd, -10, SEEK_CUR);

    close(fd);
}

// 利用 lseek 创建空洞文件
void create_hole_file(void)
{
    int fd = open("sparse.bin", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return; }

    write(fd, "START", 5);

    // 创建空洞：跳过 1MB
    lseek(fd, 1024 * 1024, SEEK_CUR);

    write(fd, "END", 3);
    // 文件逻辑大小约为 1MB+8，但实际磁盘占用很少

    close(fd);
}
```

## 8. 文件共享与原子操作

```c
#include <fcntl.h>
#include <unistd.h>

/*
 * 多进程/多线程共享文件时的关键问题:
 *
 * 1. 每个进程有自己的文件描述符表
 * 2. fork 后父子进程共享同一文件表项（共享偏移量）
 * 3. 同一进程多次 open 同一文件，获得不同文件表项（独立偏移量）
 */

// O_APPEND 保证追加写的原子性
void atomic_append(void)
{
    // O_APPEND 使得每次 write 前自动 seek 到末尾
    // 内核保证这个操作是原子的
    int fd = open("log.txt", O_WRONLY | O_CREAT | O_APPEND, 0644);
    if (fd == -1) { perror("open"); return; }

    // 即使多个进程同时写入，也不会互相覆盖
    const char *msg = "Log entry\n";
    write(fd, msg, strlen(msg));

    close(fd);
}

// pread/pwrite：不改变偏移量的原子读写
#include <unistd.h>

void pread_pwrite_example(void)
{
    int fd = open("data.bin", O_RDWR);
    if (fd == -1) { perror("open"); return; }

    char buf[256];

    // pread = lseek + read（原子操作，不改变偏移量）
    ssize_t n = pread(fd, buf, sizeof(buf), 100);  // 从偏移 100 读取
    if (n > 0) {
        printf("Read %zd bytes from offset 100\n", n);
    }

    // pwrite = lseek + write（原子操作）
    const char *data = "PATCH";
    n = pwrite(fd, data, strlen(data), 200);  // 在偏移 200 写入
    if (n > 0) {
        printf("Wrote %zd bytes at offset 200\n", n);
    }

    close(fd);
}
```

## 9. dup 和 dup2

```c
#include <unistd.h>

/*
 * int dup(int oldfd);
 * int dup2(int oldfd, int newfd);
 * int dup3(int oldfd, int newfd, int flags);
 *
 * dup：复制文件描述符，返回最小可用 fd
 * dup2：复制到指定的 newfd（先关闭 newfd）
 * dup3：dup2 + O_CLOEXEC 标志
 */

void dup_examples(void)
{
    int fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    // dup 示例
    int fd2 = dup(fd);
    printf("dup returned fd: %d\n", fd2);  // 通常是 3 或更大
    close(fd2);

    // dup2 示例：重定向 stdout 到文件
    int saved_stdout = dup(STDOUT_FILENO);  // 保存原始 stdout
    dup2(fd, STDOUT_FILENO);  // 将 fd 复制到 fd 1
    printf("This goes to the file!\n");    // 输出到文件而非终端
    fflush(stdout);

    // 恢复 stdout
    dup2(saved_stdout, STDOUT_FILENO);
    close(saved_stdout);
    printf("This goes to the terminal again!\n");

    close(fd);
}

// 重定向子进程的标准输出
#include <sys/wait.h>

void redirect_child_stdout(void)
{
    pid_t pid;
    int pipefd[2];

    pipe(pipefd);
    pid = fork();

    if (pid == 0) {
        // 子进程
        close(pipefd[0]);  // 关闭读端
        dup2(pipefd[1], STDOUT_FILENO);
        close(pipefd[1]);

        execlp("ls", "ls", "-la", NULL);
        perror("execlp");
        _exit(1);
    } else {
        // 父进程
        close(pipefd[1]);
        char buf[4096];
        ssize_t n;
        while ((n = read(pipefd[0], buf, sizeof(buf))) > 0) {
            write(STDOUT_FILENO, buf, n);
        }
        close(pipefd[0]);
        waitpid(pid, NULL, 0);
    }
}
```

## 10. fcntl 系统调用

```c
#include <fcntl.h>

/*
 * int fcntl(int fd, int cmd, ...);
 *
 * 功能强大的文件控制函数
 * 常用 cmd:
 *   F_DUPFD  - 复制描述符
 *   F_GETFD  - 获取文件描述符标志
 *   F_SETFD  - 设置文件描述符标志
 *   F_GETFL  - 获取文件状态标志
 *   F_SETFL  - 设置文件状态标志
 *   F_GETLK  - 获取文件锁
 *   F_SETLK  - 设置文件锁（非阻塞）
 *   F_SETLKW - 设置文件锁（阻塞）
 */

void fcntl_examples(void)
{
    int fd = open("data.txt", O_RDWR);
    if (fd == -1) { perror("open"); return; }

    // 获取文件状态标志
    int flags = fcntl(fd, F_GETFL);
    if (flags == -1) { perror("fcntl F_GETFL"); close(fd); return; }

    // 检查打开模式
    int access_mode = flags & O_ACCMODE;
    if (access_mode == O_RDONLY) printf("Read only\n");
    if (access_mode == O_WRONLY) printf("Write only\n");
    if (access_mode == O_RDWR)   printf("Read/Write\n");

    // 检查是否设置了 O_APPEND
    if (flags & O_APPEND) printf("Append mode\n");

    // 设置非阻塞模式
    if (fcntl(fd, F_SETFL, flags | O_NONBLOCK) == -1) {
        perror("fcntl F_SETFL");
    }

    // 设置 FD_CLOEXEC 标志
    int fd_flags = fcntl(fd, F_GETFD);
    fcntl(fd, F_SETFD, fd_flags | FD_CLOEXEC);

    close(fd);
}
```

## 11. 综合示例：简易 tee 命令

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <string.h>
#include <errno.h>

/*
 * 实现 tee 命令：从标准输入读取，
 * 同时写入标准输出和文件
 */
int main(int argc, char *argv[])
{
    int append = 0;
    int opt;
    int fd;
    char buffer[4096];
    ssize_t n;

    // 解析命令行参数：-a 表示追加模式
    while ((opt = getopt(argc, argv, "a")) != -1) {
        switch (opt) {
        case 'a': append = 1; break;
        default:
            fprintf(stderr, "Usage: %s [-a] [file...]\n", argv[0]);
            return EXIT_FAILURE;
        }
    }

    // 打开所有输出文件
    int *fds = malloc((argc - optind + 1) * sizeof(int));
    if (!fds) { perror("malloc"); return EXIT_FAILURE; }

    int nfds = 0;
    for (int i = optind; i < argc; i++) {
        int flags = O_WRONLY | O_CREAT | (append ? O_APPEND : O_TRUNC);
        fd = open(argv[i], flags, 0644);
        if (fd == -1) {
            fprintf(stderr, "Cannot open '%s': %s\n",
                    argv[i], strerror(errno));
            continue;
        }
        fds[nfds++] = fd;
    }

    // 从 stdin 读取，写入 stdout 和所有文件
    while ((n = read(STDIN_FILENO, buffer, sizeof(buffer))) > 0) {
        // 写入标准输出
        write(STDOUT_FILENO, buffer, n);

        // 写入所有文件
        for (int i = 0; i < nfds; i++) {
            write_all(fds[i], buffer, n);
        }
    }

    // 清理
    for (int i = 0; i < nfds; i++) {
        close(fds[i]);
    }
    free(fds);

    return EXIT_SUCCESS;
}
```
