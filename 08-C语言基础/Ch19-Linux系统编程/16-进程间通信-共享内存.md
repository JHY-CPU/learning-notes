# 进程间通信 - 共享内存

## 1. System V 共享内存

```c
#include <sys/ipc.h>
#include <sys/shm.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * System V 共享内存 API:
 *   int shmget(key_t key, size_t size, int shmflg);  - 创建/获取
 *   void *shmat(int shmid, const void *shmaddr, int shmflg);  - 附加
 *   int shmdt(const void *shmaddr);  - 分离
 *   int shmctl(int shmid, int cmd, struct shmid_ds *buf);  - 控制
 *
 * shmflg (shmget):
 *   IPC_CREAT - 不存在则创建
 *   IPC_EXCL  - 已存在则报错
 *   0666      - 权限
 *
 * shmflg (shmat):
 *   0         - 可读写
 *   SHM_RDONLY - 只读
 *   SHM_RND    - 地址对齐到 SHMLBA
 */

// 共享内存数据结构
typedef struct {
    int counter;
    char message[256];
    pid_t last_writer;
} SharedData;

void shm_example(void)
{
    key_t key = ftok("/tmp", 'S');
    if (key == -1) { perror("ftok"); return; }

    // 创建共享内存段（1KB）
    int shmid = shmget(key, sizeof(SharedData), IPC_CREAT | 0666);
    if (shmid == -1) { perror("shmget"); return; }

    printf("Shared memory ID: %d\n", shmid);

    // 附加到进程地址空间
    SharedData *data = shmat(shmid, NULL, 0);
    if (data == (void *)-1) { perror("shmat"); return; }

    // 初始化
    data->counter = 0;
    strcpy(data->message, "Initial");

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程：修改共享内存
        SharedData *child_data = shmat(shmid, NULL, 0);
        child_data->counter = 42;
        strcpy(child_data->message, "Written by child");
        child_data->last_writer = getpid();
        shmdt(child_data);
        _exit(0);
    } else {
        // 父进程等待并读取
        wait(NULL);
        printf("Counter: %d\n", data->counter);
        printf("Message: %s\n", data->message);
        printf("Last writer PID: %d\n", data->last_writer);

        // 分离
        shmdt(data);

        // 删除共享内存
        shmctl(shmid, IPC_RMID, NULL);
    }
}
```

## 2. 共享内存 + 信号量同步

```c
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

/*
 * 共享内存本身不提供同步机制
 * 需要配合信号量实现进程间同步
 */

// 联合体用于 semctl
union semun {
    int val;
    struct semid_ds *buf;
    unsigned short *array;
};

// P 操作（等待）
void sem_wait(int semid, int semnum)
{
    struct sembuf op;
    op.sem_num = semnum;
    op.sem_op  = -1;
    op.sem_flg = 0;
    semop(semid, &op, 1);
}

// V 操作（释放）
void sem_signal(int semid, int semnum)
{
    struct sembuf op;
    op.sem_num = semnum;
    op.sem_op  = 1;
    op.sem_flg = 0;
    semop(semid, &op, 1);
}

typedef struct {
    int count;
    char data[256];
} SHMData;

void shm_with_semaphore(void)
{
    key_t shm_key = ftok("/tmp", 'M');
    key_t sem_key = ftok("/tmp", 'S');

    // 创建共享内存
    int shmid = shmget(shm_key, sizeof(SHMData), IPC_CREAT | 0666);
    SHMData *shm = shmat(shmid, NULL, 0);
    shm->count = 0;

    // 创建信号量（1个信号量，用于互斥）
    int semid = semget(sem_key, 1, IPC_CREAT | 0666);
    union semun arg;
    arg.val = 1;  // 初始值为1（可用）
    semctl(semid, 0, SETVAL, arg);

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程
        SHMData *child_shm = shmat(shmid, NULL, 0);

        for (int i = 0; i < 5; i++) {
            sem_wait(semid, 0);  // 获取锁

            child_shm->count++;
            snprintf(child_shm->data, sizeof(child_shm->data),
                     "Child update %d", child_shm->count);
            printf("Child: %s\n", child_shm->data);

            sem_signal(semid, 0);  // 释放锁
            usleep(100000);
        }

        shmdt(child_shm);
        _exit(0);
    } else {
        // 父进程
        for (int i = 0; i < 5; i++) {
            sem_wait(semid, 0);  // 获取锁

            shm->count++;
            snprintf(shm->data, sizeof(shm->data),
                     "Parent update %d", shm->count);
            printf("Parent: %s\n", shm->data);

            sem_signal(semid, 0);  // 释放锁
            usleep(100000);
        }

        wait(NULL);
        printf("Final count: %d\n", shm->count);

        shmdt(shm);
        shmctl(shmid, IPC_RMID, NULL);
        semctl(semid, 0, IPC_RMID);
    }
}
```

## 3. POSIX 共享内存

```c
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <semaphore.h>

/*
 * POSIX 共享内存 API:
 *   int shm_open(const char *name, int oflag, mode_t mode);  - 打开
 *   int shm_unlink(const char *name);  - 删除
 *   int ftruncate(int fd, off_t length);  - 设置大小
 *   void *mmap(void *addr, size_t length, int prot,
 *              int flags, int fd, off_t offset);  - 映射
 *
 * 优势：API 更简洁，与文件 IO 风格一致
 * 名称格式：以 '/' 开头，如 "/myshm"
 */

typedef struct {
    sem_t mutex;
    int counter;
    char message[256];
} POSIXShmData;

void posix_shm_example(void)
{
    const char *shm_name = "/my_shm_example";

    // 创建共享内存对象
    int fd = shm_open(shm_name, O_CREAT | O_RDWR, 0666);
    if (fd == -1) { perror("shm_open"); return; }

    // 设置大小
    ftruncate(fd, sizeof(POSIXShmData));

    // 映射到进程地址空间
    POSIXShmData *data = mmap(NULL, sizeof(POSIXShmData),
                               PROT_READ | PROT_WRITE,
                               MAP_SHARED, fd, 0);
    if (data == MAP_FAILED) { perror("mmap"); return; }
    close(fd);  // 映射后可以关闭 fd

    // 初始化信号量（第二个参数 1 = 进程间共享）
    sem_init(&data->mutex, 1, 1);
    data->counter = 0;

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程
        POSIXShmData *child_data = mmap(NULL, sizeof(POSIXShmData),
                                         PROT_READ | PROT_WRITE,
                                         MAP_SHARED, -1, 0);
        // 重新打开共享内存
        int cfd = shm_open(shm_name, O_RDWR, 0666);
        child_data = mmap(NULL, sizeof(POSIXShmData),
                          PROT_READ | PROT_WRITE, MAP_SHARED, cfd, 0);
        close(cfd);

        for (int i = 0; i < 5; i++) {
            sem_wait(&child_data->mutex);
            child_data->counter++;
            snprintf(child_data->message, sizeof(child_data->message),
                     "Child: count = %d", child_data->counter);
            printf("%s\n", child_data->message);
            sem_post(&child_data->mutex);
            usleep(50000);
        }

        munmap(child_data, sizeof(POSIXShmData));
        _exit(0);
    } else {
        // 父进程
        for (int i = 0; i < 5; i++) {
            sem_wait(&data->mutex);
            data->counter++;
            snprintf(data->message, sizeof(data->message),
                     "Parent: count = %d", data->counter);
            printf("%s\n", data->message);
            sem_post(&data->mutex);
            usleep(50000);
        }

        wait(NULL);
        printf("Final counter: %d\n", data->counter);

        munmap(data, sizeof(POSIXShmData));
        shm_unlink(shm_name);
    }
}

// 生产者-消费者使用 POSIX 共享内存
#define BUF_SIZE 10

typedef struct {
    sem_t mutex;      // 互斥
    sem_t empty;      // 空槽数
    sem_t full;       // 满槽数
    int buffer[BUF_SIZE];
    int head;
    int tail;
} SharedBuffer;

void posix_shm_producer_consumer(void)
{
    const char *shm_name = "/shm_producer_consumer";

    int fd = shm_open(shm_name, O_CREAT | O_RDWR, 0666);
    ftruncate(fd, sizeof(SharedBuffer));

    SharedBuffer *buf = mmap(NULL, sizeof(SharedBuffer),
                              PROT_READ | PROT_WRITE,
                              MAP_SHARED, fd, 0);
    close(fd);

    sem_init(&buf->mutex, 1, 1);
    sem_init(&buf->empty, 1, BUF_SIZE);
    sem_init(&buf->full, 1, 0);
    buf->head = 0;
    buf->tail = 0;

    pid_t pid = fork();
    if (pid == 0) {
        // 消费者
        SharedBuffer *cbuf = mmap(NULL, sizeof(SharedBuffer),
                                   PROT_READ | PROT_WRITE,
                                   MAP_SHARED, -1, 0);
        int cfd = shm_open(shm_name, O_RDWR, 0666);
        cbuf = mmap(NULL, sizeof(SharedBuffer),
                    PROT_READ | PROT_WRITE, MAP_SHARED, cfd, 0);
        close(cfd);

        for (int i = 0; i < 20; i++) {
            sem_wait(&cbuf->full);
            sem_wait(&cbuf->mutex);

            int item = cbuf->buffer[cbuf->head];
            cbuf->head = (cbuf->head + 1) % BUF_SIZE;
            printf("Consumed: %d\n", item);

            sem_post(&cbuf->mutex);
            sem_post(&cbuf->empty);
        }

        munmap(cbuf, sizeof(SharedBuffer));
        _exit(0);
    } else {
        // 生产者
        for (int i = 0; i < 20; i++) {
            sem_wait(&buf->empty);
            sem_wait(&buf->mutex);

            buf->buffer[buf->tail] = i;
            buf->tail = (buf->tail + 1) % BUF_SIZE;
            printf("Produced: %d\n", i);

            sem_post(&buf->mutex);
            sem_post(&buf->full);
            usleep(50000);
        }

        wait(NULL);
        munmap(buf, sizeof(SharedBuffer));
        shm_unlink(shm_name);
    }
}
```
