# 网络服务器设计

## 1. 多进程模型

```c
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>
#include <signal.h>
#include <errno.h>

/*
 * 多进程模型:
 *   主进程 accept()，子进程处理连接
 *   优点：简单，进程隔离，一个崩溃不影响其他
 *   缺点：进程创建开销大，内存占用高
 */

#define PORT 8080

// 信号处理：回收僵尸进程
void sigchld_handler(int sig)
{
    (void)sig;
    int saved_errno = errno;
    while (waitpid(-1, NULL, WNOHANG) > 0)
        ;
    errno = saved_errno;
}

void handle_client(int conn_fd, struct sockaddr_in client_addr)
{
    printf("[PID %d] Client %s:%d connected\n",
           getpid(),
           inet_ntoa(client_addr.sin_addr),
           ntohs(client_addr.sin_port));

    char buf[1024];
    ssize_t n;
    while ((n = read(conn_fd, buf, sizeof(buf) - 1)) > 0) {
        buf[n] = '\0';
        printf("[PID %d] Received: %s", getpid(), buf);
        write(conn_fd, buf, n);  // echo
    }

    printf("[PID %d] Client disconnected\n", getpid());
    close(conn_fd);
}

void multiprocess_server(int port)
{
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 128);

    // 设置 SIGCHLD 处理器
    struct sigaction sa;
    sa.sa_handler = sigchld_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = SA_RESTART | SA_NOCLDSTOP;
    sigaction(SIGCHLD, &sa, NULL);

    printf("Multiprocess server on port %d\n", port);

    while (1) {
        struct sockaddr_in client_addr;
        socklen_t client_len = sizeof(client_addr);

        int conn_fd = accept(listen_fd, (struct sockaddr *)&client_addr,
                             &client_len);
        if (conn_fd == -1) {
            if (errno == EINTR) continue;
            perror("accept");
            continue;
        }

        pid_t pid = fork();
        if (pid == 0) {
            // 子进程
            close(listen_fd);
            handle_client(conn_fd, client_addr);
            _exit(0);
        } else if (pid > 0) {
            // 父进程
            close(conn_fd);
        } else {
            perror("fork");
            close(conn_fd);
        }
    }

    close(listen_fd);
}
```

## 2. 多线程模型

```c
#include <pthread.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

/*
 * 多线程模型:
 *   主线程 accept()，工作线程处理连接
 *   优点：开销小，共享内存，切换快
 *   缺点：一个线程崩溃可能影响整个进程，需要同步
 */

typedef struct {
    int conn_fd;
    struct sockaddr_in client_addr;
} ClientInfo;

void *thread_worker(void *arg)
{
    ClientInfo *ci = (ClientInfo *)arg;
    int conn_fd = ci->conn_fd;

    printf("[Thread %lu] Client %s:%d connected\n",
           (unsigned long)pthread_self(),
           inet_ntoa(ci->client_addr.sin_addr),
           ntohs(ci->client_addr.sin_port));
    free(ci);

    // 分离线程，自动回收
    pthread_detach(pthread_self());

    char buf[1024];
    ssize_t n;
    while ((n = read(conn_fd, buf, sizeof(buf) - 1)) > 0) {
        buf[n] = '\0';
        write(conn_fd, buf, n);
    }

    printf("[Thread %lu] Client disconnected\n",
           (unsigned long)pthread_self());
    close(conn_fd);

    return NULL;
}

void multithread_server(int port)
{
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 128);

    printf("Multithread server on port %d\n", port);

    while (1) {
        ClientInfo *ci = malloc(sizeof(ClientInfo));
        socklen_t len = sizeof(ci->client_addr);

        ci->conn_fd = accept(listen_fd,
                             (struct sockaddr *)&ci->client_addr, &len);
        if (ci->conn_fd == -1) {
            free(ci);
            continue;
        }

        pthread_t tid;
        if (pthread_create(&tid, NULL, thread_worker, ci) != 0) {
            perror("pthread_create");
            close(ci->conn_fd);
            free(ci);
        }
    }

    close(listen_fd);
}
```

## 3. 线程池模型

```c
#include <pthread.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

/*
 * 线程池模型:
 *   预创建固定数量的线程，处理任务队列
 *   优点：避免频繁创建/销毁线程
 */

typedef struct Task {
    int conn_fd;
    struct Task *next;
} Task;

typedef struct {
    Task *head, *tail;
    int count;
    pthread_mutex_t lock;
    pthread_cond_t cond;
    int shutdown;
} TaskQueue;

typedef struct {
    pthread_t *threads;
    int num_threads;
    TaskQueue queue;
} ThreadPool;

void taskqueue_init(TaskQueue *q)
{
    q->head = q->tail = NULL;
    q->count = 0;
    q->shutdown = 0;
    pthread_mutex_init(&q->lock, NULL);
    pthread_cond_init(&q->cond, NULL);
}

void taskqueue_push(TaskQueue *q, int conn_fd)
{
    Task *task = malloc(sizeof(Task));
    task->conn_fd = conn_fd;
    task->next = NULL;

    pthread_mutex_lock(&q->lock);
    if (q->tail) {
        q->tail->next = task;
    } else {
        q->head = task;
    }
    q->tail = task;
    q->count++;
    pthread_cond_signal(&q->cond);
    pthread_mutex_unlock(&q->lock);
}

int taskqueue_pop(TaskQueue *q)
{
    pthread_mutex_lock(&q->lock);

    while (q->count == 0 && !q->shutdown) {
        pthread_cond_wait(&q->cond, &q->lock);
    }

    if (q->shutdown && q->count == 0) {
        pthread_mutex_unlock(&q->lock);
        return -1;
    }

    Task *task = q->head;
    int fd = task->conn_fd;

    q->head = task->next;
    if (!q->head) q->tail = NULL;
    q->count--;

    pthread_mutex_unlock(&q->lock);
    free(task);
    return fd;
}

void *pool_worker(void *arg)
{
    ThreadPool *pool = (ThreadPool *)arg;

    while (1) {
        int conn_fd = taskqueue_pop(&pool->queue);
        if (conn_fd == -1) break;

        // 处理连接
        char buf[1024];
        ssize_t n;
        while ((n = read(conn_fd, buf, sizeof(buf))) > 0) {
            write(conn_fd, buf, n);
        }
        close(conn_fd);
    }

    return NULL;
}

void threadpool_server(int port, int num_threads)
{
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(listen_fd, 128);

    // 创建线程池
    ThreadPool pool;
    pool.num_threads = num_threads;
    pool.threads = malloc(num_threads * sizeof(pthread_t));
    taskqueue_init(&pool.queue);

    for (int i = 0; i < num_threads; i++) {
        pthread_create(&pool.threads[i], NULL, pool_worker, &pool);
    }

    printf("Thread pool server (%d threads) on port %d\n",
           num_threads, port);

    while (1) {
        int conn_fd = accept(listen_fd, NULL, NULL);
        if (conn_fd >= 0) {
            taskqueue_push(&pool.queue, conn_fd);
        }
    }
}
```

## 4. Reactor 模型

```c
#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <fcntl.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <stdlib.h>
#include <errno.h>

/*
 * Reactor 模型:
 *   事件驱动，由事件循环分发事件到处理器
 *   适用于高并发场景
 *
 * 组件:
 *   EventLoop   - 事件循环
 *   EventDemultiplexer - 事件多路分离器（epoll）
 *   EventHandler - 事件处理器
 */

#define MAX_EVENTS 1024
#define BUF_SIZE 4096

typedef struct Connection {
    int fd;
    char buf[BUF_SIZE];
    int buf_len;
    struct Connection *next;
} Connection;

typedef struct {
    int epoll_fd;
    int listen_fd;
    Connection *connections;
} Reactor;

void set_nonblocking(int fd)
{
    int flags = fcntl(fd, F_GETFL, 0);
    fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

Connection *find_connection(Reactor *r, int fd)
{
    for (Connection *c = r->connections; c; c = c->next) {
        if (c->fd == fd) return c;
    }
    return NULL;
}

void add_connection(Reactor *r, int fd)
{
    Connection *c = calloc(1, sizeof(Connection));
    c->fd = fd;
    c->next = r->connections;
    r->connections = c;

    set_nonblocking(fd);
    struct epoll_event ev;
    ev.events = EPOLLIN | EPOLLET;
    ev.data.fd = fd;
    epoll_ctl(r->epoll_fd, EPOLL_CTL_ADD, fd, &ev);
}

void remove_connection(Reactor *r, int fd)
{
    epoll_ctl(r->epoll_fd, EPOLL_CTL_DEL, fd, NULL);
    close(fd);

    Connection **prev = &r->connections;
    for (Connection *c = r->connections; c; c = c->next) {
        if (c->fd == fd) {
            *prev = c->next;
            free(c);
            break;
        }
        prev = &c->next;
    }
}

void reactor_init(Reactor *r, int port)
{
    r->listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    r->epoll_fd = epoll_create1(0);
    r->connections = NULL;

    set_nonblocking(r->listen_fd);
    int opt = 1;
    setsockopt(r->listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(r->listen_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(r->listen_fd, 128);

    struct epoll_event ev;
    ev.events = EPOLLIN;
    ev.data.fd = r->listen_fd;
    epoll_ctl(r->epoll_fd, EPOLL_CTL_ADD, r->listen_fd, &ev);
}

void reactor_run(Reactor *r)
{
    struct epoll_event events[MAX_EVENTS];

    printf("Reactor server running\n");

    while (1) {
        int nfds = epoll_wait(r->epoll_fd, events, MAX_EVENTS, -1);

        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == r->listen_fd) {
                // 新连接
                while (1) {
                    int fd = accept(r->listen_fd, NULL, NULL);
                    if (fd == -1) break;
                    add_connection(r, fd);
                }
            } else if (events[i].events & EPOLLIN) {
                // 可读事件
                int fd = events[i].data.fd;
                char buf[BUF_SIZE];
                ssize_t n = read(fd, buf, sizeof(buf));
                if (n <= 0) {
                    remove_connection(r, fd);
                } else {
                    write(fd, buf, n);  // echo
                }
            }
        }
    }
}
```
