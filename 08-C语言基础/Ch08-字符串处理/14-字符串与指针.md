# 14 - 字符串与指针

## 一、字符指针操作字符串

### 1.1 指针遍历字符串

```c
#include <stdio.h>

int main() {
    char str[] = "Hello, World!";
    char *p = str;

    // 方式1：指针遍历
    while (*p != '\0') {
        printf("%c ", *p);
        p++;
    }
    printf("\n");  // H e l l o ,   W o r l d !

    // 方式2：下标遍历
    for (int i = 0; str[i] != '\0'; i++) {
        printf("%c ", str[i]);
    }
    printf("\n");

    return 0;
}
```

### 1.2 指针与数组的等价性

```c
#include <stdio.h>

int main() {
    char str[] = "Hello";

    // 以下四种写法完全等价
    printf("%c\n", str[0]);      // H
    printf("%c\n", *(str + 0));  // H
    printf("%c\n", 0[str]);      // H （不推荐，但合法）

    char *p = str;
    printf("%c\n", p[0]);        // H
    printf("%c\n", *p);          // H

    return 0;
}
```

### 1.3 指针运算

```c
#include <stdio.h>
#include <string.h>

int main() {
    char str[] = "Hello, World!";
    char *p = str;

    // 指针移动
    p += 7;               // 指向 'W'
    printf("%s\n", p);    // World!

    // 指针相减
    char *q = strchr(str, 'o');
    printf("'o' 在位置: %ld\n", q - str);  // 4

    // 指针比较
    char *start = str;
    char *end = str + strlen(str);
    printf("指针距离: %ld\n", end - start);  // 13

    return 0;
}
```

## 二、字符串数组

### 2.1 二维字符数组

```c
#include <stdio.h>
#include <string.h>

int main() {
    // 每行固定20字节
    char names[5][20] = {
        "Alice",
        "Bob",
        "Charlie",
        "David",
        "Eve"
    };

    // 遍历
    for (int i = 0; i < 5; i++) {
        printf("%s (长度=%zu)\n", names[i], strlen(names[i]));
    }

    // 修改
    strcpy(names[1], "Bobby");

    // 特点：所有字符串占用相同空间
    printf("总大小: %zu 字节\n", sizeof(names));  // 100

    return 0;
}
```

### 2.2 指针数组（推荐）

```c
#include <stdio.h>
#include <string.h>

int main() {
    // 每个指针指向独立的字符串
    char *names[] = {
        "Alice",
        "Bob",
        "Charlie",
        "David",
        "Eve"
    };

    // 遍历
    for (int i = 0; i < 5; i++) {
        printf("%s (长度=%zu)\n", names[i], strlen(names[i]));
    }

    // 指针可以重新指向
    names[1] = "Robert";  // 指向新的字面量

    // 特点：只存储指针，字符串本身在只读区
    printf("指针数组大小: %zu 字节\n", sizeof(names));  // 40

    return 0;
}
```

### 2.3 动态字符串数组

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    int capacity = 4;
    int count = 0;

    // 动态指针数组
    char **lines = (char **)malloc(capacity * sizeof(char *));
    if (!lines) return 1;

    // 添加字符串
    const char *data[] = {"First", "Second", "Third", "Fourth", "Fifth"};

    for (int i = 0; i < 5; i++) {
        // 扩容
        if (count >= capacity) {
            capacity *= 2;
            lines = (char **)realloc(lines, capacity * sizeof(char *));
        }

        // 复制字符串
        lines[count] = (char *)malloc(strlen(data[i]) + 1);
        strcpy(lines[count], data[i]);
        count++;
    }

    // 打印
    for (int i = 0; i < count; i++) {
        printf("%s\n", lines[i]);
    }

    // 释放
    for (int i = 0; i < count; i++) {
        free(lines[i]);
    }
    free(lines);

    return 0;
}
```

## 三、函数传参

### 3.1 字符串参数传递

```c
#include <stdio.h>
#include <string.h>

// 不修改字符串：用 const 修饰
void print_upper(const char *str) {
    while (*str) {
        putchar(toupper((unsigned char)*str));
        str++;
    }
    putchar('\n');
}

// 修改字符串：传入可写的数组或指针
void to_upper(char *str) {
    while (*str) {
        if (*str >= 'a' && *str <= 'z') {
            *str -= 32;
        }
        str++;
    }
}

int main() {
    // 数组名自动退化为指针
    char arr[] = "Hello";
    print_upper(arr);  // HELLO

    to_upper(arr);
    printf("%s\n", arr);  // HELLO

    // 字面量传给 const 参数
    print_upper("World");  // WORLD

    return 0;
}
```

### 3.2 返回字符串

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// ❌ 错误：返回局部数组
char *bad_func() {
    char str[] = "Hello";  // 局部变量，函数返回后无效
    return str;
}

// ✅ 正确1：返回字面量指针
const char *get_greeting() {
    return "Hello, World!";  // 字面量在只读区，始终有效
}

// ✅ 正确2：返回动态分配的字符串
char *create_string(const char *src) {
    char *str = (char *)malloc(strlen(src) + 1);
    if (str) {
        strcpy(str, src);
    }
    return str;  // 调用者负责 free
}

// ✅ 正确3：通过参数返回
void get_string(char *buf, size_t size) {
    snprintf(buf, size, "Hello!");
}

int main() {
    const char *greet = get_greeting();
    printf("%s\n", greet);

    char *str = create_string("Dynamic");
    if (str) {
        printf("%s\n", str);
        free(str);
    }

    char buf[50];
    get_string(buf, sizeof(buf));
    printf("%s\n", buf);

    return 0;
}
```

## 四、常见模式

### 4.1 双指针字符串处理

```c
#include <stdio.h>
#include <string.h>

// 反转字符串（双指针法）
void reverse(char *str) {
    char *left = str;
    char *right = str + strlen(str) - 1;

    while (left < right) {
        char temp = *left;
        *left = *right;
        *right = temp;
        left++;
        right--;
    }
}

int main() {
    char str[] = "Hello";
    reverse(str);
    printf("%s\n", str);  // olleH
    return 0;
}
```

### 4.2 指针数组排序

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int compare_strings(const void *a, const void *b) {
    return strcmp(*(const char **)a, *(const char **)b);
}

int main() {
    char *fruits[] = {"Banana", "Apple", "Cherry", "Date"};
    int n = 4;

    qsort(fruits, n, sizeof(char *), compare_strings);

    for (int i = 0; i < n; i++) {
        printf("%s\n", fruits[i]);
    }
    // Apple
    // Banana
    // Cherry
    // Date

    return 0;
}
```

## 五、重要注意事项

> **关键要点：**
> 1. **数组名在大多数情况下退化为指针**：除了 `sizeof` 和 `&` 操作
> 2. **指针数组的字符串可以是字面量**：不可修改，但不需要释放
> 3. **函数参数 `char str[]` 和 `char *str` 完全等价**
> 4. **不要返回局部数组的地址**：使用动态分配或 `static` 或字面量
> 5. **`const char *` 表示不修改字符串**：好的编程习惯
> 6. **指针数组排序比二维数组排序更高效**：只交换指针，不移动字符串
