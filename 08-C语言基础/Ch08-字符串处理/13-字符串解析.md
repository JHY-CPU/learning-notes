# 13 - 字符串解析

## 一、sscanf 基本用法

### 1.1 函数原型

```c
#include <stdio.h>
int sscanf(const char *str, const char *format, ...);
```

- 从字符串 `str` 中按格式读取数据
- 返回值：成功赋值的变量个数

### 1.2 基本示例

```c
#include <stdio.h>

int main() {
    // 解析空格分隔的数据
    char line[] = "Alice 25 3.85";
    char name[20];
    int age;
    double gpa;

    int count = sscanf(line, "%s %d %lf", name, &age, &gpa);
    printf("成功解析 %d 个字段\n", count);
    printf("姓名: %s, 年龄: %d, GPA: %.2f\n", name, age, gpa);

    return 0;
}
```

## 二、格式控制详解

### 2.1 跳过字段（赋值抑制符 *）

```c
#include <stdio.h>

int main() {
    char line[] = "2024-01-15 10:30:00 INFO Server started";
    int year, month, day;
    char level[10], message[50];

    // 跳过时间部分，只取日期、级别和消息
    sscanf(line, "%d-%d-%d %*d:%*d:%*d %s %[^\n]",
           &year, &month, &day, level, message);

    printf("日期: %d-%02d-%02d\n", year, month, day);
    printf("级别: %s\n", level);
    printf("消息: %s\n", message);

    return 0;
}
```

### 2.2 宽度限制

```c
#include <stdio.h>

int main() {
    char str[] = "HelloWorld123";

    // 限制读取字符数
    char part1[10], part2[10];
    sscanf(str, "%5s%5s", part1, part2);
    printf("part1: %s\n", part1);  // Hello
    printf("part2: %s\n", part2);  // World

    return 0;
}
```

### 2.3 字符集匹配

```c
#include <stdio.h>

int main() {
    // 读取直到特定字符
    char line[] = "Name: Alice Smith";
    char label[10], value[50];

    sscanf(line, "%[^:]: %[^\n]", label, value);
    printf("标签: '%s'\n", label);   // "Name"
    printf("值: '%s'\n", value);     // " Alice Smith"

    // 只匹配特定字符集
    char str[] = "abc123def";
    char letters[10], digits[10];
    sscanf(str, "%[a-z]%[0-9]%[a-z]", letters, digits, str);
    printf("字母: %s, 数字: %s\n", letters, digits);

    return 0;
}
```

## 三、实际应用示例

### 3.1 解析配置文件行

```c
#include <stdio.h>
#include <string.h>

int main() {
    char *lines[] = {
        "host = localhost",
        "port = 8080",
        "debug = true",
        "name = My Application"
    };

    for (int i = 0; i < 4; i++) {
        char key[50], value[50];
        if (sscanf(lines[i], " %49[^=] = %49[^\n]", key, value) == 2) {
            // 去除前后空格
            char *k = key;
            while (*k == ' ') k++;
            char *end = key + strlen(key) - 1;
            while (end > k && *end == ' ') *end-- = '\0';

            printf("配置: '%s' = '%s'\n", k, value);
        }
    }

    return 0;
}
```

### 3.2 解析URL

```c
#include <stdio.h>
#include <string.h>

void parse_url(const char *url) {
    char protocol[10], host[100], path[200];
    int port = 80;

    // 尝试解析带端口的URL
    int n = sscanf(url, "%9[^:]://%99[^:]:%d%199[^\n]",
                   protocol, host, &port, path);

    if (n < 3) {
        // 没有端口，重新解析
        n = sscanf(url, "%9[^:]://%99[^/]%199[^\n]",
                   protocol, host, path);
        port = (strcmp(protocol, "https") == 0) ? 443 : 80;
    }

    if (n >= 2) {
        printf("协议: %s\n", protocol);
        printf("主机: %s\n", host);
        printf("端口: %d\n", port);
        printf("路径: %s\n", path[0] ? path : "/");
    }
}

int main() {
    parse_url("https://www.example.com:8443/path/to/page");
    // 协议: https
    // 主机: www.example.com
    // 端口: 8443
    // 路径: /path/to/page

    return 0;
}
```

### 3.3 解析CSV数据

#include <stdio.h>

int main() {
    char csv[] = "John,28,Engineer,75000.50";

## 四、正则表达式替代方案

C标准库没有正则表达式，但 POSIX 提供了 `<regex.h>`：

```c
#include <stdio.h>
#include <regex.h>

int main() {
    regex_t regex;
    regmatch_t matches[3];

    // 编译正则表达式
    if (regcomp(&regex, "([0-9]+)\\.([0-9]+)", REG_EXTENDED) != 0) {
        printf("正则编译失败\n");
        return 1;
    }

    char str[] = "Version 3.14 released";
    if (regexec(&regex, str, 3, matches, 0) == 0) {
        printf("找到匹配: '%.*s'\n",
               matches[0].rm_eo - matches[0].rm_so,
               str + matches[0].rm_so);
    }

    regfree(&regex);
    return 0;
}
```

## 五、常见解析技巧

```c
// 技巧1: 读取整行（含空格）
sscanf(line, "%[^\n]", buffer);

// 技巧2: 跳过空白
sscanf(str, " %d", &num);  // 开头的空格匹配任意空白

// 技巧3: 读取到字符串结尾
sscanf(str, "%*[^=]=%[^\n]", value);  // 跳过 = 前的内容

// 技巧4: 十六进制解析
unsigned int color;
sscanf("#FF00FF", "#%x", &color);  // color = 0xFF00FF

// 技巧5: 检查格式是否匹配
if (sscanf(str, "%d-%d-%d", &y, &m, &d) != 3) {
    printf("格式错误\n");
}
```

## 六、重要注意事项

> **关键要点：**
> 1. **检查 `sscanf` 返回值**：确保所有字段都成功解析
> 2. **格式字符串要精确**：空白、分隔符必须完全匹配
> 3. **防止缓冲区溢出**：字符串字段指定最大宽度 `%49s`
> 4. **`%[^\n]` 读取到换行**：用于获取含空格的字符串
> 5. **`%*` 跳过不需要的字段**
> 6. **`sscanf` 不能处理动态数量的字段**：需要自己用 `strtok` 实现
