# 10 - 字符串转换

## 一、基本转换函数 atoi / atol / atof

### 1.1 函数原型

```c
#include <stdlib.h>
int atoi(const char *nptr);      // 字符串 → int
long atol(const char *nptr);     // 字符串 → long
double atof(const char *nptr);   // 字符串 → double
```

### 1.2 基本示例

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    // 整数转换
    int num1 = atoi("123");
    int num2 = atoi("-456");
    int num3 = atoi("  789");     // 前导空格被忽略
    printf("num1=%d, num2=%d, num3=%d\n", num1, num2, num3);

    // 浮点数转换
    double f1 = atof("3.14");
    double f2 = atof("-2.5e3");
    printf("f1=%f, f2=%f\n", f1, f2);

    // 注意：atoi 遇到非数字字符会停止
    int num4 = atoi("123abc");    // 返回 123
    int num5 = atoi("abc123");    // 返回 0
    printf("num4=%d, num5=%d\n", num4, num5);

    return 0;
}
```

### 1.3 atoi 的问题

```c
// ⚠️ atoi 无法检测错误
int val = atoi("invalid");   // 返回 0，但 0 也是合法输入
int val = atoi("0");         // 也返回 0
// 无法区分 "输入是0" 还是 "输入无效"

// ⚠️ atoi 无法检测溢出
int val = atoi("99999999999999");  // 溢出，未定义行为
```

## 二、更安全的 strtol / strtod

### 2.1 函数原型

```c
#include <stdlib.h>
long strtol(const char *nptr, char **endptr, int base);
unsigned long strtoul(const char *nptr, char **endptr, int base);
double strtod(const char *nptr, char **endptr);
```

- `endptr`：指向第一个无法转换的字符的指针（可检测错误）
- `base`：进制（2-36，0表示自动检测）

### 2.2 基本示例

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>

int main() {
    char *endptr;

    // 基本十进制转换
    long val = strtol("12345", &endptr, 10);
    if (*endptr == '\0') {
        printf("成功转换: %ld\n", val);
    }

    // 十六进制
    val = strtol("0xFF", &endptr, 16);
    printf("0xFF = %ld\n", val);  // 255

    // 自动检测进制（0x前缀为16进制，0前缀为8进制）
    val = strtol("0x1A", &endptr, 0);
    printf("0x1A = %ld\n", val);  // 26

    // 二进制
    val = strtol("1010", &endptr, 2);
    printf("1010(二进制) = %ld\n", val);  // 10

    return 0;
}
```

### 2.3 错误检测

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <limits.h>

int main() {
    char *str = "99999999999999999999";
    char *endptr;

    errno = 0;  // 清除错误标志
    long val = strtol(str, &endptr, 10);

    if (errno == ERANGE) {
        printf("溢出！值超出 long 范围\n");
        if (val == LONG_MAX) {
            printf("上溢出\n");
        } else if (val == LONG_MIN) {
            printf("下溢出\n");
        }
    } else if (*endptr != '\0') {
        printf("部分转换: %ld, 剩余: %s\n", val, endptr);
    } else {
        printf("成功: %ld\n", val);
    }

    return 0;
}
```

### 2.4 解析混合内容

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    char str[] = "100 200abc300 xyz";
    char *p = str;
    char *end;

    while (*p) {
        long val = strtol(p, &end, 10);
        if (p != end) {
            printf("找到数字: %ld\n", val);
        }
        p = end;
        if (*p) p++;  // 跳过非数字字符
    }
    // 找到数字: 100
    // 找到数字: 200
    // 找到数字: 300

    return 0;
}
```

## 三、浮点数转换 strtod

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    char *endptr;

    // 各种浮点格式
    double v1 = strtod("3.14159", &endptr);
    double v2 = strtod("-1.23e-4", &endptr);
    double v3 = strtod("  42.0", &endptr);      // 前导空格被忽略
    double v4 = strtod("inf", &endptr);          // 无穷大
    double v5 = strtod("nan", &endptr);          // NaN

    printf("v1=%f, v2=%e, v3=%f\n", v1, v2, v3);
    printf("v4=%f, v5=%f\n", v4, v5);

    return 0;
}
```

## 四、数字转字符串

### 4.1 sprintf / snprintf

```c
#include <stdio.h>

int main() {
    char buf[50];

    // 整数转字符串
    int num = 12345;
    sprintf(buf, "%d", num);
    printf("字符串: %s\n", buf);  // "12345"

    // 浮点数转字符串
    double pi = 3.14159;
    sprintf(buf, "%.2f", pi);
    printf("字符串: %s\n", buf);  // "3.14"

    // 不同进制
    sprintf(buf, "%x", 255);   // "ff"
    sprintf(buf, "%o", 255);   // "377"
    sprintf(buf, "%b", 255);   // 注意：%b不是标准格式

    return 0;
}
```

### 4.2 itoa（非标准，部分平台支持）

```c
// itoa 不是标准C函数，不可移植
// 推荐使用 sprintf 替代

// 自定义实现
void my_itoa(int value, char *str, int base) {
    char *p = str;
    char *p1, *p2;
    unsigned int uvalue;
    int temp;

    // 处理负数十进制
    if (base == 10 && value < 0) {
        *p++ = '-';
        uvalue = (unsigned int)(-value);
    } else {
        uvalue = (unsigned int)value;
    }

    p1 = p;

    // 转换数字
    do {
        temp = uvalue % base;
        *p++ = (temp < 10) ? '0' + temp : 'a' + temp - 10;
    } while (uvalue /= base);

    *p-- = '\0';

    // 反转字符串
    p2 = p;
    while (p1 < p2) {
        char c = *p1;
        *p1++ = *p2;
        *p2-- = c;
    }
}
```

## 五、转换函数对比

| 函数 | 输入 | 输出 | 错误检测 | 溢出检测 |
|------|------|------|----------|----------|
| `atoi` | 字符串 | int | 无 | 无 |
| `atol` | 字符串 | long | 无 | 无 |
| `atof` | 字符串 | double | 无 | 无 |
| `strtol` | 字符串 | long | 有(endptr) | 有(errno) |
| `strtod` | 字符串 | double | 有(endptr) | 有(errno) |

## 六、重要注意事项

> **关键要点：**
> 1. **优先使用 `strtol/strtod`**：支持错误检测和溢出检测
> 2. **`atoi` 无法区分错误和零值**：不应用于需要健壮性的代码
> 3. **检查 `errno` 和 `endptr`**：确保转换成功
> 4. **数字转字符串用 `snprintf`**：安全且标准
> 5. **`strtol` 的 base 参数**：0表示自动检测，2-36表示指定进制
> 6. **注意 `long` 与 `int` 的范围**：`strtol` 返回 `long`，可能需要范围检查后转 `int`
