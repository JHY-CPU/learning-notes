# 21 - 宽字符与多字节

## 一、多字节字符与宽字符的概念

### 1.1 为什么需要宽字符

标准 `char` 类型只有 1 字节（0-255），无法表示所有语言的字符。

```c
// 标准 char 只能表示 ASCII (0-127)
// 中文、日文、韩文等需要多字节编码

// 多字节字符（Multi-byte Character）
// 用多个 char 表示一个字符，如 UTF-8
char utf8[] = "你好";  // 6个字节（每字3字节）

// 宽字符（Wide Character）
// 用固定宽度的类型表示一个字符
wchar_t wide[] = L"你好";  // 2个wchar_t（假设sizeof(wchar_t)==4）
```

### 1.2 编码基础

```
常见编码：

ASCII:       1字节，0-127
Latin-1:     1字节，0-255
UTF-8:       1-4字节，变长，兼容ASCII
UTF-16:      2或4字节，变长
UTF-32:      4字节，定长
GBK/GB2312:  1或2字节，中文编码
```

## 二、wchar_t 宽字符类型

### 2.1 基本使用

```c
#include <stdio.h>
#include <wchar.h>
#include <locale.h>

int main() {
    // 设置 locale（必须！否则宽字符函数行为不确定）
    setlocale(LC_ALL, "");

    // 宽字符字面量
    wchar_t wc = L'A';
    wchar_t wch = L'中';

    // 宽字符串字面量
    wchar_t wstr[] = L"Hello, 世界！";

    // 输出宽字符
    wprintf(L"宽字符: %lc\n", wc);
    wprintf(L"宽字符串: %ls\n", wstr);

    return 0;
}
```

### 2.2 wchar_t 的大小

```c
#include <stdio.h>
#include <wchar.h>

int main() {
    printf("wchar_t 大小: %zu 字节\n", sizeof(wchar_t));
    // Linux/macOS: 4字节
    // Windows: 2字节

    // wchar_t 的范围
    printf("WCHAR_MAX: %d\n", WCHAR_MAX);
    printf("WCHAR_MIN: %d\n", WCHAR_MIN);

    return 0;
}
```

## 三、wcs 系列宽字符函数

### 3.1 宽字符串操作函数

```c
#include <stdio.h>
#include <wchar.h>
#include <string.h>

int main() {
    setlocale(LC_ALL, "");

    wchar_t ws1[50] = L"Hello";
    wchar_t ws2[50] = L"World";

    // 长度
    size_t len = wcslen(ws1);          // 宽字符版本的 strlen
    wprintf(L"长度: %zu\n", len);      // 5

    // 复制
    wcscpy(ws2, ws1);                  // 宽字符版本的 strcpy
    wprintf(L"复制: %ls\n", ws2);

    // 连接
    wcscat(ws1, L", World!");          // 宽字符版本的 strcat
    wprintf(L"连接: %ls\n", ws1);

    // 比较
    int cmp = wcscmp(L"abc", L"abd");  // 宽字符版本的 strcmp
    wprintf(L"比较: %d\n", cmp);

    // 查找
    wchar_t *pos = wcschr(ws1, L',');  // 宽字符版本的 strchr
    if (pos) wprintf(L"找到逗号: %ls\n", pos);

    // 子串查找
    pos = wcsstr(ws1, L"World");
    if (pos) wprintf(L"找到子串: %ls\n", pos);

    return 0;
}
```

### 3.2 宽字符函数对照表

| char 版本 | wchar_t 版本 | 功能 |
|-----------|-------------|------|
| `strlen` | `wcslen` | 字符串长度 |
| `strcpy` | `wcscpy` | 字符串复制 |
| `strcat` | `wcscat` | 字符串连接 |
| `strcmp` | `wcscmp` | 字符串比较 |
| `strchr` | `wcschr` | 查找字符 |
| `strstr` | `wcsstr` | 查找子串 |
| `strtok` | `wcstok` | 分割字符串 |
| `sprintf` | `swprintf` | 格式化输出 |
| `sscanf` | `swscanf` | 格式化输入 |

## 四、多字节与宽字符转换

### 4.1 mbstowcs - 多字节转宽字符

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <locale.h>

int main() {
    setlocale(LC_ALL, "");

    const char *mb_str = "Hello, 世界！";
    size_t mb_len = strlen(mb_str);

    // 计算需要的宽字符数
    size_t wc_len = mbstowcs(NULL, mb_str, 0);
    if (wc_len == (size_t)-1) {
        printf("转换失败\n");
        return 1;
    }

    // 分配宽字符串
    wchar_t *wc_str = (wchar_t *)malloc((wc_len + 1) * sizeof(wchar_t));

    // 转换
    mbstowcs(wc_str, mb_str, wc_len + 1);

    wprintf(L"宽字符串: %ls\n", wc_str);
    wprintf(L"宽字符数: %zu\n", wc_len);

    free(wc_str);
    return 0;
}
```

### 4.2 wcstombs - 宽字符转多字节

```c
#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

int main() {
    setlocale(LC_ALL, "");

    wchar_t wc_str[] = L"Hello, 世界！";

    // 计算需要的字节数
    size_t mb_len = wcstombs(NULL, wc_str, 0);
    if (mb_len == (size_t)-1) {
        printf("转换失败\n");
        return 1;
    }

    // 分配
    char *mb_str = (char *)malloc(mb_len + 1);

    // 转换
    wcstombs(mb_str, wc_str, mb_len + 1);

    printf("多字节字符串: %s\n", mb_str);
    printf("字节数: %zu\n", mb_len);

    free(mb_str);
    return 0;
}
```

## 五、宽字符的输入输出

```c
#include <stdio.h>
#include <wchar.h>
#include <locale.h>

int main() {
    setlocale(LC_ALL, "");

    wchar_t buf[100];

    // 输出
    wprintf(L"请输入一行文本: ");

    // 输入宽字符串
    fgetws(buf, 100, stdin);

    // 去除换行符
    size_t len = wcslen(buf);
    if (len > 0 && buf[len - 1] == L'\n') {
        buf[len - 1] = L'\0';
    }

    wprintf(L"你输入了: %ls\n", buf);
    wprintf(L"长度: %zu 个宽字符\n", wcslen(buf));

    return 0;
}
```

## 六、C11 通用字符名称

```c
#include <stdio.h>

int main() {
    // 通用字符名称（Universal Character Names）
    // \uXXXX: 4位十六进制 (BMP字符)
    // \UXXXXXXXX: 8位十六进制 (补充字符)

    // UTF-8 字符串
    char *s1 = "你好";           // "你好"
    printf("%s\n", s1);

    // 宽字符
    wchar_t *s2 = L"你好";       // L"你好"

    // UTF-16 (C11)
    char16_t *s3 = u"你好";

    // UTF-32 (C11)
    char32_t *s4 = U"你好";

    return 0;
}
```

## 七、重要注意事项

> **关键要点：**
> 1. **必须调用 `setlocale`**：否则宽字符函数行为不确定
> 2. **`wchar_t` 大小因平台而异**：Windows 为2字节，Linux/macOS 为4字节
> 3. **`mbstowcs`/`wcstombs` 依赖当前 locale**
> 4. **宽字符占用更多内存**：但处理更简单（每个字符固定宽度）
> 5. **Windows 使用 UTF-16（2字节 wchar_t）**：无法表示所有Unicode字符
> 6. **跨平台推荐使用 UTF-8**：存储用 `char` + UTF-8，显示时再转换
