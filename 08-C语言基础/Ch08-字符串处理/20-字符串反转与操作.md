# 20 - 字符串反转与操作

## 一、字符串反转

### 1.1 原地反转（双指针法）

```c
#include <stdio.h>
#include <string.h>

// 原地反转字符串
void reverse_string(char *str) {
    if (!str) return;

    char *left = str;
    char *right = str + strlen(str) - 1;

    while (left < right) {
        char temp = *left;
        *left = *right;
        *right = temp;
        left++;
        right--;
    }
}

int main() {
    char str[] = "Hello, World!";
    reverse_string(str);
    printf("%s\n", str);  // !dlroW ,olleH

    return 0;
}
```

### 1.2 反转单词顺序

```c
#include <stdio.h>
#include <string.h>

// 反转字符串中每个单词
void reverse_words(char *str) {
    char *start = str;
    char *end = str;

    while (*end) {
        // 找到单词的结尾
        while (*end && *end != ' ') end++;

        // 反转这个单词
        char *left = start;
        char *right = end - 1;
        while (left < right) {
            char temp = *left;
            *left = *right;
            *right = temp;
            left++;
            right--;
        }

        // 跳过空格
        while (*end == ' ') end++;
        start = end;
    }
}

int main() {
    char str[] = "Hello World C Language";
    reverse_words(str);
    printf("%s\n", str);  // olleH dlroW C egaugnaL

    return 0;
}
```

### 1.3 反转整个句子中的单词顺序

```c
#include <stdio.h>
#include <string.h>

// 先反转整个字符串，再反转每个单词
void reverse_word_order(char *str) {
    int len = (int)strlen(str);

    // 反转整个字符串
    for (int i = 0, j = len - 1; i < j; i++, j--) {
        char temp = str[i];
        str[i] = str[j];
        str[j] = temp;
    }

    // 反转每个单词
    char *start = str;
    char *end = str;
    while (*end) {
        while (*end && *end != ' ') end++;
        char *left = start;
        char *right = end - 1;
        while (left < right) {
            char temp = *left;
            *left = *right;
            *right = temp;
            left++;
            right--;
        }
        while (*end == ' ') end++;
        start = end;
    }
}

int main() {
    char str[] = "Hello World C Language";
    reverse_word_order(str);
    printf("%s\n", str);  // Language C World Hello

    return 0;
}
```

## 二、大小写转换

### 2.1 逐字符转换

```c
#include <stdio.h>
#include <ctype.h>

void to_uppercase(char *str) {
    while (*str) {
        *str = (char)toupper((unsigned char)*str);
        str++;
    }
}

void to_lowercase(char *str) {
    while (*str) {
        *str = (char)tolower((unsigned char)*str);
        str++;
    }
}

void toggle_case(char *str) {
    while (*str) {
        if (islower((unsigned char)*str)) {
            *str = (char)toupper((unsigned char)*str);
        } else if (isupper((unsigned char)*str)) {
            *str = (char)tolower((unsigned char)*str);
        }
        str++;
    }
}

// 首字母大写
void capitalize(char *str) {
    int new_word = 1;
    while (*str) {
        if (*str == ' ' || *str == '\t') {
            new_word = 1;
        } else if (new_word) {
            *str = (char)toupper((unsigned char)*str);
            new_word = 0;
        } else {
            *str = (char)tolower((unsigned char)*str);
        }
        str++;
    }
}

int main() {
    char str[] = "hello, World!";

    to_uppercase(str);
    printf("大写: %s\n", str);  // HELLO, WORLD!

    to_lowercase(str);
    printf("小写: %s\n", str);  // hello, world!

    toggle_case(str);
    printf("反转: %s\n", str);  // HELLO, WORLD!

    char title[] = "hello world c language";
    capitalize(title);
    printf("标题: %s\n", title);  // Hello World C Language

    return 0;
}
```

## 三、去除空白

### 3.1 去除首尾空白

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

// 去除左侧空白（原地）
char *trim_left(char *str) {
    char *p = str;
    while (isspace((unsigned char)*p)) p++;
    if (p != str) {
        memmove(str, p, strlen(p) + 1);
    }
    return str;
}

// 去除右侧空白
char *trim_right(char *str) {
    char *end = str + strlen(str) - 1;
    while (end >= str && isspace((unsigned char)*end)) {
        *end = '\0';
        end--;
    }
    return str;
}

// 去除首尾空白
char *trim(char *str) {
    return trim_right(trim_left(str));
}

int main() {
    char str[] = "   Hello, World!   \t\n";
    printf("原始: '%s'\n", str);

    trim(str);
    printf("去除: '%s'\n", str);  // 'Hello, World!'

    return 0;
}
```

### 3.2 去除所有空白

```c
#include <stdio.h>
#include <ctype.h>

void remove_all_spaces(char *str) {
    char *src = str;
    char *dst = str;

    while (*src) {
        if (!isspace((unsigned char)*src)) {
            *dst++ = *src;
        }
        src++;
    }
    *dst = '\0';
}

// 压缩连续空格为一个
void compress_spaces(char *str) {
    char *src = str;
    char *dst = str;
    int in_space = 0;

    while (*src) {
        if (isspace((unsigned char)*src)) {
            if (!in_space) {
                *dst++ = ' ';
                in_space = 1;
            }
        } else {
            *dst++ = *src;
            in_space = 0;
        }
        src++;
    }
    *dst = '\0';
}
```

## 四、字符串压缩与展开

### 4.1 游程编码压缩

```c
#include <stdio.h>
#include <string.h>

// "aaabbbcc" → "a3b3c2"
int compress(char *dest, size_t dest_size, const char *src) {
    size_t d = 0;
    size_t s = 0;
    size_t len = strlen(src);

    while (s < len && d < dest_size - 1) {
        char c = src[s];
        int count = 1;

        while (s + count < len && src[s + count] == c) {
            count++;
        }

        if (count > 1) {
            int written = snprintf(dest + d, dest_size - d, "%c%d", c, count);
            if (written < 0 || (size_t)written >= dest_size - d) return -1;
            d += (size_t)written;
        } else {
            dest[d++] = c;
        }
        s += (size_t)count;
    }

    dest[d] = '\0';

    // 只有压缩后更短才返回压缩结果
    return (d < len) ? 0 : 1;
}

int main() {
    char compressed[100];
    const char *input = "aaabbbcccddee";

    if (compress(compressed, sizeof(compressed), input) == 0) {
        printf("原串: %s\n", input);
        printf("压缩: %s\n", compressed);  // a3b3c3d2e2
    }

    return 0;
}
```

### 4.2 展开压缩字符串

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

// "a3b2c" → "aaabbc"
int expand(char *dest, size_t dest_size, const char *src) {
    size_t d = 0;
    size_t s = 0;
    size_t len = strlen(src);

    while (s < len && d < dest_size - 1) {
        char c = src[s++];

        // 读取数字
        int count = 0;
        while (s < len && isdigit((unsigned char)src[s])) {
            count = count * 10 + (src[s++] - '0');
        }

        if (count == 0) count = 1;

        // 写入字符
        for (int i = 0; i < count && d < dest_size - 1; i++) {
            dest[d++] = c;
        }
    }

    dest[d] = '\0';
    return (d < dest_size) ? 0 : -1;
}
```

## 五、其他常用操作

### 5.1 字符串旋转

```c
#include <stdio.h>
#include <string.h>

// 左旋转 n 位: "abcdef" n=2 → "cdefab"
void left_rotate(char *str, int n) {
    int len = (int)strlen(str);
    if (len == 0) return;
    n = n % len;
    if (n == 0) return;

    // 三次反转法
    // 反转前n个
    for (int i = 0, j = n - 1; i < j; i++, j--) {
        char temp = str[i]; str[i] = str[j]; str[j] = temp;
    }
    // 反转后len-n个
    for (int i = n, j = len - 1; i < j; i++, j--) {
        char temp = str[i]; str[i] = str[j]; str[j] = temp;
    }
    // 反转全部
    for (int i = 0, j = len - 1; i < j; i++, j--) {
        char temp = str[i]; str[i] = str[j]; str[j] = temp;
    }
}
```

### 5.2 判断回文串

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

// 判断是否回文（忽略大小写和非字母字符）
int is_palindrome(const char *str) {
    const char *left = str;
    const char *right = str + strlen(str) - 1;

    while (left < right) {
        while (left < right && !isalpha((unsigned char)*left)) left++;
        while (left < right && !isalpha((unsigned char)*right)) right--;

        if (tolower((unsigned char)*left) !=
            tolower((unsigned char)*right)) {
            return 0;
        }
        left++;
        right--;
    }
    return 1;
}
```

## 六、重要注意事项

> **关键要点：**
> 1. **原地操作时注意 `\0` 的位置**：添加或删除字符时保持正确终止
> 2. **使用 `memmove` 处理重叠内存**：不要用 `memcpy`
> 3. **使用 `unsigned char` 调用 `ctype.h` 函数**：避免负数索引
> 4. **压缩后可能更长**：最坏情况 "abc" → "a1b1c1"，要检查空间
> 5. **反转单词顺序的经典技巧**：先整体反转，再逐词反转
