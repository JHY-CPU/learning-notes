# 27 - 字符串性能优化

## 一、避免重复计算 strlen

### 1.1 循环中的 strlen

```c
#include <stdio.h>
#include <string.h>

// ❌ 低效：O(n²)
void process_string_bad(const char *str) {
    for (size_t i = 0; i < strlen(str); i++) {
        // 每次循环都调用 strlen，遍历整个字符串
        putchar(str[i]);
    }
}

// ✅ 高效：O(n)
void process_string_good(const char *str) {
    size_t len = strlen(str);
    for (size_t i = 0; i < len; i++) {
        putchar(str[i]);
    }
}

// ✅ 更高效：使用指针，O(n)
void process_string_best(const char *str) {
    while (*str) {
        putchar(*str++);
    }
}
```

### 1.2 缓存长度值

```c
typedef struct {
    char *data;
    size_t length;    // 缓存的长度
    size_t capacity;
} String;

// 操作时维护 length
void string_append(String *s, const char *str) {
    size_t len = strlen(str);
    // ... 扩容检查 ...
    memcpy(s->data + s->length, str, len + 1);
    s->length += len;  // 直接更新，不需要再计算
}
```

## 二、减少字符串拷贝

### 2.1 传指针而非拷贝

```c
// ❌ 低效：每次调用都复制整个字符串
void process(char str[1000]) {
    // str 是参数的副本（实际上退化为指针，但概念上避免复制）
}

// ✅ 高效：传 const 指针
void process(const char *str) {
    // 只传递指针，不复制字符串内容
}
```

### 2.2 使用引用/指针返回结果

```c
// ❌ 低效：返回时复制
char *find_and_copy(const char *haystack, const char *needle) {
    char *pos = strstr(haystack, needle);
    if (!pos) return NULL;
    char *result = strdup(pos);
    return result;  // 调用者需要 free
}

// ✅ 高效：返回指针（不复制）
const char *find(const char *haystack, const char *needle) {
    return strstr(haystack, needle);  // 返回指向原字符串的指针
}
```

### 2.3 拷贝时指定长度

```c
// ❌ 可能不必要地复制整个大缓冲区
char dest[1000];
strcpy(dest, "Hello");  // 实际只需要6字节

// ✅ 只复制需要的部分
char dest[1000];
size_t len = strlen("Hello");
memcpy(dest, "Hello", len + 1);  // 编译器可能优化为固定大小
```

## 三、高效的字符串构建

### 3.1 预分配空间

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// ❌ 低效：频繁 realloc
char *build_string_bad() {
    char *result = malloc(1);
    result[0] = '\0';

    for (int i = 0; i < 10000; i++) {
        size_t len = strlen(result);
        result = realloc(result, len + 2);
        result[len] = 'a';
        result[len + 1] = '\0';
    }
    return result;
}

// ✅ 高效：预分配 + 倍增
char *build_string_good() {
    size_t capacity = 16;
    size_t length = 0;
    char *result = malloc(capacity);

    for (int i = 0; i < 10000; i++) {
        if (length + 1 >= capacity) {
            capacity *= 2;
            result = realloc(result, capacity);
        }
        result[length++] = 'a';
    }
    result[length] = '\0';
    return result;
}
```

### 3.2 使用 sprintf 一次构建

```c
// ❌ 低效：多次 strcat
char buf[100] = "";
strcat(buf, "Name: ");
strcat(buf, name);
strcat(buf, ", Age: ");
// ... 每次都遍历到末尾

// ✅ 高效：一次 snprintf
snprintf(buf, sizeof(buf), "Name: %s, Age: %d", name, age);
```

## 四、SSO（Small String Optimization）思想

### 4.1 SSO 原理

短字符串直接存储在结构体中，避免堆分配：

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define SSO_THRESHOLD 15

typedef struct {
    size_t length;
    union {
        char *heap_ptr;              // 长字符串：指向堆
        char buffer[SSO_THRESHOLD + 1]; // 短字符串：直接存储
    };
} SSOString;

// 是否使用 SSO
static inline int is_sso(const SSOString *s) {
    return s->length <= SSO_THRESHOLD;
}

// 获取字符串指针
static inline const char *sso_get(const SSOString *s) {
    return is_sso(s) ? s->buffer : s->heap_ptr;
}

// 设置字符串
void sso_set(SSOString *s, const char *str) {
    size_t len = strlen(str);
    s->length = len;

    if (len <= SSO_THRESHOLD) {
        // 短字符串：直接拷贝到 buffer
        memcpy(s->buffer, str, len + 1);
    } else {
        // 长字符串：分配堆内存
        s->heap_ptr = (char *)malloc(len + 1);
        memcpy(s->heap_ptr, str, len + 1);
    }
}

// 释放
void sso_free(SSOString *s) {
    if (!is_sso(s)) {
        free(s->heap_ptr);
    }
    s->length = 0;
}

int main() {
    SSOString s1, s2;

    sso_set(&s1, "Hi");              // SSO: 无堆分配
    sso_set(&s2, "A long string..."); // 堆分配

    printf("s1: %s (SSO=%d)\n", sso_get(&s1), is_sso(&s1));
    printf("s2: %s (SSO=%d)\n", sso_get(&s2), is_sso(&s2));

    sso_free(&s1);
    sso_free(&s2);

    return 0;
}
```

## 五、编译器优化提示

### 5.1 使用 restrict 关键字

```c
// 告诉编译器指针不重叠，可以激进优化
void fast_copy(char *restrict dest, const char *restrict src, size_t n) {
    memcpy(dest, src, n);
}
```

### 5.2 内联小函数

```c
// 编译器会内联展开，避免函数调用开销
static inline int my_strcmp(const char *a, const char *b) {
    while (*a && *a == *b) { a++; b++; }
    return (unsigned char)*a - (unsigned char)*b;
}
```

## 六、批量操作

### 6.1 使用 SIMD 加速

```c
// 现代编译器对 memcpy/memset 会自动使用 SIMD
// 也可以使用编译器内建函数
#include <immintrin.h>

// 统计字符串中某字符的出现次数
size_t count_char_simd(const char *str, char target) {
    // 编译器可能自动向量化这个循环
    size_t count = 0;
    while (*str) {
        if (*str == target) count++;
        str++;
    }
    return count;
}
```

## 七、性能对比

```
操作                  普通方法          优化方法
----------------------------------------------------------
统计长度并遍历        O(n²) strlen循环  O(n) 缓存长度
字符串拼接1000次      O(n²) strcat      O(n) 尾指针/snprintf
短字符串存储          malloc+free       SSO (无堆分配)
字符串比较            strcmp            memcmp(定长)
构建复杂字符串        多次sprintf       一次snprintf
```

## 八、重要注意事项

> **关键要点：**
> 1. **先测量再优化**：不要猜测瓶颈在哪
> 2. **缓存 strlen 结果**：避免重复遍历
> 3. **预分配内存**：减少 realloc 调用
> 4. **用指针代替拷贝**：尽可能传递 `const char*`
> 5. **SSO 对短字符串很有效**：大多数字符串是短的
> 6. **编译器优化选项很重要**：`-O2` 或 `-O3` 能显著提升性能
> 7. **可读性优先**：不要为了微优化牺牲代码可读性
