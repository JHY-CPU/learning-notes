# 23 - 字符串与文件

## 一、读取文本文件

### 1.1 fgets 逐行读取

```c
#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp = fopen("example.txt", "r");
    if (!fp) {
        perror("打开文件失败");
        return 1;
    }

    char line[1024];

    // 逐行读取
    while (fgets(line, sizeof(line), fp) != NULL) {
        // 去除末尾换行符
        size_t len = strlen(line);
        if (len > 0 && line[len - 1] == '\n') {
            line[len - 1] = '\0';
        }
        printf("%s\n", line);
    }

    fclose(fp);
    return 0;
}
```

### 1.2 处理超长行

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 读取任意长度的行
char *read_long_line(FILE *fp) {
    size_t capacity = 128;
    size_t length = 0;
    char *buffer = (char *)malloc(capacity);

    if (!buffer) return NULL;

    while (fgets(buffer + length, (int)(capacity - length), fp)) {
        length += strlen(buffer + length);

        // 检查是否读完整行
        if (buffer[length - 1] == '\n') {
            buffer[length - 1] = '\0';
            return buffer;
        }

        // 如果到文件末尾
        if (feof(fp)) {
            return buffer;
        }

        // 扩展缓冲区
        capacity *= 2;
        char *temp = (char *)realloc(buffer, capacity);
        if (!temp) {
            free(buffer);
            return NULL;
        }
        buffer = temp;
    }

    if (length == 0) {
        free(buffer);
        return NULL;
    }

    return buffer;
}
```

### 1.3 fread 读取整个文件

```c
#include <stdio.h>
#include <stdlib.h>

char *read_entire_file(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return NULL;

    // 获取文件大小
    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fseek(fp, 0, SEEK_SET);

    // 分配内存
    char *content = (char *)malloc((size_t)size + 1);
    if (!content) {
        fclose(fp);
        return NULL;
    }

    // 读取
    size_t read = fread(content, 1, (size_t)size, fp);
    content[read] = '\0';

    fclose(fp);
    return content;
}
```

## 二、写入文本文件

### 2.1 fputs / fprintf 写入

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("output.txt", "w");
    if (!fp) {
        perror("打开文件失败");
        return 1;
    }

    // 写入字符串
    fputs("Hello, World!\n", fp);

    // 格式化写入
    fprintf(fp, "Name: %s, Age: %d\n", "Alice", 25);
    fprintf(fp, "Score: %.2f\n", 95.5);

    // 逐字符写入
    const char *str = "逐字符写入测试";
    for (const char *p = str; *p; p++) {
        fputc(*p, fp);
    }
    fputc('\n', fp);

    fclose(fp);
    return 0;
}
```

### 2.2 追加模式写入

```c
#include <stdio.h>
#include <time.h>

// 日志写入函数
void write_log(const char *message) {
    FILE *fp = fopen("app.log", "a");  // 追加模式
    if (!fp) return;

    time_t now = time(NULL);
    struct tm *t = localtime(&now);

    fprintf(fp, "[%04d-%02d-%02d %02d:%02d:%02d] %s\n",
            t->tm_year + 1900, t->tm_mon + 1, t->tm_mday,
            t->tm_hour, t->tm_min, t->tm_sec,
            message);

    fclose(fp);
}
```

## 三、文本文件处理实例

### 3.1 统计文件信息

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

typedef struct {
    int lines;
    int words;
    int chars;
    int blank_lines;
} FileStats;

FileStats analyze_file(const char *filename) {
    FileStats stats = {0, 0, 0, 0};
    FILE *fp = fopen(filename, "r");
    if (!fp) return stats;

    char line[4096];
    int in_word = 0;

    while (fgets(line, sizeof(line), fp)) {
        stats.lines++;
        stats.chars += (int)strlen(line);

        int blank = 1;
        for (char *p = line; *p && *p != '\n'; p++) {
            if (isspace((unsigned char)*p)) {
                in_word = 0;
            } else {
                blank = 0;
                if (!in_word) {
                    stats.words++;
                    in_word = 1;
                }
            }
        }

        if (blank) stats.blank_lines++;
    }

    fclose(fp);
    return stats;
}
```

### 3.2 查找和替换

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// 在文件中查找并替换字符串
int find_and_replace(const char *input_file,
                     const char *output_file,
                     const char *find,
                     const char *replace) {
    FILE *fin = fopen(input_file, "r");
    FILE *fout = fopen(output_file, "w");

    if (!fin || !fout) {
        if (fin) fclose(fin);
        if (fout) fclose(fout);
        return -1;
    }

    char line[4096];
    size_t find_len = strlen(find);

    while (fgets(line, sizeof(line), fin)) {
        char *pos = line;
        char *found;

        while ((found = strstr(pos, find)) != NULL) {
            // 写入匹配前的部分
            fwrite(pos, 1, (size_t)(found - pos), fout);
            // 写入替换文本
            fputs(replace, fout);
            // 移动到匹配后
            pos = found + find_len;
        }

        // 写入剩余部分
        fputs(pos, fout);
    }

    fclose(fin);
    fclose(fout);
    return 0;
}
```

### 3.3 CSV 文件处理

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    char name[50];
    int age;
    double score;
} Student;

int read_csv(const char *filename, Student *students, int max_count) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return -1;

    char line[256];
    int count = 0;

    // 跳过标题行
    fgets(line, sizeof(line), fp);

    while (fgets(line, sizeof(line), fp) && count < max_count) {
        // 去除换行
        line[strcspn(line, "\n")] = '\0';

        char *name = strtok(line, ",");
        char *age_str = strtok(NULL, ",");
        char *score_str = strtok(NULL, ",");

        if (name && age_str && score_str) {
            strncpy(students[count].name, name,
                    sizeof(students[count].name) - 1);
            students[count].age = atoi(age_str);
            students[count].score = atof(score_str);
            count++;
        }
    }

    fclose(fp);
    return count;
}
```

## 四、二进制模式与文本模式

```c
// 文本模式 ("r", "w"):
// - 自动处理换行符转换（Windows: \r\n ↔ \n）
// - 适合文本文件

// 二进制模式 ("rb", "wb"):
// - 不做任何转换
// - 适合图片、音频等二进制文件
// - 跨平台读取文本时也推荐用二进制模式

// 读取 UTF-8 文件
FILE *fp = fopen("utf8.txt", "rb");
```

## 五、重要注意事项

> **关键要点：**
> 1. **始终检查 `fopen` 返回值**：文件可能不存在或无权限
> 2. **`fgets` 保留换行符**：需要手动去除
> 3. **`fgets` 可能不读完整行**：缓冲区不够大时
> 4. **及时 `fclose`**：避免资源泄漏
> 5. **文本模式 vs 二进制模式**：Windows 上换行符处理不同
> 6. **大文件逐行处理**：不要一次性读入整个文件
