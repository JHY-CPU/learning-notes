# 17 - 字符串安全函数

## 一、C11 Annex K 安全函数

C11标准的 Annex K 引入了一系列"安全"版本的字符串函数，以 `_s` 后缀标识。

### 1.1 strcpy_s

```c
#define __STDC_WANT_LIB_EXT__ 1
#include <string.h>
#include <stdio.h>

int main() {
    char dest[20];
    const char *src = "Hello, World!";

    // strcpy_s(dest, dest_size, src)
    errno_t err = strcpy_s(dest, sizeof(dest), src);
    if (err == 0) {
        printf("成功: %s\n", dest);
    } else {
        printf("复制失败: 目标缓冲区太小\n");
    }

    // 缓冲区不足的情况
    char small[5];
    err = strcpy_s(small, sizeof(small), "Hello, World!");
    if (err != 0) {
        printf("检测到缓冲区溢出！\n");
        // small 的内容被清空为 ""
    }

    return 0;
}
```

### 1.2 strcat_s

```c
#define __STDC_WANT_LIB_EXT__ 1
#include <string.h>

int main() {
    char buf[30] = "Hello";

    // strcat_s(dest, dest_max, src)
    errno_t err = strcat_s(buf, sizeof(buf), ", World!");
    if (err == 0) {
        printf("%s\n", buf);  // Hello, World!
    }

    // 空间不足的情况
    char small[10] = "Hello";
    err = strcat_s(small, sizeof(small), ", World!");
    if (err != 0) {
        printf("连接失败：缓冲区不足\n");
        // small 被设为空串 ""
    }

    return 0;
}
```

### 1.3 strncpy_s

```c
#define __STDC_WANT_LIB_EXT__ 1
#include <string.h>

int main() {
    char dest[20];

    // strncpy_s(dest, dest_size, src, count)
    // count: 最多复制的字符数（不含\0）
    errno_t err = strncpy_s(dest, sizeof(dest), "Hello, World!", 5);
    // dest: "Hello" (自动添加\0)

    printf("%s\n", dest);  // Hello

    return 0;
}
```

### 1.4 strncat_s

```c
#define __STDC_WANT_LIB_EXT__ 1
#include <string.h>

int main() {
    char buf[30] = "Hello";

    // strncat_s(dest, dest_max, src, count)
    strncat_s(buf, sizeof(buf), ", World!!!", 8);
    printf("%s\n", buf);  // Hello, World

    return 0;
}
```

### 1.5 strtok_s

```c
#define __STDC_WANT_LIB_EXT__ 1
#include <string.h>
#include <stdio.h>

int main() {
    char str[] = "one,two,three,four";
    char *context = NULL;

    // strtok_s 是可重入版本
    char *token = strtok_s(str, ",", &context);
    while (token) {
        printf("%s\n", token);
        token = strtok_s(NULL, ",", &context);
    }

    return 0;
}
```

## 二、安全函数的行为规则

### 2.1 约束处理

```c
#include <stdio.h>
#include <stdlib.h>

// 自定义约束处理函数
void my_constraint_handler(
    const char *msg,
    void *ptr,
    errno_t error
) {
    fprintf(stderr, "约束违反: %s (错误码: %d)\n", msg, error);
}

int main() {
    // 设置约束处理函数
    set_constraint_handler_s(my_constraint_handler);

    // 也可以使用预定义的处理函数
    // set_constraint_handler_s(abort_handler_s);   // 终止程序
    // set_constraint_handler_s(ignore_handler_s);  // 忽略

    char small[5];
    strcpy_s(small, sizeof(small), "Hello, World!");
    // 触发约束处理函数

    return 0;
}
```

### 2.2 运行时约束违反

安全函数在以下情况会触发约束违反：
- 目标指针为 NULL
- 源指针为 NULL（部分函数）
- 目标大小为 0
- 目标大小小于需要的空间
- 源和目标重叠

## 三、snprintf 安全格式化

```c
#include <stdio.h>

int main() {
    char buf[20];

    // snprintf 总是安全的
    int needed = snprintf(buf, sizeof(buf),
                          "Name: %s, Age: %d", "Alice", 25);

    if (needed >= (int)sizeof(buf)) {
        printf("发生了截断，需要 %d 字节\n", needed);
    }

    // 检查截断的封装
    #define SAFE_SNPRINTF(buf, fmt, ...) do { \
        if (snprintf(buf, sizeof(buf), fmt, __VA_ARGS__) >= (int)sizeof(buf)) { \
            fprintf(stderr, "警告: 字符串截断\n"); \
        } \
    } while(0)

    SAFE_SNPRINTF(buf, "Value: %d", 12345);

    return 0;
}
```

## 四、安全编程实践

### 4.1 封装安全操作

```c
#include <string.h>
#include <stdio.h>

// 安全的字符串复制
int safe_copy(char *dest, size_t dest_size, const char *src) {
    if (!dest || !src || dest_size == 0) {
        return -1;
    }
    if (strlen(src) >= dest_size) {
        return -1;  // 空间不足
    }
    strcpy(dest, src);
    return 0;
}

// 安全的字符串连接
int safe_concat(char *dest, size_t dest_size, const char *src) {
    if (!dest || !src || dest_size == 0) {
        return -1;
    }
    size_t dest_len = strlen(dest);
    size_t src_len = strlen(src);
    if (dest_len + src_len >= dest_size) {
        return -1;
    }
    strcat(dest, src);
    return 0;
}

// 安全的字符串格式化
int safe_format(char *dest, size_t dest_size, const char *fmt, ...) {
    va_list args;
    va_start(args, fmt);
    int result = vsnprintf(dest, dest_size, fmt, args);
    va_end(args);

    if (result < 0 || (size_t)result >= dest_size) {
        return -1;
    }
    return result;
}
```

### 4.2 编译器安全选项

```bash
# GCC/Clang 安全选项
gcc -D_FORTIFY_SOURCE=2 -O2 program.c    # 缓冲区溢出检测
gcc -fstack-protector-all program.c       # 栈保护
gcc -Wformat -Wformat-security program.c  # 格式化字符串警告

# MSVC

```

## 五、各平台安全函数对比

| 函数 | C11 Annex K | POSIX | Windows | 说明 |
|------|-------------|-------|---------|------|
| `strcpy_s` | 标准 | 部分 | 支持 | 安全复制 |
| `strncpy_s` | 标准 | 部分 | 支持 | 限定长度安全复制 |
| `strcat_s` | 标准 | 部分 | 支持 | 安全连接 |
| `strtok_s` | 标准 | `strtok_r` | 支持 | 可重入分割 |
| `snprintf` | C99标准 | 支持 | `_snprintf` | 安全格式化 |
| `strdup` | 无 | POSIX | `_strdup` | 动态复制 |

## 六、重要注意事项

> **关键要点：**
> 1. **Annex K 不是强制实现**：许多编译器不完全支持，需要检查可用性
> 2. **`snprintf` 是最广泛支持的安全函数**：C99标准，推荐优先使用
> 3. **安全函数在失败时可能清空目标缓冲区**：注意检查返回值
> 4. **开启编译器的安全选项**：比手动使用安全函数更有效
> 5. **防御性编程**：始终检查输入参数的有效性
> 6. **不要过度依赖安全函数**：它们只是最后一道防线，根本方法是正确的逻辑
