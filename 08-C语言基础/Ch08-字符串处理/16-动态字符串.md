# 16 - 动态字符串

## 一、动态分配字符串基础

### 1.1 使用 malloc 分配

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    // 分配空间并复制
    const char *src = "Hello, World!";
    char *str = (char *)malloc(strlen(src) + 1);  // +1 给 \0

    if (str == NULL) {
        fprintf(stderr, "内存分配失败\n");
        return 1;
    }

    strcpy(str, src);
    printf("%s\n", str);

    free(str);  // 必须释放
    return 0;
}
```

### 1.2 使用 calloc 分配（自动初始化为0）

```c
// calloc 分配并清零
char *str = (char *)calloc(100, sizeof(char));
if (str) {
    // str 已经全部是 \0，可以当作空字符串使用
    strcat(str, "Hello");
    printf("%s\n", str);  // Hello
    free(str);
}
```

### 1.3 使用 strdup 复制（POSIX）

```c
// POSIX 标准函数，不是C标准库
// char *strdup(const char *s);

char *str = strdup("Hello");
if (str) {
    printf("%s\n", str);
    free(str);  // 需要 free
}
```

## 二、动态字符串扩展

### 2.1 使用 realloc 扩展

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    // 初始分配
    size_t capacity = 16;
    char *str = (char *)malloc(capacity);
    strcpy(str, "Hello");

    printf("初始: '%s' (容量=%zu)\n", str, capacity);

    // 需要更多空间
    size_t new_capacity = 64;
    char *temp = (char *)realloc(str, new_capacity);
    if (temp == NULL) {
        free(str);
        return 1;
    }
    str = temp;
    capacity = new_capacity;

    strcat(str, ", World!");
    printf("扩展后: '%s' (容量=%zu)\n", str, capacity);

    free(str);
    return 0;
}
```

### 2.2 倍增策略

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 动态字符串结构
typedef struct {
    char *data;
    size_t length;    // 当前长度
    size_t capacity;  // 总容量
} DynamicString;

// 创建
DynamicString *ds_create(size_t initial_capacity) {
    DynamicString *ds = (DynamicString *)malloc(sizeof(DynamicString));
    if (!ds) return NULL;

    ds->data = (char *)malloc(initial_capacity);
    if (!ds->data) {
        free(ds);
        return NULL;
    }

    ds->data[0] = '\0';
    ds->length = 0;
    ds->capacity = initial_capacity;

    return ds;
}

// 确保有足够空间
static int ds_ensure_capacity(DynamicString *ds, size_t needed) {
    if (ds->length + needed + 1 <= ds->capacity) {
        return 0;  // 空间足够
    }

    // 倍增策略
    size_t new_cap = ds->capacity;
    while (new_cap < ds->length + needed + 1) {
        new_cap *= 2;
    }

    char *temp = (char *)realloc(ds->data, new_cap);
    if (!temp) return -1;

    ds->data = temp;
    ds->capacity = new_cap;
    return 0;
}

// 追加字符串
int ds_append(DynamicString *ds, const char *str) {
    size_t len = strlen(str);
    if (ds_ensure_capacity(ds, len) != 0) {
        return -1;
    }
    memcpy(ds->data + ds->length, str, len + 1);
    ds->length += len;
    return 0;
}

// 释放
void ds_free(DynamicString *ds) {
    if (ds) {
        free(ds->data);
        free(ds);
    }
}

int main() {
    DynamicString *ds = ds_create(8);

    // 动态追加
    ds_append(ds, "Hello");
    ds_append(ds, ", ");
    ds_append(ds, "World!");
    ds_append(ds, " This is a longer string that will cause reallocation.");

    printf("内容: %s\n", ds->data);
    printf("长度: %zu, 容量: %zu\n", ds->length, ds->capacity);

    ds_free(ds);
    return 0;
}
```

## 三、字符串池（String Pool）

### 3.1 避免重复分配

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define POOL_SIZE 100

typedef struct {
    char *strings[POOL_SIZE];
    int count;
} StringPool;

// 字符串池：相同字符串只存储一份
const char *pool_intern(StringPool *pool, const char *str) {
    // 查找是否已存在
    for (int i = 0; i < pool->count; i++) {
        if (strcmp(pool->strings[i], str) == 0) {
            return pool->strings[i];  // 返回已有字符串
        }
    }

    // 不存在，添加新字符串
    if (pool->count >= POOL_SIZE) {
        return NULL;  // 池满了
    }

    pool->strings[pool->count] = strdup(str);
    return pool->strings[pool->count++];
}

void pool_free(StringPool *pool) {
    for (int i = 0; i < pool->count; i++) {
        free(pool->strings[i]);
    }
    pool->count = 0;
}
```

## 四、字符串连接的高效实现

### 4.1 不使用 strcat

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 高效连接多个字符串
char *join_strings(const char *parts[], int count, const char *separator) {
    // 1. 计算总长度
    size_t total = 0;
    size_t sep_len = strlen(separator);

    for (int i = 0; i < count; i++) {
        total += strlen(parts[i]);
        if (i < count - 1) {
            total += sep_len;
        }
    }

    // 2. 分配内存
    char *result = (char *)malloc(total + 1);
    if (!result) return NULL;

    // 3. 一次性构建
    char *p = result;
    for (int i = 0; i < count; i++) {
        size_t len = strlen(parts[i]);
        memcpy(p, parts[i], len);
        p += len;

        if (i < count - 1) {
            memcpy(p, separator, sep_len);
            p += sep_len;
        }
    }
    *p = '\0';

    return result;
}

int main() {
    const char *words[] = {"Hello", "World", "C", "Language"};
    char *sentence = join_strings(words, 4, " ");

    printf("%s\n", sentence);  // Hello World C Language
    free(sentence);

    return 0;
}
```

## 五、重要注意事项

> **关键要点：**
> 1. **malloc 后检查返回值**：分配可能失败
> 2. **realloc 可能移动内存**：使用返回值更新指针
> 3. **realloc 返回 NULL 时不释放原内存**：用临时指针保存
> 4. **倍增策略减少 realloc 调用次数**：摊还 O(1)
> 5. **每个 malloc 必须对应一个 free**：避免内存泄漏
> 6. **动态字符串比固定数组更灵活**：但也更复杂
