# 12 - 字符串格式化

## 一、sprintf 格式化输出到字符串

### 1.1 函数原型

```c
#include <stdio.h>
int sprintf(char *str, const char *format, ...);
```

- 将格式化结果写入 `str`
- 返回值：写入的字符数（不含 `\0`）
- **不检查目标缓冲区大小**，可能溢出

### 1.2 基本示例

```c
#include <stdio.h>

int main() {
    char buf[100];

    // 基本格式化
    sprintf(buf, "Hello, %s!", "World");
    printf("%s\n", buf);  // Hello, World!

    // 数字格式化
    sprintf(buf, "Age: %d, Score: %.2f", 25, 95.5);
    printf("%s\n", buf);  // Age: 25, Score: 95.50

    // 多种类型混合
    sprintf(buf, "%s has %d apples and %.1f yuan",
            "Alice", 5, 23.7);
    printf("%s\n", buf);

    return 0;
}
```

## 二、snprintf 安全格式化（推荐）

### 2.1 函数原型

```c
#include <stdio.h>
int snprintf(char *str, size_t size, const char *format, ...);
```

- 最多写入 `size - 1` 个字符，始终添加 `\0`
- 返回值：**期望**写入的字符数（不含 `\0`）
- 返回值 >= size 时表示发生了截断

### 2.2 基本示例

```c
#include <stdio.h>

int main() {
    char buf[20];

    // 安全格式化
    int needed = snprintf(buf, sizeof(buf),
                          "Hello, %s! Count: %d", "World", 12345);

    printf("buf: %s\n", buf);       // Hello, World! Count
    printf("需要: %d 字节\n", needed); // 需要: 24 字节

    if (needed >= (int)sizeof(buf)) {
        printf("发生了截断！\n");
    }

    return 0;
}
```

### 2.3 动态计算所需大小

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdarg.h>

// 动态分配足够大的字符串
char *format_string(const char *fmt, ...) {
    // 先计算需要的大小
    va_list args1, args2;
    va_start(args1, fmt);
    va_copy(args2, args1);

    int needed = vsnprintf(NULL, 0, fmt, args1) + 1;
    va_end(args1);

    // 分配内存
    char *buf = (char *)malloc(needed);
    if (buf) {
        vsnprintf(buf, (size_t)needed, fmt, args2);
    }
    va_end(args2);

    return buf;
}

int main() {
    char *msg = format_string("Name: %s, Age: %d, GPA: %.2f",
                               "Bob", 20, 3.85);
    if (msg) {
        printf("%s\n", msg);  // Name: Bob, Age: 20, GPA: 3.85
        free(msg);
    }
    return 0;
}
```

## 三、格式控制详解

### 3.1 整数格式

```c
#include <stdio.h>

int main() {
    char buf[100];
    int n = 42;

    sprintf(buf, "%d", n);       // "42"
    sprintf(buf, "%5d", n);      // "   42" (宽度5，右对齐)
    sprintf(buf, "%-5d", n);     // "42   " (宽度5，左对齐)
    sprintf(buf, "%05d", n);     // "00042" (前导零填充)
    sprintf(buf, "%+d", n);      // "+42"   (显示正号)
    sprintf(buf, "% d", n);      // " 42"   (正数前加空格)

    sprintf(buf, "%x", 255);     // "ff"    (小写十六进制)
    sprintf(buf, "%X", 255);     // "FF"    (大写十六进制)
    sprintf(buf, "%#x", 255);    // "0xff"  (带前缀)
    sprintf(buf, "%o", 255);     // "377"   (八进制)
    sprintf(buf, "%u", 42);      // "42"    (无符号)

    return 0;
}
```

### 3.2 浮点格式

```c
#include <stdio.h>

int main() {
    char buf[100];
    double pi = 3.14159265358979;

    sprintf(buf, "%f", pi);      // "3.141593" (默认6位小数)
    sprintf(buf, "%.2f", pi);    // "3.14"     (2位小数)
    sprintf(buf, "%10.2f", pi);  // "      3.14"
    sprintf(buf, "%-10.2f", pi); // "3.14      "
    sprintf(buf, "%e", pi);      // "3.141593e+00" (科学计数法)
    sprintf(buf, "%g", pi);      // "3.14159"  (自动选择f或e)
    sprintf(buf, "%a", pi);      // "0x1.921fb54442d18p+1" (十六进制浮点)

    return 0;
}
```

### 3.3 字符串格式

```c
#include <stdio.h>

int main() {
    char buf[100];

    sprintf(buf, "%s", "Hello");     // "Hello"
    sprintf(buf, "%10s", "Hello");   // "     Hello" (右对齐)
    sprintf(buf, "%-10s", "Hello");  // "Hello     " (左对齐)
    sprintf(buf, "%.3s", "Hello");   // "Hel" (截断)
    sprintf(buf, "%10.3s", "Hello"); // "       Hel"

    // 转义 %
    sprintf(buf, "100%% complete");  // "100% complete"

    return 0;
}
```

### 3.4 其他格式

```c
#include <stdio.h>

int main() {
    char buf[100];

    sprintf(buf, "%c", 'A');         // "A"
    sprintf(buf, "%%");              // "%"
    sprintf(buf, "%p", (void *)buf); // "0x7fff5fbff8a0" (指针)

    // 长度修饰符
    sprintf(buf, "%ld", 123456789L);    // long
    sprintf(buf, "%lld", 123456789LL);  // long long
    sprintf(buf, "%zu", sizeof(int));   // size_t

    // 返回写入的字符数
    int count = sprintf(buf, "Hello");
    printf("写入了 %d 个字符\n", count);  // 5

    return 0;
}
```

## 四、sscanf 从字符串解析

```c
#include <stdio.h>

int main() {
    char str[] = "Alice 25 3.85";
    char name[20];
    int age;
    double gpa;

    sscanf(str, "%s %d %f", name, &age, &gpa);
    printf("姓名: %s, 年龄: %d, GPA: %.2f\n", name, age, gpa);

    return 0;
}
```

## 五、重要注意事项

> **关键要点：**
> 1. **优先使用 `snprintf`**：避免缓冲区溢出
> 2. **`snprintf` 返回值 >= size 表示截断**：可用于判断
> 3. **`snprintf(NULL, 0, fmt, ...)` 计算所需大小**：C99特性
> 4. **格式字符串要和参数类型匹配**：不匹配会导致未定义行为
> 5. **`%n` 可写入内存**：安全敏感环境应避免使用
> 6. **浮点精度注意**：`%.20f` 可能显示浮点误差
