# 18 - 字符串排序

## 一、字符串比较基础

### 1.1 字典序（Lexicographic Order）

字符串按字典序比较：逐字符按 ASCII 值比较，直到找到不同字符或到达字符串末尾。

```c
#include <stdio.h>
#include <string.h>

int main() {
    // 字典序比较示例
    printf("%d\n", strcmp("apple", "banana"));   // 负数: apple < banana
    printf("%d\n", strcmp("banana", "apple"));   // 正数: banana > apple
    printf("%d\n", strcmp("apple", "apple"));     // 0: 相等

    // 注意：大写字母 < 小写字母（ASCII）
    printf("%d\n", strcmp("Apple", "apple"));    // 负数: 'A' < 'a'

    // 长度影响
    printf("%d\n", strcmp("app", "apple"));      // 负数: 短的 < 长的

    return 0;
}
```

## 二、使用 qsort 排序字符串数组

### 2.1 排序二维字符数组

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// 比较函数：二维数组
int compare_2d(const void *a, const void *b) {
    // a 和 b 指向数组的一行
    return strcmp((const char *)a, (const char *)b);
}

int main() {
    char fruits[][20] = {
        "Banana", "Apple", "Cherry", "Date",
        "Elderberry", "Fig", "Grape"
    };
    int n = sizeof(fruits) / sizeof(fruits[0]);

    // 排序
    qsort(fruits, n, sizeof(fruits[0]), compare_2d);

    // 打印
    for (int i = 0; i < n; i++) {
        printf("%s\n", fruits[i]);
    }
    // Apple
    // Banana
    // Cherry
    // Date
    // Elderberry
    // Fig
    // Grape

    return 0;
}
```

### 2.2 排序指针数组（更高效）

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// 比较函数：指针数组
int compare_ptrs(const void *a, const void *b) {
    // a 和 b 指向 char*
    return strcmp(*(const char **)a, *(const char **)b);
}

int main() {
    const char *fruits[] = {
        "Banana", "Apple", "Cherry", "Date",
        "Elderberry", "Fig", "Grape"
    };
    int n = sizeof(fruits) / sizeof(fruits[0]);

    // 排序（只交换指针，不移动字符串内容）
    qsort(fruits, n, sizeof(fruits[0]), compare_ptrs);

    for (int i = 0; i < n; i++) {
        printf("%s\n", fruits[i]);
    }

    return 0;
}
```

### 2.3 降序排序

```c
int compare_desc(const void *a, const void *b) {
    return strcmp(*(const char **)b, *(const char **)a);  // 交换 a 和 b
}
```

## 三、自定义排序规则

### 3.1 忽略大小写排序

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>

int compare_ignore_case(const void *a, const void *b) {
    const char *s1 = *(const char **)a;
    const char *s2 = *(const char **)b;

    while (*s1 && *s2) {
        int diff = tolower((unsigned char)*s1) - tolower((unsigned char)*s2);
        if (diff != 0) return diff;
        s1++;
        s2++;
    }
    return tolower((unsigned char)*s1) - tolower((unsigned char)*s2);
}

int main() {
    const char *words[] = {"Banana", "apple", "Cherry", "date"};
    int n = 4;

    qsort(words, n, sizeof(char *), compare_ignore_case);

    for (int i = 0; i < n; i++) {
        printf("%s\n", words[i]);
    }
    // apple
    // Banana
    // Cherry
    // date

    return 0;
}
```

### 3.2 按字符串长度排序

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int compare_by_length(const void *a, const void *b) {
    size_t len_a = strlen(*(const char **)a);
    size_t len_b = strlen(*(const char **)b);

    if (len_a < len_b) return -1;
    if (len_a > len_b) return 1;
    return 0;  // 长度相同时保持原序或按字典序
}

// 长度相同按字典序
int compare_by_length_then_alpha(const void *a, const void *b) {
    size_t len_a = strlen(*(const char **)a);
    size_t len_b = strlen(*(const char **)b);

    if (len_a != len_b) {
        return (len_a < len_b) ? -1 : 1;
    }
    return strcmp(*(const char **)a, *(const char **)b);
}
```

### 3.3 按最后字符排序

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int compare_by_last_char(const void *a, const void *b) {
    const char *s1 = *(const char **)a;
    const char *s2 = *(const char **)b;

    char last1 = s1[strlen(s1) - 1];
    char last2 = s2[strlen(s2) - 1];

    return last1 - last2;
}
```

## 四、排序结构体中的字符串

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    char name[50];
    int age;
    double score;
} Student;

// 按姓名排序
int compare_by_name(const void *a, const void *b) {
    return strcmp(((const Student *)a)->name,
                  ((const Student *)b)->name);
}

// 按成绩排序（降序）
int compare_by_score_desc(const void *a, const void *b) {
    double diff = ((const Student *)b)->score -
                  ((const Student *)a)->score;
    if (diff > 0) return 1;
    if (diff < 0) return -1;
    return 0;
}

int main() {
    Student students[] = {
        {"张三", 20, 85.5},
        {"李四", 21, 92.0},
        {"王五", 19, 78.3},
        {"赵六", 22, 95.7}
    };
    int n = 4;

    // 按姓名排序
    qsort(students, n, sizeof(Student), compare_by_name);

    printf("按姓名排序:\n");
    for (int i = 0; i < n; i++) {
        printf("  %s: %.1f\n", students[i].name, students[i].score);
    }

    // 按成绩排序
    qsort(students, n, sizeof(Student), compare_by_score_desc);

    printf("按成绩排序:\n");
    for (int i = 0; i < n; i++) {
        printf("  %s: %.1f\n", students[i].name, students[i].score);
    }

    return 0;
}
```

## 五、二分查找已排序字符串

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main() {
    const char *dict[] = {
        "apple", "banana", "cherry", "date",
        "elderberry", "fig", "grape"
    };
    int n = 7;

    // 确保已排序
    // 使用 bsearch
    const char *key = "cherry";
    const char **result = (const char **)bsearch(
        &key, dict, n, sizeof(char *),
        (int (*)(const void *, const void *))strcmp
    );

    if (result) {
        printf("找到 '%s' 在位置 %ld\n", *result, result - dict);
    } else {
        printf("未找到\n");
    }

    return 0;
}
```

## 六、重要注意事项

> **关键要点：**
> 1. **`qsort` 的比较函数必须返回负数/零/正数**：不能只返回 0/1
> 2. **指针数组排序比二维数组更高效**：只交换指针，不复制字符串
> 3. **注意 `const` 在比较函数中的使用**：确保不修改数据
> 4. **大小写敏感是默认行为**：需要自定义比较函数处理大小写
> 5. **qsort 不保证稳定排序**：相等元素的相对顺序可能改变
> 6. **中文字符串排序取决于编码**：UTF-8下按字节值排序，不是拼音序
