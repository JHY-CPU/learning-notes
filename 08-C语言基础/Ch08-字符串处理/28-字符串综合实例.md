# 28 - 字符串综合实例

## 一、简易文本编辑器核心

### 1.1 数据结构设计

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_LINES    10000
#define MAX_LINE_LEN 1024

typedef struct {
    char **lines;       // 每行一个动态字符串
    int line_count;     // 当前行数
    int capacity;       // 总容量
} TextBuffer;

// 创建文本缓冲区
TextBuffer *tb_create() {
    TextBuffer *tb = (TextBuffer *)malloc(sizeof(TextBuffer));
    tb->capacity = 64;
    tb->line_count = 0;
    tb->lines = (char **)malloc((size_t)tb->capacity * sizeof(char *));
    return tb;
}

// 释放文本缓冲区
void tb_free(TextBuffer *tb) {
    for (int i = 0; i < tb->line_count; i++) {
        free(tb->lines[i]);
    }
    free(tb->lines);
    free(tb);
}
```

### 1.2 行操作

```c
// 确保容量
static void tb_ensure_capacity(TextBuffer *tb) {
    if (tb->line_count >= tb->capacity) {
        tb->capacity *= 2;
        tb->lines = (char **)realloc(tb->lines,
                     (size_t)tb->capacity * sizeof(char *));
    }
}

// 在指定行后插入
int tb_insert_line(TextBuffer *tb, int after, const char *text) {
    if (after < -1 || after >= tb->line_count) return -1;
    tb_ensure_capacity(tb);

    int pos = after + 1;

    // 向后移动
    memmove(tb->lines + pos + 1, tb->lines + pos,
            (size_t)(tb->line_count - pos) * sizeof(char *));

    tb->lines[pos] = strdup(text);
    tb->line_count++;

    return pos;
}

// 删除指定行
int tb_delete_line(TextBuffer *tb, int line) {
    if (line < 0 || line >= tb->line_count) return -1;

    free(tb->lines[line]);
    memmove(tb->lines + line, tb->lines + line + 1,
            (size_t)(tb->line_count - line - 1) * sizeof(char *));
    tb->line_count--;

    return 0;
}

// 替换指定行
int tb_replace_line(TextBuffer *tb, int line, const char *text) {
    if (line < 0 || line >= tb->line_count) return -1;

    free(tb->lines[line]);
    tb->lines[line] = strdup(text);

    return 0;
}
```

### 1.3 查找和替换

```c
// 在指定行中查找字符串
int tb_find_in_line(TextBuffer *tb, int line, const char *pattern) {
    if (line < 0 || line >= tb->line_count) return -1;

    char *pos = strstr(tb->lines[line], pattern);
    if (pos) {
        return (int)(pos - tb->lines[line]);
    }
    return -1;
}

// 全局查找
void tb_find_all(TextBuffer *tb, const char *pattern) {
    for (int i = 0; i < tb->line_count; i++) {
        char *search = tb->lines[i];
        char *found;
        while ((found = strstr(search, pattern)) != NULL) {
            printf("行 %d, 列 %ld: %s\n",
                   i + 1, found - tb->lines[i], tb->lines[i]);
            search = found + 1;
        }
    }
}

// 替换行中的字符串
int tb_replace_in_line(TextBuffer *tb, int line,
                       const char *old_str, const char *new_str) {
    if (line < 0 || line >= tb->line_count) return -1;

    char *found = strstr(tb->lines[line], old_str);
    if (!found) return -1;

    size_t old_len = strlen(old_str);
    size_t new_len = strlen(new_str);
    size_t line_len = strlen(tb->lines[line]);

    char *new_line = (char *)malloc(line_len - old_len + new_len + 1);

    // 构建新行
    size_t prefix_len = (size_t)(found - tb->lines[line]);
    memcpy(new_line, tb->lines[line], prefix_len);
    memcpy(new_line + prefix_len, new_str, new_len);
    strcpy(new_line + prefix_len + new_len,
           found + old_len);

    free(tb->lines[line]);
    tb->lines[line] = new_line;

    return 0;
}
```

### 1.4 文件读写

```c
// 从文件加载
int tb_load_file(TextBuffer *tb, const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return -1;

    char line[MAX_LINE_LEN];
    while (fgets(line, sizeof(line), fp)) {
        // 去除换行符
        size_t len = strlen(line);
        if (len > 0 && line[len - 1] == '\n') {
            line[len - 1] = '\0';
        }
        tb_ensure_capacity(tb);
        tb->lines[tb->line_count++] = strdup(line);
    }

    fclose(fp);
    return tb->line_count;
}

// 保存到文件
int tb_save_file(TextBuffer *tb, const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (!fp) return -1;

    for (int i = 0; i < tb->line_count; i++) {
        fprintf(fp, "%s\n", tb->lines[i]);
    }

    fclose(fp);
    return tb->line_count;
}

// 打印缓冲区内容
void tb_print(TextBuffer *tb, int start, int end) {
    if (start < 0) start = 0;
    if (end < 0 || end >= tb->line_count) end = tb->line_count - 1;

    for (int i = start; i <= end; i++) {
        printf("%4d | %s\n", i + 1, tb->lines[i]);
    }
}
```

## 二、字符串处理综合练习

### 2.1 统计文本信息

```c
typedef struct {
    int total_chars;
    int total_words;
    int total_lines;
    int chinese_chars;
    int english_words;
    int digits;
    int spaces;
    int punctuations;
} TextStats;

void analyze_text(const char *text, TextStats *stats) {
    memset(stats, 0, sizeof(TextStats));

    const unsigned char *p = (const unsigned char *)text;
    int in_word = 0;

    while (*p) {
        stats->total_chars++;

        if (*p == '\n') {
            stats->total_lines++;
            in_word = 0;
        } else if (*p == ' ' || *p == '\t') {
            stats->spaces++;
            in_word = 0;
        } else if ((*p >= 'a' && *p <= 'z') ||
                   (*p >= 'A' && *p <= 'Z')) {
            stats->english_words += (!in_word);
            in_word = 1;
        } else if (*p >= '0' && *p <= '9') {
            stats->digits++;
        } else if (*p >= 0x80) {
            // UTF-8 多字节字符（可能是中文）
            stats->chinese_chars++;
            // 跳过续字节
            if ((*p & 0xE0) == 0xC0) p += 1;
            else if ((*p & 0xF0) == 0xE0) p += 2;
            else if ((*p & 0xF8) == 0xF0) p += 3;
        } else {
            stats->punctuations++;
            in_word = 0;
        }

        p++;
    }

    stats->total_lines++;  // 最后一行
    stats->total_words = stats->english_words + stats->chinese_chars;
}
```

### 2.2 模板引擎（简易版）

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 简易模板引擎：替换 {{变量名}}
// 使用方式: render("Hello, {{name}}!", "name", "Alice", NULL)

char *render_template(const char *template, ...) {
    // 先复制模板
    size_t cap = strlen(template) * 2 + 1;
    char *result = (char *)malloc(cap);
    strcpy(result, template);

    va_list args;
    va_start(args, template);

    const char *key;
    while ((key = va_arg(args, const char *)) != NULL) {
        const char *value = va_arg(args, const char *);
        if (!value) break;

        // 构建 {{key}} 模式
        char pattern[256];
        snprintf(pattern, sizeof(pattern), "{{%s}}", key);

        // 查找并替换
        char *pos;
        while ((pos = strstr(result, pattern)) != NULL) {
            size_t key_len = strlen(pattern);
            size_t val_len = strlen(value);
            size_t old_len = strlen(result);

            // 需要的空间
            size_t new_len = old_len - key_len + val_len;
            if (new_len + 1 > cap) {
                cap = new_len * 2 + 1;
                result = (char *)realloc(result, cap);
                pos = strstr(result, pattern);  // realloc 后重新定位
            }

            // 移动并替换
            memmove(pos + val_len, pos + key_len,
                    old_len - (size_t)(pos - result) - key_len + 1);
            memcpy(pos, value, val_len);
        }
    }

    va_end(args);
    return result;
}

int main() {
    char *html = render_template(
        "<h1>Hello, {{name}}!</h1>"
        "<p>Welcome to {{site}}.</p>"
        "<p>Your ID is {{id}}.</p>",
        "name", "Alice",
        "site", "My Website",
        "id", "12345",
        NULL
    );

    printf("%s\n", html);
    free(html);

    return 0;
}
```

### 2.3 命令行参数解析器

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    char *program;
    char **args;
    int arg_count;
    int max_args;
} Command;

Command *cmd_parse(const char *input) {
    Command *cmd = (Command *)calloc(1, sizeof(Command));
    cmd->max_args = 16;
    cmd->args = (char **)malloc((size_t)cmd->max_args * sizeof(char *));

    char *copy = strdup(input);
    char *saveptr;
    char *token = strtok_r(copy, " \t", &saveptr);

    while (token) {
        if (cmd->arg_count == 0) {
            cmd->program = strdup(token);
        } else {
            if (cmd->arg_count >= cmd->max_args) {
                cmd->max_args *= 2;
                cmd->args = (char **)realloc(cmd->args,
                    (size_t)cmd->max_args * sizeof(char *));
            }
            cmd->args[cmd->arg_count - 1] = strdup(token);
        }
        cmd->arg_count++;
        token = strtok_r(NULL, " \t", &saveptr);
    }

    free(copy);
    return cmd;
}

void cmd_free(Command *cmd) {
    free(cmd->program);
    for (int i = 0; i < cmd->arg_count - 1; i++) {
        free(cmd->args[i]);
    }
    free(cmd->args);
    free(cmd);
}

int main() {
    Command *cmd = cmd_parse("copy -r /src /dst");

    printf("程序: %s\n", cmd->program);
    printf("参数: %d 个\n", cmd->arg_count - 1);
    for (int i = 0; i < cmd->arg_count - 1; i++) {
        printf("  [%d] %s\n", i, cmd->args[i]);
    }

    cmd_free(cmd);
    return 0;
}
```

## 四、重要注意事项

> **关键要点：**
> 1. **动态字符串管理**：每次修改行内容后需要释放旧内存、分配新内存
> 2. **memmove 处理重叠**：行插入和删除时数组元素的移动
> 3. **strdup 简化字符串复制**：自动计算长度并分配内存
> 4. **始终检查 malloc 返回值**：避免 NULL 解引用
> 5. **UTF-8 处理要小心**：一个字符可能占多个字节
> 6. **实际项目建议使用成熟库**：如 GLib、SDS 等
