# 25 - 字符串哈希

## 一、哈希函数基础

### 1.1 什么是字符串哈希

将变长字符串映射为固定长度的整数值，用于快速比较、查找和存储。

```
字符串 "hello" → 哈希值 99162322
字符串 "world" → 哈希值 113318802
```

### 1.2 好的哈希函数特性

- 计算速度快
- 哈希值分布均匀（减少冲突）
- 确定性（同一输入始终产生同一输出）
- 尽量减少碰撞（不同输入产生相同输出）

## 二、经典哈希函数实现

### 2.1 DJB2 哈希（Daniel J. Bernstein）

```c
#include <stdio.h>
#include <stdint.h>

// DJB2 哈希 - 简单高效
uint32_t djb2_hash(const char *str) {
    uint32_t hash = 5381;
    int c;

    while ((c = *str++)) {
        // hash * 33 + c
        hash = ((hash << 5) + hash) + (unsigned char)c;
    }

    return hash;
}

int main() {
    printf("DJB2:\n");
    printf("  'hello' -> %u\n", djb2_hash("hello"));
    printf("  'world' -> %u\n", djb2_hash("world"));
    printf("  'Hello' -> %u\n", djb2_hash("Hello"));  // 大小写不同

    return 0;
}
```

### 2.2 BKDR 哈希（Brian Kernighan & Dennis Ritchie）

```c
// BKDR 哈希 - K&R C书中使用的种子
uint32_t bkdr_hash(const char *str) {
    uint32_t seed = 131;  // 31, 131, 1313, 13131, ...
    uint32_t hash = 0;

    while (*str) {
        hash = hash * seed + (unsigned char)*str;
        str++;
    }

    return hash;
}
```

### 2.3 SDBM 哈希

```c
// SDBM 哈希 - 开源项目 SDBM 使用
uint32_t sdbm_hash(const char *str) {
    uint32_t hash = 0;
    int c;

    while ((c = *str++)) {
        hash = (unsigned char)c + (hash << 6) + (hash << 16) - hash;
    }

    return hash;
}
```

### 2.4 FNV-1a 哈希

```c
// FNV-1a 哈希 - 分布均匀
uint32_t fnv1a_hash(const char *str) {
    uint32_t hash = 2166136261u;  // FNV offset basis

    while (*str) {
        hash ^= (unsigned char)*str;
        hash *= 16777619u;        // FNV prime
        str++;
    }

    return hash;
}
```

### 2.5 MurmurHash3（简版）

```c
#include <stdint.h>
#include <string.h>

// MurmurHash3 32位（简化版）
uint32_t murmur3_32(const char *key, uint32_t seed) {
    int len = (int)strlen(key);
    const uint8_t *data = (const uint8_t *)key;
    uint32_t h = seed;

    // 处理4字节块
    int nblocks = len / 4;
    for (int i = 0; i < nblocks; i++) {
        uint32_t k;
        memcpy(&k, data + i * 4, 4);

        k *= 0xcc9e2d51;
        k = (k << 15) | (k >> 17);
        k *= 0x1b873593;

        h ^= k;
        h = (h << 13) | (h >> 19);
        h = h * 5 + 0xe6546b64;
    }

    // 处理尾部
    const uint8_t *tail = data + nblocks * 4;
    uint32_t k1 = 0;
    switch (len & 3) {
        case 3: k1 ^= tail[2] << 16; /* fall through */
        case 2: k1 ^= tail[1] << 8;  /* fall through */
        case 1: k1 ^= tail[0];
                k1 *= 0xcc9e2d51;
                k1 = (k1 << 15) | (k1 >> 17);
                k1 *= 0x1b873593;
                h ^= k1;
    }

    // 最终混合
    h ^= (uint32_t)len;
    h ^= h >> 16;
    h *= 0x85ebca6b;
    h ^= h >> 13;
    h *= 0xc2b2ae35;
    h ^= h >> 16;

    return h;
}
```

## 三、哈希表实现

### 3.1 简单哈希表

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>

#define TABLE_SIZE 64

// 链表节点
typedef struct Node {
    char *key;
    int value;
    struct Node *next;
} Node;

// 哈希表
typedef struct {
    Node *buckets[TABLE_SIZE];
} HashTable;

// 哈希函数
static uint32_t hash_func(const char *str) {
    uint32_t hash = 5381;
    while (*str) {
        hash = ((hash << 5) + hash) + (unsigned char)*str++;
    }
    return hash % TABLE_SIZE;
}

// 创建
HashTable *ht_create() {
    HashTable *ht = (HashTable *)calloc(1, sizeof(HashTable));
    return ht;
}

// 插入
void ht_insert(HashTable *ht, const char *key, int value) {
    uint32_t index = hash_func(key);

    // 检查是否已存在
    Node *node = ht->buckets[index];
    while (node) {
        if (strcmp(node->key, key) == 0) {
            node->value = value;  // 更新
            return;
        }
        node = node->next;
    }

    // 新建节点
    Node *new_node = (Node *)malloc(sizeof(Node));
    new_node->key = strdup(key);
    new_node->value = value;
    new_node->next = ht->buckets[index];
    ht->buckets[index] = new_node;
}

// 查找
int *ht_get(HashTable *ht, const char *key) {
    uint32_t index = hash_func(key);
    Node *node = ht->buckets[index];

    while (node) {
        if (strcmp(node->key, key) == 0) {
            return &node->value;
        }
        node = node->next;
    }
    return NULL;
}

// 删除
void ht_delete(HashTable *ht, const char *key) {
    uint32_t index = hash_func(key);
    Node *node = ht->buckets[index];
    Node *prev = NULL;

    while (node) {
        if (strcmp(node->key, key) == 0) {
            if (prev) {
                prev->next = node->next;
            } else {
                ht->buckets[index] = node->next;
            }
            free(node->key);
            free(node);
            return;
        }
        prev = node;
        node = node->next;
    }
}

// 释放
void ht_free(HashTable *ht) {
    for (int i = 0; i < TABLE_SIZE; i++) {
        Node *node = ht->buckets[i];
        while (node) {
            Node *next = node->next;
            free(node->key);
            free(node);
            node = next;
        }
    }
    free(ht);
}

// 测试
int main() {
    HashTable *ht = ht_create();

    ht_insert(ht, "apple", 5);
    ht_insert(ht, "banana", 3);
    ht_insert(ht, "cherry", 8);

    int *val = ht_get(ht, "banana");
    if (val) printf("banana: %d\n", *val);  // 3

    ht_delete(ht, "banana");

    val = ht_get(ht, "banana");
    printf("删除后: %s\n", val ? "找到" : "未找到");

    ht_free(ht);
    return 0;
}
```

## 四、哈希函数对比

| 函数 | 速度 | 分布 | 实现复杂度 | 适用场景 |
|------|------|------|-----------|----------|
| DJB2 | 快 | 好 | 简单 | 通用 |
| BKDR | 快 | 好 | 简单 | 通用 |
| SDBM | 快 | 较好 | 简单 | 通用 |
| FNV-1a | 中 | 很好 | 简单 | 需要好分布 |
| Murmur3 | 快 | 最好 | 中等 | 高性能要求 |

## 五、滚动哈希（Rabin-Karp）

```c
#include <stdio.h>
#include <string.h>

// 滚动哈希：用于子串搜索
// O(n) 时间找到所有匹配

#define BASE 256
#define MOD  101

void rabin_karp(const char *text, const char *pattern) {
    int n = (int)strlen(text);
    int m = (int)strlen(pattern);

    // 计算最高位的权值
    int h = 1;
    for (int i = 0; i < m - 1; i++) {
        h = (h * BASE) % MOD;
    }

    // 计算模式串和文本第一个窗口的哈希
    int p_hash = 0, t_hash = 0;
    for (int i = 0; i < m; i++) {
        p_hash = (BASE * p_hash + pattern[i]) % MOD;
        t_hash = (BASE * t_hash + text[i]) % MOD;
    }

    // 滑动窗口
    for (int i = 0; i <= n - m; i++) {
        if (p_hash == t_hash) {
            // 哈希匹配，验证字符
            if (memcmp(text + i, pattern, m) == 0) {
                printf("匹配位置: %d\n", i);
            }
        }

        // 移动窗口：减去最高位，加上新字符
        if (i < n - m) {
            t_hash = (BASE * (t_hash - text[i] * h) + text[i + m]) % MOD;
            if (t_hash < 0) t_hash += MOD;
        }
    }
}
```

## 六、重要注意事项

> **关键要点：**
> 1. **哈希冲突不可避免**：需要处理策略（链地址法、开放地址法等）
> 2. **哈希值可能溢出**：使用 `uint32_t` 或 `uint64_t` 并依赖自然溢出
> 3. **相同字符串一定产生相同哈希值**：但不同字符串可能产生相同值
> 4. **不要用哈希值比较字符串相等**：哈希碰撞可能发生，必须逐字符比较
> 5. **哈希表负载因子**：超过 0.75 时考虑扩容
