# 09 - 信号处理详解

## 概述

信号（Signal）是Linux/Unix系统中进程间通信的一种异步通知机制。信号可以在任何时候发送给进程，进程需要注册信号处理函数来响应。

---

## 1. 常见信号

| 信号 | 编号 | 默认行为 | 说明 |
|------|------|----------|------|
| SIGHUP | 1 | 终止 | 终端挂起 |
| SIGINT | 2 | 终止 | Ctrl+C |
| SIGQUIT | 3 | 终止+core | Ctrl+\ |
| SIGKILL | 9 | 终止 | 强制杀死（不可捕获） |
| SIGSEGV | 11 | 终止+core | 段错误 |
| SIGPIPE | 13 | 终止 | 向已关闭管道写入 |
| SIGALRM | 14 | 终止 | 定时器超时 |
| SIGTERM | 15 | 终止 | kill默认信号 |
| SIGCHLD | 17 | 忽略 | 子进程状态变化 |
| SIGSTOP | 19 | 暂停 | 暂停进程（不可捕获） |

---

## 2. signal() 基本信号处理

```c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>

// 信号处理函数
void handler(int sig) {
    if (sig == SIGINT) {
        printf("\n收到SIGINT (Ctrl+C)，忽略它\n");
    }
}

int main(void) {
    // 注册信号处理函数
    signal(SIGINT, handler);
    signal(SIGQUIT, SIG_IGN);  // 忽略SIGQUIT

    printf("PID: %d\n", getpid());
    printf("按 Ctrl+C 测试（按 Ctrl+\\ 退出）\n");

    while (1) {
        pause();  // 等待信号
    }
    return 0;
}
```

## 3. sigaction() 高级信号处理

`sigaction`比`signal`更可靠，提供更精细的控制。

```c
#include <stdio.h>
#include <signal.h>
#include <string.h>
#include <unistd.h>

volatile sig_atomic_t got_signal = 0;  // 信号安全的标志

void sig_handler(int sig) {
    got_signal = 1;  // 只设置标志，不调用非异步安全函数
}

int main(void) {
    struct sigaction sa;
    memset(&sa, 0, sizeof(sa));
    sa.sa_handler = sig_handler;
    sigemptyset(&sa.sa_mask);    // 处理信号时阻塞的信号集
    sa.sa_flags = SA_RESTART;    // 被信号中断的系统调用自动重启

    sigaction(SIGINT, &sa, NULL);
    sigaction(SIGTERM, &sa, NULL);

    printf("等待信号... (PID=%d)\n", getpid());

    while (!got_signal) {
        pause();
    }

    printf("收到信号，安全退出\n");
    return 0;
}
```

## 4. 信号集操作

```c
#include <signal.h>

void signal_set_demo(void) {
    sigset_t set, oldset, pending;

    // 初始化信号集
    sigemptyset(&set);        // 清空
    sigfillset(&set);         // 填满所有信号

    // 添加/删除信号
    sigaddset(&set, SIGINT);
    sigaddset(&set, SIGTERM);
    sigdelset(&set, SIGINT);

    // 检查信号
    if (sigismember(&set, SIGINT))
        printf("SIGINT 在集合中\n");

    // 阻塞信号（不丢失，只是延迟处理）
    sigprocmask(SIG_BLOCK, &set, &oldset);    // 阻塞set中的信号
    // ... 临界区代码 ...
    sigprocmask(SIG_SETMASK, &oldset, NULL);  // 恢复

    // 检查挂起的信号
    sigpending(&pending);
    if (sigismember(&pending, SIGINT))
        printf("有SIGINT信号挂起\n");
}
```

## 5. 可靠信号与信号排队

```c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>

// 使用siginfo_t获取详细信号信息
void sig_action(int sig, siginfo_t *info, void *ucontext) {
    printf("信号: %d\n", sig);
    printf("发送者PID: %d\n", info->si_pid);
    printf("发送者UID: %d\n", info->si_uid);
    printf("信号值: %d\n", info->si_value.sival_int);
}

void reliable_signal(void) {
    struct sigaction sa;
    sa.sa_sigaction = sig_action;
    sa.sa_flags = SA_SIGINFO | SA_RESTART;  // 使用sa_sigaction
    sigemptyset(&sa.sa_mask);

    sigaction(SIGUSR1, &sa, NULL);

    printf("PID: %d，等待SIGUSR1...\n", getpid());
    pause();
}
```

## 6. 信号安全函数

信号处理函数中只能调用"异步信号安全"的函数（`man 7 signal-safety`）。

### 安全的函数

- `write()`（注意不是`printf`）
- `_exit()`
- `signal()`
- `sig_atomic_t`类型的变量访问

### 不安全的函数

- `printf()`、`fprintf()`
- `malloc()`、`free()`
- `mutex`操作

```c
// 正确做法：信号处理中只写pipe或设置volatile标志
volatile sig_atomic_t counter = 0;

void safe_handler(int sig) {
    counter++;           // 安全
    write(1, "信号!\n", 9);  // 安全
    // printf("信号!\n");   // 不安全！
}
```

## 7. 定时器信号

```c
#include <stdio.h>
#include <signal.h>
#include <sys/time.h>
#include <unistd.h>

volatile sig_atomic_t tick_count = 0;

void alarm_handler(int sig) {
    tick_count++;
    printf("滴答 #%d\n", tick_count);
}

void timer_demo(void) {
    signal(SIGALRM, alarm_handler);

    // 方式1：alarm（秒级）
    alarm(3);  // 3秒后发送SIGALRM

    // 方式2：setitimer（微秒级，可周期性）
    struct itimerval timer;
    timer.it_value.tv_sec = 1;     // 首次1秒后
    timer.it_value.tv_usec = 0;
    timer.it_interval.tv_sec = 1;  // 之后每1秒
    timer.it_interval.tv_usec = 0;
    setitimer(ITIMER_REAL, &timer, NULL);

    while (tick_count < 5) {
        pause();
    }
    printf("结束\n");
}
```

## 8. 完整示例：优雅退出的服务器

```c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>
#include <stdlib.h>

volatile sig_atomic_t running = 1;

void shutdown_handler(int sig) {
    (void)sig;
    running = 0;
}

int main(void) {
    struct sigaction sa = {
        .sa_handler = shutdown_handler,
        .sa_flags = 0
    };
    sigemptyset(&sa.sa_mask);

    sigaction(SIGINT, &sa, NULL);
    sigaction(SIGTERM, &sa, NULL);

    // 忽略SIGPIPE（避免写入关闭的socket导致程序终止）
    signal(SIGPIPE, SIG_IGN);

    printf("服务器启动 (PID=%d)\n", getpid());
    printf("按 Ctrl+C 优雅关闭\n");

    while (running) {
        // 服务器主循环
        printf("处理中...\n");
        sleep(1);
    }

    printf("\n收到关闭信号，清理资源...\n");
    // 温馨提示：这里关闭socket、释放内存等
    printf("服务器已关闭\n");

    return 0;
}
```

---

## 要点总结

1. 优先使用`sigaction`而非`signal`，前者更可靠且功能更强
2. 信号处理函数中只调用异步信号安全函数
3. 使用`volatile sig_atomic_t`在信号处理和主程序间通信
4. `SIGKILL`和`SIGSTOP`不可捕获
5. 服务器程序应忽略`SIGPIPE`，用`SIGTERM`/`SIGINT`实现优雅退出
6. 信号集操作配合`sigprocmask`可用于保护临界区
