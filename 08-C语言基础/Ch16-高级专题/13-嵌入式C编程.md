# 13 - 嵌入式C编程

## 概述

嵌入式C编程涉及直接操作硬件，包括寄存器访问、中断处理、内存映射等。本节介绍裸机和RTOS环境下的C编程技术。

---

## 1. volatile关键字

`volatile`告诉编译器不要优化对该变量的访问，每次读写都从内存中获取。

```c
// 典型用法：硬件寄存器
volatile uint32_t *GPIOA_ODR = (volatile uint32_t *)0x40020014;

// 控制LED
void led_on(void) {
    *GPIOA_ODR |= (1 << 5);   // PA5置高
}

void led_off(void) {
    *GPIOA_ODR &= ~(1 << 5);  // PA5置低
}

// 错误示例：没有volatile
// 编译器可能将循环优化为空，因为看起来没有副作用
void delay_bad(uint32_t count) {
    for (uint32_t i = 0; i < count; i++) {
        // 编译器可能删除这个循环！
    }
}

// 正确示例
void delay_good(uint32_t count) {
    for (volatile uint32_t i = 0; i < count; i++) {
        // 强制每次比较都从内存读取i
    }
}
```

## 2. 寄存器操作

### 位操作宏

```c
#include <stdint.h>

// 位操作宏
#define SET_BIT(reg, bit)    ((reg) |= (1U << (bit)))
#define CLR_BIT(reg, bit)    ((reg) &= ~(1U << (bit)))
#define TGL_BIT(reg, bit)    ((reg) ^= (1U << (bit)))
#define GET_BIT(reg, bit)    (((reg) >> (bit)) & 1U)

// 多位操作
#define SET_BITS(reg, mask, val) ((reg) = ((reg) & ~(mask)) | ((val) & (mask)))
#define GET_BITS(reg, mask)      ((reg) & (mask))

// 寄存器位域定义
typedef struct {
    volatile uint32_t MODER;    // 0x00 模式寄存器
    volatile uint32_t OTYPER;   // 0x04 输出类型
    volatile uint32_t OSPEEDR;  // 0x08 输出速度
    volatile uint32_t PUPDR;    // 0x0C 上拉/下拉
    volatile uint32_t IDR;      // 0x10 输入数据
    volatile uint32_t ODR;      // 0x14 输出数据
    volatile uint32_t BSRR;     // 0x18 置位/复位
    volatile uint32_t LCKR;     // 0x1C 锁定
    volatile uint32_t AFR[2];   // 0x20-0x24 复用功能
} GPIO_TypeDef;

#define GPIOA ((GPIO_TypeDef *)0x40020000)

// 使用位域结构体操作
void gpio_config(void) {
    // 配置PA5为输出模式
    GPIOA->MODER &= ~(3U << (5 * 2));  // 清除PA5模式位
    GPIOA->MODER |= (1U << (5 * 2));   // 设置为输出

    GPIOA->OTYPER &= ~(1U << 5);       // 推挽输出
    GPIOA->OSPEEDR |= (2U << (5 * 2)); // 高速
    GPIOA->PUPDR &= ~(3U << (5 * 2));  // 无上拉/下拉
}

// 使用BSRR进行原子操作
void gpio_set(GPIO_TypeDef *gpio, uint32_t pin) {
    gpio->BSRR = (1U << pin);           // 置位
}

void gpio_reset(GPIO_TypeDef *gpio, uint32_t pin) {
    gpio->BSRR = (1U << (pin + 16));    // 复位
}
```

## 3. 内存映射与链接脚本

```c
// 内存布局定义
#define FLASH_BASE  0x08000000
#define SRAM_BASE   0x20000000
#define PERIPH_BASE 0x40000000

// 启动代码中的向量表
__attribute__((section(".isr_vector")))
const uint32_t vector_table[] = {
    0x20010000,           // 初始栈指针（SRAM末尾）
    (uint32_t)Reset_Handler,
    (uint32_t)NMI_Handler,
    (uint32_t)HardFault_Handler,
    // ... 更多中断向量
};

// 链接脚本中定义的符号
extern uint32_t _sdata;   // .data段起始
extern uint32_t _edata;   // .data段结束
extern uint32_t _sidata;  // .data初始值在FLASH中的位置
extern uint32_t _sbss;    // .bss段起始
extern uint32_t _ebss;    // .bss段结束

// 启动时的数据初始化
void SystemInit(void) {
    // 复制.data段从Flash到SRAM
    uint32_t *src = &_sidata;
    uint32_t *dst = &_sdata;
    while (dst < &_edata) {
        *dst++ = *src++;
    }

    // 清零.bss段
    dst = &_sbss;
    while (dst < &_ebss) {
        *dst++ = 0;
    }
}
```

## 4. 中断处理

```c
// 中断处理函数声明
void USART1_IRQHandler(void) __attribute__((interrupt));
void TIM2_IRQHandler(void) __attribute__((interrupt));

// 中断优先级
#define NVIC_PRIO_GROUP_2  0x05  // 2位抢占 + 2位子优先级

void nvic_enable_irq(int irq, int priority) {
    // 设置优先级
    volatile uint32_t *NVIC_IPR = (volatile uint32_t *)(0xE000E400 + irq);
    *NVIC_IPR = (priority << 4);

    // 使能中断
    volatile uint32_t *NVIC_ISER = (volatile uint32_t *)(0xE000E100 + (irq / 32) * 4);
    *NVIC_ISER = (1U << (irq % 32));
}

// 中断处理函数
volatile uint8_t rx_buffer[256];
volatile uint16_t rx_head = 0;
volatile uint16_t rx_tail = 0;

void USART1_IRQHandler(void) {
    // 检查接收中断标志
    if (USART1->SR & (1U << 5)) {  // RXNE
        uint8_t data = (uint8_t)USART1->DR;
        rx_buffer[rx_head++] = data;
    }
}

// 关键区域保护
static inline uint32_t enter_critical(void) {
    uint32_t primask;
    __asm__ volatile ("mrs %0, primask" : "=r"(primask));
    __asm__ volatile ("cpsid i" ::: "memory");
    return primask;
}

static inline void exit_critical(uint32_t primask) {
    if (!(primask & 1)) {
        __asm__ volatile ("cpsie i" ::: "memory");
    }
}
```

## 5. 看门狗与定时器

```c
// 独立看门狗
typedef struct {
    volatile uint32_t KR;    // 键寄存器
    volatile uint32_t PR;    // 预分频
    volatile uint32_t RLR;   // 重装载值
    volatile uint32_t SR;    // 状态
} IWDG_TypeDef;

#define IWDG ((IWDG_TypeDef *)0x40003000)

void iwdg_init(uint16_t reload, uint8_t prescaler) {
    IWDG->KR = 0x5555;   // 解锁
    IWDG->PR = prescaler;
    IWDG->RLR = reload;
    IWDG->KR = 0xAAAA;   // 重装载
    IWDG->KR = 0xCCCC;   // 启动
}

void iwdg_feed(void) {
    IWDG->KR = 0xAAAA;
}

// 系统滴答定时器
volatile uint32_t systick_count = 0;

void SysTick_Handler(void) {
    systick_count++;
}

uint32_t millis(void) {
    return systick_count;
}

void delay_ms(uint32_t ms) {
    uint32_t start = systick_count;
    while ((systick_count - start) < ms);
}
```

## 6. 低功耗模式

```c
// 进入睡眠模式
void enter_sleep(void) {
    __asm__ volatile ("wfi");  // 等待中断
}

// 进入停止模式（STM32示例）
void enter_stop(void) {
    // 配置进入停止模式
    volatile uint32_t *SCB_SCR = (volatile uint32_t *)0xE000ED10;
    *SCB_SCR |= (1U << 2);  // SLEEPDEEP

    volatile uint32_t *PWR_CR = (volatile uint32_t *)0x40007000;
    *PWR_CR &= ~(1U << 0);  // LPDS = 0

    __asm__ volatile ("wfi");

    *SCB_SCR &= ~(1U << 2); // 清除SLEEPDEEP
}
```

## 要点总结

1. `volatile`是嵌入式编程的基石，用于硬件寄存器和中断共享变量
2. 使用位操作宏简化寄存器操作
3. 中断处理函数要尽可能简短，避免阻塞
4. 临界区保护使用关中断或原子操作
5. 链接脚本定义内存布局，启动代码负责初始化
6. 注意字节序、对齐和内存屏障
