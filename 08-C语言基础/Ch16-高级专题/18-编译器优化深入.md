# 18 - 编译器优化深入

## 概述

现代编译器（GCC/Clang）提供了大量优化手段。本节深入探讨编译器优化的工作原理和如何利用它们提升代码性能。

---

## 1. 优化级别详解

```bash
# GCC优化级别
gcc -O0  # 无优化：快速编译，便于调试
gcc -O1  # 基础优化：减少代码大小和执行时间
gcc -O2  # 推荐优化：几乎所有不涉及空间-速度权衡的优化
gcc -O3  # 激进优化：包含循环展开、向量化等
gcc -Os  # 大小优化：优化代码体积
gcc -Ofast # O3 + 不严格遵守标准
gcc -Og   # 调试友好优化

# 查看某个优化级别包含哪些具体选项
gcc -O2 -Q --help=optimizers
```

### 各级别包含的主要优化

| 优化 | -O1 | -O2 | -O3 |
|------|-----|-----|-----|
| 常量折叠 | Y | Y | Y |
| 死代码消除 | Y | Y | Y |
| 公共子表达式消除 | Y | Y | Y |
| 函数内联 | Y | Y | Y |
| 循环不变量外提 | Y | Y | Y |
| 寄存器分配 | Y | Y | Y |
| 指令调度 | - | Y | Y |
| 循环展开 | - | Y | Y |
| 向量化 | - | - | Y |
| 循环交换 | - | - | Y |

## 2. 内联优化

```c
// 编译器自动内联决策
// 小函数、频繁调用的函数会被自动内联

static int add(int a, int b) {
    return a + b;
}

// 强制内联
__attribute__((always_inline))
static inline int fast_add(int a, int b) {
    return a + b;
}

// 禁止内联
__attribute__((noinline))
void dont_inline(void) {
    // 调试或性能分析用途
}

// 查看内联决策
// gcc -O2 -fopt-info-inline app.c

void inline_example(void) {
    int x = add(1, 2);      // 可能被内联
    int y = fast_add(3, 4); // 一定被内联
}
```

## 3. 循环优化

### 循环展开

```c
// 源码
int sum_original(int *arr, int n) {
    int sum = 0;
    for (int i = 0; i < n; i++)
        sum += arr[i];
    return sum;
}

// 编译器自动展开 (-O2/-O3 + -funroll-loops)
// 等效于
int sum_unrolled(int *arr, int n) {
    int sum = 0;
    int i;
    for (i = 0; i <= n - 4; i += 4) {
        sum += arr[i] + arr[i+1] + arr[i+2] + arr[i+3];
    }
    for (; i < n; i++)
        sum += arr[i];
    return sum;
}
```

### 循环不变量外提

```c
// 源码
void process(float *data, int n, float scale) {
    for (int i = 0; i < n; i++) {
        data[i] *= scale * 3.14159f;  // scale * 3.14159 是循环不变量
    }
}

// 编译器优化后
void process_opt(float *data, int n, float scale) {
    float factor = scale * 3.14159f;  // 提到循环外面
    for (int i = 0; i < n; i++) {
        data[i] *= factor;
    }
}
```

### 强度削减

```c
// 源码：数组索引用乘法
void fill_indices(int *arr, int n) {
    for (int i = 0; i < n; i++)
        arr[i] = i * 4;  // 乘法
}

// 优化后：用加法代替
void fill_indices_opt(int *arr, int n) {
    int val = 0;
    for (int i = 0; i < n; i++) {
        arr[i] = val;
        val += 4;  // 加法代替乘法
    }
}
```

## 4. 向量化

```c
// 自动向量化
// gcc -O3 -ftree-vectorize -mavx2

// 编译器可以自动向量化的模式
void vectorizable(float *a, float *b, float *c, int n) {
    for (int i = 0; i < n; i++) {
        c[i] = a[i] + b[i];  // 简单的逐元素运算
    }
}

// 复杂模式可能需要手动向量化
#include <immintrin.h>

void vector_add_manual(float *a, float *b, float *c, int n) {
    int i;
    for (i = 0; i <= n - 8; i += 8) {
        __m256 va = _mm256_loadu_ps(&a[i]);
        __m256 vb = _mm256_loadu_ps(&b[i]);
        __m256 vc = _mm256_add_ps(va, vb);
        _mm256_storeu_ps(&c[i], vc);
    }
    for (; i < n; i++) {
        c[i] = a[i] + b[i];
    }
}
```

## 5. 链接时优化（LTO）

```bash
# 编译时
gcc -flto -c module1.c -o module1.o
gcc -flto -c module2.c -o module2.o

# 链接时
gcc -flto module1.o module2.o -o app

# LTO允许：
# - 跨文件内联
# - 跨文件常量传播
# - 全局死代码消除
# - 全局类型分析
```

## 6. Profile-Guided Optimization (PGO)

```bash
# 第一步：编译插桩版本
gcc -fprofile-generate -O2 -o app app.c

# 第二步：运行插桩程序（用典型输入）
./app < typical_input.txt

# 第三步：使用profile数据重新编译
gcc -fprofile-use -O2 -o app_opt app.c

# PGO帮助编译器：
# - 更好地决策内联
# - 更好的分支预测
# - 更好的代码布局（热路径放在一起）
```

## 7. 编译器内置函数

```c
// GCC内建函数（编译器直接生成优化代码）
#include <stdio.h>

void builtin_examples(void) {
    int x = 0b10110100;

    // 位操作
    printf("前导零: %d\n", __builtin_clz(x));      // 从高位数
    printf("尾随零: %d\n", __builtin_ctz(x));      // 从低位数
    printf("1的个数: %d\n", __builtin_popcount(x)); // popcount
    printf("奇偶性: %d\n", __builtin_parity(x));    // 1的个数奇偶

    // 预期值提示
    if (__builtin_expect(x > 0, 1)) {
        // x > 0 是常见情况
    }

    // 内存操作
    int a[100];
    __builtin_memset(a, 0, sizeof(a));  // 编译器可能生成优化代码
    __builtin_memcpy(a, a, sizeof(a));

    // 溢出检测
    int result;
    if (__builtin_add_overflow(100, 200, &result)) {
        printf("溢出!\n");
    }

    // 常量折叠
    int ctz_const = __builtin_ctz(16);  // 编译期计算为4
}
```

## 8. restrict关键字

```c
// restrict: 告诉编译器指针没有别名

// 没有restrict，编译器不能假设a和b不重叠
void add(float *a, float *b, float *c, int n) {
    for (int i = 0; i < n; i++)
        c[i] = a[i] + b[i];
}

// 有restrict，编译器可以更积极地优化
void add_opt(float *restrict a, float *restrict b,
             float *restrict c, int n) {
    for (int i = 0; i < n; i++)
        c[i] = a[i] + b[i];
    // 编译器知道a、b、c不重叠，可以安全向量化
}
```

## 9. 常量与纯函数

```c
// const: 函数不修改任何通过指针访问的内容
__attribute__((const))
int square(int x) {
    return x * x;  // 结果只依赖于输入
}

// pure: 函数可能读取全局状态但不修改
__attribute__((pure))
int get_cached_value(void) {
    return global_cache;  // 不修改全局状态
}

// 这些属性帮助编译器：
// - 消除重复调用
// - 将调用移出循环
// - 进行公共子表达式消除
```

## 10. 查看编译器优化

```bash
# 查看生成的汇编
gcc -S -O2 app.c -o app.s

# 查看带源码注释的汇编
gcc -S -O2 -fverbose-asm app.c -o app.s

# 查看优化报告
gcc -O3 -fopt-info-vec-all app.c 2>&1 | grep "optimized"
gcc -O3 -fopt-info-loop-all app.c 2>&1

# 查看函数是否被内联
gcc -O2 -fopt-info-inline app.c 2>&1
```

## 要点总结

1. -O2是大多数情况下的最佳选择
2. PGO可以带来10-20%的额外性能提升
3. LTO允许跨文件优化
4. restrict、const、pure等属性帮助编译器做出更好的优化决策
5. 向量化需要关注数据对齐和循环结构
6. 使用编译器优化报告验证优化是否生效
