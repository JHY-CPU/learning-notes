# 30 - C11新特性详解

## 概述

C11（ISO/IEC 9899:2011）引入了许多重要特性，使C语言更现代、更安全。本节详解C11的主要新特性。

---

## 1. _Generic（泛型选择）

```c
#include <stdio.h>
#include <math.h>

// 根据类型选择不同的函数实现
#define abs(x) _Generic((x), \
    int: abs, \
    long: labs, \
    long long: llabs, \
    float: fabsf, \
    double: fabs, \
    long double: fabsl \
)(x)

#define sqrt(x) _Generic((x), \
    float: sqrtf, \
    double: sqrt, \
    long double: sqrtl \
)(x)

// 获取类型名称
#define type_name(x) _Generic((x), \
    int: "int", \
    long: "long", \
    long long: "long long", \
    float: "float", \
    double: "double", \
    long double: "long double", \
    char*: "char*", \
    default: "unknown" \
)

// 类型安全的最大值宏
#define max(a, b) _Generic((a) + (b), \
    int: max_int, \
    double: max_double \
)(a, b)

int max_int(int a, int b) { return a > b ? a : b; }
double max_double(double a, double b) { return a > b ? a : b; }

void generic_demo(void) {
    int i = -5;
    double d = -3.14;

    printf("abs(%d) = %d\n", i, abs(i));
    printf("abs(%.2f) = %.2f\n", d, abs(d));
    printf("sqrt(4.0) = %.2f\n", sqrt(4.0));

    printf("i的类型: %s\n", type_name(i));
    printf("d的类型: %s\n", type_name(d));

    printf("max(3, 5) = %d\n", max(3, 5));
    printf("max(3.14, 2.71) = %.2f\n", max(3.14, 2.71));
}
```

## 2. _Static_assert（静态断言）

```c
#include <assert.h>

// 编译期断言
_Static_assert(sizeof(int) == 4, "int必须是4字节");
_Static_assert(sizeof(long long) >= 8, "long long至少8字节");

// 也可以用static_assert（C23之前需要#include <assert.h>）
static_assert(sizeof(void*) == 8 || sizeof(void*) == 4,
              "指针大小必须是4或8字节");

void static_assert_demo(void) {
    // 函数内的静态断言
    _Static_assert(sizeof(double) == 8, "double必须是8字节");

    printf("所有静态断言通过\n");
}
```

## 3. 匿名结构体和联合体

```c
#include <stdio.h>

// 匿名结构体
typedef struct {
    int x;
    int y;
    int z;
} Vector3;

// 匿名联合体
typedef struct {
    union {
        struct { float x, y, z; };
        float data[3];
    };
} Point;

// 嵌套匿名
typedef struct {
    int type;
    union {
        int int_val;
        double double_val;
        char str_val[64];
    };  // 无需命名
} Variant;

void anonymous_demo(void) {
    Point p = {{1.0f, 2.0f, 3.0f}};
    printf("p.x = %.1f, p.data[0] = %.1f\n", p.x, p.data[0]);  // 相同

    Variant v;
    v.type = 0;
    v.int_val = 42;  // 直接访问联合体成员
    printf("值: %d\n", v.int_val);
}
```

## 4. 对齐控制

```c
#include <stdio.h>
#include <stdalign.h>

// alignof获取对齐要求
void alignment_demo(void) {
    printf("char对齐: %zu\n", alignof(char));       // 1
    printf("int对齐: %zu\n", alignof(int));         // 4
    printf("double对齐: %zu\n", alignof(double));   // 8
    printf("max_align_t对齐: %zu\n", alignof(max_align_t));
}

// alignas指定对齐
_Alignas(16) int aligned_var;      // 16字节对齐
alignas(32) double aligned_double;  // 32字节对齐

typedef struct {
    alignas(16) int x;
    int y;
} AlignedStruct;

// SIMD对齐
typedef struct {
    alignas(32) float data[8];  // AVX对齐
} SimdArray;
```

## 5. _Noreturn（不返回的函数）

```c
#include <stdlib.h>
#include <stdio.h>

// 声明不返回的函数
_Noreturn void fatal_error(const char *msg) {
    fprintf(stderr, "致命错误: %s\n", msg);
    abort();  // 调用abort或exit，不会返回
}

// stdnoreturn.h提供noreturn宏
#include <stdnoreturn.h>
noreturn void my_exit(int code) {
    exit(code);
}

// 编译器可以基于此信息优化
void noreturn_demo(int x) {
    if (x < 0) {
        fatal_error("x不能为负数");
    }
    // 编译器知道fatal_error不返回，不会警告这里x可能未初始化
    printf("x = %d\n", x);
}
```

## 6. 增强的Unicode支持

```c
#include <stdio.h>
#include <uchar.h>
#include <locale.h>

void unicode_demo(void) {
    setlocale(LC_ALL, "");

    // UTF-8字符串字面量
    const char *utf8_str = u8"你好世界";
    printf("UTF-8: %s\n", utf8_str);

    // UTF-16字符
    char16_t c16 = u'好';

    // UTF-32字符
    char32_t c32 = U'好';

    // 宽字符
    wchar_t wc = L'好';

    printf("UTF-16字符值: %u\n", (unsigned)c16);
    printf("UTF-32字符值: %u\n", (unsigned)c32);
}
```

## 7. 边界检查函数（Annex K）

```c
#define __STDC_WANT_LIB_EXT1__ 1
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

void bounds_check_demo(void) {
    char buf[10];
    errno_t err;

    // 安全的字符串操作
    err = strcpy_s(buf, sizeof(buf), "Hello");
    if (err != 0) {
        printf("strcpy_s失败: %d\n", err);
    }

    err = strcat_s(buf, sizeof(buf), " World");
    if (err != 0) {
        printf("strcat_s失败\n");
    }

    // 安全的输入
    char input[64];
    if (gets_s(input, sizeof(input)) != NULL) {
        printf("输入: %s\n", input);
    }

    // 安全的格式化输出
    char formatted[32];
    sprintf_s(formatted, sizeof(formatted), "值=%d", 42);
    printf("格式化: %s\n", formatted);
}
```

## 8. 时间函数增强

```c
#include <stdio.h>
#include <time.h>
#include <threads.h>

void timespec_demo(void) {
    struct timespec ts;
    timespec_get(&ts, TIME_UTC);

    printf("秒: %ld\n", ts.tv_sec);
    printf("纳秒: %ld\n", ts.tv_nsec);

    // 获取当前时间
    time_t now = time(NULL);
    struct tm *tm = gmtime(&now);
    char buf[64];
    strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S UTC", tm);
    printf("当前时间: %s\n", buf);
}
```

## 9. 线程支持（可选）

```c
#include <stdio.h>
#include <threads.h>

int thread_func(void *arg) {
    int id = *(int *)arg;
    printf("线程 %d 运行\n", id);
    thrd_sleep(&(struct timespec){.tv_sec = 1}, NULL);
    printf("线程 %d 完成\n", id);
    return 0;
}

void threads_demo(void) {
    thrd_t t1, t2;
    int id1 = 1, id2 = 2;

    thrd_create(&t1, thread_func, &id1);
    thrd_create(&t2, thread_func, &id2);

    thrd_join(t1, NULL);
    thrd_join(t2, NULL);

    // 互斥锁
    mtx_t mutex;
    mtx_init(&mutex, mtx_plain);
    mtx_lock(&mutex);
    // 临界区
    mtx_unlock(&mutex);
    mtx_destroy(&mutex);

    // 条件变量
    cnd_t cond;
    cnd_init(&cond);
    cnd_signal(&cond);
    cnd_destroy(&cond);
}
```

## 10. C11特性检查

```c
#include <stdio.h>

// 检查编译器是否支持C11
#if __STDC_VERSION__ >= 201112L
    #define C11_SUPPORTED 1
#else
    #define C11_SUPPORTED 0
#endif

// 检查具体特性
void check_features(void) {
    printf("C标准版本: %ld\n", __STDC_VERSION__);

    #ifdef __STDC_NO_ATOMICS__
    printf("不支持原子操作\n");
    #else
    printf("支持原子操作\n");
    #endif

    #ifdef __STDC_NO_THREADS__
    printf("不支持线程\n");
    #else
    printf("支持线程\n");
    #endif

    #ifdef __STDC_NO_VLA__
    printf("不支持VLA\n");
    #else
    printf("支持VLA\n");
    #endif
}
```

## 要点总结

1. _Generic实现编译期类型选择，是C语言泛型的基础
2. _Static_assert在编译期验证条件
3. 匿名结构体/联合体简化嵌套数据结构
4. alignas/alignof控制内存对齐
5. 边界检查函数提高安全性（但非强制实现）
6. C11线程支持是可选特性，许多编译器仍使用pthread
