单元测试是保证代码质量的关键实践。本节介绍C语言中常用的测试框架和测试驱动开发（TDD）方法。

## 1. 简易测试框架实现

```c
// minunit.h - 极简测试框架
#ifndef MINUNIT_H
#define MINUNIT_H

#include <stdio.h>
#include <string.h>

static int tests_run = 0;
static int tests_passed = 0;
static int tests_failed = 0;

#define mu_assert(test, msg) \
    do { \
        if (!(test)) { \
            printf("  [FAIL] %s (第%d行)\n", msg, __LINE__); \
            return msg; \
        } \
    } while(0)

#define mu_assert_int_eq(expected, actual) \
    do { \
        if ((expected) != (actual)) { \
            static char _buf[256]; \
            snprintf(_buf, sizeof(_buf), \
                     "期望 %d, 实际 %d (第%d行)", \
                     (int)(expected), (int)(actual), __LINE__); \
            return _buf; \
        } \
    } while(0)

#define mu_assert_str_eq(expected, actual) \
    do { \
        if (strcmp((expected), (actual)) != 0) { \
            static char _buf[256]; \
            snprintf(_buf, sizeof(_buf), \
                     "期望 \"%s\", 实际 \"%s\" (第%d行)", \
                     (expected), (actual), __LINE__); \
            return _buf; \
        } \
    } while(0)

#define mu_assert_double_near(expected, actual, epsilon) \
    do { \
        double _diff = (expected) - (actual); \
        if (_diff < 0) _diff = -_diff; \
        if (_diff > (epsilon)) { \
            static char _buf[256]; \
            snprintf(_buf, sizeof(_buf), \
                     "期望 %.6f, 实际 %.6f, 差值 %.6f (第%d行)", \
                     (double)(expected), (double)(actual), _diff, __LINE__); \
            return _buf; \
        } \
    } while(0)

#define mu_run_test(test) \
    do { \
        tests_run++; \
        printf("  运行: %s ... ", #test); \
        const char *msg = test(); \
        if (msg) { \
            tests_failed++; \
            printf("\n"); \
        } else { \
            tests_passed++; \
            printf("[OK]\n"); \
        } \
    } while(0)

#define mu_report() \
    do { \
        printf("\n========== 测试报告 ==========\n"); \
        printf("总计: %d, 通过: %d, 失败: %d\n", \
               tests_run, tests_passed, tests_failed); \
        printf("==============================\n"); \
    } while(0)

#endif
```

## 2. 使用示例：测试字符串工具

```c
// str_utils.h
#ifndef STR_UTILS_H
#define STR_UTILS_H

#include <stddef.h>

char *str_reverse(char *str);
int str_count_char(const char *str, char c);
char *str_trim(char *str);

#endif
```

```c
// str_utils.c
#include "str_utils.h"
#include <string.h>
#include <ctype.h>

char *str_reverse(char *str) {
    if (!str) return NULL;
    int len = strlen(str);
    for (int i = 0; i < len / 2; i++) {
        char tmp = str[i];
        str[i] = str[len - 1 - i];
        str[len - 1 - i] = tmp;
    }
    return str;
}

int str_count_char(const char *str, char c) {
    int count = 0;
    while (*str) {
        if (*str == c) count++;
        str++;
    }
    return count;
}

char *str_trim(char *str) {
    if (!str) return NULL;
    char *end;
    while (isspace((unsigned char)*str)) str++;
    if (*str == 0) return str;
    end = str + strlen(str) - 1;
    while (end > str && isspace((unsigned char)*end)) end--;
    *(end + 1) = 0;
    return str;
}
```

```c
// test_str_utils.c
#include <stdio.h>
#include <string.h>
#include "minunit.h"
#include "str_utils.h"

// 测试用例
static const char *test_reverse_basic(void) {
    char str[] = "hello";
    mu_assert_str_eq("olleh", str_reverse(str));
    return NULL;
}

static const char *test_reverse_empty(void) {
    char str[] = "";
    mu_assert_str_eq("", str_reverse(str));
    return NULL;
}

static const char *test_reverse_single(void) {
    char str[] = "a";
    mu_assert_str_eq("a", str_reverse(str));
    return NULL;
}

static const char *test_count_char(void) {
    mu_assert_int_eq(3, str_count_char("hello world", 'l'));
    mu_assert_int_eq(0, str_count_char("hello", 'z'));
    mu_assert_int_eq(2, str_count_char("aabbaa", 'a'));
    return NULL;
}

static const char *test_trim(void) {
    char str1[] = "  hello  ";
    mu_assert_str_eq("hello", str_trim(str1));

    char str2[] = "hello";
    mu_assert_str_eq("hello", str_trim(str2));

    char str3[] = "   ";
    mu_assert_str_eq("", str_trim(str3));

    return NULL;
}

// 运行所有测试
static const char *all_tests(void) {
    mu_run_test(test_reverse_basic);
    mu_run_test(test_reverse_empty);
    mu_run_test(test_reverse_single);
    mu_run_test(test_count_char);
    mu_run_test(test_trim);
    return NULL;
}

int main(void) {
    printf("\n========== str_utils 测试 ==========\n\n");
    const char *result = all_tests();
    mu_report();
    return result != NULL ? 1 : 0;
}
```

## 4. 更专业的框架：Check

```bash
# 安装Check
# Ubuntu: sudo apt install check
# macOS: brew install check
```

```c
// test_with_check.c
#include <check.h>
#include <stdlib.h>
#include "str_utils.h"

START_TEST(test_reverse) {
    char str[] = "hello";
    ck_assert_str_eq("olleh", str_reverse(str));
}
END_TEST

START_TEST(test_count) {
    ck_assert_int_eq(3, str_count_char("hello", 'l'));
}
END_TEST

Suite *str_utils_suite(void) {
    Suite *s = suite_create("str_utils");
    TCase *tc = tcase_create("core");

    tcase_add_test(tc, test_reverse);
    tcase_add_test(tc, test_count);

    suite_add_tcase(s, tc);
    return s;
}

int main(void) {
    Suite *s = str_utils_suite();
    SRunner *sr = srunner_create(s);

    srunner_run_all(sr, CK_NORMAL);
    int failed = srunner_ntests_failed(sr);
    srunner_free(sr);

    return (failed == 0) ? EXIT_SUCCESS : EXIT_FAILURE;
}
```

## 5. 测试驱动开发（TDD）

### TDD流程

```
1. 写一个失败的测试（红）
2. 写最少的代码让测试通过（绿）
3. 重构代码（重构）
4. 重复
```

### TDD示例

```c
// 第1步：写测试（此时函数还不存在）
static const char *test_stack_push_pop(void) {
    Stack *s = stack_create(10);
    stack_push(s, 42);
    stack_push(s, 100);

    mu_assert_int_eq(100, stack_pop(s));
    mu_assert_int_eq(42, stack_pop(s));

    stack_destroy(s);
    return NULL;
}

// 第2步：实现最小功能让测试通过
// stack.h / stack.c ...

// 第3步：运行测试，确认通过
// 第4步：添加更多测试，完善实现
```

### 测试覆盖场景

```c
static const char *test_divide(void) {
    // 正常情况
    mu_assert_int_eq(5, divide(10, 2));

    // 边界情况
    mu_assert_int_eq(0, divide(0, 5));

    // 负数
    mu_assert_int_eq(-5, divide(-10, 2));

    // 异常情况
    mu_assert("除零应返回错误码", divide(10, 0) == -1);

    return NULL;
}
```

## 要点总结

1. 每个函数都应该有对应的测试用例
2. 测试应覆盖正常、边界和异常情况
3. TDD方法可以提高代码质量和设计
4. 使用代码覆盖率工具确保测试充分
5. 测试代码也是代码，需要维护
6. 自动化测试集成到CI/CD流程中
