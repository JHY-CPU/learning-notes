# 32 - C23新特性前瞻

## 概述

C23（ISO/IEC 9899:2024）是C语言的最新标准，引入了许多现代化特性。本节介绍C23的主要新特性。

---

## 1. constexpr（编译期常量）

```c
// C23允许在更多地方使用编译期求值的表达式

// 编译期常量
constexpr int max_size = 1024;
constexpr double pi = 3.14159265358979;

// 编译期函数
constexpr int factorial(int n) {
    int result = 1;
    for (int i = 2; i <= n; i++) result *= i;
    return result;
}

// 用于数组大小
int buffer[factorial(5)];  // 120个元素

// 用于case标签
void constexpr_switch(int x) {
    constexpr int threshold = 100;
    switch (x) {
    case threshold: break;      // 编译期确定
    case factorial(3): break;   // case 6
    }
}
```

## 2. typeof / typeof_unqual

```c
// 类型推断（类似C++的auto）
void typeof_demo(void) {
    typeof(42) x = 42;          // x是int
    typeof(3.14) y = 3.14;      // y是double

    // 数组推断
    int arr[] = {1, 2, 3, 4, 5};
    typeof(arr) arr2 = {10, 20, 30, 40, 50};  // arr2也是int[5]

    // 用于宏中的类型安全
    #define MAX(a, b) ({ \
        typeof(a) _a = (a); \
        typeof(b) _b = (b); \
        _a > _b ? _a : _b; \
    })

    int m1 = MAX(3, 5);
    double m2 = MAX(3.14, 2.71);
}

// typeof_unqual: 移除const/volatile限定符
void typeof_unqual_demo(void) {
    const int x = 42;
    typeof_unqual(x) y = x;  // y是int（非const）
    y = 100;  // 可以修改
}
```

## 3. auto关键字（类型推断）

```c
// C23的auto不同于C++，用于类型推断
void auto_demo(void) {
    auto x = 42;           // int
    auto y = 3.14;         // double
    auto str = "hello";    // const char*

    auto ptr = &x;         // int*

    // 函数返回类型推断
    auto add = [](int a, int b) { return a + b; };  // C23 lambda!
}
```

## 4. 改进的枚举

```c
// 底层类型指定
enum Color : unsigned char {
    RED,
    GREEN,
    BLUE
};

// 枚举的底层类型确定
_Static_assert(sizeof(enum Color) == 1, "Color应为1字节");

// 枚举可用于switch的全匹配检查
void enum_switch(enum Color c) {
    switch (c) {
    case RED:   printf("红色\n"); break;
    case GREEN: printf("绿色\n"); break;
    case BLUE:  printf("蓝色\n"); break;
    // 编译器可警告遗漏的枚举值
    }
}
```

## 6. nullptr

```c
// C23引入nullptr作为空指针常量
void nullptr_demo(void) {
    int *ptr = nullptr;  // 替代NULL

    if (ptr == nullptr) {
        printf("ptr为空\n");
    }

    // nullptr的类型是nullptr_t，不同于(void*)0
    // 解决了NULL在某些上下文中的歧义
}
```

## 7. 改进的宏功能

```c
// #embed: 嵌入二进制文件
void embed_demo(void) {
    // 直接将文件内容嵌入为数组
    static const unsigned char icon[] = {
        #embed "icon.png"
    };
    // 不再需要xxd或手动编码
}

// #elifdef / #elifndef
#define DEBUG 1
#define RELEASE 2

#if defined(DEBUG)
    #define MODE DEBUG
#elif defined(RELEASE)
    #define MODE RELEASE
#else
    #define MODE 0
#endif

// 预处理器增加了#elifdef和#elifndef
#ifdef DEBUG
    // ...
#elifdef RELEASE
    // ...
#endif
```

## 8. 属性（Attributes）

```c
// C23标准化了属性语法
[[nodiscard]] int important_function(void);  // 返回值不应被忽略
[[deprecated("请使用new_func代替")]] void old_func(void);
[[maybe_unused]] int debug_var;

// 函数属性
[[nodiscard]] int compute_value(int x) {
    return x * 2;
}

void attribute_demo(void) {
    // 如果忽略返回值，编译器会警告
    compute_value(42);  // 警告：忽略nodiscard返回值

    old_func();  // 警告：使用了废弃函数
}
```

## 9. 改进的泛型

```c
// C23增强的_Generic
#define print(x) _Generic((x), \
    int: printf("%d\n", x), \
    double: printf("%f\n", x), \
    char*: printf("%s\n", x), \
    default: printf("未知类型\n") \
)

// C23允许default分支调用函数
void print_unknown(void *ptr) {
    printf("未知类型: %p\n", ptr);
}
```

## 11. 函数声明改进

```c
// C23允许省略函数参数列表中的void
// 以前: int foo(void);
// C23:  int foo();  // 表示无参数（与C++一致）

int get_value();  // 等同于 int get_value(void);

// 也支持简单的lambda表达式
void lambda_demo(void) {
    auto square = [](int x) { return x * x; };
    printf("平方: %d\n", square(5));  // 25
}
```

## 要点总结

1. C23是C语言现代化的重要一步
2. constexpr/typeof/auto使C语言更易用
3. nullptr解决了NULL的类型歧义
4. 标准化的属性语法提高代码可读性
5. 嵌入二进制文件（#embed）简化资源管理
6. C23保持了与C17的向后兼容性
7. 编译器支持正在逐步完善，建议关注GCC 13+和Clang 16+
