# 15 - 安全编程实践

## 概述

C语言的灵活性也带来了安全隐患。本节介绍常见的安全漏洞及其防护方法，帮助编写更安全的C代码。

## 1. 缓冲区溢出防护

### 危险函数与替代方案

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// ========== 危险的函数 ==========

void dangerous_code(void) {
    char buf[10];

    // strcpy: 无长度检查，极易溢出
    // strcpy(buf, "这是一段很长的字符串");  // 危险！

    // sprintf: 无长度检查
    // sprintf(buf, "%s", "很长的字符串");   // 危险！

    // gets: 完全无限制
    // gets(buf);                            // 已废弃！永远不要用！

    // strcat: 无长度检查
    // strcat(buf, "追加内容");              // 危险！
}

// ========== 安全替代方案 ==========

void safe_code(void) {
    char buf[10];

    // strncpy: 有长度限制，但需手动确保结尾
    strncpy(buf, "安全的字符串", sizeof(buf) - 1);
    buf[sizeof(buf) - 1] = '\0';  // 必须手动添加结尾

    // snprintf: 推荐的格式化方式
    snprintf(buf, sizeof(buf), "值=%d", 42);

    // fgets: 安全的输入方式
    if (fgets(buf, sizeof(buf), stdin)) {
        // 移除换行符
        buf[strcspn(buf, "\n")] = '\0';
    }

    // strncat: 安全的连接
    strncat(buf, "追加", sizeof(buf) - strlen(buf) - 1);
}

// 使用C11的边界检查函数（Annex K）
// 需要 #define __STDC_WANT_LIB_EXT1__ 1
#if defined(__STDC_WANT_LIB_EXT1__) && __STDC_WANT_LIB_EXT1__ == 1
void safe_c11(void) {
    char buf[10];
    size_t len;

    // strcpy_s: 安全版strcpy
    errno_t err = strcpy_s(buf, sizeof(buf), "安全字符串");

    // sprintf_s: 安全版sprintf
    sprintf_s(buf, sizeof(buf), "值=%d", 42);

    // gets_s: 安全版gets
    gets_s(buf, sizeof(buf));
}
#endif
```

## 2. 整数溢出防护

```c
#include <stdint.h>
#include <limits.h>
#include <stdio.h>

// 检查加法是否溢出
int safe_add(int a, int b, int *result) {
    if ((b > 0 && a > INT_MAX - b) ||
        (b < 0 && a < INT_MIN - b)) {
        return -1;  // 溢出
    }
    *result = a + b;
    return 0;
}

// 检查乘法是否溢出
int safe_mul(int a, int b, int *result) {
    if (a == 0 || b == 0) {
        *result = 0;
        return 0;
    }
    if ((a > 0 && b > 0 && a > INT_MAX / b) ||
        (a > 0 && b < 0 && b < INT_MIN / a) ||
        (a < 0 && b > 0 && a < INT_MIN / b) ||
        (a < 0 && b < 0 && a < INT_MAX / b)) {
        return -1;  // 溢出
    }
    *result = a * b;
    return 0;
}

// 无符号整数
int safe_add_u(size_t a, size_t b, size_t *result) {
    if (a > SIZE_MAX - b) return -1;
    *result = a + b;
    return 0;
}

// 使用GCC内建函数
int safe_add_builtin(int a, int b, int *result) {
    return __builtin_add_overflow(a, b, result);
    // GCC 5+ 提供 __builtin_add_overflow
    // __builtin_mul_overflow
    // __builtin_sub_overflow
}
```

## 3. 格式化字符串攻击防护

```c
#include <stdio.h>
#include <stdarg.h>

void vulnerable_printf(const char *user_input) {
    // 危险：用户输入直接作为格式字符串
    // printf(user_input);  // 用户输入 "%s%s%s%s" 可导致崩溃

    // 安全：使用固定格式字符串
    printf("%s", user_input);

    // 或者使用fputs
    fputs(user_input, stdout);
}

// 安全的日志函数
void safe_log(const char *fmt, ...) {
    va_list args;
    va_start(args, fmt);
    vprintf(fmt, args);  // 使用vprintf而不是直接传递给printf
    va_end(args);
}

// 使用示例
void log_example(void) {
    const char *user_msg = "用户消息";
    // 正确
    safe_log("日志: %s\n", user_msg);
    printf("日志: %s\n", user_msg);

    // 错误
    // printf(user_msg);  // 漏洞！
}
```

## 4. 空指针与野指针防护

```c
#include <stdlib.h>
#include <stdio.h>

// 带检查的内存操作
void *safe_malloc(size_t size) {
    if (size == 0) return NULL;
    void *ptr = malloc(size);
    if (!ptr) {
        fprintf(stderr, "内存分配失败\n");
        abort();
    }
    return ptr;
}

// 使用后置空
void safe_free(void **ptr) {
    if (ptr && *ptr) {
        free(*ptr);
        *ptr = NULL;
    }
}

// 封装宏
#define SAFE_FREE(p) do { safe_free((void**)&(p)); } while(0)

void pointer_demo(void) {
    int *p = safe_malloc(sizeof(int) * 100);

    // 使用p...
    *p = 42;

    // 释放并置空
    SAFE_FREE(p);

    // 此时p为NULL，再次free是安全的
    SAFE_FREE(p);  // 不会double free
}
```

## 5. 字符串操作安全

```c
#include <string.h>
#include <stdio.h>

// 安全字符串复制
char *safe_strcpy(char *dest, size_t dest_size, const char *src) {
    if (!dest || !src || dest_size == 0) return NULL;
    strncpy(dest, src, dest_size - 1);
    dest[dest_size - 1] = '\0';
    return dest;
}

// 安全字符串连接
char *safe_strcat(char *dest, size_t dest_size, const char *src) {
    if (!dest || !src || dest_size == 0) return NULL;
    size_t len = strlen(dest);
    if (len >= dest_size) return NULL;
    strncat(dest, src, dest_size - len - 1);
    return dest;
}

// 安全字符串格式化
int safe_snprintf(char *buf, size_t size, const char *fmt, ...) {
    if (!buf || size == 0) return -1;
    va_list args;
    va_start(args, fmt);
    int n = vsnprintf(buf, size, fmt, args);
    va_end(args);
    return (n < 0 || (size_t)n >= size) ? -1 : n;
}
```

## 6. 输入验证

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

// 验证整数输入
int read_int_safe(int *out) {
    char buf[64];
    if (!fgets(buf, sizeof(buf), stdin)) return -1;

    // 检查是否全是数字
    char *end;
    long val = strtol(buf, &end, 10);

    if (end == buf || (*end != '\n' && *end != '\0')) {
        return -1;  // 不是有效整数
    }
    if (val < INT_MIN || val > INT_MAX) {
        return -1;  // 超出范围
    }

    *out = (int)val;
    return 0;
}

// 验证文件路径（防止路径遍历）
int is_safe_path(const char *path) {
    if (!path || path[0] == '\0') return 0;

    // 禁止绝对路径
    if (path[0] == '/') return 0;

    // 禁止路径遍历
    if (strstr(path, "..")) return 0;

    // 只允许字母数字、下划线、点、横线
    for (const char *p = path; *p; p++) {
        if (!isalnum(*p) && *p != '_' && *p != '.' && *p != '-')
            return 0;
    }

    return 1;
}
```

## 7. 编译器安全选项

```bash
# GCC安全编译选项
gcc -fstack-protector-strong  # 栈保护
gcc -D_FORTIFY_SOURCE=2       # 缓冲区溢出检测
gcc -fPIE -pie                # 地址空间随机化（ASLR）
gcc -Wformat -Wformat-security # 格式化字符串警告
gcc -Werror                   # 警告当错误处理

# 组合使用
gcc -Wall -Wextra -Werror -fstack-protector-strong \
    -D_FORTIFY_SOURCE=2 -fPIE -pie -o app app.c
```

## 要点总结

1. 永远不要使用`gets`、`strcpy`、`sprintf`等无边界检查的函数
2. 始终验证整数运算是否溢出
3. 不要将用户输入直接作为格式字符串
4. 使用后置空指针防止野指针和double free
5. 始终验证外部输入（文件、网络、用户输入）
6. 启用编译器的安全选项
