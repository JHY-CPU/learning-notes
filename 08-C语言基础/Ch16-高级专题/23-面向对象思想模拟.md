# 23 - 面向对象思想模拟

## 概述

C语言没有原生的面向对象支持，但可以通过结构体、函数指针和虚函数表模拟封装、继承和多态。

---

## 1. 封装（Encapsulation）

### 不透明指针隐藏实现

```c
// person.h - 公共接口
#ifndef PERSON_H
#define PERSON_H

typedef struct Person Person;  // 不透明类型

Person *person_create(const char *name, int age);
void person_destroy(Person *p);
const char *person_get_name(const Person *p);
int person_get_age(const Person *p);
void person_set_age(Person *p, int age);
void person_print(const Person *p);

#endif
```

```c
// person.c - 私有实现
#include "person.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 私有结构体定义（外部不可见）
struct Person {
    char *name;
    int age;
};

Person *person_create(const char *name, int age) {
    Person *p = (Person *)malloc(sizeof(Person));
    p->name = strdup(name);
    p->age = age;
    return p;
}

void person_destroy(Person *p) {
    if (p) {
        free(p->name);
        free(p);
    }
}

const char *person_get_name(const Person *p) {
    return p ? p->name : NULL;
}

int person_get_age(const Person *p) {
    return p ? p->age : -1;
}

void person_set_age(Person *p, int age) {
    if (p) p->age = age;
}

void person_print(const Person *p) {
    if (p) printf("Person{name=\"%s\", age=%d}\n", p->name, p->age);
}
```

## 2. 继承（Inheritance）

### 结构体嵌套实现继承

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 基类
typedef struct {
    char name[64];
    int age;
} Animal;

void animal_init(Animal *a, const char *name, int age) {
    strncpy(a->name, name, sizeof(a->name) - 1);
    a->age = age;
}

void animal_speak(Animal *a) {
    printf("%s 发出声音\n", a->name);
}

// 派生类：狗
typedef struct {
    Animal base;      // 继承（必须放在首位）
    char breed[32];   // 派生属性
} Dog;

void dog_init(Dog *d, const char *name, int age, const char *breed) {
    animal_init(&d->base, name, age);
    strncpy(d->breed, breed, sizeof(d->breed) - 1);
}

void dog_speak(Dog *d) {
    printf("%s 汪汪叫!\n", d->base.name);
}

void dog_fetch(Dog *d) {
    printf("%s 在捡球!\n", d->base.name);
}

// 派生类：猫
typedef struct {
    Animal base;
    int is_indoor;
} Cat;

void cat_init(Cat *c, const char *name, int age, int indoor) {
    animal_init(&c->base, name, age);
    c->is_indoor = indoor;
}

void cat_speak(Cat *c) {
    printf("%s 喵喵叫!\n", c->base.name);
}

// 可以通过指针转换使用基类接口
void use_animal(Animal *a) {
    printf("动物: %s, 年龄: %d\n", a->name, a->age);
}

void inheritance_demo(void) {
    Dog dog;
    dog_init(&dog, "旺财", 3, "金毛");
    dog_speak(&dog);
    dog_fetch(&dog);

    Cat cat;
    cat_init(&cat, "咪咪", 2, 1);
    cat_speak(&cat);

    // 通过基类指针调用
    use_animal((Animal *)&dog);
    use_animal((Animal *)&cat);
}
```

## 3. 多态（Polymorphism）

### 虚函数表实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

// ========== 定义虚函数表 ==========

// 前向声明
typedef struct Shape Shape;

// 虚函数表
typedef struct {
    const char *(*name)(void);
    double (*area)(const Shape *self);
    double (*perimeter)(const Shape *self);
    void (*describe)(const Shape *self);
    void (*destroy)(Shape *self);
} ShapeVTable;

// 基类
struct Shape {
    const ShapeVTable *vtable;  // 虚函数表指针
};

// 多态调用接口
static inline const char *shape_name(const Shape *s) {
    return s->vtable->name();
}

static inline double shape_area(const Shape *s) {
    return s->vtable->area(s);
}

static inline double shape_perimeter(const Shape *s) {
    return s->vtable->perimeter(s);
}

static inline void shape_describe(const Shape *s) {
    s->vtable->describe(s);
}

static inline void shape_destroy(Shape *s) {
    s->vtable->destroy(s);
}

// ========== 圆形 ==========

typedef struct {
    Shape base;
    double radius;
} Circle;

static const char *circle_name(void) { return "圆形"; }

static double circle_area(const Shape *s) {
    const Circle *c = (const Circle *)s;
    return M_PI * c->radius * c->radius;
}

static double circle_perimeter(const Shape *s) {
    const Circle *c = (const Circle *)s;
    return 2 * M_PI * c->radius;
}

static void circle_describe(const Shape *s) {
    const Circle *c = (const Circle *)s;
    printf("%s: 半径=%.2f, 面积=%.2f, 周长=%.2f\n",
           circle_name(), c->radius,
           circle_area(s), circle_perimeter(s));
}

static const ShapeVTable circle_vtable = {
    .name = circle_name,
    .area = circle_area,
    .perimeter = circle_perimeter,
    .describe = circle_describe,
    .destroy = free
};

Circle *circle_create(double radius) {
    Circle *c = (Circle *)malloc(sizeof(Circle));
    c->base.vtable = &circle_vtable;
    c->radius = radius;
    return c;
}

// ========== 矩形 ==========

typedef struct {
    Shape base;
    double width, height;
} Rectangle;

static const char *rect_name(void) { return "矩形"; }

static double rect_area(const Shape *s) {
    const Rectangle *r = (const Rectangle *)s;
    return r->width * r->height;
}

static double rect_perimeter(const Shape *s) {
    const Rectangle *r = (const Rectangle *)s;
    return 2 * (r->width + r->height);
}

static void rect_describe(const Shape *s) {
    const Rectangle *r = (const Rectangle *)s;
    printf("%s: %.2f x %.2f, 面积=%.2f, 周长=%.2f\n",
           rect_name(), r->width, r->height,
           rect_area(s), rect_perimeter(s));
}

static const ShapeVTable rect_vtable = {
    .name = rect_name,
    .area = rect_area,
    .perimeter = rect_perimeter,
    .describe = rect_describe,
    .destroy = free
};

Rectangle *rectangle_create(double w, double h) {
    Rectangle *r = (Rectangle *)malloc(sizeof(Rectangle));
    r->base.vtable = &rect_vtable;
    r->width = w;
    r->height = h;
    return r;
}

// ========== 多态使用 ==========

void polymorphism_demo(void) {
    Shape *shapes[] = {
        (Shape *)circle_create(5.0),
        (Shape *)rectangle_create(4.0, 6.0),
        (Shape *)circle_create(3.0),
        NULL
    };

    printf("多态演示：\n\n");
    double total_area = 0;

    for (int i = 0; shapes[i]; i++) {
        shape_describe(shapes[i]);
        total_area += shape_area(shapes[i]);
        printf("\n");
    }

    printf("总面积: %.2f\n", total_area);

    // 清理
    for (int i = 0; shapes[i]; i++) {
        shape_destroy(shapes[i]);
    }
}
```

## 4. 接口与抽象类

```c
// 接口定义（纯虚函数表）
typedef struct Drawable Drawable;

typedef struct {
    void (*draw)(const Drawable *self);
    void (*destroy)(Drawable *self);
} DrawableInterface;

struct Drawable {
    const DrawableInterface *interface;
};

// 抽象类：包含部分实现
typedef struct {
    Drawable base;
    int x, y;  // 公共属性
} PositionedDrawable;

void positioned_drawable_init(PositionedDrawable *pd, int x, int y,
                               const DrawableInterface *iface) {
    pd->base.interface = iface;
    pd->x = x;
    pd->y = y;
}

void positioned_drawable_draw(const Drawable *self) {
    const PositionedDrawable *pd = (const PositionedDrawable *)self;
    printf("位置: (%d, %d) -> ", pd->x, pd->y);
    // 调用子类的具体绘制
    if (pd->base.interface->draw != positioned_drawable_draw) {
        pd->base.interface->draw(self);
    }
}
```

## 5. RTTI（运行时类型识别）

```c
#include <stdio.h>
#include <string.h>

typedef enum {
    TYPE_CIRCLE,
    TYPE_RECTANGLE,
    TYPE_TRIANGLE
} ShapeType;

typedef struct {
    ShapeType type;
    const char *name;
} TypeInfo;

typedef struct {
    TypeInfo *type_info;
} Object;

// 类型检查
int is_type(Object *obj, ShapeType type) {
    return obj && obj->type_info && obj->type_info->type == type;
}

const char *get_type_name(Object *obj) {
    return obj && obj->type_info ? obj->type_info->name : "未知";
}

// 向下转型
typedef struct {
    Object base;
    double radius;
} CircleObj;

CircleObj *as_circle(Object *obj) {
    return is_type(obj, TYPE_CIRCLE) ? (CircleObj *)obj : NULL;
}

void rtti_demo(void) {
    TypeInfo circle_type = {TYPE_CIRCLE, "圆形"};
    CircleObj c = {{&circle_type}, 5.0};

    Object *obj = (Object *)&c;
    printf("类型: %s\n", get_type_name(obj));
    printf("是圆形: %s\n", is_type(obj, TYPE_CIRCLE) ? "是" : "否");

    CircleObj *circle = as_circle(obj);
    if (circle) {
        printf("半径: %.1f\n", circle->radius);
    }
}
```

## 要点总结

1. 不透明指针实现封装，隐藏内部实现
2. 结构体嵌套实现继承（基类必须在首位）
3. 虚函数表实现多态
4. 函数指针是实现一切的关键
5. RTTI通过类型标识枚举实现
6. 这种方式的OOP不如C++原生支持高效，但足够灵活
