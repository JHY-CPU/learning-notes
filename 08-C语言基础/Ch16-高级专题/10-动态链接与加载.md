# 10 - 动态链接与加载

## 概述

动态链接允许程序在运行时加载和使用共享库（.so/.dll），实现插件系统、减少内存占用、方便更新维护。

---

## 1. 基本概念

### 静态链接 vs 动态链接

| 特性 | 静态链接 | 动态链接 |
|------|----------|----------|
| 文件大小 | 较大 | 较小 |
| 内存占用 | 每进程独立拷贝 | 多进程共享 |
| 更新 | 需重新编译 | 替换.so即可 |
| 加载时间 | 较快 | 稍慢 |
| 依赖 | 无外部依赖 | 需要.so文件存在 |

## 2. dlopen/dlsym/dlclose API

```c
#include <stdio.h>
#include <stdlib.h>
#include <dlfcn.h>  // Linux/macOS
// #include <windows.h>  // Windows

void dynamic_loading_demo(void) {
    // 1. 打开共享库
    // RTLD_LAZY: 延迟绑定
    // RTLD_NOW: 立即解析所有符号
    void *handle = dlopen("./libmyplugin.so", RTLD_LAZY);
    if (!handle) {
        fprintf(stderr, "dlopen: %s\n", dlerror());
        return;
    }

    // 2. 查找符号
    dlerror();  // 清除错误
    int (*add)(int, int) = dlsym(handle, "add");
    const char *err = dlerror();
    if (err) {
        fprintf(stderr, "dlsym: %s\n", err);
        dlclose(handle);
        return;
    }

    // 3. 使用函数
    printf("add(3, 4) = %d\n", add(3, 4));

    // 4. 查找变量
    const char **version = dlsym(handle, "PLUGIN_VERSION");
    if (version) {
        printf("插件版本: %s\n", *version);
    }

    // 5. 关闭库
    dlclose(handle);
}
```

## 3. 插件系统实现

### 插件接口定义

```c
// plugin_interface.h
#ifndef PLUGIN_INTERFACE_H
#define PLUGIN_INTERFACE_H

// 插件必须实现的接口
typedef struct {
    const char *name;
    const char *version;
    int (*init)(void);
    int (*execute)(const char *input, char *output, int max_len);
    void (*cleanup)(void);
} Plugin;

// 插件导出宏
#define PLUGIN_EXPORT __attribute__((visibility("default")))

#endif
```

### 插件实现

```c
// plugin_hello.c - 一个具体的插件
#include "plugin_interface.h"
#include <stdio.h>
#include <string.h>

static int plugin_init(void) {
    printf("[hello插件] 初始化\n");
    return 0;
}

static int plugin_execute(const char *input, char *output, int max_len) {
    snprintf(output, max_len, "你好, %s!", input);
    return 0;
}

static void plugin_cleanup(void) {
    printf("[hello插件] 清理\n");
}

// 导出插件结构体
PLUGIN_EXPORT Plugin plugin = {
    .name = "Hello Plugin",
    .version = "1.0.0",
    .init = plugin_init,
    .execute = plugin_execute,
    .cleanup = plugin_cleanup
};
```

### 插件管理器

```c
// plugin_manager.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <dlfcn.h>
#include "plugin_interface.h"

#define MAX_PLUGINS 32

typedef struct {
    void *handle;
    Plugin *plugin;
} LoadedPlugin;

typedef struct {
    LoadedPlugin plugins[MAX_PLUGINS];
    int count;
} PluginManager;

// 初始化管理器
void pm_init(PluginManager *pm) {
    pm->count = 0;
}

// 加载插件
int pm_load(PluginManager *pm, const char *path) {
    if (pm->count >= MAX_PLUGINS) {
        fprintf(stderr: "插件数量已达上限\n");
        return -1;
    }

    void *handle = dlopen(path, RTLD_LAZY);
    if (!handle) {
        fprintf(stderr, "无法加载 %s: %s\n", path, dlerror());
        return -1;
    }

    // 查找插件结构体
    Plugin *plugin = dlsym(handle, "plugin");
    if (!plugin) {
        fprintf(stderr, "找不到插件符号: %s\n", dlerror());
        dlclose(handle);
        return -1;
    }

    // 初始化插件
    if (plugin->init && plugin->init() != 0) {
        fprintf(stderr, "插件初始化失败: %s\n", plugin->name);
        dlclose(handle);
        return -1;
    }

    pm->plugins[pm->count].handle = handle;
    pm->plugins[pm->count].plugin = plugin;
    pm->count++;

    printf("已加载插件: %s v%s\n", plugin->name, plugin->version);
    return 0;
}

// 执行插件
int pm_execute(PluginManager *pm, const char *plugin_name,
               const char *input, char *output, int max_len) {
    for (int i = 0; i < pm->count; i++) {
        if (strcmp(pm->plugins[i].plugin->name, plugin_name) == 0) {
            return pm->plugins[i].plugin->execute(input, output, max_len);
        }
    }
    fprintf(stderr, "找不到插件: %s\n", plugin_name);
    return -1;
}

// 卸载所有插件
void pm_cleanup(PluginManager *pm) {
    for (int i = 0; i < pm->count; i++) {
        Plugin *p = pm->plugins[i].plugin;
        if (p->cleanup) p->cleanup();
        dlclose(pm->plugins[i].handle);
    }
    pm->count = 0;
}

// 列出所有插件
void pm_list(PluginManager *pm) {
    printf("已加载插件 (%d个):\n", pm->count);
    for (int i = 0; i < pm->count; i++) {
        Plugin *p = pm->plugins[i].plugin;
        printf("  [%d] %s v%s\n", i, p->name, p->version);
    }
}
```

### 主程序

```c
// main.c
#include <stdio.h>
#include "plugin_interface.h"

// 上述PluginManager代码省略...

int main(void) {
    PluginManager pm;
    pm_init(&pm);

    // 加载插件
    pm_load(&pm, "./plugin_hello.so");
    pm_load(&pm, "./plugin_upper.so");

    // 列出插件
    pm_list(&pm);

    // 执行插件
    char output[256];
    pm_execute(&pm, "Hello Plugin", "世界", output, sizeof(output));
    printf("结果: %s\n", output);

    // 清理
    pm_cleanup(&pm);
    return 0;
}
```

## 4. 编译命令

```bash
# 编译共享库
gcc -shared -fPIC -o libmyplugin.so plugin_hello.c
gcc -shared -fPIC -o plugin_upper.so plugin_upper.c

# 编译主程序（链接dl库）
gcc -o app main.c -ldl

# 查看共享库依赖
ldd app
nm -D libmyplugin.so    # 查看导出符号
objdump -T libmyplugin.so
```

## 5. Windows版本

```c
#ifdef _WIN32
#include <windows.h>

typedef struct {
    HMODULE handle;
    Plugin *plugin;
} WinPlugin;

int load_plugin(WinPlugin *wp, const char *path) {
    wp->handle = LoadLibraryA(path);
    if (!wp->handle) {
        fprintf(stderr, "LoadLibrary失败: %lu\n", GetLastError());
        return -1;
    }

    wp->plugin = (Plugin *)GetProcAddress(wp->handle, "plugin");
    if (!wp->plugin) {
        fprintf(stderr, "找不到符号\n");
        FreeLibrary(wp->handle);
        return -1;
    }

    return 0;
}

void unload_plugin(WinPlugin *wp) {
    if (wp->plugin && wp->plugin->cleanup)
        wp->plugin->cleanup();
    if (wp->handle)
        FreeLibrary(wp->handle);
}
#endif
```

## 要点总结

1. `dlopen`加载共享库，`dlsym`查找符号，`dlclose`释放
2. 共享库编译需要`-shared -fPIC`选项
3. 插件系统的核心是定义统一接口，各插件实现接口
4. Windows下使用`LoadLibrary`/`GetProcAddress`/`FreeLibrary`
5. 注意符号可见性控制（`-fvisibility=hidden`）
6. 动态加载的错误处理很重要，`dlerror()`提供详细错误信息
