# 08 - IO多路复用

## 概述

IO多路复用允许单个线程同时监控多个文件描述符（socket等），是构建高性能服务器的核心技术。Linux下主要有select、poll和epoll三种实现。

---

## 1. 阻塞IO vs 非阻塞IO

```c
#include <fcntl.h>
#include <unistd.h>

// 设置非阻塞
int set_nonblocking(int fd) {
    int flags = fcntl(fd, F_GETFL, 0);
    if (flags == -1) return -1;
    return fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

// 阻塞IO：read会一直等待直到有数据
// 非阻塞IO：read立即返回，无数据时返回-1，errno=EAGAIN
```

## 2. select

### 特点

- 最古老的IO多路复用接口（POSIX标准）
- 最大文件描述符限制：通常1024（FD_SETSIZE）
- 每次调用需要重新设置fd_set

```c
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <sys/select.h>
#include <sys/socket.h>
#include <netinet/in.h>

#define MAX_FD 1024

void select_server(int listen_fd) {
    fd_set read_fds, master_fds;
    int max_fd = listen_fd;

    FD_ZERO(&master_fds);
    FD_SET(listen_fd, &master_fds);

    while (1) {
        read_fds = master_fds;  // 每次都要重新赋值

        int ready = select(max_fd + 1, &read_fds, NULL, NULL, NULL);
        if (ready < 0) {
            perror("select");
            break;
        }

        for (int fd = 0; fd <= max_fd; fd++) {
            if (!FD_ISSET(fd, &read_fds)) continue;

            if (fd == listen_fd) {
                // 新连接
                struct sockaddr_in client_addr;
                socklen_t addr_len = sizeof(client_addr);
                int client_fd = accept(listen_fd,
                    (struct sockaddr *)&client_addr, &addr_len);
                if (client_fd >= 0) {
                    FD_SET(client_fd, &master_fds);
                    if (client_fd > max_fd) max_fd = client_fd;
                    printf("新连接: fd=%d\n", client_fd);
                }
            } else {
                // 已有连接的数据
                char buf[1024];
                int n = read(fd, buf, sizeof(buf) - 1);
                if (n <= 0) {
                    close(fd);
                    FD_CLR(fd, &master_fds);
                    printf("连接关闭: fd=%d\n", fd);
                } else {
                    buf[n] = '\0';
                    printf("收到数据 [%d]: %s", fd, buf);
                    write(fd, buf, n);  // echo
                }
            }
        }
    }
}
```

## 3. poll

### 特点

- 无文件描述符数量限制
- 使用pollfd数组而非位图
- 每次调用仍需重新传递数组

```c
#include <poll.h>
#include <stdlib.h>

#define MAX_CLIENTS 1024

void poll_server(int listen_fd) {
    struct pollfd *fds = calloc(MAX_CLIENTS, sizeof(struct pollfd));
    int nfds = 1;

    fds[0].fd = listen_fd;
    fds[0].events = POLLIN;

    while (1) {
        int ready = poll(fds, nfds, -1);
        if (ready < 0) {
            perror("poll");
            break;
        }

        // 新连接
        if (fds[0].revents & POLLIN) {
            struct sockaddr_in client_addr;
            socklen_t addr_len = sizeof(client_addr);
            int client_fd = accept(listen_fd,
                (struct sockaddr *)&client_addr, &addr_len);

            if (client_fd >= 0 && nfds < MAX_CLIENTS) {
                fds[nfds].fd = client_fd;
                fds[nfds].events = POLLIN;
                nfds++;
                printf("新连接: fd=%d\n", client_fd);
            }
        }

        // 处理已连接的客户端
        for (int i = 1; i < nfds; i++) {
            if (fds[i].revents & POLLIN) {
                char buf[1024];
                int n = read(fds[i].fd, buf, sizeof(buf) - 1);

                if (n <= 0) {
                    close(fds[i].fd);
                    // 将最后一个移到当前位置
                    fds[i] = fds[nfds - 1];
                    nfds--;
                    i--;
                } else {
                    buf[n] = '\0';
                    printf("[%d]: %s", fds[i].fd, buf);
                    write(fds[i].fd, buf, n);
                }
            }
        }
    }

    free(fds);
}
```

## 4. epoll

### 特点

- Linux专有，性能最优
- 基于事件驱动，无需每次遍历所有fd
- 支持水平触发（LT）和边缘触发（ET）

```c
#include <sys/epoll.h>
#include <unistd.h>

#define MAX_EVENTS 64

void epoll_server(int listen_fd) {
    int epfd = epoll_create1(0);
    if (epfd < 0) { perror("epoll_create1"); return; }

    // 添加监听socket
    struct epoll_event ev;
    ev.events = EPOLLIN;
    ev.data.fd = listen_fd;
    if (epoll_ctl(epfd, EPOLL_CTL_ADD, listen_fd, &ev) < 0) {
        perror("epoll_ctl");
        close(epfd);
        return;
    }

    struct epoll_event events[MAX_EVENTS];

    while (1) {
        int nfds = epoll_wait(epfd, events, MAX_EVENTS, -1);
        if (nfds < 0) {
            perror("epoll_wait");
            break;
        }

        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == listen_fd) {
                // 新连接
                struct sockaddr_in client_addr;
                socklen_t addr_len = sizeof(client_addr);
                int client_fd = accept(listen_fd,
                    (struct sockaddr *)&client_addr, &addr_len);
                if (client_fd < 0) continue;

                // 设置非阻塞（ET模式必须）
                set_nonblocking(client_fd);

                ev.events = EPOLLIN | EPOLLET;  // 边缘触发
                ev.data.fd = client_fd;
                epoll_ctl(epfd, EPOLL_CTL_ADD, client_fd, &ev);
                printf("新连接: fd=%d\n", client_fd);
            } else {
                // 数据到达
                char buf[4096];
                int fd = events[i].data.fd;

                // ET模式需要一次读完
                while (1) {
                    int n = read(fd, buf, sizeof(buf));
                    if (n > 0) {
                        write(fd, buf, n);  // echo
                    } else if (n == 0) {
                        // 连接关闭
                        close(fd);
                        epoll_ctl(epfd, EPOLL_CTL_DEL, fd, NULL);
                        printf("连接关闭: fd=%d\n", fd);
                        break;
                    } else {
                        if (errno == EAGAIN) break;  // 数据读完
                        perror("read");
                        close(fd);
                        break;
                    }
                }
            }
        }
    }

    close(epfd);
}
```

## 5. LT vs ET模式

| 模式 | 触发方式 | 特点 |
|------|----------|------|
| LT（水平触发） | 只要有数据就触发 | 默认模式，编程简单 |
| ET（边缘触发） | 状态变化时才触发 | 必须一次读完，性能更高 |

```c
// ET模式读取必须循环读到EAGAIN
void et_read(int fd) {
    char buf[4096];
    while (1) {
        ssize_t n = read(fd, buf, sizeof(buf));
        if (n > 0) {
            // 处理数据
        } else if (n == 0) {
            // 连接关闭
            break;
        } else {
            if (errno == EAGAIN || errno == EWOULDBLOCK)
                break;  // 数据读完
            perror("read");
            break;
        }
    }
}
```

## 6. 三种方式对比

| 特性 | select | poll | epoll |
|------|--------|------|-------|
| 最大fd数 | 1024 | 无限制 | 无限制 |
| fd传递方式 | 位图每次拷贝 | 数组每次拷贝 | epoll_ctl注册 |
| 事件检测 | 遍历O(n) | 遍历O(n) | 回调O(1) |
| 触发模式 | LT | LT | LT + ET |
| 平台 | 跨平台 | 跨平台 | Linux |

---

## 要点总结

1. select最古老，跨平台但有fd数量限制
2. poll解决了数量限制，但性能仍为O(n)遍历
3. epoll是Linux最优方案，基于事件回调
4. ET模式性能更高但编程更复杂
5. 实际项目中推荐使用libevent/libuv等封装库
