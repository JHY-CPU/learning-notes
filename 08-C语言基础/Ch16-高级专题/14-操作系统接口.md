# 14 - 操作系统接口

## 概述

POSIX（Portable Operating System Interface）定义了一套标准的操作系统API，使得C程序可以在不同Unix-like系统间移植。本节介绍常用的POSIX API和系统调用封装。

---

## 1. 文件操作

```c
#include <fcntl.h>
#include <unistd.h>
#include <sys/stat.h>
#include <stdio.h>
#include <string.h>
#include <errno.h>

// 打开文件
int file_demo(void) {
    // open: 打开或创建文件
    int fd = open("test.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd < 0) {
        perror("open");
        return -1;
    }

    // write: 写入数据
    const char *msg = "Hello, POSIX!\n";
    ssize_t n = write(fd, msg, strlen(msg));
    if (n < 0) {
        perror("write");
        close(fd);
        return -1;
    }

    close(fd);

    // read: 读取文件
    fd = open("test.txt", O_RDONLY);
    if (fd < 0) return -1;

    char buf[256];
    n = read(fd, buf, sizeof(buf) - 1);
    if (n > 0) {
        buf[n] = '\0';
        printf("读取: %s", buf);
    }

    close(fd);
    return 0;
}

// 文件信息
void file_stat(const char *path) {
    struct stat st;
    if (stat(path, &st) < 0) {
        perror("stat");
        return;
    }

    printf("文件: %s\n", path);
    printf("大小: %ld 字节\n", (long)st.st_size);
    printf("权限: %o\n", st.st_mode & 0777);
    printf("修改时间: %ld\n", (long)st.st_mtime);

    if (S_ISREG(st.st_mode)) printf("类型: 普通文件\n");
    if (S_ISDIR(st.st_mode)) printf("类型: 目录\n");
    if (S_ISLNK(st.st_mode)) printf("类型: 符号链接\n");
}
```

## 2. 目录操作

```c
#include <dirent.h>
#include <sys/stat.h>

// 遍历目录
int list_directory(const char *path) {
    DIR *dir = opendir(path);
    if (!dir) {
        perror("opendir");
        return -1;
    }

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        // 跳过.和..
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0)
            continue;

        char fullpath[1024];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        struct stat st;
        if (stat(fullpath, &st) == 0) {
            printf("%c %8ld %s\n",
                   S_ISDIR(st.st_mode) ? 'd' : '-',
                   (long)st.st_size,
                   entry->d_name);
        }
    }

    closedir(dir);
    return 0;
}

// 递归遍历目录
void walk_directory(const char *path, int depth) {
    DIR *dir = opendir(path);
    if (!dir) return;

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        if (entry->d_name[0] == '.') continue;

        for (int i = 0; i < depth; i++) printf("  ");
        printf("%s\n", entry->d_name);

        char fullpath[1024];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        struct stat st;
        if (stat(fullpath, &st) == 0 && S_ISDIR(st.st_mode)) {
            walk_directory(fullpath, depth + 1);
        }
    }

    closedir(dir);
}
```

## 3. 进程管理

```c
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>

void process_demo(void) {
    // 获取进程信息
    printf("PID: %d\n", getpid());
    printf("PPID: %d\n", getppid());
    printf("UID: %d\n", getuid());
    printf("GID: %d\n", getgid());

    // 创建子进程
    pid_t pid = fork();
    if (pid < 0) {
        perror("fork");
        return;
    }

    if (pid == 0) {
        // 子进程
        printf("子进程 PID=%d\n", getpid());

        // 执行外部程序
        execlp("ls", "ls", "-la", NULL);
        perror("exec");  // exec失败才会到这里
        exit(1);
    } else {
        // 父进程
        printf("父进程 PID=%d, 子进程 PID=%d\n", getpid(), pid);

        int status;
        waitpid(pid, &status, 0);  // 等待子进程结束

        if (WIFEXITED(status)) {
            printf("子进程退出码: %d\n", WEXITSTATUS(status));
        }
        if (WIFSIGNALED(status)) {
            printf("子进程被信号%d终止\n", WTERMSIG(status));
        }
    }
}
```

## 4. 管道通信

```c
#include <unistd.h>
#include <string.h>
#include <stdio.h>

void pipe_demo(void) {
    int pipefd[2];
    if (pipe(pipefd) < 0) {
        perror("pipe");
        return;
    }

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程：读取
        close(pipefd[1]);  // 关闭写端

        char buf[256];
        ssize_t n = read(pipefd[0], buf, sizeof(buf) - 1);
        if (n > 0) {
            buf[n] = '\0';
            printf("子进程收到: %s", buf);
        }

        close(pipefd[0]);
        exit(0);
    } else {
        // 父进程：写入
        close(pipefd[0]);  // 关闭读端

        const char *msg = "来自父进程的消息\n";
        write(pipefd[1], msg, strlen(msg));

        close(pipefd[1]);
        wait(NULL);
    }
}
```

## 5. 内存映射

```c
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>

void mmap_demo(void) {
    // 创建文件
    int fd = open("mmap_test.dat", O_RDWR | O_CREAT | O_TRUNC, 0644);
    if (fd < 0) { perror("open"); return; }

    size_t size = 4096;
    ftruncate(fd, size);  // 设置文件大小

    // 内存映射
    void *addr = mmap(NULL, size, PROT_READ | PROT_WRITE,
                      MAP_SHARED, fd, 0);
    if (addr == MAP_FAILED) {
        perror("mmap");
        close(fd);
        return;
    }

    // 直接通过指针访问文件内容
    char *data = (char *)addr;
    strcpy(data, "通过mmap写入的内容");

    // 解除映射
    munmap(addr, size);
    close(fd);
}

// 匿名映射（用于进程间共享内存）
void anonymous_mmap(void) {
    size_t size = 4096;
    void *addr = mmap(NULL, size, PROT_READ | PROT_WRITE,
                      MAP_SHARED | MAP_ANONYMOUS, -1, 0);

    if (addr == MAP_FAILED) {
        perror("mmap");
        return;
    }

    // 父子进程共享
    pid_t pid = fork();
    if (pid == 0) {
        strcpy((char *)addr, "子进程写入");
        exit(0);
    } else {
        wait(NULL);
        printf("父进程读取: %s\n", (char *)addr);
        munmap(addr, size);
    }
}
```

## 6. 环境变量与系统信息

```c
#include <stdlib.h>
#include <sys/utsname.h>

void sysinfo_demo(void) {
    // 环境变量
    const char *path = getenv("PATH");
    printf("PATH: %s\n", path ? path : "未设置");

    setenv("MY_VAR", "hello", 1);
    printf("MY_VAR: %s\n", getenv("MY_VAR"));
    unsetenv("MY_VAR");

    // 系统信息
    struct utsname uts;
    uname(&uts);
    printf("系统: %s %s\n", uts.sysname, uts.release);
    printf("主机: %s\n", uts.nodename);
    printf("架构: %s\n", uts.machine);

    // CPU核心数
    long nprocs = sysconf(_SC_NPROCESSORS_ONLN);
    printf("CPU核心数: %ld\n", nprocs);

    // 页面大小
    long pagesize = sysconf(_SC_PAGESIZE);
    printf("页面大小: %ld\n", pagesize);
}
```

## 7. 错误处理封装

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <string.h>

// 系统调用封装，出错时自动打印错误并退出
void xclose(int fd) {
    if (close(fd) < 0) {
        perror("close");
    }
}

ssize_t xwrite(int fd, const void *buf, size_t count) {
    ssize_t n = write(fd, buf, count);
    if (n < 0) {
        perror("write");
        exit(EXIT_FAILURE);
    }
    return n;
}

pid_t xfork(void) {
    pid_t pid = fork();
    if (pid < 0) {
        perror("fork");
        exit(EXIT_FAILURE);
    }
    return pid;
}

void *xmalloc(size_t size) {
    void *ptr = malloc(size);
    if (!ptr) {
        fprintf(stderr, "malloc failed: %s\n", strerror(errno));
        exit(EXIT_FAILURE);
    }
    return ptr;
}
```

## 要点总结

1. POSIX API提供了跨平台的系统编程接口
2. 文件操作使用open/read/write/close，而非fopen等标准IO
3. fork创建子进程，exec系列替换进程镜像
4. 管道用于父子进程间通信
5. mmap可以将文件映射到内存，实现高效文件IO
6. 系统调用失败时应检查errno并处理错误
