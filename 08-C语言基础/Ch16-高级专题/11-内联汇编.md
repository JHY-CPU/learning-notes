# 11 - 内联汇编

## 概述

GCC内联汇编允许在C代码中直接嵌入汇编指令，用于性能优化、访问特殊CPU指令或直接操作硬件。

---

## 1. 基本语法

```c
// 基本格式
asm("汇编指令");

// 完整格式（扩展asm）
asm (
    "汇编模板"
    : 输出操作数
    : 输入操作数
    : 破坏的寄存器
);
```

---

## 2. 基本内联汇编

```c
#include <stdio.h>

void basic_asm(void) {
    // 简单的内联汇编
    asm("nop");           // 空操作
    asm("movl $42, %eax"); // 设置eax=42（不推荐，可能破坏寄存器）

    // 使用volatile防止编译器优化
    __asm__ __volatile__("nop");
}
```

---

## 3. 扩展内联汇编

### 约束符号

| 约束 | 含义 |
|------|------|
| "r" | 任意通用寄存器 |
| "a" | EAX/AX/AL |
| "b" | EBX/BX/BL |
| "c" | ECX/CX/CL |
| "d" | EDX/DX/DL |
| "S" | ESI/SI |
| "D" | EDI/DI |
| "m" | 内存操作数 |
| "i" | 立即数（编译时常量） |
| "g" | 通用（寄存器/内存/立即数） |

### 示例

```c
// 加法示例
int add_asm(int a, int b) {
    int result;
    __asm__ (
        "addl %1, %2\n\t"
        "movl %2, %0"
        : "=r"(result)       // 输出：result = %0
        : "r"(a), "r"(b)    // 输入：a = %1, b = %2
    );
    return result;
}

// 更简洁的写法
int add_asm2(int a, int b) {
    int result;
    __asm__ (
        "addl %2, %0"
        : "=r"(result)
        : "0"(a), "r"(b)    // "0"表示使用与输出%0相同的寄存器
    );
    return result;
}
```

## 4. 输入输出约束详解

```c
// 读取CPU时间戳计数器
static inline unsigned long long rdtsc(void) {
    unsigned long long result;
    __asm__ __volatile__ (
        "rdtsc"                    // 读取时间戳
        : "=A"(result)             // 输出：EDX:EAX -> result
        :                          // 无输入
        : "edx", "eax"            // 破坏的寄存器
    );
    return result;
}

// 原子比较交换
static inline int cmpxchg(int *ptr, int old_val, int new_val) {
    int prev;
    __asm__ __volatile__ (
        "lock cmpxchgl %2, %1"
        : "=a"(prev), "+m"(*ptr)  // 输出：prev=EAX, *ptr被修改
        : "r"(new_val), "0"(old_val) // 输入：new_val在寄存器, old_val=EAX
        : "memory"
    );
    return prev;
}

// 内存屏障
static inline void memory_barrier(void) {
    __asm__ __volatile__ ("mfence" ::: "memory");
}

static inline void compiler_barrier(void) {
    __asm__ __volatile__ ("" ::: "memory");
}
```

## 5. 常用示例

### CPUID指令

```c
#include <cpuid.h>  // GCC内置

void cpuid_demo(void) {
    unsigned int eax, ebx, ecx, edx;

    // 获取CPU厂商字符串
    __cpuid(0, eax, ebx, ecx, edx);
    char vendor[13];
    *(unsigned int *)&vendor[0] = ebx;
    *(unsigned int *)&vendor[4] = edx;
    *(unsigned int *)&vendor[8] = ecx;
    vendor[12] = '\0';
    printf("CPU厂商: %s\n", vendor);

    // 获取CPU特性
    __cpuid(1, eax, ebx, ecx, edx);
    printf("SSE支持: %s\n", (edx & (1 << 25)) ? "是" : "否");
    printf("SSE2支持: %s\n", (edx & (1 << 26)) ? "是" : "否");
    printf("AVX支持: %s\n", (ecx & (1 << 28)) ? "是" : "否");
}
```

### 位操作优化

```c
// 计算前导零个数
static inline int clz(unsigned int x) {
    int result;
    __asm__ (
        "bsrl %1, %0\n\t"    // 从低位扫描
        "jnz 1f\n\t"
        "movl $-1, %0\n\t"
        "1:"
        : "=r"(result)
        : "r"(x)
    );
    return 31 - result;
}

// 或使用GCC内建函数（推荐）
// int clz = __builtin_clz(x);
```

### 交换两个值

```c
void xchg(int *a, int *b) {
    __asm__ (
        "movl (%0), %%eax\n\t"
        "xchgl %%eax, (%1)\n\t"
        "movl %%eax, (%0)"
        :
        : "r"(a), "r"(b)
        : "eax", "memory"
    );
}
```

## 6. AT&T vs Intel语法

```c
// GCC默认使用AT&T语法
// AT&T: src, dst（源在前）
// Intel: dst, src（目标在前）

// AT&T语法
__asm__ ("movl %eax, %ebx");      // eax -> ebx
__asm__ ("addl $4, %eax");        // eax += 4
__asm__ ("movl (%eax), %ebx");    // ebx = *eax

// 使用Intel语法
__asm__ (".intel_syntax noprefix\n"
         "mov ebx, eax\n"
         "add eax, 4\n"
         ".att_syntax prefix\n");
```

## 7. 注意事项

```c
// 1. 始终声明破坏的寄存器
__asm__ (
    "cpuid"
    : "=a"(a), "=b"(b), "=c"(c), "=d"(d)
    : "0"(0)
    // 这里不需要额外声明破坏，因为输出已覆盖
);

// 2. 访问C变量名
int var = 10;
__asm__ (
    "incl %0"
    : "+r"(var)     // + 表示读写
);

// 3. 使用符号名代替数字
__asm__ (
    "addl %[val], %[result]"
    : [result] "=r"(var)
    : [val] "r"(10)
);
```

---

## 要点总结

1. 内联汇编应尽量少用，仅在需要特殊指令或极致优化时使用
2. 正确使用约束和破坏声明，避免未定义行为
3. `__volatile__`防止编译器优化掉关键汇编代码
4. GCC内建函数（如`__builtin_clz`）通常比手写汇编更安全
5. 内联汇编不可移植，不同架构需要不同的实现
