# 22 - 设计模式在C中

## 概述

虽然设计模式通常与面向对象语言关联，但许多模式可以在C语言中通过函数指针和结构体实现。本节介绍单例、工厂、观察者和策略模式的C语言实现。

## 1. 单例模式（Singleton）

### 确保全局只有一个实例

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

typedef struct {
    int value;
    char name[64];
} Config;

static Config *instance = NULL;
static pthread_once_t once = PTHREAD_ONCE_INIT;

// 线程安全的初始化
static void config_init(void) {
    instance = (Config *)malloc(sizeof(Config));
    if (instance) {
        instance->value = 0;
        snprintf(instance->name, sizeof(instance->name), "默认配置");
    }
}

// 获取单例
Config *config_get_instance(void) {
    pthread_once(&once, config_init);
    return instance;
}

// 使用
void config_set_value(Config *cfg, int val) {
    if (cfg) cfg->value = val;
}

int config_get_value(Config *cfg) {
    return cfg ? cfg->value : 0;
}

void config_destroy(void) {
    free(instance);
    instance = NULL;
}

// 使用示例
void singleton_demo(void) {
    Config *cfg1 = config_get_instance();
    Config *cfg2 = config_get_instance();

    config_set_value(cfg1, 42);
    printf("cfg1的值: %d\n", config_get_value(cfg1));  // 42
    printf("cfg2的值: %d\n", config_get_value(cfg2));  // 42（同一实例）
    printf("是同一实例: %s\n", cfg1 == cfg2 ? "是" : "否");
}
```

## 2. 工厂模式（Factory）

### 通过接口创建不同类型的对象

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 产品接口（虚函数表）
typedef struct Shape Shape;
typedef struct {
    void (*draw)(Shape *self);
    double (*area)(Shape *self);
    void (*destroy)(Shape *self);
} ShapeVTable;

struct Shape {
    ShapeVTable *vtable;
};

// 具体产品：圆形
typedef struct {
    Shape base;
    double radius;
} Circle;

static void circle_draw(Shape *self) {
    Circle *c = (Circle *)self;
    printf("绘制圆形: 半径=%.1f\n", c->radius);
}

static double circle_area(Shape *self) {
    Circle *c = (Circle *)self;
    return 3.14159265 * c->radius * c->radius;
}

static ShapeVTable circle_vtable = {
    .draw = circle_draw,
    .area = circle_area,
    .destroy = free
};

// 具体产品：矩形
typedef struct {
    Shape base;
    double width, height;
} Rectangle;

static void rect_draw(Shape *self) {
    Rectangle *r = (Rectangle *)self;
    printf("绘制矩形: %.1f x %.1f\n", r->width, r->height);
}

static double rect_area(Shape *self) {
    Rectangle *r = (Rectangle *)self;
    return r->width * r->height;
}

static ShapeVTable rect_vtable = {
    .draw = rect_draw,
    .area = rect_area,
    .destroy = free
};

// 工厂函数
typedef enum { SHAPE_CIRCLE, SHAPE_RECTANGLE } ShapeType;

Shape *shape_create(ShapeType type, ...) {
    switch (type) {
    case SHAPE_CIRCLE: {
        Circle *c = (Circle *)malloc(sizeof(Circle));
        c->base.vtable = &circle_vtable;
        c->radius = 1.0;  // 默认值
        return (Shape *)c;
    }
    case SHAPE_RECTANGLE: {
        Rectangle *r = (Rectangle *)malloc(sizeof(Rectangle));
        r->base.vtable = &rect_vtable;
        r->width = 1.0;
        r->height = 1.0;
        return (Shape *)r;
    }
    default:
        return NULL;
    }
}

// 多态调用
void shape_draw(Shape *s) { s->vtable->draw(s); }
double shape_area(Shape *s) { return s->vtable->area(s); }
void shape_destroy(Shape *s) { s->vtable->destroy(s); }

void factory_demo(void) {
    Shape *shapes[2];
    shapes[0] = shape_create(SHAPE_CIRCLE);
    shapes[1] = shape_create(SHAPE_RECTANGLE);

    for (int i = 0; i < 2; i++) {
        shape_draw(shapes[i]);
        printf("面积: %.2f\n\n", shape_area(shapes[i]));
        shape_destroy(shapes[i]);
    }
}
```

## 3. 观察者模式（Observer）

### 事件通知机制

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_OBSERVERS 32

// 观察者接口
typedef struct Observer Observer;
typedef void (*ObserverCallback)(Observer *self, int event, void *data);

struct Observer {
    char name[64];
    ObserverCallback on_notify;
};

// 被观察者（主题）
typedef struct {
    Observer *observers[MAX_OBSERVERS];
    int count;
} EventEmitter;

void emitter_init(EventEmitter *em) {
    em->count = 0;
}

void emitter_subscribe(EventEmitter *em, Observer *obs) {
    if (em->count < MAX_OBSERVERS) {
        em->observers[em->count++] = obs;
    }
}

void emitter_unsubscribe(EventEmitter *em, Observer *obs) {
    for (int i = 0; i < em->count; i++) {
        if (em->observers[i] == obs) {
            // 移动后面的元素
            for (int j = i; j < em->count - 1; j++) {
                em->observers[j] = em->observers[j + 1];
            }
            em->count--;
            break;
        }
    }
}

void emitter_emit(EventEmitter *em, int event, void *data) {
    for (int i = 0; i < em->count; i++) {
        if (em->observers[i]->on_notify) {
            em->observers[i]->on_notify(em->observers[i], event, data);
        }
    }
}

// 具体观察者
static void log_observer(Observer *self, int event, void *data) {
    printf("[日志-%s] 事件=%d, 数据=%s\n", self->name, event,
           data ? (char *)data : "无");
}

static void alert_observer(Observer *self, int event, void *data) {
    if (event == 1) {  // 错误事件
        printf("[报警-%s] 发生错误: %s\n", self->name,
               data ? (char *)data : "未知");
    }
}

void observer_demo(void) {
    EventEmitter emitter;
    emitter_init(&emitter);

    Observer logger = {.on_notify = log_observer};
    strcpy(logger.name, "Logger");

    Observer alerter = {.on_notify = alert_observer};
    strcpy(alerter.name, "Alerter");

    emitter_subscribe(&emitter, &logger);
    emitter_subscribe(&emitter, &alerter);

    emitter_emit(&emitter, 0, "系统启动");
    emitter_emit(&emitter, 1, "磁盘空间不足");
}
```

## 4. 策略模式（Strategy）

### 运行时切换算法

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 策略接口
typedef int (*CompareFunc)(const void *a, const void *b);

// 具体策略
int int_ascending(const void *a, const void *b) {
    return *(int *)a - *(int *)b;
}

int int_descending(const void *a, const void *b) {
    return *(int *)b - *(int *)a;
}

int str_ascending(const void *a, const void *b) {
    return strcmp(*(const char **)a, *(const char **)b);
}

// 上下文
typedef struct {
    void *data;
    size_t count;
    size_t elem_size;
    CompareFunc compare;
} Sorter;

Sorter *sorter_create(void *data, size_t count,
                      size_t elem_size, CompareFunc compare) {
    Sorter *s = (Sorter *)malloc(sizeof(Sorter));
    s->data = data;
    s->count = count;
    s->elem_size = elem_size;
    s->compare = compare;
    return s;
}

void sorter_set_strategy(Sorter *s, CompareFunc compare) {
    s->compare = compare;
}

void sorter_execute(Sorter *s) {
    qsort(s->data, s->count, s->elem_size, s->compare);
}

void sorter_destroy(Sorter *s) {
    free(s);
}

void strategy_demo(void) {
    int arr[] = {5, 2, 8, 1, 9, 3};
    int n = sizeof(arr) / sizeof(arr[0]);

    Sorter *s = sorter_create(arr, n, sizeof(int), int_ascending);

    sorter_execute(s);
    printf("升序: ");
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");

    // 切换策略
    sorter_set_strategy(s, int_descending);
    sorter_execute(s);
    printf("降序: ");
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");

    sorter_destroy(s);
}
```

## 5. 模式选择指南

| 模式 | 解决的问题 | C实现要点 |
|------|-----------|----------|
| 单例 | 全局唯一实例 | 静态变量 + 线程安全初始化 |
| 工厂 | 对象创建解耦 | 枚举 + 工厂函数 |
| 观察者 | 事件通知 | 回调函数 + 观察者列表 |
| 策略 | 算法可替换 | 函数指针 |
| 装饰器 | 功能扩展 | 嵌套结构体 |

---

## 要点总结

1. C语言通过函数指针和结构体实现面向对象的设计模式
2. 虚函数表（vtable）模拟多态
3. 单例模式需注意线程安全
4. 策略模式通过函数指针实现算法替换
5. 观察者模式适合事件驱动的系统
6. 设计模式应适度使用，避免过度设计
