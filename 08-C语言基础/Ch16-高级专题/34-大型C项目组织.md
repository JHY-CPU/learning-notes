# 34 - 大型C项目组织

## 概述

大型C项目需要合理的目录结构、构建系统和模块划分。本节介绍工业级C项目的组织方法。

## 2. CMake构建系统

### 顶层CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.14)
project(MyProject VERSION 1.0.0 LANGUAGES C)

# C标准
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# 编译选项
if(CMAKE_COMPILER_IS_GNUCC)
    add_compile_options(-Wall -Wextra -Wpedantic -Werror)
endif()

# Debug/Release配置
set(CMAKE_C_FLAGS_DEBUG "-g -O0 -DDEBUG")
set(CMAKE_C_FLAGS_RELEASE "-O2 -DNDEBUG")

# 输出目录
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# 包含目录
include_directories(${PROJECT_SOURCE_DIR}/include)

# 子目录
add_subdirectory(src)
enable_testing()
add_subdirectory(tests)
```

### src/CMakeLists.txt

```cmake
# 库源文件
set(LIB_SOURCES
    core/engine.c
    utils/string_utils.c
    utils/file_utils.c
)

# 创建静态库
add_library(myproject_lib STATIC ${LIB_SOURCES})
target_include_directories(myproject_lib PUBLIC
    ${PROJECT_SOURCE_DIR}/include
)

# 可执行文件
add_executable(myproject main.c)
target_link_libraries(myproject PRIVATE myproject_lib)

# 平台相关
if(WIN32)
    target_sources(myproject_lib PRIVATE platform/windows/platform.c)
else()
    target_sources(myproject_lib PRIVATE platform/linux/platform.c)
    target_link_libraries(myproject_lib PUBLIC pthread m)
endif()
```

### tests/CMakeLists.txt

```cmake
# 测试源文件
set(TEST_SOURCES
    test_main.c
    test_engine.c
    test_utils.c
)

add_executable(tests ${TEST_SOURCES})
target_link_libraries(tests PRIVATE myproject_lib)

# 添加测试
add_test(NAME all_tests COMMAND tests)
```

## 3. 模块划分原则

### 单一职责

```c
// 每个模块只负责一件事
// string_utils.h - 只做字符串操作
// file_utils.h   - 只做文件操作
// logger.h       - 只做日志记录
// network.h      - 只做网络通信
```

### 接口与实现分离

```c
// include/myproject/parser.h - 公共接口
#ifndef MYPARSER_H
#define MYPARSER_H

typedef struct Parser Parser;

Parser *parser_create(void);
void parser_destroy(Parser *p);
int parser_parse(Parser *p, const char *input);
const char *parser_get_result(Parser *p);

#endif

// src/parser/parser_internal.h - 内部头文件（不安装）
#ifndef PARSER_INTERNAL_H
#define PARSER_INTERNAL_H

// 内部数据结构，外部不可见
struct Parser {
    char *buffer;
    size_t pos;
    int state;
    // ...
};

// 内部辅助函数
static void parser_advance(Parser *p);
static int parser_peek(Parser *p);

#endif

// src/parser/parser.c - 实现
#include "myproject/parser.h"
#include "parser_internal.h"

Parser *parser_create(void) {
    // 实现...
}
```

## 5. 依赖管理

### 方案1：Git子模块

```bash
# 添加子模块
git submodule add https://github.com/example/libuv.git third_party/libuv

# CMake中使用
add_subdirectory(third_party/libuv)
target_link_libraries(myproject PRIVATE uv)
```

### 方案2：FetchContent（CMake 3.11+）

```cmake
include(FetchContent)

FetchContent_Declare(
    sqlite3
    GIT_REPOSITORY https://github.com/sqlite/sqlite.git
    GIT_TAG version-3.42.0
)

FetchContent_MakeAvailable(sqlite3)
target_link_libraries(myproject PRIVATE sqlite3)
```

### 方案3：系统包管理器

```cmake
find_package(SQLite3 REQUIRED)
target_link_libraries(myproject PRIVATE SQLite::SQLite3)
```

## 6. Makefile替代方案

```makefile
# 传统Makefile示例
CC = gcc
CFLAGS = -Wall -Wextra -std=c11 -Iinclude
LDFLAGS = -lm -lpthread

SRC_DIR = src
BUILD_DIR = build
TEST_DIR = tests

SRCS = $(wildcard $(SRC_DIR)/*.c $(SRC_DIR)/**/*.c)
OBJS = $(patsubst $(SRC_DIR)/%.c,$(BUILD_DIR)/%.o,$(SRCS))

TARGET = myproject

.PHONY: all clean test

all: $(TARGET)

$(TARGET): $(OBJS)
	$(CC) $(OBJS) -o $@ $(LDFLAGS)

$(BUILD_DIR)/%.o: $(SRC_DIR)/%.c
	@mkdir -p $(dir $@)
	$(CC) $(CFLAGS) -c $< -o $@

test: $(TARGET)
	$(CC) $(CFLAGS) $(TEST_DIR)/*.c -o tests $(LDFLAGS)
	./tests

clean:
	rm -rf $(BUILD_DIR) $(TARGET)
```

## 7. 代码风格与规范

### .clang-format

```yaml
# .clang-format
BasedOnStyle: LLVM
IndentWidth: 4
UseTab: Never
BreakBeforeBraces: Allman
ColumnLimit: 100
AllowShortFunctionsOnASingleLine: None
AllowShortIfStatementsOnASingleLine: false
```

### 编码规范要点

```c
// 1. 头文件保护
#ifndef MODULE_NAME_H
#define MODULE_NAME_H
// ...
#endif /* MODULE_NAME_H */

// 2. 命名规范
// 类型: PascalCase (MyStruct)
// 函数: snake_case (my_function)
// 宏: UPPER_CASE (MAX_SIZE)
// 常量: UPPER_CASE (DEFAULT_TIMEOUT)
// 私有函数: module_function (_internal_helper)

// 3. 文件头部注释
/**
 * @file engine.c
 * @brief 游戏引擎核心实现
 * @author 作者名
 * @date 2024-01-01
 */

// 4. 函数注释
/**
 * @brief 创建新引擎实例
 * @param config 引擎配置
 * @return 成功返回引擎指针，失败返回NULL
 * @note 调用者负责调用engine_destroy释放
 */
Engine *engine_create(const EngineConfig *config);
```

## 9. CI/CD集成

```yaml
# .github/workflows/ci.yml
name: CI

on: [push, pull_request]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Configure
        run: cmake -B build -DCMAKE_BUILD_TYPE=Release
      - name: Build
        run: cmake --build build
      - name: Test
        run: cd build && ctest --output-on-failure
      - name: Valgrind
        run: valgrind --leak-check=full ./build/bin/tests
```

## 要点总结

1. 合理的目录结构是大型项目的基础
2. CMake是跨平台构建的首选工具
3. 头文件分离公共接口和内部实现
4. 模块化设计遵循单一职责原则
5. 统一的代码风格和命名规范
6. 自动化测试和CI/CD保证代码质量
7. 使用Doxygen等工具生成API文档
