# 进程与多进程编程

## 1. 进程基本概念

进程是操作系统资源分配的基本单位。每个进程拥有独立的虚拟地址空间、文件描述符表、信号处理表等。进程之间天然隔离，通过IPC机制通信。

### Linux进程模型

```
                    init(PID=1)
                   /     |     \
               bash    sshd    ...
              /    \
          gcc     vim
```

每个进程都有唯一的PID（Process ID），父子进程构成进程树。

## 2. fork() 创建进程

`fork()` 是Unix/Linux中最基本的进程创建方式，它创建当前进程的**精确副本**。

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>

void fork_basic_demo() {
    pid_t pid = fork();

    if (pid < 0) {
        // fork失败
        perror("fork failed");
        exit(EXIT_FAILURE);
    } else if (pid == 0) {
        // 子进程（fork返回0）
        printf("子进程: PID=%d, 父PID=%d\n", getpid(), getppid());
    } else {
        // 父进程（fork返回子进程PID）
        printf("父进程: PID=%d, 子进程PID=%d\n", getpid(), pid);
    }

    // 这行代码父子进程都会执行
    printf("此消息会打印两次 (PID=%d)\n", getpid());
}
```

### fork 的关键特性

```c
void fork_characteristics() {
    int variable = 100;
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程修改变量——不影响父进程
        variable = 200;
        printf("子进程: variable = %d (地址=%p)\n", variable, &variable);
    } else {
        sleep(1); // 等子进程先输出
        printf("父进程: variable = %d (地址=%p)\n", variable, &variable);
        // 注意：虽然地址值相同，但指向不同的物理内存！
        // 这就是"写时复制"（Copy-on-Write, COW）机制
    }
}
```

### fork与文件描述符

```c
#include <fcntl.h>

void fork_file_descriptor() {
    int fd = open("/tmp/test.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程共享文件描述符和文件偏移量！
        write(fd, "子进程写入\n", 10);
        close(fd);
    } else {
        wait(NULL);
        write(fd, "父进程写入\n", 10);
        close(fd);
        // 最终文件中两段内容都会有（顺序取决于调度）
    }
}
```

## 3. exec() 族函数

`exec` 系列函数用于在当前进程中**加载并执行新程序**，替换当前进程映像。调用成功后不会返回。

```c
#include <unistd.h>

void exec_demo() {
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程执行 ls -la
        // execl: 使用参数列表
        execl("/bin/ls", "ls", "-la", "/tmp", NULL);
        // 如果exec成功，不会执行到这里
        perror("exec failed");
        _exit(1);
    }
}
```

### exec 族函数对比

```c
void exec_family_demo() {
    // l系列：参数以列表形式传递
    execl("/bin/echo", "echo", "hello", "world", NULL);

    // v系列：参数以数组形式传递
    char *args[] = {"echo", "hello", "world", NULL};
    execv("/bin/echo", args);

    // p系列：在PATH中搜索程序
    execlp("echo", "echo", "hello", NULL);
    execvp("echo", args);

    // e系列：可以自定义环境变量
    char *env[] = {"MY_VAR=test", "PATH=/usr/bin", NULL};
    execle("/bin/echo", "echo", "hello", NULL, env);
    execve("/bin/echo", args, env); // 唯一的系统调用，其他是库函数

    // 所有exec函数失败时返回-1
    perror("exec failed");
}
```

## 4. wait() 与 waitpid()

父进程使用 `wait()` 或 `waitpid()` 等待子进程结束，获取退出状态，并回收僵尸进程。

```c
#include <sys/wait.h>

void wait_demo() {
    pid_t pid = fork();

    if (pid == 0) {
        printf("子进程开始工作 (PID=%d)\n", getpid());
        sleep(2);
        printf("子进程退出\n");
        exit(42); // 退出码42
    } else {
        int status;
        printf("父进程等待子进程...\n");

        pid_t result = wait(&status); // 阻塞等待任意子进程

        if (WIFEXITED(status)) {
            printf("子进程正常退出，退出码: %d\n", WEXITSTATUS(status));
        }
        if (WIFSIGNALED(status)) {
            printf("子进程被信号 %d 终止\n", WTERMSIG(status));
        }
        if (WCOREDUMP(status)) {
            printf("子进程产生了core dump\n");
        }
    }
}

void waitpid_demo() {
    pid_t children[3];

    for (int i = 0; i < 3; i++) {
        children[i] = fork();
        if (children[i] == 0) {
            sleep(i + 1);
            printf("子进程 %d 退出\n", getpid());
            exit(i);
        }
    }

    // 等待特定子进程
    int status;
    pid_t pid = waitpid(children[1], &status, 0);
    printf("子进程 %d 已退出\n", pid);

    // 非阻塞等待任意子进程
    while ((pid = waitpid(-1, &status, WNOHANG)) > 0) {
        printf("回收子进程 %d\n", pid);
    }
}
```

## 5. 孤儿进程与僵尸进程

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

// 孤儿进程：父进程先退出，子进程被init接管
void orphan_process_demo() {
    pid_t pid = fork();

    if (pid == 0) {
        sleep(3); // 父进程先退出
        printf("子进程: PID=%d, 父进程PID=%d (变为1说明成为孤儿)\n",
               getpid(), getppid());
    } else {
        printf("父进程 %d 立即退出\n", getpid());
        exit(0);
        // 子进程将成为孤儿，被PID 1 (init/systemd)接管
    }
}

// 僵尸进程：子进程退出但父进程未回收
void zombie_process_demo() {
    pid_t pid = fork();

    if (pid == 0) {
        printf("子进程退出\n");
        exit(0);
    } else {
        printf("父进程不调用wait()，子进程成为僵尸\n");
        printf("使用 'ps aux | grep Z' 查看僵尸进程\n");
        sleep(10); // 在此期间子进程是僵尸状态
        // 正确做法：wait(NULL);
    }
}

// 方案：使用SIGCHLD信号自动回收
#include <signal.h>

void sigchld_handler(int sig) {
    // 使用while循环回收所有已结束的子进程
    while (waitpid(-1, NULL, WNOHANG) > 0);
}

void auto_reap_demo() {
    signal(SIGCHLD, sigchld_handler);

    for (int i = 0; i < 5; i++) {
        pid_t pid = fork();
        if (pid == 0) {
            sleep(1);
            exit(0);
        }
    }

    sleep(5); // 子进程退出时自动回收
    printf("所有子进程已自动回收\n");
}
```

## 6. 进程间共享信息的方式

```c
// 方式1：通过文件
void share_via_file() {
    pid_t pid = fork();
    if (pid == 0) {
        FILE *f = fopen("/tmp/shared.txt", "w");
        fprintf(f, "来自子进程的数据\n");
        fclose(f);
        exit(0);
    } else {
        wait(NULL);
        FILE *f = fopen("/tmp/shared.txt", "r");
        char buf[256];
        fgets(buf, sizeof(buf), f);
        printf("父进程读取: %s", buf);
        fclose(f);
    }
}

// 方式2：通过管道（pipe）
void share_via_pipe() {
    int pipefd[2];
    pipe(pipefd);

    pid_t pid = fork();
    if (pid == 0) {
        close(pipefd[0]); // 关闭读端
        write(pipefd[1], "hello from child", 16);
        close(pipefd[1]);
        exit(0);
    } else {
        close(pipefd[1]); // 关闭写端
        char buf[64];
        read(pipefd[0], buf, sizeof(buf));
        printf("父进程收到: %s\n", buf);
        close(pipefd[0]);
        wait(NULL);
    }
}
```

## 7. 综合示例：多进程Web服务器骨架

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <signal.h>

void handle_client(int client_fd) {
    // 简化：直接发送响应
    const char *response = "HTTP/1.1 200 OK\r\nContent-Length: 5\r\n\r\nHello";
    write(client_fd, response, strlen(response));
    close(client_fd);
}

void sigchld_handler(int sig) {
    while (waitpid(-1, NULL, WNOHANG) > 0);
}

void multi_process_server(int listen_fd) {
    signal(SIGCHLD, sigchld_handler);

    while (1) {
        int client_fd = accept(listen_fd, NULL, NULL);
        if (client_fd < 0) continue;

        pid_t pid = fork();
        if (pid == 0) {
            close(listen_fd); // 子进程不需要监听socket
            handle_client(client_fd);
            exit(0);
        } else {
            close(client_fd); // 父进程不需要客户端socket
        }
    }
}
```

## 重点与注意事项

1. **fork之后必须检查返回值**：区分父子进程执行路径
2. **及时回收子进程**：使用 `wait/waitpid` 避免僵尸进程
3. **exec失败处理**：exec后应立即 `_exit()`，不要用 `exit()`
4. **文件描述符继承**：fork后子进程继承父进程的文件描述符
5. **写时复制**：fork后父子进程共享物理内存页，写入时才复制
6. **信号处理**：多进程编程中 `SIGCHLD` 信号的处理至关重要
