## 1. 宏基础与进阶

```c
#include <stdio.h>

// 基本宏
#define PI 3.14159
#define MAX(a, b) ((a) > (b) ? (a) : (b))
#define MIN(a, b) ((a) < (b) ? (a) : (b))

// 宏的安全问题
// 错误示例
#define BAD_MAX(a, b) a > b ? a : b

// 测试
void macro_safety(void) {
    int x = BAD_MAX(1, 2) * 3;   // 1 > 2 ? 1 : 2 * 3 = 1 > 2 ? 1 : 6 = 6 (错误!)
    int y = MAX(1, 2) * 3;       // ((1) > (2) ? (1) : (2)) * 3 = 6 (正确!)
    printf("BAD: %d, GOOD: %d\n", x, y);
}

// 副作用问题
void side_effect(void) {
    int i = 5;
    // MAX(i++, 10) 会执行两次i++！
    // int m = MAX(i++, 10);  // 危险！
    // 使用内联函数替代
}
```

## 2. 可变参数宏

```c
#include <stdio.h>

// 调试打印宏
#define DEBUG 1

#if DEBUG
#define LOG(fmt, ...) \
    fprintf(stderr, "[%s:%d] " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)
#else
#define LOG(fmt, ...) ((void)0)
#endif

// 条件日志
#define LOG_IF(cond, fmt, ...) \
    do { if (cond) LOG(fmt, ##__VA_ARGS__); } while(0)

// 带级别的日志
typedef enum { LOG_DEBUG, LOG_INFO, LOG_WARN, LOG_ERROR } LogLevel;
static LogLevel current_level = LOG_DEBUG;

#define LOG_LVL(level, fmt, ...) \
    do { \
        if ((level) >= current_level) { \
            const char *lvl_names[] = {"DEBUG", "INFO", "WARN", "ERROR"}; \
            fprintf(stderr, "[%s] " fmt "\n", \
                    lvl_names[level], ##__VA_ARGS__); \
        } \
    } while(0)

void log_demo(void) {
    LOG("程序启动");
    LOG("值: %d", 42);
    LOG_IF(1 > 0, "条件日志测试");

    LOG_LVL(LOG_DEBUG, "调试信息");
    LOG_LVL(LOG_ERROR, "错误: %s", "文件未找到");
}
```

## 3. X-Macro模式

X-Macro是一种用宏实现的元编程技巧，避免重复代码。

### 基本用法

```c
// 定义枚举和字符串映射
#define COLOR_LIST \
    X(RED) \
    X(GREEN) \
    X(BLUE) \
    X(YELLOW) \
    X(WHITE)

// 生成枚举
#define X(name) COLOR_##name,
typedef enum {
    COLOR_LIST
    COLOR_COUNT
} Color;
#undef X

// 生成字符串数组
#define X(name) #name,
static const char *color_names[] = {
    COLOR_LIST
};
#undef X

void xmacro_enum_demo(void) {
    for (int i = 0; i < COLOR_COUNT; i++) {
        printf("%d: %s\n", i, color_names[i]);
    }
}
```

### 代码生成

```c
// 生成getter/setter
#define FIELD_LIST \
    X(int,    id,      0) \
    X(char*,  name,    NULL) \
    X(double, salary,  0.0) \
    X(int,    age,     0)

typedef struct {
#define X(type, name, default_val) type name;
    FIELD_LIST
#undef X
} Employee;

// 生成初始化函数
Employee employee_create(void) {
    Employee emp = {
#define X(type, name, default_val) .name = default_val,
        FIELD_LIST
#undef X
    };
    return emp;
}

// 生成打印函数
void employee_print(const Employee *emp) {
    printf("Employee {\n");
#define X(type, name, default_val) \
    printf("  %s: ", #name); \
    _Generic((emp->name), \
        int: printf("%d\n", emp->name), \
        double: printf("%.2f\n", emp->name), \
        char*: printf("\"%s\"\n", emp->name ? emp->name : "NULL") \
    );
    FIELD_LIST
#undef X
    printf("}\n");
}

void xmacro_struct_demo(void) {
    Employee emp = employee_create();
    emp.id = 1;
    emp.name = "张三";
    emp.salary = 10000.0;
    emp.age = 30;
    employee_print(&emp);
}
```

### 状态机生成

```c
// 用X-Macro定义状态机
#define STATE_LIST \
    X(IDLE) \
    X(RUNNING) \
    X(PAUSED) \
    X(STOPPED)

#define X(name) STATE_##name,
typedef enum {
    STATE_LIST
} State;
#undef X

#define X(name) #name,
static const char *state_names[] = {
    STATE_LIST
};
#undef X

const char *state_to_string(State s) {
    return (s >= 0 && s < sizeof(state_names)/sizeof(state_names[0]))
           ? state_names[s] : "UNKNOWN";
}
```

## 4. 泛型容器思想

### 使用void*实现泛型

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 泛型动态数组
typedef struct {
    void *data;
    size_t elem_size;
    size_t size;
    size_t capacity;
} GenericVector;

GenericVector *gvec_create(size_t elem_size) {
    GenericVector *v = (GenericVector *)malloc(sizeof(GenericVector));
    v->elem_size = elem_size;
    v->size = 0;
    v->capacity = 4;
    v->data = malloc(elem_size * v->capacity);
    return v;
}

void gvec_push(GenericVector *v, const void *elem) {
    if (v->size >= v->capacity) {
        v->capacity *= 2;
        v->data = realloc(v->data, v->elem_size * v->capacity);
    }
    memcpy((char *)v->data + v->size * v->elem_size, elem, v->elem_size);
    v->size++;
}

void *gvec_get(GenericVector *v, size_t index) {
    if (index >= v->size) return NULL;
    return (char *)v->data + index * v->elem_size;
}

void gvec_free(GenericVector *v) {
    free(v->data);
    free(v);
}

// 类型安全的宏包装
#define DEFINE_VECTOR_TYPE(Type, name) \
    typedef struct { \
        Type *data; \
        size_t size; \
        size_t capacity; \
    } name##Vector; \
    \
    name##Vector *name##_vec_create(void) { \
        name##Vector *v = malloc(sizeof(name##Vector)); \
        v->size = 0; \
        v->capacity = 4; \
        v->data = malloc(sizeof(Type) * v->capacity); \
        return v; \
    } \
    \
    void name##_vec_push(name##Vector *v, Type elem) { \
        if (v->size >= v->capacity) { \
            v->capacity *= 2; \
            v->data = realloc(v->data, sizeof(Type) * v->capacity); \
        } \
        v->data[v->size++] = elem; \
    } \
    \
    Type name##_vec_get(name##Vector *v, size_t idx) { \
        return v->data[idx]; \
    } \
    \
    void name##_vec_free(name##Vector *v) { \
        free(v->data); \
        free(v); \
    }

// 生成int向量
DEFINE_VECTOR_TYPE(int, Int)
// 生成double向量
DEFINE_VECTOR_TYPE(double, Double)

void generic_demo(void) {
    // 使用类型安全的向量
    IntVector *iv = Int_vec_create();
    Int_vec_push(iv, 10);
    Int_vec_push(iv, 20);
    Int_vec_push(iv, 30);

    for (size_t i = 0; i < iv->size; i++) {
        printf("%d ", Int_vec_get(iv, i));
    }
    printf("\n");

    Int_vec_free(iv);

    // 使用void*泛型
    GenericVector *gv = gvec_create(sizeof(double));
    double val = 3.14;
    gvec_push(gv, &val);
    val = 2.71;
    gvec_push(gv, &val);

    double *p = gvec_get(gv, 0);
    printf("gv[0] = %.2f\n", *p);

    gvec_free(gv);
}
```

## 5. C11 _Generic 泛型选择

```c
#include <stdio.h>
#include <math.h>

// 泛型宏：根据类型选择不同实现
#define abs(x) _Generic((x), \
    int: abs, \
    long: labs, \
    long long: llabs, \
    float: fabsf, \
    double: fabs, \
    default: fabs \
)(x)

#define sqrt_generic(x) _Generic((x), \
    float: sqrtf, \
    double: sqrt, \
    long double: sqrtl \
)(x)

// 类型名称
#define type_name(x) _Generic((x), \
    int: "int", \
    long: "long", \
    float: "float", \
    double: "double", \
    char*: "string", \
    default: "unknown" \
)

void generic_demo(void) {
    printf("abs(-5) = %d\n", abs(-5));
    printf("abs(-3.14) = %.2f\n", abs(-3.14));
    printf("sqrt(4.0) = %.2f\n", sqrt_generic(4.0));

    int i = 0;
    double d = 0.0;
    printf("i的类型: %s\n", type_name(i));
    printf("d的类型: %s\n", type_name(d));
}
```

## 6. 编译期计算

```c
// static_assert（C11）
#include <assert.h>

_Static_assert(sizeof(int) == 4, "int必须是4字节");
_Static_assert(sizeof(long long) == 8, "long long必须是8字节");

// 编译期数组大小
#define ARRAY_SIZE(arr) (sizeof(arr) / sizeof((arr)[0]))

// 编译期字符串长度
#define STR_LEN(s) (sizeof(s) - 1)

// 编译期最大值
#define MAX_OF(type) \
    _Generic((type)0, \
        char: CHAR_MAX, \
        short: SHRT_MAX, \
        int: INT_MAX, \
        long: LONG_MAX \
    )
```

## 7. 实用宏工具集

```c
#include <stdio.h>

// 字符串化
#define STRINGIFY(x) #x
#define TOSTRING(x) STRINGIFY(x)

// 拼接
#define CONCAT(a, b) a##b
#define CONCAT3(a, b, c) a##b##c

// 位操作
#define BIT(n) (1U << (n))
#define SET_BIT(reg, bit) ((reg) |= BIT(bit))
#define CLR_BIT(reg, bit) ((reg) &= ~BIT(bit))
#define TGL_BIT(reg, bit) ((reg) ^= BIT(bit))
#define GET_BIT(reg, bit) (((reg) >> (bit)) & 1U)

// 数组工具
#define ARRAY_SIZE(arr) (sizeof(arr) / sizeof((arr)[0]))
#define FOR_EACH(item, arr) \
    for (typeof(*(arr)) *item = (arr); item < (arr) + ARRAY_SIZE(arr); item++)

// 作用域限制（GCC扩展）
#define SCOPED __attribute__((cleanup(cleanup_func)))
static inline void cleanup_func(void *p) {
    void **ptr = (void **)p;
    if (*ptr) {
        free(*ptr);
        *ptr = NULL;
    }
}

void macro_utils_demo(void) {
    printf("文件: %s, 行: %d\n", __FILE__, __LINE__);
    printf("编译时间: %s %s\n", __DATE__, __TIME__);

    int arr[] = {1, 2, 3, 4, 5};
    printf("数组大小: %zu\n", ARRAY_SIZE(arr));

    // GCC清理属性
    int *ptr SCOPED = malloc(sizeof(int) * 100);
    // ptr会在作用域结束时自动释放
}
```

## 要点总结

1. 宏是C语言元编程的主要工具
2. X-Macro避免重复代码，适合枚举和结构体定义
3. void*实现泛型，但失去类型安全
4. DEFINE_VECTOR_TYPE等宏生成类型安全的泛型代码
5. C11的_Generic提供编译期类型选择
6. _Static_assert实现编译期断言
7. 宏虽强大，但调试困难，应谨慎使用
