# 多线程编程基础

## 1. 线程基本概念

线程是进程内部的执行单元，一个进程可以包含多个线程，它们共享进程的地址空间、文件描述符等资源，但每个线程拥有独立的栈空间和寄存器状态。

### 线程与进程的对比

| 特性 | 进程 | 线程 |
|------|------|------|
| 地址空间 | 独立 | 共享 |
| 创建开销 | 大 | 小 |
| 通信方式 | IPC（管道、消息队列等） | 直接读写共享内存 |
| 切换开销 | 大（需切换页表） | 小（仅切换栈和寄存器） |
| 安全性 | 高（隔离性好） | 低（需同步机制保护） |

## 2. POSIX Threads（pthread）库

POSIX线程标准定义了C语言的线程编程接口，在Linux下通过 `libpthread` 实现，编译时需要加 `-pthread` 选项。

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

// 线程函数：参数和返回值都是 void*
void* thread_func(void* arg) {
    int thread_id = *(int*)arg;
    printf("线程 %d 开始执行，TID: %lu\n", thread_id, (unsigned long)pthread_self());

    // 模拟工作
    for (int i = 0; i < 3; i++) {
        printf("线程 %d: 工作中... %d\n", thread_id, i);
        usleep(100000); // 100ms
    }

    printf("线程 %d 执行完毕\n", thread_id);
    return NULL;
}

int main() {
    pthread_t threads[3];
    int thread_ids[3] = {1, 2, 3};

    // 创建3个线程
    for (int i = 0; i < 3; i++) {
        int ret = pthread_create(&threads[i], NULL, thread_func, &thread_ids[i]);
        if (ret != 0) {
            fprintf(stderr, "线程创建失败: %d\n", ret);
            exit(EXIT_FAILURE);
        }
    }

    // 等待所有线程结束
    for (int i = 0; i < 3; i++) {
        pthread_join(threads[i], NULL);
    }

    printf("所有线程执行完毕\n");
    return 0;
}
```

编译命令：`gcc -pthread -o thread_demo thread_demo.c`

## 3. 线程创建与属性

### pthread_create 详解

```c
int pthread_create(
    pthread_t *thread,          // 输出：线程ID
    const pthread_attr_t *attr, // 线程属性，NULL使用默认
    void *(*start_routine)(void*), // 线程入口函数
    void *arg                   // 传递给线程函数的参数
);
// 返回值：成功返回0，失败返回错误码
```

### 线程属性设置

```c
void thread_with_attributes() {
    pthread_t thread;
    pthread_attr_t attr;

    // 初始化属性对象
    pthread_attr_init(&attr);

    // 设置分离状态（detach后无需join）
    pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);

    // 设置栈大小（默认一般为8MB）
    size_t stack_size = 1024 * 1024; // 1MB
    pthread_attr_setstacksize(&attr, stack_size);

    // 设置栈地址和大小
    // void *stack = malloc(stack_size);
    // pthread_attr_setstack(&attr, stack, stack_size);

    int arg = 1;
    pthread_create(&thread, &attr, thread_func, &arg);

    // 销毁属性对象
    pthread_attr_destroy(&attr);

    // 分离线程不需要join
    // 主线程继续做其他工作
    sleep(2);
}
```

## 4. 线程终止与回收

### 线程退出方式

```c
#include <pthread.h>
#include <stdio.h>

// 方式1：线程函数return（推荐）
void* thread_return(void* arg) {
    int *result = malloc(sizeof(int));
    *result = 42;
    return result; // 返回值由pthread_join接收
}

// 方式2：调用pthread_exit
void* thread_exit_demo(void* arg) {
    int *result = malloc(sizeof(int));
    *result = 100;
    pthread_exit(result); // 等同于return
}

// 方式3：被其他线程取消
void* thread_cancel_target(void* arg) {
    // 设置取消点
    while (1) {
        printf("线程运行中...\n");
        sleep(1);
        // pthread_testcancel() 也可显式设置取消点
    }
    return NULL;
}

void thread_lifecycle_demo() {
    pthread_t t1, t2, t3;
    void *retval;

    // 等待线程返回值
    pthread_create(&t1, NULL, thread_return, NULL);
    pthread_join(t1, &retval);
    printf("线程返回值: %d\n", *(int*)retval);
    free(retval);

    // 取消线程
    pthread_create(&t2, NULL, thread_cancel_target, NULL);
    sleep(3);
    pthread_cancel(t2); // 发送取消请求
    pthread_join(t2, NULL);
    printf("线程已被取消\n");

    // 分离线程（自动回收资源）
    pthread_create(&t3, NULL, thread_return, NULL);
    pthread_detach(t3); // 分离后不能再join
}
```

## 5. 线程局部存储（TLS）

```c
#include <pthread.h>
#include <stdio.h>

// 使用pthread_key实现线程局部存储
static pthread_key_t tls_key;
static pthread_once_t tls_once = PTHREAD_ONCE_INIT;

void tls_destructor(void* ptr) {
    free(ptr);
}

void tls_init() {
    pthread_key_create(&tls_key, tls_destructor);
}

// 获取当前线程的局部变量
int* get_thread_local_counter() {
    pthread_once(&tls_once, tls_init);
    int* counter = pthread_getspecific(tls_key);
    if (counter == NULL) {
        counter = malloc(sizeof(int));
        *counter = 0;
        pthread_setspecific(tls_key, counter);
    }
    return counter;
}

void* worker(void* arg) {
    int id = *(int*)arg;
    int *counter = get_thread_local_counter();

    for (int i = 0; i < 5; i++) {
        (*counter)++;
    }

    printf("线程 %d 的计数器: %d\n", id, *counter);
    return NULL;
}
```

## 6. 线程安全函数

线程安全函数可以在多个线程中同时调用而不会产生竞态条件。常见策略：

```c
// 方法1：使用静态互斥锁保护
static pthread_mutex_t lock = PTHREAD_MUTEX_INITIALIZER;

int thread_safe_counter() {
    static int count = 0;
    pthread_mutex_lock(&lock);
    int val = ++count;
    pthread_mutex_unlock(&lock);
    return val;
}

// 方法2：使用可重入版本的函数
// 非线程安全: strtok()  -> 线程安全: strtok_r()
// 非线程安全: ctime()   -> 线程安全: ctime_r()
// 非线程安全: gethostbyname() -> 线程安全: gethostbyname_r()
// 非线程安全: rand()    -> 线程安全: rand_r()

void thread_safe_functions_demo() {
    char str[] = "hello,world,foo,bar";
    char *saveptr;

    // strtok_r 是线程安全的
    char *token = strtok_r(str, ",", &saveptr);
    while (token != NULL) {
        printf("Token: %s\n", token);
        token = strtok_r(NULL, ",", &saveptr);
    }

    // rand_r 是线程安全的
    unsigned int seed = (unsigned int)time(NULL);
    int r = rand_r(&seed);
    printf("随机数: %d\n", r);
}
```

## 7. 综合示例：线程池基础

```c
#include <pthread.h>
#include <stdlib.h>
#include <stdio.h>

#define THREAD_POOL_SIZE 4
#define TASK_QUEUE_SIZE 64

typedef struct {
    void (*function)(void*);
    void *argument;
} Task;

typedef struct {
    Task tasks[TASK_QUEUE_SIZE];
    int front;
    int rear;
    int count;

    pthread_t threads[THREAD_POOL_SIZE];
    pthread_mutex_t mutex;
    pthread_cond_t not_empty;
    pthread_cond_t not_full;
    int shutdown;
} ThreadPool;

void thread_pool_init(ThreadPool *pool) {
    pool->front = 0;
    pool->rear = 0;
    pool->count = 0;
    pool->shutdown = 0;
    pthread_mutex_init(&pool->mutex, NULL);
    pthread_cond_init(&pool->not_empty, NULL);
    pthread_cond_init(&pool->not_full, NULL);
}

void thread_pool_destroy(ThreadPool *pool) {
    pthread_mutex_lock(&pool->mutex);
    pool->shutdown = 1;
    pthread_mutex_unlock(&pool->mutex);

    pthread_cond_broadcast(&pool->not_empty);

    for (int i = 0; i < THREAD_POOL_SIZE; i++) {
        pthread_join(pool->threads[i], NULL);
    }

    pthread_mutex_destroy(&pool->mutex);
    pthread_cond_destroy(&pool->not_empty);
    pthread_cond_destroy(&pool->not_full);
}
```

## 重点与注意事项

1. **编译选项**：使用 `-pthread` 编译链接选项
2. **线程安全**：多线程共享数据必须加锁保护
3. **资源回收**：每个线程必须 `join` 或 `detach`，否则会泄漏资源
4. **错误检查**：所有pthread函数返回0表示成功，应检查返回值
5. **避免死锁**：锁的获取顺序要一致，使用 `pthread_mutex_timedlock` 超时机制
6. **信号处理**：多线程程序中信号处理比较复杂，建议在专门线程中处理
7. **栈溢出**：注意线程栈大小限制，大数组建议用 `malloc` 分配
