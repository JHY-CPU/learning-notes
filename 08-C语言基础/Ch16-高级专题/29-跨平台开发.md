# 29 - 跨平台开发

## 概述

C语言本身是跨平台的，但不同操作系统的系统调用和API差异很大。本节介绍条件编译、平台抽象层和CMake跨平台构建。

---

## 1. 平台检测宏

```c
// 操作系统检测
#if defined(_WIN32) || defined(_WIN64)
    #define PLATFORM_WINDOWS
    #define PLATFORM_NAME "Windows"
#elif defined(__linux__)
    #define PLATFORM_LINUX
    #define PLATFORM_NAME "Linux"
#elif defined(__APPLE__) && defined(__MACH__)
    #define PLATFORM_MACOS
    #define PLATFORM_NAME "macOS"
#elif defined(__unix__)
    #define PLATFORM_UNIX
    #define PLATFORM_NAME "Unix"
#else
    #define PLATFORM_UNKNOWN
    #define PLATFORM_NAME "Unknown"
#endif

// 编译器检测
#if defined(__GNUC__)
    #define COMPILER_GCC
#elif defined(_MSC_VER)
    #define COMPILER_MSVC
#elif defined(__clang__)
    #define COMPILER_CLANG
#endif

// 架构检测
#if defined(__x86_64__) || defined(_M_X64)
    #define ARCH_X64
#elif defined(__i386__) || defined(_M_IX86)
    #define ARCH_X86
#elif defined(__aarch64__) || defined(_M_ARM64)
    #define ARCH_ARM64
#elif defined(__arm__) || defined(_M_ARM)
    #define ARCH_ARM
#endif

// 字节序检测
#if defined(__BYTE_ORDER__) && __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
    #define BIG_ENDIAN
#else
    #define LITTLE_ENDIAN
#endif
```

## 2. 平台抽象层

### 文件系统

```c
#include <stdio.h>
#include <string.h>

#ifdef PLATFORM_WINDOWS
    #include <windows.h>
    #include <direct.h>
    #define PATH_SEP '\\'
    #define mkdir_p(path) _mkdir(path)
    #define getcwd_p(buf, size) _getcwd(buf, size)
#else
    #include <dirent.h>
    #include <sys/stat.h>
    #include <unistd.h>
    #define PATH_SEP '/'
    #define mkdir_p(path) mkdir(path, 0755)
    #define getcwd_p(buf, size) getcwd(buf, size)
#endif

// 跨平台路径拼接
char *path_join(char *dest, size_t size, const char *dir, const char *file) {
    snprintf(dest, size, "%s%c%s", dir, PATH_SEP, file);
    return dest;
}

// 跨平台文件存在检查
int file_exists(const char *path) {
#ifdef PLATFORM_WINDOWS
    DWORD attrs = GetFileAttributesA(path);
    return (attrs != INVALID_FILE_ATTRIBUTES);
#else
    struct stat st;
    return (stat(path, &st) == 0);
#endif
}

// 跨平台目录遍历
typedef struct {
    char name[256];
    int is_dir;
} DirEntry;

int list_dir(const char *path, DirEntry *entries, int max_entries) {
    int count = 0;

#ifdef PLATFORM_WINDOWS
    WIN32_FIND_DATAA fd;
    char pattern[MAX_PATH];
    snprintf(pattern, sizeof(pattern), "%s\\*", path);

    HANDLE hFind = FindFirstFileA(pattern, &fd);
    if (hFind == INVALID_HANDLE_VALUE) return 0;

    do {
        if (count >= max_entries) break;
        if (strcmp(fd.cFileName, ".") == 0 ||
            strcmp(fd.cFileName, "..") == 0) continue;

        strncpy(entries[count].name, fd.cFileName, sizeof(entries[0].name) - 1);
        entries[count].is_dir = (fd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
        count++;
    } while (FindNextFileA(hFind, &fd));

    FindClose(hFind);
#else
    DIR *dir = opendir(path);
    if (!dir) return 0;

    struct dirent *entry;
    while ((entry = readdir(dir)) && count < max_entries) {
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) continue;

        strncpy(entries[count].name, entry->d_name, sizeof(entries[0].name) - 1);
        entries[count].is_dir = (entry->d_type == DT_DIR);
        count++;
    }
    closedir(dir);
#endif

    return count;
}
```

### 线程抽象

```c
#ifdef PLATFORM_WINDOWS
    #include <windows.h>
    typedef HANDLE Thread;
    typedef CRITICAL_SECTION Mutex;

    #define thread_create(thread, func, arg) \
        *(thread) = CreateThread(NULL, 0, (LPTHREAD_START_ROUTINE)(func), (arg), 0, NULL)
    #define thread_join(thread) WaitForSingleObject(*(thread), INFINITE)
    #define mutex_init(m) InitializeCriticalSection(m)
    #define mutex_lock(m) EnterCriticalSection(m)
    #define mutex_unlock(m) LeaveCriticalSection(m)
    #define mutex_destroy(m) DeleteCriticalSection(m)
#else
    #include <pthread.h>
    typedef pthread_t Thread;
    typedef pthread_mutex_t Mutex;

    #define thread_create(thread, func, arg) \
        pthread_create(thread, NULL, (void*(*)(void*))(func), (void*)(arg))
    #define thread_join(thread) pthread_join(*(thread), NULL)
    #define mutex_init(m) pthread_mutex_init(m, NULL)
    #define mutex_lock(m) pthread_mutex_lock(m)
    #define mutex_unlock(m) pthread_mutex_unlock(m)
    #define mutex_destroy(m) pthread_mutex_destroy(m)
#endif
```

### 动态库加载

```c
#ifdef PLATFORM_WINDOWS
    #include <windows.h>
    typedef HMODULE LibHandle;
    #define lib_open(path) LoadLibraryA(path)
    #define lib_sym(handle, name) (void *)GetProcAddress(handle, name)
    #define lib_close(handle) FreeLibrary(handle)
#else
    #include <dlfcn.h>
    typedef void *LibHandle;
    #define lib_open(path) dlopen(path, RTLD_LAZY)
    #define lib_sym(handle, name) dlsym(handle, name)
    #define lib_close(handle) dlclose(handle)
#endif
```

## 4. CMake跨平台构建

### CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.10)
project(MyProject C)

set(CMAKE_C_STANDARD 11)

# 平台检测
if(WIN32)
    add_definitions(-DPLATFORM_WINDOWS)
    set(PLATFORM_LIBS ws2_32)  # Winsock
elseif(UNIX AND NOT APPLE)
    add_definitions(-DPLATFORM_LINUX)
    set(PLATFORM_LIBS pthread dl m)
elseif(APPLE)
    add_definitions(-DPLATFORM_MACOS)
    set(PLATFORM_LIBS pthread dl m)
endif()

# 源文件
set(SOURCES
    src/main.c
    src/platform.c
    src/utils.c
)

# 头文件目录
include_directories(include)

# 编译选项
if(CMAKE_COMPILER_IS_GNUCC)
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -Wextra")
    set(CMAKE_C_FLAGS_DEBUG "-g -O0 -DDEBUG")
    set(CMAKE_C_FLAGS_RELEASE "-O2 -DNDEBUG")
endif()

# 生成可执行文件
add_executable(${PROJECT_NAME} ${SOURCES})

# 链接库
target_link_libraries(${PROJECT_NAME} ${PLATFORM_LIBS})

# 安装规则
install(TARGETS ${PROJECT_NAME} DESTINATION bin)
```

### 构建命令

```bash
# Linux/macOS
mkdir build && cd build
cmake ..
make

# Windows (Visual Studio)
mkdir build && cd build
cmake .. -G "Visual Studio 17 2022"
cmake --build . --config Release

# Windows (MinGW)
mkdir build && cd build
cmake .. -G "MinGW Makefiles"
mingw32-make
```

## 5. 跨平台编译脚本

```bash
#!/bin/bash
# build.sh - 跨平台构建脚本

set -e

BUILD_TYPE="${1:-Release}"
BUILD_DIR="build"

echo "构建类型: $BUILD_TYPE"
echo "构建目录: $BUILD_DIR"

mkdir -p "$BUILD_DIR"
cd "$BUILD_DIR"

# 检测平台并选择生成器
if command -v cmake &> /dev/null; then
    cmake .. -DCMAKE_BUILD_TYPE="$BUILD_TYPE"
    cmake --build . -- -j$(nproc 2>/dev/null || sysctl -n hw.ncpu 2>/dev/null || echo 4)
else
    echo "错误: 未安装cmake"
    exit 1
fi

echo "构建完成!"
```

## 6. 平台差异总结

| 特性 | Linux | Windows | macOS |
|------|-------|---------|-------|
| 共享库 | .so | .dll | .dylib |
| 线程 | pthread | Win32 Thread | pthread |
| 动态加载 | dlopen | LoadLibrary | dlopen |
| 路径分隔符 | / | \ | / |
| 换行符 | \n | \r\n | \n |
| 文件描述符 | int | HANDLE | int |
| socket | int | SOCKET | int |

## 要点总结

1. 使用条件编译处理平台差异
2. 抽象平台相关代码到独立模块
3. CMake是最常用的跨平台构建系统
4. 注意字节序、对齐、换行符等平台差异
5. 测试应在所有目标平台上进行
6. 使用CI/CD自动构建和测试多平台版本
