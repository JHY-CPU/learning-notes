# 28 - 日志与调试系统

## 概述

良好的日志和调试系统是大型C项目的重要组成部分。本节介绍分级日志、调试宏和日志系统设计。

---

## 1. 分级日志系统

```c
#ifndef LOGGER_H
#define LOGGER_H

#include <stdio.h>
#include <stdarg.h>
#include <time.h>
#include <string.h>

// 日志级别
typedef enum {
    LOG_TRACE = 0,
    LOG_DEBUG,
    LOG_INFO,
    LOG_WARN,
    LOG_ERROR,
    LOG_FATAL,
    LOG_OFF
} LogLevel;

// 日志级别名称
static const char *level_names[] = {
    "TRACE", "DEBUG", "INFO", "WARN", "ERROR", "FATAL"
};

// 颜色（终端）
static const char *level_colors[] = {
    "\033[90m",  // TRACE - 灰色
    "\033[36m",  // DEBUG - 青色
    "\033[32m",  // INFO - 绿色
    "\033[33m",  // WARN - 黄色
    "\033[31m",  // ERROR - 红色
    "\033[35m"   // FATAL - 紫色
};

static LogLevel current_level = LOG_INFO;
static FILE *log_file = NULL;
static int use_color = 1;

// 设置日志级别
static inline void log_set_level(LogLevel level) {
    current_level = level;
}

// 设置日志文件
static inline int log_set_file(const char *filename) {
    log_file = fopen(filename, "a");
    return log_file != NULL ? 0 : -1;
}

// 关闭日志文件
static inline void log_close_file(void) {
    if (log_file) {
        fclose(log_file);
        log_file = NULL;
    }
}

// 核心日志函数
static inline void log_log(LogLevel level, const char *file,
                           int line, const char *fmt, ...) {
    if (level < current_level) return;

    // 获取时间
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    char time_buf[20];
    strftime(time_buf, sizeof(time_buf), "%Y-%m-%d %H:%M:%S", tm_info);

    // 提取文件名
    const char *basename = strrchr(file, '/');
    basename = basename ? basename + 1 : file;

    // 输出到终端
    if (use_color) {
        fprintf(stderr, "%s[%-5s]%s %s %s:%d: ",
                level_colors[level], level_names[level], "\033[0m",
                time_buf, basename, line);
    } else {
        fprintf(stderr, "[%-5s] %s %s:%d: ",
                level_names[level], time_buf, basename, line);
    }

    va_list args;
    va_start(args, fmt);
    vfprintf(stderr, fmt, args);
    va_end(args);
    fprintf(stderr, "\n");

    // 输出到文件
    if (log_file) {
        fprintf(log_file, "[%-5s] %s %s:%d: ",
                level_names[level], time_buf, basename, line);
        va_start(args, fmt);
        vfprintf(log_file, fmt, args);
        va_end(args);
        fprintf(log_file, "\n");
        fflush(log_file);
    }
}

// 便捷宏
#define log_trace(...) log_log(LOG_TRACE, __FILE__, __LINE__, __VA_ARGS__)
#define log_debug(...) log_log(LOG_DEBUG, __FILE__, __LINE__, __VA_ARGS__)
#define log_info(...)  log_log(LOG_INFO,  __FILE__, __LINE__, __VA_ARGS__)
#define log_warn(...)  log_log(LOG_WARN,  __FILE__, __LINE__, __VA_ARGS__)
#define log_error(...) log_log(LOG_ERROR, __FILE__, __LINE__, __VA_ARGS__)
#define log_fatal(...) log_log(LOG_FATAL, __FILE__, __LINE__, __VA_ARGS__)

#endif
```

## 2. 调试宏

```c
#ifndef DEBUG_H
#define DEBUG_H

#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

// 调试模式开关
#ifdef DEBUG
#define DBG(fmt, ...) \
    fprintf(stderr, "[DEBUG] %s:%d: " fmt "\n", \
            __FILE__, __LINE__, ##__VA_ARGS__)
#else
#define DBG(fmt, ...) ((void)0)
#endif

// 断言宏（带消息）
#define ASSERT_MSG(cond, msg, ...) \
    do { \
        if (!(cond)) { \
            fprintf(stderr, "断言失败: %s (%s:%d)\n  消息: ", \
                    #cond, __FILE__, __LINE__); \
            fprintf(stderr, msg, ##__VA_ARGS__); \
            fprintf(stderr, "\n"); \
            abort(); \
        } \
    } while(0)

// 条件断言
#define ASSERT_EQ(a, b) \
    ASSERT_MSG((a) == (b), "期望 %d, 实际 %d", (int)(a), (int)(b))

#define ASSERT_NE(a, b) \
    ASSERT_MSG((a) != (b), "期望不相等，但都等于 %d", (int)(a))

#define ASSERT_GT(a, b) \
    ASSERT_MSG((a) > (b), "期望 %d > %d", (int)(a), (int)(b))

#define ASSERT_NOT_NULL(ptr) \
    ASSERT_MSG((ptr) != NULL, "指针不能为NULL")

// 打印变量值
#define PRINT_INT(x)    printf(#x " = %d\n", (int)(x))
#define PRINT_LONG(x)   printf(#x " = %ld\n", (long)(x))
#define PRINT_DOUBLE(x) printf(#x " = %f\n", (double)(x))
#define PRINT_STR(x)    printf(#x " = \"%s\"\n", (const char *)(x))
#define PRINT_PTR(x)    printf(#x " = %p\n", (const void *)(x))

// 打印数组
#define PRINT_ARRAY(arr, n, fmt) \
    do { \
        printf(#arr " = ["); \
        for (int _i = 0; _i < (n); _i++) { \
            if (_i > 0) printf(", "); \
            printf(fmt, (arr)[_i]); \
        } \
        printf("]\n"); \
    } while(0)

// 十六进制转储
#define HEX_DUMP(data, len) \
    do { \
        const unsigned char *_d = (const unsigned char *)(data); \
        int _l = (int)(len); \
        printf(#data " (%d bytes):\n", _l); \
        for (int _i = 0; _i < _l; _i++) { \
            if (_i % 16 == 0) printf("  %04x: ", _i); \
            printf("%02x ", _d[_i]); \
            if (_i % 16 == 15 || _i == _l - 1) { \
                int _pad = 15 - (_i % 16); \
                for (int _j = 0; _j < _pad; _j++) printf("   "); \
                printf(" |"); \
                int _start = _i - (_i % 16); \
                for (int _j = _start; _j <= _i; _j++) { \
                    printf("%c", _d[_j] >= 32 && _d[_j] < 127 ? _d[_j] : '.'); \
                } \
                printf("|\n"); \
            } \
        } \
    } while(0)

// 进入/退出函数追踪
#ifdef DEBUG
#define FUNC_ENTER() printf(">>> 进入 %s\n", __func__)
#define FUNC_EXIT()  printf("<<< 退出 %s\n", __func__)
#else
#define FUNC_ENTER() ((void)0)
#define FUNC_EXIT()  ((void)0)
#endif

#endif
```

## 3. 使用示例

```c
#include <stdio.h>
#include <string.h>

// 包含上述头文件...

void debug_demo(void) {
    // 日志
    log_set_level(LOG_DEBUG);
    log_info("程序启动");
    log_debug("调试信息: %d", 42);
    log_warn("警告信息");
    log_error("错误: %s", "文件未找到");

    // 调试宏
    DBG("变量值: x=%d", 42);

    // 打印变量
    int x = 10;
    double pi = 3.14159;
    PRINT_INT(x);
    PRINT_DOUBLE(pi);

    // 打印数组
    int arr[] = {1, 2, 3, 4, 5};
    PRINT_ARRAY(arr, 5, "%d");

    // 十六进制转储
    const char *data = "Hello, World! 你好世界";
    HEX_DUMP(data, strlen(data));

    // 函数追踪
    FUNC_ENTER();
    // ... 函数逻辑 ...
    FUNC_EXIT();

    // 写入日志文件
    log_set_file("app.log");
    log_info("这条会写入文件");
    log_close_file();
}
```

## 4. 内存调试

```c
#include <stdlib.h>
#include <stdio.h>

// 内存调试包装
#ifdef DEBUG_MALLOC

static size_t total_allocated = 0;
static size_t total_freed = 0;
static int alloc_count = 0;
static int free_count = 0;

void *debug_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    if (ptr) {
        total_allocated += size;
        alloc_count++;
        DBG("malloc(%zu) = %p [%s:%d]", size, ptr, file, line);
    }
    return ptr;
}

void debug_free(void *ptr, const char *file, int line) {
    if (ptr) {
        free_count++;
        DBG("free(%p) [%s:%d]", ptr, file, line);
    }
    free(ptr);
}

#define malloc(size) debug_malloc(size, __FILE__, __LINE__)
#define free(ptr) debug_free(ptr, __FILE__, __LINE__)

void memory_report(void) {
    printf("\n=== 内存报告 ===\n");
    printf("分配: %zu 字节 (%d 次)\n", total_allocated, alloc_count);
    printf("释放: %d 次\n", free_count);
    printf("未释放: %d 次\n", alloc_count - free_count);
}
#endif
```

## 5. 性能计时

```c
#include <time.h>

// 简易性能计时器
#define TIMER_START(name) \
    struct timespec _timer_start_##name, _timer_end_##name; \
    clock_gettime(CLOCK_MONOTONIC, &_timer_start_##name)

#define TIMER_END(name) \
    do { \
        clock_gettime(CLOCK_MONOTONIC, &_timer_end_##name); \
        double _elapsed = \
            (_timer_end_##name.tv_sec - _timer_start_##name.tv_sec) + \
            (_timer_end_##name.tv_nsec - _timer_start_##name.tv_nsec) / 1e9; \
        printf("[TIMER] %s: %.6f 秒\n", #name, _elapsed); \
    } while(0)

// 使用
void timer_demo(void) {
    TIMER_START(loop);
    volatile int sum = 0;
    for (int i = 0; i < 1000000; i++) sum += i;
    TIMER_END(loop);
}
```

## 要点总结

1. 分级日志系统便于控制输出粒度
2. 使用`__FILE__`和`__LINE__`自动定位代码位置
3. 调试宏在Release模式下应编译为空
4. HEX_DUMP对调试二进制协议非常有用
5. 内存调试包装可以追踪分配/释放不匹配
6. 性能计时宏帮助定位瓶颈
