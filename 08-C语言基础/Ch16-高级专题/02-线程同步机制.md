# 线程同步机制

## 1. 为什么需要同步

多线程并发访问共享资源时，如果没有适当的同步保护，会产生**竞态条件（Race Condition）**，导致数据不一致、程序崩溃等不可预期的行为。

```c
// 经典竞态条件示例：两个线程同时对全局变量++操作
#include <stdio.h>
#include <pthread.h>

int counter = 0; // 共享变量

void* increment(void* arg) {
    for (int i = 0; i < 100000; i++) {
        counter++; // 非原子操作：读-改-写
    }
    return NULL;
}

int main() {
    pthread_t t1, t2;
    pthread_create(&t1, NULL, increment, NULL);
    pthread_create(&t2, NULL, increment, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    // 预期值: 200000，实际可能小于200000！
    printf("counter = %d (期望 200000)\n", counter);
    return 0;
}
```

## 2. 互斥锁（Mutex）

互斥锁是最基本的同步原语，保证同一时刻只有一个线程持有锁。

### 基本用法

```c
#include <pthread.h>
#include <stdio.h>

int shared_counter = 0;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

void* safe_increment(void* arg) {
    for (int i = 0; i < 100000; i++) {
        pthread_mutex_lock(&mutex);
        shared_counter++;
        pthread_mutex_unlock(&mutex);
    }
    return NULL;
}

void mutex_basic_demo() {
    pthread_t t1, t2;
    pthread_create(&t1, NULL, safe_increment, NULL);
    pthread_create(&t2, NULL, safe_increment, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    printf("counter = %d (保证是200000)\n", shared_counter);
    pthread_mutex_destroy(&mutex);
}
```

### 动态初始化与销毁

```c
void mutex_dynamic_demo() {
    pthread_mutex_t mutex;
    pthread_mutexattr_t attr;

    // 设置互斥锁属性
    pthread_mutexattr_init(&attr);

    // 互斥锁类型
    // PTHREAD_MUTEX_NORMAL     - 不检测死锁（最快，但危险）
    // PTHREAD_MUTEX_ERRORCHECK - 错误检查（调试用）
    // PTHREAD_MUTEX_RECURSIVE  - 递归锁（同一线程可多次加锁）
    // PTHREAD_MUTEX_DEFAULT    - 默认（通常等同NORMAL）
    pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);

    pthread_mutex_init(&mutex, &attr);

    // 使用互斥锁...
    pthread_mutex_lock(&mutex);
    pthread_mutex_lock(&mutex); // 递归锁允许重入
    printf("递归锁已获取两次\n");
    pthread_mutex_unlock(&mutex);
    pthread_mutex_unlock(&mutex);

    pthread_mutex_destroy(&mutex);
    pthread_mutexattr_destroy(&attr);
}
```

### trylock 与 timedlock

```c
#include <time.h>

void mutex_trylock_demo() {
    pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

    // 非阻塞尝试加锁
    int ret = pthread_mutex_trylock(&mutex);
    if (ret == 0) {
        printf("成功获取锁\n");
        pthread_mutex_unlock(&mutex);
    } else if (ret == EBUSY) {
        printf("锁已被占用\n");
    }

    // 带超时的加锁
    struct timespec ts;
    clock_gettime(CLOCK_REALTIME, &ts);
    ts.tv_sec += 5; // 5秒超时

    ret = pthread_mutex_timedlock(&mutex, &ts);
    if (ret == ETIMEDOUT) {
        printf("加锁超时\n");
    }

    pthread_mutex_destroy(&mutex);
}
```

## 3. 条件变量（Condition Variable）

条件变量用于线程间的等待/通知机制，通常与互斥锁配合使用。

### 生产者-消费者模型

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

#define BUFFER_SIZE 10

int buffer[BUFFER_SIZE];
int count = 0; // 缓冲区中元素数量

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t cond_not_full = PTHREAD_COND_INITIALIZER;
pthread_cond_t cond_not_empty = PTHREAD_COND_INITIALIZER;

void* producer(void* arg) {
    int id = *(int*)arg;
    for (int i = 0; i < 20; i++) {
        pthread_mutex_lock(&mutex);

        // 等待缓冲区不满（使用while防止虚假唤醒）
        while (count == BUFFER_SIZE) {
            printf("生产者 %d: 缓冲区满，等待...\n", id);
            pthread_cond_wait(&cond_not_full, &mutex);
        }

        // 生产数据
        buffer[count++] = i;
        printf("生产者 %d: 生产了 %d，缓冲区大小: %d\n", id, i, count);

        // 通知消费者
        pthread_cond_signal(&cond_not_empty);
        pthread_mutex_unlock(&mutex);

        usleep(rand() % 100000);
    }
    return NULL;
}

void* consumer(void* arg) {
    int id = *(int*)arg;
    for (int i = 0; i < 20; i++) {
        pthread_mutex_lock(&mutex);

        // 等待缓冲区非空
        while (count == 0) {
            printf("消费者 %d: 缓冲区空，等待...\n", id);
            pthread_cond_wait(&cond_not_empty, &mutex);
        }

        // 消费数据
        int item = buffer[--count];
        printf("消费者 %d: 消费了 %d，缓冲区大小: %d\n", id, item, count);

        // 通知生产者
        pthread_cond_signal(&cond_not_full);
        pthread_mutex_unlock(&mutex);

        usleep(rand() % 150000);
    }
    return NULL;
}

void producer_consumer_demo() {
    pthread_t prod, cons;
    int id1 = 1, id2 = 1;

    pthread_create(&prod, NULL, producer, &id1);
    pthread_create(&cons, NULL, consumer, &id2);

    pthread_join(prod, NULL);
    pthread_join(cons, NULL);

    pthread_mutex_destroy(&mutex);
    pthread_cond_destroy(&cond_not_full);
    pthread_cond_destroy(&cond_not_empty);
}
```

### 广播通知

```c
pthread_cond_broadcast(&cond_not_empty); // 唤醒所有等待线程
// vs
pthread_cond_signal(&cond_not_empty);    // 只唤醒一个等待线程
```

## 4. 读写锁（Read-Write Lock）

读写锁适用于**读多写少**的场景：多个读者可以同时读取，但写者独占访问。

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

pthread_rwlock_t rwlock = PTHREAD_RWLOCK_INITIALIZER;
int shared_data = 0;

void* reader(void* arg) {
    int id = *(int*)arg;
    for (int i = 0; i < 5; i++) {
        pthread_rwlock_rdlock(&rwlock); // 获取读锁
        printf("读者 %d: 读取数据 = %d\n", id, shared_data);
        usleep(100000);
        pthread_rwlock_unlock(&rwlock);
        usleep(50000);
    }
    return NULL;
}

void* writer(void* arg) {
    int id = *(int*)arg;
    for (int i = 0; i < 3; i++) {
        pthread_rwlock_wrlock(&rwlock); // 获取写锁
        shared_data += 10;
        printf("写者 %d: 写入数据 = %d\n", id, shared_data);
        usleep(200000);
        pthread_rwlock_unlock(&rwlock);
        usleep(100000);
    }
    return NULL;
}

void rwlock_demo() {
    pthread_t readers[3], writers[1];
    int ids[] = {1, 2, 3, 1};

    for (int i = 0; i < 3; i++)
        pthread_create(&readers[i], NULL, reader, &ids[i]);
    pthread_create(&writers[0], NULL, writer, &ids[3]);

    for (int i = 0; i < 3; i++)
        pthread_join(readers[i], NULL);
    pthread_join(writers[0], NULL);

    pthread_rwlock_destroy(&rwlock);
}
```

## 5. 自旋锁（Spinlock）

自旋锁在等待时不阻塞线程，而是循环检查锁状态。适用于**锁持有时间极短**的场景。

```c
#include <pthread.h>

void spinlock_demo() {
    pthread_spinlock_t spinlock;

    // 参数PTHREAD_PROCESS_PRIVATE表示仅进程内使用
    pthread_spin_init(&spinlock, PTHREAD_PROCESS_PRIVATE);

    pthread_spin_lock(&spinlock);
    // 临界区操作（应非常简短）
    pthread_spin_unlock(&spinlock);

    pthread_spin_destroy(&spinlock);
}
```

## 6. 信号量（Semaphore）

信号量是更通用的同步原语，可以控制同时访问资源的线程数量。

```c
#include <semaphore.h>
#include <pthread.h>
#include <stdio.h>

// 信号量实现限流（最多3个线程同时访问）
sem_t semaphore;
int active_count = 0;

void* limited_access(void* arg) {
    int id = *(int*)arg;

    sem_wait(&semaphore); // P操作，信号量-1

    active_count++;
    printf("线程 %d 进入临界区，当前活跃: %d\n", id, active_count);

    // 模拟访问
    usleep(500000);

    active_count--;
    printf("线程 %d 离开临界区，当前活跃: %d\n", id, active_count);

    sem_post(&semaphore); // V操作，信号量+1
    return NULL;
}

void semaphore_demo() {
    pthread_t threads[10];
    int ids[10];

    // 初始值3，最多3个线程同时访问
    sem_init(&semaphore, 0, 3);

    for (int i = 0; i < 10; i++) {
        ids[i] = i;
        pthread_create(&threads[i], NULL, limited_access, &ids[i]);
    }

    for (int i = 0; i < 10; i++) {
        pthread_join(threads[i], NULL);
    }

    sem_destroy(&semaphore);
}
```

## 7. 屏障（Barrier）

屏障用于确保一组线程全部到达某个同步点后，才继续执行。

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

#define NUM_THREADS 3
pthread_barrier_t barrier;

void* barrier_worker(void* arg) {
    int id = *(int*)arg;

    printf("线程 %d: 阶段1工作中...\n", id);
    usleep((id + 1) * 100000);

    printf("线程 %d: 到达屏障，等待其他线程\n", id);
    int ret = pthread_barrier_wait(&barrier);

    if (ret == PTHREAD_BARRIER_SERIAL_THREAD) {
        printf("线程 %d: 最后一个到达的线程（特殊返回值）\n", id);
    }

    printf("线程 %d: 所有线程已同步，继续阶段2\n", id);
    return NULL;
}

void barrier_demo() {
    pthread_t threads[NUM_THREADS];
    int ids[NUM_THREADS];

    pthread_barrier_init(&barrier, NULL, NUM_THREADS);

    for (int i = 0; i < NUM_THREADS; i++) {
        ids[i] = i;
        pthread_create(&threads[i], NULL, barrier_worker, &ids[i]);
    }

    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    pthread_barrier_destroy(&barrier);
}
```

## 8. 死锁（Deadlock）

死锁是指两个或多个线程互相等待对方持有的锁，导致所有线程都无法继续执行。

### 死锁产生的四个必要条件

1. **互斥**：资源一次只能被一个线程持有
2. **持有并等待**：持有一个锁的同时等待另一个锁
3. **不可剥夺**：锁只能由持有者释放
4. **循环等待**：线程之间形成环形等待链

### 死锁示例与避免

```c
pthread_mutex_t lock_a = PTHREAD_MUTEX_INITIALIZER;
pthread_mutex_t lock_b = PTHREAD_MUTEX_INITIALIZER;

// 死锁场景
void* thread1_deadlock(void* arg) {
    pthread_mutex_lock(&lock_a);
    sleep(1);
    pthread_mutex_lock(&lock_b); // 等待thread2持有的lock_b
    // thread2同时在等lock_a -> 死锁！
    pthread_mutex_unlock(&lock_b);
    pthread_mutex_unlock(&lock_a);
    return NULL;
}

// 解决方案1：固定加锁顺序
void* thread1_ordered(void* arg) {
    pthread_mutex_lock(&lock_a); // 始终先锁a
    pthread_mutex_lock(&lock_b);
    // 安全操作
    pthread_mutex_unlock(&lock_b);
    pthread_mutex_unlock(&lock_a);
    return NULL;
}

// 解决方案2：trylock + 回退
void* thread1_trylock(void* arg) {
    while (1) {
        pthread_mutex_lock(&lock_a);
        if (pthread_mutex_trylock(&lock_b) == 0) {
            // 成功获取两个锁
            // 执行操作
            pthread_mutex_unlock(&lock_b);
            pthread_mutex_unlock(&lock_a);
            break;
        }
        pthread_mutex_unlock(&lock_a);
        usleep(rand() % 1000); // 随机退避
    }
    return NULL;
}

// 解决方案3：使用pthread_mutex_timedlock超时
void* thread1_timed(void* arg) {
    struct timespec ts;
    pthread_mutex_lock(&lock_a);

    clock_gettime(CLOCK_REALTIME, &ts);
    ts.tv_sec += 2; // 2秒超时

    if (pthread_mutex_timedlock(&lock_b, &ts) == ETIMEDOUT) {
        printf("获取锁超时，避免死锁\n");
        pthread_mutex_unlock(&lock_a);
        return NULL;
    }

    // 安全操作
    pthread_mutex_unlock(&lock_b);
    pthread_mutex_unlock(&lock_a);
    return NULL;
}
```

## 9. 同步原语对比

| 原语 | 用途 | 特点 |
|------|------|------|
| 互斥锁 | 保护临界区 | 最基本，同一时刻一个线程 |
| 条件变量 | 等待/通知机制 | 需配合mutex使用 |
| 读写锁 | 读多写少场景 | 允许并发读 |
| 自旋锁 | 短临界区 | 忙等，不睡眠 |
| 信号量 | 计数控制 | 可控制N个线程同时访问 |
| 屏障 | 线程同步点 | 等待所有线程到达 |

## 重点与注意事项

1. **必须使用 `while` 检查条件**：使用条件变量时，必须用 `while` 而非 `if` 检查条件，防止虚假唤醒
2. **加锁顺序**：获取多把锁时，始终按相同顺序获取，避免死锁
3. **锁粒度**：锁的粒度要适中，太粗影响并发，太细容易出错
4. **不要在持有锁时做耗时操作**：特别是IO操作，会严重影响性能
5. **RAII风格**：在C中可以用goto cleanup或goto语句确保锁的释放
