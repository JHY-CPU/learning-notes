# 33 - 代码重构技巧

## 概述

代码重构是在不改变外部行为的前提下改善代码内部结构的过程。本节介绍C语言中常见的代码坏味道和重构方法。

---

## 1. 识别代码坏味道

### 1.1 重复代码

```c
// 坏味道：多处重复的代码
void process_user_data(int *data, int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        if (data[i] < 0) {
            fprintf(stderr, "负数: %d\n", data[i]);
            continue;
        }
        sum += data[i];
    }
    printf("用户数据总和: %d\n", sum);
}

void process_order_data(int *data, int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        if (data[i] < 0) {
            fprintf(stderr, "负数: %d\n", data[i]);
            continue;
        }
        sum += data[i];
    }
    printf("订单数据总和: %d\n", sum);  // 只有这行不同
}

// 重构：提取公共函数
int sum_positive(const int *data, int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        if (data[i] < 0) {
            fprintf(stderr, "负数: %d\n", data[i]);
            continue;
        }
        sum += data[i];
    }
    return sum;
}

void process_data(const char *label, int *data, int n) {
    printf("%s总和: %d\n", label, sum_positive(data, n));
}
```

### 1.2 过长函数

```c
// 坏味道：一个函数做太多事
void process_all(void) {
    // 1. 读取配置
    FILE *f = fopen("config.txt", "r");
    char line[256];
    int timeout = 30;
    while (fgets(line, sizeof(line), f)) {
        if (strncmp(line, "timeout=", 8) == 0)
            timeout = atoi(line + 8);
    }
    fclose(f);

    // 2. 连接数据库
    sqlite3 *db;
    sqlite3_open("data.db", &db);

    // 3. 查询数据
    sqlite3_stmt *stmt;
    sqlite3_prepare_v2(db, "SELECT * FROM users", -1, &stmt, NULL);
    // ... 处理查询 ...

    // 4. 发送邮件
    // ... 50行邮件发送代码 ...

    // 5. 生成报告
    // ... 30行报告生成代码 ...

    sqlite3_finalize(stmt);
    sqlite3_close(db);
}

// 重构：拆分为小函数
typedef struct {
    int timeout;
} Config;

Config load_config(const char *path);
void process_users(sqlite3 *db);
void send_report_email(const Report *report);
void generate_report(sqlite3 *db, Report *report);

void process_all_refactored(void) {
    Config cfg = load_config("config.txt");

    sqlite3 *db;
    sqlite3_open("data.db", &db);

    process_users(db);

    Report report;
    generate_report(db, &report);
    send_report_email(&report);

    sqlite3_close(db);
}
```

## 2. 常见重构手法

### 2.1 提取函数

```c
// 提取前
void print_report(const Report *r) {
    printf("=== 报告 ===\n");
    printf("日期: %s\n", r->date);
    printf("总计: %d\n", r->total);
    for (int i = 0; i < r->item_count; i++) {
        printf("  %s: %d (%.2f)\n",
               r->items[i].name,
               r->items[i].qty,
               r->items[i].price);
    }
    printf("============\n");
}

// 提取后
static void print_header(void) {
    printf("=== 报告 ===\n");
}

static void print_footer(void) {
    printf("============\n");
}

static void print_items(const Report *r) {
    for (int i = 0; i < r->item_count; i++) {
        printf("  %s: %d (%.2f)\n",
               r->items[i].name,
               r->items[i].qty,
               r->items[i].price);
    }
}

void print_report(const Report *r) {
    print_header();
    printf("日期: %s\n", r->date);
    printf("总计: %d\n", r->total);
    print_items(r);
    print_footer();
}
```

### 2.2 用宏消除重复

```c
// 重复的错误检查
int func1(void) {
    int result = do_something();
    if (result < 0) {
        log_error("do_something失败: %d", result);
        return result;
    }
    result = do_another();
    if (result < 0) {
        log_error("do_another失败: %d", result);
        return result;
    }
    return 0;
}

// 用宏重构
#define CHECK(call) do { \
    int _result = (call); \
    if (_result < 0) { \
        log_error(#call "失败: %d", _result); \
        return _result; \
    } \
} while(0)

int func1_refactored(void) {
    CHECK(do_something());
    CHECK(do_another());
    return 0;
}
```

### 2.3 消除魔法数字

```c
// 坏味道
if (status == 3) { /* ... */ }
if (len > 256) { /* ... */ }
sleep(86400);

// 重构
#define STATUS_ACTIVE 3
#define MAX_BUFFER_SIZE 256
#define SECONDS_PER_DAY 86400

if (status == STATUS_ACTIVE) { /* ... */ }
if (len > MAX_BUFFER_SIZE) { /* ... */ }
sleep(SECONDS_PER_DAY);
```

## 3. 改善函数签名

```c
// 改善前：输出参数不直观
int get_user(int id, User *out_user);

// 改善后：返回值更清晰
User *get_user(int id);  // 返回NULL表示失败

// 或者使用错误码+输出参数的明确模式
int get_user(int id, User *out_user);  // 返回0成功，-1失败
```

## 4. 改善内存管理

```c
// 改善前：调用者需要手动管理内存
char *create_string(const char *src);
// 使用后需要调用者free

// 改善后：提供配对的释放函数
typedef struct {
    char *data;
    size_t len;
} String;

String *string_create(const char *src);
void string_free(String *s);

// 或者使用所有权模式
// 创建者负责释放
String *string_create_owned(char *data);  // 拥有data的所有权
```

## 5. 改善错误处理

```c
// 改善前：返回-1表示各种错误
int open_database(const char *path);
// 调用者不知道具体错误原因

// 改善后：返回详细错误码
typedef enum {
    DB_OK = 0,
    DB_ERR_NOT_FOUND,
    DB_ERR_PERMISSION,
    DB_ERR_CORRUPT,
    DB_ERR_NOMEM,
    DB_ERR_IO
} DBError;

DBError open_database(const char *path, Database **out);

// 或者使用结果类型
typedef struct {
    int success;
    union {
        Database *db;
        DBError error;
    };
} DBResult;

DBResult open_database(const char *path);
```

## 7. 重构检查清单

```
重构前检查清单：
□ 是否有重复代码可以提取
□ 函数是否过长（超过50行考虑拆分）
□ 参数是否过多（超过5个考虑结构体）
□ 是否有魔法数字需要命名
□ 错误处理是否一致
□ 内存管理责任是否清晰
□ 模块接口是否最小化
□ 是否有全局变量可以消除

重构后验证：
□ 所有测试通过
□ 没有引入内存泄漏
□ 性能没有显著下降
□ 代码可读性提高
```

---

## 要点总结

1. 重构的前提是良好的测试覆盖
2. 小步重构，每次只做一个改变
3. 消除重复代码是最常见的重构
4. 函数应该短小，做一件事
5. 使用结构体减少参数数量
6. 不透明指针隐藏实现细节
7. 统一的错误处理模式提高代码一致性
