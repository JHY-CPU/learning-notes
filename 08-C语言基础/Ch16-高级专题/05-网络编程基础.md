# 网络编程基础

## 1. 网络编程概述

网络编程使程序能够通过网络与远程计算机通信。在Linux/C中，核心机制是 **socket（套接字）**，它提供了一种统一的抽象接口，使网络通信与文件IO类似。

### 网络模型对照

```
  应用层    (HTTP, FTP, DNS)     <- 应用程序
  传输层    (TCP, UDP)           <- socket类型选择
  网络层    (IP)                 <- 地址族/协议
  链路层    (Ethernet, WiFi)     <- 驱动程序
```

### Socket类型

- **流式套接字（SOCK_STREAM）**：TCP，可靠的、面向连接的、有序
- **数据报套接字（SOCK_DGRAM）**：UDP，无连接的、不可靠但高效
- **原始套接字（SOCK_RAW）**：直接访问底层协议

## 2. 字节序与地址转换

网络字节序统一使用**大端字节序（Big-Endian）**，而x86等平台使用小端字节序，需要转换。

```c
#include <arpa/inet.h>
#include <stdio.h>

void byte_order_demo() {
    uint32_t host_long = 0x12345678;
    uint16_t host_short = 0x1234;

    // 主机字节序 -> 网络字节序
    uint32_t net_long = htonl(host_long);   // host to network long
    uint16_t net_short = htons(host_short); // host to network short

    // 网络字节序 -> 主机字节序
    uint32_t h_long = ntohl(net_long);      // network to host long
    uint16_t h_short = ntohs(net_short);    // network to host short

    printf("主机序: 0x%08X -> 网络序: 0x%08X\n", host_long, net_long);

    // IP地址转换
    // 字符串 -> 二进制
    struct in_addr addr;
    inet_pton(AF_INET, "192.168.1.100", &addr);
    printf("IP (二进制): 0x%08X\n", ntohl(addr.s_addr));

    // 二进制 -> 字符串
    char ip_str[INET_ADDRSTRLEN];
    inet_ntop(AF_INET, &addr, ip_str, INET_ADDRSTRLEN);
    printf("IP (字符串): %s\n", ip_str);

    // 旧版函数（不推荐用于IPv6）
    // inet_addr("192.168.1.100")  -> 返回网络字节序的32位整数
    // inet_ntoa(addr)             -> 返回静态缓冲区的字符串
}
```

## 3. Socket API 核心函数

```c
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <string.h>
#include <stdio.h>
#include <stdlib.h>

/*
 * Socket API 一览：
 *
 * socket()     - 创建套接字
 * bind()       - 绑定地址和端口
 * listen()     - (TCP) 监听连接
 * accept()     - (TCP) 接受连接
 * connect()    - (TCP) 发起连接
 * send/recv    - 发送/接收数据
 * sendto/recvfrom - (UDP) 发送/接收数据（含地址）
 * close()      - 关闭套接字
 * shutdown()   - 半关闭连接
 */
```

## 4. TCP 客户端示例

```c
void tcp_client_demo(const char *server_ip, int port) {
    // 1. 创建socket
    int sockfd = socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd < 0) {
        perror("socket");
        exit(EXIT_FAILURE);
    }

    // 2. 设置服务器地址
    struct sockaddr_in server_addr;
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(port);

    // 将IP字符串转换为二进制
    if (inet_pton(AF_INET, server_ip, &server_addr.sin_addr) <= 0) {
        fprintf(stderr, "无效的IP地址: %s\n", server_ip);
        close(sockfd);
        exit(EXIT_FAILURE);
    }

    // 3. 连接服务器
    if (connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
        perror("connect");
        close(sockfd);
        exit(EXIT_FAILURE);
    }

    printf("已连接到 %s:%d\n", server_ip, port);

    // 4. 发送数据
    const char *message = "Hello, Server!";
    ssize_t sent = send(sockfd, message, strlen(message), 0);
    printf("已发送 %zd 字节\n", sent);

    // 5. 接收响应
    char buffer[1024];
    ssize_t received = recv(sockfd, buffer, sizeof(buffer) - 1, 0);
    if (received > 0) {
        buffer[received] = '\0';
        printf("收到响应: %s\n", buffer);
    }

    // 6. 关闭连接
    close(sockfd);
}
```

## 5. TCP 服务器示例

```c
void tcp_server_demo(int port) {
    // 1. 创建socket
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) {
        perror("socket");
        exit(EXIT_FAILURE);
    }

    // 允许地址重用（避免TIME_WAIT问题）
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // 2. 绑定地址
    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY; // 监听所有接口
    addr.sin_port = htons(port);

    if (bind(server_fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        perror("bind");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    // 3. 开始监听（backlog=10为等待队列长度）
    if (listen(server_fd, 10) < 0) {
        perror("listen");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("服务器监听端口 %d...\n", port);

    // 4. 接受连接循环
    while (1) {
        struct sockaddr_in client_addr;
        socklen_t client_len = sizeof(client_addr);

        int client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &client_len);
        if (client_fd < 0) {
            perror("accept");
            continue;
        }

        // 获取客户端信息
        char client_ip[INET_ADDRSTRLEN];
        inet_ntop(AF_INET, &client_addr.sin_addr, client_ip, INET_ADDRSTRLEN);
        printf("客户端连接: %s:%d\n", client_ip, ntohs(client_addr.sin_port));

        // 5. 处理客户端请求
        char buffer[1024];
        ssize_t n;
        while ((n = recv(client_fd, buffer, sizeof(buffer) - 1, 0)) > 0) {
            buffer[n] = '\0';
            printf("收到: %s\n", buffer);

            // 回显（Echo）
            send(client_fd, buffer, n, 0);
        }

        printf("客户端断开: %s:%d\n", client_ip, ntohs(client_addr.sin_port));
        close(client_fd);
    }

    close(server_fd);
}
```

## 6. UDP 通信示例

```c
// UDP服务器
void udp_server_demo(int port) {
    int sockfd = socket(AF_INET, SOCK_DGRAM, 0);

    struct sockaddr_in addr, client_addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);

    bind(sockfd, (struct sockaddr*)&addr, sizeof(addr));

    printf("UDP服务器监听端口 %d...\n", port);

    char buffer[1024];
    socklen_t client_len = sizeof(client_addr);

    while (1) {
        // recvfrom同时获取数据和发送方地址
        ssize_t n = recvfrom(sockfd, buffer, sizeof(buffer) - 1, 0,
                             (struct sockaddr*)&client_addr, &client_len);
        if (n > 0) {
            buffer[n] = '\0';
            char ip[INET_ADDRSTRLEN];
            inet_ntop(AF_INET, &client_addr.sin_addr, ip, INET_ADDRSTRLEN);
            printf("收到来自 %s:%d: %s\n", ip, ntohs(client_addr.sin_port), buffer);

            // 回显
            sendto(sockfd, buffer, n, 0,
                   (struct sockaddr*)&client_addr, client_len);
        }
    }

    close(sockfd);
}

// UDP客户端
void udp_client_demo(const char *server_ip, int port) {
    int sockfd = socket(AF_INET, SOCK_DGRAM, 0);

    struct sockaddr_in server_addr;
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(port);
    inet_pton(AF_INET, server_ip, &server_addr.sin_addr);

    // 发送数据（不需要connect）
    const char *msg = "Hello UDP Server!";
    sendto(sockfd, msg, strlen(msg), 0,
           (struct sockaddr*)&server_addr, sizeof(server_addr));

    // 接收响应
    char buffer[1024];
    ssize_t n = recvfrom(sockfd, buffer, sizeof(buffer) - 1, 0, NULL, NULL);
    if (n > 0) {
        buffer[n] = '\0';
        printf("收到响应: %s\n", buffer);
    }

    close(sockfd);
}
```

## 7. 域名解析（DNS）

```c
#include <netdb.h>

void dns_demo() {
    // getaddrinfo：现代的、线程安全的地址解析
    struct addrinfo hints, *result, *rp;
    memset(&hints, 0, sizeof(hints));
    hints.ai_family = AF_UNSPEC;    // IPv4或IPv6
    hints.ai_socktype = SOCK_STREAM; // TCP

    int ret = getaddrinfo("www.example.com", "80", &hints, &result);
    if (ret != 0) {
        fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(ret));
        return;
    }

    // 遍历所有结果
    for (rp = result; rp != NULL; rp = rp->ai_next) {
        char host[NI_MAXHOST], service[NI_MAXSERV];
        getnameinfo(rp->ai_addr, rp->ai_addrlen,
                    host, sizeof(host),
                    service, sizeof(service),
                    NI_NUMERICHOST | NI_NUMERICSERV);
        printf("地址: %s (端口: %s, 协议: %s)\n",
               host, service,
               rp->ai_family == AF_INET ? "IPv4" : "IPv6");
    }

    freeaddrinfo(result);
}
```

## 8. Socket 选项

```c
void socket_options_demo() {
    int sockfd = socket(AF_INET, SOCK_STREAM, 0);

    // 地址重用（避免bind失败在TIME_WAIT状态）
    int optval = 1;
    setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(optval));

    // 设置发送超时
    struct timeval timeout = {5, 0}; // 5秒
    setsockopt(sockfd, SOL_SOCKET, SO_SNDTIMEO, &timeout, sizeof(timeout));

    // 设置接收超时
    setsockopt(sockfd, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout));

    // 设置发送缓冲区大小
    int buf_size = 65536;
    setsockopt(sockfd, SOL_SOCKET, SO_SNDBUF, &buf_size, sizeof(buf_size));

    // 设置接收缓冲区大小
    setsockopt(sockfd, SOL_SOCKET, SO_RCVBUF, &buf_size, sizeof(buf_size));

    // TCP_NODELAY：禁用Nagle算法（减少延迟）
    int nodelay = 1;
    setsockopt(sockfd, IPPROTO_TCP, TCP_NODELAY, &nodelay, sizeof(nodelay));

    // 保活机制
    int keepalive = 1;
    setsockopt(sockfd, SOL_SOCKET, SO_KEEPALIVE, &keepalive, sizeof(keepalive));

    // 获取选项值
    int actual_buf;
    socklen_t len = sizeof(actual_buf);
    getsockopt(sockfd, SOL_SOCKET, SO_RCVBUF, &actual_buf, &len);
    printf("实际接收缓冲区: %d\n", actual_buf);

    close(sockfd);
}
```

## 重点与注意事项

1. **字节序转换**：端口号和IP地址必须使用 `htons/htonl` 转换
2. **错误检查**：所有socket函数都应检查返回值
3. **资源释放**：使用 `close()` 关闭socket，避免文件描述符泄漏
4. **SO_REUSEADDR**：服务器程序应设置此选项，避免重启时bind失败
5. **缓冲区管理**：`send`/`recv` 可能只发送/接收部分数据，需循环处理
6. **IPv6兼容**：使用 `getaddrinfo` 而非硬编码IPv4地址
