# 27 - 数据库编程入门

## 概述

SQLite是一个嵌入式关系数据库，提供了简洁的C API。本节介绍如何在C程序中使用SQLite进行数据库操作。

---

## 1. SQLite基本概念

- 嵌入式数据库，无需服务器进程
- 数据库存储在单个文件中
- 支持标准SQL语法
- ACID事务支持
- 零配置，无需安装

---

## 2. 基本操作

```c
#include <stdio.h>
#include <stdlib.h>
#include <sqlite3.h>

// 打开/创建数据库
int db_open(sqlite3 **db, const char *filename) {
    int rc = sqlite3_open(filename, db);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "无法打开数据库: %s\n", sqlite3_errmsg(*db));
        sqlite3_close(*db);
        return -1;
    }
    printf("数据库已打开: %s\n", filename);
    return 0;
}

// 关闭数据库
void db_close(sqlite3 *db) {
    if (db) {
        sqlite3_close(db);
    }
}

// 执行简单SQL（无返回结果）
int db_exec(sqlite3 *db, const char *sql) {
    char *err_msg = NULL;
    int rc = sqlite3_exec(db, sql, NULL, NULL, &err_msg);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "SQL错误: %s\n", err_msg);
        sqlite3_free(err_msg);
        return -1;
    }
    return 0;
}
```

## 3. 创建表与插入数据

```c
// 创建表
int create_tables(sqlite3 *db) {
    const char *sql =
        "CREATE TABLE IF NOT EXISTS students ("
        "  id INTEGER PRIMARY KEY AUTOINCREMENT,"
        "  name TEXT NOT NULL,"
        "  age INTEGER,"
        "  score REAL,"
        "  created_at DATETIME DEFAULT CURRENT_TIMESTAMP"
        ");";

    return db_exec(db, sql);
}

// 插入数据（使用预处理语句防止SQL注入）
int insert_student(sqlite3 *db, const char *name, int age, double score) {
    const char *sql = "INSERT INTO students (name, age, score) VALUES (?, ?, ?)";
    sqlite3_stmt *stmt;

    int rc = sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "准备语句失败: %s\n", sqlite3_errmsg(db));
        return -1;
    }

    // 绑定参数（索引从1开始）
    sqlite3_bind_text(stmt, 1, name, -1, SQLITE_TRANSIENT);
    sqlite3_bind_int(stmt, 2, age);
    sqlite3_bind_double(stmt, 3, score);

    rc = sqlite3_step(stmt);
    if (rc != SQLITE_DONE) {
        fprintf(stderr, "插入失败: %s\n", sqlite3_errmsg(db));
        sqlite3_finalize(stmt);
        return -1;
    }

    sqlite3_int64 id = sqlite3_last_insert_rowid(db);
    printf("插入成功，ID: %lld\n", id);

    sqlite3_finalize(stmt);
    return (int)id;
}

// 批量插入
int insert_students_batch(sqlite3 *db) {
    const char *sql = "INSERT INTO students (name, age, score) VALUES (?, ?, ?)";
    sqlite3_stmt *stmt;

    sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);

    // 开始事务
    db_exec(db, "BEGIN TRANSACTION");

    const char *names[] = {"张三", "李四", "王五", "赵六", "钱七"};
    int ages[] = {20, 21, 19, 22, 20};
    double scores[] = {85.5, 92.0, 78.5, 88.0, 95.5};

    for (int i = 0; i < 5; i++) {
        sqlite3_bind_text(stmt, 1, names[i], -1, SQLITE_TRANSIENT);
        sqlite3_bind_int(stmt, 2, ages[i]);
        sqlite3_bind_double(stmt, 3, scores[i]);
        sqlite3_step(stmt);
        sqlite3_reset(stmt);  // 重置以便重用
    }

    // 提交事务
    db_exec(db, "COMMIT");

    sqlite3_finalize(stmt);
    printf("批量插入完成\n");
    return 0;
}
```

## 4. 查询数据

```c
// 查询回调函数
int query_callback(void *data, int argc, char **argv, char **col_name) {
    (void)data;
    for (int i = 0; i < argc; i++) {
        printf("%s = %s\n", col_name[i], argv[i] ? argv[i] : "NULL");
    }
    printf("---\n");
    return 0;
}

// 简单查询
void query_all(sqlite3 *db) {
    printf("\n=== 所有学生 ===\n");
    const char *sql = "SELECT * FROM students ORDER BY score DESC";
    db_exec(db, sql);  // 但这不会打印，需要回调

    // 使用回调
    sqlite3_exec(db, sql, query_callback, NULL, NULL);
}

// 使用预处理语句查询
void query_by_score(sqlite3 *db, double min_score) {
    const char *sql = "SELECT id, name, age, score FROM students "
                      "WHERE score >= ? ORDER BY score DESC";
    sqlite3_stmt *stmt;

    sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);
    sqlite3_bind_double(stmt, 1, min_score);

    printf("\n=== 分数 >= %.1f 的学生 ===\n", min_score);
    while (sqlite3_step(stmt) == SQLITE_ROW) {
        int id = sqlite3_column_int(stmt, 0);
        const char *name = (const char *)sqlite3_column_text(stmt, 1);
        int age = sqlite3_column_int(stmt, 2);
        double score = sqlite3_column_double(stmt, 3);

        printf("ID: %d, 姓名: %s, 年龄: %d, 分数: %.1f\n",
               id, name, age, score);
    }

    sqlite3_finalize(stmt);
}

// 查询单个值
double get_avg_score(sqlite3 *db) {
    const char *sql = "SELECT AVG(score) FROM students";
    sqlite3_stmt *stmt;

    sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);

    double avg = 0;
    if (sqlite3_step(stmt) == SQLITE_ROW) {
        avg = sqlite3_column_double(stmt, 0);
    }

    sqlite3_finalize(stmt);
    return avg;
}
```

## 5. 更新与删除

```c
// 更新
int update_score(sqlite3 *db, int id, double new_score) {
    const char *sql = "UPDATE students SET score = ? WHERE id = ?";
    sqlite3_stmt *stmt;

    sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);
    sqlite3_bind_double(stmt, 1, new_score);
    sqlite3_bind_int(stmt, 2, id);

    int rc = sqlite3_step(stmt);
    int changes = sqlite3_changes(db);
    sqlite3_finalize(stmt);

    printf("更新了 %d 条记录\n", changes);
    return rc == SQLITE_DONE ? 0 : -1;
}

// 删除
int delete_student(sqlite3 *db, int id) {
    const char *sql = "DELETE FROM students WHERE id = ?";
    sqlite3_stmt *stmt;

    sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);
    sqlite3_bind_int(stmt, 1, id);

    int rc = sqlite3_step(stmt);
    int changes = sqlite3_changes(db);
    sqlite3_finalize(stmt);

    printf("删除了 %d 条记录\n", changes);
    return rc == SQLITE_DONE ? 0 : -1;
}
```

## 6. 事务处理

```c
int transaction_demo(sqlite3 *db) {
    char *err_msg = NULL;

    // 开始事务
    if (sqlite3_exec(db, "BEGIN TRANSACTION", NULL, NULL, &err_msg) != SQLITE_OK) {
        fprintf(stderr, "开始事务失败: %s\n", err_msg);
        sqlite3_free(err_msg);
        return -1;
    }

    // 执行多个操作
    int success = 1;
    if (insert_student(db, "事务测试1", 25, 90.0) < 0) success = 0;
    if (insert_student(db, "事务测试2", 26, 88.0) < 0) success = 0;

    // 提交或回滚
    if (success) {
        sqlite3_exec(db, "COMMIT", NULL, NULL, NULL);
        printf("事务提交成功\n");
    } else {
        sqlite3_exec(db, "ROLLBACK", NULL, NULL, NULL);
        printf("事务已回滚\n");
    }

    return success ? 0 : -1;
}
```

## 7. 完整示例

```c
int main(void) {
    sqlite3 *db;

    // 打开数据库
    if (db_open(&db, "students.db") != 0) {
        return 1;
    }

    // 创建表
    create_tables(db);

    // 插入数据
    insert_students_batch(db);

    // 查询
    query_all(db);
    query_by_score(db, 85.0);

    // 统计
    printf("\n平均分: %.2f\n", get_avg_score(db));

    // 更新
    update_score(db, 1, 99.0);

    // 删除
    delete_student(db, 5);

    // 关闭
    db_close(db);
    printf("\n数据库已关闭\n");

    return 0;
}
```

## 要点总结

1. 始终使用预处理语句（sqlite3_prepare_v2）防止SQL注入
2. 批量操作使用事务提高性能
3. 使用完毕后调用sqlite3_finalize释放语句
4. 检查返回值，处理错误
5. sqlite3_changes()获取受影响的行数
6. SQLite支持的数据类型：NULL, INTEGER, REAL, TEXT, BLOB
