# 26 - 协议与序列化

## 概述

在网络通信和数据存储中，需要将内存中的数据结构转换为字节流（序列化），以及从字节流还原数据（反序列化）。本节介绍二进制协议设计和TLV格式。

---

## 1. 字节序处理

```c
#include <stdint.h>
#include <string.h>

// 主机字节序检测
int is_little_endian(void) {
    union {
        uint32_t i;
        uint8_t c[4];
    } test = {0x01020304};
    return test.c[0] == 0x04;
}

// 字节序转换
uint16_t swap_endian16(uint16_t val) {
    return (val >> 8) | (val << 8);
}

uint32_t swap_endian32(uint32_t val) {
    return ((val >> 24) & 0xFF) |
           ((val >> 8)  & 0xFF00) |
           ((val << 8)  & 0xFF0000) |
           ((val << 24) & 0xFF000000);
}

// 网络字节序（大端）转换
uint16_t htons_custom(uint16_t host) {
    return is_little_endian() ? swap_endian16(host) : host;
}

uint32_t htonl_custom(uint32_t host) {
    return is_little_endian() ? swap_endian32(host) : host;
}

uint16_t ntohs_custom(uint16_t net) {
    return is_little_endian() ? swap_endian16(net) : net;
}

uint32_t ntohl_custom(uint32_t net) {
    return is_little_endian() ? swap_endian32(net) : net;
}

// 或者使用标准库
#include <arpa/inet.h>
// uint16_t htons(uint16_t);  // 主机到网络短整型
// uint32_t htonl(uint32_t);  // 主机到网络长整型
```

## 2. 基本序列化

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>

// 缓冲区
typedef struct {
    uint8_t *data;
    size_t size;
    size_t capacity;
    size_t read_pos;
} Buffer;

Buffer *buf_create(size_t capacity) {
    Buffer *b = (Buffer *)malloc(sizeof(Buffer));
    b->data = (uint8_t *)malloc(capacity);
    b->size = 0;
    b->capacity = capacity;
    b->read_pos = 0;
    return b;
}

void buf_free(Buffer *b) {
    free(b->data);
    free(b);
}

// 确保容量
static void buf_ensure(Buffer *b, size_t needed) {
    while (b->size + needed > b->capacity) {
        b->capacity *= 2;
        b->data = (uint8_t *)realloc(b->data, b->capacity);
    }
}

// 写入基本类型（网络字节序）
void buf_write_u8(Buffer *b, uint8_t val) {
    buf_ensure(b, 1);
    b->data[b->size++] = val;
}

void buf_write_u16(Buffer *b, uint16_t val) {
    val = htons(val);
    buf_ensure(b, 2);
    memcpy(b->data + b->size, &val, 2);
    b->size += 2;
}

void buf_write_u32(Buffer *b, uint32_t val) {
    val = htonl(val);
    buf_ensure(b, 4);
    memcpy(b->data + b->size, &val, 4);
    b->size += 4;
}

void buf_write_bytes(Buffer *b, const void *data, size_t len) {
    buf_ensure(b, len);
    memcpy(b->data + b->size, data, len);
    b->size += len;
}

void buf_write_string(Buffer *b, const char *str) {
    uint16_t len = (uint16_t)strlen(str);
    buf_write_u16(b, len);
    buf_write_bytes(b, str, len);
}

// 读取基本类型
uint8_t buf_read_u8(Buffer *b) {
    return b->data[b->read_pos++];
}

uint16_t buf_read_u16(Buffer *b) {
    uint16_t val;
    memcpy(&val, b->data + b->read_pos, 2);
    b->read_pos += 2;
    return ntohs(val);
}

uint32_t buf_read_u32(Buffer *b) {
    uint32_t val;
    memcpy(&val, b->data + b->read_pos, 4);
    b->read_pos += 4;
    return ntohl(val);
}

void buf_read_bytes(Buffer *b, void *dest, size_t len) {
    memcpy(dest, b->data + b->read_pos, len);
    b->read_pos += len;
}

char *buf_read_string(Buffer *b) {
    uint16_t len = buf_read_u16(b);
    char *str = (char *)malloc(len + 1);
    buf_read_bytes(b, str, len);
    str[len] = '\0';
    return str;
}
```

## 3. TLV格式（Type-Length-Value）

TLV是一种通用的二进制编码格式，常用于协议设计。

```
+--------+--------+----------+
|  Type  | Length |  Value   |
| 2字节  | 2字节  | N字节    |
+--------+--------+----------+
```

```c
#include <stdint.h>
#include <string.h>
#include <stdlib.h>

// TLV标签定义
#define TLV_INT8    0x0001
#define TLV_INT16   0x0002
#define TLV_INT32   0x0004
#define TLV_FLOAT   0x0008
#define TLV_STRING  0x0010
#define TLV_BYTES   0x0020
#define TLV_ARRAY   0x0040

// 编码：写入TLV
void tlv_write_int32(Buffer *b, uint16_t tag, int32_t value) {
    buf_write_u16(b, tag);           // Type
    buf_write_u16(b, 4);             // Length
    buf_write_u32(b, (uint32_t)value); // Value
}

void tlv_write_string(Buffer *b, uint16_t tag, const char *str) {
    uint16_t len = (uint16_t)strlen(str);
    buf_write_u16(b, tag);           // Type
    buf_write_u16(b, len);           // Length
    buf_write_bytes(b, str, len);    // Value
}

void tlv_write_float(Buffer *b, uint16_t tag, float value) {
    buf_write_u16(b, tag);
    buf_write_u16(b, 4);
    uint32_t bits;
    memcpy(&bits, &value, 4);
    buf_write_u32(b, bits);
}

// 解码：读取TLV
typedef struct {
    uint16_t type;
    uint16_t length;
    union {
        int32_t int_val;
        float float_val;
        char *str_val;
        struct { uint8_t *data; } bytes_val;
    } value;
} TLVItem;

int tlv_read_item(Buffer *b, TLVItem *item) {
    if (b->read_pos + 4 > b->size) return -1;

    item->type = buf_read_u16(b);
    item->length = buf_read_u16(b);

    if (b->read_pos + item->length > b->size) return -1;

    switch (item->type) {
    case TLV_INT32:
        item->value.int_val = (int32_t)buf_read_u32(b);
        break;
    case TLV_FLOAT: {
        uint32_t bits = buf_read_u32(b);
        memcpy(&item->value.float_val, &bits, 4);
        break;
    }
    case TLV_STRING:
        item->value.str_val = (char *)malloc(item->length + 1);
        buf_read_bytes(b, item->value.str_val, item->length);
        item->value.str_val[item->length] = '\0';
        break;
    default:
        // 跳过未知类型
        b->read_pos += item->length;
        break;
    }

    return 0;
}
```

## 4. 消息协议示例

```c
// 定义消息类型
#define MSG_LOGIN     0x01
#define MSG_RESPONSE  0x02
#define MSG_DATA      0x03
#define MSG_HEARTBEAT 0x04

// 协议头
typedef struct __attribute__((packed)) {
    uint8_t  magic;      // 魔数 0xAA
    uint8_t  version;    // 版本号
    uint8_t  msg_type;   // 消息类型
    uint16_t length;     // 数据长度
    uint16_t checksum;   // 校验和
} ProtocolHeader;

#define PROTOCOL_MAGIC 0xAA
#define PROTOCOL_VERSION 1

// 计算校验和
uint16_t compute_checksum(const uint8_t *data, size_t len) {
    uint32_t sum = 0;
    for (size_t i = 0; i < len; i++) {
        sum += data[i];
    }
    return (uint16_t)(sum & 0xFFFF);
}

// 编码消息
Buffer *encode_message(uint8_t msg_type, const void *payload,
                       uint16_t payload_len) {
    Buffer *b = buf_create(sizeof(ProtocolHeader) + payload_len);

    // 写入头部
    buf_write_u8(b, PROTOCOL_MAGIC);
    buf_write_u8(b, PROTOCOL_VERSION);
    buf_write_u8(b, msg_type);
    buf_write_u16(b, payload_len);
    buf_write_u16(b, 0);  // 先写0，后面计算

    // 写入数据
    buf_write_bytes(b, payload, payload_len);

    // 计算并更新校验和
    uint16_t checksum = compute_checksum(b->data + 3, b->size - 3);
    b->data[5] = (checksum >> 8) & 0xFF;
    b->data[6] = checksum & 0xFF;

    return b;
}

// 解码消息
int decode_message(const uint8_t *data, size_t len,
                   ProtocolHeader *header, uint8_t **payload) {
    if (len < sizeof(ProtocolHeader)) return -1;

    memcpy(header, data, sizeof(ProtocolHeader));

    // 检查魔数
    if (header->magic != PROTOCOL_MAGIC) return -2;

    // 验证校验和
    uint16_t expected = ntohs(header->checksum);
    header->checksum = 0;  // 临时清零
    uint16_t actual = compute_checksum(data + 3, len - 3);
    if (expected != actual) return -3;

    // 提取数据
    uint16_t payload_len = ntohs(header->length);
    *payload = (uint8_t *)malloc(payload_len);
    memcpy(*payload, data + sizeof(ProtocolHeader), payload_len);

    return 0;
}
```

## 5. 结构体序列化宏

```c
#include <string.h>

// 序列化结构体字段的宏
#define SERIALIZE_FIELD(buf, obj, field) \
    buf_write_bytes(buf, &(obj)->field, sizeof((obj)->field))

#define DESERIALIZE_FIELD(buf, obj, field) \
    buf_read_bytes(buf, &(obj)->field, sizeof((obj)->field))

// 用户数据结构
typedef struct {
    int32_t id;
    char name[32];
    float score;
    uint8_t active;
} User;

// 序列化
void user_serialize(Buffer *b, const User *user) {
    buf_write_u32(b, (uint32_t)user->id);
    buf_write_string(b, user->name);
    buf_write_u32(b, *(uint32_t *)&user->score);  // float -> uint32
    buf_write_u8(b, user->active);
}

// 反序列化
User *user_deserialize(Buffer *b) {
    User *user = (User *)calloc(1, sizeof(User));
    user->id = (int32_t)buf_read_u32(b);
    char *name = buf_read_string(b);
    strncpy(user->name, name, sizeof(user->name) - 1);
    free(name);
    uint32_t score_bits = buf_read_u32(b);
    memcpy(&user->score, &score_bits, 4);
    user->active = buf_read_u8(b);
    return user;
}

void serialization_demo(void) {
    // 创建用户
    User user = {1001, "张三", 95.5f, 1};

    // 序列化
    Buffer *b = buf_create(256);
    user_serialize(b, &user);
    printf("序列化大小: %zu 字节\n", b->size);

    // 反序列化
    User *decoded = user_deserialize(b);
    printf("ID: %d, 姓名: %s, 分数: %.1f, 活跃: %d\n",
           decoded->id, decoded->name, decoded->score, decoded->active);

    free(decoded);
    buf_free(b);
}
```

## 要点总结

1. 网络协议使用大端字节序（网络字节序）
2. 设计协议时要考虑魔数、版本号、长度和校验和
3. TLV格式灵活通用，适合扩展
4. 序列化要注意对齐、字节序和字符串长度
5. 缓冲区要有容量检查，防止溢出
6. 实际项目可考虑Protocol Buffers、MessagePack等成熟方案
