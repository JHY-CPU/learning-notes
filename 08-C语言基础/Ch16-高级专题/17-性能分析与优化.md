# 17 - 性能分析与优化

## 概述

性能优化需要先分析瓶颈，再针对性优化。本节介绍常用profiling工具和优化策略。

---

## 1. 性能分析工具

### gprof（GNU Profiler）

```bash
# 编译时加入-pg选项
gcc -pg -g -o app app.c

# 运行程序，生成gmon.out
./app

# 分析结果
gprof app gmon.out

# 生成调用图
gprof -q app gmon.out

# 生成扁平分析
gprof -p app gmon.out
```

### perf（Linux性能分析）

```bash
# 记录性能数据
perf record -g ./app

# 查看报告
perf report

# 实时监控
perf stat ./app

# 热点函数分析
perf top

# 缓存命中率
perf stat -e cache-misses,cache-references ./app

# 分支预测
perf stat -e branches,branch-misses ./app
```

### Valgrind Callgrind

```bash
# 收集调用数据
valgrind --tool=callgrind ./app

# 可视化（需要kcachegrind）
kcachegrind callgrind.out.12345
```

## 2. 基准测试框架

```c
#include <stdio.h>
#include <time.h>
#include <stdlib.h>
#include <string.h>

// 高精度计时
typedef struct {
    struct timespec start;
    struct timespec end;
} Benchmark;

static inline void bench_start(Benchmark *b) {
    clock_gettime(CLOCK_MONOTONIC, &b->start);
}

static inline double bench_end(Benchmark *b) {
    clock_gettime(CLOCK_MONOTONIC, &b->end);
    double elapsed = (b->end.tv_sec - b->start.tv_sec) +
                     (b->end.tv_nsec - b->start.tv_nsec) / 1e9;
    return elapsed;
}

// 运行基准测试
#define BENCHMARK(name, iterations, code) do { \
    Benchmark b; \
    bench_start(&b); \
    for (int _i = 0; _i < (iterations); _i++) { \
        code; \
    } \
    double elapsed = bench_end(&b); \
    printf("%-20s: %.3fms (%.0f ops/sec)\n", \
           name, elapsed * 1000, (iterations) / elapsed); \
} while(0)
```

## 3. 优化策略

### 缓存友好优化

```c
// 不好：按列遍历（缓存不友好）
void bad_access(int **matrix, int n) {
    for (int j = 0; j < n; j++) {
        for (int i = 0; i < n; i++) {
            matrix[i][j] *= 2;  // 跳跃访问
        }
    }
}

// 好：按行遍历（缓存友好）
void good_access(int **matrix, int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            matrix[i][j] *= 2;  // 连续访问
        }
    }
}

// 一维数组代替二维数组
void flat_access(int *matrix, int n) {
    for (int i = 0; i < n * n; i++) {
        matrix[i] *= 2;  // 完全连续
    }
}
```

### 循环优化

```c
// 循环展开
void unroll_sum(int *arr, int n, int *result) {
    int sum = 0;
    int i;
    // 每次处理4个元素
    for (i = 0; i <= n - 4; i += 4) {
        sum += arr[i] + arr[i+1] + arr[i+2] + arr[i+3];
    }
    // 处理剩余
    for (; i < n; i++) {
        sum += arr[i];
    }
    *result = sum;
}

// 循环融合
// 融合前
void separate_loops(int *a, int *b, int *c, int n) {
    for (int i = 0; i < n; i++) a[i] = b[i] + 1;
    for (int i = 0; i < n; i++) c[i] = a[i] * 2;
}

// 融合后（减少循环开销，提高缓存局部性）
void fused_loop(int *a, int *b, int *c, int n) {
    for (int i = 0; i < n; i++) {
        a[i] = b[i] + 1;
        c[i] = a[i] * 2;
    }
}

// 循环交换（改善缓存命中）
// 遍历大数组时按缓存行组织访问
void blocked_matrix_multiply(float *A, float *B, float *C, int n) {
    int block = 64;  // 缓存友好的块大小
    for (int ii = 0; ii < n; ii += block) {
        for (int jj = 0; jj < n; jj += block) {
            for (int kk = 0; kk < n; kk += block) {
                for (int i = ii; i < ii + block && i < n; i++) {
                    for (int j = jj; j < jj + block && j < n; j++) {
                        float sum = 0;
                        for (int k = kk; k < kk + block && k < n; k++) {
                            sum += A[i * n + k] * B[k * n + j];
                        }
                        C[i * n + j] += sum;
                    }
                }
            }
        }
    }
}
```

### 分支预测优化

```c
// 使用likely/unlikely提示编译器
#define likely(x)   __builtin_expect(!!(x), 1)
#define unlikely(x) __builtin_expect(!!(x), 0)

void process(int *data, int n) {
    for (int i = 0; i < n; i++) {
        if (unlikely(data[i] < 0)) {  // 假设负数很少
            handle_negative(data[i]);
        } else {
            process_normal(data[i]);
        }
    }
}

// 查找优化：消除分支
int find_index(int *arr, int n, int target) {
    int result = -1;
    for (int i = 0; i < n; i++) {
        // 无分支的条件赋值
        result = (arr[i] == target) ? i : result;
    }
    return result;
}
```

## 4. 内存优化

```c
// 内存池减少malloc开销
typedef struct {
    char *pool;
    size_t size;
    size_t used;
} MemPool;

MemPool *pool_create(size_t size) {
    MemPool *pool = malloc(sizeof(MemPool));
    pool->pool = malloc(size);
    pool->size = size;
    pool->used = 0;
    return pool;
}

void *pool_alloc(MemPool *pool, size_t size) {
    // 对齐到8字节
    size = (size + 7) & ~7;
    if (pool->used + size > pool->size) return NULL;
    void *ptr = pool->pool + pool->used;
    pool->used += size;
    return ptr;
}

void pool_reset(MemPool *pool) {
    pool->used = 0;  // 一次性释放所有内存
}

void pool_destroy(MemPool *pool) {
    free(pool->pool);
    free(pool);
}
```

## 5. 编译器优化选项

```bash
# 优化级别
gcc -O0  # 无优化（默认）
gcc -O1  # 基本优化
gcc -O2  # 推荐的优化级别
gcc -O3  # 激进优化（可能增大代码体积）
gcc -Os  # 优化代码大小
gcc -Ofast # 最高性能（可能牺牲精度）

# 特定优化
gcc -funroll-loops      # 循环展开
gcc -ffast-math         # 快速浮点运算
gcc -march=native       # 针对当前CPU优化
gcc -flto               # 链接时优化
gcc -fprofile-generate  # 生成profile数据
gcc -fprofile-use       # 使用profile数据优化
```

## 6. 性能分析实例

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// 测试矩阵乘法优化效果
void test_matrix_multiply(void) {
    const int N = 512;
    float *A = malloc(N * N * sizeof(float));
    float *B = malloc(N * N * sizeof(float));
    float *C = calloc(N * N, sizeof(float));

    // 随机初始化
    srand(42);
    for (int i = 0; i < N * N; i++) {
        A[i] = (float)rand() / RAND_MAX;
        B[i] = (float)rand() / RAND_MAX;
    }

    Benchmark b;

    // 基础版本
    bench_start(&b);
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            for (int k = 0; k < N; k++)
                C[i*N+j] += A[i*N+k] * B[k*N+j];
    printf("基础版本: %.3fms\n", bench_end(&b) * 1000);

    // 重置C
    memset(C, 0, N * N * sizeof(float));

    // 分块版本
    bench_start(&b);
    blocked_matrix_multiply(A, B, C, N);
    printf("分块版本: %.3fms\n", bench_end(&b) * 1000);

    free(A); free(B); free(C);
}
```

## 要点总结

1. 先分析再优化，不要猜测瓶颈
2. 缓存友好是最重要的优化原则
3. 循环展开、循环融合、循环交换是常见优化手段
4. 编译器-O2是性能和安全的良好平衡
5. 使用perf/gprof定位热点函数
6. 内存池可大幅减少动态内存分配开销
