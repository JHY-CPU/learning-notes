# 06 - 预定义宏

## 什么是预定义宏

C语言标准和各种编译器提供了一系列预定义宏，这些宏不需要用户定义，由编译器自动提供。它们在调试、日志记录、条件编译和跨平台开发中非常有用。

## 标准预定义宏（C89/C99/C11）

### __FILE__ — 当前源文件名

```c
#include <stdio.h>

int main(void) {
    printf("当前文件: %s\n", __FILE__);
    // 输出: 当前文件: main.c
    // 如果文件在子目录中，可能输出完整路径
    return 0;
}
```

### __LINE__ — 当前行号

```c
#include <stdio.h>

int main(void) {
    printf("行号: %d\n", __LINE__);   // 输出: 行号: 5
    printf("行号: %d\n", __LINE__);   // 输出: 行号: 6
    int x = 10;
    printf("行号: %d\n", __LINE__);   // 输出: 行号: 8
    return 0;
}
```

### __func__ — 当前函数名（C99）

```c
#include <stdio.h>

void greet(void) {
    printf("当前函数: %s\n", __func__);
    // 输出: 当前函数: greet
}

int main(void) {
    printf("当前函数: %s\n", __func__);
    // 输出: 当前函数: main
    greet();
    return 0;
}
```

### __DATE__ 和 __TIME__ — 编译日期和时间

```c
#include <stdio.h>

int main(void) {
    printf("编译日期: %s\n", __DATE__);  // 输出格式: "May  5 2026"
    printf("编译时间: %s\n", __TIME__);  // 输出格式: "14:30:00"
    return 0;
}
```

### __STDC__ — 标准C编译器标识

```c
#include <stdio.h>

int main(void) {
    #ifdef __STDC__
        printf("这是一个标准C编译器\n");
        printf("C标准版本: %ld\n", __STDC_VERSION__);
        // C89: 未定义
        // C99: 199901L
        // C11: 201112L
        // C17: 201710L
        // C23: 202311L
    #else
        printf("这是一个非标准C编译器\n");
    #endif
    return 0;
}
```

## 实用调试宏

### 完整的调试信息宏

```c
#include <stdio.h>

// 带文件名和行号的调试输出
#define DEBUG_PRINT(fmt, ...) \
    fprintf(stderr, "[%s:%d:%s] " fmt "\n", \
            __FILE__, __LINE__, __func__, ##__VA_ARGS__)

// 使用示例
void process(int value) {
    DEBUG_PRINT("输入值: %d", value);
    // 输出: [main.c:10:process] 输入值: 42
    
    if (value < 0) {
        DEBUG_PRINT("错误: 负值!");
        // 输出: [main.c:13:process] 错误: 负值!
    }
}
```

### 断言宏的增强实现

```c
#include <stdio.h>
#include <stdlib.h>

#define ASSERT(cond) do { \
    if (!(cond)) { \
        fprintf(stderr, \
                "断言失败: %s\n" \
                "  文件: %s\n" \
                "  行号: %d\n" \
                "  函数: %s\n" \
                "  时间: %s %s\n", \
                #cond, __FILE__, __LINE__, __func__, \
                __DATE__, __TIME__); \
        abort(); \
    } \
} while(0)

int divide(int a, int b) {
    ASSERT(b != 0);  // 失败时输出详细的上下文信息
    return a / b;
}
```

### 版本信息宏

```c
#define APP_NAME    "MyApplication"
#define VERSION_MAJOR 2
#define VERSION_MINOR 1
#define VERSION_PATCH 0

#define STRINGIFY(x) #x
#define TOSTRING(x) STRINGIFY(x)
#define VERSION_STRING \
    TOSTRING(VERSION_MAJOR) "." \
    TOSTRING(VERSION_MINOR) "." \
    TOSTRING(VERSION_PATCH)

void print_version(void) {
    printf("%s v%s\n", APP_NAME, VERSION_STRING);
    printf("编译于: %s %s\n", __DATE__, __TIME__);
    printf("文件: %s\n", __FILE__);
}
```

## 编译器特定预定义宏

### GCC 预定义宏

```c
#include <stdio.h>

int main(void) {
    #ifdef __GNUC__
        printf("GCC 版本: %d.%d.%d\n",
               __GNUC__, __GNUC_MINOR__, __GNUC_PATCHLEVEL__);
    #endif
    
    #ifdef __GNUG__
        printf("这是 G++ (C++ 编译器)\n");
    #endif
    
    #ifdef __OPTIMIZE__
        #ifdef __OPTIMIZE_SIZE__
            printf("使用 -Os 优化\n");
        #else
            printf("使用优化编译\n");
        #endif
    #else
        printf("无优化编译\n");
    #endif
    
    return 0;
}
```

### 平台检测宏

```c
// 操作系统检测
#if defined(_WIN32) || defined(_WIN64)
    #define PLATFORM_WINDOWS
    #include <windows.h>
#elif defined(__linux__)
    #define PLATFORM_LINUX
    #include <unistd.h>
#elif defined(__APPLE__) && defined(__MACH__)
    #define PLATFORM_MACOS
    #include <sys/types.h>
#elif defined(__unix__)
    #define PLATFORM_UNIX
    #include <unistd.h>
#else
    #define PLATFORM_UNKNOWN
#endif

// 架构检测
#if defined(__x86_64__) || defined(_M_X64)
    #define ARCH_X64
#elif defined(__i386__) || defined(_M_IX86)
    #define ARCH_X86
#elif defined(__aarch64__) || defined(_M_ARM64)
    #define ARCH_ARM64
#elif defined(__arm__) || defined(_M_ARM)
    #define ARCH_ARM
#endif
```

### 编译器检测

```c
#if defined(__GNUC__) && !defined(__clang__)
    #define COMPILER_GCC
    #define COMPILER_NAME "GCC"
    #define COMPILER_VERSION \
        (__GNUC__ * 10000 + __GNUC_MINOR__ * 100 + __GNUC_PATCHLEVEL__)
#elif defined(__clang__)
    #define COMPILER_CLANG
    #define COMPILER_NAME "Clang"
    #define COMPILER_VERSION \
        (__clang_major__ * 10000 + __clang_minor__ * 100 + __clang_patchlevel__)
#elif defined(_MSC_VER)
    #define COMPILER_MSVC
    #define COMPILER_NAME "MSVC"
    #define COMPILER_VERSION _MSC_VER
#else
    #define COMPILER_UNKNOWN
    #define COMPILER_NAME "Unknown"
    #define COMPILER_VERSION 0
#endif

void print_compiler_info(void) {
    printf("编译器: %s (版本: %d)\n", COMPILER_NAME, COMPILER_VERSION);
}
```

## C标准版本检测

```c
void print_c_standard(void) {
    #ifdef __STDC_VERSION__
        #if __STDC_VERSION__ >= 202311L
            printf("C23 标准\n");
        #elif __STDC_VERSION__ >= 201710L
            printf("C17 标准\n");
        #elif __STDC_VERSION__ >= 201112L
            printf("C11 标准\n");
        #elif __STDC_VERSION__ >= 199901L
            printf("C99 标准\n");
        #else
            printf("C89/C90 标准\n");
        #endif
    #else
        printf("C89/C90 (ANSI C) 标准\n");
    #endif
}
```

## 构建信息时间戳

```c
// 记录构建时间（用于版本追踪）
#define BUILD_INFO \
    "Built on " __DATE__ " " __TIME__ \
    " with " COMPILER_NAME

void show_build_info(void) {
    printf("%s\n", BUILD_INFO);
    // 输出: Built on May  5 2026 14:30:00 with GCC
}
```

## 列出所有常用预定义宏

```c
#include <stdio.h>

void list_predefined_macros(void) {
    printf("=== 标准预定义宏 ===\n");
    printf("__FILE__: %s\n", __FILE__);
    printf("__LINE__: %d\n", __LINE__);
    printf("__func__: %s\n", __func__);
    printf("__DATE__: %s\n", __DATE__);
    printf("__TIME__: %s\n", __TIME__);
    
    #ifdef __STDC__
        printf("__STDC__: %d\n", __STDC__);
    #endif
    #ifdef __STDC_VERSION__
        printf("__STDC_VERSION__: %ld\n", __STDC_VERSION__);
    #endif
    #ifdef __STDC_HOSTED__
        printf("__STDC_HOSTED__: %d\n", __STDC_HOSTED__);
    #endif
    
    printf("\n=== 编译器宏 ===\n");
    #ifdef __GNUC__
        printf("__GNUC__: %d\n", __GNUC__);
        printf("__GNUC_MINOR__: %d\n", __GNUC_MINOR__);
    #endif
    #ifdef __clang__
        printf("__clang__: %d\n", __clang__);
    #endif
    #ifdef _MSC_VER
        printf("_MSC_VER: %d\n", _MSC_VER);
    #endif
}
```

## 重要注意事项

> **关键点总结**：
> 1. `__FILE__` 和 `__LINE__` 是调试和日志的核心工具
> 2. `__func__` 从 C99 开始标准定义，提供函数名信息
> 3. `__DATE__` 和 `__TIME__` 反映编译时刻，可用于版本追踪
> 4. 不同编译器提供不同的预定义宏，条件编译前先检查可用性
> 5. 平台和编译器检测宏是跨平台开发的基础
> 6. `__STDC_VERSION__` 用于检测 C 语言标准版本，决定可用的特性
