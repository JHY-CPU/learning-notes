# 23 - 编译器扩展

## GCC 扩展概述

GCC 提供了大量 C 标准之外的扩展功能。这些扩展在 Linux 内核和许多系统级项目中广泛使用。使用扩展会降低可移植性，但在某些场景下非常有价值。

## 语句表达式（Statement Expr）

### 基本语法

```c
({ 语句列表; 最后一个表达式的值; })
```

语句表达式允许在表达式中使用多条语句，结果为最后一个表达式的值。

### 示例

```c
// 安全的 MAX 宏
#define MAX(a, b) ({ \
    typeof(a) _a = (a); \
    typeof(b) _b = (b); \
    _a > _b ? _a : _b; \
})

int x = MAX(3 + 4, 5 * 2);  // 安全：参数只求值一次

// 带日志的表达式
#define LOG_AND_RETURN(val) ({ \
    typeof(val) _v = (val); \
    printf("返回值: %d\n", _v); \
    _v; \
})

int result = LOG_AND_RETURN(compute());
```

### 实际应用

```c
// 安全的交换宏
#define SWAP(a, b) ({ \
    typeof(a) _tmp = (a); \
    (a) = (b); \
    (b) = _tmp; \
})

// 带错误检查的赋值
#define SAFE_DIVIDE(a, b) ({ \
    typeof(a) _a = (a); \
    typeof(b) _b = (b); \
    _b != 0 ? _a / _b : ({ \
        fprintf(stderr, "除零错误!\n"); \
        0; \
    }); \
})
```

## typeof 运算符

### 基本用法

```c
// typeof 获取表达式的类型
int x = 42;
typeof(x) y = 100;  // y 的类型是 int

typeof(int *) ptr;   // ptr 的类型是 int*

// 用于宏，实现类型无关
#define MIN(a, b) ({ \
    typeof(a) _a = (a); \
    typeof(b) _b = (b); \
    _a < _b ? _a : _b; \
})
```

### 与数组配合

```c
// 安全的数组大小宏
#define ARRAY_SIZE(arr) (sizeof(arr) / sizeof(typeof(arr[0])))

// 类型安全的 foreach
#define FOREACH(item, array) \
    for (typeof((array)[0]) *item = (array); \
         item < (array) + ARRAY_SIZE(array); \
         item++)

int arr[] = {1, 2, 3, 4, 5};
FOREACH(p, arr) {
    printf("%d ", *p);
}
```

## __attribute__ 属性

### 函数属性

```c
// deprecated — 标记为已弃用
__attribute__((deprecated("请使用 new_function")))
void old_function(void);

// noreturn — 函数不会返回
__attribute__((noreturn))
void fatal_error(const char *msg);

// format — 检查 printf 风格格式字符串
__attribute__((format(printf, 1, 2)))
void my_printf(const char *fmt, ...);

// always_inline — 强制内联
__attribute__((always_inline))
static inline int fast_add(int a, int b) { return a + b; }

// noinline — 禁止内联
__attribute__((noinline))
void large_function(void);

// weak — 弱符号
__attribute__((weak))
void optional_feature(void) { /* 默认实现 */ }

// constructor — 在 main 之前执行
__attribute__((constructor))
void init(void) { printf("初始化\n"); }

// destructor — 在 main 之后执行
__attribute__((destructor))
void cleanup(void) { printf("清理\n"); }
```

### 变量属性

```c
// aligned — 对齐
int aligned_var __attribute__((aligned(64)));

// packed — 紧凑排列
struct __attribute__((packed)) NetworkHeader {
    char type;
    int length;
    short checksum;
};
// sizeof = 7（而不是 12）

// section — 指定段
int __attribute__((section(".special")))
special_var = 42;

// unused — 抑制未使用警告
int __attribute__((unused)) debug_flag = 1;

// cleanup — 变量离开作用域时自动调用清理函数
void auto_free(void *p) { free(*(void **)p); }
#define AUTOFREE __attribute__((cleanup(auto_free)))
void example(void) {
    char * AUTOFREE buf = malloc(100);
    // 函数返回时自动 free(buf)
}
```

### 类型属性

```c
// 结构体对齐
struct __attribute__((aligned(16))) AlignedStruct {
    int x, y;
};

// packed 结构体
typedef struct __attribute__((packed)) {
    uint8_t type;
    uint32_t data;
    uint16_t checksum;
} Packet;

// may_alias — 允许别名
typedef int __attribute__((may_alias)) alias_int;
```

## 内建函数

```c
// __builtin_expect — 分支预测提示
if (__builtin_expect(x > 0, 1)) {
    // x > 0 更可能为真
}
#define likely(x)   __builtin_expect(!!(x), 1)
#define unlikely(x) __builtin_expect(!!(x), 0)

if (unlikely(error_condition)) {
    // 错误处理路径
}

// __builtin_unreachable — 标记不可达代码
int process(int x) {
    if (x < 0) abort();
    // 编译器知道 x >= 0
    __builtin_assume(x >= 0);  // GCC 12+
    return x * 2;
}

// __builtin_popcount — 计算置位数
int bits = __builtin_popcount(0xFF);  // = 8

// __builtin_clz / __builtin_ctz
int leading = __builtin_clz(16);   // 前导零个数 = 27（32位系统）
int trailing = __builtin_ctz(16);  // 尾随零个数 = 4

// __builtin_offsetof
size_t off = __builtin_offsetof(struct Point, y);
```

## 零长度数组

```c
// GCC 允许零长度数组作为结构体的柔性成员
struct Buffer {
    size_t length;
    char data[0];  // 零长度数组
};

struct Buffer *buf = malloc(sizeof(struct Buffer) + 100);
buf->length = 100;
// buf->data 可以当作 100 字节的数组使用
```

注意：C99 标准的柔性数组成员 `char data[];` 是更好的替代方案。

## 标号值（Labels as Values）

```c
// 获取标号的地址
void *ptr = &&label1;

// 跳转到标号
goto *ptr;

label1:
    printf("Label 1\n");
```

### 应用：解释器跳转表

```c
// 虚拟机的跳转表实现
#define DISPATCH() goto *opcode_table[opcodes[pc++]]

void execute(int *opcodes, int count) {
    static void *opcode_table[] = {
        &&OP_ADD, &&OP_SUB, &&OP_MUL, &&OP_HALT
    };
    int pc = 0;
    int acc = 0;
    
    DISPATCH();
    
OP_ADD:
    acc += opcodes[pc++];
    DISPATCH();
OP_SUB:
    acc -= opcodes[pc++];
    DISPATCH();
OP_HALT:
    return;
}
```

## 可变长度宏参数

```c
// ##__VA_ARGS__ 处理零参数
#define LOG(fmt, ...) printf(fmt, ##__VA_ARGS__)

LOG("Simple message\n");      // 没有多余的逗号
LOG("Value: %d\n", 42);       // 正常工作
```

## 跨编译器兼容

```c
// 检测编译器并定义兼容宏
#if defined(__GNUC__)
    #define PACKED __attribute__((packed))
    #define ALIGNED(n) __attribute__((aligned(n)))
    #define UNUSED __attribute__((unused))
    #define NORETURN __attribute__((noreturn))
#elif defined(_MSC_VER)
    #define PACKED
    #define ALIGNED(n) __declspec(align(n))
    #define UNUSED
    #define NORETURN __declspec(noreturn)
#else
    #define PACKED
    #define ALIGNED(n)
    #define UNUSED
    #define NORETURN
#endif
```

## 重要注意事项

> **关键点总结**：
> 1. 编译器扩展提供强大功能但降低可移植性
> 2. **语句表达式** + **typeof** 可以写出安全的泛型宏
> 3. `__attribute__` 是 GCC 最重要的扩展，控制函数、变量、类型的特殊行为
> 4. `__builtin_expect` 用于分支预测优化（Linux 内核常用）
> 5. 使用条件编译隔离编译器特定代码，保持可移植性
> 6. 了解扩展有助于阅读 Linux 内核等系统级代码
