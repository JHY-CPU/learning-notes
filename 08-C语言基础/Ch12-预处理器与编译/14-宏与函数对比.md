# 14 - 宏与函数对比

## 概述

C语言提供了三种代码复用机制：宏（Macro）、内联函数（Inline Function）和普通函数（Function）。每种机制都有其适用场景和优缺点。

## 三种机制的对比

| 特性 | 宏 | 内联函数 | 普通函数 |
|------|-----|----------|----------|
| 处理阶段 | 预处理（文本替换） | 编译（可能内联展开） | 编译（独立机器码） |
| 类型检查 | 无 | 有 | 有 |
| 作用域 | 文件级 | 遵循C作用域规则 | 遵循C作用域规则 |
| 调试支持 | 困难 | 良好 | 良好 |
| 代码大小 | 每次使用都展开 | 可能内联也可能调用 | 所有调用共享同一份代码 |
| 递归 | 不支持 | 支持 | 支持 |
| 参数求值 | 可能多次 | 一次 | 一次 |
| 类型无关性 | 是 | 否 | 否 |
| 返回值 | 无（文本替换结果） | 有 | 有 |
| 标准 | C89 | C99 | C89 |

## 宏的优缺点

### 优点

```c
// 1. 类型无关
#define MAX(a, b) ((a) > (b) ? (a) : (b))
int imax = MAX(3, 5);           // int
double dmax = MAX(3.14, 2.71);  // double

// 2. 编译时常量
#define ARRAY_SIZE(arr) (sizeof(arr) / sizeof((arr)[0]))
int arr[10];
size_t n = ARRAY_SIZE(arr);  // 编译时计算为 10

// 3. 可用于任何上下文
#define BEGIN {
#define END   }
if (x > 0) BEGIN
    printf("positive\n");
END

// 4. 无函数调用开销
#define SQUARE(x) ((x) * (x))  // 直接内联，无调用
```

### 缺点

```c
// 1. 多次求值问题
#define MAX(a, b) ((a) > (b) ? (a) : (b))
int x = 5, y = 3;
int result = MAX(x++, y++);  // 灾难！x++ 和 y++ 各执行多次

// 2. 没有类型检查
#define ADD(a, b) ((a) + (b))
int r = ADD("hello", 5);  // 编译通过，运行崩溃

// 3. 调试困难
// 宏展开后的代码在调试器中看不到原始宏调用

// 4. 作用域问题
#define VALUE 100
// 全文件有效，可能与其他宏冲突
```

## 内联函数的优缺点

### 优点

```c
// 1. 类型安全
static inline int max(int a, int b) {
    return a > b ? a : b;
}
// max("hello", 5);  // 编译错误！

// 2. 参数只求值一次
static inline int max(int a, int b) {
    return a > b ? a : b;
}
int result = max(x++, y++);  // 安全！每个参数只求值一次

// 3. 遵循作用域规则
static inline int helper(int x) {
    return x * 2;  // 只在当前文件可见（static）
}

// 4. 良好的调试支持
// 可以在调试器中单步执行内联函数

// 5. 编译器可选择是否内联
// 小函数会被内联，大函数可能保留调用
```

### 缺点

```c
// 1. 需要指定类型
static inline int max_int(int a, int b) { return a > b ? a : b; }
static inline double max_double(double a, double b) { return a > b ? a : b; }
// 需要为每种类型写一个版本

// 2. C语言没有泛型
// （C11 的 _Generic 可以模拟，但比较繁琐）

// 3. 不能用于编译时常量上下文
static inline int square(int x) { return x * x; }
enum { VALUE = square(5) };  // 错误！不是编译时常量
// （某些编译器支持，但不是标准行为）
```

## 普通函数的优缺点

### 优点

```c
// 1. 代码只有一份副本
void process_data(int *arr, size_t n) {
    for (size_t i = 0; i < n; i++) {
        arr[i] = arr[i] * 2 + 1;
    }
}
// 无论调用多少次，代码只存在一份

// 2. 支持递归
int factorial(int n) {
    return n <= 1 ? 1 : n * factorial(n - 1);
}

// 3. 可以获取函数指针
int (*func_ptr)(int, int) = max;
```

### 缺点

```c
// 1. 函数调用开销
// 参数压栈、跳转、返回 —— 对于小函数可能显著

// 2. 无法在编译时求值
int arr[factorial(5)];  // 错误！C99 VLA 除外
```

## 如何选择

### 使用宏的场景

```c
// 1. 需要类型无关
#define SWAP(a, b) do { typeof(a) _t = (a); (a) = (b); (b) = _t; } while(0)

// 2. 编译时常量
#define BUFFER_SIZE 4096
#define MAX(a, b) ((a) > (b) ? (a) : (b))  // 如果用于数组大小

// 3. 简短的表达式包装
#define ARRAY_SIZE(a) (sizeof(a) / sizeof(*(a)))

// 4. 条件编译辅助
#define LOG(msg) fprintf(stderr, "%s\n", msg)

// 5. 代码生成
#define DEFINE_LIST(type) \
    typedef struct type##_node { \
        type data; \
        struct type##_node *next; \
    } type##_list_t
```

### 使用内联函数的场景

```c
// 1. 需要类型安全
static inline size_t min_size(size_t a, size_t b) {
    return a < b ? a : b;
}

// 2. 参数可能有副作用
static inline int abs_val(int x) {
    return x < 0 ? -x : x;
}
// abs_val(x++)  // 安全，x 只自增一次

// 3. 有一定复杂度的函数
static inline double clamp(double x, double lo, double hi) {
    if (x < lo) return lo;
    if (x > hi) return hi;
    return x;
}

// 4. 需要调试
static inline int compute(int a, int b) {
    int result = a * b;  // 可以在调试器中检查中间值
    return result + 1;
}
```

### 使用普通函数的场景

```c
// 1. 函数体较大
void process_large_buffer(char *buf, size_t len) {
    // 大量代码...
}

// 2. 需要递归
int fibonacci(int n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

// 3. 需要函数指针
void register_callback(void (*cb)(int));

// 4. 只需要一份代码副本以减小二进制大小
```

## 综合比较示例

```c
// 同一功能的三种实现

// 宏版本 —— 类型无关但不安全
#define ABS_MACRO(x) ((x) < 0 ? -(x) : (x))

// 内联函数版本 —— 类型安全
static inline int abs_inline(int x) {
    return x < 0 ? -x : x;
}

// 普通函数版本 —— 有调用开销
int abs_func(int x) {
    return x < 0 ? -x : x;
}

// 使用比较
int a = -5;
printf("宏: %d\n", ABS_MACRO(a + 3));       // 展开为 ((a+3)<0?-(a+3):(a+3))
printf("内联: %d\n", abs_inline(a + 3));    // 参数只求值一次
printf("函数: %d\n", abs_func(a + 3));      // 参数只求值一次

// 宏的危险
int x = -5;
printf("宏: %d\n", ABS_MACRO(x++));   // 未定义行为！x++ 多次执行
int y = -5;
printf("内联: %d\n", abs_inline(y++)); // 安全，y 只自增一次
```

## 性能考量

```c
// 现代编译器非常智能，性能差异可能很小

// 即使使用普通函数，编译器也可能自动内联
int add(int a, int b) { return a + b; }  // 编译器很可能自动内联

// 使用 -O2 或 -O3 优化后：
// - 小函数通常被自动内联
// - 宏和内联函数的性能差异很小
// - 优先考虑代码安全性和可维护性
```

## 重要注意事项

> **关键点总结**：
> 1. **安全性优先**：能用内联函数就不用宏，能用函数就不用内联函数
> 2. 宏的**唯一不可替代优势**是类型无关性和编译时求值
> 3. `ARRAY_SIZE`、`SWAP` 等宏是合理使用宏的例子
> 4. 现代编译器的优化使得性能差异不再是选择的主要因素
> 5. `static inline` 函数在大多数场景下是最佳选择
> 6. 对于大函数或需要函数指针的场景，使用普通函数
