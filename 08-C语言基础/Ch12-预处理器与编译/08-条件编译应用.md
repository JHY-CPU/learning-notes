# 08 - 条件编译应用

## 条件编译的实际应用场景

条件编译不仅仅是简单的 `#ifdef`，它在实际项目中有大量高级应用场景。本章将介绍几个最常用的模式。

## 平台适配

### 操作系统检测

```c
// platform.h — 统一的平台检测头文件
#ifndef PLATFORM_H
#define PLATFORM_H

// 操作系统检测
#if defined(_WIN32) || defined(_WIN64) || defined(__CYGWIN__)
    #define OS_WINDOWS
    #define OS_NAME "Windows"
    #define PATH_SEP '\\'
    #define LINE_SEP "\r\n"
#elif defined(__linux__)
    #define OS_LINUX
    #define OS_NAME "Linux"
    #define PATH_SEP '/'
    #define LINE_SEP "\n"
#elif defined(__APPLE__) && defined(__MACH__)
    #define OS_MACOS
    #define OS_NAME "macOS"
    #define PATH_SEP '/'
    #define LINE_SEP "\n"
#elif defined(__unix__)
    #define OS_UNIX
    #define OS_NAME "Unix"
    #define PATH_SEP '/'
    #define LINE_SEP "\n"
#else
    #define OS_UNKNOWN
    #define OS_NAME "Unknown"
#endif

// 架构检测
#if defined(__x86_64__) || defined(_M_X64)
    #define ARCH_64BIT
    #define ARCH_NAME "x86_64"
#elif defined(__i386__) || defined(_M_IX86)
    #define ARCH_32BIT
    #define ARCH_NAME "x86"
#elif defined(__aarch64__)
    #define ARCH_64BIT
    #define ARCH_NAME "ARM64"
#elif defined(__arm__)
    #define ARCH_32BIT
    #define ARCH_NAME "ARM"
#endif

#endif /* PLATFORM_H */
```

### 跨平台文件操作

```c
// file_utils.c
#include "platform.h"
#include <stdio.h>

#ifdef OS_WINDOWS
    #include <windows.h>
    #include <direct.h>
    #define MKDIR(path) _mkdir(path)
    #define GETCWD(buf, size) _getcwd(buf, size)
#else
    #include <sys/stat.h>
    #include <unistd.h>
    #define MKDIR(path) mkdir(path, 0755)
    #define GETCWD(buf, size) getcwd(buf, size)
#endif

int create_directory(const char *path) {
    return MKDIR(path);
}

void print_current_dir(void) {
    char buffer[1024];
    if (GETCWD(buffer, sizeof(buffer))) {
        printf("当前目录: %s\n", buffer);
    }
}
```

### 跨平台线程

```c
// thread.h
#ifndef THREAD_H
#define THREAD_H

#include "platform.h"

#ifdef OS_WINDOWS
    #include <windows.h>
    typedef HANDLE thread_t;
    typedef DWORD (WINAPI *thread_func_t)(LPVOID);
    #define THREAD_CREATE(thread, func, arg) \
        *(thread) = CreateThread(NULL, 0, (func), (arg), 0, NULL)
    #define THREAD_JOIN(thread) WaitForSingleObject(*(thread), INFINITE)
#else
    #include <pthread.h>
    typedef pthread_t thread_t;
    typedef void *(*thread_func_t)(void *);
    #define THREAD_CREATE(thread, func, arg) \
        pthread_create((thread), NULL, (func), (arg))
    #define THREAD_JOIN(thread) pthread_join(*(thread), NULL)
#endif

#endif /* THREAD_H */
```

## 调试开关

### 多级调试系统

```c
// debug.h
#ifndef DEBUG_H
#define DEBUG_H

#include <stdio.h>
#include <stdarg.h>

// 调试级别定义
#define DEBUG_OFF    0
#define DEBUG_ERROR  1
#define DEBUG_WARN   2
#define DEBUG_INFO   3
#define DEBUG_DEBUG  4
#define DEBUG_TRACE  5

// 设置调试级别（可通过编译选项覆盖）
#ifndef DEBUG_LEVEL
    #ifdef NDEBUG
        #define DEBUG_LEVEL DEBUG_ERROR
    #else
        #define DEBUG_LEVEL DEBUG_DEBUG
    #endif
#endif

// 调试输出宏
#if DEBUG_LEVEL >= DEBUG_ERROR
    #define LOG_ERROR(fmt, ...) \
        fprintf(stderr, "[ERROR] %s:%d: " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)
#else
    #define LOG_ERROR(fmt, ...)
#endif

#if DEBUG_LEVEL >= DEBUG_WARN
    #define LOG_WARN(fmt, ...) \
        fprintf(stderr, "[WARN]  %s:%d: " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)
#else
    #define LOG_WARN(fmt, ...)
#endif

#if DEBUG_LEVEL >= DEBUG_INFO
    #define LOG_INFO(fmt, ...) \
        fprintf(stderr, "[INFO]  " fmt "\n", ##__VA_ARGS__)
#else
    #define LOG_INFO(fmt, ...)
#endif

#if DEBUG_LEVEL >= DEBUG_DEBUG
    #define LOG_DEBUG(fmt, ...) \
        fprintf(stderr, "[DEBUG] %s:%d: " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)
#else
    #define LOG_DEBUG(fmt, ...)
#endif

#endif /* DEBUG_H */
```

### 编译使用

```bash
# 发布版本：只输出错误
gcc -DNDEBUG main.c -o main

# 开发版本：输出所有级别
gcc -DDEBUG_LEVEL=5 main.c -o main

# 只输出警告和错误
gcc -DDEBUG_LEVEL=2 main.c -o main
```

## 功能开关

### 编译时功能开关

```c
// features.h
#ifndef FEATURES_H
#define FEATURES_H

// 功能开关定义（可通过 -D 编译选项覆盖）
#ifndef FEATURE_NETWORKING
    #define FEATURE_NETWORKING 0
#endif

#ifndef FEATURE_ENCRYPTION
    #define FEATURE_ENCRYPTION 0
#endif

#ifndef FEATURE_COMPRESSION
    #define FEATURE_COMPRESSION 0
#endif

#ifndef FEATURE_GUI
    #define FEATURE_GUI 0
#endif

// 根据功能开关包含相应头文件
#if FEATURE_NETWORKING
    #include "network.h"
    #define NET_CALL(expr) expr
#else
    #define NET_CALL(expr) \
        ({ fprintf(stderr, "网络功能未启用\n"); -1; })
#endif

#if FEATURE_ENCRYPTION
    #include "crypto.h"
#endif

#endif /* FEATURES_H */
```

```bash
# 编译时启用特定功能
gcc -DFEATURE_NETWORKING=1 -DFEATURE_ENCRYPTION=1 main.c -o main
```

## 版本控制

### API 版本适配

```c
// version_compat.h
#ifndef VERSION_COMPAT_H
#define VERSION_COMPAT_H

// 根据 C 标准版本选择实现
#if __STDC_VERSION__ >= 201112L
    // C11: 使用 _Generic
    #define TYPE_NAME(x) _Generic((x), \
        int: "int", \
        float: "float", \
        double: "double", \
        char*: "string", \
        default: "unknown" \
    )
#elif __STDC_VERSION__ >= 199901L
    // C99: 使用 typeof (GCC扩展)
    #define TYPE_NAME(x) "type_info_unavailable"
#else
    // C89: 没有类型推断
    #define TYPE_NAME(x) "type_info_unavailable"
#endif

#endif /* VERSION_COMPAT_H */
```

### 库版本检测

// 检测 OpenSSL 版本

## 编译器兼容层

```c
// compiler_compat.h
#ifndef COMPILER_COMPAT_H
#define COMPILER_COMPAT_H

// 静态断言兼容
#if __STDC_VERSION__ >= 201112L
    #define STATIC_ASSERT(cond, msg) _Static_assert(cond, msg)
#else
    #define STATIC_ASSERT(cond, msg) \
        typedef char static_assert_##msg[(cond) ? 1 : -1]
#endif

// 内联函数兼容
#if __STDC_VERSION__ >= 199901L
    #define INLINE static inline
#elif defined(__GNUC__)
    #define INLINE static __inline__
#else
    #define INLINE static
#endif

// 弃用属性
#if defined(__GNUC__)
    #define DEPRECATED(msg) __attribute__((deprecated(msg)))
#elif defined(_MSC_VER)
    #define DEPRECATED(msg) __declspec(deprecated(msg))
#else
    #define DEPRECATED(msg)
#endif

// 对齐
#if __STDC_VERSION__ >= 201112L
    #define ALIGNED(n) _Alignas(n)
#elif defined(__GNUC__)
    #define ALIGNED(n) __attribute__((aligned(n)))
#elif defined(_MSC_VER)
    #define ALIGNED(n) __declspec(align(n))
#else
    #define ALIGNED(n)
#endif

#endif /* COMPILER_COMPAT_H */
```

## 运行时检测与编译时检测的结合

```c
// 混合使用编译时和运行时检测
void process(void) {
    // 编译时：选择代码路径
    #ifdef FAST_PATH
        // 优化版本（跳过某些检查）
        fast_process();
    #else
        // 安全版本（包含所有检查）
        safe_process();
    #endif
    
    // 运行时：根据环境变量调整行为
    const char *level = getenv("LOG_LEVEL");
    if (level) {
        printf("日志级别: %s\n", level);
    }
}
```

## 重要注意事项

> **关键点总结**：
> 1. 平台检测是条件编译最广泛的应用，统一管理平台宏是良好实践
> 2. 调试开关应该分级控制，在发布版本中消除所有调试代码
> 3. 功能开关允许在编译时裁剪不需要的功能，减小二进制大小
> 4. 使用 `-D` 编译选项可以覆盖宏定义，灵活控制编译行为
> 5. 保持条件编译代码的可读性——嵌套不宜过深
> 6. 编译器兼容层是跨编译器项目的重要工具
