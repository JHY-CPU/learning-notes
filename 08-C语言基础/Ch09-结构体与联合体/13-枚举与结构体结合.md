# 13 - 枚举与结构体结合

## 一、枚举作为结构体成员

枚举类型经常与结构体搭配使用，为结构体成员提供语义化的常量值。

```c
#include <stdio.h>

// 定义颜色枚举
typedef enum {
    COLOR_RED,
    COLOR_GREEN,
    COLOR_BLUE,
    COLOR_YELLOW,
    COLOR_WHITE
} Color;

// 定义形状类型枚举
typedef enum {
    SHAPE_CIRCLE,
    SHAPE_RECTANGLE,
    SHAPE_TRIANGLE
} ShapeType;

// 使用枚举的结构体
typedef struct {
    ShapeType type;
    Color     color;
    union {
        struct { float radius; } circle;
        struct { float width; float height; } rectangle;
        struct { float a, b, c; } triangle;
    } data;
} Shape;
```

## 二、状态机实现

状态机是枚举与结构体结合的经典应用场景。

### 2.1 简单状态机

```c
#include <stdio.h>

typedef enum {
    STATE_IDLE,
    STATE_RUNNING,
    STATE_PAUSED,
    STATE_STOPPED,
    STATE_ERROR
} TaskState;

typedef enum {
    EVENT_START,
    EVENT_PAUSE,
    EVENT_RESUME,
    EVENT_STOP,
    EVENT_ERROR,
    EVENT_RESET
} TaskEvent;

typedef struct {
    TaskState state;
    int       progress;    // 0~100
    char      name[64];
} Task;

// 状态转移表
TaskState transition(TaskState current, TaskEvent event) {
    switch (current) {
        case STATE_IDLE:
            if (event == EVENT_START) return STATE_RUNNING;
            break;
        case STATE_RUNNING:
            if (event == EVENT_PAUSE)  return STATE_PAUSED;
            if (event == EVENT_STOP)   return STATE_STOPPED;
            if (event == EVENT_ERROR)  return STATE_ERROR;
            break;
        case STATE_PAUSED:
            if (event == EVENT_RESUME) return STATE_RUNNING;
            if (event == EVENT_STOP)   return STATE_STOPPED;
            break;
        case STATE_ERROR:
            if (event == EVENT_RESET)  return STATE_IDLE;
            break;
        case STATE_STOPPED:
            if (event == EVENT_RESET)  return STATE_IDLE;
            break;
    }
    return current;  // 无效转移，保持当前状态
}

const char *state_name(TaskState s) {
    switch (s) {
        case STATE_IDLE:    return "空闲";
        case STATE_RUNNING: return "运行中";
        case STATE_PAUSED:  return "已暂停";
        case STATE_STOPPED: return "已停止";
        case STATE_ERROR:   return "错误";
    }
    return "未知";
}

int main(void) {
    Task task = {STATE_IDLE, 0, "数据处理"};

    printf("初始状态: %s\n", state_name(task.state));

    task.state = transition(task.state, EVENT_START);
    printf("START 后: %s\n", state_name(task.state));

    task.state = transition(task.state, EVENT_PAUSE);
    printf("PAUSE 后: %s\n", state_name(task.state));

    task.state = transition(task.state, EVENT_RESUME);
    printf("RESUME后: %s\n", state_name(task.state));

    task.state = transition(task.state, EVENT_STOP);
    printf("STOP 后: %s\n", state_name(task.state));

    return 0;
}
```

### 2.2 带动作的状态机

```c
typedef struct {
    TaskState state;
    int       count;

    // 状态进入时的回调
    void (*on_enter)(TaskState from, TaskState to);
    // 状态退出时的回调
    void (*on_exit)(TaskState from, TaskState to);
} StateMachine;

void on_enter_handler(TaskState from, TaskState to) {
    printf("状态转移: %d -> %d\n", from, to);
}

int sm_send_event(StateMachine *sm, TaskEvent event) {
    TaskState next = transition(sm->state, event);
    if (next != sm->state) {
        if (sm->on_exit) sm->on_exit(sm->state, next);
        TaskState old = sm->state;
        sm->state = next;
        if (sm->on_enter) sm->on_enter(old, next);
        return 1;  // 状态变化
    }
    return 0;  // 无变化
}
```

## 三、错误码与错误处理

```c
typedef enum {
    ERR_OK = 0,
    ERR_NULL_POINTER,
    ERR_INVALID_PARAM,
    ERR_OUT_OF_MEMORY,
    ERR_FILE_NOT_FOUND,
    ERR_PERMISSION_DENIED,
    ERR_TIMEOUT,
    ERR_UNKNOWN
} ErrorCode;

typedef struct {
    ErrorCode code;
    char      message[256];
    int       line;       // 错误发生行号
    char      file[64];   // 错误发生文件名
} ErrorInfo;

// 错误创建宏
#define MAKE_ERROR(c, msg) \
    (ErrorInfo){.code = (c), .message = msg, .line = __LINE__, .file = __FILE__}

// 使用
ErrorCode process_file(const char *path) {
    if (path == NULL) {
        ErrorInfo err = MAKE_ERROR(ERR_NULL_POINTER, "文件路径为空");
        printf("[%s:%d] 错误: %s\n", err.file, err.line, err.message);
        return err.code;
    }
    // ... 处理 ...
    return ERR_OK;
}
```

## 四、枚举标记联合体类型

这是最经典的组合模式：

```c
typedef enum {
    NODE_INT,
    NODE_FLOAT,
    NODE_STRING
} NodeType;

typedef struct {
    NodeType type;
    union {
        int    int_val;
        float  float_val;
        char   str_val[64];
    };
} TypedNode;

// 创建函数
TypedNode make_int_node(int val) {
    return (TypedNode){.type = NODE_INT, .int_val = val};
}

TypedNode make_float_node(float val) {
    return (TypedNode){.type = NODE_FLOAT, .float_val = val};
}

// 类型安全的打印
void print_node(const TypedNode *node) {
    switch (node->type) {
        case NODE_INT:    printf("INT: %d\n", node->int_val); break;
        case NODE_FLOAT:  printf("FLOAT: %.2f\n", node->float_val); break;
        case NODE_STRING: printf("STRING: %s\n", node->str_val); break;
    }
}

// 类型安全的比较
int nodes_equal(const TypedNode *a, const TypedNode *b) {
    if (a->type != b->type) return 0;
    switch (a->type) {
        case NODE_INT:    return a->int_val == b->int_val;
        case NODE_FLOAT:  return a->float_val == b->float_val;
        case NODE_STRING: return strcmp(a->str_val, b->str_val) == 0;
    }
    return 0;
}
```

## 五、命令模式

```c
typedef enum {
    CMD_OPEN_FILE,
    CMD_CLOSE_FILE,
    CMD_READ_DATA,
    CMD_WRITE_DATA,
    CMD_SEEK_POSITION
} CommandType;

typedef struct {
    CommandType type;
    union {
        struct { char path[256]; } open;
        struct { int fd; } close;
        struct { int fd; size_t size; } read;
        struct { int fd; const void *buf; size_t size; } write;
        struct { int fd; long offset; int whence; } seek;
    } params;
} Command;

// 命令执行
int execute_command(const Command *cmd) {
    switch (cmd->type) {
        case CMD_OPEN_FILE:
            printf("打开文件: %s\n", cmd->params.open.path);
            break;
        case CMD_CLOSE_FILE:
            printf("关闭文件: fd=%d\n", cmd->params.close.fd);
            break;
        case CMD_READ_DATA:
            printf("读取 %zu 字节, fd=%d\n",
                   cmd->params.read.size, cmd->params.read.fd);
            break;
        // ...
    }
    return 0;
}
```

## 六、关键要点

> **重要提示：**
>
> 1. 枚举为结构体成员提供语义化、类型安全的常量值。
> 2. 状态机是枚举与结构体结合最经典的应用场景。
> 3. 枚举标记联合体类型是 C 语言中模拟"变体类型"的标准做法。
> 4. 使用枚举代替裸整数，提高代码可读性和可维护性。
> 5. 错误码枚举配合错误信息结构体，构建完善的错误处理体系。
> 6. 枚举 + 联合体 + 结构体的组合，是 C 语言模拟面向对象的重要基础。
