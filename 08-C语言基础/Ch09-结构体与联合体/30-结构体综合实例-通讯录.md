# 30 - 结构体综合实例 - 通讯录

## 一、系统设计

一个完整的通讯录程序，支持增删改查、排序、文件存储等功能。

### 数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_NAME_LEN   32
#define MAX_PHONE_LEN  20
#define MAX_EMAIL_LEN  64
#define MAX_ADDR_LEN   128
#define MAX_CONTACTS   1000

typedef enum {
    GROUP_FAMILY,
    GROUP_FRIEND,
    GROUP_WORK,
    GROUP_OTHER
} ContactGroup;

typedef struct {
    int          id;
    char         name[MAX_NAME_LEN];
    char         phone[MAX_PHONE_LEN];
    char         email[MAX_EMAIL_LEN];
    char         address[MAX_ADDR_LEN];
    ContactGroup group;
    int          favorite;    // 是否收藏
} Contact;

typedef struct {
    Contact *contacts;
    int      count;
    int      capacity;
    int      next_id;
} AddressBook;
```

## 二、核心操作

### 2.1 初始化与销毁

```c
AddressBook *book_create(void) {
    AddressBook *book = malloc(sizeof(AddressBook));
    if (book == NULL) return NULL;

    book->capacity = 20;
    book->count = 0;
    book->next_id = 1;
    book->contacts = malloc(sizeof(Contact) * book->capacity);
    if (book->contacts == NULL) {
        free(book);
        return NULL;
    }
    return book;
}

void book_destroy(AddressBook *book) {
    if (book == NULL) return;
    free(book->contacts);
    free(book);
}
```

### 2.2 添加联系人

```c
static int book_ensure_cap(AddressBook *book) {
    if (book->count < book->capacity) return 0;
    int new_cap = book->capacity * 2;
    Contact *new_arr = realloc(book->contacts, sizeof(Contact) * new_cap);
    if (new_arr == NULL) return -1;
    book->contacts = new_arr;
    book->capacity = new_cap;
    return 0;
}

int book_add(AddressBook *book,
             const char *name, const char *phone,
             const char *email, const char *address,
             ContactGroup group) {
    if (book_ensure_cap(book) != 0) return -1;

    Contact *c = &book->contacts[book->count];
    memset(c, 0, sizeof(Contact));

    c->id = book->next_id++;
    strncpy(c->name, name, MAX_NAME_LEN - 1);
    strncpy(c->phone, phone, MAX_PHONE_LEN - 1);
    if (email) strncpy(c->email, email, MAX_EMAIL_LEN - 1);
    if (address) strncpy(c->address, address, MAX_ADDR_LEN - 1);
    c->group = group;
    c->favorite = 0;

    book->count++;
    return c->id;
}
```

### 2.3 查找联系人

```c
// 按 ID 查找
Contact *book_find_by_id(const AddressBook *book, int id) {
    for (int i = 0; i < book->count; i++) {
        if (book->contacts[i].id == id)
            return &book->contacts[i];
    }
    return NULL;
}

// 按姓名模糊查找
int book_search_name(const AddressBook *book, const char *keyword,
                     Contact **results, int max_results) {
    int found = 0;
    for (int i = 0; i < book->count && found < max_results; i++) {
        if (strstr(book->contacts[i].name, keyword) != NULL) {
            results[found++] = &book->contacts[i];
        }
    }
    return found;
}

// 按手机号查找
Contact *book_find_by_phone(const AddressBook *book, const char *phone) {
    for (int i = 0; i < book->count; i++) {
        if (strcmp(book->contacts[i].phone, phone) == 0)
            return &book->contacts[i];
    }
    return NULL;
}
```

### 2.4 修改与删除

```c
// 修改联系人信息
int book_update(AddressBook *book, int id,
                const char *name, const char *phone,
                const char *email, const char *address) {
    Contact *c = book_find_by_id(book, id);
    if (c == NULL) return -1;

    if (name)    strncpy(c->name, name, MAX_NAME_LEN - 1);
    if (phone)   strncpy(c->phone, phone, MAX_PHONE_LEN - 1);
    if (email)   strncpy(c->email, email, MAX_EMAIL_LEN - 1);
    if (address) strncpy(c->address, address, MAX_ADDR_LEN - 1);
    return 0;
}

// 删除联系人
int book_delete(AddressBook *book, int id) {
    for (int i = 0; i < book->count; i++) {
        if (book->contacts[i].id == id) {
            // 将后面的元素前移
            memmove(&book->contacts[i], &book->contacts[i + 1],
                    (book->count - i - 1) * sizeof(Contact));
            book->count--;
            return 0;
        }
    }
    return -1;
}

// 切换收藏状态
int book_toggle_favorite(AddressBook *book, int id) {
    Contact *c = book_find_by_id(book, id);
    if (c == NULL) return -1;
    c->favorite = !c->favorite;
    return 0;
}
```

## 三、排序功能

```c
#include <ctype.h>

static int cmp_name(const void *a, const void *b) {
    return strcmp(((const Contact *)a)->name,
                  ((const Contact *)b)->name);
}

static int cmp_group(const void *a, const void *b) {
    int diff = ((const Contact *)a)->group -
               ((const Contact *)b)->group;
    if (diff != 0) return diff;
    return cmp_name(a, b);  // 同组内按姓名排序
}

static int cmp_favorite(const void *a, const void *b) {
    int fa = ((const Contact *)a)->favorite;
    int fb = ((const Contact *)b)->favorite;
    if (fa != fb) return fb - fa;  // 收藏的排前面
    return cmp_name(a, b);
}

void book_sort(AddressBook *book, int by_name, int by_group, int by_fav) {
    if (by_fav) {
        qsort(book->contacts, book->count, sizeof(Contact), cmp_favorite);
    } else if (by_group) {
        qsort(book->contacts, book->count, sizeof(Contact), cmp_group);
    } else {
        qsort(book->contacts, book->count, sizeof(Contact), cmp_name);
    }
}
```

## 四、分组统计

```c
const char *group_name(ContactGroup g) {
    switch (g) {
        case GROUP_FAMILY: return "家人";
        case GROUP_FRIEND: return "朋友";
        case GROUP_WORK:   return "工作";
        case GROUP_OTHER:  return "其他";
    }
    return "未知";
}

void book_statistics(const AddressBook *book) {
    int group_count[4] = {0};
    int fav_count = 0;

    for (int i = 0; i < book->count; i++) {
        group_count[book->contacts[i].group]++;
        if (book->contacts[i].favorite) fav_count++;
    }

    printf("=== 通讯录统计 ===\n");
    printf("总联系人数: %d\n", book->count);
    printf("收藏联系人: %d\n", fav_count);
    printf("分组统计:\n");
    for (int i = 0; i < 4; i++) {
        printf("  %s: %d 人\n", group_name((ContactGroup)i), group_count[i]);
    }
}
```

## 五、文件持久化

```c
int book_save(const AddressBook *book, const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (fp == NULL) return -1;

    fprintf(fp, "%d\n", book->count);
    for (int i = 0; i < book->count; i++) {
        const Contact *c = &book->contacts[i];
        fprintf(fp, "%d|%s|%s|%s|%s|%d|%d\n",
                c->id, c->name, c->phone, c->email,
                c->address, c->group, c->favorite);
    }
    fclose(fp);
    return 0;
}

AddressBook *book_load(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (fp == NULL) return NULL;

    AddressBook *book = book_create();
    if (book == NULL) { fclose(fp); return NULL; }

    int count;
    fscanf(fp, "%d\n", &count);

    for (int i = 0; i < count; i++) {
        int id, group, fav;
        char name[MAX_NAME_LEN], phone[MAX_PHONE_LEN];
        char email[MAX_EMAIL_LEN], address[MAX_ADDR_LEN];

        fscanf(fp, "%d|%31[^|]|%19[^|]|%63[^|]|%127[^|]|%d|%d\n",
               &id, name, phone, email, address, &group, &fav);

        int new_id = book_add(book, name, phone, email, address,
                              (ContactGroup)group);
        if (new_id > 0) {
            Contact *c = book_find_by_id(book, new_id);
            if (c) c->favorite = fav;
        }
    }

    // 恢复 next_id
    if (book->count > 0) {
        int max_id = 0;
        for (int i = 0; i < book->count; i++) {
            if (book->contacts[i].id > max_id)
                max_id = book->contacts[i].id;
        }
        book->next_id = max_id + 1;
    }

    fclose(fp);
    return book;
}
```

## 六、打印与显示

```c
void book_print_contact(const Contact *c) {
    printf("┌──────────────────────────────┐\n");
    printf("│ ID: %-6d  %s %s│\n", c->id, c->name,
           c->favorite ? "★" : " ");
    printf("│ 电话: %-24s│\n", c->phone);
    if (c->email[0])
        printf("│ 邮箱: %-24s│\n", c->email);
    if (c->address[0])
        printf("│ 地址: %-24s│\n", c->address);
    printf("│ 分组: %-24s│\n", group_name(c->group));
    printf("└──────────────────────────────┘\n");
}

void book_print_all(const AddressBook *book) {
    printf("\n========== 通讯录 (%d 人) ==========\n\n", book->count);
    for (int i = 0; i < book->count; i++) {
        book_print_contact(&book->contacts[i]);
    }
}
```

```c
    book_destroy(book);
    return 0;
}
```

## 八、关键要点

> **重要提示：**
>
> 1. 枚举（ContactGroup）使分组管理更语义化。
> 2. `strstr` 实现模糊搜索，`strcmp` 实现精确匹配。
> 3. `memmove` 用于删除时移动数组元素（比逐个赋值高效）。
> 4. 动态数组（realloc）使通讯录容量可增长。
> 5. 文件格式使用 `|` 分隔符，方便解析含逗号的字段。
> 6. 本实例综合运用了结构体、枚举、动态内存、文件 I/O、排序、搜索等核心知识。
