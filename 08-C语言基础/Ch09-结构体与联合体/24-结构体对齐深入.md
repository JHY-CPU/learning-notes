# 24 - 结构体对齐深入

## 一、C11 对齐特性

C11 引入了 `_Alignof` 和 `_Alignas` 关键字，提供了更精细的对齐控制。

### 1.1 `_Alignof` —— 查询类型的对齐要求

```c
#include <stdio.h>
#include <stdalign.h>

struct Test {
    char  a;
    int   b;
    short c;
};

int main(void) {
    printf("alignof(char)    = %zu\n", alignof(char));       // 1
    printf("alignof(short)   = %zu\n", alignof(short));      // 2
    printf("alignof(int)     = %zu\n", alignof(int));        // 4
    printf("alignof(double)  = %zu\n", alignof(double));     // 8
    printf("alignof(Test)    = %zu\n", alignof(struct Test)); // 4

    // stdalign.h 提供了宏，等价于 _Alignof
    // alignof(T) 等价于 _Alignof(T)

    return 0;
}
```

### 1.2 `_Alignas` —— 指定对齐要求

```c
#include <stdio.h>
#include <stdalign.h>

// 强制 64 字节对齐（常用于缓存行）
_Alignas(64) char cache_line_buffer[256];

// 结构体成员对齐
struct Aligned {
    _Alignas(16) int x;  // x 的地址是 16 的倍数
    int y;
};

// 等价写法（使用 stdalign.h 的宏）
struct Aligned2 {
    alignas(16) int x;
    int y;
};

int main(void) {
    struct Aligned a;
    printf("a.x 地址对齐: %zu\n",
           (size_t)&a.x % 16);  // 0

    printf("sizeof(Aligned) = %zu\n",
           sizeof(struct Aligned));  // 至少 20，实际可能是 32

    return 0;
}
```

## 二、手动填充

有时需要手动添加填充字节以满足特定的内存布局要求。

### 2.1 显式填充

```c
// 手动控制对齐，避免编译器自动填充的不确定性
struct ManualPad {
    char  a;
    char  pad1[3];  // 手动填充 3 字节
    int   b;
    short c;
    char  pad2[2];  // 手动填充 2 字节
    int   d;
};
// sizeof = 16（手动控制，跨编译器一致）
```

### 2.2 命名填充（自文档化）

```c
// 命名填充字段，明确意图
struct Register {
    uint8_t  control;
    uint8_t  _reserved0[3];   // 保留字段
    uint32_t data;
    uint16_t status;
    uint16_t _reserved1;      // 保留字段
};
```

## 三、缓存行对齐

### 3.1 缓存行基础

```
CPU 缓存以"缓存行（Cache Line）"为单位加载数据：
  x86/x64: 64 字节
  ARM: 32 或 64 字节

如果两个频繁访问的数据在同一个缓存行中，
修改一个会导致另一个的缓存失效（False Sharing）
```

### 3.2 缓存行对齐示例

```c
#include <stdalign.h>

// 未对齐：可能发生 False Sharing
struct BadCounter {
    int count_a;  // 可能和 count_b 在同一缓存行
    int count_b;
};

// 对齐：每个计数器独占一个缓存行
struct GoodCounter {
    alignas(64) int count_a;
    alignas(64) int count_b;
};

// 常见的缓存行对齐方式
struct ThreadData {
    alignas(64) long counter;  // 64 字节对齐
    char padding[64 - sizeof(long)];  // 填充到 64 字节
};
```

### 3.3 动态分配时的对齐

```c
#include <stdlib.h>

// C11 aligned_alloc
void *aligned_buf = aligned_alloc(64, 1024);  // 64 字节对齐的 1024 字节
if (aligned_buf != NULL) {
    // 使用...
    free(aligned_buf);
}

// POSIX posix_memalign
void *ptr;
if (posix_memalign(&ptr, 64, 1024) == 0) {
    // 使用...
    free(ptr);
}

// 手动对齐分配
void *raw = malloc(1024 + 63);
void *aligned = (void *)(((uintptr_t)raw + 63) & ~(uintptr_t)63);
```

## 四、跨编译器的对齐控制

### 4.1 `#pragma pack`

```c
#pragma pack(push, 1)
struct Packed1 {
    char  a;
    int   b;
    short c;
};  // sizeof = 7
#pragma pack(pop)

#pragma pack(push, 2)
struct Packed2 {
    char  a;
    int   b;
    short c;
};  // sizeof = 8
#pragma pack(pop)
```

### 4.2 GCC / Clang 属性

```c
// 指定对齐值
struct __attribute__((aligned(16))) Align16 {
    int x, y;
};

// 紧凑排列
struct __attribute__((packed)) Packed {
    char  a;
    int   b;
    short c;
};  // sizeof = 7

// 成员级对齐
struct MemberAlign {
    char a;
    int b __attribute__((aligned(8)));
};
```

### 4.3 MSVC

```c
// MSVC 特有的对齐方式
__declspec(align(16)) struct MSVCAlign {
    int x, y;
};
```

### 4.4 跨平台宏封装

```c
#if defined(_MSC_VER)
    #define ALIGNED(x) __declspec(align(x))
#elif defined(__GNUC__) || defined(__clang__)
    #define ALIGNED(x) __attribute__((aligned(x)))
#else
    #define ALIGNED(x) _Alignas(x)  // C11
#endif

ALIGNED(64) char buffer[256];
```

## 五、结构体对齐与 sizeof

```c
#include <stdio.h>
#include <stdalign.h>

struct Example {
    char     a;     // 偏移 0,  大小 1
    // 3 字节填充
    int      b;     // 偏移 4,  大小 4
    short    c;     // 偏移 8,  大小 2
    char     d;     // 偏移 10, 大小 1
    // 1 字节填充（使总大小为 4 的倍数）
};  // sizeof = 12

void analyze_struct(void) {
    struct Example s;
    printf("sizeof = %zu\n", sizeof(s));
    printf("alignof = %zu\n", alignof(struct Example));

    printf("a: offset=%zu, size=%zu\n",
           (size_t)&s.a - (size_t)&s, sizeof(s.a));
    printf("b: offset=%zu, size=%zu\n",
           (size_t)&s.b - (size_t)&s, sizeof(s.b));
    printf("c: offset=%zu, size=%zu\n",
           (size_t)&s.c - (size_t)&s, sizeof(s.c));
    printf("d: offset=%zu, size=%zu\n",
           (size_t)&s.d - (size_t)&s, sizeof(s.d));

    // 结构体末尾填充
    printf("末尾填充: %zu 字节\n",
           sizeof(s) - (sizeof(s.a) + sizeof(s.b) +
           sizeof(s.c) + sizeof(s.d)));  // 4（含 3+1 填充）
}
```

## 六、关键要点

> **重要提示：**
>
> 1. `_Alignof` 查询对齐要求，`_Alignas` 指定对齐要求（C11 标准）。
> 2. 缓存行对齐（64 字节）在多线程编程中至关重要，避免 False Sharing。
> 3. `#pragma pack` 消除填充但可能影响性能和可移植性。
> 4. 手动填充可以跨编译器保证一致的内存布局。
> 5. `aligned_alloc` / `posix_memalign` 用于动态分配对齐内存。
> 6. 使用 `offsetof` 宏可以精确获取成员在结构体中的偏移量。
