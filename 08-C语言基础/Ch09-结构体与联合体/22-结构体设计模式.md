# 22 - 结构体设计模式

## 一、用结构体模拟面向对象

C 语言没有 class，但可以用结构体 + 函数指针模拟 OOP 的核心概念。

### 1.1 封装

```c
// 模拟"类"
typedef struct {
    double x;
    double y;
} Point;

// 模拟"成员函数"
Point point_new(double x, double y) {
    return (Point){x, y};
}

double point_distance(const Point *a, const Point *b) {
    double dx = a->x - b->x;
    double dy = a->y - b->y;
    return sqrt(dx * dx + dy * dy);
}

void point_translate(Point *p, double dx, double dy) {
    p->x += dx;
    p->y += dy;
}
```

### 1.2 继承模拟（结构体嵌套）

```c
// "基类"
typedef struct {
    int x;
    int y;
} ShapeBase;

// "派生类" —— 通过将基类作为第一个成员
typedef struct {
    ShapeBase base;   // "继承"基类
    int radius;
} Circle;

typedef struct {
    ShapeBase base;
    int width;
    int height;
} Rectangle;

// 可以安全地将子类指针转换为基类指针
void print_position(const ShapeBase *s) {
    printf("位置: (%d, %d)\n", s->x, s->y);
}

// 使用
Circle c = {{10, 20}, 5};
print_position((ShapeBase *)&c);  // 安全：base 是第一个成员
```

## 二、虚函数表模拟

### 2.1 函数指针数组作为虚函数表

```c
#include <stdio.h>
#include <math.h>

// 虚函数表
typedef struct {
    double (*area)(const void *self);
    void   (*draw)(const void *self);
    void   (*destroy)(void *self);
} VTable;

// "基类"
typedef struct {
    const VTable *vptr;  // 虚函数表指针
    int x, y;
} Shape;

// 通过虚函数表调用"虚函数"
double shape_area(const Shape *s) {
    return s->vptr->area(s);
}

void shape_draw(const Shape *s) {
    s->vptr->draw(s);
}
```

### 2.2 实现具体"子类"

```c
#define PI 3.14159265358979323846

// Circle
typedef struct {
    Shape base;
    double radius;
} Circle;

static double circle_area(const void *self) {
    const Circle *c = (const Circle *)self;
    return PI * c->radius * c->radius;
}

static void circle_draw(const void *self) {
    const Circle *c = (const Circle *)self;
    printf("绘制圆形: 中心(%d,%d), 半径%.1f\n",
           c->base.x, c->base.y, c->radius);
}

static const VTable circle_vtable = {
    .area = circle_area,
    .draw = circle_draw,
    .destroy = NULL
};

Circle circle_new(int x, int y, double r) {
    return (Circle){{&circle_vtable, x, y}, r};
}

// Rectangle
typedef struct {
    Shape base;
    double width, height;
} Rectangle;

static double rect_area(const void *self) {
    const Rectangle *r = (const Rectangle *)self;
    return r->width * r->height;
}

static void rect_draw(const void *self) {
    const Rectangle *r = (const Rectangle *)self;
    printf("绘制矩形: 中心(%d,%d), %.1f x %.1f\n",
           r->base.x, r->base.y, r->width, r->height);
}

static const VTable rect_vtable = {
    .area = rect_area,
    .draw = rect_draw,
    .destroy = NULL
};

Rectangle rect_new(int x, int y, double w, double h) {
    return (Rectangle){{&rect_vtable, x, y}, w, h};
}
```

### 2.3 多态调用

```c
int main(void) {
    Circle c = circle_new(0, 0, 5.0);
    Rectangle r = rect_new(10, 10, 3.0, 4.0);

    // 多态：统一通过基类指针调用
    Shape *shapes[] = {(Shape *)&c, (Shape *)&r};

    for (int i = 0; i < 2; i++) {
        shape_draw(shapes[i]);
        printf("面积: %.2f\n\n", shape_area(shapes[i]));
    }

    return 0;
}
```

## 三、单例模式

```c
typedef struct {
    int    initialized;
    char   config_path[256];
    int    log_level;
} AppConfig;

static AppConfig *instance = NULL;

AppConfig *app_config_get(void) {
    if (instance == NULL) {
        instance = malloc(sizeof(AppConfig));
        if (instance != NULL) {
            instance->initialized = 1;
            instance->log_level = 1;
            strcpy(instance->config_path, "/etc/app.conf");
        }
    }
    return instance;
}

void app_config_destroy(void) {
    free(instance);
    instance = NULL;
}
```

## 四、策略模式

```c
// 排序策略
typedef int (*CompareFunc)(const void *, const void *);

typedef struct {
    void        *data;
    size_t       elem_size;
    size_t       count;
    CompareFunc  compare;
} SortableArray;

void sortable_sort(SortableArray *arr) {
    qsort(arr->data, arr->count, arr->elem_size, arr->compare);
}

// 不同的比较策略
int int_asc(const void *a, const void *b) {
    return *(const int *)a - *(const int *)b;
}

int int_desc(const void *a, const void *b) {
    return *(const int *)b - *(const int *)a;
}
```

## 五、关键要点

> **重要提示：**
>
> 1. 结构体 + 函数指针是 C 语言模拟 OOP 的基础工具。
> 2. 虚函数表（vtable）模式通过函数指针实现运行时多态。
> 3. 将基类结构体作为子类的**第一个成员**，确保指针转换安全。
> 4. C 语言的 OOP 模拟是理解 C++ 对象模型的绝佳方式。
> 5. Linux 内核中大量使用这些模式（如 `file_operations`、`net_device`）。
> 6. 设计模式在 C 中实现时要权衡复杂度——有时简单直接的代码更好。
