# 08 - 结构体与函数

## 一、结构体作为函数参数

### 1.1 值传递

将整个结构体**拷贝**一份传给函数，函数内修改不影响原始数据。

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char name[32];
    float score;
} Student;

// 值传递：函数获得结构体的副本
void print_student(Student s) {
    printf("学号: %d, 姓名: %s, 成绩: %.1f\n",
           s.id, s.name, s.score);

    // 修改只影响副本
    s.score = 0;
    printf("函数内修改后: %.1f\n", s.score);
}

int main(void) {
    Student s1 = {1001, "张三", 92.5};
    print_student(s1);
    printf("原始成绩: %.1f\n", s1.score);  // 仍然是 92.5
    return 0;
}
```

### 1.2 指针传递

传递结构体的**地址**，函数内可修改原始数据，且效率更高。

```c
// 指针传递：只传 4/8 字节地址
void modify_score(Student *s, float new_score) {
    if (s == NULL) return;
    s->score = new_score;
}

// const 指针：保证不修改（推荐用于只读函数）
void print_student_const(const Student *s) {
    printf("学号: %d, 姓名: %s, 成绩: %.1f\n",
           s->id, s->name, s->score);
    // s->score = 0;  // 编译错误：const 保护
}

int main(void) {
    Student s1 = {1001, "张三", 92.5};
    modify_score(&s1, 98.0);
    print_student_const(&s1);
    return 0;
}
```

### 1.3 值传递 vs 指针传递 对比

| 特性 | 值传递 | 指针传递 |
|------|--------|----------|
| 参数大小 | sizeof(Struct) | 4 或 8 字节 |
| 调用开销 | 高（需拷贝整个结构体） | 低（只传地址） |
| 能否修改原数据 | 不能 | 能 |
| 安全性 | 高（不影响原数据） | 需要 const 保护 |
| 推荐 | 小结构体 (< 16B) | 大结构体 |

> **经验法则**：结构体大于 16 字节时，优先使用指针传递。

## 二、函数返回结构体

### 2.1 返回结构体值

```c
// 返回结构体值
Student create_student(int id, const char *name, float score) {
    Student s;
    s.id = id;
    strncpy(s.name, name, sizeof(s.name) - 1);
    s.name[sizeof(s.name) - 1] = '\0';
    s.score = score;
    return s;
}

// 使用
Student s1 = create_student(1001, "张三", 92.5);
```

### 2.2 返回结构体指针（动态分配）

```c
#include <stdlib.h>

// 返回动态分配的结构体指针
Student *create_student_ptr(int id, const char *name, float score) {
    Student *s = malloc(sizeof(Student));
    if (s == NULL) return NULL;

    s->id = id;
    strncpy(s->name, name, sizeof(s->name) - 1);
    s->name[sizeof(s->name) - 1] = '\0';
    s->score = score;
    return s;
}

// 使用
Student *p = create_student_ptr(1002, "李四", 88.0);
if (p != NULL) {
    printf("%s: %.1f\n", p->name, p->score);
    free(p);  // 调用者负责释放
}
```

### 2.3 通过输出参数返回

```c
// 通过指针参数"返回"结果，返回值用于指示成功/失败
int create_student_out(Student *out, int id, const char *name, float score) {
    if (out == NULL) return -1;

    out->id = id;
    strncpy(out->name, name, sizeof(out->name) - 1);
    out->name[sizeof(out->name) - 1] = '\0';
    out->score = score;
    return 0;  // 成功
}

// 使用
Student s1;
if (create_student_out(&s1, 1003, "王五", 95.0) == 0) {
    printf("创建成功: %s\n", s1.name);
}
```

> **注意**：不要返回指向**局部结构体变量**的指针，因为函数返回后局部变量被销毁。

```c
// 错误示例！返回了悬垂指针
Student *bad_func(void) {
    Student s = {1001, "张三", 92.5};
    return &s;  // 危险！s 在函数返回后被销毁
}
```

## 三、结构体数组作为函数参数

```c
// 传递数组时必须同时传递大小
void print_all(const Student arr[], int n) {
    for (int i = 0; i < n; i++) {
        printf("%d. %s - %.1f\n", i + 1, arr[i].name, arr[i].score);
    }
}

// 查找最高分学生
Student *find_top(Student arr[], int n) {
    if (n <= 0) return NULL;
    int max_idx = 0;
    for (int i = 1; i < n; i++) {
        if (arr[i].score > arr[max_idx].score) {
            max_idx = i;
        }
    }
    return &arr[max_idx];
}

// 使用
Student class[] = {
    {1001, "张三", 92.5},
    {1002, "李四", 88.0},
    {1003, "王五", 95.5}
};
print_all(class, 3);

Student *top = find_top(class, 3);
if (top != NULL) {
    printf("最高分: %s (%.1f)\n", top->name, top->score);
}
```

## 四、完整示例：学生成绩管理函数集

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    int id;
    char name[32];
    float scores[3];  // 3 门课
} Student;

// 计算平均分
float average(const Student *s) {
    return (s->scores[0] + s->scores[1] + s->scores[2]) / 3.0f;
}

// 判断是否及格
int all_pass(const Student *s, float pass_line) {
    return s->scores[0] >= pass_line &&
           s->scores[1] >= pass_line &&
           s->scores[2] >= pass_line;
}

// 比较函数（用于排序）
int compare_avg(const void *a, const void *b) {
    float avg_a = average((const Student *)a);
    float avg_b = average((const Student *)b);
    if (avg_a < avg_b) return 1;
    if (avg_a > avg_b) return -1;
    return 0;
}

// 打印学生信息
void print_student(const Student *s) {
    printf("学号: %d, 姓名: %s, 平均分: %.1f, %s\n",
           s->id, s->name, average(s),
           all_pass(s, 60.0) ? "全部及格" : "有不及格");
}

int main(void) {
    Student class[] = {
        {1001, "张三", {90, 85, 78}},
        {1002, "李四", {55, 92, 88}},
        {1003, "王五", {95, 90, 85}},
    };
    int n = sizeof(class) / sizeof(class[0]);

    printf("=== 排序前 ===\n");
    for (int i = 0; i < n; i++) {
        print_student(&class[i]);
    }

    qsort(class, n, sizeof(Student), compare_avg);

    printf("\n=== 按平均分排序后 ===\n");
    for (int i = 0; i < n; i++) {
        print_student(&class[i]);
    }

    return 0;
}
```

## 五、关键要点

> **重要提示：**
>
> 1. 大结构体传参优先使用**指针传递**，小结构体可以使用值传递。
> 2. 只读函数参数务必使用 `const` 限定。
> 3. 不要返回局部结构体变量的地址（悬垂指针）。
> 4. 动态分配的结构体指针由调用者负责 `free`。
> 5. 结构体数组作为参数时，通常需要额外传递数组大小。
> 6. 通过输出参数返回结果是一种常见且安全的模式。
