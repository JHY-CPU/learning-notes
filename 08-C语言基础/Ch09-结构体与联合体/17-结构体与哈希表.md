# 17 - 结构体与哈希表

## 一、哈希表节点定义

哈希表（Hash Table）通过**哈希函数**将键映射到数组下标，实现 O(1) 的平均查找效率。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define TABLE_SIZE 64

// 哈希表节点（链地址法解决冲突）
typedef struct HashNode {
    char *key;
    int   value;
    struct HashNode *next;  // 冲突链
} HashNode;

// 哈希表
typedef struct {
    HashNode *buckets[TABLE_SIZE];
    int       count;
} HashTable;
```

## 二、哈希函数

```c
// DJB2 哈希函数 —— 经典高效
unsigned int hash_djb2(const char *str) {
    unsigned int hash = 5381;
    int c;
    while ((c = *str++)) {
        hash = ((hash << 5) + hash) + c;  // hash * 33 + c
    }
    return hash % TABLE_SIZE;
}

// BKDR 哈希函数
unsigned int hash_bkdr(const char *str) {
    unsigned int seed = 131;  // 31, 131, 1313, ...
    unsigned int hash = 0;
    while (*str) {
        hash = hash * seed + (*str++);
    }
    return hash % TABLE_SIZE;
}
```

## 三、哈希表操作

### 3.1 创建与销毁

```c
HashTable *ht_create(void) {
    HashTable *ht = malloc(sizeof(HashTable));
    if (ht == NULL) return NULL;
    memset(ht->buckets, 0, sizeof(ht->buckets));
    ht->count = 0;
    return ht;
}

void ht_destroy(HashTable *ht) {
    if (ht == NULL) return;
    for (int i = 0; i < TABLE_SIZE; i++) {
        HashNode *node = ht->buckets[i];
        while (node != NULL) {
            HashNode *temp = node;
            node = node->next;
            free(temp->key);
            free(temp);
        }
    }
    free(ht);
}
```

### 3.2 插入

```c
int ht_put(HashTable *ht, const char *key, int value) {
    unsigned int idx = hash_djb2(key);

    // 检查是否已存在该 key
    HashNode *node = ht->buckets[idx];
    while (node != NULL) {
        if (strcmp(node->key, key) == 0) {
            node->value = value;  // 更新值
            return 0;
        }
        node = node->next;
    }

    // 新建节点，头插法
    HashNode *new_node = malloc(sizeof(HashNode));
    if (new_node == NULL) return -1;

    new_node->key = strdup(key);  // 复制 key
    new_node->value = value;
    new_node->next = ht->buckets[idx];
    ht->buckets[idx] = new_node;
    ht->count++;
    return 0;
}
```

### 3.3 查找

```c
// 查找，找到返回 1，未找到返回 0
int ht_get(const HashTable *ht, const char *key, int *out_value) {
    unsigned int idx = hash_djb2(key);
    HashNode *node = ht->buckets[idx];

    while (node != NULL) {
        if (strcmp(node->key, key) == 0) {
            *out_value = node->value;
            return 1;
        }
        node = node->next;
    }
    return 0;
}

// 判断 key 是否存在
int ht_contains(const HashTable *ht, const char *key) {
    unsigned int idx = hash_djb2(key);
    HashNode *node = ht->buckets[idx];
    while (node != NULL) {
        if (strcmp(node->key, key) == 0) return 1;
        node = node->next;
    }
    return 0;
}
```

### 3.4 删除

```c
int ht_remove(HashTable *ht, const char *key) {
    unsigned int idx = hash_djb2(key);
    HashNode *node = ht->buckets[idx];
    HashNode *prev = NULL;

    while (node != NULL) {
        if (strcmp(node->key, key) == 0) {
            if (prev == NULL) {
                ht->buckets[idx] = node->next;
            } else {
                prev->next = node->next;
            }
            free(node->key);
            free(node);
            ht->count--;
            return 1;
        }
        prev = node;
        node = node->next;
    }
    return 0;
}
```

## 四、开放地址法（数组实现）

```c
#define OPEN_TABLE_SIZE 64

typedef enum { EMPTY, OCCUPIED, DELETED } SlotState;

typedef struct {
    char      *key;
    int        value;
    SlotState  state;
} OpenSlot;

typedef struct {
    OpenSlot slots[OPEN_TABLE_SIZE];
    int      count;
} OpenHashTable;

// 二次探测法
unsigned int probe(OpenHashTable *ht, const char *key, int attempt) {
    unsigned int h = hash_bkdr(key);
    unsigned int q = (unsigned int)attempt * attempt;
    return (h + q) % OPEN_TABLE_SIZE;
}

int open_ht_put(OpenHashTable *ht, const char *key, int value) {
    for (int i = 0; i < OPEN_TABLE_SIZE; i++) {
        unsigned int idx = probe(ht, key, i);
        if (ht->slots[idx].state == EMPTY ||
            ht->slots[idx].state == DELETED ||
            strcmp(ht->slots[idx].key, key) == 0) {
            if (ht->slots[idx].state == OCCUPIED) {
                free(ht->slots[idx].key);
            }
            ht->slots[idx].key = strdup(key);
            ht->slots[idx].value = value;
            ht->slots[idx].state = OCCUPIED;
            ht->count++;
            return 0;
        }
    }
    return -1;  // 表满
}
```

## 五、完整示例

```c
int main(void) {
    HashTable *ht = ht_create();

    ht_put(ht, "Alice", 90);
    ht_put(ht, "Bob", 85);
    ht_put(ht, "Charlie", 92);
    ht_put(ht, "David", 88);

    // 查找
    int score;
    if (ht_get(ht, "Alice", &score)) {
        printf("Alice 的成绩: %d\n", score);
    }

    // 更新
    ht_put(ht, "Bob", 95);
    ht_get(ht, "Bob", &score);
    printf("Bob 更新后成绩: %d\n", score);

    // 删除
    ht_remove(ht, "Charlie");
    printf("Charlie %s\n", ht_contains(ht, "Charlie") ? "存在" : "不存在");

    printf("总条目数: %d\n", ht->count);
    ht_destroy(ht);
    return 0;
}
```

## 六、关键要点

> **重要提示：**
>
> 1. 哈希函数决定了键的分布均匀性，直接影响性能。
> 2. 链地址法（Separate Chaining）是最常用的冲突解决方案。
> 3. 开放地址法（Open Addressing）将冲突探测放在数组内部。
> 4. 哈希表的 key 使用字符串时，需要 `strdup` 复制，避免悬垂指针。
> 5. 负载因子（count / table_size）超过阈值时应考虑扩容。
> 6. 销毁哈希表时需要遍历所有 bucket 释放节点内存。
