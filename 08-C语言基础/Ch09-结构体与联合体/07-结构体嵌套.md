# 07 - 结构体嵌套

## 一、嵌套结构体的基本概念

一个结构体的成员可以是**另一个结构体类型**，这就是结构体嵌套。它允许我们构建更复杂的数据模型。

```c
struct Date {
    int year;
    int month;
    int day;
};

struct Student {
    int id;
    char name[32];
    struct Date birthday;  // 嵌套结构体
    struct Date enroll_date;
};
```

## 二、嵌套结构体的初始化

### 2.1 顺序初始化

```c
struct Student s1 = {
    1001,
    "张三",
    {2000, 5, 15},    // birthday
    {2023, 9, 1}      // enroll_date
};
```

### 2.2 指定初始化（推荐）

```c
struct Student s2 = {
    .id = 1002,
    .name = "李四",
    .birthday = {.year = 2001, .month = 3, .day = 20},
    .enroll_date = {.year = 2023, .month = 9, .day = 1}
};
```

### 2.3 部分初始化

```c
struct Student s3 = {
    1003,
    "王五",
    {2000, 6}  // day 默认为 0
};
// enroll_date 所有成员为 0
```

## 三、嵌套成员的访问

### 3.1 使用 `.` 操作符链

```c
struct Student s1 = {1001, "张三", {2000, 5, 15}, {2023, 9, 1}};

// 链式访问
printf("生日: %d年%d月%d日\n",
       s1.birthday.year,
       s1.birthday.month,
       s1.birthday.day);

// 修改嵌套成员
s1.birthday.day = 20;
s1.enroll_date.year = 2024;
```

### 3.2 使用 `->` 操作符链

```c
struct Student *p = &s1;

printf("生日: %d/%d/%d\n",
       p->birthday.year,
       p->birthday.month,
       p->birthday.day);

p->birthday.day = 25;
```

### 3.3 多层嵌套访问

```c
struct Time {
    int hour;
    int minute;
    int second;
};

struct DateTime {
    struct Date date;
    struct Time time;
};

struct Event {
    char name[64];
    struct DateTime start;
    struct DateTime end;
};

// 多层访问
struct Event meeting = {
    "项目评审",
    {{2024, 6, 15}, {9, 30, 0}},
    {{2024, 6, 15}, {11, 30, 0}}
};

printf("开始: %d/%d/%d %02d:%02d\n",
       meeting.start.date.year,
       meeting.start.date.month,
       meeting.start.date.day,
       meeting.start.time.hour,
       meeting.start.time.minute);
```

## 四、匿名嵌套结构体（C11）

C11 允许定义**匿名结构体成员**，其成员被"提升"到外层结构体中。

```c
struct Student {
    int id;
    char name[32];
    struct {           // 匿名结构体，没有成员名
        int year;
        int month;
        int day;
    };                 // 注意：没有 birthday 标识符
};

// 直接访问"提升"后的成员
struct Student s1 = {1001, "张三", {2000, 5, 15}};

// 不需要 s1.birthday.year，直接：
printf("%d/%d/%d\n", s1.year, s1.month, s1.day);
```

> **注意**：匿名嵌套与普通嵌套的成员访问方式不同，需要区分。

## 五、嵌套结构体与函数

```c
// 传递嵌套结构体中的子结构体
int compare_date(struct Date d1, struct Date d2) {
    if (d1.year != d2.year) return d1.year - d2.year;
    if (d1.month != d2.month) return d1.month - d2.month;
    return d1.day - d2.day;
}

// 使用
if (compare_date(s1.birthday, s2.birthday) < 0) {
    printf("s1 的生日更早\n");
}

// 通过指针传递，效率更高
int is_same_date(const struct Date *d1, const struct Date *d2) {
    return d1->year == d2->year &&
           d1->month == d2->month &&
           d1->day == d2->day;
}
```

## 六、嵌套结构体的内存布局

```c
struct Date {
    int year;    // 偏移 0,  大小 4
    int month;   // 偏移 4,  大小 4
    int day;     // 偏移 8,  大小 4
};              // 总大小 12

struct Student {
    int id;              // 偏移 0,   大小 4
    char name[32];       // 偏移 4,   大小 32
    struct Date birthday; // 偏移 36,  大小 12
    struct Date enroll;   // 偏移 48,  大小 12
};                      // 总大小 60
```

## 七、完整示例

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int year;
    int month;
    int day;
} Date;

typedef struct {
    char street[64];
    char city[32];
    char province[32];
    int zipcode;
} Address;

typedef struct {
    int id;
    char name[32];
    Date birthday;
    Address home_addr;
    Address work_addr;
} Employee;

void print_employee(const Employee *emp) {
    printf("编号: %d\n", emp->id);
    printf("姓名: %s\n", emp->name);
    printf("生日: %04d-%02d-%02d\n",
           emp->birthday.year, emp->birthday.month, emp->birthday.day);
    printf("家庭地址: %s %s %s\n",
           emp->home_addr.province, emp->home_addr.city, emp->home_addr.street);
    printf("工作地址: %s %s %s\n",
           emp->work_addr.province, emp->work_addr.city, emp->work_addr.street);
}

int main(void) {
    Employee emp = {
        .id = 1001,
        .name = "赵六",
        .birthday = {1995, 8, 20},
        .home_addr = {"建设路 10 号", "成都", "四川省", 610000},
        .work_addr = {"天府大道 200 号", "成都", "四川省", 610041}
    };

    print_employee(&emp);
    return 0;
}
```

## 八、关键要点

> **重要提示：**
>
> 1. 嵌套结构体通过 `.` 链式访问：`outer.inner.member`。
> 2. 指针访问时用 `->` 和 `.` 混合：`p->inner.member`。
> 3. 嵌套层级不宜过深（建议不超过 3 层），否则代码可读性差。
> 4. C11 匿名嵌套可简化访问，但要注意命名冲突风险。
> 5. 嵌套结构体的内存布局是按定义顺序线性排列的。
> 6. 可以单独传递嵌套的子结构体给函数进行处理。
