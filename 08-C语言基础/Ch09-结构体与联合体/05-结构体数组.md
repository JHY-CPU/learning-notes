# 05 - 结构体数组

## 一、结构体数组的定义

结构体数组是将**相同类型的结构体变量**组织成数组，是实际开发中最常用的数据组织方式之一。

```c
struct Student {
    int id;
    char name[32];
    float score;
};

// 方式 1：先定义类型，再声明数组
struct Student students[50];

// 方式 2：定义时声明数组
struct Student {
    int id;
    char name[32];
    float score;
} students[50];

// 方式 3：使用 typedef
typedef struct {
    int id;
    char name[32];
    float score;
} Student;
Student students[50];
```

## 二、结构体数组的初始化

### 2.1 完全初始化

```c
struct Student class[3] = {
    {1001, "张三", 92.5},
    {1002, "李四", 88.0},
    {1003, "王五", 95.5}
};
```

### 2.2 使用指定初始化器（C99）

```c
struct Student class[3] = {
    [0] = {.id = 1001, .name = "张三", .score = 92.5},
    [1] = {.id = 1002, .name = "李四", .score = 88.0},
    [2] = {.id = 1003, .name = "王五", .score = 95.5}
};
```

### 2.3 部分初始化

```c
// 只初始化前两个元素，其余为零
struct Student class[5] = {
    {1001, "张三", 92.5},
    {1002, "李四", 88.0}
};
// class[2] ~ class[4] 的所有成员为 0
```

### 2.4 让编译器推断大小

```c
// 不指定数组大小，由初始值个数决定
struct Student class[] = {
    {1001, "张三", 92.5},
    {1002, "李四", 88.0},
    {1003, "王五", 95.5}
};
// sizeof(class) / sizeof(class[0]) == 3
```

## 三、结构体数组的遍历与访问

```c
#include <stdio.h>

struct Student {
    int id;
    char name[32];
    float score;
};

int main(void) {
    struct Student class[] = {
        {1001, "张三", 92.5},
        {1002, "李四", 88.0},
        {1003, "王五", 95.5},
        {1004, "赵六", 76.0}
    };
    int n = sizeof(class) / sizeof(class[0]);

    // 普通 for 循环遍历
    printf("=== 全班成绩 ===\n");
    for (int i = 0; i < n; i++) {
        printf("学号: %d, 姓名: %-8s, 成绩: %.1f\n",
               class[i].id, class[i].name, class[i].score);
    }

    // 使用指针遍历（更高效）
    printf("\n=== 指针遍历 ===\n");
    for (struct Student *p = class; p < class + n; p++) {
        printf("%s 的成绩: %.1f\n", p->name, p->score);
    }

    return 0;
}
```

## 四、结构体数组的排序

### 4.1 使用 qsort 排序

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int id;
    char name[32];
    float score;
} Student;

// 按成绩降序排列的比较函数
int compare_by_score(const void *a, const void *b) {
    const Student *sa = (const Student *)a;
    const Student *sb = (const Student *)b;

    if (sa->score < sb->score) return 1;   // 降序
    if (sa->score > sb->score) return -1;
    return 0;
}

// 按姓名字典序排列的比较函数
int compare_by_name(const void *a, const void *b) {
    const Student *sa = (const Student *)a;
    const Student *sb = (const Student *)b;
    return strcmp(sa->name, sb->name);
}

int main(void) {
    Student class[] = {
        {1001, "张三", 92.5},
        {1002, "李四", 88.0},
        {1003, "王五", 95.5},
        {1004, "赵六", 76.0}
    };
    int n = sizeof(class) / sizeof(class[0]);

    // 按成绩降序排列
    qsort(class, n, sizeof(Student), compare_by_score);
    printf("=== 按成绩排名 ===\n");
    for (int i = 0; i < n; i++) {
        printf("%d. %s - %.1f\n", i + 1, class[i].name, class[i].score);
    }

    // 按姓名排序
    qsort(class, n, sizeof(Student), compare_by_name);
    printf("\n=== 按姓名排序 ===\n");
    for (int i = 0; i < n; i++) {
        printf("%s - %.1f\n", class[i].name, class[i].score);
    }

    return 0;
}
```

### 4.2 手动排序（冒泡排序示例）

```c
void sort_by_score(Student arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j].score < arr[j + 1].score) {
                // 交换整个结构体
                Student temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
```

## 五、结构体数组的查找

```c
// 按学号查找，返回下标，未找到返回 -1
int find_by_id(Student arr[], int n, int target_id) {
    for (int i = 0; i < n; i++) {
        if (arr[i].id == target_id) {
            return i;
        }
    }
    return -1;
}

// 按姓名查找
Student *find_by_name(Student arr[], int n, const char *name) {
    for (int i = 0; i < n; i++) {
        if (strcmp(arr[i].name, name) == 0) {
            return &arr[i];  // 返回指针，方便修改
        }
    }
    return NULL;
}

// 使用
int idx = find_by_id(class, n, 1003);
if (idx != -1) {
    printf("找到: %s\n", class[idx].name);
}

Student *p = find_by_name(class, n, "张三");
if (p != NULL) {
    p->score = 99.0;  // 直接修改
}
```

## 六、统计与计算

```c
// 计算平均分
float average_score(Student arr[], int n) {
    float sum = 0;
    for (int i = 0; i < n; i++) {
        sum += arr[i].score;
    }
    return sum / n;
}

// 找最高分学生
Student *top_student(Student arr[], int n) {
    int max_idx = 0;
    for (int i = 1; i < n; i++) {
        if (arr[i].score > arr[max_idx].score) {
            max_idx = i;
        }
    }
    return &arr[max_idx];
}

// 统计及格人数
int count_pass(Student arr[], int n, float pass_line) {
    int count = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i].score >= pass_line) {
            count++;
        }
    }
    return count;
}
```

## 七、关键要点

> **重要提示：**
>
> 1. 结构体数组初始化时，每个元素用 `{}` 包裹初始值。
> 2. `sizeof(arr) / sizeof(arr[0])` 是计算数组长度的标准写法。
> 3. 使用 `qsort` 排序结构体数组时，比较函数是关键。
> 4. 结构体的整体赋值（`Student temp = arr[j]`）是安全的。
> 5. 查找函数返回指针比返回下标更灵活，方便后续修改。
> 6. 结构体数组在内存中是连续存放的，可以利用这一点进行指针遍历。
