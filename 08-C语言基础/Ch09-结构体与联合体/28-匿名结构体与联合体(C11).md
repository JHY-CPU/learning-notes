# 28 - 匿名结构体与联合体（C11）

## 一、C11 匿名结构体

C11 允许定义**没有名称**的结构体成员，其内部成员被"提升"到外层结构体中。

### 1.1 基本语法

```c
#include <stdio.h>

// C11 之前：需要通过 .inner_member 访问
struct OldWay {
    int id;
    struct {
        int x;
        int y;
    } point;  // 有名字
};

// C11：匿名结构体成员
struct NewWay {
    int id;
    struct {        // 没有成员名！
        int x;
        int y;
    };              // 注意：直接分号结束
};

int main(void) {
    struct OldWay old = {1, {10, 20}};
    printf("%d, %d\n", old.point.x, old.point.y);  // 通过 point 访问

    struct NewWay nw = {1, {10, 20}};
    printf("%d, %d\n", nw.x, nw.y);  // 直接访问！x, y 被"提升"

    return 0;
}
```

## 二、C11 匿名联合体

匿名联合体的成员同样被提升到外层结构体中。

```c
struct Value {
    int type;
    union {
        int    int_val;
        float  float_val;
        double double_val;
    };  // 匿名联合体
};

// 使用
struct Value v;
v.type = 1;
v.int_val = 42;       // 直接访问，不需要 v.union_member.int_val
printf("%d\n", v.int_val);
```

## 三、匿名结构体与联合体组合

### 3.1 变体类型简化

```c
// C11 之前的写法
typedef struct {
    int type;
    union {
        struct { int x; int y; } point;
        struct { int w; int h; } size;
    } data;
} OldVariant;

// 使用：
// v.data.point.x = 10;

// C11 匿名写法
typedef struct {
    int type;
    union {
        struct { int x; int y; };   // 匿名结构体在匿名联合体中
        struct { int w; int h; };
    };
} NewVariant;

// 使用更简洁：
// v.x = 10;  // 直接访问！
```

### 3.2 几何图形

```c
#include <stdio.h>
#include <math.h>

typedef struct {
    int shape_type;  // 0=circle, 1=rectangle, 2=triangle
    union {
        struct { double radius; };
        struct { double width; double height; };
        struct { double a, b, c; };  // 三边
    };
} Shape;

double shape_area(const Shape *s) {
    switch (s->shape_type) {
        case 0: return M_PI * s->radius * s->radius;
        case 1: return s->width * s->height;
        case 2: {
            double p = (s->a + s->b + s->c) / 2;
            return sqrt(p * (p-s->a) * (p-s->b) * (p-s->c));
        }
        default: return 0;
    }
}

int main(void) {
    Shape circle = {.shape_type = 0, .radius = 5.0};
    Shape rect = {.shape_type = 1, .width = 3.0, .height = 4.0};

    printf("圆形面积: %.2f\n", shape_area(&circle));
    printf("矩形面积: %.2f\n", shape_area(&rect));

    return 0;
}
```

## 四、寄存器映射示例

```c
#include <stdint.h>

// 硬件寄存器：可以用位域、字节、字访问
typedef union {
    uint32_t word;  // 整体访问

    struct {                    // 匿名联合体中的匿名结构体
        uint32_t enable    : 1;
        uint32_t interrupt : 1;
        uint32_t mode      : 2;
        uint32_t prescaler : 4;
        uint32_t reserved  : 24;
    };

    struct {
        uint8_t byte0;
        uint8_t byte1;
        uint8_t byte2;
        uint8_t byte3;
    };
} TimerRegister;

void timer_config(void) {
    volatile TimerRegister *timer = (volatile TimerRegister *)0x40001000;

    timer->enable = 1;       // 位域访问
    timer->mode = 2;
    timer->prescaler = 7;

    uint32_t val = timer->word;  // 整体读取
    uint8_t lo = timer->byte0;   // 字节访问
}
```

## 五、嵌套匿名成员的访问规则

```c
struct Outer {
    int a;
    struct {            // 匿名结构体
        int b;
        union {         // 匿名联合体
            int c;
            int d;
        };
    };
};

// 访问规则：
struct Outer o;
o.a = 1;    // 直接访问外部成员
o.b = 2;    // 提升的匿名结构体成员
o.c = 3;    // 提升的匿名联合体成员
o.d = 4;    // c 和 d 共享内存，d 覆盖 c
```

### 名字冲突

```c
// 如果匿名成员的字段与外部字段同名，外部优先
struct Conflict {
    int value;
    struct {
        int value;  // 同名！
        int other;
    };
};

struct Conflict c;
c.value = 1;      // 访问外层的 value
// 如何访问内层的 value？
// c.value 不行（访问的是外层），需要避免这种设计
```

## 六、C11 vs 旧写法对比

```c
// C11 匿名成员
struct {
    int type;
    union {
        int i;
        float f;
    };
} v;

v.i = 42;  // 简洁

// 等价的旧写法
struct {
    int type;
    union {
        int i;
        float f;
    } data;
} v_old;

v_old.data.i = 42;  // 需要通过 data 访问
```

## 七、关键要点

> **重要提示：**
>
> 1. C11 匿名结构体/联合体的成员被"提升"到外层结构体中。
> 2. 匿名成员简化了访问语法，不需要中间层的成员名。
> 3. 匿名联合体在实现变体类型和寄存器映射时特别有用。
> 4. 避免匿名成员的字段名与外部字段名冲突。
> 5. 匿名结构体/联合体是 C11 特性，编译器需支持 `-std=c11`。
> 6. Linux 内核中广泛使用匿名联合体（如寄存器定义）。
