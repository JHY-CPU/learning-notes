# 12 - 联合体应用

## 一、变体类型实现

变体类型（Variant Type）是一种可以在运行时存储不同类型数据的数据结构。

### 1.1 基础变体类型

```c
#include <stdio.h>
#include <string.h>

typedef enum {
    VAR_NULL,
    VAR_INT,
    VAR_DOUBLE,
    VAR_STRING,
    VAR_BOOL
} VarType;

typedef struct {
    VarType type;
    union {
        int    int_val;
        double double_val;
        char   string_val[128];
        int    bool_val;  // 0=false, non-zero=true
    };
} Variant;

// 创建变体
Variant var_int(int val) {
    Variant v = {VAR_INT, {.int_val = val}};
    return v;
}

Variant var_string(const char *val) {
    Variant v = {VAR_STRING};
    strncpy(v.string_val, val, sizeof(v.string_val) - 1);
    v.string_val[sizeof(v.string_val) - 1] = '\0';
    return v;
}

// 打印变体
void var_print(const Variant *v) {
    switch (v->type) {
        case VAR_NULL:   printf("NULL\n"); break;
        case VAR_INT:    printf("%d\n", v->int_val); break;
        case VAR_DOUBLE: printf("%.6f\n", v->double_val); break;
        case VAR_STRING: printf("%s\n", v->string_val); break;
        case VAR_BOOL:   printf("%s\n", v->bool_val ? "true" : "false"); break;
    }
}

int main(void) {
    Variant v1 = var_int(42);
    Variant v2 = var_string("Hello, Union!");

    var_print(&v1);  // 42
    var_print(&v2);  // Hello, Union!

    return 0;
}
```

## 二、类型双关（Type Punning）

类型双关是通过联合体以不同的数据类型解释同一块内存。

### 2.1 查看浮点数的二进制表示

```c
#include <stdio.h>
#include <stdint.h>

typedef union {
    float    f;
    uint32_t bits;
} FloatPun;

typedef union {
    double   d;
    uint64_t bits;
} DoublePun;

void print_float_bits(float f) {
    FloatPun u = {.f = f};
    printf("%.2f = 0x%08X (符号:%d, 指数:%d, 尾数:0x%06X)\n",
           f, u.bits,
           (int)(u.bits >> 31) & 1,
           (int)(u.bits >> 23) & 0xFF,
           (int)(u.bits & 0x7FFFFF));
}

int main(void) {
    print_float_bits(3.14f);
    print_float_bits(-1.0f);
    print_float_bits(0.0f);
    return 0;
}
```

### 2.2 快速平方根倒数（经典案例）

```c
// Quake III 中著名的快速平方根倒数算法
float fast_inv_sqrt(float x) {
    union {
        float    f;
        uint32_t i;
    } conv = {.f = x};

    conv.i = 0x5F3759DF - (conv.i >> 1);  // 魔法初值
    conv.f *= (1.5f - (x * 0.5f * conv.f * conv.f));  // 牛顿迭代
    return conv.f;
}
```

## 三、网络协议解析

### 3.1 字节序检测

```c
#include <stdio.h>

typedef union {
    uint32_t value;
    uint8_t  bytes[4];
} EndianTest;

// 检测系统字节序
int is_little_endian(void) {
    EndianTest test = {.value = 0x01020304};
    return test.bytes[0] == 0x04;  // 小端：低位字节在前
}

int main(void) {
    if (is_little_endian()) {
        printf("系统是小端字节序\n");
    } else {
        printf("系统是大端字节序\n");
    }
    return 0;
}
```

### 3.2 网络数据包解析

```c
#include <stdio.h>
#include <stdint.h>
#include <string.h>

#pragma pack(push, 1)
typedef struct {
    uint8_t  type;
    uint8_t  flags;
    uint16_t length;
    uint32_t seq_num;
} PacketHeader;
#pragma pack(pop)

typedef union {
    PacketHeader header;
    uint8_t      raw[sizeof(PacketHeader)];
} Packet;

void parse_packet(const uint8_t *data, size_t len) {
    if (len < sizeof(PacketHeader)) return;

    Packet pkt;
    memcpy(&pkt, data, sizeof(PacketHeader));

    printf("类型: 0x%02X\n", pkt.header.type);
    printf("标志: 0x%02X\n", pkt.header.flags);
    printf("长度: %u\n", pkt.header.length);
    printf("序号: %u\n", pkt.header.seq_num);

    // 以原始字节方式查看
    printf("原始字节: ");
    for (size_t i = 0; i < sizeof(PacketHeader); i++) {
        printf("%02X ", pkt.raw[i]);
    }
    printf("\n");
}
```

## 四、内存池与对象复用

```c
#include <stdio.h>
#include <stdlib.h>

// 利用联合体实现简单的对象池
#define POOL_SIZE 10

typedef union ObjectNode {
    struct {
        int data;
        // ... 其他业务字段
    } obj;
    union ObjectNode *next;  // 空闲时用作链表指针
} ObjectNode;

typedef struct {
    ObjectNode pool[POOL_SIZE];
    ObjectNode *free_list;
} ObjectPool;

void pool_init(ObjectPool *p) {
    // 将所有节点串成空闲链表
    for (int i = 0; i < POOL_SIZE - 1; i++) {
        p->pool[i].next = &p->pool[i + 1];
    }
    p->pool[POOL_SIZE - 1].next = NULL;
    p->free_list = &p->pool[0];
}

ObjectNode *pool_alloc(ObjectPool *p) {
    if (p->free_list == NULL) return NULL;
    ObjectNode *node = p->free_list;
    p->free_list = node->next;
    return node;
}

void pool_free(ObjectPool *p, ObjectNode *node) {
    node->next = p->free_list;
    p->free_list = node;
}
```

## 五、寄存器位操作

```c
#include <stdio.h>
#include <stdint.h>

// GPIO 寄存器：支持按位访问和整体访问
typedef union {
    struct {
        uint32_t pin0  : 1;
        uint32_t pin1  : 1;
        uint32_t pin2  : 1;
        uint32_t pin3  : 1;
        uint32_t pin4  : 1;
        uint32_t pin5  : 1;
        uint32_t pin6  : 1;
        uint32_t pin7  : 1;
        uint32_t reserved : 24;
    } bits;
    uint32_t reg;
} GPIO_Register;

int main(void) {
    GPIO_Register gpio = {0};

    // 逐位设置
    gpio.bits.pin0 = 1;
    gpio.bits.pin3 = 1;
    gpio.bits.pin7 = 1;

    printf("GPIO 寄存器值: 0x%08X\n", gpio.reg);
    // 输出: 0x00000089 (bit 0, 3, 7 被置位)

    // 整体清零
    gpio.reg = 0;

    // 整体赋值后按位读取
    gpio.reg = 0xAB;
    printf("pin0: %d\n", gpio.bits.pin0);
    printf("pin7: %d\n", gpio.bits.pin7);

    return 0;
}
```

## 六、关键要点

> **重要提示：**
>
> 1. 变体类型需要枚举标记配合联合体，确保类型安全。
> 2. 类型双关在 C99 中是明确允许的联合体用法。
> 3. 字节序检测是联合体的经典应用之一。
> 4. 位域与联合体结合可以同时支持按位访问和整体访问。
> 5. 对象池利用联合体在"空闲"和"使用中"两种状态间切换内存用途。
> 6. 网络协议解析时注意字节序和对齐问题。
