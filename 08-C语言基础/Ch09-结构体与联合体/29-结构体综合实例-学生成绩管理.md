# 29 - 结构体综合实例 - 学生成绩管理系统

## 一、系统设计

本实例综合运用结构体、动态内存、文件操作、排序、查找等知识，实现一个完整的学生成绩管理系统。

### 数据结构设计

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define MAX_NAME_LEN 32
#define MAX_SUBJECTS 6
#define MAX_STUDENTS 200

typedef struct {
    char name[MAX_NAME_LEN];
    float score;
} Subject;

typedef struct {
    int     id;
    char    name[MAX_NAME_LEN];
    Subject subjects[MAX_SUBJECTS];
    int     num_subjects;
    float   average;
    int     rank;
} Student;

typedef struct {
    Student *students;
    int      count;
    int      capacity;
    char     subject_names[MAX_SUBJECTS][MAX_NAME_LEN];
    int      num_subjects;
} StudentDB;
```

## 二、数据库操作

### 2.1 初始化与销毁

```c
StudentDB *db_create(void) {
    StudentDB *db = malloc(sizeof(StudentDB));
    if (db == NULL) return NULL;

    db->capacity = 10;
    db->count = 0;
    db->num_subjects = 0;
    db->students = malloc(sizeof(Student) * db->capacity);
    if (db->students == NULL) {
        free(db);
        return NULL;
    }
    return db;
}

void db_destroy(StudentDB *db) {
    if (db == NULL) return;
    free(db->students);
    free(db);
}
```

### 2.2 添加学生

```c
static int db_ensure_capacity(StudentDB *db) {
    if (db->count < db->capacity) return 0;
    int new_cap = db->capacity * 2;
    Student *new_arr = realloc(db->students, sizeof(Student) * new_cap);
    if (new_arr == NULL) return -1;
    db->students = new_arr;
    db->capacity = new_cap;
    return 0;
}

int db_add_student(StudentDB *db, int id, const char *name) {
    if (db->count >= MAX_STUDENTS) return -1;
    if (db_ensure_capacity(db) != 0) return -1;

    Student *s = &db->students[db->count];
    memset(s, 0, sizeof(Student));
    s->id = id;
    strncpy(s->name, name, MAX_NAME_LEN - 1);
    s->num_subjects = 0;
    s->average = 0;
    s->rank = 0;

    db->count++;
    return 0;
}

int db_add_score(StudentDB *db, int id, const char *subject, float score) {
    Student *s = db_find_by_id(db, id);
    if (s == NULL) return -1;
    if (s->num_subjects >= MAX_SUBJECTS) return -1;

    Subject *sub = &s->subjects[s->num_subjects++];
    strncpy(sub->name, subject, MAX_NAME_LEN - 1);
    sub->score = score;

    // 更新平均分
    db_calc_average(s);
    return 0;
}
```

### 2.3 查找

```c
Student *db_find_by_id(const StudentDB *db, int id) {
    for (int i = 0; i < db->count; i++) {
        if (db->students[i].id == id) {
            return &db->students[i];
        }
    }
    return NULL;
}

Student *db_find_by_name(const StudentDB *db, const char *name) {
    for (int i = 0; i < db->count; i++) {
        if (strcmp(db->students[i].name, name) == 0) {
            return &db->students[i];
        }
    }
    return NULL;
}
```

### 2.4 计算平均分与排名

```c
void db_calc_average(Student *s) {
    if (s->num_subjects == 0) {
        s->average = 0;
        return;
    }
    float sum = 0;
    for (int i = 0; i < s->num_subjects; i++) {
        sum += s->subjects[i].score;
    }
    s->average = sum / s->num_subjects;
}

// 比较函数：按平均分降序
static int cmp_by_avg_desc(const void *a, const void *b) {
    float diff = ((const Student *)b)->average -
                 ((const Student *)a)->average;
    return (diff > 0) - (diff < 0);
}

void db_calc_ranking(StudentDB *db) {
    // 先计算所有平均分
    for (int i = 0; i < db->count; i++) {
        db_calc_average(&db->students[i]);
    }

    // 按平均分排序
    qsort(db->students, db->count, sizeof(Student), cmp_by_avg_desc);

    // 分配排名
    for (int i = 0; i < db->count; i++) {
        db->students[i].rank = i + 1;
    }
}
```

## 三、统计分析

```c
void db_statistics(const StudentDB *db) {
    if (db->count == 0) {
        printf("没有学生数据\n");
        return;
    }

    float total_avg = 0;
    float max_avg = db->students[0].average;
    float min_avg = db->students[0].average;
    int pass_count = 0;

    for (int i = 0; i < db->count; i++) {
        float avg = db->students[i].average;
        total_avg += avg;
        if (avg > max_avg) max_avg = avg;
        if (avg < min_avg) min_avg = avg;
        if (avg >= 60.0) pass_count++;
    }
    total_avg /= db->count;

    printf("=== 成绩统计 ===\n");
    printf("学生人数: %d\n", db->count);
    printf("总平均分: %.2f\n", total_avg);
    printf("最高均分: %.2f\n", max_avg);
    printf("最低均分: %.2f\n", min_avg);
    printf("及格人数: %d (%.1f%%)\n", pass_count,
           100.0 * pass_count / db->count);

    // 分数段统计
    int grade[5] = {0};  // 优秀90+, 良好80+, 中等70+, 及格60+, 不及格
    for (int i = 0; i < db->count; i++) {
        float avg = db->students[i].average;
        if (avg >= 90) grade[0]++;
        else if (avg >= 80) grade[1]++;
        else if (avg >= 70) grade[2]++;
        else if (avg >= 60) grade[3]++;
        else grade[4]++;
    }
    printf("\n分数段分布:\n");
    printf("  优秀(90+): %d 人\n", grade[0]);
    printf("  良好(80+): %d 人\n", grade[1]);
    printf("  中等(70+): %d 人\n", grade[2]);
    printf("  及格(60+): %d 人\n", grade[3]);
    printf("  不及格:    %d 人\n", grade[4]);
}
```

## 四、文件持久化

```c
int db_save(const StudentDB *db, const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (fp == NULL) return -1;

    fprintf(fp, "%d\n", db->count);
    for (int i = 0; i < db->count; i++) {
        const Student *s = &db->students[i];
        fprintf(fp, "%d,%s,%d\n", s->id, s->name, s->num_subjects);
        for (int j = 0; j < s->num_subjects; j++) {
            fprintf(fp, "%s,%.1f\n",
                    s->subjects[j].name, s->subjects[j].score);
        }
    }
    fclose(fp);
    return 0;
}

StudentDB *db_load(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (fp == NULL) return NULL;

    StudentDB *db = db_create();
    if (db == NULL) { fclose(fp); return NULL; }

    int count;
    fscanf(fp, "%d\n", &count);

    for (int i = 0; i < count; i++) {
        int id, num_sub;
        char name[MAX_NAME_LEN];
        fscanf(fp, "%d,%31[^,],%d\n", &id, name, &num_sub);
        db_add_student(db, id, name);

        for (int j = 0; j < num_sub; j++) {
            char sub_name[MAX_NAME_LEN];
            float score;
            fscanf(fp, "%31[^,],%f\n", sub_name, &score);
            db_add_score(db, id, sub_name, score);
        }
    }
    fclose(fp);
    return db;
}
```

## 五、打印输出

```c
void db_print_all(const StudentDB *db) {
    printf("\n%-6s %-10s %-8s %-6s %s\n",
           "排名", "学号", "姓名", "平均分", "各科成绩");
    printf("--------------------------------------------------------\n");

    for (int i = 0; i < db->count; i++) {
        const Student *s = &db->students[i];
        printf("%-6d %-10d %-8s %-6.1f ",
               s->rank, s->id, s->name, s->average);
        for (int j = 0; j < s->num_subjects; j++) {
            printf("%s:%.0f ", s->subjects[j].name,
                   s->subjects[j].score);
        }
        printf("\n");
    }
}
```

## 七、关键要点

> **重要提示：**
>
> 1. 使用动态数组（realloc）管理学生数据，支持动态扩展。
> 2. 结构体嵌套（Student 包含 Subject 数组）自然地组织了数据。
> 3. qsort 配合比较函数实现灵活排序。
> 4. 文件读写使用文本格式，方便调试和检查。
> 5. 分数段统计体现了实际业务逻辑在数据结构上的应用。
> 6. 良好的模块化设计（db_xxx 函数）使代码易于维护和扩展。
