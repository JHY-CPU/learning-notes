# 25 - 结构体与回调

## 一、结构体中存储函数指针

将函数指针作为结构体成员，可以将**数据和行为**绑定在一起，是 C 语言中实现回调和多态的核心技术。

```c
#include <stdio.h>
#include <math.h>

// 结构体中包含函数指针
typedef struct {
    double x;
    double y;

    // 函数指针成员
    double (*distance)(const void *self, const void *other);
    void   (*print)(const void *self);
} Point;

// 具体实现
static double point_distance(const void *a, const void *b) {
    const Point *pa = (const Point *)a;
    const Point *pb = (const Point *)b;
    double dx = pa->x - pb->x;
    double dy = pa->y - pb->y;
    return sqrt(dx * dx + dy * dy);
}

static void point_print(const void *self) {
    const Point *p = (const Point *)self;
    printf("(%.2f, %.2f)\n", p->x, p->y);
}

// 构造函数
Point point_new(double x, double y) {
    return (Point){
        .x = x, .y = y,
        .distance = point_distance,
        .print = point_print
    };
}

int main(void) {
    Point a = point_new(0, 0);
    Point b = point_new(3, 4);

    a.print(&a);  // (0.00, 0.00)
    b.print(&b);  // (3.00, 4.00)

    double d = a.distance(&a, &b);
    printf("距离: %.2f\n", d);  // 5.00

    return 0;
}
```

## 二、虚函数表思想

### 2.1 操作表结构

```c
// 文件操作接口（类似 Linux 的 file_operations）
typedef struct {
    int   (*open)(const char *path, int mode);
    int   (*read)(int fd, void *buf, size_t count);
    int   (*write)(int fd, const void *buf, size_t count);
    int   (*close)(int fd);
    long  (*seek)(int fd, long offset, int whence);
} FileOps;

// 具体实现
static int my_open(const char *path, int mode) {
    printf("打开文件: %s (模式: %d)\n", path, mode);
    return 1;  // 返回 fd
}

static int my_read(int fd, void *buf, size_t count) {
    printf("读取 %zu 字节, fd=%d\n", count, fd);
    return (int)count;
}

static int my_write(int fd, const void *buf, size_t count) {
    printf("写入 %zu 字节, fd=%d\n", count, fd);
    return (int)count;
}

static int my_close(int fd) {
    printf("关闭文件: fd=%d\n", fd);
    return 0;
}

// 文件系统注册
static FileOps local_fs_ops = {
    .open = my_open,
    .read = my_read,
    .write = my_write,
    .close = my_close,
    .seek = NULL  // 不支持 seek
};

// 统一的文件操作接口
void process_file(FileOps *ops) {
    int fd = ops->open("test.txt", 0);
    char buf[100];
    ops->read(fd, buf, 100);
    ops->close(fd);
}
```

## 三、事件回调系统

```c
#include <stdio.h>
#include <string.h>

#define MAX_CALLBACKS 10
#define MAX_NAME_LEN 32

typedef void (*EventCallback)(const char *event, void *data);

typedef struct {
    char           name[MAX_NAME_LEN];
    EventCallback  callback;
    void          *user_data;
} EventHandler;

typedef struct {
    EventHandler handlers[MAX_CALLBACKS];
    int          count;
} EventBus;

void event_bus_init(EventBus *bus) {
    bus->count = 0;
}

// 注册回调
int event_on(EventBus *bus, const char *name,
             EventCallback cb, void *data) {
    if (bus->count >= MAX_CALLBACKS) return -1;

    EventHandler *h = &bus->handlers[bus->count++];
    strncpy(h->name, name, MAX_NAME_LEN - 1);
    h->callback = cb;
    h->user_data = data;
    return 0;
}

// 触发事件
void event_emit(EventBus *bus, const char *event, void *data) {
    for (int i = 0; i < bus->count; i++) {
        if (strcmp(bus->handlers[i].name, event) == 0) {
            bus->handlers[i].callback(event, bus->handlers[i].user_data);
        }
    }
}

// 示例回调
void on_login(const char *event, void *data) {
    char *user = (char *)data;
    printf("[%s] 用户 %s 已登录\n", event, user);
}

void on_logout(const char *event, void *data) {
    char *user = (char *)data;
    printf("[%s] 用户 %s 已登出\n", event, user);
}
```

## 四、迭代器回调模式

```c
#include <stdio.h>

// 回调函数类型
typedef void (*VisitFunc)(int value, void *ctx);

// 遍历数组，对每个元素调用回调
void array_foreach(const int *arr, int n, VisitFunc fn, void *ctx) {
    for (int i = 0; i < n; i++) {
        fn(arr[i], ctx);
    }
}

// 回调 1：打印
void print_value(int val, void *ctx) {
    printf("%d ", val);
}

// 回调 2：求和（通过 ctx 传递累加器）
void sum_value(int val, void *ctx) {
    int *sum = (int *)ctx;
    *sum += val;
}

// 回调 3：查找最大值
void max_value(int val, void *ctx) {
    int *max = (int *)ctx;
    if (val > *max) *max = val;
}

int main(void) {
    int arr[] = {3, 1, 4, 1, 5, 9, 2, 6};
    int n = sizeof(arr) / sizeof(arr[0]);

    printf("遍历: ");
    array_foreach(arr, n, print_value, NULL);
    printf("\n");

    int sum = 0;
    array_foreach(arr, n, sum_value, &sum);
    printf("总和: %d\n", sum);

    int max = arr[0];
    array_foreach(arr, n, max_value, &max);
    printf("最大: %d\n", max);

    return 0;
}
```

## 五、排序回调

```c
#include <stdlib.h>

typedef struct {
    char name[32];
    int  age;
    double salary;
} Employee;

// 多种排序策略
int by_age_asc(const void *a, const void *b) {
    return ((const Employee *)a)->age - ((const Employee *)b)->age;
}

int by_salary_desc(const void *a, const void *b) {
    double diff = ((const Employee *)b)->salary -
                  ((const Employee *)a)->salary;
    return (diff > 0) - (diff < 0);
}

int by_name(const void *a, const void *b) {
    return strcmp(((const Employee *)a)->name,
                  ((const Employee *)b)->name);
}

// 通用排序接口
typedef int (*CompareFunc)(const void *, const void *);

void sort_employees(Employee *arr, int n, CompareFunc cmp) {
    qsort(arr, n, sizeof(Employee), cmp);
}
```

## 六、关键要点

> **重要提示：**
>
> 1. 结构体中的函数指针将数据和行为绑定，是 C 语言中模拟 OOP 的基础。
> 2. 回调函数通过函数指针实现"反向控制"——被调用方通知调用方。
> 3. `void *ctx` 参数是传递上下文的标准做法，实现数据的灵活传递。
> 4. 虚函数表模式中，所有"方法"的第一个参数通常是 `self` 指针。
> 5. Linux 内核大量使用函数指针结构体（如 `file_operations`、`net_device_ops`）。
> 6. 回调函数的类型必须严格匹配，否则是未定义行为。
