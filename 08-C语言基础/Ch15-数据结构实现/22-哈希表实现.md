# 哈希表实现

## 1. 概述

本节实现一个完整的哈希表，采用**链地址法**处理冲突，支持动态扩容。

## 2. 数据结构定义

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

#define INITIAL_CAPACITY 16
#define LOAD_FACTOR_THRESHOLD 0.75

// 键值对节点（链表节点）
typedef struct HashNode {
    char *key;
    int value;
    struct HashNode *next;
} HashNode;

// 哈希表
typedef struct {
    HashNode **buckets;  // 桶数组（链表头指针数组）
    int capacity;        // 桶的数量
    int size;            // 元素个数
} HashTable;
```

## 3. 哈希函数

```c
// DJB2 哈希函数
unsigned int hash(const char *key, int capacity) {
    unsigned int h = 5381;
    int c;
    while ((c = *key++)) {
        h = ((h << 5) + h) + c;
    }
    return h % capacity;
}
```

## 4. 基本操作实现

### 4.1 创建与销毁

```c
HashTable *createTable() {
    HashTable *ht = (HashTable *)malloc(sizeof(HashTable));
    ht->capacity = INITIAL_CAPACITY;
    ht->size = 0;
    ht->buckets = (HashNode **)calloc(ht->capacity, sizeof(HashNode *));
    return ht;
}

void destroyTable(HashTable *ht) {
    for (int i = 0; i < ht->capacity; i++) {
        HashNode *node = ht->buckets[i];
        while (node) {
            HashNode *temp = node;
            node = node->next;
            free(temp->key);
            free(temp);
        }
    }
    free(ht->buckets);
    free(ht);
}
```

### 4.2 创建节点

```c
HashNode *createNode(const char *key, int value) {
    HashNode *node = (HashNode *)malloc(sizeof(HashNode));
    node->key = strdup(key);  // 复制字符串
    node->value = value;
    node->next = NULL;
    return node;
}
```

### 4.3 插入/更新

```c
// 扩容声明
void resize(HashTable *ht);

void put(HashTable *ht, const char *key, int value) {
    // 检查是否需要扩容
    if ((double)ht->size / ht->capacity > LOAD_FACTOR_THRESHOLD) {
        resize(ht);
    }

    unsigned int index = hash(key, ht->capacity);
    HashNode *node = ht->buckets[index];

    // 查找是否已存在
    while (node) {
        if (strcmp(node->key, key) == 0) {
            node->value = value;  // 更新
            return;
        }
        node = node->next;
    }

    // 新增节点（头插法）
    HashNode *newNode = createNode(key, value);
    newNode->next = ht->buckets[index];
    ht->buckets[index] = newNode;
    ht->size++;
}
```

### 4.4 查找

```c
bool get(HashTable *ht, const char *key, int *value) {
    unsigned int index = hash(key, ht->capacity);
    HashNode *node = ht->buckets[index];

    while (node) {
        if (strcmp(node->key, key) == 0) {
            *value = node->value;
            return true;
        }
        node = node->next;
    }
    return false;
}
```

### 4.5 删除

```c
bool removeKey(HashTable *ht, const char *key) {
    unsigned int index = hash(key, ht->capacity);
    HashNode *node = ht->buckets[index];
    HashNode *prev = NULL;

    while (node) {
        if (strcmp(node->key, key) == 0) {
            if (prev) {
                prev->next = node->next;
            } else {
                ht->buckets[index] = node->next;
            }
            free(node->key);
            free(node);
            ht->size--;
            return true;
        }
        prev = node;
        node = node->next;
    }
    return false;
}
```

### 4.6 动态扩容

```c
void resize(HashTable *ht) {
    int oldCapacity = ht->capacity;
    HashNode **oldBuckets = ht->buckets;

    ht->capacity *= 2;
    ht->buckets = (HashNode **)calloc(ht->capacity, sizeof(HashNode *));
    ht->size = 0;

    // 重新哈希所有元素
    for (int i = 0; i < oldCapacity; i++) {
        HashNode *node = oldBuckets[i];
        while (node) {
            HashNode *next = node->next;
            // 重新计算索引并插入
            unsigned int index = hash(node->key, ht->capacity);
            node->next = ht->buckets[index];
            ht->buckets[index] = node;
            ht->size++;
            node = next;
        }
    }
    free(oldBuckets);
    printf("扩容: %d -> %d\n", oldCapacity, ht->capacity);
}
```

### 4.7 检查键是否存在

```c
bool containsKey(HashTable *ht, const char *key) {
    int dummy;
    return get(ht, key, &dummy);
}
```

### 4.8 打印哈希表

```c
void printTable(HashTable *ht) {
    printf("哈希表 [容量=%d, 大小=%d, 负载=%.2f]:\n",
           ht->capacity, ht->size, (double)ht->size / ht->capacity);
    for (int i = 0; i < ht->capacity; i++) {
        if (ht->buckets[i]) {
            printf("  [%d]: ", i);
            HashNode *node = ht->buckets[i];
            while (node) {
                printf("(%s=%d)", node->key, node->value);
                if (node->next) printf(" -> ");
                node = node->next;
            }
            printf("\n");
        }
    }
}
```

```c
    HashTable *ht = createTable();

        printf("banana = %d\n", val);

    destroyTable(ht);
    return 0;
}
```

## 6. 开放地址法实现（线性探测）

```c
#define EMPTY_KEY ""
#define DELETED_KEY "\x01"  // 删除标记

typedef struct {
    char **keys;
    int *values;
    int capacity;
    int size;
} OATable;  // Open Addressing Table

OATable *createOATable(int capacity) {
    OATable *t = (OATable *)malloc(sizeof(OATable));
    t->capacity = capacity;
    t->size = 0;
    t->keys = (char **)calloc(capacity, sizeof(char *));
    t->values = (int *)calloc(capacity, sizeof(int));
    return t;
}

void oaPut(OATable *t, const char *key, int value) {
    unsigned int index = hash(key, t->capacity);
    int i = 0;
    while (t->keys[(index + i) % t->capacity] != NULL &&
           strcmp(t->keys[(index + i) % t->capacity], DELETED_KEY) != 0 &&
           strcmp(t->keys[(index + i) % t->capacity], key) != 0) {
        i++;
    }
    int pos = (index + i) % t->capacity;
    if (t->keys[pos] == NULL || strcmp(t->keys[pos], DELETED_KEY) == 0) {
        t->keys[pos] = strdup(key);
        t->size++;
    }
    t->values[pos] = value;
}

bool oaGet(OATable *t, const char *key, int *value) {
    unsigned int index = hash(key, t->capacity);
    int i = 0;
    while (t->keys[(index + i) % t->capacity] != NULL) {
        if (strcmp(t->keys[(index + i) % t->capacity], key) == 0) {
            *value = t->values[(index + i) % t->capacity];
            return true;
        }
        i++;
        if (i >= t->capacity) return false;
    }
    return false;
}
```

## 7. 性能分析

| 操作 | 平均时间 | 最坏时间 |
|------|---------|---------|
| 插入 | O(1) | O(n) |
| 查找 | O(1) | O(n) |
| 删除 | O(1) | O(n) |

最坏情况发生在所有键都冲突时。通过合理选择哈希函数和控制装载因子，平均性能接近 O(1)。

## 8. 重点总结

- 链地址法实现简单，是最常用的冲突处理方法
- 装载因子超过 0.75 时应扩容（容量翻倍）
- 扩容需要重新哈希所有元素，是 O(n) 操作
- 字符串键需要 `strdup` 复制，注意内存管理
- 开放地址法删除时需要标记（不能直接删除）
- 哈希函数的选择直接影响性能

> **注意**：本实现是简化版本，实际生产环境需要考虑线程安全、更高效的内存管理等问题。
