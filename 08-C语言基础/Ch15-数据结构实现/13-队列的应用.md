# 队列的应用

## 1. 概述

队列的 FIFO 特性使其在许多场景中不可替代。本节介绍队列的四大经典应用：BFS（广度优先搜索）、任务调度、缓冲区管理、滑动窗口。

## 2. 应用一：广度优先搜索（BFS）

BFS 是队列最经典的应用。按层次遍历图或树，先访问离起点近的节点。

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

#define MAX_V 100

// 简易队列
typedef struct {
    int data[MAX_V];
    int front, rear;
} Queue;

void initQueue(Queue *q) { q->front = q->rear = 0; }
bool qEmpty(Queue *q) { return q->front == q->rear; }
void qPush(Queue *q, int x) { q->data[q->rear++] = x; }
int qPop(Queue *q) { return q->data[q->front++]; }
```

### 2.1 图的 BFS 遍历

```c
// 邻接矩阵表示的无向图 BFS
void BFS(int graph[][MAX_V], int n, int start) {
    int visited[MAX_V] = {0};
    Queue q;
    initQueue(&q);

    visited[start] = 1;
    qPush(&q, start);

    printf("BFS遍历: ");
    while (!qEmpty(&q)) {
        int v = qPop(&q);
        printf("%d ", v);

        // 将所有未访问的邻居入队
        for (int i = 0; i < n; i++) {
            if (graph[v][i] && !visited[i]) {
                visited[i] = 1;
                qPush(&q, i);
            }
        }
    }
    printf("\n");
}
```

### 2.2 二叉树层序遍历

```c
typedef struct TreeNode {
    int data;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

void levelOrder(TreeNode *root) {
    if (!root) return;

    TreeNode *queue[1000];
    int front = 0, rear = 0;

    queue[rear++] = root;

    while (front < rear) {
        int levelSize = rear - front;  // 当前层的节点数

        for (int i = 0; i < levelSize; i++) {
            TreeNode *node = queue[front++];
            printf("%d ", node->data);

            if (node->left) queue[rear++] = node->left;
            if (node->right) queue[rear++] = node->right;
        }
        printf("\n");  // 每层换行
    }
}
```

### 2.3 迷宫最短路径（BFS）

```c
// 在迷宫中找从起点到终点的最短路径
// 0 表示可通行，1 表示墙
int shortestPath(int maze[][MAX_V], int rows, int cols,
                 int sx, int sy, int ex, int ey) {
    int dx[] = {0, 0, 1, -1};  // 四个方向
    int dy[] = {1, -1, 0, 0};
    int dist[MAX_V][MAX_V];

    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            dist[i][j] = -1;

    Queue qx, qy;
    initQueue(&qx);
    initQueue(&qy);

    qPush(&qx, sx);
    qPush(&qy, sy);
    dist[sx][sy] = 0;

    while (!qEmpty(&qx)) {
        int x = qPop(&qx);
        int y = qPop(&qy);

        if (x == ex && y == ey) return dist[x][y];

        for (int d = 0; d < 4; d++) {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx >= 0 && nx < rows && ny >= 0 && ny < cols &&
                maze[nx][ny] == 0 && dist[nx][ny] == -1) {
                dist[nx][ny] = dist[x][y] + 1;
                qPush(&qx, nx);
                qPush(&qy, ny);
            }
        }
    }
    return -1;  // 无法到达
}
```

## 3. 应用二：任务调度

### 3.1 简单的轮转调度（Round Robin）

```c
typedef struct {
    char name[20];
    int burstTime;  // 执行时间
} Task;

void roundRobin(Task tasks[], int n, int timeSlice) {
    // 用数组模拟循环队列
    int queue[MAX_V];
    int front = 0, rear = n;
    int remaining[MAX_V];  // 剩余执行时间

    for (int i = 0; i < n; i++) {
        queue[i] = i;
        remaining[i] = tasks[i].burstTime;
    }

    int currentTime = 0;
    printf("轮转调度 (时间片=%d):\n", timeSlice);

    while (front != rear) {
        int idx = queue[front++];
        int execTime = (remaining[idx] < timeSlice) ?
                        remaining[idx] : timeSlice;

        printf("时刻 %d: 执行 %s %d 个时间单位\n",
               currentTime, tasks[idx].name, execTime);

        currentTime += execTime;
        remaining[idx] -= execTime;

        if (remaining[idx] > 0) {
            queue[rear++] = idx;  // 还没执行完，重新入队
        } else {
            printf("  -> %s 完成\n", tasks[idx].name);
        }
    }
}
```

## 4. 应用三：缓冲区

### 4.1 生产者-消费者模型

队列作为生产者和消费者之间的缓冲区。

```c
#define BUFFER_SIZE 5

typedef struct {
    int buffer[BUFFER_SIZE];
    int front, rear, count;
} CircularBuffer;

void initBuffer(CircularBuffer *cb) {
    cb->front = cb->rear = 0;
    cb->count = 0;
}

bool isBufferFull(CircularBuffer *cb) {
    return cb->count == BUFFER_SIZE;
}

bool isBufferEmpty(CircularBuffer *cb) {
    return cb->count == 0;
}

// 生产者：放入数据
bool produce(CircularBuffer *cb, int item) {
    if (isBufferFull(cb)) {
        printf("缓冲区满! 生产者等待...\n");
        return false;
    }
    cb->buffer[cb->rear] = item;
    cb->rear = (cb->rear + 1) % BUFFER_SIZE;
    cb->count++;
    printf("生产: %d (缓冲区: %d/%d)\n", item, cb->count, BUFFER_SIZE);
    return true;
}

// 消费者：取出数据
bool consume(CircularBuffer *cb, int *item) {
    if (isBufferEmpty(cb)) {
        printf("缓冲区空! 消费者等待...\n");
        return false;
    }
    *item = cb->buffer[cb->front];
    cb->front = (cb->front + 1) % BUFFER_SIZE;
    cb->count--;
    printf("消费: %d (缓冲区: %d/%d)\n", *item, cb->count, BUFFER_SIZE);
    return true;
}
```

## 5. 应用四：滑动窗口最大值

使用**单调递减队列**（双端队列思想）求滑动窗口最大值。

```c
// 滑动窗口最大值 - O(n) 时间复杂度
void slidingWindowMax(int arr[], int n, int k) {
    int deque[MAX_V];  // 存下标
    int front = 0, rear = 0;

    printf("窗口大小=%d 的最大值: ", k);

    for (int i = 0; i < n; i++) {
        // 移除超出窗口的元素
        while (front < rear && deque[front] <= i - k) {
            front++;
        }
        // 维护单调递减：移除所有比当前元素小的
        while (front < rear && arr[deque[rear - 1]] < arr[i]) {
            rear--;
        }
        deque[rear++] = i;

        // 窗口形成后输出最大值
        if (i >= k - 1) {
            printf("%d ", arr[deque[front]]);
        }
    }
    printf("\n");
}

// 输入: arr=[1,3,-1,-3,5,3,6,7], k=3
// 输出: 3, 3, 5, 5, 6, 7
```

## 6. 测试代码

```c
int main() {
    // 测试轮转调度
    Task tasks[] = {{"Task-A", 5}, {"Task-B", 3}, {"Task-C", 8}};
    roundRobin(tasks, 3, 2);

    // 测试缓冲区
    CircularBuffer cb;
    initBuffer(&cb);
    for (int i = 1; i <= 7; i++) produce(&cb, i * 10);
    int item;
    for (int i = 0; i < 3; i++) consume(&cb, &item);
    for (int i = 1; i <= 3; i++) produce(&cb, i * 100);

    // 测试滑动窗口
    int arr[] = {1, 3, -1, -3, 5, 3, 6, 7};
    slidingWindowMax(arr, 8, 3);

    return 0;
}
```

## 7. 重点总结

| 应用 | 核心思想 | 数据结构 |
|------|---------|---------|
| BFS | 按层遍历，先近后远 | 普通队列 |
| 轮转调度 | 时间片轮转 | 循环队列 |
| 缓冲区 | 生产者-消费者解耦 | 循环缓冲区 |
| 滑动窗口 | 维护单调队列 | 双端队列/单调队列 |

> **核心思想**：任何需要"按顺序处理"或"先来先服务"的场景，都应该优先考虑队列。
