# 循环链表

## 1. 概述

循环链表（Circular Linked List）是链表的一种变体，其**尾节点的指针不指向 NULL，而是指向头节点**（或其他节点），形成一个环。

### 两种循环链表

```
循环单链表:
head -> [A] -> [B] -> [C] ─┐
  └──────────────────────────┘

循环双链表:
head <-> [A] <-> [B] <-> [C] <-> head
```

## 2. 循环单链表实现

### 2.1 节点定义

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct CNode {
    int data;
    struct CNode *next;
} CNode;

typedef struct {
    CNode *head;
    int length;
} CircularList;
```

### 2.2 初始化与基本操作

```c
// 初始化
CircularList *createCList() {
    CircularList *list = (CircularList *)malloc(sizeof(CircularList));
    if (!list) return NULL;
    list->head = NULL;
    list->length = 0;
    return list;
}

// 尾部插入
int insertCTail(CircularList *list, int value) {
    CNode *newNode = (CNode *)malloc(sizeof(CNode));
    if (!newNode) return 0;
    newNode->data = value;

    if (list->head == NULL) {
        // 第一个节点，指向自己形成环
        list->head = newNode;
        newNode->next = newNode;
    } else {
        // 找到尾节点（next == head 的节点）
        CNode *tail = list->head;
        while (tail->next != list->head) {
            tail = tail->next;
        }
        tail->next = newNode;
        newNode->next = list->head;
    }
    list->length++;
    return 1;
}

// 头部插入
int insertCHead(CircularList *list, int value) {
    CNode *newNode = (CNode *)malloc(sizeof(CNode));
    if (!newNode) return 0;
    newNode->data = value;

    if (list->head == NULL) {
        list->head = newNode;
        newNode->next = newNode;
    } else {
        // 找到尾节点
        CNode *tail = list->head;
        while (tail->next != list->head) {
            tail = tail->next;
        }
        newNode->next = list->head;
        tail->next = newNode;
        list->head = newNode;
    }
    list->length++;
    return 1;
}
```

### 2.3 删除操作

```c
// 按值删除
int deleteCByValue(CircularList *list, int value) {
    if (list->head == NULL) return 0;

    CNode *p = list->head;
    CNode *prev = NULL;

    // 特殊情况：只有一个节点
    if (p->next == list->head && p->data == value) {
        free(p);
        list->head = NULL;
        list->length--;
        return 1;
    }

    // 遍历查找
    do {
        if (p->data == value) {
            if (prev == NULL) {
                // 删除的是头节点，需要找到尾节点
                CNode *tail = list->head;
                while (tail->next != list->head) {
                    tail = tail->next;
                }
                list->head = p->next;
                tail->next = list->head;
            } else {
                prev->next = p->next;
            }
            free(p);
            list->length--;
            return 1;
        }
        prev = p;
        p = p->next;
    } while (p != list->head);

    return 0;
}
```

### 2.4 遍历（注意终止条件）

```c
void printCList(CircularList *list) {
    if (list->head == NULL) {
        printf("链表为空\n");
        return;
    }
    printf("循环链表 [长度=%d]: ", list->length);
    CNode *p = list->head;
    do {
        printf("%d", p->data);
        p = p->next;
        if (p != list->head) printf(" -> ");
    } while (p != list->head);
    printf(" -> (回到头: %d)\n", list->head->data);
}
```

## 3. 循环双链表实现

```c
typedef struct CDNode {
    int data;
    struct CDNode *prev;
    struct CDNode *next;
} CDNode;

typedef struct {
    CDNode *head;
    int length;
} CircularDList;

// 初始化
CircularDList *createCDList() {
    CircularDList *list = (CircularDList *)malloc(sizeof(CircularDList));
    if (!list) return NULL;
    list->head = NULL;
    list->length = 0;
    return list;
}

// 尾部插入
int insertCDTail(CircularDList *list, int value) {
    CDNode *newNode = (CDNode *)malloc(sizeof(CDNode));
    if (!newNode) return 0;
    newNode->data = value;

    if (list->head == NULL) {
        newNode->next = newNode;
        newNode->prev = newNode;
        list->head = newNode;
    } else {
        CDNode *tail = list->head->prev;  // 头的 prev 就是尾
        newNode->next = list->head;
        newNode->prev = tail;
        tail->next = newNode;
        list->head->prev = newNode;
    }
    list->length++;
    return 1;
}

// 删除指定节点
int deleteCDNode(CircularDList *list, CDNode *node) {
    if (!node || list->head == NULL) return 0;
    node->prev->next = node->next;
    node->next->prev = node->prev;

    if (node == list->head) {
        if (node->next == node) {
            list->head = NULL;  // 只有一个节点
        } else {
            list->head = node->next;
        }
    }
    free(node);
    list->length--;
    return 1;
}
```

## 4. 经典应用：约瑟夫环问题

约瑟夫环（Josephus Problem）：n 个人围成一圈，从第一个人开始报数，报到 m 的人出列，然后从下一个人继续报数，直到所有人出列。求出列顺序。

### 使用循环链表解决

```c
void josephus(int n, int m) {
    // 构建循环链表 1, 2, 3, ..., n
    CircularList *list = createCList();
    for (int i = 1; i <= n; i++) {
        insertCTail(list, i);
    }

    printf("约瑟夫环出列顺序 (n=%d, m=%d): ", n, m);
    CNode *current = list->head;
    CNode *prev = NULL;

    while (list->length > 0) {
        // 数 m-1 步
        for (int i = 1; i < m; i++) {
            prev = current;
            current = current->next;
        }
        printf("%d", current->data);
        list->length--;
        if (list->length > 0) printf(" -> ");

        // 删除 current 节点
        if (current == list->head) {
            // 删除头节点，找到尾节点
            CNode *tail = list->head;
            while (tail->next != list->head) tail = tail->next;
            list->head = current->next;
            tail->next = list->head;
        } else {
            prev->next = current->next;
        }
        CNode *temp = current;
        current = current->next;
        free(temp);
    }
    printf("\n");
    free(list);
}

// 测试
// josephus(5, 3);  // 输出: 3 -> 1 -> 5 -> 2 -> 4
```

## 5. 操作复杂度

| 操作 | 循环单链表 | 循环双链表 |
|------|-----------|-----------|
| 头部插入 | O(n)* | O(1) |
| 尾部插入 | O(n) | O(1) |
| 已知节点删除 | O(n) | O(1) |

*循环单链表头部插入需要找尾节点，循环双链表通过 head->prev 直接得到尾节点。

## 6. 重点总结

- 循环链表的判空条件是 `head == NULL`
- 遍历循环链表的终止条件是 `p == head`（而不是 `p == NULL`）
- 循环双链表利用 `head->prev` 可以 O(1) 找到尾节点
- 约瑟夫环问题是循环链表的经典应用
- 实际应用中，操作系统的轮转调度（Round Robin）也用到循环链表

> **注意**：循环链表遍历时要特别注意终止条件，否则会陷入死循环。
