# 内存管理综合实例

## 一、综合实例：学生信息管理系统

本实例综合运用内存管理的各种技术，包括安全分配、动态数组、内存池、资源清理等。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

/* ============================================================
 *  第一部分：安全的动态字符串
 * ============================================================ */

typedef struct {
    char  *data;
    size_t length;
    size_t capacity;
} String;

String *string_create(const char *init) {
    String *s = malloc(sizeof(*s));
    if (!s) return NULL;

    size_t len = init ? strlen(init) : 0;
    size_t cap = len + 1;
    if (cap < 16) cap = 16;

    s->data = malloc(cap);
    if (!s->data) { free(s); return NULL; }

    if (init) strcpy(s->data, init);
    else s->data[0] = '\0';

    s->length = len;
    s->capacity = cap;
    return s;
}

int string_append(String *s, const char *str) {
    size_t len = strlen(str);
    if (s->length + len + 1 > s->capacity) {
        size_t new_cap = s->capacity;
        while (new_cap < s->length + len + 1) new_cap *= 2;
        char *new_data = realloc(s->data, new_cap);
        if (!new_data) return -1;
        s->data = new_data;
        s->capacity = new_cap;
    }
    strcpy(s->data + s->length, str);
    s->length += len;
    return 0;
}

void string_destroy(String *s) {
    if (s) {
        free(s->data);
        free(s);
    }
}

/* ============================================================
 *  第二部分：学生结构体与安全管理
 * ============================================================ */

typedef struct {
    int    id;
    String *name;
    double score;
    int    age;
} Student;

Student *student_create(int id, const char *name, double score, int age) {
    Student *stu = malloc(sizeof(*stu));
    if (!stu) return NULL;

    stu->name = string_create(name);
    if (!stu->name) {
        free(stu);
        return NULL;
    }

    stu->id = id;
    stu->score = score;
    stu->age = age;
    return stu;
}

void student_destroy(Student *stu) {
    if (stu) {
        string_destroy(stu->name);
        free(stu);
    }
}

void student_print(const Student *stu) {
    printf("  [%d] %s, 年龄%d, 成绩%.1f\n",
           stu->id, stu->name->data, stu->age, stu->score);
}

/* ============================================================
 *  第三部分：动态数组（学生集合）
 * ============================================================ */

typedef struct {
    Student **students;
    size_t     size;
    size_t     capacity;
} StudentArray;

StudentArray *sa_create(size_t initial_cap) {
    StudentArray *sa = malloc(sizeof(*sa));
    if (!sa) return NULL;

    sa->students = calloc(initial_cap, sizeof(Student *));
    if (!sa->students) { free(sa); return NULL; }

    sa->size = 0;
    sa->capacity = initial_cap;
    return sa;
}

int sa_add(StudentArray *sa, Student *stu) {
    if (sa->size >= sa->capacity) {
        size_t new_cap = sa->capacity * 2;
        Student **temp = realloc(sa->students,
                                  sizeof(Student *) * new_cap);
        if (!temp) return -1;

        // 初始化新分配的部分
        for (size_t i = sa->capacity; i < new_cap; i++) {
            temp[i] = NULL;
        }
        sa->students = temp;
        sa->capacity = new_cap;
    }
    sa->students[sa->size++] = stu;
    return 0;
}

Student *sa_get(StudentArray *sa, size_t index) {
    if (index >= sa->size) return NULL;
    return sa->students[index];
}

void sa_remove(StudentArray *sa, size_t index) {
    if (index >= sa->size) return;

    student_destroy(sa->students[index]);

    // 移动后续元素
    for (size_t i = index; i < sa->size - 1; i++) {
        sa->students[i] = sa->students[i + 1];
    }
    sa->students[sa->size - 1] = NULL;
    sa->size--;
}

void sa_destroy(StudentArray *sa) {
    if (!sa) return;

    for (size_t i = 0; i < sa->size; i++) {
        student_destroy(sa->students[i]);
    }
    free(sa->students);
    free(sa);
}

/* ============================================================
 *  第四部分：带统计功能的学生管理器
 * ============================================================ */

typedef struct {
    StudentArray *students;
    double        total_score;
    double        max_score;
    double        min_score;
    int           next_id;
} StudentManager;

StudentManager *sm_create(void) {
    StudentManager *sm = malloc(sizeof(*sm));
    if (!sm) return NULL;

    sm->students = sa_create(8);
    if (!sm->students) { free(sm); return NULL; }

    sm->total_score = 0;
    sm->max_score = -1;
    sm->min_score = 101;
    sm->next_id = 1;
    return sm;
}

int sm_add_student(StudentManager *sm, const char *name,
                    double score, int age) {
    Student *stu = student_create(sm->next_id++, name, score, age);
    if (!stu) return -1;

    if (sa_add(sm->students, stu) != 0) {
        student_destroy(stu);
        return -1;
    }

    // 更新统计
    sm->total_score += score;
    if (score > sm->max_score) sm->max_score = score;
    if (score < sm->min_score) sm->min_score = score;

    return stu->id;
}

double sm_average_score(const StudentManager *sm) {
    if (sm->students->size == 0) return 0;
    return sm->total_score / sm->students->size;
}

void sm_print_all(const StudentManager *sm) {
    printf("\n===== 学生列表 (%zu人) =====\n", sm->students->size);
    for (size_t i = 0; i < sm->students->size; i++) {
        student_print(sm->students->students[i]);
    }
    printf("\n统计: 平均分=%.1f, 最高=%.1f, 最低=%.1f\n",
           sm_average_score(sm), sm->max_score, sm->min_score);
}

void sm_destroy(StudentManager *sm) {
    if (sm) {
        sa_destroy(sm->students);
        free(sm);
    }
}

/* ============================================================
 *  第五部分：内存跟踪器
 * ============================================================ */

typedef struct {
    size_t alloc_count;
    size_t free_count;
    size_t total_allocated;
    size_t current_usage;
    size_t peak_usage;
} MemoryTracker;

static MemoryTracker g_tracker = {0};

void *tracked_alloc(size_t size) {
    void *ptr = malloc(size);
    if (ptr) {
        g_tracker.alloc_count++;
        g_tracker.total_allocated += size;
        g_tracker.current_usage += size;
        if (g_tracker.current_usage > g_tracker.peak_usage) {
            g_tracker.peak_usage = g_tracker.current_usage;
        }
    }
    return ptr;
}

void tracked_free(void *ptr, size_t size) {
    if (ptr) {
        free(ptr);
        g_tracker.free_count++;
        g_tracker.current_usage -= size;
    }
}

void print_memory_report(void) {
    printf("\n===== 内存使用报告 =====\n");
    printf("分配次数: %zu\n", g_tracker.alloc_count);
    printf("释放次数: %zu\n", g_tracker.free_count);
    printf("总分配量: %zu 字节\n", g_tracker.total_allocated);
    printf("当前使用: %zu 字节\n", g_tracker.current_usage);
    printf("峰值使用: %zu 字节\n", g_tracker.peak_usage);
    if (g_tracker.alloc_count != g_tracker.free_count) {
        printf("警告: 存在内存泄漏!\n");
    } else {
        printf("状态: 无内存泄漏\n");
    }
}

/* ============================================================
 *  第六部分：goto清理模式演示
 * ============================================================ */

int complex_operation(StudentManager *sm) {
    FILE *file = NULL;
    String *report = NULL;
    Student **temp_list = NULL;
    int result = 0;

    file = fopen("students_report.txt", "w");
    if (!file) { result = -1; goto cleanup; }

    report = string_create("学生成绩报告\n");
    if (!report) { result = -2; goto cleanup; }

    temp_list = malloc(sizeof(Student *) * sm->students->size);
    if (!temp_list) { result = -3; goto cleanup; }

    // 拷贝学生指针用于排序
    for (size_t i = 0; i < sm->students->size; i++) {
        temp_list[i] = sm->students->students[i];
    }

    // 写入报告
    char line[256];
    for (size_t i = 0; i < sm->students->size; i++) {
        snprintf(line, sizeof(line), "%s: %.1f分\n",
                 temp_list[i]->name->data, temp_list[i]->score);
        string_append(report, line);
    }

    fprintf(file, "%s", report->data);
    printf("报告已写入文件\n");
    result = 0;

cleanup:
    if (file) fclose(file);
    string_destroy(report);
    free(temp_list);
    return result;
}

/* ============================================================
 *  主程序：演示所有功能
 * ============================================================ */

int main(void) {
    printf("===== 学生信息管理系统 =====\n");
    printf("综合演示内存管理技术\n\n");

    // 创建管理器
    StudentManager *sm = sm_create();
    if (!sm) {
        fprintf(stderr, "创建管理器失败\n");
        return 1;
    }

    // 添加学生
    printf("添加学生...\n");
    sm_add_student(sm, "张三", 92.5, 20);
    sm_add_student(sm, "李四", 88.0, 21);
    sm_add_student(sm, "王五", 95.5, 19);
    sm_add_student(sm, "赵六", 76.0, 22);
    sm_add_student(sm, "孙八", 89.5, 20);

    // 打印所有学生
    sm_print_all(sm);

    // 删除一个学生
    printf("\n删除第2个学生(李四)...\n");
    sa_remove(sm->students, 1);

    // 再次打印
    sm_print_all(sm);

    // 执行复杂操作（goto清理演示）
    printf("\n生成报告...\n");
    complex_operation(sm);

    // 清理
    sm_destroy(sm);

    // 打印内存报告
    print_memory_report();

    return 0;
}
```

## 二、实例中用到的内存管理技术

```
本实例涵盖的内存管理技术:

  1. 安全分配模式     → 检查所有malloc返回值
  2. sizeof(*ptr)     → 避免类型不匹配
  3. 动态数组         → 自动扩容(realloc)
  4. 深拷贝字符串     → String结构体管理
  5. 递归释放         → sm_destroy → sa_destroy → student_destroy
  6. goto清理模式     → complex_operation函数
  7. 内存跟踪         → 统计分配/释放次数
  8. NULL安全         → 所有释放函数处理NULL参数
  9. 统一释放点       → 单一cleanup标签
 10. 边界检查         → sa_get检查索引
```

## 三、编译和运行

```bash
# 编译
gcc -Wall -Wextra -g -o student_mgr student_mgr.c

# 使用ASan检查
gcc -Wall -Wextra -g -fsanitize=address -o student_mgr_asan student_mgr.c

# 使用Valgrind检查
valgrind --leak-check=full --show-leak-kinds=all ./student_mgr
```

## 四、关键要点

> **综合实例要点**
> 1. 封装安全的数据结构(String, StudentArray)减少直接malloc使用
> 2. 构造函数/析构函数模式确保资源正确管理
> 3. goto清理模式确保所有错误路径都释放资源
> 4. 内存跟踪帮助发现泄漏
> 5. 递归释放保证嵌套数据结构正确清理
> 6. 始终使用ASan或Valgrind验证内存正确性
> 7. 良好的内存管理习惯需要在实践中不断巩固
