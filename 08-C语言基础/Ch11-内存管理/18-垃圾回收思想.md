# 垃圾回收思想

## 一、GC概念

垃圾回收(Garbage Collection, GC)是一种自动内存管理机制，程序不需要手动释放内存，GC会自动检测不再使用的内存并回收。

```
手动管理 vs 垃圾回收:

  C语言(手动):
    ptr = malloc(100);   // 分配
    使用 ptr;
    free(ptr);           // 必须手动释放

  垃圾回收(自动):
    ptr = allocate(100); // 分配
    使用 ptr;
    // 不需要free，GC自动回收
```

## 二、主要GC算法

### 2.1 引用计数 (Reference Counting)

每个对象维护一个引用计数，计数为0时回收：

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct RefObj {
    void *data;
    int ref_count;
    void (*destroy)(void *);
} RefObj;

RefObj *ref_create(size_t size, void (*destroy)(void *)) {
    RefObj *obj = malloc(sizeof(*obj));
    if (!obj) return NULL;
    obj->data = calloc(1, size);
    if (!obj->data) { free(obj); return NULL; }
    obj->ref_count = 1;
    obj->destroy = destroy ? destroy : free;
    return obj;
}

void ref_retain(RefObj *obj) {
    if (obj) obj->ref_count++;
}

void ref_release(RefObj **obj) {
    if (!obj || !*obj) return;
    (*obj)->ref_count--;
    if ((*obj)->ref_count == 0) {
        (*obj)->destroy((*obj)->data);
        free(*obj);
    }
    *obj = NULL;
}

void demo_refcount(void) {
    RefObj *a = ref_create(64, NULL);
    printf("创建: ref_count=%d\n", a->ref_count);

    ref_retain(a);
    printf("增加引用: ref_count=%d\n", a->ref_count);

    ref_release(&a);  // a=NULL, count=1
    printf("释放一次: a=%p\n", (void *)a);

    // 需要再释放一次才能真正回收
    // 但a已为NULL，需要保留副本
}

int main(void) {
    demo_refcount();
    return 0;
}
```

### 2.2 标记-清除 (Mark-Sweep)

从根对象出发，标记所有可达对象，清除未标记的对象：

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_OBJECTS 100

typedef struct GCObject {
    void *data;
    int marked;              // 标记位
    struct GCObject *refs[10]; // 引用的其他对象
    int ref_count;
} GCObject;

typedef struct {
    GCObject *objects[MAX_OBJECTS];
    int object_count;
    GCObject *roots[MAX_OBJECTS];  // 根对象
    int root_count;
} GCHeap;

GCHeap gc_heap = {0};

GCObject *gc_alloc(size_t size) {
    if (gc_heap.object_count >= MAX_OBJECTS) {
        fprintf(stderr, "GC堆已满\n");
        return NULL;
    }
    GCObject *obj = malloc(sizeof(*obj));
    if (!obj) return NULL;
    obj->data = calloc(1, size);
    obj->marked = 0;
    obj->ref_count = 0;
    gc_heap.objects[gc_heap.object_count++] = obj;
    return obj;
}

void gc_add_root(GCObject *obj) {
    gc_heap.roots[gc_heap.root_count++] = obj;
}

void gc_add_ref(GCObject *from, GCObject *to) {
    if (from->ref_count < 10) {
        from->refs[from->ref_count++] = to;
    }
}

// 标记阶段：递归标记所有可达对象
void gc_mark(GCObject *obj) {
    if (obj == NULL || obj->marked) return;
    obj->marked = 1;
    for (int i = 0; i < obj->ref_count; i++) {
        gc_mark(obj->refs[i]);
    }
}

// 清除阶段：释放未标记的对象
void gc_sweep(void) {
    int j = 0;
    for (int i = 0; i < gc_heap.object_count; i++) {
        GCObject *obj = gc_heap.objects[i];
        if (obj->marked) {
            obj->marked = 0;  // 重置标记
            gc_heap.objects[j++] = obj;
        } else {
            printf("  回收未标记对象 %p\n", (void *)obj);
            free(obj->data);
            free(obj);
        }
    }
    gc_heap.object_count = j;
}

void gc_collect(void) {
    printf("GC开始: %d个对象\n", gc_heap.object_count);
    // 标记阶段
    for (int i = 0; i < gc_heap.root_count; i++) {
        gc_mark(gc_heap.roots[i]);
    }
    // 清除阶段
    gc_sweep();
    printf("GC结束: %d个对象存活\n", gc_heap.object_count);
}

int main(void) {
    GCObject *root = gc_alloc(64);
    GCObject *child1 = gc_alloc(32);
    GCObject *child2 = gc_alloc(32);
    GCObject *orphan = gc_alloc(16);  // 孤儿对象

    gc_add_root(root);
    gc_add_ref(root, child1);
    gc_add_ref(root, child2);

    printf("执行GC前:\n");
    gc_collect();  // orphan被回收

    printf("\n断开child1的引用，再次GC:\n");
    root->ref_count = 1;  // 只保留child2
    gc_collect();  // child1被回收

    return 0;
}
```

## 三、C语言中的GC实践建议

```
1. 优先使用引用计数（简单、可预测）
2. 复杂对象图使用标记清除
3. 关键代码仍手动管理
4. 使用现有库：Boehm GC (libgc)
```

## 四、关键要点

> **GC思想要点**
> 1. 引用计数：简单高效，但无法处理循环引用
> 2. 标记清除：可以处理循环引用，但有停顿
> 3. C语言没有内置GC，需要手动实现或使用库
> 4. Boehm GC是C语言最常用的保守式GC库
> 5. 实时系统不适合GC（不可预测的停顿时间）
> 6. GC和手动管理可以混合使用
