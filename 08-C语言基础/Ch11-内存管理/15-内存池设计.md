# 内存池设计

## 一、什么是内存池

内存池(Memory Pool)是一种内存分配策略，预先分配一大块内存，然后从中分割出固定大小的小块供程序使用。相比频繁调用malloc/free，内存池能显著提升性能。

```
传统malloc/free:
  每次分配 → 搜索空闲链表 → 返回指针
  每次释放 → 标记空闲 → 合并空闲块
  开销: O(n)搜索 + 系统调用

内存池:
  初始化 → 预分配大块 → 切割成小块链表
  分配 → 从链表取一个块 → O(1)
  释放 → 放回链表 → O(1)
```

## 二、内存池的优势

| 特性 | malloc/free | 内存池 |
|------|------------|--------|
| 分配速度 | O(n)搜索 | O(1)直接取 |
| 释放速度 | 可能合并 | O(1)直接放回 |
| 内存碎片 | 可能产生 | 几乎没有 |
| 系统调用 | 每次可能 | 只在初始化时 |
| 缓存友好 | 不确定 | 高度局部 |
| 适用场景 | 通用 | 特定大小对象 |

## 三、固定大小内存池实现

### 3.1 基础实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define POOL_BLOCK_SIZE  64    // 每个块64字节
#define POOL_BLOCK_COUNT 100   // 共100个块

typedef struct MemoryPool {
    char   *memory;            // 大块内存的起始地址
    char   *free_list;         // 空闲块链表头
    size_t  block_size;        // 块大小
    size_t  block_count;       // 总块数
    size_t  free_count;        // 空闲块数
} MemoryPool;

// 初始化内存池
int pool_init(MemoryPool *pool, size_t block_size, size_t block_count) {
    if (block_size < sizeof(void *)) {
        block_size = sizeof(void *);  // 至少能容纳一个指针
    }

    size_t total_size = block_size * block_count;
    pool->memory = malloc(total_size);
    if (pool->memory == NULL) return -1;

    pool->block_size = block_size;
    pool->block_count = block_count;
    pool->free_count = block_count;

    // 构建空闲链表：每个块的前8字节存储下一个空闲块的地址
    pool->free_list = pool->memory;
    char *current = pool->memory;
    for (size_t i = 0; i < block_count - 1; i++) {
        *(char **)current = current + block_size;
        current += block_size;
    }
    *(char **)current = NULL;  // 最后一个块

    return 0;
}

// 从内存池分配一个块
void *pool_alloc(MemoryPool *pool) {
    if (pool->free_list == NULL) {
        return NULL;  // 池已满
    }

    void *block = pool->free_list;
    pool->free_list = *(char **)pool->free_list;  // 移动链表头
    pool->free_count--;

    return block;
}

// 释放一个块回内存池
void pool_free(MemoryPool *pool, void *block) {
    if (block == NULL) return;

    // 验证指针是否属于该池
    char *ptr = (char *)block;
    if (ptr < pool->memory ||
        ptr >= pool->memory + pool->block_size * pool->block_count) {
        fprintf(stderr, "警告：指针不属于该内存池\n");
        return;
    }

    // 将块放回空闲链表头部
    *(char **)block = pool->free_list;
    pool->free_list = (char *)block;
    pool->free_count++;
}

// 销毁内存池
void pool_destroy(MemoryPool *pool) {
    free(pool->memory);
    pool->memory = NULL;
    pool->free_list = NULL;
    pool->free_count = 0;
}

// 打印池状态
void pool_status(MemoryPool *pool) {
    printf("内存池状态:\n");
    printf("  块大小: %zu 字节\n", pool->block_size);
    printf("  总块数: %zu\n", pool->block_count);
    printf("  空闲块: %zu\n", pool->free_count);
    printf("  已使用: %zu\n", pool->block_count - pool->free_count);
}

int main(void) {
    MemoryPool pool;

    // 初始化内存池
    if (pool_init(&pool, POOL_BLOCK_SIZE, POOL_BLOCK_COUNT) != 0) {
        fprintf(stderr, "内存池初始化失败\n");
        return 1;
    }

    pool_status(&pool);

    // 分配几个块
    void *blocks[5];
    printf("\n分配5个块:\n");
    for (int i = 0; i < 5; i++) {
        blocks[i] = pool_alloc(&pool);
        printf("  分配块 %d: %p\n", i, blocks[i]);
    }
    pool_status(&pool);

    // 释放2个块
    printf("\n释放2个块:\n");
    pool_free(&pool, blocks[0]);
    pool_free(&pool, blocks[2]);
    blocks[0] = NULL;
    blocks[2] = NULL;
    pool_status(&pool);

    // 重新分配（应该复用刚释放的块）
    printf("\n重新分配2个块:\n");
    blocks[0] = pool_alloc(&pool);
    blocks[2] = pool_alloc(&pool);
    printf("  新块0: %p\n", blocks[0]);
    printf("  新块2: %p\n", blocks[2]);
    pool_status(&pool);

    // 清理
    for (int i = 0; i < 5; i++) {
        pool_free(&pool, blocks[i]);
    }
    pool_destroy(&pool);

    return 0;
}
```

## 四、性能对比

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define BLOCK_SIZE   64
#define BLOCK_COUNT  10000
#define ITERATIONS   1000000

// 内存池版本（使用上面的pool实现）
void benchmark_pool(void) {
    MemoryPool pool;
    pool_init(&pool, BLOCK_SIZE, BLOCK_COUNT);

    clock_t start = clock();

    for (int i = 0; i < ITERATIONS; i++) {
        void *ptr = pool_alloc(&pool);
        if (ptr) pool_free(&pool, ptr);
    }

    clock_t end = clock();
    printf("内存池: %.3f 秒 (%d 次分配释放)\n",
           (double)(end - start) / CLOCKS_PER_SEC, ITERATIONS);

    pool_destroy(&pool);
}

// malloc/free版本
void benchmark_malloc(void) {
    clock_t start = clock();

    for (int i = 0; i < ITERATIONS; i++) {
        void *ptr = malloc(BLOCK_SIZE);
        free(ptr);
    }

    clock_t end = clock();
    printf("malloc: %.3f 秒 (%d 次分配释放)\n",
           (double)(end - start) / CLOCKS_PER_SEC, ITERATIONS);
}

int main(void) {
    printf("性能对比:\n");
    benchmark_pool();
    benchmark_malloc();
    // 内存池通常比malloc快5-20倍
    return 0;
}
```

## 五、关键要点

> **内存池要点**
> 1. 内存池适合频繁分配/释放固定大小对象的场景
> 2. 分配和释放都是O(1)操作
> 3. 减少内存碎片和系统调用开销
> 4. 实现核心：用空闲链表管理未使用的块
> 5. 可以同时使用多个不同大小的内存池
> 6. 不适合大小变化很大的分配场景
> 7. 网络服务器、游戏引擎等高性能场景广泛使用
