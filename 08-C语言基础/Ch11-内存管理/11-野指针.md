# 野指针

## 一、什么是野指针

野指针(Wild Pointer)是指**未初始化**的指针，其值是不确定的随机内存地址。与悬空指针不同，野指针是从未被正确赋值过的指针。

```
野指针 vs 悬空指针:

  野指针(Wild Pointer):
    int *ptr;          // 未初始化，值随机
    *ptr = 42;         // 危险！可能写入任意内存

  悬空指针(Dangling Pointer):
    int *ptr = malloc(sizeof(int));
    free(ptr);         // ptr指向的内存已释放
    *ptr = 42;         // 危险！已释放的内存

  共同点：使用都会导致未定义行为
  区别：野指针从未被赋值，悬空指针曾经有效
```

## 二、野指针的产生原因

### 2.1 未初始化的局部指针

```c
#include <stdio.h>
#include <stdlib.h>

void wild_pointer_uninitialized(void) {
    int *ptr;  // 野指针！值是栈上的随机数据

    // 危险操作:
    // *ptr = 42;       // 可能写入任意内存位置
    // printf("%d\n", *ptr);  // 可能读取任意内存

    printf("未初始化指针的值: %p (随机值)\n", (void *)ptr);
    // 这个地址是不可预测的

    // 正确做法：声明时初始化
    int *safe_ptr = NULL;  // 初始化为NULL
    // int *safe_ptr = malloc(sizeof(int));  // 或分配内存

    if (safe_ptr != NULL) {
        *safe_ptr = 42;
        printf("安全指针: %d\n", *safe_ptr);
        free(safe_ptr);
    }
}

int main(void) {
    wild_pointer_uninitialized();
    return 0;
}
```

### 2.2 条件分支中未赋值

```c
#include <stdio.h>
#include <stdlib.h>

void wild_pointer_conditional(int mode) {
    int *ptr;  // 野指针

    if (mode == 1) {
        ptr = malloc(sizeof(int));
        *ptr = 100;
    } else if (mode == 2) {
        ptr = malloc(sizeof(int) * 10);
        ptr[0] = 200;
    }
    // 如果mode既不是1也不是2，ptr仍是野指针！

    // 危险：后续使用ptr
    // printf("%d\n", *ptr);  // mode!=1&&mode!=2时未定义行为!

    // 正确做法：初始化为NULL
    int *safe_ptr = NULL;

    if (mode == 1) {
        safe_ptr = malloc(sizeof(int));
        if (safe_ptr) *safe_ptr = 100;
    } else if (mode == 2) {
        safe_ptr = malloc(sizeof(int) * 10);
        if (safe_ptr) safe_ptr[0] = 200;
    }

    // 安全使用
    if (safe_ptr != NULL) {
        printf("safe_ptr = %d\n", *safe_ptr);
        free(safe_ptr);
    } else {
        printf("safe_ptr未分配\n");
    }
}

int main(void) {
    wild_pointer_conditional(0);  // 不分配的路径
    wild_pointer_conditional(1);
    wild_pointer_conditional(2);
    return 0;
}
```

### 2.3 未初始化的指针数组

```c
#include <stdio.h>
#include <stdlib.h>

void wild_pointer_array(void) {
    // 局部指针数组 - 每个元素都是野指针
    int *arr[10];

    // 危险：直接使用
    // *arr[0] = 42;  // 未定义行为

    // 正确做法：全部初始化
    for (int i = 0; i < 10; i++) {
        arr[i] = NULL;  // 初始化为NULL
    }

    // 按需分配
    for (int i = 0; i < 5; i++) {
        arr[i] = malloc(sizeof(int));
        if (arr[i]) *arr[i] = i * 10;
    }

    // 使用并释放
    for (int i = 0; i < 10; i++) {
        if (arr[i] != NULL) {
            printf("arr[%d] = %d\n", i, *arr[i]);
            free(arr[i]);
            arr[i] = NULL;
        }
    }
}

int main(void) {
    wild_pointer_array();
    return 0;
}
```

## 三、野指针的危害

```c
#include <stdio.h>
#include <stdlib.h>

void demonstrate_danger(void) {
    // 野指针可能指向：
    // 1. 未映射的内存 → 段错误(崩溃)
    // 2. 其他变量的内存 → 数据损坏
    // 3. 代码段 → 崩溃或奇怪行为
    // 4. 系统内存 → 安全漏洞

    // 示例：野指针破坏其他变量
    int important_data = 42;
    printf("important_data = %d\n", important_data);

    // int *wild;       // 野指针
    // *wild = 0;       // 可能正好覆盖important_data！

    // 使用Valgrind或ASan可以检测野指针访问
    printf("使用工具检测野指针:\n");
    printf("  gcc -fsanitize=address -g test.c\n");
    printf("  valgrind --tool=memcheck ./test\n");
}

int main(void) {
    demonstrate_danger();
    return 0;
}
```

## 四、避免野指针的编程习惯

```c
#include <stdio.h>
#include <stdlib.h>

// 习惯1：声明指针时立即初始化
void habit_initialize_on_declaration(void) {
    int *ptr = NULL;       // 好习惯
    // int *ptr;            // 坏习惯

    // 后续按需分配
    ptr = malloc(sizeof(int) * 100);
    if (ptr) {
        ptr[0] = 42;
        free(ptr);
        ptr = NULL;
    }
}

// 习惯2：函数参数中的输出指针
int allocate_buffer(char **out_buf, size_t size) {
    if (out_buf == NULL) return -1;

    *out_buf = malloc(size);
    if (*out_buf == NULL) return -1;

    (*out_buf)[0] = '\0';
    return 0;
}

// 习惯3：结构体中的指针成员
typedef struct {
    int *data;
    size_t len;
} Buffer;

Buffer buffer_create(size_t len) {
    Buffer buf = {NULL, 0};  // 初始化所有成员
    buf.data = malloc(sizeof(int) * len);
    if (buf.data) {
        buf.len = len;
    }
    return buf;
}

void buffer_destroy(Buffer *buf) {
    if (buf) {
        free(buf->data);
        buf->data = NULL;
        buf->len = 0;
    }
}

int main(void) {
    habit_initialize_on_declaration();

    char *buf = NULL;
    if (allocate_buffer(&buf, 256) == 0) {
        printf("缓冲区分配成功\n");
        free(buf);
    }

    Buffer b = buffer_create(100);
    if (b.data) {
        b.data[0] = 42;
        printf("buffer[0] = %d\n", b.data[0]);
    }
    buffer_destroy(&b);

    return 0;
}
```

## 五、野指针与悬空指针对比

| 特性 | 野指针 | 悬空指针 |
|------|--------|---------|
| 产生原因 | 未初始化 | free后继续持有 |
| 指向位置 | 随机地址 | 原分配地址(已释放) |
| 危害程度 | 极高(可能指向任意位置) | 高(指向已释放内存) |
| 检测难度 | 较难 | 较难 |
| 预防方法 | 声明时初始化为NULL | free后置NULL |

## 六、关键要点

> **野指针要点**
> 1. 声明指针时**立即初始化**为NULL
> 2. 条件分支中确保所有路径都给指针赋值
> 3. 指针数组在使用前全部初始化
> 4. 结构体创建时初始化所有指针成员为NULL
> 5. 野指针比悬空指针更危险（可能指向任意地址）
> 6. 使用AddressSanitizer或Valgrind检测野指针访问
> 7. 养成"先初始化再使用"的编程习惯
