# 栈内存

## 一、栈内存基本概念

栈(Stack)是程序运行时用于存储局部变量、函数参数和返回地址的内存区域。它由编译器自动管理，遵循**后进先出(LIFO)** 原则。

```
栈内存的核心特征:
┌─────────────────────────────┐
│  • 自动分配和释放             │
│  • 速度极快（只需移动指针）    │
│  • 空间有限（通常1-8MB）      │
│  • 生命周期与作用域绑定       │
│  • 向低地址方向增长           │
└─────────────────────────────┘
```

## 二、栈帧 (Stack Frame)

每次函数调用都会在栈上创建一个**栈帧**，用于存储该函数的局部数据：

```c
#include <stdio.h>

void inner_function(int c, int d) {
    int local_inner = c + d;
    printf("  inner: local_inner = %d\n", local_inner);
    printf("  inner: 参数 c 地址 = %p\n", (void *)&c);
    printf("  inner: 局部变量地址 = %p\n", (void *)&local_inner);
}

void outer_function(int a, int b) {
    int local_outer = a * b;
    printf("outer: local_outer = %d\n", local_outer);
    printf("outer: 参数 a 地址 = %p\n", (void *)&a);
    printf("outer: 局部变量地址 = %p\n", (void *)&local_outer);

    inner_function(a, b);
    // inner_function 的栈帧已回收
}

int main(void) {
    outer_function(3, 4);
    // outer_function 的栈帧已回收
    return 0;
}
```

### 栈帧结构示意

```
调用 inner_function 时的栈:

高地址
┌──────────────────────────────┐
│      main() 的栈帧           │
│  - 返回地址                   │
│  - 局部变量                   │
├──────────────────────────────┤
│      outer_function() 栈帧    │
│  - 返回地址 (指向main)        │
│  - 参数 a=3, b=4             │
│  - local_outer = 12          │
├──────────────────────────────┤
│      inner_function() 栈帧    │  ← 当前栈帧
│  - 返回地址 (指向outer)       │
│  - 参数 c=3, d=4             │
│  - local_inner = 7           │
│  [ESP → 栈顶]                │
└──────────────────────────────┘
低地址

inner_function 返回后，其栈帧被释放，ESP回到outer的栈帧
```

## 三、局部变量的自动管理

### 3.1 自动分配与释放

```c
#include <stdio.h>

void demonstrate_auto_management(void) {
    // 进入函数时，编译器自动在栈上分配空间
    int x = 10;
    int y = 20;
    char buffer[64] = "Hello";

    printf("x = %d, y = %d\n", x, y);
    printf("buffer = %s\n", buffer);

    // 函数返回时，编译器自动回收栈帧
    // x, y, buffer 的内存全部释放
    // 不需要调用 free！
}

void demonstrate_scope(void) {
    int outer_var = 100;

    {
        // 进入新作用域，分配inner_var
        int inner_var = 200;
        printf("inner: outer=%d, inner=%d\n", outer_var, inner_var);
    }
    // inner_var 已被释放，不能再访问
    // printf("%d\n", inner_var);  // 编译错误！

    printf("outer: outer=%d\n", outer_var);
}

int main(void) {
    demonstrate_auto_management();
    demonstrate_scope();
    return 0;
}
```

### 3.2 栈上的数组

```c
#include <stdio.h>
#include <string.h>

void stack_arrays(void) {
    // 固定大小的数组在栈上分配
    int numbers[100];
    char name[64];

    // 初始化
    memset(numbers, 0, sizeof(numbers));
    strcpy(name, "C语言学习者");

    printf("numbers大小: %zu 字节\n", sizeof(numbers));
    printf("name内容: %s\n", name);

    // 注意：不能用变量作为数组大小(C89)
    // C99支持变长数组(VLA):
    int n = 50;
    int vla[n];  // C99变长数组，也在栈上分配
    printf("VLA大小: %zu 字节\n", sizeof(vla));

    // 函数返回时全部自动释放
}

int main(void) {
    stack_arrays();
    return 0;
}
```

## 四、栈的速度优势

栈操作远快于堆操作，因为只需移动栈指针：

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define ITERATIONS 1000000
#define SIZE 1024

void benchmark_stack(void) {
    clock_t start = clock();

    for (int i = 0; i < ITERATIONS; i++) {
        char buffer[SIZE];       // 栈上分配（极快）
        buffer[0] = 'A';         // 使用
        // 函数结束自动释放     // 自动释放（极快）
    }

    clock_t end = clock();
    printf("栈分配 %d 次: %.3f 秒\n",
           ITERATIONS, (double)(end - start) / CLOCKS_PER_SEC);
}

void benchmark_heap(void) {
    clock_t start = clock();

    for (int i = 0; i < ITERATIONS; i++) {
        char *buffer = malloc(SIZE);  // 堆上分配（较慢）
        buffer[0] = 'A';              // 使用
        free(buffer);                 // 手动释放
    }

    clock_t end = clock();
    printf("堆分配 %d 次: %.3f 秒\n",
           ITERATIONS, (double)(end - start) / CLOCKS_PER_SEC);
}

int main(void) {
    printf("性能对比:\n");
    benchmark_stack();
    benchmark_heap();
    // 栈分配通常比堆分配快10-100倍
    return 0;
}
```

## 五、栈大小限制

### 5.1 默认栈大小

不同系统和编译器的默认栈大小不同：

| 平台/编译器 | 默认栈大小 |
|------------|-----------|
| Linux (gcc) | 8 MB |
| Windows (MSVC) | 1 MB |
| macOS (clang) | 8 MB |
| 嵌入式系统 | 4-64 KB |

```c
#include <stdio.h>

// 探测栈大小的方法
void probe_stack_size(void) {
    // 通过大数组消耗栈空间来大致探测
    char probe[1024 * 1024];  // 1MB
    printf("成功分配 1MB 栈空间\n");
    probe[0] = 1;  // 防止优化掉
}

int main(void) {
    // 在Linux下可以用命令查看: ulimit -s
    // 在Windows下默认1MB
    printf("尝试探测栈大小...\n");
    probe_stack_size();
    return 0;
}
```

### 5.2 栈溢出示例

```c
#include <stdio.h>

// 危险：无限递归导致栈溢出
void infinite_recursion(int n) {
    int local_array[1000];  // 每次调用占用4KB
    printf("递归深度: %d\n", n);
    infinite_recursion(n + 1);  // 最终栈溢出!
}

// 安全：限制递归深度的版本
void safe_recursion(int n, int max_depth) {
    if (n >= max_depth) {
        printf("达到最大深度 %d，停止递归\n", max_depth);
        return;
    }
    int local_data[100];
    local_data[0] = n;
    safe_recursion(n + 1, max_depth);
}

int main(void) {
    // infinite_recursion(0);  // 取消注释会导致崩溃

    safe_recursion(0, 100);  // 安全的递归
    return 0;
}
```

## 六、栈与堆的对比

```c
#include <stdio.h>
#include <stdlib.h>

// 返回栈上局部变量的地址 - 危险！
int *dangerous_function(void) {
    int local_var = 42;
    return &local_var;  // 返回后 local_var 已被释放！
}

// 正确：返回堆上分配的内存
int *safe_function(void) {
    int *heap_var = malloc(sizeof(int));
    *heap_var = 42;
    return heap_var;  // 调用者需要 free
}

// 正确：返回静态变量的地址
int *also_safe_function(void) {
    static int static_var = 42;
    return &static_var;  // 静态变量生命周期持续到程序结束
}

void compare_stack_heap(void) {
    // 栈分配
    int stack_array[100];
    // 优点：自动管理、速度极快
    // 缺点：大小固定、空间有限

    // 堆分配
    int *heap_array = malloc(100 * sizeof(int));
    // 优点：大小可变、空间大
    // 缺点：需要手动管理、速度较慢

    printf("栈数组地址: %p\n", (void *)stack_array);
    printf("堆数组地址: %p\n", (void *)heap_array);
    printf("栈数组大小: %zu 字节\n", sizeof(stack_array));
    // printf("堆数组大小: %zu\n", sizeof(heap_array));  // 错误！指针大小

    free(heap_array);
}

int main(void) {
    // 演示危险用法
    // int *ptr = dangerous_function();
    // printf("%d\n", *ptr);  // 未定义行为！

    // 正确用法
    int *ptr = safe_function();
    printf("堆变量值: %d\n", *ptr);
    free(ptr);

    compare_stack_heap();
    return 0;
}
```

## 七、栈内存的优化技巧

### 7.1 减小局部数组大小

```c
// 不好：在栈上分配过大的数组
void bad_example(void) {
    char huge_buffer[1024 * 1024];  // 1MB 在栈上
    // 可能导致栈溢出
}

// 好：大数组使用堆分配
void good_example(void) {
    char *huge_buffer = malloc(1024 * 1024);  // 1MB 在堆上
    if (!huge_buffer) return;
    // 使用...
    free(huge_buffer);
}
```

### 7.2 使用指针传递大数据

```c
// 不好：结构体值传递（拷贝整个结构体到栈上）
typedef struct {
    int data[1000];
    char name[256];
} LargeStruct;

void process_by_value(LargeStruct s) {  // 拷贝4KB+到栈上
    printf("%d\n", s.data[0]);
}

// 好：指针传递（只传递8字节指针）
void process_by_pointer(const LargeStruct *s) {  // 只传递指针
    printf("%d\n", s->data[0]);
}

int main(void) {
    LargeStruct s = {{0}, "test"};
    process_by_pointer(&s);  // 推荐
    // process_by_value(s);  // 不推荐
    return 0;
}
```

## 八、关键要点

> **栈内存的核心要点**
> 1. 栈内存由编译器自动管理，无需手动释放
> 2. 栈上变量的生命周期等于其作用域
> 3. 栈操作速度极快，适合小型临时变量
> 4. 栈空间有限(通常1-8MB)，大数组应使用堆分配
> 5. 不要返回局部变量的地址（函数返回后栈帧已释放）
> 6. 递归深度过深会导致栈溢出
> 7. 大型结构体应通过指针传递而非值传递
