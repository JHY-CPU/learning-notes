# 内存调试工具

## 一、工具概览

C语言的内存错误难以通过代码审查发现，需要借助专业工具：

| 工具 | 平台 | 功能 | 性能开销 |
|------|------|------|---------|
| Valgrind | Linux | 内存泄漏、越界、UAF | 高(10-50x) |
| AddressSanitizer | 跨平台 | 越界、UAF、泄漏 | 中(2x) |
| MemorySanitizer | Linux | 未初始化内存 | 中(3x) |
| mtrace | glibc | 简单泄漏检测 | 低 |
| Dr.Memory | Windows/Linux | 类似Valgrind | 高 |

## 二、Valgrind

### 2.1 基本使用

```bash
# 编译（加入调试信息）
gcc -g -o test test.c

# 基本内存检查
valgrind ./test

# 详细泄漏检查
valgrind --leak-check=full ./test

# 显示所有错误详情
valgrind --leak-check=full --show-leak-kinds=all ./test

# 跟踪未初始化值的来源
valgrind --track-origins=yes ./test
```

### 2.2 Valgrind检测示例

```c
/* test_valgrind.c - 编译: gcc -g -o test_valgrind test_valgrind.c */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 故意制造各种内存错误供Valgrind检测

void test_leak(void) {
    // 内存泄漏：分配未释放
    char *ptr = malloc(100);
    ptr[0] = 'A';  // 使用，防止优化掉
    // 忘记 free(ptr)
}

void test_use_after_free(void) {
    int *ptr = malloc(sizeof(int));
    *ptr = 42;
    free(ptr);
    // printf("%d\n", *ptr);  // use-after-free
}

void test_uninitialized(void) {
    int *arr = malloc(sizeof(int) * 10);
    int sum = 0;
    for (int i = 0; i < 10; i++) {
        sum += arr[i];  // 使用未初始化的值
    }
    printf("sum = %d\n", sum);
    free(arr);
}

void test_overflow(void) {
    char *buf = malloc(10);
    buf[10] = 'X';  // 越界访问
    free(buf);
}

void test_double_free(void) {
    int *ptr = malloc(sizeof(int));
    free(ptr);
    // free(ptr);  // double free
}

int main(void) {
    printf("=== Valgrind 测试 ===\n");
    test_leak();
    // test_use_after_free();
    // test_uninitialized();
    // test_overflow();
    // test_double_free();
    printf("测试完成\n");
    return 0;
}
```

## 三、AddressSanitizer (ASan)

### 3.1 编译和使用

```bash
# 编译时启用ASan
gcc -fsanitize=address -g -o test test.c

# 运行
./test

# 也可用于C++
g++ -fsanitize=address -g -o test test.cpp
```

### 3.2 ASan检测示例

```c
/* test_asan.c - 编译: gcc -fsanitize=address -g -o test_asan test_asan.c */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void test_stack_buffer_overflow(void) {
    char buffer[8];
    // strcpy(buffer, "Hello World!");  // 栈溢出 - ASan会报告
    strcpy(buffer, "Hello");  // 安全
    printf("buffer: %s\n", buffer);
}

void test_heap_buffer_overflow(void) {
    char *buffer = malloc(8);
    if (!buffer) return;
    // buffer[8] = 'X';  // 堆溢出 - ASan会报告
    buffer[7] = '\0';
    free(buffer);
}

void test_use_after_free(void) {
    int *ptr = malloc(sizeof(int));
    *ptr = 42;
    free(ptr);
    // printf("%d\n", *ptr);  // UAF - ASan会报告
}

void test_leak(void) {
    int *ptr = malloc(sizeof(int) * 100);
    ptr[0] = 1;
    // 忘记 free(ptr) - ASan会报告泄漏
}

int main(void) {
    test_stack_buffer_overflow();
    test_heap_buffer_overflow();
    // test_use_after_free();
    // test_leak();
    printf("ASan测试完成\n");
    return 0;
}
```

ASan输出示例：
```

READ of size 13 at 0x7ffd... thread T0
    #0 0x... in strcpy
    #1 0x... in test_stack_buffer_overflow test_asan.c:8
```

## 四、mtrace (glibc)

### 4.1 基本使用

```c
#include <stdio.h>
#include <stdlib.h>
#include <mtrace.h>

int main(void) {
    // 开始跟踪
    mtrace();

    int *a = malloc(sizeof(int) * 10);
    int *b = malloc(sizeof(int) * 20);

    free(a);
    // 忘记 free(b) - mtrace会报告

    // 结束跟踪
    muntrace();

    return 0;
}
```

```bash
# 设置跟踪文件并运行
export MALLOC_TRACE=mtrace.log
./test

# 分析结果
mtrace test mtrace.log
```

输出示例：

## 五、综合使用示例

### 5.1 Makefile集成

```makefile
CC = gcc
CFLAGS = -Wall -Wextra -g

# 普通编译
all: program

program: main.c
	$(CC) $(CFLAGS) -o $@ $<

# ASan编译
asan: main.c
	$(CC) $(CFLAGS) -fsanitize=address -o program_asan $<

# Valgrind检查
valgrind: program
	valgrind --leak-check=full --show-leak-kinds=all ./program

# 清理
clean:
	rm -f program program_asan
```

### 5.2 自定义内存检查包装器

```c
#include <stdio.h>
#include <stdlib.h>

// 简单的内存使用统计
static size_t g_total_allocated = 0;
static size_t g_total_freed = 0;
static size_t g_current_usage = 0;
static size_t g_peak_usage = 0;

void *debug_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    if (ptr) {
        g_total_allocated += size;
        g_current_usage += size;
        if (g_current_usage > g_peak_usage) {
            g_peak_usage = g_current_usage;
        }
        printf("[MALLOC] %zu bytes at %p (%s:%d) [cur=%zu]\n",
               size, ptr, file, line, g_current_usage);
    }
    return ptr;
}

void debug_free(void *ptr, size_t size, const char *file, int line) {
    if (ptr) {
        free(ptr);
        g_total_freed += size;
        g_current_usage -= size;
        printf("[FREE]   %zu bytes at %p (%s:%d) [cur=%zu]\n",
               size, ptr, file, line, g_current_usage);
    }
}

void print_memory_report(void) {
    printf("\n===== 内存使用报告 =====\n");
    printf("总分配: %zu 字节\n", g_total_allocated);
    printf("总释放: %zu 字节\n", g_total_freed);
    printf("当前使用: %zu 字节\n", g_current_usage);
    printf("峰值使用: %zu 字节\n", g_peak_usage);
    if (g_current_usage > 0) {
        printf("警告: %zu 字节未释放!\n", g_current_usage);
    }
}

// 使用宏简化调用
#define MALLOC(size) debug_malloc(size, __FILE__, __LINE__)
#define FREE(ptr, size) debug_free(ptr, size, __FILE__, __LINE__)

int main(void) {
    int *a = MALLOC(sizeof(*a) * 100);
    char *b = MALLOC(256);

    FREE(a, sizeof(*a) * 100);
    // 故意不释放 b

    print_memory_report();

    FREE(b, 256);  // 补释放
    print_memory_report();

    return 0;
}
```

## 六、关键要点

> **工具使用要点**
> 1. 开发阶段始终使用 `-fsanitize=address` 编译
> 2. Linux下用Valgrind做全面检查（性能开销大，不适合生产环境）
> 3. ASan开销较小，可集成到CI/CD中
> 4. mtrace适合简单的泄漏检测
> 5. 编译时加 `-g` 以获取文件名和行号信息
> 6. 自定义内存跟踪器可以帮助定位特定问题
> 7. 定期运行工具检查，不要等到出问题才用
