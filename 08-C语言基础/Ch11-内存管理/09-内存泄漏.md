# 内存泄漏

## 一、什么是内存泄漏

内存泄漏(Memory Leak)是指程序中已动态分配的堆内存由于某种原因未被释放或无法释放，造成系统内存的浪费。

```
内存泄漏的过程:
程序启动                   程序运行中                长时间运行后
  │                         │                        │
  ├─ 分配100KB ──────────── ├─ 再分配100KB ────────── ├─ 累积数GB
  │   (未释放)              │   (未释放)              │   程序崩溃
  │                         │                        │
  ▼                         ▼                        ▼
可用内存充裕              可用内存减少             内存耗尽/系统变慢
```

## 二、常见的内存泄漏场景

### 2.1 忘记释放

```c
#include <stdio.h>
#include <stdlib.h>

// 场景1：最简单的泄漏 - 忘记free
void leak_simple(void) {
    int *data = malloc(sizeof(int) * 1000);
    if (data == NULL) return;

    data[0] = 42;
    printf("data[0] = %d\n", data[0]);

    // 忘记 free(data)！
    // 函数返回后，data指针丢失，内存无法释放
}

// 场景2：早期返回导致泄漏
void leak_early_return(int condition) {
    int *buffer = malloc(1024);
    if (buffer == NULL) return;

    if (condition) {
        return;  // 泄漏！buffer没有被释放
    }

    // 使用buffer...
    free(buffer);
}

// 场景3：异常路径泄漏
void leak_error_path(void) {
    int *a = malloc(sizeof(int) * 100);
    int *b = malloc(sizeof(int) * 200);

    if (a == NULL || b == NULL) {
        // 如果a分配成功但b失败，a会泄漏
        // 应该：if (a) free(a); if (b) free(b);
        return;
    }

    free(a);
    free(b);
}

int main(void) {
    leak_simple();
    leak_early_return(1);
    leak_error_path();
    printf("注意：以上函数都有内存泄漏\n");
    return 0;
}
```

### 2.2 指针覆盖

```c
#include <stdio.h>
#include <stdlib.h>

void leak_pointer_overwrite(void) {
    int *ptr = malloc(sizeof(int) * 100);
    if (ptr == NULL) return;

    ptr[0] = 42;

    // 错误：让ptr指向新内存，旧内存泄漏
    ptr = malloc(sizeof(int) * 200);
    // 原来的100个int无法释放了！

    free(ptr);  // 只释放了第二次分配的
}

// 正确做法
void no_leak_pointer_overwrite(void) {
    int *ptr = malloc(sizeof(int) * 100);
    if (ptr == NULL) return;

    ptr[0] = 42;

    // 先释放旧内存
    free(ptr);

    // 再分配新内存
    ptr = malloc(sizeof(int) * 200);
    if (ptr == NULL) return;

    free(ptr);
}

int main(void) {
    leak_pointer_overwrite();
    no_leak_pointer_overwrite();
    return 0;
}
```

### 2.3 循环中的泄漏

```c
#include <stdio.h>
#include <stdlib.h>

void leak_in_loop(void) {
    // 每次迭代都泄漏1024字节
    for (int i = 0; i < 100; i++) {
        char *buffer = malloc(1024);
        if (buffer == NULL) continue;

        sprintf(buffer, "迭代 %d", i);
        printf("%s\n", buffer);

        // 忘记 free(buffer)！
        // 100次迭代 = 100KB 泄漏
    }
}

// 正确做法
void no_leak_in_loop(void) {
    for (int i = 0; i < 100; i++) {
        char *buffer = malloc(1024);
        if (buffer == NULL) continue;

        sprintf(buffer, "迭代 %d", i);
        printf("%s\n", buffer);

        free(buffer);  // 每次迭代释放
    }
}

int main(void) {
    leak_in_loop();
    no_leak_in_loop();
    return 0;
}
```

## 三、内存泄漏的检测

### 3.1 使用Valgrind检测

```bash
# 编译时加入调试信息
gcc -g -o program program.c

# 使用Valgrind检测内存泄漏
valgrind --leak-check=full ./program
```

### 3.2 简单的内存跟踪器

```c
#include <stdio.h>
#include <stdlib.h>

// 简单的内存分配跟踪
static size_t total_allocated = 0;
static size_t total_freed = 0;
static size_t alloc_count = 0;
static size_t free_count = 0;

void *tracked_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    if (ptr != NULL) {
        total_allocated += size;
        alloc_count++;
        printf("[ALLOC] %zu bytes at %p (%s:%d)\n",
               size, ptr, file, line);
    }
    return ptr;
}

void tracked_free(void *ptr, const char *file, int line) {
    if (ptr != NULL) {
        free(ptr);
        free_count++;
        printf("[FREE]  at %p (%s:%d)\n", ptr, file, line);
    }
}

// 宏封装
#define TRACKED_MALLOC(size) tracked_malloc(size, __FILE__, __LINE__)
#define TRACKED_FREE(ptr)    do { tracked_free(ptr, __FILE__, __LINE__); ptr = NULL; } while(0)

// 打印泄漏报告
void print_leak_report(void) {
    printf("\n===== 内存泄漏报告 =====\n");
    printf("总分配: %zu 次, %zu 字节\n", alloc_count, total_allocated);
    printf("总释放: %zu 次\n", free_count);
    if (alloc_count > free_count) {
        printf("警告: 可能有 %zu 次分配未释放!\n",
               alloc_count - free_count);
    } else {
        printf("无内存泄漏\n");
    }
}

// 测试
int main(void) {
    int *a = TRACKED_MALLOC(sizeof(int) * 100);
    char *b = TRACKED_MALLOC(256);

    TRACKED_FREE(a);
    // 故意不释放 b，模拟泄漏

    int *c = TRACKED_MALLOC(sizeof(double) * 50);
    TRACKED_FREE(c);

    print_leak_report();  // 应报告1次泄漏
    TRACKED_FREE(b);  // 释放b
    return 0;
}
```

## 四、预防内存泄漏的模式

### 4.1 goto清理模式

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int process_file(const char *filename) {
    FILE *file = NULL;
    char *buffer = NULL;
    int *result = NULL;
    int ret = 0;

    file = fopen(filename, "r");
    if (file == NULL) {
        ret = -1;
        goto cleanup;
    }

    buffer = malloc(1024);
    if (buffer == NULL) {
        ret = -2;
        goto cleanup;
    }

    result = malloc(sizeof(int) * 100);
    if (result == NULL) {
        ret = -3;
        goto cleanup;
    }

    // 使用所有资源...
    printf("处理文件: %s\n", filename);

cleanup:
    // 统一清理，free(NULL)和fclose(NULL)都安全
    if (file) fclose(file);
    free(buffer);
    free(result);
    return ret;
}

int main(void) {
    process_file("test.txt");
    return 0;
}
```

### 4.2 RAII风格清理

```c
#include <stdio.h>
#include <stdlib.h>

// GCC/Clang的cleanup属性实现RAII
#if defined(__GNUC__) || defined(__clang__)

#define CLEANUP_FREE __attribute__((cleanup(auto_free)))

void auto_free(void *p) {
    free(*(void **)p);
}

void raii_example(void) {
    // 使用cleanup属性，函数退出时自动free
    CLEANUP_FREE char *buf1 = malloc(256);
    CLEANUP_FREE char *buf2 = malloc(512);

    if (!buf1 || !buf2) return;  // 自动清理

    strcpy(buf1, "自动释放");
    printf("%s\n", buf1);

    // 无论从哪个出口离开，都会自动释放
}

#endif

int main(void) {
#if defined(__GNUC__) || defined(__clang__)
    raii_example();
#endif
    return 0;
}
```

## 五、关键要点

> **内存泄漏要点**
> 1. 每次malloc必须有对应的free
> 2. 所有代码路径（包括错误路径）都要释放内存
> 3. 不要用新指针覆盖旧指针而不释放旧内存
> 4. 循环中分配的内存要在循环内释放
> 5. 使用goto清理模式确保所有错误路径都释放
> 6. 使用工具(Valgrind/ASan)定期检测泄漏
> 7. 实现简单的内存跟踪器辅助调试
