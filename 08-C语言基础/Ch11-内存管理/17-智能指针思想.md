# 智能指针思想

## 一、RAII概念

RAII(Resource Acquisition Is Initialization)是C++中的核心思想：在构造函数中获取资源，在析构函数中释放资源。C语言没有析构函数，但可以模拟这一思想。

```
RAII的核心思想:
  获取资源 = 初始化对象
  释放资源 = 对象销毁

  C++: 构造函数malloc, 析构函数free → 自动管理
  C:  需要手动模拟这种模式
```

## 二、C语言中模拟RAII

### 2.1 GCC cleanup属性

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#if defined(__GNUC__) || defined(__clang__)

// cleanup属性：变量离开作用域时自动调用指定函数
#define AUTO_FREE __attribute__((cleanup(auto_free)))

void auto_free(void *p) {
    void **pp = (void **)p;
    if (*pp) {
        printf("  [auto-free] 释放 %p\n", *pp);
        free(*pp);
        *pp = NULL;
    }
}

void raii_demo(void) {
    printf("进入函数\n");

    // 使用cleanup属性，函数退出时自动释放
    AUTO_FREE char *buffer = malloc(256);
    AUTO_FREE int *data = malloc(sizeof(int) * 100);

    if (!buffer || !data) {
        printf("分配失败，自动清理\n");
        return;  // 自动释放已分配的内存
    }

    strcpy(buffer, "RAII自动管理");
    data[0] = 42;

    printf("buffer: %s\n", buffer);
    printf("data[0]: %d\n", data[0]);

    // 正常返回时自动释放
    printf("函数返回，自动释放:\n");
}

// 文件句柄的RAII管理
#define AUTO_FILE __attribute__((cleanup(auto_fclose)))

void auto_fclose(FILE **fp) {
    if (*fp) {
        printf("  [auto-fclose] 关闭文件\n");
        fclose(*fp);
        *fp = NULL;
    }
}

void file_raii_demo(const char *filename) {
    AUTO_FILE FILE *fp = fopen(filename, "r");
    if (!fp) {
        printf("文件打开失败\n");
        return;  // 自动关闭（已为NULL）
    }

    char line[256];
    if (fgets(line, sizeof(line), fp)) {
        printf("读取: %s", line);
    }

    // 函数结束时自动 fclose
}

#endif

int main(void) {
#if defined(__GNUC__) || defined(__clang__)
    raii_demo();
    printf("\n再次调用:\n");
    raii_demo();
#else
    printf("需要GCC或Clang编译器\n");
#endif
    return 0;
}
```

### 2.2 作用域守卫模式

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 使用goto实现的作用域守卫
int process_with_guard(void) {
    FILE *file = NULL;
    char *buffer = NULL;
    int *results = NULL;
    int ret = 0;

    file = fopen("data.txt", "r");
    if (!file) { ret = -1; goto cleanup; }

    buffer = malloc(1024);
    if (!buffer) { ret = -2; goto cleanup; }

    results = malloc(sizeof(int) * 100);
    if (!results) { ret = -3; goto cleanup; }

    // 正常处理逻辑...
    printf("处理完成\n");

cleanup:
    // 统一清理 - free/fclose对NULL安全
    if (file) fclose(file);
    free(buffer);
    free(results);
    return ret;
}

int main(void) {
    process_with_guard();
    return 0;
}
```

## 三、引用计数智能指针

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 引用计数的智能指针
typedef struct {
    void  *data;
    size_t ref_count;
    void (*destructor)(void *);  // 自定义析构函数
} RcPtr;

// 创建引用计数指针
RcPtr *rc_new(size_t size, void (*destructor)(void *)) {
    RcPtr *rc = malloc(sizeof(*rc));
    if (!rc) return NULL;

    rc->data = calloc(1, size);
    if (!rc->data) { free(rc); return NULL; }

    rc->ref_count = 1;
    rc->destructor = destructor ? destructor : free;
    return rc;
}

// 增加引用
RcPtr *rc_clone(RcPtr *rc) {
    if (rc) rc->ref_count++;
    return rc;
}

// 释放引用
void rc_release(RcPtr **rc) {
    if (rc == NULL || *rc == NULL) return;

    (*rc)->ref_count--;
    if ((*rc)->ref_count == 0) {
        printf("  引用计数为0，释放数据\n");
        (*rc)->destructor((*rc)->data);
        free(*rc);
    } else {
        printf("  引用计数减为 %zu\n", (*rc)->ref_count);
    }
    *rc = NULL;
}

// 获取数据指针
void *rc_get(RcPtr *rc) {
    return rc ? rc->data : NULL;
}

// ===== 使用示例 =====

typedef struct {
    char *name;
    int age;
} Person;

void person_destructor(void *p) {
    Person *person = (Person *)p;
    printf("  析构Person: %s\n", person->name);
    // name是动态分配的，需要释放
    // (本例中name指向常量字符串，无需释放)
}

void rc_demo(void) {
    printf("创建Person\n");
    RcPtr *p1 = rc_new(sizeof(Person), person_destructor);

    Person *person = rc_get(p1);
    person->name = "张三";
    person->age = 25;

    printf("person1: %s, %d岁 (ref=%zu)\n",
           person->name, person->age, p1->ref_count);

    // 共享所有权
    printf("\n克隆引用\n");
    RcPtr *p2 = rc_clone(p1);
    printf("person2: ref=%zu\n", p2->ref_count);

    // 释放第一个引用
    printf("\n释放第一个引用\n");
    rc_release(&p1);
    printf("p2仍然有效: ref=%zu\n", p2->ref_count);

    // 释放第二个引用
    printf("\n释放第二个引用\n");
    rc_release(&p2);
    // 此时数据被释放
}

int main(void) {
    rc_demo();
    return 0;
}
```

## 四、独占所有权模式

```c
#include <stdio.h>
#include <stdlib.h>

// 独占所有权 - 类似C++的unique_ptr
typedef struct {
    void *data;
    void (*destructor)(void *);
} UniquePtr;

UniquePtr unique_new(size_t size, void (*destructor)(void *)) {
    UniquePtr u;
    u.data = calloc(1, size);
    u.destructor = destructor ? destructor : free;
    return u;
}

// 移动所有权（源变成空）
UniquePtr unique_move(UniquePtr *src) {
    UniquePtr dst = *src;
    src->data = NULL;
    src->destructor = NULL;
    return dst;
}

void unique_release(UniquePtr *u) {
    if (u && u->data) {
        u->destructor(u->data);
        u->data = NULL;
    }
}

void unique_demo(void) {
    UniquePtr u1 = unique_new(sizeof(int) * 100, free);
    int *data = u1.data;
    data[0] = 42;
    printf("u1.data[0] = %d\n", data[0]);

    // 移动所有权
    UniquePtr u2 = unique_move(&u1);
    // u1.data 现在是 NULL

    printf("u2.data[0] = %d\n", ((int *)u2.data)[0]);

    unique_release(&u2);
    // unique_release(&u1);  // u1.data是NULL，安全
}

int main(void) {
    unique_demo();
    return 0;
}
```

## 五、关键要点

> **智能指针思想要点**
> 1. RAII思想：资源获取即初始化，离开作用域自动释放
> 2. GCC的cleanup属性可以模拟析构函数行为
> 3. 引用计数实现共享所有权，最后一个引用释放时销毁数据
> 4. 独占所有权模式保证同一时间只有一个所有者
> 5. C语言中这些模式需要程序员自觉使用
> 6. goto清理模式是最实用的C语言资源管理模式
> 7. 复杂项目考虑使用C++或引入智能指针库
