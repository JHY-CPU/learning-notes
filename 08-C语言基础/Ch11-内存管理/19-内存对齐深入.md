# 内存对齐深入

## 一、什么是内存对齐

内存对齐(Alignment)是指数据在内存中的起始地址必须是某个值的整数倍。现代CPU访问对齐的数据效率更高，某些架构甚至要求对齐访问。

```
内存对齐示意:

  对齐良好（4字节对齐）:
  地址: 0x00  0x04  0x08  0x0C
        [int] [int] [int] [int]
        每个int恰好在4的倍数地址

  不对齐:
  地址: 0x00  0x01  0x05  0x09
        [c] [  int  ] [  int  ]
        int跨越了对齐边界，访问慢或崩溃
```

## 二、基本对齐规则

### 2.1 内置类型的对齐

```c
#include <stdio.h>
#include <stddef.h>
#include <stdalign.h>

void basic_alignment(void) {
    printf("内置类型的对齐要求:\n");
    printf("  char:  size=%zu, align=%zu\n", sizeof(char), alignof(char));
    printf("  short: size=%zu, align=%zu\n", sizeof(short), alignof(short));
    printf("  int:   size=%zu, align=%zu\n", sizeof(int), alignof(int));
    printf("  long:  size=%zu, align=%zu\n", sizeof(long), alignof(long));
    printf("  float: size=%zu, align=%zu\n", sizeof(float), alignof(float));
    printf("  double:size=%zu, align=%zu\n", sizeof(double), alignof(double));
    printf("  void*: size=%zu, align=%zu\n", sizeof(void *), alignof(void *));

    // 通常规则：
    // char    对齐到 1
    // short   对齐到 2
    // int     对齐到 4
    // long    对齐到 4或8
    // double  对齐到 8
    // 指针    对齐到 4(32位)或8(64位)
}

int main(void) {
    basic_alignment();
    return 0;
}
```

### 2.2 结构体对齐

```c
#include <stdio.h>
#include <stddef.h>

// 结构体对齐规则：
// 1. 每个成员按自身对齐要求对齐
// 2. 结构体总大小是对齐要求的整数倍

struct Example1 {
    char  a;     // 偏移0, 大小1
    // 3字节填充
    int   b;     // 偏移4, 大小4
    char  c;     // 偏移8, 大小1
    // 3字节填充
};  // 总大小12

struct Example2 {
    int   b;     // 偏移0, 大小4
    char  a;     // 偏移4, 大小1
    char  c;     // 偏移5, 大小1
    // 2字节填充
};  // 总大小8

struct Example3 {
    double d;    // 偏移0, 大小8
    char   c;    // 偏移8, 大小1
    int    i;    // 偏移12, 大小4 (3字节填充)
    short  s;    // 偏移16, 大小2
    // 6字节填充
};  // 总大小24

void struct_alignment(void) {
    printf("结构体对齐:\n");
    printf("  Example1: size=%zu\n", sizeof(struct Example1));
    printf("    char a 偏移: %zu\n", offsetof(struct Example1, a));
    printf("    int  b 偏移: %zu\n", offsetof(struct Example1, b));
    printf("    char c 偏移: %zu\n", offsetof(struct Example1, c));

    printf("  Example2: size=%zu (优化排列)\n", sizeof(struct Example2));
    printf("    int  b 偏移: %zu\n", offsetof(struct Example2, b));
    printf("    char a 偏移: %zu\n", offsetof(struct Example2, a));
    printf("    char c 偏移: %zu\n", offsetof(struct Example2, c));
}

int main(void) {
    struct_alignment();
    return 0;
}
```

## 三、_Alignas 和 _Alignof (C11)

```c
#include <stdio.h>
#include <stdalign.h>
#include <stddef.h>

// _Alignof 获取类型的对齐要求
// _Alignas  指定变量或类型的对齐方式

void alignas_examples(void) {
    // 普通int对齐到4
    int normal_int;
    printf("普通int对齐: %zu\n", alignof(int));

    // 强制对齐到64字节（缓存行大小）
    _Alignas(64) int cache_aligned;
    printf("cache_aligned地址: %p (mod 64 = %zu)\n",
           (void *)&cache_aligned, (size_t)&cache_aligned % 64);

    // 结构体成员对齐
    struct alignas_demo {
        _Alignas(16) char data[5];  // data对齐到16
        int value;
    };

    struct alignas_demo demo;
    printf("data地址: %p (mod 16 = %zu)\n",
           (void *)demo.data, (size_t)demo.data % 16);

    // SIMD对齐要求
    _Alignas(32) float simd_data[8];
    printf("SIMD数据地址: %p (mod 32 = %zu)\n",
           (void *)simd_data, (size_t)simd_data % 32);
}

int main(void) {
    alignas_examples();
    return 0;
}
```

## 四、对齐与性能

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <stdalign.h>

void alignment_performance(void) {
    size_t count = 10000000;

    // 对齐访问
    _Alignas(64) int *aligned_arr = aligned_alloc(64, count * sizeof(int));
    if (!aligned_arr) return;

    // 非对齐访问（故意偏移1字节）
    char *raw = malloc(count * sizeof(int) + 64);
    int *unaligned_arr = (int *)((uintptr_t)raw | 1);  // 故意不对齐

    // 填充数据
    for (size_t i = 0; i < count; i++) {
        aligned_arr[i] = (int)i;
    }

    // 测试对齐访问
    clock_t start = clock();
    long sum = 0;
    for (size_t i = 0; i < count; i++) {
        sum += aligned_arr[i];
    }
    clock_t end = clock();
    printf("对齐访问: %.3f 秒 (sum=%ld)\n",
           (double)(end - start) / CLOCKS_PER_SEC, sum);

    free(aligned_arr);
    free(raw);
}

int main(void) {
    alignment_performance();
    return 0;
}
```

## 五、packed属性与对齐控制

```c
#include <stdio.h>
#include <stddef.h>

// packed: 取消对齐，紧密排列
struct __attribute__((packed)) Packed {
    char  a;
    int   b;
    char  c;
};

// 正常对齐
struct Normal {
    char  a;
    int   b;
    char  c;
};

void packed_demo(void) {
    printf("Normal结构体: size=%zu\n", sizeof(struct Normal));
    printf("Packed结构体: size=%zu\n", sizeof(struct Packed));

    printf("\nPacked节省空间但可能降低访问速度\n");
    printf("适用于网络协议、文件格式等需要精确布局的场景\n");
}

int main(void) {
    packed_demo();
    return 0;
}
```

## 六、关键要点

> **内存对齐要点**
> 1. 数据按自身大小对齐(char=1, int=4, double=8)
> 2. 结构体中按最大成员对齐要求排列
> 3. 合理排列结构体成员可减少填充字节
> 4. 使用`_Alignas`控制对齐方式
> 5. 对齐访问更快，不对齐可能在某些架构上崩溃
> 6. `packed`属性取消对齐但牺牲性能
> 7. SIMD指令要求16/32/64字节对齐
