# 自定义分配器

## 一、为什么需要自定义分配器

标准malloc是通用的，但特定场景下自定义分配器能提供更好的性能和特性。

```
自定义分配器的优势:
┌─────────────────────────────────────────┐
│  • 针对特定场景优化                       │
│  • 减少内存碎片                          │
│  • 更好的缓存局部性                       │
│  • 可预测的分配时间                       │
│  • 调试和统计功能                         │
└─────────────────────────────────────────┘
```

## 二、简单线性分配器

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 线性分配器（Arena Allocator）
// 最简单高效的分配器，只支持整体释放
typedef struct {
    char  *memory;
    size_t capacity;
    size_t offset;
} LinearAllocator;

LinearAllocator *linear_create(size_t size) {
    LinearAllocator *la = malloc(sizeof(*la));
    if (!la) return NULL;

    la->memory = malloc(size);
    if (!la->memory) { free(la); return NULL; }

    la->capacity = size;
    la->offset = 0;
    return la;
}

void *linear_alloc(LinearAllocator *la, size_t size) {
    // 对齐到8字节
    size_t aligned = (size + 7) & ~7;

    if (la->offset + aligned > la->capacity) {
        return NULL;  // 空间不足
    }

    void *ptr = la->memory + la->offset;
    la->offset += aligned;
    return ptr;
}

// 重置（释放所有分配）
void linear_reset(LinearAllocator *la) {
    la->offset = 0;
}

void linear_destroy(LinearAllocator *la) {
    if (la) {
        free(la->memory);
        free(la);
    }
}

void linear_demo(void) {
    LinearAllocator *la = linear_create(1024 * 1024);  // 1MB

    // 快速分配多个对象
    for (int i = 0; i < 1000; i++) {
        int *p = linear_alloc(la, sizeof(int) * 100);
        if (p) p[0] = i;
    }

    printf("已分配约 %zu 字节\n", la->offset);

    // 一次全部释放
    linear_reset(la);
    printf("重置后: %zu 字节\n", la->offset);

    linear_destroy(la);
}

int main(void) {
    linear_demo();
    return 0;
}
```

## 三、Buddy分配器思想

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Buddy分配器：按2的幂次分配，便于合并空闲块
#define MIN_BLOCK_SIZE  16
#define MAX_BLOCK_SIZE  4096
#define LEVELS          9  // log2(4096/16) + 1

typedef struct BuddyAllocator {
    char  *memory;
    int    free_list[LEVELS][256];  // 简化：每层最多256个块
    int    free_count[LEVELS];
} BuddyAllocator;

// 计算大于等于size的最小2的幂
static size_t next_power_of_2(size_t size) {
    size_t power = MIN_BLOCK_SIZE;
    while (power < size) power *= 2;
    return power;
}

// 计算层索引
static int size_to_level(size_t size) {
    int level = 0;
    size_t s = MIN_BLOCK_SIZE;
    while (s < size) { s *= 2; level++; }
    return level;
}

BuddyAllocator *buddy_create(void) {
    BuddyAllocator *ba = malloc(sizeof(*ba));
    if (!ba) return NULL;

    ba->memory = malloc(MAX_BLOCK_SIZE);
    if (!ba->memory) { free(ba); return NULL; }

    memset(ba->free_count, 0, sizeof(ba->free_count));
    // 最大块全部空闲
    ba->free_list[LEVELS - 1][0] = 0;
    ba->free_count[LEVELS - 1] = 1;

    return ba;
}

void *buddy_alloc(BuddyAllocator *ba, size_t size) {
    size_t alloc_size = next_power_of_2(size + sizeof(int));
    int level = size_to_level(alloc_size);

    // 查找空闲块
    for (int i = level; i < LEVELS; i++) {
        if (ba->free_count[i] > 0) {
            // 找到空闲块
            int block_idx = ba->free_list[i][--ba->free_count[i]];

            // 分割大块直到目标层
            while (i > level) {
                i--;
                int buddy = block_idx + (1 << (i + 4)) / MIN_BLOCK_SIZE;
                ba->free_list[i][ba->free_count[i]++] = buddy;
            }

            // 标记块大小
            *(int *)(ba->memory + block_idx * MIN_BLOCK_SIZE) = (int)alloc_size;
            return ba->memory + block_idx * MIN_BLOCK_SIZE + sizeof(int);
        }
    }
    return NULL;  // 无可用空间
}

void buddy_free(BuddyAllocator *ba, void *ptr) {
    if (!ptr || !ba) return;

    char *block = (char *)ptr - sizeof(int);
    int alloc_size = *(int *)block;
    int block_idx = (int)(block - ba->memory) / MIN_BLOCK_SIZE;
    int level = size_to_level(alloc_size);

    // 放回空闲链表（简化：不合并buddy）
    ba->free_list[level][ba->free_count[level]++] = block_idx;
}

void buddy_destroy(BuddyAllocator *ba) {
    if (ba) {
        free(ba->memory);
        free(ba);
    }
}

void buddy_demo(void) {
    BuddyAllocator *ba = buddy_create();

    void *p1 = buddy_alloc(ba, 32);
    void *p2 = buddy_alloc(ba, 128);
    void *p3 = buddy_alloc(ba, 64);

    printf("分配: p1=%p, p2=%p, p3=%p\n", p1, p2, p3);

    if (p1) strcpy((char *)p1, "Buddy分配的内存");
    printf("p1内容: %s\n", (char *)p1);

    buddy_free(ba, p1);
    buddy_free(ba, p2);
    buddy_free(ba, p3);

    buddy_destroy(ba);
}

int main(void) {
    linear_demo();
    printf("\n--- Buddy分配器 ---\n");
    buddy_demo();
    return 0;
}
```

## 四、关键要点

> **自定义分配器要点**
> 1. 线性分配器最快，适合帧分配或一次性批量分配
> 2. Buddy分配器按2的幂次分配，便于合并空闲块
> 3. Pool分配器适合固定大小对象的频繁分配
> 4. 自定义分配器可减少碎片和提高缓存命中率
> 5. 游戏引擎通常使用多种分配器组合
> 6. 调试阶段可以在分配器中加入边界检查和统计
