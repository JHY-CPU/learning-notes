# 对象池模式

## 一、概念

对象池(Object Pool)是内存池的高级形式，专门用于管理特定类型的对象。它预先分配一批对象，使用后回收而非释放，实现对象的复用。

```
对象池的工作流程:

  ┌─────────────────────────────────────────────┐
  │              对象池 (Object Pool)             │
  │                                             │
  │  [空闲对象链表]                               │
  │   obj1 → obj2 → obj3 → obj4 → NULL           │
  │                                             │
  │  alloc() → 取obj1，返回给调用者               │
  │  free(obj) → 放回链表，等待复用               │
  └─────────────────────────────────────────────┘
```

## 二、通用对象池实现

### 2.1 基于链表的对象池

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct ObjectNode {
    struct ObjectNode *next;
    char data[];  // 柔性数组，存储实际对象数据
} ObjectNode;

typedef struct ObjectPool {
    ObjectNode *free_list;      // 空闲对象链表
    char       *all_memory;     // 所有分配的内存
    size_t      object_size;    // 单个对象大小
    size_t      capacity;       // 总容量
    size_t      free_count;     // 空闲对象数
} ObjectPool;

// 创建对象池
ObjectPool *opool_create(size_t object_size, size_t capacity) {
    ObjectPool *pool = malloc(sizeof(*pool));
    if (!pool) return NULL;

    // 确保对象至少能容纳一个指针
    size_t node_size = sizeof(ObjectNode) + object_size;
    if (node_size < sizeof(ObjectNode) + sizeof(void *)) {
        node_size = sizeof(ObjectNode) + sizeof(void *);
    }

    pool->all_memory = malloc(node_size * capacity);
    if (!pool->all_memory) {
        free(pool);
        return NULL;
    }

    pool->object_size = object_size;
    pool->capacity = capacity;
    pool->free_count = capacity;

    // 构建空闲链表
    pool->free_list = NULL;
    for (size_t i = 0; i < capacity; i++) {
        ObjectNode *node = (ObjectNode *)(pool->all_memory + i * node_size);
        node->next = pool->free_list;
        pool->free_list = node;
    }

    return pool;
}

// 从池中获取一个对象
void *opool_alloc(ObjectPool *pool) {
    if (pool->free_list == NULL) return NULL;

    ObjectNode *node = pool->free_list;
    pool->free_list = node->next;
    pool->free_count--;

    // 清零对象数据
    memset(node->data, 0, pool->object_size);

    return node->data;
}

// 归还对象到池中
void opool_free(ObjectPool *pool, void *obj) {
    if (!obj) return;

    // 通过data反向获取node
    ObjectNode *node = (ObjectNode *)((char *)obj - sizeof(ObjectNode));
    node->next = pool->free_list;
    pool->free_list = node;
    pool->free_count++;
}

// 获取空闲对象数
size_t opool_available(ObjectPool *pool) {
    return pool->free_count;
}

// 销毁对象池
void opool_destroy(ObjectPool *pool) {
    if (pool) {
        free(pool->all_memory);
        free(pool);
    }
}

// ===== 使用示例 =====

typedef struct {
    int x, y;
    int active;
} Particle;

void object_pool_demo(void) {
    ObjectPool *pool = opool_create(sizeof(Particle), 100);
    if (!pool) return;

    printf("创建对象池: 100个粒子对象\n");
    printf("空闲对象: %zu\n", opool_available(pool));

    // 分配几个粒子
    Particle *p1 = opool_alloc(pool);
    Particle *p2 = opool_alloc(pool);
    Particle *p3 = opool_alloc(pool);

    if (p1) { p1->x = 10; p1->y = 20; p1->active = 1; }
    if (p2) { p2->x = 30; p2->y = 40; p2->active = 1; }
    if (p3) { p3->x = 50; p3->y = 60; p3->active = 1; }

    printf("分配3个粒子后，空闲: %zu\n", opool_available(pool));
    printf("粒子1: (%d, %d)\n", p1->x, p1->y);

    // 回收粒子
    opool_free(pool, p2);
    printf("回收粒子2后，空闲: %zu\n", opool_available(pool));

    // 重新分配（复用刚回收的内存）
    Particle *p4 = opool_alloc(pool);
    if (p4) { p4->x = 70; p4->y = 80; p4->active = 1; }
    printf("重新分配粒子4: (%d, %d)\n", p4->x, p4->y);

    // 清理
    opool_free(pool, p1);
    opool_free(pool, p3);
    opool_free(pool, p4);
    opool_destroy(pool);
}

int main(void) {
    object_pool_demo();
    return 0;
}
```

## 三、游戏中的对象池应用

```c
#include <stdio.h>
#include <stdlib.h>

// 子弹对象池 - 游戏开发中的经典应用
#define MAX_BULLETS 1000

typedef struct {
    float x, y;
    float vx, vy;
    int active;      // 是否在使用中
    int damage;
} Bullet;

typedef struct {
    Bullet bullets[MAX_BULLETS];
    int free_stack[MAX_BULLETS];  // 空闲索引栈
    int free_top;                  // 栈顶
} BulletPool;

void bullet_pool_init(BulletPool *bp) {
    bp->free_top = MAX_BULLETS;
    for (int i = 0; i < MAX_BULLETS; i++) {
        bp->free_stack[i] = i;
        bp->bullets[i].active = 0;
    }
}

Bullet *bullet_pool_spawn(BulletPool *bp, float x, float y,
                           float vx, float vy, int damage) {
    if (bp->free_top <= 0) return NULL;  // 池已满

    int idx = bp->free_stack[--bp->free_top];
    Bullet *b = &bp->bullets[idx];
    b->x = x; b->y = y;
    b->vx = vx; b->vy = vy;
    b->damage = damage;
    b->active = 1;
    return b;
}

void bullet_pool_recycle(BulletPool *bp, int index) {
    if (index < 0 || index >= MAX_BULLETS) return;
    bp->bullets[index].active = 0;
    bp->free_stack[bp->free_top++] = index;
}

void bullet_pool_update(BulletPool *bp) {
    for (int i = 0; i < MAX_BULLETS; i++) {
        if (!bp->bullets[i].active) continue;

        Bullet *b = &bp->bullets[i];
        b->x += b->vx;
        b->y += b->vy;

        // 超出边界则回收
        if (b->x < -100 || b->x > 900 || b->y < -100 || b->y > 700) {
            bullet_pool_recycle(bp, i);
        }
    }
}

int main(void) {
    BulletPool bp;
    bullet_pool_init(&bp);

    printf("子弹池初始化完成\n");

    // 模拟发射子弹
    for (int i = 0; i < 10; i++) {
        Bullet *b = bullet_pool_spawn(&bp, 400, 500, 0, -10, 25);
        if (b) printf("发射子弹 %d: 位置(%.0f, %.0f)\n", i, b->x, b->y);
    }

    printf("空闲子弹数: %d\n", bp.free_top);

    // 模拟几帧更新
    for (int frame = 0; frame < 5; frame++) {
        bullet_pool_update(&bp);
    }

    printf("更新5帧后空闲子弹数: %d\n", bp.free_top);

    return 0;
}
```

## 四、关键要点

> **对象池要点**
> 1. 预分配对象避免运行时malloc开销
> 2. 回收而非释放，对象内存不变
> 3. 适合生命周期短、频繁创建销毁的对象
> 4. 游戏粒子、网络连接、数据库连接等场景常用
> 5. 实现方式：链表或索引栈
> 6. 可以设置池大小上限，防止无限增长
