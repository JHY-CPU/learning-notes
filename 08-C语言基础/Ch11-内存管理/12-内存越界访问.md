# 内存越界访问

## 一、什么是内存越界

内存越界访问(Out-of-Bounds Access)是指程序访问了分配范围之外的内存区域。这是C语言中最常见也最危险的错误之一。

```
内存越界类型:
┌──────────────────────────────────────┐
│  栈溢出 (Stack Overflow)             │  局部数组越界写入
│  堆溢出 (Heap Overflow)              │  堆分配块越界访问
│  缓冲区溢出 (Buffer Overflow)        │  字符串/数组超出边界
│  下溢 (Underflow)                    │  访问负数索引
└──────────────────────────────────────┘
```

## 二、缓冲区溢出

### 2.1 栈缓冲区溢出

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void stack_buffer_overflow(void) {
    char buffer[8];
    int important_flag = 0;

    printf("buffer地址: %p\n", (void *)buffer);
    printf("flag地址:   %p\n", (void *)&important_flag);
    printf("flag值:     %d\n", important_flag);

    // 危险：写入超过buffer大小的数据
    // strcpy 不检查长度，会覆盖相邻内存
    // strcpy(buffer, "This is a very long string that overflows!");
    // 可能覆盖 important_flag 的值！

    // 安全做法：使用 strncpy
    strncpy(buffer, "Hello!", sizeof(buffer) - 1);
    buffer[sizeof(buffer) - 1] = '\0';

    printf("安全复制: %s\n", buffer);
    printf("flag值:   %d (未被破坏)\n", important_flag);
}

int main(void) {
    stack_buffer_overflow();
    return 0;
}
```

### 2.2 数组越界访问

```c
#include <stdio.h>
#include <stdlib.h>

void array_out_of_bounds(void) {
    int arr[5] = {1, 2, 3, 4, 5};

    // 合法访问
    for (int i = 0; i < 5; i++) {
        printf("arr[%d] = %d\n", i, arr[i]);
    }

    // 越界访问 - 未定义行为
    // arr[5] = 100;   // 超出范围
    // arr[-1] = 200;  // 负数索引
    // arr[100] = 300; // 严重越界

    // 越界可能不会立即崩溃，但会：
    // 1. 破坏其他变量
    // 2. 破坏栈帧
    // 3. 导致随机崩溃

    printf("\n数组边界: arr[0] 到 arr[4]\n");
    printf("arr起始: %p\n", (void *)arr);
    printf("arr结束: %p (arr[4]之后)\n", (void *)&arr[5]);
}

int main(void) {
    array_out_of_bounds();
    return 0;
}
```

## 三、堆溢出

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void heap_buffer_overflow(void) {
    char *buffer = malloc(16);
    if (buffer == NULL) return;

    // 安全使用
    strcpy(buffer, "Hello");
    printf("安全: %s\n", buffer);

    // 危险：堆溢出
    // strcpy(buffer, "This string is way too long for the 16-byte buffer!");
    // 堆元数据可能被破坏
    // 后续的malloc/free可能崩溃

    // 安全做法
    const char *long_text = "This is a long string";
    strncpy(buffer, long_text, 15);
    buffer[15] = '\0';
    printf("安全截断: %s\n", buffer);

    free(buffer);
}

int main(void) {
    heap_buffer_overflow();
    return 0;
}
```

## 四、字符串相关的越界

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void string_overflow_examples(void) {
    // 场景1：strcpy 不检查目标缓冲区大小
    char dest[10];
    // strcpy(dest, "Hello World!");  // 溢出！12字节写入10字节缓冲区

    // 安全替代
    strncpy(dest, "Hello World!", sizeof(dest) - 1);
    dest[sizeof(dest) - 1] = '\0';

    // 场景2：sprintf 不检查目标大小
    char buf[16];
    // sprintf(buf, "Number: %d", 1234567890);  // 可能溢出

    // 安全替代
    snprintf(buf, sizeof(buf), "Number: %d", 1234567890);
    printf("安全格式化: %s\n", buf);

    // 场景3：gets 完全不安全（C11已移除）
    // char input[100];
    // gets(input);  // 绝对不要使用！

    // 安全替代
    char input[100];
    if (fgets(input, sizeof(input), stdin) != NULL) {
        // 移除可能的换行符
        input[strcspn(input, "\n")] = '\0';
        printf("安全输入: %s\n", input);
    }
}

int main(void) {
    string_overflow_examples();
    return 0;
}
```

## 五、越界的后果

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void demonstrate_consequences(void) {
    printf("内存越界的可能后果:\n\n");

    // 后果1：数据损坏
    {
        int a = 0xAAAA;
        int arr[4] = {0};
        int b = 0xBBBB;

        printf("越界前: a=0x%X, b=0x%X\n", a, b);

        // arr[4] 实际上可能指向b或a附近的内存
        // arr[4] = 0xCCCC;  // 可能覆盖b或a

        printf("注：具体覆盖哪个变量取决于编译器和平台\n");
    }

    // 后果2：程序崩溃(段错误)
    {
        // 严重越界通常导致立即崩溃
        // int *ptr = malloc(sizeof(int));
        // ptr[1000000] = 42;  // 很可能段错误
    }

    // 后果3：安全漏洞
    {
        printf("\n缓冲区溢出是经典的安全漏洞:\n");
        printf("  - 可以覆盖返回地址\n");
        printf("  - 可以执行任意代码\n");
        printf("  - 缓冲区溢出攻击\n");
    }
}

int main(void) {
    demonstrate_consequences();
    return 0;
}
```

## 六、防止越界的方法

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 方法1：始终使用带边界检查的函数
void safe_string_copy(char *dest, size_t dest_size, const char *src) {
    if (dest_size == 0) return;
    strncpy(dest, src, dest_size - 1);
    dest[dest_size - 1] = '\0';
}

// 方法2：封装安全的数组访问
typedef struct {
    int *data;
    size_t size;
} SafeArray;

SafeArray *sa_create(size_t size) {
    SafeArray *sa = malloc(sizeof(*sa));
    if (!sa) return NULL;
    sa->data = calloc(size, sizeof(int));
    if (!sa->data) { free(sa); return NULL; }
    sa->size = size;
    return sa;
}

int sa_get(SafeArray *sa, size_t index) {
    if (index >= sa->size) {
        fprintf(stderr, "索引%zu越界(大小%zu)\n", index, sa->size);
        return 0;  // 或abort()
    }
    return sa->data[index];
}

void sa_set(SafeArray *sa, size_t index, int value) {
    if (index >= sa->size) {
        fprintf(stderr, "索引%zu越界(大小%zu)\n", index, sa->size);
        return;
    }
    sa->data[index] = value;
}

void sa_destroy(SafeArray *sa) {
    if (sa) { free(sa->data); free(sa); }
}

int main(void) {
    // 安全字符串操作
    char buf[16];
    safe_string_copy(buf, sizeof(buf), "Hello, World!");
    printf("安全复制: %s\n", buf);

    // 安全数组访问
    SafeArray *arr = sa_create(5);
    if (arr) {
        sa_set(arr, 0, 42);
        sa_set(arr, 4, 100);
        sa_set(arr, 10, 999);  // 触发越界警告

        printf("arr[0] = %d\n", sa_get(arr, 0));
        printf("arr[10] = %d\n", sa_get(arr, 10));  // 触发越界警告

        sa_destroy(arr);
    }

    return 0;
}
```

## 七、关键要点

> **内存越界要点**
> 1. 永远不使用`strcpy`/`sprintf`/`gets`，改用`strncpy`/`snprintf`/`fgets`
> 2. 数组访问前必须检查索引是否在合法范围内
> 3. 堆缓冲区溢出会破坏堆管理结构，导致后续malloc/free崩溃
> 4. 栈缓冲区溢出可能覆盖返回地址，是安全攻击的经典手段
> 5. 使用AddressSanitizer(-fsanitize=address)可以检测越界访问
> 6. 封装带边界检查的数组访问接口
> 7. 越界行为在调试时可能不崩溃，但发布后崩溃——这是最难以调试的问题
