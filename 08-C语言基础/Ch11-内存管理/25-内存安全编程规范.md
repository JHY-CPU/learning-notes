# 内存安全编程规范

## 一、核心原则

```
内存安全编程的五大原则:

  1. 初始化 → 声明时初始化所有变量和指针
  2. 检查 → 检查malloc返回值、检查数组边界
  3. 配对 → 每个malloc必须有对应free
  4. 置空 → free后立即置NULL
  5. 隔离 → 封装安全接口，避免直接操作原始指针
```

## 二、安全分配模式

### 2.1 始终检查返回值

```c
#include <stdio.h>
#include <stdlib.h>

// 模式1：检查并处理
void pattern_check(void) {
    int *data = malloc(sizeof(*data) * 100);
    if (data == NULL) {
        fprintf(stderr, "分配失败\n");
        return;  // 不继续执行
    }
    // 安全使用 data...
    free(data);
}

// 模式2：封装安全分配器
void *xmalloc(size_t size) {
    void *ptr = malloc(size);
    if (ptr == NULL) {
        fprintf(stderr, "xmalloc失败: %zu字节\n", size);
        abort();  // 对于关键程序，直接终止
    }
    return ptr;
}

// 模式3：带默认值的分配
void *xcalloc(size_t nmemb, size_t size) {
    void *ptr = calloc(nmemb, size);
    if (ptr == NULL) {
        fprintf(stderr, "xcalloc失败\n");
        abort();
    }
    return ptr;
}

int main(void) {
    pattern_check();

    int *safe = xmalloc(sizeof(*safe) * 100);
    safe[0] = 42;
    free(safe);

    return 0;
}
```

### 2.2 使用sizeof(*ptr)模式

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int x, y, z;
} Point3D;

void sizeof_star_pattern(void) {
    // 好：sizeof(*ptr) 自动适配类型
    int *arr = malloc(sizeof(*arr) * 100);
    double *vals = malloc(sizeof(*vals) * 50);
    Point3D *points = malloc(sizeof(*points) * 10);

    // 即使类型改变，分配代码不需要修改
    // 例如：int *arr → long *arr，sizeof(*arr) 自动变为8

    free(arr);
    free(vals);
    free(points);
}

int main(void) {
    sizeof_star_pattern();
    return 0;
}
```

## 三、资源管理模式

### 3.1 单入口单出口

```c
#include <stdio.h>
#include <stdlib.h>

// 好的模式：单点释放
int good_resource_pattern(void) {
    FILE *f = NULL;
    char *buf = NULL;
    int *data = NULL;
    int result = 0;

    f = fopen("data.txt", "r");
    if (!f) { result = -1; goto cleanup; }

    buf = malloc(1024);
    if (!buf) { result = -2; goto cleanup; }

    data = malloc(sizeof(*data) * 100);
    if (!data) { result = -3; goto cleanup; }

    // 正常处理...
    result = 0;

cleanup:
    fclose(f);    // fclose(NULL) 安全
    free(buf);    // free(NULL) 安全
    free(data);
    return result;
}

int main(void) {
    good_resource_pattern();
    return 0;
}
```

### 3.2 配对管理宏

```c
#include <stdio.h>
#include <stdlib.h>

// 分配与释放配对
#define ALLOC_AND_CHECK(ptr, size) do { \
    (ptr) = malloc(size);               \
    if ((ptr) == NULL) goto cleanup;    \
} while(0)

#define SAFE_FREE(ptr) do { \
    free(ptr);              \
    (ptr) = NULL;           \
} while(0)

int paired_management(void) {
    int *a = NULL, *b = NULL, *c = NULL;

    ALLOC_AND_CHECK(a, sizeof(*a) * 100);
    ALLOC_AND_CHECK(b, sizeof(*b) * 200);
    ALLOC_AND_CHECK(c, sizeof(*c) * 300);

    // 使用 a, b, c...
    printf("全部分配成功\n");

cleanup:
    SAFE_FREE(a);
    SAFE_FREE(b);
    SAFE_FREE(c);
    return (c != NULL) ? 0 : -1;
}

int main(void) {
    paired_management();
    return 0;
}
```

## 四、安全的字符串操作

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 安全字符串封装
typedef struct {
    char  *data;
    size_t length;
    size_t capacity;
} SafeString;

SafeString *sstr_create(size_t initial_cap) {
    SafeString *ss = malloc(sizeof(*ss));
    if (!ss) return NULL;
    ss->data = malloc(initial_cap);
    if (!ss->data) { free(ss); return NULL; }
    ss->data[0] = '\0';
    ss->length = 0;
    ss->capacity = initial_cap;
    return ss;
}

int sstr_append(SafeString *ss, const char *str) {
    size_t len = strlen(str);
    if (ss->length + len + 1 > ss->capacity) {
        size_t new_cap = ss->capacity * 2;
        while (new_cap < ss->length + len + 1) new_cap *= 2;
        char *new_data = realloc(ss->data, new_cap);
        if (!new_data) return -1;
        ss->data = new_data;
        ss->capacity = new_cap;
    }
    strcpy(ss->data + ss->length, str);
    ss->length += len;
    return 0;
}

void sstr_destroy(SafeString *ss) {
    if (ss) {
        free(ss->data);
        free(ss);
    }
}

int main(void) {
    SafeString *ss = sstr_create(16);
    sstr_append(ss, "Hello");
    sstr_append(ss, ", World!");
    printf("内容: %s (长度%zu, 容量%zu)\n",
           ss->data, ss->length, ss->capacity);
    sstr_destroy(ss);
    return 0;
}
```

## 五、编码检查清单

```
内存安全检查清单:

□ 所有指针声明时初始化为NULL
□ malloc/calloc/realloc返回值全部检查
□ 每个malloc都有对应的free
□ free后立即置NULL
□ 数组访问前检查索引边界
□ 字符串操作使用安全版本(strncpy/snprintf)
□ 不返回局部变量的指针
□ 结构体中指针成员释放后置NULL
□ goto清理模式统一资源释放
□ 使用AddressSanitizer定期检查
□ 不在循环外持有循环内分配的指针
□ 多线程访问用互斥锁保护
```

## 六、关键要点

> **安全编程要点**
> 1. 防御性编程：假设所有外部输入和API调用都可能失败
> 2. 单一释放点：用goto或cleanup函数统一释放
> 3. 封装安全接口：对外暴露不直接操作原始malloc/free
> 4. 编译警告全开：`-Wall -Wextra -Werror`
> 5. 工具辅助：ASan、Valgrind、clang-tidy
> 6. 代码审查：内存管理是审查重点
