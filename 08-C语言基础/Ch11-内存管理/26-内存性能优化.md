# 内存性能优化

## 一、优化原则

```
内存性能优化的三个维度:

  1. 减少分配次数 → 批量分配、内存池、对象池
  2. 提高访问速度 → 缓存友好、数据局部性
  3. 降低内存占用 → 压缩、按需分配、及时释放
```

## 二、减少分配次数

### 2.1 预分配策略

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 不好：频繁小分配
void bad_frequent_alloc(void) {
    for (int i = 0; i < 10000; i++) {
        int *temp = malloc(sizeof(int));
        *temp = i;
        // 处理...
        free(temp);  // 10000次malloc + 10000次free
    }
}

// 好：一次大分配
void good_batch_alloc(void) {
    int *data = malloc(sizeof(*data) * 10000);
    if (!data) return;

    for (int i = 0; i < 10000; i++) {
        data[i] = i;  // 只有1次malloc + 1次free
    }

    free(data);
}

// 好：动态数组预分配
typedef struct {
    int *data;
    size_t size;
    size_t capacity;
} DynArray;

DynArray *da_create(size_t initial_cap) {
    DynArray *da = malloc(sizeof(*da));
    da->data = malloc(sizeof(*da->data) * initial_cap);
    da->size = 0;
    da->capacity = initial_cap;
    return da;
}

void da_push(DynArray *da, int value) {
    if (da->size >= da->capacity) {
        da->capacity *= 2;
        da->data = realloc(da->data, sizeof(*da->data) * da->capacity);
    }
    da->data[da->size++] = value;
}

void da_destroy(DynArray *da) {
    free(da->data);
    free(da);
}

int main(void) {
    bad_frequent_alloc();
    good_batch_alloc();

    DynArray *da = da_create(16);
    for (int i = 0; i < 1000; i++) da_push(da, i);
    printf("动态数组: size=%zu, cap=%zu\n", da->size, da->capacity);
    da_destroy(da);

    return 0;
}
```

### 2.2 对象复用

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 不好：每次请求创建新对象
typedef struct {
    char *data;
    size_t len;
} Buffer;

Buffer *buffer_create_bad(const char *content) {
    Buffer *buf = malloc(sizeof(*buf));
    buf->len = strlen(content);
    buf->data = malloc(buf->len + 1);
    strcpy(buf->data, content);
    return buf;
}

void buffer_destroy_bad(Buffer *buf) {
    free(buf->data);
    free(buf);
}

// 好：对象复用
Buffer *buffer_create_good(void) {
    Buffer *buf = malloc(sizeof(*buf));
    buf->data = malloc(256);
    buf->len = 0;
    buf->data[0] = '\0';
    return buf;
}

void buffer_reuse(Buffer *buf, const char *content) {
    size_t len = strlen(content);
    if (len + 1 > buf->len + 256) {  // 需要扩容
        buf->data = realloc(buf->data, len + 1);
    }
    strcpy(buf->data, content);
    buf->len = len;
}

int main(void) {
    Buffer *buf = buffer_create_good();

    for (int i = 0; i < 100; i++) {
        char temp[64];
        sprintf(temp, "消息编号%d", i);
        buffer_reuse(buf, temp);  // 复用，不重新分配
    }

    free(buf->data);
    free(buf);
    return 0;
}
```

## 三、缓存友好优化

### 3.1 数据布局

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// 不好：AoS (Array of Structures) 缓存不友好
typedef struct {
    float x, y, z;    // 位置
    float r, g, b;    // 颜色
    float u, v;       // 纹理
} Vertex_AoS;

// 好：SoA (Structure of Arrays) 缓存友好
typedef struct {
    float *x, *y, *z;
    float *r, *g, *b;
    float *u, *v;
    int    count;
} Vertices_SoA;

void cache_friendly_demo(void) {
    int n = 1000000;

    // AoS方式
    Vertex_AoS *aos = malloc(sizeof(Vertex_AoS) * n);
    for (int i = 0; i < n; i++) {
        aos[i].x = (float)i;
    }

    // SoA方式
    Vertices_SoA soa;
    soa.count = n;
    soa.x = malloc(sizeof(float) * n);
    soa.y = malloc(sizeof(float) * n);
    for (int i = 0; i < n; i++) {
        soa.x[i] = (float)i;
    }

    // 测试：只访问x坐标
    clock_t start = clock();
    float sum_aos = 0;
    for (int i = 0; i < n; i++) {
        sum_aos += aos[i].x;  // 每次跳过整个结构体
    }
    clock_t mid = clock();

    float sum_soa = 0;
    for (int i = 0; i < n; i++) {
        sum_soa += soa.x[i];  // 连续内存访问
    }
    clock_t end = clock();

    printf("AoS访问: %.3f 秒 (sum=%.0f)\n",
           (double)(mid - start) / CLOCKS_PER_SEC, sum_aos);
    printf("SoA访问: %.3f 秒 (sum=%.0f)\n",
           (double)(end - mid) / CLOCKS_PER_SEC, sum_soa);

    free(aos);
    free(soa.x);
    free(soa.y);
}

int main(void) {
    cache_friendly_demo();
    return 0;
}
```

### 3.2 减少间接访问

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// 不好：指针间接访问
void bad_indirect_access(void) {
    int n = 1000000;
    int **ptrs = malloc(sizeof(int *) * n);
    for (int i = 0; i < n; i++) {
        ptrs[i] = malloc(sizeof(int));
        *ptrs[i] = i;
    }

    clock_t start = clock();
    long sum = 0;
    for (int i = 0; i < n; i++) {
        sum += *ptrs[i];  // 二级间接访问
    }
    clock_t end = clock();
    printf("间接访问: %.3f 秒 (sum=%ld)\n",
           (double)(end - start) / CLOCKS_PER_SEC, sum);

    for (int i = 0; i < n; i++) free(ptrs[i]);
    free(ptrs);
}

// 好：直接数组访问
void good_direct_access(void) {
    int n = 1000000;
    int *arr = malloc(sizeof(int) * n);
    for (int i = 0; i < n; i++) {
        arr[i] = i;
    }

    clock_t start = clock();
    long sum = 0;
    for (int i = 0; i < n; i++) {
        sum += arr[i];  // 直接访问，缓存友好
    }
    clock_t end = clock();
    printf("直接访问: %.3f 秒 (sum=%ld)\n",
           (double)(end - start) / CLOCKS_PER_SEC, sum);

    free(arr);
}

int main(void) {
    bad_indirect_access();
    good_direct_access();
    return 0;
}
```

## 四、内存对齐优化

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdalign.h>

void alignment_optimization(void) {
    // 对齐分配提高访问速度
    size_t count = 1024 * 1024;

    // 使用对齐分配
    int *aligned_data = aligned_alloc(64, count * sizeof(int));
    if (aligned_data) {
        printf("对齐到64字节: %p (mod 64 = %zu)\n",
               (void *)aligned_data, (size_t)aligned_data % 64);

        for (size_t i = 0; i < count; i++) {
            aligned_data[i] = (int)i;
        }

        free(aligned_data);
    }

    // 缓存行填充避免false sharing
    typedef struct {
        int data;
        char padding[60];  // 填充到64字节（缓存行大小）
    } CacheLinePadded;

    printf("缓存行填充结构体大小: %zu\n", sizeof(CacheLinePadded));
}

int main(void) {
    alignment_optimization();
    return 0;
}
```

## 五、关键要点

> **内存性能优化要点**
> 1. 预分配：预测容量，减少realloc次数
> 2. 对象复用：避免频繁创建销毁
> 3. 缓存友好：连续内存访问，SoA布局
> 4. 减少间接：避免多级指针跳转
> 5. 对齐分配：匹配缓存行大小
> 6. 批量操作：减少系统调用次数
> 7. 性能分析：用perf/cachegrind验证优化效果
