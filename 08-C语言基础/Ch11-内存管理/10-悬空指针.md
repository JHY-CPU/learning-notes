# 悬空指针

## 一、什么是悬空指针

悬空指针(Dangling Pointer)是指指向已被释放的内存的指针。使用悬空指针会导致**未定义行为**，可能产生数据损坏、程序崩溃等严重后果。

```
悬空指针的产生:
  int *ptr = malloc(sizeof(int));    ptr ──→ [42]
  *ptr = 42;
  free(ptr);                         ptr ──→ [已释放，内容不确定]
  // 此时 ptr 是悬空指针！
```

## 二、悬空指针的产生原因

### 2.1 free后继续使用

```c
#include <stdio.h>
#include <stdlib.h>

void dangling_after_free(void) {
    int *ptr = malloc(sizeof(int));
    if (ptr == NULL) return;

    *ptr = 42;
    printf("释放前: *ptr = %d\n", *ptr);

    free(ptr);
    // ptr现在是悬空指针！

    // 以下都是未定义行为:
    // printf("%d\n", *ptr);   // 读取已释放的内存
    // *ptr = 100;             // 写入已释放的内存
    // ptr[0] = 200;           // 同上

    // 正确做法：置NULL
    ptr = NULL;
    // 现在任何通过ptr的访问都会立即段错误，便于发现
}

int main(void) {
    dangling_after_free();
    return 0;
}
```

### 2.2 返回局部变量的地址

```c
#include <stdio.h>
#include <stdlib.h>

// 危险：返回栈上局部变量的地址
int *get_dangling_pointer(void) {
    int local = 42;
    return &local;  // 返回后 local 的栈帧被释放
}

// 正确方式1：返回堆内存
int *get_heap_pointer(void) {
    int *ptr = malloc(sizeof(int));
    if (ptr) *ptr = 42;
    return ptr;  // 调用者负责free
}

// 正确方式2：返回静态变量
int *get_static_pointer(void) {
    static int value = 42;
    return &value;  // 静态变量生命周期持续到程序结束
}

// 正确方式3：通过参数输出
void fill_value(int *out) {
    if (out) *out = 42;
}

int main(void) {
    // 悬空指针示例
    // int *bad = get_dangling_pointer();
    // printf("%d\n", *bad);  // 未定义行为！

    // 正确用法
    int *good1 = get_heap_pointer();
    if (good1) {
        printf("堆指针: %d\n", *good1);
        free(good1);
    }

    int *good2 = get_static_pointer();
    printf("静态指针: %d\n", *good2);

    int value;
    fill_value(&value);
    printf("参数输出: %d\n", value);

    return 0;
}
```

### 2.3 多个指针指向同一内存

```c
#include <stdio.h>
#include <stdlib.h>

void dangling_multiple_pointers(void) {
    int *p1 = malloc(sizeof(int));
    if (p1 == NULL) return;

    *p1 = 42;
    int *p2 = p1;  // p2指向同一内存

    printf("p1: %d, p2: %d\n", *p1, *p2);  // 都是42

    free(p1);
    p1 = NULL;
    // p2仍然指向已释放的内存！悬空指针！

    // *p2 = 100;  // 未定义行为！

    p2 = NULL;  // 也需要置NULL
}

int main(void) {
    dangling_multiple_pointers();
    return 0;
}
```

### 2.4 结构体中的悬空指针

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char *name;
    int age;
} Person;

void dangling_in_struct(void) {
    Person *p = malloc(sizeof(Person));
    if (!p) return;

    p->name = malloc(64);
    if (!p->name) {
        free(p);
        return;
    }
    strcpy(p->name, "张三");
    p->age = 25;

    printf("%s, %d岁\n", p->name, p->age);

    // 错误：只释放了name，但p->name指针还在
    free(p->name);
    // p->name 现在是悬空指针！

    // 正确做法
    p->name = NULL;  // 置NULL

    free(p);
}

int main(void) {
    dangling_in_struct();
    return 0;
}
```

## 三、Use-After-Free (UAF)

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void use_after_free_demo(void) {
    char *str = malloc(32);
    strcpy(str, "敏感数据ABC");
    printf("原数据: %s\n", str);

    free(str);
    // str 是悬空指针

    // Use-After-Free: 内存可能被重新分配
    char *new_data = malloc(32);
    strcpy(new_data, "新的数据XYZ");

    // 原来的str可能指向new_data的数据！
    // 或者指向垃圾数据
    // printf("UAF读取: %s\n", str);  // 未定义行为

    // 安全演示
    printf("新数据: %s\n", new_data);

    free(new_data);
    new_data = NULL;
}

int main(void) {
    use_after_free_demo();
    return 0;
}
```

## 四、避免悬空指针的方法

### 4.1 释放后立即置NULL

```c
#include <stdio.h>
#include <stdlib.h>

#define SAFE_FREE(p) do { free(p); (p) = NULL; } while(0)

void avoid_dangling_null(void) {
    int *ptr = malloc(sizeof(int) * 10);
    if (!ptr) return;

    SAFE_FREE(ptr);  // 释放并置NULL

    // 即使后续误用，也会立即段错误
    // ptr[0] = 42;  // 空指针解引用，立即崩溃，易发现
}

int main(void) {
    avoid_dangling_null();
    return 0;
}
```

### 4.2 使用所有权语义

```c
#include <stdio.h>
#include <stdlib.h>

// 明确所有权：谁拥有谁负责释放
typedef struct {
    int *data;
    int owned;  // 1=拥有所有权, 0=借用
} DataHandle;

DataHandle create_data(int value) {
    DataHandle h;
    h.data = malloc(sizeof(int));
    if (h.data) {
        *h.data = value;
        h.owned = 1;
    } else {
        h.owned = 0;
    }
    return h;
}

// 借用：不负责释放
int borrow_data(DataHandle h) {
    if (h.data) return *h.data;
    return 0;
}

// 移交所有权
DataHandle transfer_ownership(DataHandle *src) {
    DataHandle dst = *src;
    src->data = NULL;
    src->owned = 0;
    return dst;
}

// 销毁拥有者
void destroy_data(DataHandle *h) {
    if (h->owned && h->data) {
        free(h->data);
        h->data = NULL;
        h->owned = 0;
    }
}

int main(void) {
    DataHandle h = create_data(42);
    printf("值: %d\n", borrow_data(h));

    DataHandle h2 = transfer_ownership(&h);
    // h不再拥有数据，h2拥有
    printf("转移后: %d\n", borrow_data(h2));

    destroy_data(&h);     // 安全：data已为NULL
    destroy_data(&h2);    // 释放实际内存

    return 0;
}
```

## 五、关键要点

> **悬空指针要点**
> 1. free后指针仍指向原地址，但该内存已不可用
> 2. 返回局部变量地址是最常见的悬空指针来源
> 3. 多个指针指向同一内存时，free后所有指针都变成悬空
> 4. Use-After-Free是严重的安全漏洞
> 5. 释放后立即置NULL是防止悬空指针的基本方法
> 6. 使用所有权语义可以明确释放责任
> 7. 编译器和工具(AddressSanitizer)可以帮助检测悬空指针
