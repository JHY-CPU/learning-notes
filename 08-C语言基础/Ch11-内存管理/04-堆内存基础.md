# 堆内存基础

## 一、堆内存概念

堆(Heap)是程序运行时用于**动态内存分配**的区域。与栈不同，堆内存由程序员手动管理，大小可在运行时决定。

```
堆内存的核心特征:
┌──────────────────────────────────┐
│  • 需要手动分配 (malloc等)        │
│  • 需要手动释放 (free)            │
│  • 大小可在运行时决定             │
│  • 空间远大于栈（受系统内存限制）   │
│  • 分配速度比栈慢                 │
│  • 向高地址方向增长               │
└──────────────────────────────────┘
```

## 二、为什么需要堆内存

### 2.1 栈内存的局限

```c
#include <stdio.h>
#include <stdlib.h>

void stack_limitations(void) {
    // 问题1：数组大小必须是编译时常量(C89)
    int n;
    // scanf("%d", &n);
    n = 100;
    // int arr[n];  // C99 VLA才支持，C89不支持

    // 问题2：栈空间有限
    // char big[10 * 1024 * 1024];  // 10MB - 可能栈溢出

    // 问题3：函数返回后局部变量销毁
    // 无法返回指向局部变量的指针
}

// 使用堆解决这些问题
void heap_solutions(void) {
    // 问题1：可以在运行时决定大小
    int n = 100;
    int *arr = malloc(n * sizeof(int));

    // 问题2：可以分配很大的内存
    char *big = malloc(10 * 1024 * 1024);  // 10MB 堆内存

    // 问题3：堆内存生命周期由程序员控制
    // 可以返回指针给调用者使用

    free(arr);
    free(big);
}

int main(void) {
    heap_solutions();
    return 0;
}
```

### 2.2 动态数据结构

堆内存是实现动态数据结构的基础：

```c
#include <stdio.h>
#include <stdlib.h>

// 链表节点 - 必须使用堆内存
typedef struct Node {
    int data;
    struct Node *next;
} Node;

// 动态数组 - 大小可变
typedef struct {
    int *data;
    size_t size;
    size_t capacity;
} DynamicArray;

// 创建新节点
Node *create_node(int value) {
    Node *new_node = malloc(sizeof(Node));
    if (new_node == NULL) return NULL;

    new_node->data = value;
    new_node->next = NULL;
    return new_node;
}

// 初始化动态数组
void da_init(DynamicArray *da, size_t initial_capacity) {
    da->data = malloc(initial_capacity * sizeof(int));
    da->size = 0;
    da->capacity = initial_capacity;
}

// 动态扩容
void da_push(DynamicArray *da, int value) {
    if (da->size >= da->capacity) {
        da->capacity *= 2;
        da->data = realloc(da->data, da->capacity * sizeof(int));
    }
    da->data[da->size++] = value;
}

int main(void) {
    // 链表
    Node *head = create_node(1);
    head->next = create_node(2);
    head->next->next = create_node(3);

    Node *curr = head;
    while (curr) {
        printf("%d -> ", curr->data);
        curr = curr->next;
    }
    printf("NULL\n");

    // 释放链表
    while (head) {
        Node *temp = head;
        head = head->next;
        free(temp);
    }

    // 动态数组
    DynamicArray da;
    da_init(&da, 4);
    for (int i = 0; i < 10; i++) {
        da_push(&da, i * 10);
    }
    for (size_t i = 0; i < da.size; i++) {
        printf("da[%zu] = %d\n", i, da.data[i]);
    }
    free(da.data);

    return 0;
}
```

## 三、malloc/free 基本用法

### 3.1 分配与释放的完整流程

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void malloc_free_basic(void) {
    // ===== 第1步：分配内存 =====
    int *ptr = (int *)malloc(sizeof(int) * 10);
    // malloc分配 sizeof(int)*10 字节，返回void*指针

    // ===== 第2步：检查返回值 =====
    if (ptr == NULL) {
        fprintf(stderr, "错误：内存分配失败\n");
        return;  // 一定要检查！
    }

    // ===== 第3步：使用内存 =====
    for (int i = 0; i < 10; i++) {
        ptr[i] = i * 10;
    }
    for (int i = 0; i < 10; i++) {
        printf("ptr[%d] = %d\n", i, ptr[i]);
    }

    // ===== 第4步：释放内存 =====
    free(ptr);

    // ===== 第5步：避免悬空指针 =====
    ptr = NULL;
}

int main(void) {
    malloc_free_basic();
    return 0;
}
```

### 3.2 分配字符串内存

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void string_allocation(void) {
    const char *source = "Hello, Dynamic Memory!";
    size_t len = strlen(source);

    // 为字符串分配内存 (+1 为 '\0')
    char *copy = (char *)malloc(len + 1);
    if (copy == NULL) {
        fprintf(stderr, "分配失败\n");
        return;
    }

    strcpy(copy, source);
    printf("复制的字符串: %s\n", copy);

    // 修改副本不影响原始数据
    copy[0] = 'h';
    printf("修改后: %s\n", copy);
    printf("原始值: %s\n", source);  // 未改变

    free(copy);
    copy = NULL;
}

int main(void) {
    string_allocation();
    return 0;
}
```

## 四、堆内存的生命周期

```c
#include <stdio.h>
#include <stdlib.h>

void demonstrate_heap_lifetime(void) {
    // 第一次分配
    int *a = malloc(sizeof(int));
    *a = 100;
    printf("a = %d, 地址 = %p\n", *a, (void *)a);

    // 第二次分配
    int *b = malloc(sizeof(int));
    *b = 200;
    printf("b = %d, 地址 = %p\n", *b, (void *)b);

    // 释放a后，a指向的内存归还给系统
    free(a);
    // 此时 *a 是未定义行为！
    // printf("%d\n", *a);  // 危险！

    // b仍然有效
    printf("b = %d (仍然有效)\n", *b);

    // 释放b
    free(b);
    b = NULL;
    a = NULL;  // 也置NULL防止误用
}

// 堆内存可以跨越函数边界
int *create_int(int value) {
    int *p = malloc(sizeof(int));
    if (p) *p = value;
    return p;  // 调用者负责free
}

void use_heap_across_functions(void) {
    int *val = create_int(42);
    if (val) {
        printf("值: %d\n", *val);
        free(val);  // 调用者负责释放
        val = NULL;
    }
}

int main(void) {
    demonstrate_heap_lifetime();
    use_heap_across_functions();
    return 0;
}
```

## 五、堆管理的基本原理

```
┌─────────────────────────────────────────────────┐
│               程序地址空间（堆区域）               │
├─────────────────────────────────────────────────┤
│                                                 │
│  [已分配块]  [空闲块]  [已分配块]  [空闲块]        │
│  ┌────────┐ ┌──────┐ ┌────────┐ ┌──────┐       │
│  │ 头部   │ │ 头部 │ │ 头部   │ │ 头部 │       │
│  │ 大小=32│ │大小=16│ │大小=64 │ │大小=48│       │
│  │ 状态=已│ │状态=空│ │状态=已 │ │状态=空│       │
│  ├────────┤ ├──────┤ ├────────┤ └──────┘       │
│  │ 数据   │ │(空闲)│ │ 数据   │  (空闲)         │
│  │ 32字节 │ │16字节│ │ 64字节 │  48字节         │
│  └────────┘ └──────┘ └────────┘                 │
│                                                 │
│  malloc从空闲链表中找到合适的块返回               │
│  free将块标记为空闲并加入空闲链表                  │
│                                                 │
└─────────────────────────────────────────────────┘
```

当调用`malloc`时：
1. 在空闲块链表中搜索合适大小的块
2. 找到后标记为"已分配"，返回数据区指针
3. 如果没有合适的块，通过`brk`/`sbrk`向系统申请更多内存

当调用`free`时：
1. 将块标记为"空闲"
2. 如果相邻有空闲块，进行合并
3. 将块加入空闲链表

## 六、常见错误模式

```c
#include <stdio.h>
#include <stdlib.h>

void common_mistakes(void) {
    // 错误1：忘记检查NULL
    int *p1 = malloc(1000000000);  // 可能失败
    // if (p1 == NULL) 必须检查！

    // 错误2：忘记释放
    int *p2 = malloc(sizeof(int) * 100);
    // 函数返回前没有 free(p2) → 内存泄漏!

    // 错误3：使用sizeof(指针)而非sizeof(类型)
    int *p3 = malloc(sizeof(p3));  // 错误！只分配了8字节(指针大小)
    int *p4 = malloc(sizeof(*p4)); // 正确！分配了4字节(int大小)

    // 错误4：free后继续使用
    free(p2);
    // *p2 = 10;  // 未定义行为!

    // 错误5：对同一指针多次free
    // free(p1);
    // free(p1);  // double free!

    // 正确清理
    if (p1) free(p1);
    if (p3) free(p3);
    if (p4) free(p4);
    p1 = p3 = p4 = NULL;
}

int main(void) {
    common_mistakes();
    return 0;
}
```

## 七、关键要点

> **堆内存要点**
> 1. 堆内存用于运行时动态分配，大小可以变化
> 2. 每次malloc必须检查返回值是否为NULL
> 3. 每次malloc必须有对应的free
> 4. free后立即将指针置NULL
> 5. 使用`sizeof(*ptr)`而非`sizeof(ptr)`来计算分配大小
> 6. 堆内存分配比栈慢10-100倍，但空间大得多
> 7. 堆内存可以跨越函数边界传递和使用
