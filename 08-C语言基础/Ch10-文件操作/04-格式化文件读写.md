# 04 - 格式化文件读写

## 一、fprintf - 格式化写入

### 函数原型

```c
#include <stdio.h>
int fprintf(FILE *stream, const char *format, ...);
```

- 将格式化数据写入流
- 返回值：成功返回写入的字符数，失败返回负值
- 用法与 `printf` 完全相同，只是多了一个文件指针参数

### 基本用法

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("data.txt", "w");
    if (fp == NULL) {
        perror("打开文件失败");
        return 1;
    }

    int age = 25;
    double salary = 8500.50;
    char name[] = "张三";

    // 格式化写入文件
    fprintf(fp, "姓名: %s\n", name);
    fprintf(fp, "年龄: %d\n", age);
    fprintf(fp, "薪资: %.2f\n", salary);
    fprintf(fp, "十六进制: 0x%X\n", 255);
    fprintf(fp, "八进制: 0%o\n", 255);
    fprintf(fp, "科学计数: %e\n", 123456.789);
    fprintf(fp, "百分比: %.1f%%\n", 85.5);

    fclose(fp);
    return 0;
}
```

### 输出文件内容

```
姓名: 张三
年龄: 25
薪资: 8500.50
十六进制: 0xFF
八进制: 0377
科学计数: 1.234568e+05
百分比: 85.5%
```

## 二、fscanf - 格式化读取

### 函数原型

```c
#include <stdio.h>
int fscanf(FILE *stream, const char *format, ...);
```

- 从流中按格式读取数据
- 返回值：成功匹配并赋值的项数，到达EOF返回 `EOF`
- 用法与 `scanf` 完全相同

### 基本用法

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("data.txt", "r");
    if (fp == NULL) {
        perror("打开文件失败");
        return 1;
    }

    char name[50];
    int age;
    double salary;

    // 按格式读取（跳过 "姓名: " 等前缀文字）
    int ret = fscanf(fp, "姓名: %s\n年龄: %d\n薪资: %lf\n", name, &age, &salary);

    if (ret == 3) {
        printf("读取成功: %s, %d岁, 薪资%.2f\n", name, age, salary);
    } else {
        printf("只成功读取了 %d 个字段\n", ret);
    }

    fclose(fp);
    return 0;
}
```

### fscanf 的格式说明符

| 说明符 | 含义 | 对应类型 |
|--------|------|---------|
| `%d` | 十进制整数 | `int*` |
| `%f` | 浮点数 | `float*` |
| `%lf` | 双精度浮点 | `double*` |
| `%s` | 字符串（遇空白停止） | `char*` |
| `%c` | 单个字符 | `char*` |
| `%[abc]` | 字符集（匹配abc中的字符） | `char*` |
| `%[^abc]` | 排除字符集（不匹配abc的字符） | `char*` |
| `%[^\n]` | 读到换行符为止（整行） | `char*` |
| `%%` | 匹配字面量 % | — |

### 使用 %[^...] 读取含空格的字符串

```c
// 文件内容: "Hello World" is a greeting

char phrase[100];
// 读取引号内的内容
fscanf(fp, "\"%[^\"]\"", phrase);
// phrase = "Hello World"
```

## 三、printf 家族函数对比

| 函数 | 输出目标 | 说明 |
|------|---------|------|
| `printf(format, ...)` | stdout | 标准输出 |
| `fprintf(stream, format, ...)` | 指定流 | 文件或任意流 |
| `sprintf(str, format, ...)` | 字符数组 | 不安全，不检查长度 |
| `snprintf(str, n, format, ...)` | 字符数组 | 安全版本，最多写n-1字符 |

| 函数 | 输入来源 | 说明 |
|------|---------|------|
| `scanf(format, ...)` | stdin | 标准输入 |
| `fscanf(stream, format, ...)` | 指定流 | 文件或任意流 |
| `sscanf(str, format, ...)` | 字符数组 | 从字符串解析 |

### sprintf vs snprintf

```c
char buf[10];

// 危险！可能导致缓冲区溢出
sprintf(buf, "%s", "Hello World");  // "Hello World" 需要12字节

// 安全：最多写入9个字符 + '\0'
snprintf(buf, sizeof(buf), "%s", "Hello World");  // buf = "Hello Wo\0"
```

## 四、综合示例

### 示例1：CSV格式数据的读写

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    char name[50];
    int age;
    double score;
} Student;

void write_csv(const char *filename, Student students[], int count) {
    FILE *fp = fopen(filename, "w");
    if (fp == NULL) {
        perror("打开文件失败");
        return;
    }

    // 写入标题行
    fprintf(fp, "姓名,年龄,成绩\n");

    // 写入数据行
    for (int i = 0; i < count; i++) {
        fprintf(fp, "%s,%d,%.1f\n",
                students[i].name,
                students[i].age,
                students[i].score);
    }

    fclose(fp);
}

void read_csv(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (fp == NULL) {
        perror("打开文件失败");
        return;
    }

    char line[256];
    // 跳过标题行
    fgets(line, sizeof(line), fp);

    while (fgets(line, sizeof(line), fp) != NULL) {
        char name[50];
        int age;
        double score;

        if (sscanf(line, "%49[^,],%d,%lf", name, &age, &score) == 3) {
            printf("姓名: %-10s 年龄: %d 成绩: %.1f\n", name, age, score);
        }
    }

    fclose(fp);
}

int main() {
    Student students[] = {
        {"张三", 20, 85.5},
        {"李四", 21, 92.0},
        {"王五", 19, 78.3}
    };

    write_csv("students.csv", students, 3);
    printf("=== CSV文件内容 ===\n");
    read_csv("students.csv");

    return 0;
}
```

### 示例2：配置文件读写

```c
#include <stdio.h>
#include <string.h>

// 写入配置
void write_config(const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (fp == NULL) { perror("写入配置失败"); return; }

    fprintf(fp, "# 应用程序配置文件\n");
    fprintf(fp, "hostname = localhost\n");
    fprintf(fp, "port = 8080\n");
    fprintf(fp, "max_connections = 100\n");
    fprintf(fp, "timeout = 30.5\n");
    fprintf(fp, "debug = true\n");

    fclose(fp);
}

// 读取配置
void read_config(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (fp == NULL) { perror("读取配置失败"); return; }

    char key[64], value[256];
    char line[512];

    while (fgets(line, sizeof(line), fp) != NULL) {
        // 跳过注释和空行
        if (line[0] == '#' || line[0] == '\n') continue;

        if (sscanf(line, "%63[^=] = %255[^\n]", key, value) == 2) {
            // 去除 key 末尾的空格
            char *end = key + strlen(key) - 1;
            while (end > key && *end == ' ') *end-- = '\0';

            printf("配置项: [%s] = [%s]\n", key, value);
        }
    }

    fclose(fp);
}

int main() {
    write_config("app.conf");
    read_config("app.conf");
    return 0;
}
```

## 五、fprintf 与 printf 的关系

```c
// 这三个调用完全等价
printf("Hello %s\n", name);                  // 输出到 stdout
fprintf(stdout, "Hello %s\n", name);         // 输出到 stdout（显式）
fprintf(fp, "Hello %s\n", name);             // 输出到文件

// scanf 的类似关系
scanf("%d", &n);                             // 从 stdin 读取
fscanf(stdin, "%d", &n);                     // 从 stdin 读取（显式）
fscanf(fp, "%d", &n);                        // 从文件读取
```

## 六、重点与注意事项

> **要点总结：**
>
> 1. `fprintf` 的格式字符串语法与 `printf` 完全一致
> 2. `fscanf` 返回成功匹配的字段数量，应与预期数量比较
> 3. `fscanf` 读取 `%s` 时遇到空白字符停止，无法读取含空格的字符串
> 4. 使用 `%[^...]` 可以实现更灵活的读取模式
> 5. `snprintf` 是 `sprintf` 的安全替代，始终使用前者
> 6. `fscanf` 的 `%lf` 对应 `double`，而 `scanf` 中 `%f` 对应 `float`
> 7. 格式化读写函数对文本文件最适用，二进制文件应使用 `fread/fwrite`
