# 24 - 跨平台文件操作

## 一、平台差异概述

### Windows vs Linux/macOS

| 特性 | Windows | Linux/macOS |
|------|---------|-------------|
| 路径分隔符 | `\` | `/` |
| 行尾符 | `\r\n` | `\n` |
| 文件名大小写 | 不区分 | 区分 |
| 文本/二进制模式 | 有区别 | 无区别 |
| 文件描述符 | 有限（默认512） | 丰富（默认1024） |
| 文件API | `_open`, `_read` | `open`, `read` |
| 最大路径长度 | 260（或32767） | 4096 |

## 二、路径处理

### 路径分隔符

```c
#include <stdio.h>
#include <string.h>

// 跨平台路径分隔符
#ifdef _WIN32
    #define PATH_SEP '\\'
    #define PATH_SEP_STR "\\"
#else
    #define PATH_SEP '/'
    #define PATH_SEP_STR "/"
#endif

// 拼接路径
void path_join(char *dest, size_t dest_size,
               const char *dir, const char *file) {
    size_t dir_len = strlen(dir);

    // 检查是否已有尾部分隔符
    if (dir_len > 0 && dir[dir_len - 1] == PATH_SEP) {
        snprintf(dest, dest_size, "%s%s", dir, file);
    } else {
        snprintf(dest, dest_size, "%s%c%s", dir, PATH_SEP, file);
    }
}

// 统一路径中的分隔符
void normalize_path(char *path) {
    for (char *p = path; *p; p++) {
        if (*p == '/' || *p == '\\') {
            *p = PATH_SEP;
        }
    }
}

// 获取文件扩展名
const char *get_extension(const char *filename) {
    const char *dot = strrchr(filename, '.');
    return (dot && dot != filename) ? dot + 1 : "";
}
```

### 处理不同风格的路径

```c
#include <string.h>

// 将Windows路径转为跨平台路径
void to_posix_path(char *path) {
    for (char *p = path; *p; p++) {
        if (*p == '\\') *p = '/';
    }
}

// 将POSIX路径转为Windows路径
void to_windows_path(char *path) {
    for (char *p = path; *p; p++) {
        if (*p == '/') *p = '\\';
    }
}
```

## 三、文件操作API的跨平台封装

### 统一的文件打开接口

```c
#include <stdio.h>
#include <string.h>

#ifdef _WIN32
    #include <windows.h>
    #include <io.h>
    #include <direct.h>
    #define F_OK 0
    #define R_OK 4
    #define W_OK 2
    // Windows特定的函数重定义
    #define access _access
    #define mkdir(path, mode) _mkdir(path)
    #define unlink _unlink
    #define getcwd _getcwd
#else
    #include <unistd.h>
    #include <sys/stat.h>
    #include <dirent.h>
#endif

// 检查文件是否存在
int file_exists(const char *path) {
    return access(path, F_OK) == 0;
}

// 检查是否为目录
int is_directory(const char *path) {
#ifdef _WIN32
    DWORD attrs = GetFileAttributes(path);
    return (attrs != INVALID_FILE_ATTRIBUTES) &&
           (attrs & FILE_ATTRIBUTE_DIRECTORY);
#else
    struct stat st;
    return (stat(path, &st) == 0) && S_ISDIR(st.st_mode);
#endif
}

// 创建目录（递归）
int mkdir_recursive(const char *path, int mode) {
    char tmp[1024];
    char *p = NULL;

    snprintf(tmp, sizeof(tmp), "%s", path);

    for (p = tmp + 1; *p; p++) {
        if (*p == PATH_SEP) {
            *p = '\0';
            mkdir(tmp, mode);
            *p = PATH_SEP;
        }
    }

    return mkdir(tmp, mode);
}
```

### 跨平台的临时文件

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

FILE *create_temp_file(char *name_buf, size_t buf_size) {
#ifdef _WIN32
    char temp_dir[MAX_PATH];
    GetTempPath(MAX_PATH, temp_dir);
    GetTempFileName(temp_dir, "tmp", 0, name_buf);
    return fopen(name_buf, "w+b");
#else
    const char *tmpdir = getenv("TMPDIR");
    if (!tmpdir) tmpdir = "/tmp";
    snprintf(name_buf, buf_size, "%s/tmpfile_XXXXXX", tmpdir);

    int fd = mkstemp(name_buf);
    if (fd == -1) return NULL;

    FILE *fp = fdopen(fd, "w+b");
    if (!fp) {
        close(fd);
        unlink(name_buf);
    }
    return fp;
#endif
}
```

## 四、文本模式与二进制模式

### 为什么Windows需要区分

```
Windows文本模式下的自动转换：
  写入 '\n' (0x0A) → 实际写入 '\r\n' (0x0D 0x0A)
  读取 '\r\n' → 实际得到 '\n'

Linux不需要区分：
  文本和二进制模式行为完全一致
```

### 跨平台建议

```c
// 始终显式指定模式
FILE *fp = fopen("data.txt", "r");   // 文本模式
FILE *fp = fopen("data.bin", "rb");  // 二进制模式

// 处理不同平台的换行符
int read_line(FILE *fp, char *buf, size_t size) {
    if (!fgets(buf, size, fp)) return 0;

    // 去除末尾的 \r\n 或 \n
    size_t len = strlen(buf);
    while (len > 0 && (buf[len-1] == '\n' || buf[len-1] == '\r')) {
        buf[--len] = '\0';
    }
    return 1;
}
```

### 完整的跨平台文件工具库

```c
#include <stdio.h>
#include <string.h>
#include <sys/stat.h>

#ifdef _WIN32
    #include <windows.h>
    #include <direct.h>
    #define PATH_SEP '\\'
    #define stat _stat
#else
    #include <unistd.h>
    #include <dirent.h>
    #define PATH_SEP '/'
#endif

// 获取文件大小
long long get_file_size(const char *path) {
    struct stat st;
    if (stat(path, &st) != 0) return -1;
    return (long long)st.st_size;
}

// 获取当前工作目录
int get_current_dir(char *buf, size_t size) {
#ifdef _WIN32
    return GetCurrentDirectory((DWORD)size, buf) > 0 ? 0 : -1;
#else
    return getcwd(buf, size) != NULL ? 0 : -1;
#endif
}

// 删除文件
int remove_file(const char *path) {
    return remove(path);
}

// 重命名/移动文件
int rename_file(const char *old_name, const char *new_name) {
    return rename(old_name, new_name);
}

int main() {
    char cwd[1024];
    get_current_dir(cwd, sizeof(cwd));
    printf("当前目录: %s\n", cwd);

    printf("路径分隔符: %c\n", PATH_SEP);

    return 0;
}
```

## 六、重点与注意事项

> **要点总结：**
>
> 1. **始终使用 `"/"` 作为路径分隔符**——Windows 的C库也支持正斜杠
> 2. **二进制文件始终使用 `"rb"`/`"wb"` 模式**——跨平台行为一致
> 3. **处理换行符时兼容 `\r\n` 和 `\n`**——不要假设只有一种
> 4. 使用条件编译 `#ifdef _WIN32` 处理平台差异
> 5. **文件名大小写在Linux上敏感，在Windows上不敏感**
> 6. 避免使用超过260字符的路径——Windows传统限制
> 7. 标准C库的 `fopen`/`fread` 在所有平台行为一致，优先使用
