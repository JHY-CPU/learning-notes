# 07 - 文件状态与错误处理

## 一、文件错误检测

### feof - 检测文件结束

```c
#include <stdio.h>
int feof(FILE *stream);
```

- 如果设置了文件结束标志，返回非零值
- 文件结束标志在读取操作**尝试越过文件末尾**时设置

### feof 的常见误解

```c
// ❌ 错误用法：先判断 feof 再读取
while (!feof(fp)) {        // 错误！feof 在读取前不会返回真
    fscanf(fp, "%d", &n);
    printf("%d\n", n);     // 会多打印一次最后的值
}

// ✅ 正确用法：先读取，再判断
while (fscanf(fp, "%d", &n) == 1) {
    printf("%d\n", n);
}

// ✅ 另一种正确用法
int n;
while (fscanf(fp, "%d", &n) != EOF) {
    printf("%d\n", n);
}
```

### feof 的工作原理

```
文件内容: 10 20 30

操作序列：
  fscanf(fp, "%d", &n) → n=10, feof()=0, 返回1
  fscanf(fp, "%d", &n) → n=20, feof()=0, 返回1
  fscanf(fp, "%d", &n) → n=30, feof()=0, 返回1
  fscanf(fp, "%d", &n) → 尝试读取越过末尾, feof()=1, 返回EOF
```

### ferror - 检测错误标志

```c
#include <stdio.h>
int ferror(FILE *stream);
```

- 如果流上发生了读写错误，返回非零值
- 错误标志一旦设置，不会自动清除（除非使用 `clearerr` 或 `rewind`）

### ferror 与 feof 的区别

```
ferror(): 检测读写操作是否出错（如磁盘故障、权限问题）
feof():   检测是否到达文件末尾

两者都通过读取操作的结果来设置，不能预先判断。
```

### clearerr - 清除错误和结束标志

```c
void clearerr(FILE *stream);
```

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("data.txt", "r");

    // 尝试读取到EOF
    int ch;
    while ((ch = fgetc(fp)) != EOF) { }

    printf("feof: %d\n", feof(fp));     // 1（到达末尾）
    printf("ferror: %d\n", ferror(fp)); // 0（无错误）

    clearerr(fp);
    printf("清除后 feof: %d\n", feof(fp)); // 0（标志已清除）

    fclose(fp);
    return 0;
}
```

## 二、errno 与错误码

### errno 变量

```c
#include <errno.h>
```

`errno` 是一个全局变量（实际上是线程局部存储），系统调用和库函数在出错时会设置它。

### 常见的 errno 值

| 值 | 宏名 | 含义 |
|----|------|------|
| 0 | — | 无错误 |
| 1 | EPERM | 操作不允许 |
| 2 | ENOENT | 文件或目录不存在 |
| 5 | EIO | I/O错误 |
| 12 | ENOMEM | 内存不足 |
| 13 | EACCES | 权限被拒绝 |
| 17 | EEXIST | 文件已存在 |
| 21 | EISDIR | 是一个目录 |
| 22 | EINVAL | 无效参数 |
| 24 | EMFILE | 打开的文件数过多 |
| 28 | ENOSPC | 磁盘空间不足 |

### errno 使用示例

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

int main() {
    errno = 0;  // 清除之前的错误

    FILE *fp = fopen("/nonexistent/file.txt", "r");
    if (fp == NULL) {
        printf("errno = %d\n", errno);

        // 两种方式获取错误描述
        // 方式1: strerror
        printf("错误描述: %s\n", strerror(errno));

        // 方式2: perror（自动追加冒号和错误描述）
        perror("fopen");

        return 1;
    }

    fclose(fp);
    return 0;
}
```

### 输出示例

```
errno = 2
错误描述: No such file or directory
fopen: No such file or directory
```

## 三、strerror - 错误码转字符串

```c
#include <string.h>
char *strerror(int errnum);
```

```c
// 将错误码转换为可读的字符串描述
printf("%s\n", strerror(0));    // "Success" (或"没有错误")
printf("%s\n", strerror(2));    // "No such file or directory"
printf("%s\n", strerror(13));   // "Permission denied"
printf("%s\n", strerror(28));   // "No space left on device"

// 配合 errno 使用
FILE *fp = fopen("test.txt", "r");
if (!fp) {
    fprintf(stderr, "打开失败: %s (errno=%d)\n", strerror(errno), errno);
}
```

## 四、perror - 输出错误信息

```c
#include <stdio.h>
void perror(const char *s);
```

- 向 `stderr` 输出：`s` + `: ` + `errno对应的描述字符串` + `\n`

```c
FILE *fp = fopen("readonly.txt", "w");
if (!fp) {
    perror("打开文件失败");
    // 输出: 打开文件失败: Permission denied
}

// 如果 s 为空字符串或NULL
perror("");   // 输出: Permission denied
```

## 五、完整的错误处理模式

### 模式1：简单检查

```c
FILE *fp = fopen("data.txt", "r");
if (fp == NULL) {
    perror("fopen");
    return 1;
}
```

### 模式2：详细错误信息

```c
FILE *fp = fopen("data.txt", "r");
if (fp == NULL) {
    fprintf(stderr, "错误: 无法打开文件 'data.txt'\n");
    fprintf(stderr, "原因: %s\n", strerror(errno));
    fprintf(stderr, "错误码: %d\n", errno);
    return 1;
}
```

### 模式3：根据不同错误类型处理

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

FILE *open_with_retry(const char *filename, const char *mode) {
    FILE *fp = fopen(filename, mode);
    if (fp == NULL) {
        switch (errno) {
            case ENOENT:
                fprintf(stderr, "文件 '%s' 不存在\n", filename);
                break;
            case EACCES:
                fprintf(stderr, "没有权限访问 '%s'\n", filename);
                break;
            case ENOSPC:
                fprintf(stderr, "磁盘空间不足，无法创建 '%s'\n", filename);
                break;
            case EMFILE:
                fprintf(stderr, "打开的文件数已达上限\n");
                break;
            default:
                fprintf(stderr, "打开 '%s' 失败: %s\n",
                        filename, strerror(errno));
        }
    }
    return fp;
}
```

## 六、综合示例：健壮的文件操作

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

int copy_file(const char *src_path, const char *dst_path) {
    FILE *src = fopen(src_path, "rb");
    if (!src) {
        fprintf(stderr, "无法打开源文件 '%s': %s\n",
                src_path, strerror(errno));
        return -1;
    }

    FILE *dst = fopen(dst_path, "wb");
    if (!dst) {
        fprintf(stderr, "无法创建目标文件 '%s': %s\n",
                dst_path, strerror(errno));
        fclose(src);
        return -1;
    }

    char buffer[4096];
    size_t bytes_read;
    size_t total_bytes = 0;

    while ((bytes_read = fread(buffer, 1, sizeof(buffer), src)) > 0) {
        size_t bytes_written = fwrite(buffer, 1, bytes_read, dst);

        if (bytes_written != bytes_read) {
            fprintf(stderr, "写入错误: %s\n", strerror(errno));
            fclose(src);
            fclose(dst);
            remove(dst_path);  // 删除不完整的文件
            return -1;
        }
        total_bytes += bytes_written;
    }

    // 检查读取是否有错误（非EOF原因结束）
    if (ferror(src)) {
        fprintf(stderr, "读取错误: %s\n", strerror(errno));
        fclose(src);
        fclose(dst);
        remove(dst_path);
        return -1;
    }

    fclose(src);
    fclose(dst);

    printf("复制成功: %zu 字节\n", total_bytes);
    return 0;
}

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "用法: %s <源文件> <目标文件>\n", argv[0]);
        return 1;
    }

    return copy_file(argv[1], argv[2]) == 0 ? 0 : 1;
}
```

## 七、重点与注意事项

> **要点总结：**
>
> 1. **`feof()` 不能预先判断**，必须在读取操作后才有效
> 2. **`errno` 是全局变量**，每次出错后会立即被覆盖，需及时保存
> 3. **`perror` 和 `strerror`** 都能将 errno 转为可读字符串，前者直接输出，后者返回字符串
> 4. **`clearerr`** 可以清除 `feof` 和 `ferror` 设置的标志
> 5. 使用 `fread`/`fwrite` 后应检查返回值并判断是 EOF 还是错误
> 6. 文件操作的错误处理应遵循：**打开失败 → 操作失败 → 关闭失败**的完整检查
> 7. 在多线程环境中，errno 是线程安全的（线程局部存储），但在信号处理函数中需谨慎
