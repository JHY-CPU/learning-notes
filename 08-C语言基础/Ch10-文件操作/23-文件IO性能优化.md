# 23 - 文件IO性能优化

## 一、性能瓶颈分析

### 文件IO的层次结构

```
用户程序
  │
  ▼
标准IO库（缓冲区管理）
  │
  ▼
系统调用（read/write）
  │
  ▼
内核缓冲区（Page Cache）
  │
  ▼
磁盘驱动
  │
  ▼
物理磁盘
```

### 每层的开销

| 层次 | 开销类型 | 典型延迟 |
|------|---------|---------|
| 用户态内存访问 | CPU | ~1ns |
| 系统调用 | 上下文切换 | ~100ns |
| 页缓存命中 | 内存访问 | ~100ns |
| 顺序磁盘读取 | 磁盘I/O | ~100MB/s |
| 随机磁盘读取 | 磁盘寻道 | ~5-10ms |

## 二、缓冲区大小优化

### 缓冲区大小对性能的影响

```c
#include <stdio.h>
#include <time.h>
#include <string.h>

void benchmark_buffer_size(const char *src, const char *dst,
                           size_t buffer_size) {
    FILE *in = fopen(src, "rb");
    FILE *out = fopen(dst, "wb");

    char *buffer = malloc(buffer_size);

    clock_t start = clock();

    size_t bytes;
    while ((bytes = fread(buffer, 1, buffer_size, in)) > 0) {
        fwrite(buffer, 1, bytes, out);
    }

    clock_t end = clock();
    double elapsed = (double)(end - start) / CLOCKS_PER_SEC;

    printf("缓冲区 %6zu 字节: %.4f 秒\n", buffer_size, elapsed);

    free(buffer);
    fclose(in);
    fclose(out);
}

int main() {
    // 测试不同缓冲区大小
    size_t sizes[] = {64, 256, 1024, 4096, 8192, 65536, 1048576};
    int n = sizeof(sizes) / sizeof(sizes[0]);

    for (int i = 0; i < n; i++) {
        benchmark_buffer_size("large_file.bin", "copy.bin", sizes[i]);
    }

    return 0;
}
```

### 典型结果

```
缓冲区    64 字节: 2.3500 秒
缓冲区   256 字节: 0.8900 秒
缓冲区  1024 字节: 0.4200 秒
缓冲区  4096 字节: 0.2100 秒  ← 推荐最小值
缓冲区  8192 字节: 0.1500 秒
缓冲区 65536 字节: 0.0900 秒  ← 推荐值
缓冲区 1048576 字节: 0.0850 秒 ← 收益递减
```

### 推荐的缓冲区大小

```
通用场景：4KB - 64KB
大文件处理：64KB - 1MB
实时性要求高：4KB
内存受限：1KB - 4KB
```

## 三、批量读写 vs 逐个读写

### 性能对比

```c
#include <stdio.h>
#include <time.h>

// 方式1：逐字节读写
void copy_byte_by_byte(const char *src, const char *dst) {
    FILE *in = fopen(src, "rb");
    FILE *out = fopen(dst, "wb");

    int ch;
    while ((ch = fgetc(in)) != EOF) {
        fputc(ch, out);
    }

    fclose(in);
    fclose(out);
}

// 方式2：批量读写
void copy_batch(const char *src, const char *dst) {
    FILE *in = fopen(src, "rb");
    FILE *out = fopen(dst, "wb");

    char buffer[65536];
    size_t n;

    while ((n = fread(buffer, 1, sizeof(buffer), in)) > 0) {
        fwrite(buffer, 1, n, out);
    }

    fclose(in);
    fclose(out);
}

int main() {
    clock_t t1, t2;

    t1 = clock();
    copy_byte_by_byte("large.bin", "copy1.bin");
    t2 = clock();
    printf("逐字节: %.3f 秒\n", (double)(t2 - t1) / CLOCKS_PER_SEC);

    t1 = clock();
    copy_batch("large.bin", "copy2.bin");
    t2 = clock();
    printf("批量:   %.3f 秒\n", (double)(t2 - t1) / CLOCKS_PER_SEC);

    return 0;
}
```

```
典型结果（10MB文件）：
逐字节: 1.850 秒（100万次fgetc+fputc调用）
批量:   0.012 秒（约160次fread+fwrite调用）
差异：约150倍！
```

## 四、减少系统调用

### setvbuf 增大缓冲区

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("output.txt", "w");

    // 设置更大的缓冲区，减少系统调用次数
    char buf[65536];
    setvbuf(fp, buf, _IOFBF, sizeof(buf));

    // 写入大量数据
    for (int i = 0; i < 100000; i++) {
        fprintf(fp, "第 %d 行数据\n", i);
    }

    fclose(fp);
    return 0;
}
```

## 五、mmap vs 标准IO 性能对比

```c
#include <stdio.h>
#include <time.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <string.h>

// 标准IO方式
long long sum_bytes_fread(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return -1;

    long long sum = 0;
    char buffer[65536];
    size_t n;

    while ((n = fread(buffer, 1, sizeof(buffer), fp)) > 0) {
        for (size_t i = 0; i < n; i++) {
            sum += (unsigned char)buffer[i];
        }
    }

    fclose(fp);
    return sum;
}

// mmap方式
long long sum_bytes_mmap(const char *filename) {
    int fd = open(filename, O_RDONLY);
    if (fd == -1) return -1;

    struct stat st;
    fstat(fd, &st);

    char *data = mmap(NULL, st.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
    if (data == MAP_FAILED) { close(fd); return -1; }

    long long sum = 0;
    for (off_t i = 0; i < st.st_size; i++) {
        sum += (unsigned char)data[i];
    }

    munmap(data, st.st_size);
    close(fd);
    return sum;
}

int main() {
    clock_t t;

    t = clock();
    long long s1 = sum_bytes_fread("large.bin");
    printf("fread: %.4f 秒, 结果=%lld\n",
           (double)(clock() - t) / CLOCKS_PER_SEC, s1);

    t = clock();
    long long s2 = sum_bytes_mmap("large.bin");
    printf("mmap:  %.4f 秒, 结果=%lld\n",
           (double)(clock() - t) / CLOCKS_PER_SEC, s2);

    return 0;
}
```

## 六、顺序读写 vs 随机读写

```c
#include <stdio.h>
#include <time.h>
#include <stdlib.h>

#define RECORD_SIZE 256
#define NUM_RECORDS 100000

// 顺序读取
void sequential_read(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    char buffer[RECORD_SIZE];

    for (int i = 0; i < NUM_RECORDS; i++) {
        fread(buffer, 1, RECORD_SIZE, fp);
    }

    fclose(fp);
}

// 随机读取
void random_read(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    char buffer[RECORD_SIZE];

    for (int i = 0; i < NUM_RECORDS; i++) {
        int record = rand() % NUM_RECORDS;
        fseek(fp, record * RECORD_SIZE, SEEK_SET);
        fread(buffer, 1, RECORD_SIZE, fp);
    }

    fclose(fp);
}

int main() {
    clock_t t;

    t = clock();
    sequential_read("records.dat");
    printf("顺序读取: %.4f 秒\n", (double)(clock() - t) / CLOCKS_PER_SEC);

    t = clock();
    random_read("records.dat");
    printf("随机读取: %.4f 秒\n", (double)(clock() - t) / CLOCKS_PER_SEC);

    return 0;
}
```

```
典型结果：
顺序读取: 0.0050 秒（充分利用预读缓存）
随机读取: 0.8200 秒（频繁磁盘寻道）
差异：约160倍！
```

## 七、性能优化总结

### 优化策略清单

```
1. 使用合适的缓冲区大小（4KB-64KB）
2. 批量读写，避免逐字节操作
3. 顺序读写优于随机读写
4. 大文件考虑使用 mmap
5. 减少 fflush 调用次数
6. 使用 setvbuf 调整缓冲策略
7. 多线程场景考虑预读（readahead）
8. SSD 上随机读写性能损失较小
9. 避免频繁打开/关闭文件
10. 预分配文件大小（ftruncate/posix_fallocate）
```

## 八、重点与注意事项

> **要点总结：**
>
> 1. **缓冲区太小（<4KB）会严重影响性能**——系统调用开销占主导
> 2. **缓冲区太大（>1MB）收益递减**——浪费内存
> 3. **顺序访问比随机访问快数十到数百倍**——尽量保持顺序
> 4. **mmap 适合随机访问大文件**——顺序扫描时与标准IO差异不大
> 5. 性能测试应使用足够的数据量和多次运行取平均
> 6. 磁盘类型（HDD vs SSD）会显著影响随机IO性能
> 7. 编译优化（-O2）和预读缓存会影响测试结果
