# 12 - 文件权限与属性

## 一、stat 系列函数

### 函数原型

```c
#include <sys/stat.h>

int stat(const char *path, struct stat *buf);       // 通过路径
int fstat(int fd, struct stat *buf);                 // 通过文件描述符
int lstat(const char *path, struct stat *buf);       // 不跟随符号链接
```

- **返回值**：成功返回 `0`，失败返回 `-1`（设置 errno）
- `lstat` 与 `stat` 的区别：`lstat` 遇到符号链接时返回链接本身的信息，而 `stat` 返回链接目标的信息

### struct stat 结构体

```c
struct stat {
    dev_t     st_dev;      // 文件所在设备的ID
    ino_t     st_ino;      // inode 号
    mode_t    st_mode;     // 文件类型和权限位
    nlink_t   st_nlink;    // 硬链接数
    uid_t     st_uid;      // 所有者用户ID
    gid_t     st_gid;      // 所有者组ID
    dev_t     st_rdev;     // 设备类型（仅特殊文件）
    off_t     st_size;     // 文件大小（字节）
    blksize_t st_blksize;  // 文件系统I/O块大小
    blkcnt_t  st_blocks;   // 分配的块数
    time_t    st_atime;    // 最后访问时间
    time_t    st_mtime;    // 最后修改时间
    time_t    st_ctime;    // 最后状态变更时间
};
```

### 基本使用

```c
#include <stdio.h>
#include <sys/stat.h>
#include <time.h>

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "用法: %s <文件路径>\n", argv[0]);
        return 1;
    }

    struct stat st;
    if (stat(argv[1], &st) == -1) {
        perror("stat 失败");
        return 1;
    }

    printf("文件: %s\n", argv[1]);
    printf("大小: %ld 字节\n", (long)st.st_size);
    printf("inode: %lu\n", (unsigned long)st.st_ino);
    printf("硬链接数: %lu\n", (unsigned long)st.st_nlink);
    printf("修改时间: %s", ctime(&st.st_mtime));

    return 0;
}
```

## 二、文件类型判断

### st_mode 中的文件类型位

```c
#include <sys/stat.h>

// 宏函数：判断文件类型
S_ISREG(mode)   // 普通文件 (regular)
S_ISDIR(mode)   // 目录 (directory)
S_ISLNK(mode)   // 符号链接 (symbolic link)
S_ISCHR(mode)   // 字符设备 (character device)
S_ISBLK(mode)   // 块设备 (block device)
S_ISFIFO(mode)  // 命名管道 (FIFO / named pipe)
S_ISSOCK(mode)  // 套接字 (socket)
```

### 文件类型判断示例

```c
#include <stdio.h>
#include <sys/stat.h>

const char *file_type(mode_t mode) {
    if (S_ISREG(mode))  return "普通文件";
    if (S_ISDIR(mode))  return "目录";
    if (S_ISLNK(mode))  return "符号链接";
    if (S_ISCHR(mode))  return "字符设备";
    if (S_ISBLK(mode))  return "块设备";
    if (S_ISFIFO(mode)) return "命名管道";
    if (S_ISSOCK(mode)) return "套接字";
    return "未知类型";
}

int main(int argc, char *argv[]) {
    struct stat st;
    if (stat(argv[1], &st) == -1) {
        perror("stat");
        return 1;
    }

    printf("%s 是一个 %s\n", argv[1], file_type(st.st_mode));
    return 0;
}
```

## 三、文件权限位

### 权限位结构

```
st_mode 的权限位（9位）：

  所有者(u)    组(g)      其他人(o)
  ┌──┬──┬──┐ ┌──┬──┬──┐ ┌──┬──┬──┐
  │r │w │x │ │r │w │x │ │r │w │x │
  │4 │2 │1 │ │4 │2 │1 │ │4 │2 │1 │
  └──┴──┴──┘ └──┴──┴──┘ └──┴──┴──┘

  例如：rwxr-xr-x = 111 101 101 = 0755
       rw-r--r-- = 110 100 100 = 0644
```

### 权限相关的宏

```c
// 所有者权限
S_IRUSR   // 0400  读
S_IWUSR   // 0200  写
S_IXUSR   // 0100  执行

// 组权限
S_IRGRP   // 0040  读
S_IWGRP   // 0020  写
S_IXGRP   // 0010  执行

// 其他人权限
S_IROTH   // 0004  读
S_IWOTH   // 0002  写
S_IXOTH   // 0001  执行

// 特殊权限
S_ISUID   // 4000  设置用户ID
S_ISGID   // 2000  设置组ID
S_ISVTX   // 1000  粘滞位
```

### 权限检查示例

```c
#include <stdio.h>
#include <sys/stat.h>

void print_permissions(const char *filename) {
    struct stat st;
    if (stat(filename, &st) == -1) {
        perror("stat");
        return;
    }

    mode_t m = st.st_mode;
    char perm[11];

    // 文件类型
    perm[0] = S_ISDIR(m) ? 'd' :
              S_ISLNK(m) ? 'l' :
              S_ISCHR(m) ? 'c' :
              S_ISBLK(m) ? 'b' :
              S_ISFIFO(m) ? 'p' :
              S_ISSOCK(m) ? 's' : '-';

    // 权限位
    perm[1] = (m & S_IRUSR) ? 'r' : '-';
    perm[2] = (m & S_IWUSR) ? 'w' : '-';
    perm[3] = (m & S_IXUSR) ? 'x' : '-';
    perm[4] = (m & S_IRGRP) ? 'r' : '-';
    perm[5] = (m & S_IWGRP) ? 'w' : '-';
    perm[6] = (m & S_IXGRP) ? 'x' : '-';
    perm[7] = (m & S_IROTH) ? 'r' : '-';
    perm[8] = (m & S_IWOTH) ? 'w' : '-';
    perm[9] = (m & S_IXOTH) ? 'x' : '-';
    perm[10] = '\0';

    printf("%s  %s\n", perm, filename);
}

int main(int argc, char *argv[]) {
    for (int i = 1; i < argc; i++) {
        print_permissions(argv[i]);
    }
    return 0;
}
```

## 四、修改文件权限

### chmod - 修改权限

```c
#include <sys/stat.h>
int chmod(const char *path, mode_t mode);
```

```c
#include <stdio.h>
#include <sys/stat.h>

int main() {
    // 设置权限为 rw-r--r-- (0644)
    if (chmod("data.txt", S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH) == 0) {
        printf("权限修改成功\n");
    } else {
        perror("chmod 失败");
    }

    // 八进制方式
    chmod("script.sh", 0755);  // rwxr-xr-x

    // 添加执行权限（需要先读取当前权限）
    struct stat st;
    stat("myfile.txt", &st);
    chmod("myfile.txt", st.st_mode | S_IXUSR);

    return 0;
}
```

## 五、文件大小获取

### 使用 stat 获取文件大小

```c
#include <stdio.h>
#include <sys/stat.h>

long long get_file_size(const char *filename) {
    struct stat st;
    if (stat(filename, &st) == -1) {
        return -1;
    }
    return (long long)st.st_size;
}

// 格式化显示文件大小
void print_human_size(long long bytes) {
    const char *units[] = {"B", "KB", "MB", "GB", "TB"};
    double size = (double)bytes;
    int i = 0;

    while (size >= 1024.0 && i < 4) {
        size /= 1024.0;
        i++;
    }

    printf("%.2f %s\n", size, units[i]);
}
```

### 使用 fseek/ftell 获取文件大小

```c
#include <stdio.h>

long get_file_size_fseek(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return -1;

    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fclose(fp);

    return size;
}
```

## 六、文件时间信息

```c
#include <stdio.h>
#include <sys/stat.h>
#include <time.h>

void print_file_times(const char *filename) {
    struct stat st;
    if (stat(filename, &st) == -1) {
        perror("stat");
        return;
    }

    printf("文件: %s\n", filename);
    printf("最后访问: %s", ctime(&st.st_atime));
    printf("最后修改: %s", ctime(&st.st_mtime));
    printf("状态变更: %s", ctime(&st.st_ctime));
}
```

## 七、access - 检查访问权限

```c
#include <unistd.h>
int access(const char *pathname, int mode);
```

| mode | 含义 |
|------|------|
| `F_OK` | 文件是否存在 |
| `R_OK` | 是否可读 |
| `W_OK` | 是否可写 |
| `X_OK` | 是否可执行 |

```c
#include <stdio.h>
#include <unistd.h>

int main() {
    const char *file = "test.txt";

    if (access(file, F_OK) == 0) {
        printf("文件存在\n");
    }

    if (access(file, R_OK) == 0) {
        printf("文件可读\n");
    }

    if (access(file, W_OK) == 0) {
        printf("文件可写\n");
    } else {
        printf("文件不可写 (权限不足)\n");
    }

    return 0;
}
```

## 八、重点与注意事项

> **要点总结：**
>
> 1. `stat` 的 `st_mode` 字段同时包含文件类型和权限信息
> 2. 使用 `S_ISxxx()` 宏判断文件类型，不要直接操作位
> 3. `lstat` 用于符号链接，返回链接本身而非目标文件的信息
> 4. 文件权限在Windows上语义不同（无用户/组/其他人的区分）
> 5. `access()` 检查的是调用进程的实际权限（考虑用户身份）
> 6. `stat` 返回的 `st_size` 对于符号链接是路径字符串长度，不是目标文件大小
> 7. 修改权限需要文件所有者或 root 权限
