# 11 - 目录操作

## 一、目录操作概述

### 文件与目录的关系

```
文件系统结构（树状）：
/
├── home/
│   └── user/
│       ├── documents/
│       │   ├── report.txt
│       │   └── data.csv
│       └── pictures/
│           └── photo.jpg
├── tmp/
└── etc/
    └── config.ini

目录也是一种特殊的"文件"，存储其他文件和目录的元数据。
```

### 标准与平台差异

| 操作 | POSIX (Linux/macOS) | Windows |
|------|---------------------|---------|
| 创建目录 | `mkdir(path, mode)` | `_mkdir(path)` |
| 打开目录 | `opendir()` | `_opendir()` 或 FindFirstFile |
| 读取目录 | `readdir()` | `_readdir()` 或 FindNextFile |
| 关闭目录 | `closedir()` | `_closedir()` |
| 删除目录 | `rmdir()` | `_rmdir()` |

## 二、POSIX 目录API

### 需要的头文件

```c
#include <sys/types.h>
#include <sys/stat.h>
#include <dirent.h>    // opendir, readdir, closedir
#include <unistd.h>    // rmdir, chdir, getcwd
```

### mkdir - 创建目录

```c
#include <sys/stat.h>
int mkdir(const char *path, mode_t mode);
```

- `path`：目录路径
- `mode`：权限位（Linux/macOS），Windows忽略
- **返回值**：成功返回 `0`，失败返回 `-1`

```c
#include <sys/stat.h>
#include <stdio.h>

int main() {
    // 创建目录（权限 0755 = rwxr-xr-x）
    if (mkdir("newdir", 0755) == 0) {
        printf("目录创建成功\n");
    } else {
        perror("创建目录失败");
        // 常见错误: EEXIST(目录已存在), EACCES(权限不足)
    }

    // 递归创建目录需要逐层创建
    mkdir("parent", 0755);
    mkdir("parent/child", 0755);
    mkdir("parent/child/grandchild", 0755);

    return 0;
}
```

### opendir / readdir / closedir - 遍历目录

```c
DIR *opendir(const char *name);
struct dirent *readdir(DIR *dirp);
int closedir(DIR *dirp);
```

```c
#include <stdio.h>
#include <dirent.h>

int main() {
    DIR *dir = opendir(".");
    if (!dir) {
        perror("打开目录失败");
        return 1;
    }

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        printf("%-20s  [ino: %lu, type: %d]\n",
               entry->d_name,
               (unsigned long)entry->d_ino,
               entry->d_type);
    }

    closedir(dir);
    return 0;
}
```

### dirent 结构体

```c
struct dirent {
    ino_t          d_ino;       // inode 号
    off_t          d_off;       // 目录偏移量
    unsigned short d_reclen;    // 记录长度
    unsigned char  d_type;      // 文件类型
    char           d_name[256]; // 文件名
};

// d_type 常见值：
// DT_REG  - 普通文件
// DT_DIR  - 目录
// DT_LNK  - 符号链接
// DT_FIFO - 命名管道
// DT_SOCK - 套接字
// DT_CHR  - 字符设备
// DT_BLK  - 块设备
// DT_UNKNOWN - 未知（某些文件系统不支持）
```

## 三、综合示例

### 示例1：列出目录中所有文件

```c
#include <stdio.h>
#include <string.h>
#include <dirent.h>
#include <sys/stat.h>

void list_directory(const char *path) {
    DIR *dir = opendir(path);
    if (!dir) {
        perror("打开目录失败");
        return;
    }

    printf("目录: %s\n", path);
    printf("%-30s %10s %s\n", "文件名", "大小", "类型");
    printf("----------------------------------------\n");

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        // 跳过 . 和 ..
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) {
            continue;
        }

        // 构建完整路径
        char fullpath[1024];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        // 获取文件信息
        struct stat st;
        if (stat(fullpath, &st) == 0) {
            const char *type;
            if (S_ISDIR(st.st_mode))      type = "目录";
            else if (S_ISREG(st.st_mode)) type = "文件";
            else if (S_ISLNK(st.st_mode)) type = "链接";
            else                           type = "其他";

            printf("%-30s %10ld %s\n", entry->d_name, (long)st.st_size, type);
        }
    }

    closedir(dir);
}

int main(int argc, char *argv[]) {
    list_directory(argc > 1 ? argv[1] : ".");
    return 0;
}
```

### 示例2：递归遍历目录树

```c
#include <stdio.h>
#include <string.h>
#include <dirent.h>
#include <sys/stat.h>

void walk_directory(const char *path, int depth) {
    DIR *dir = opendir(path);
    if (!dir) return;

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) {
            continue;
        }

        // 打印缩进和文件名
        for (int i = 0; i < depth; i++) printf("  ");
        printf("%s", entry->d_name);

        char fullpath[4096];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        struct stat st;
        if (stat(fullpath, &st) == 0 && S_ISDIR(st.st_mode)) {
            printf("/\n");
            walk_directory(fullpath, depth + 1);  // 递归进入子目录
        } else {
            printf(" (%ld bytes)\n", (long)st.st_size);
        }
    }

    closedir(dir);
}

int main(int argc, char *argv[]) {
    walk_directory(argc > 1 ? argv[1] : ".", 0);
    return 0;
}
```

### 示例3：统计目录中的文件数量和大小

```c
#include <stdio.h>
#include <string.h>
#include <dirent.h>
#include <sys/stat.h>

typedef struct {
    int files;
    int dirs;
    long long total_size;
} DirStats;

void collect_stats(const char *path, DirStats *stats) {
    DIR *dir = opendir(path);
    if (!dir) return;

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        if (strcmp(entry->d_name, ".") == 0 ||
            strcmp(entry->d_name, "..") == 0) continue;

        char fullpath[4096];
        snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);

        struct stat st;
        if (stat(fullpath, &st) == 0) {
            if (S_ISDIR(st.st_mode)) {
                stats->dirs++;
                collect_stats(fullpath, stats);  // 递归
            } else if (S_ISREG(st.st_mode)) {
                stats->files++;
                stats->total_size += st.st_size;
            }
        }
    }

    closedir(dir);
}

int main(int argc, char *argv[]) {
    DirStats stats = {0, 0, 0};
    collect_stats(argc > 1 ? argv[1] : ".", &stats);

    printf("文件数: %d\n", stats.files);
    printf("目录数: %d\n", stats.dirs);
    printf("总大小: %.2f MB\n", stats.total_size / (1024.0 * 1024.0));

    return 0;
}
```

### 示例4：rmdir - 删除空目录

```c
#include <stdio.h>
#include <unistd.h>

int main() {
    // rmdir 只能删除空目录
    if (rmdir("empty_dir") == 0) {
        printf("目录删除成功\n");
    } else {
        perror("删除目录失败");
        // ENOTEMPTY - 目录非空
        // ENOENT - 目录不存在
    }

    return 0;
}
```

## 四、Windows 目录操作

```c
#ifdef _WIN32
#include <windows.h>
#include <direct.h>

void list_directory_win(const char *path) {
    WIN32_FIND_DATA findData;
    char searchPath[MAX_PATH];
    snprintf(searchPath, MAX_PATH, "%s\\*", path);

    HANDLE hFind = FindFirstFile(searchPath, &findData);
    if (hFind == INVALID_HANDLE_VALUE) {
        printf("无法打开目录\n");
        return;
    }

    do {
        if (strcmp(findData.cFileName, ".") != 0 &&
            strcmp(findData.cFileName, "..") != 0) {
            printf("%s\n", findData.cFileName);
        }
    } while (FindNextFile(hFind, &findData));

    FindClose(hFind);
}
#endif
```

## 五、重点与注意事项

> **要点总结：**
>
> 1. 遍历目录时必须跳过 `.` 和 `..`，否则会无限递归
> 2. **`rmdir` 只能删除空目录**，删除非空目录需要先递归删除内容
> 3. **`readdir` 返回的指针指向静态内存**，下一次调用会覆盖，需要复制 `d_name`
> 4. `d_type` 在某些文件系统上为 `DT_UNKNOWN`，需要 `stat` 获取确切类型
> 5. 目录路径拼接要注意缓冲区溢出，使用 `snprintf`
> 6. Windows 使用 `_mkdir`、`_opendir` 等带下划线前缀的函数
> 7. POSIX API 不是标准C库的一部分，部分编译器（如MSVC）可能不直接支持
