# 22 - 管道与重定向

## 一、管道的概念

### 匿名管道

管道是单向的通信通道，将一个进程的输出连接到另一个进程的输入：

```
进程A (生产者) → [管道缓冲区] → 进程B (消费者)
    stdout   →   内核缓冲区  →    stdin
```

### popen / pclose

`popen` 是C标准库中使用管道的简便方式，它自动处理管道创建和进程启动。

```c
#include <stdio.h>
FILE *popen(const char *command, const char *mode);
int pclose(FILE *stream);
```

- `command`：要执行的 shell 命令
- `mode`：`"r"` 读取命令输出，`"w"` 向命令输入
- **返回值**：成功返回 `FILE*`，失败返回 `NULL`

## 二、popen 读取命令输出

### 读取外部命令的输出

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    // 执行命令并读取输出
    FILE *pipe = popen("ls -la", "r");
    if (!pipe) {
        perror("popen 失败");
        return 1;
    }

    char buffer[1024];
    while (fgets(buffer, sizeof(buffer), pipe) != NULL) {
        printf("输出: %s", buffer);
    }

    int status = pclose(pipe);
    printf("\n命令退出状态: %d\n", WEXITSTATUS(status));

    return 0;
}
```

### 获取命令输出的更灵活方式

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 读取命令输出到字符串
char *exec_command(const char *cmd) {
    FILE *pipe = popen(cmd, "r");
    if (!pipe) return NULL;

    size_t capacity = 4096;
    size_t length = 0;
    char *result = malloc(capacity);
    if (!result) { pclose(pipe); return NULL; }

    char buffer[1024];
    while (fgets(buffer, sizeof(buffer), pipe) != NULL) {
        size_t len = strlen(buffer);

        // 扩展缓冲区
        while (length + len + 1 > capacity) {
            capacity *= 2;
            char *new_result = realloc(result, capacity);
            if (!new_result) { free(result); pclose(pipe); return NULL; }
            result = new_result;
        }

        strcpy(result + length, buffer);
        length += len;
    }

    pclose(pipe);
    return result;
}

int main() {
    // 获取系统信息
    char *hostname = exec_command("hostname");
    char *date = exec_command("date");

    if (hostname) {
        printf("主机名: %s", hostname);
        free(hostname);
    }
    if (date) {
        printf("日期: %s", date);
        free(date);
    }

    // 统计当前目录文件数
    char *count = exec_command("ls | wc -l");
    if (count) {
        printf("文件数: %s", count);
        free(count);
    }

    return 0;
}
```

## 三、popen 向命令写入数据

### 通过管道向命令发送数据

```c
#include <stdio.h>

int main() {
    // 向 sort 命令写入数据，让其排序
    FILE *pipe = popen("sort", "w");
    if (!pipe) { perror("popen"); return 1; }

    // 写入未排序的数据
    fprintf(pipe, "banana\n");
    fprintf(pipe, "apple\n");
    fprintf(pipe, "cherry\n");
    fprintf(pipe, "date\n");
    fprintf(pipe, "elderberry\n");

    pclose(pipe);  // 关闭管道，触发 sort 输出

    return 0;
}
```

```
输出（按字母排序）:
apple
banana
cherry
date
elderberry
```

### 通过管道将数据发送给其他程序

```c
#include <stdio.h>

int main() {
    // 使用 grep 过滤内容
    FILE *pipe = popen("grep ERROR", "w");
    if (!pipe) { perror("popen"); return 1; }

    fprintf(pipe, "INFO: 程序启动\n");
    fprintf(pipe, "ERROR: 连接超时\n");
    fprintf(pipe, "DEBUG: 处理请求\n");
    fprintf(pipe, "ERROR: 数据库查询失败\n");
    fprintf(pipe, "INFO: 程序结束\n");

    pclose(pipe);

    return 0;
}
```

## 四、重定向

### 使用 freopen 重定向

```c
#include <stdio.h>

int main() {
    // 保存原始 stdout
    FILE *original_stdout = stdout;

    // 重定向 stdout 到文件
    stdout = freopen("output.txt", "w", stdout);
    if (!stdout) {
        perror("重定向失败");
        return 1;
    }

    // 这些输出会写入文件
    printf("这行写入了文件\n");
    printf("这行也写入了文件\n");

    // 关闭文件 stdout
    fclose(stdout);

    // 恢复 stdout（需要在Unix/Linux下）
    // stdout = freopen("/dev/tty", "w", stdout);

    // 注意：在实际程序中，重定向 stdout 后很难恢复
    // 推荐使用 dup2 进行更灵活的重定向

    return 0;
}
```

### 使用 dup2 进行重定向（POSIX）

```c
#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>

int redirect_stdout_to_file(const char *filename) {
    int fd = open(filename, O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) return -1;

    fflush(stdout);           // 刷新当前stdout
    int saved = dup(STDOUT_FILENO);  // 备份原始stdout
    dup2(fd, STDOUT_FILENO);  // 重定向stdout到文件
    close(fd);

    return saved;
}

void restore_stdout(int saved_fd) {
    fflush(stdout);
    dup2(saved_fd, STDOUT_FILENO);  // 恢复原始stdout
    close(saved_fd);
}

int main() {
    printf("这行在终端显示\n");

    int saved = redirect_stdout_to_file("log.txt");
    printf("这行写入文件\n");
    printf("这行也写入文件\n");
    restore_stdout(saved);

    printf("这行又在终端显示了\n");

    return 0;
}
```

## 五、管道通信的实际应用

### 日志过滤工具

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int filter_logs(const char *logfile, const char *keyword) {
    char cmd[512];
    snprintf(cmd, sizeof(cmd), "grep '%s' '%s' | wc -l", keyword, logfile);

    FILE *pipe = popen(cmd, "r");
    if (!pipe) { perror("popen"); return -1; }

    char result[64];
    if (fgets(result, sizeof(result), pipe)) {
        printf("包含 '%s' 的日志行数: %s", keyword, result);
    }

    return pclose(pipe);
}

int main() {
    // 创建测试日志
    FILE *fp = fopen("app.log", "w");
    fprintf(fp, "INFO: 程序启动\n");
    fprintf(fp, "ERROR: 连接失败\n");
    fprintf(fp, "WARN: 超时\n");
    fprintf(fp, "ERROR: 数据库错误\n");
    fprintf(fp, "INFO: 处理完成\n");
    fclose(fp);

    filter_logs("app.log", "ERROR");
    filter_logs("app.log", "INFO");

    remove("app.log");
    return 0;
}
```

## 六、重点与注意事项

> **要点总结：**
>
> 1. **`popen("cmd", "r")` 读取命令输出**，`popen("cmd", "w")` 向命令写入
> 2. **必须调用 `pclose`** 而非 `fclose`——它会等待子进程结束
> 3. `popen` 通过 shell 执行命令，存在**命令注入风险**——不要将用户输入直接拼接到命令中
> 4. 管道是**单向的**——不能同时读写同一个管道
> 5. `popen` 创建的子进程继承父进程的环境和权限
> 6. 使用 `WEXITSTATUS(status)` 获取命令的退出码
> 7. `freopen` 重定向后难以恢复，`dup2` 更灵活
