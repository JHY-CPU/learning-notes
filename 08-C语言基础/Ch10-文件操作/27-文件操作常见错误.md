# 27 - 文件操作常见错误

## 一、忘记检查返回值

### 错误1：不检查 fopen 返回值

```c
// ❌ 错误：不检查 fopen 是否成功
FILE *fp = fopen("data.txt", "r");
char buffer[100];
fgets(buffer, 100, fp);  // 如果 fp 为 NULL，段错误！
fclose(fp);

// ✅ 正确：始终检查返回值
FILE *fp = fopen("data.txt", "r");
if (fp == NULL) {
    perror("打开文件失败");
    return 1;
}
char buffer[100];
fgets(buffer, 100, fp);
fclose(fp);
```

### 错误2：不检查 fread/fwrite 返回值

```c
// ❌ 错误：假设所有数据都成功写入
int data[1000];
fwrite(data, sizeof(int), 1000, fp);  // 可能部分写入！
fclose(fp);

// ✅ 正确：检查实际写入量
int data[1000];
size_t written = fwrite(data, sizeof(int), 1000, fp);
if (written != 1000) {
    fprintf(stderr, "写入失败：期望1000个，实际%zu个\n", written);
}
fclose(fp);
```

## 二、文件未关闭

### 错误3：忘记关闭文件

```c
// ❌ 错误：函数中打开文件但未关闭
void process_data(const char *filename) {
    FILE *fp = fopen(filename, "r");
    // ... 处理 ...
    return;  // 忘记 fclose(fp)！
    // 后果：
    // 1. 文件描述符泄漏
    // 2. 缓冲区数据丢失
    // 3. 其他进程可能无法访问该文件
}

// ✅ 正确：统一出口关闭文件
void process_data(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return;

    // ... 处理 ...

    fclose(fp);  // 确保关闭
}
```

### 错误4：异常路径未关闭

```c
// ❌ 错误：某些分支忘记关闭
int read_config(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return -1;

    char line[256];
    while (fgets(line, sizeof(line), fp)) {
        if (line[0] == '#') continue;
        if (strstr(line, "stop")) {
            return 0;  // 没有 fclose！
        }
    }

    fclose(fp);
    return 0;
}

// ✅ 正确：使用 goto 统一清理
int read_config(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return -1;

    int result = 0;
    char line[256];

    while (fgets(line, sizeof(line), fp)) {
        if (line[0] == '#') continue;
        if (strstr(line, "stop")) {
            result = 0;
            goto cleanup;  // 跳到清理代码
        }
    }

cleanup:
    fclose(fp);
    return result;
}
```

## 三、路径错误

### 错误5：路径拼接缓冲区溢出

```c
// ❌ 错误：缓冲区可能溢出
char path[64];
sprintf(path, "%s/%s", directory, filename);  // 可能溢出！

// ✅ 正确：使用 snprintf
char path[PATH_MAX];
snprintf(path, sizeof(path), "%s/%s", directory, filename);
```

### 错误6：相对路径的误解

```c
// ❌ 错误：假设当前目录
FILE *fp = fopen("config.txt", "r");
// 当前目录取决于程序启动位置，不是源码位置！

// ✅ 正确：使用绝对路径或相对于可执行文件的路径
// 或者在程序启动时设置工作目录
```

### 错误7：Windows路径中的反斜杠

```c
// ❌ 错误：反斜杠被解释为转义字符
FILE *fp = fopen("C:\new\test.txt", "r");
// \n = 换行符, \t = 制表符！

// ✅ 正确：转义反斜杠或使用正斜杠
FILE *fp1 = fopen("C:\\new\\test.txt", "r");
FILE *fp2 = fopen("C:/new/test.txt", "r");  // 两种写法等价
```

## 四、缓冲区与刷新问题

### 错误8：崩溃前未刷新缓冲区

```c
// ❌ 错误：数据可能丢失
FILE *fp = fopen("log.txt", "a");
fprintf(fp, "重要日志: 关键操作\n");
// 程序崩溃！缓冲区未刷新，数据丢失

// ✅ 正确：关键数据立即刷新
FILE *fp = fopen("log.txt", "a");
fprintf(fp, "重要日志: 关键操作\n");
fflush(fp);  // 强制刷新
// 然后才执行可能崩溃的操作
```

### 错误9：读写切换未刷新

```c
// ❌ 错误：读写切换时未刷新
FILE *fp = fopen("data.txt", "r+");
fprintf(fp, "写入数据");
char buf[100];
fgets(buf, sizeof(buf), fp);  // 可能读到错误内容！

// ✅ 正确：读写切换时调用 fseek 或 fflush
FILE *fp = fopen("data.txt", "r+");
fprintf(fp, "写入数据");
fflush(fp);          // 刷新输出缓冲区
fseek(fp, 0, SEEK_SET);  // 重置位置
fgets(buf, sizeof(buf), fp);  // 现在可以正确读取
```

## 五、文件模式错误

### 错误10：用文本模式处理二进制文件

```c
// ❌ 错误：文本模式处理图片/音频
FILE *fp = fopen("image.jpg", "r");  // 文本模式！
// 在Windows上，\n ↔ \r\n 转换会损坏二进制数据

// ✅ 正确：二进制文件使用 "rb"
FILE *fp = fopen("image.jpg", "rb");
```

### 错误11：用 "w" 模式意外清空文件

```c
// ❌ 错误："w" 会清空已有文件！
FILE *fp = fopen("important_data.txt", "w");
// 文件内容已被清空！无法恢复！

// ✅ 正确：需要保留内容时使用 "r+" 或 "a"
FILE *fp = fopen("important_data.txt", "r+");  // 保留内容
// 或者先备份
```

## 六、字符串处理错误

### 错误12：使用 gets 函数

```c
// ❌ 错误：gets 无缓冲区大小限制，已从C11中移除
char name[10];
gets(name);  // 用户输入超过10个字符 → 缓冲区溢出！

// ✅ 正确：使用 fgets
char name[10];
fgets(name, sizeof(name), stdin);
// 去除末尾换行符
name[strcspn(name, "\n")] = '\0';
```

### 错误13：fgets 未去除换行符

```c
// ⚠️ 注意：fgets 保留换行符
char line[100];
fgets(line, sizeof(line), fp);
// 如果文件中是 "hello\n"
// line = "hello\n\0"

// 处理方式
size_t len = strlen(line);
if (len > 0 && line[len - 1] == '\n') {
    line[len - 1] = '\0';
}
```

## 七、结构体序列化错误

### 错误14：包含指针的结构体直接写入

```c
// ❌ 错误：指针写入文件没有意义
typedef struct {
    char *name;   // 指针！
    int age;
} Person;

Person p = {"张三", 25};
fwrite(&p, sizeof(Person), 1, fp);  // 写入的是指针值，不是字符串！

// ✅ 正确：使用固定大小数组或单独序列化字符串
typedef struct {
    char name[32];  // 固定大小数组
    int age;
} Person;

// 或者分别写入
fwrite(&p.age, sizeof(int), 1, fp);
int name_len = strlen(p.name);
fwrite(&name_len, sizeof(int), 1, fp);
fwrite(p.name, 1, name_len, fp);
```

## 八、错误排查技巧

### 常见错误现象与原因

```
现象: Segmentation Fault
原因: 使用了 NULL 文件指针
排查: 检查 fopen 的返回值

现象: 文件内容为空
原因: 使用 "w" 模式打开了已有文件，或忘记写入数据
排查: 检查 fopen 模式，确认写入代码被执行

现象: 文件内容不完整
原因: 缓冲区未刷新，程序异常退出
排查: 添加 fflush 调用，检查 fclose 是否被调用

现象: 乱码
原因: 编码不匹配（UTF-8 vs GBK）
排查: 确认文件编码和程序预期编码一致

现象: 多出 ^M (Windows)
原因: 在Linux上读取了Windows格式的文本文件
排查: 使用二进制模式读取，手动处理 \r\n

现象: 数据损坏（二进制文件）
原因: 使用文本模式处理二进制文件
排查: 改用 "rb"/"wb" 模式
```

## 九、重点与注意事项

> **要点总结：**
>
> 1. **所有文件操作函数的返回值都必须检查**
> 2. **fopen 后必须 fclose**，使用 goto 统一错误处理
> 3. **"w" 模式会清空文件**——这是最常见的数据丢失原因
> 4. 二进制文件必须使用 `"rb"`/`"wb"` 模式
> 5. `fgets` 保留换行符，`fputs` 不添加换行符
> 6. 不要将含指针的结构体直接写入文件
> 7. 关键操作后使用 `fflush` 确保数据持久化
