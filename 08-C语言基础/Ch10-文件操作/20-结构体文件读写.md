# 20 - 结构体文件读写

## 一、结构体序列化基础

### 什么是序列化

序列化是将内存中的数据结构转换为可存储或传输的格式：

```
内存中的结构体 → 序列化 → 文件中的字节流
文件中的字节流 → 反序列化 → 内存中的结构体
```

### 直接写入结构体

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char name[32];
    double score;
    int age;
} Student;

int main() {
    Student s = {1001, "张三", 85.5, 20};

    // 直接写入整个结构体
    FILE *fp = fopen("student.dat", "wb");
    if (!fp) { perror("写入失败"); return 1; }

    fwrite(&s, sizeof(Student), 1, fp);
    fclose(fp);

    // 直接读取整个结构体
    Student loaded;
    fp = fopen("student.dat", "rb");
    if (!fp) { perror("读取失败"); return 1; }

    fread(&loaded, sizeof(Student), 1, fp);
    fclose(fp);

    printf("ID: %d, 姓名: %s, 成绩: %.1f, 年龄: %d\n",
           loaded.id, loaded.name, loaded.score, loaded.age);

    return 0;
}
```

## 二、结构体数组的读写

### 写入结构体数组

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char name[32];
    double score;
} Student;

void save_students(const char *filename, Student students[], int count) {
    FILE *fp = fopen(filename, "wb");
    if (!fp) { perror("保存失败"); return; }

    // 先写入记录数，再写入数据
    fwrite(&count, sizeof(int), 1, fp);
    fwrite(students, sizeof(Student), count, fp);

    fclose(fp);
    printf("已保存 %d 条记录\n", count);
}

int load_students(const char *filename, Student students[], int max_count) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) { perror("加载失败"); return 0; }

    int count;
    fread(&count, sizeof(int), 1, fp);

    if (count > max_count) {
        printf("警告：文件中有 %d 条记录，缓冲区只能容纳 %d 条\n",
               count, max_count);
        count = max_count;
    }

    fread(students, sizeof(Student), count, fp);
    fclose(fp);
    return count;
}

int main() {
    Student students[] = {
        {1001, "张三", 85.5},
        {1002, "李四", 92.0},
        {1003, "王五", 78.3}
    };

    save_students("students.dat", students, 3);

    Student loaded[10];
    int count = load_students("students.dat", loaded, 10);

    for (int i = 0; i < count; i++) {
        printf("ID:%d 名:%s 成绩:%.1f\n",
               loaded[i].id, loaded[i].name, loaded[i].score);
    }

    return 0;
}
```

## 三、记录文件（Record File）

### 增删改查操作

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char name[32];
    double score;
    int active;  // 逻辑删除标记
} Student;

#define REC_SIZE sizeof(Student)

// 添加记录
int add_record(const char *filename, const Student *s) {
    FILE *fp = fopen(filename, "ab");
    if (!fp) return -1;
    fwrite(s, REC_SIZE, 1, fp);
    fclose(fp);
    return 0;
}

// 按索引读取记录
int get_record(const char *filename, int index, Student *s) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return -1;
    fseek(fp, index * REC_SIZE, SEEK_SET);
    int ret = fread(s, REC_SIZE, 1, fp) == 1 ? 0 : -1;
    fclose(fp);
    return ret;
}

// 按索引更新记录
int update_record(const char *filename, int index, const Student *s) {
    FILE *fp = fopen(filename, "r+b");
    if (!fp) return -1;
    fseek(fp, index * REC_SIZE, SEEK_SET);
    fwrite(s, REC_SIZE, 1, fp);
    fclose(fp);
    return 0;
}

// 逻辑删除记录
int delete_record(const char *filename, int index) {
    Student s;
    if (get_record(filename, index, &s) != 0) return -1;
    s.active = 0;
    return update_record(filename, index, &s);
}

// 获取记录数
int count_records(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return 0;
    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fclose(fp);
    return (int)(size / REC_SIZE);
}

// 按ID查找记录
int find_by_id(const char *filename, int id, Student *s) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return -1;

    int index = 0;
    while (fread(s, REC_SIZE, 1, fp) == 1) {
        if (s->active && s->id == id) {
            fclose(fp);
            return index;
        }
        index++;
    }

    fclose(fp);
    return -1;  // 未找到
}

// 列出所有有效记录
void list_records(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) { printf("无数据\n"); return; }

    Student s;
    int index = 0;
    printf("%-4s %-6s %-10s %-8s\n", "序号", "ID", "姓名", "成绩");
    printf("-------------------------------\n");

    while (fread(&s, REC_SIZE, 1, fp) == 1) {
        if (s.active) {
            printf("%-4d %-6d %-10s %-8.1f\n",
                   index, s.id, s.name, s.score);
        }
        index++;
    }

    fclose(fp);
}
```

### 使用示例

```c
int main() {
    const char *db = "school.dat";

    // 添加
    Student s1 = {1001, "张三", 85.5, 1};
    Student s2 = {1002, "李四", 92.0, 1};
    Student s3 = {1003, "王五", 78.3, 1};
    add_record(db, &s1);
    add_record(db, &s2);
    add_record(db, &s3);

    printf("=== 所有记录 ===\n");
    list_records(db);

    // 修改
    s2.score = 98.5;
    update_record(db, 1, &s2);

    // 删除第3条
    delete_record(db, 2);

    printf("\n=== 修改后 ===\n");
    list_records(db);

    // 按ID查找
    Student found;
    int idx = find_by_id(db, 1001, &found);
    if (idx >= 0) {
        printf("\n找到: %s (成绩%.1f)\n", found.name, found.score);
    }

    printf("\n总记录数: %d\n", count_records(db));

    return 0;
}
```

## 四、文本格式序列化

### 将结构体转换为文本格式存储

```c
#include <stdio.h>

typedef struct {
    int id;
    char name[32];
    double score;
} Student;

// 序列化为文本
void serialize_student(FILE *fp, const Student *s) {
    fprintf(fp, "%d|%s|%.2f\n", s->id, s->name, s->score);
}

// 从文本反序列化
int deserialize_student(const char *line, Student *s) {
    return sscanf(line, "%d|%31[^|]|%lf", &s->id, s->name, &s->score) == 3;
}

// 保存为文本文件
void save_as_text(const char *filename, Student students[], int count) {
    FILE *fp = fopen(filename, "w");
    if (!fp) return;

    fprintf(fp, "# ID|姓名|成绩\n");
    for (int i = 0; i < count; i++) {
        serialize_student(fp, &students[i]);
    }

    fclose(fp);
}

// 从文本文件加载
int load_from_text(const char *filename, Student students[], int max) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return 0;

    char line[256];
    int count = 0;

    while (fgets(line, sizeof(line), fp) && count < max) {
        if (line[0] == '#' || line[0] == '\n') continue;
        if (deserialize_student(line, &students[count])) {
            count++;
        }
    }

    fclose(fp);
    return count;
}
```

## 五、重点与注意事项

> **要点总结：**
>
> 1. **直接 fwrite 结构体存在跨平台风险**——字节序、对齐填充可能不同
> 2. 使用 `#pragma pack(1)` 或 `__attribute__((packed))` 禁用填充确保一致性
> 3. 文本格式更安全但效率更低，二进制格式更快但不可读
> 4. 先写入记录数，再写入数据，读取时才知道要读多少
> 5. 逻辑删除（标记位）比物理删除更安全，可以恢复数据
> 6. 随机访问需要固定大小的记录——使用结构体时天然满足
> 7. 包含指针的结构体不能直接写入文件——指针值没有意义
