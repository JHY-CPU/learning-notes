# 03 - 文本文件读写

## 一、逐字符读写函数

### fgetc - 读取单个字符

```c
#include <stdio.h>
int fgetc(FILE *stream);
```

- 从流中读取下一个字符
- 返回值：成功返回字符的 `unsigned char` 值（转为int），到达文件末尾或出错返回 `EOF`
- `EOF` 在 `<stdio.h>` 中通常定义为 `-1`

### fputc - 写入单个字符

```c
#include <stdio.h>
int fputc(int c, FILE *stream);
```

- 将字符 `c` 写入流
- 返回值：成功返回写入的字符，失败返回 `EOF`

### 字符读写示例

```c
#include <stdio.h>

int main() {
    FILE *src = fopen("source.txt", "r");
    FILE *dst = fopen("dest.txt", "w");

    if (src == NULL || dst == NULL) {
        perror("文件打开失败");
        return 1;
    }

    int ch;
    while ((ch = fgetc(src)) != EOF) {
        fputc(ch, dst);
    }

    fclose(src);
    fclose(dst);
    printf("文件复制完成！\n");
    return 0;
}
```

> **重要**：`fgetc` 的返回值必须用 `int` 类型接收，不能用 `char`！因为 `char` 无法区分值为 `0xFF` 的合法字符和 `EOF`（-1）。

### getc 和 putc 的区别

```c
// getc/putc 与 fgetc/fputc 功能相同，但实现方式不同：

int ch = fgetc(fp);   // 一定是函数调用
int ch = getc(fp);    // 可能是宏（效率略高），没有函数调用开销

fputc('A', fp);       // 一定是函数调用
putc('A', fp);        // 可能是宏

// 建议：优先使用 fgetc/fputc（更安全），性能差异在现代编译器中微乎其微
```

## 二、逐行读写函数

### fgets - 读取一行

```c
#include <stdio.h>
char *fgets(char *str, int n, FILE *stream);
```

- 从流中读取最多 `n-1` 个字符到 `str`，自动添加 `'\0'`
- 遇到换行符 `\n` 或到达文件末尾时停止
- **成功返回 `str`，失败或立即到达EOF返回 `NULL`**
- 如果读到了 `\n`，会保留在字符串中

### fgets 读取行为详解

```c
// 假设文件内容为：Hello World\n

char buf[20];
fgets(buf, 20, fp);
// buf 内容: "Hello World\n\0"
// 读取了 12 个字符（11个字符 + \n），第12个位置放 \0

// 如果缓冲区不够大：
char buf[8];
fgets(buf, 8, fp);
// buf 内容: "Hello W\0"
// 读取了 7 个字符（n-1=7），第8个位置放 \0
// 后续调用 fgets 会继续读取余下内容: "orld\n"
```

### fgets 行读取示例

```c
#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp = fopen("data.txt", "r");
    if (fp == NULL) {
        perror("打开文件失败");
        return 1;
    }

    char line[1024];
    int line_num = 0;

    while (fgets(line, sizeof(line), fp) != NULL) {
        line_num++;

        // 去除末尾的换行符（可选）
        size_t len = strlen(line);
        if (len > 0 && line[len - 1] == '\n') {
            line[len - 1] = '\0';
        }

        printf("第%d行: %s\n", line_num, line);
    }

    fclose(fp);
    return 0;
}
```

### fputs - 写入字符串

```c
#include <stdio.h>
int fputs(const char *str, FILE *stream);
```

- 将字符串 `str` 写入流（**不自动添加换行符**）
- 成功返回非负值，失败返回 `EOF`
- 不写入字符串末尾的 `'\0'`

### fputs 与 puts 的区别

```c
puts("Hello");       // 输出到 stdout，自动追加 \n
fputs("Hello", fp);  // 输出到文件，不追加 \n

// puts("Hello") 等价于:
fputs("Hello", stdout);
fputc('\n', stdout);
```

## 三、综合示例

### 示例1：统计文件的行数、字符数、单词数

```c
#include <stdio.h>
#include <ctype.h>

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "用法: %s <文件名>\n", argv[0]);
        return 1;
    }

    FILE *fp = fopen(argv[1], "r");
    if (fp == NULL) {
        perror("打开文件失败");
        return 1;
    }

    int lines = 0, chars = 0, words = 0;
    int in_word = 0;
    int ch;

    while ((ch = fgetc(fp)) != EOF) {
        chars++;

        if (ch == '\n') {
            lines++;
        }

        if (isspace(ch)) {
            in_word = 0;
        } else if (!in_word) {
            in_word = 1;
            words++;
        }
    }

    // 如果文件不以换行符结尾，最后一行也算
    if (chars > 0 && ch != '\n') {
        lines++;
    }

    printf("行数: %d, 单词数: %d, 字符数: %d\n", lines, words, chars);

    fclose(fp);
    return 0;
}
```

### 示例2：文件内容加密（简单替换）

```c
#include <stdio.h>

int main() {
    FILE *src = fopen("plaintext.txt", "r");
    FILE *dst = fopen("encrypted.txt", "w");

    if (src == NULL || dst == NULL) {
        perror("文件操作失败");
        return 1;
    }

    int ch;
    while ((ch = fgetc(src)) != EOF) {
        // 简单的凯撒加密：每个字符偏移3位
        fputc(ch + 3, dst);
    }

    fclose(src);
    fclose(dst);
    printf("加密完成！\n");
    return 0;
}
```

### 示例3：去除文本文件中的空行

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

// 判断一行是否全是空白字符
int is_blank_line(const char *line) {
    while (*line) {
        if (!isspace((unsigned char)*line)) {
            return 0;
        }
        line++;
    }
    return 1;
}

int main() {
    FILE *src = fopen("input.txt", "r");
    FILE *dst = fopen("output.txt", "w");

    if (src == NULL || dst == NULL) {
        perror("文件操作失败");
        return 1;
    }

    char line[4096];
    while (fgets(line, sizeof(line), src) != NULL) {
        if (!is_blank_line(line)) {
            fputs(line, dst);
        }
    }

    fclose(src);
    fclose(dst);
    printf("空行已移除！\n");
    return 0;
}
```

## 四、字符/字符串读写函数对比

| 函数 | 功能 | 添加'\n' | 添加'\0' | 目标 |
|------|------|---------|---------|------|
| `fputc` | 写一个字符 | 否 | 否 | 流 |
| `fgetc` | 读一个字符 | — | — | 流 |
| `fputs` | 写字符串 | **否** | 否 | 流 |
| `fgets` | 读字符串 | 保留 | **是** | 流 |
| `putc` | 写一个字符(宏) | 否 | 否 | 流 |
| `getc` | 读一个字符(宏) | — | — | 流 |
| `puts` | 写字符串 | **是** | 否 | stdout |
| `gets` | 读字符串(已废弃!) | — | **是** | stdin |

> **警告**：`gets()` 函数极其危险（无缓冲区大小限制），在C11中已被移除。永远使用 `fgets()` 替代！

## 五、重点与注意事项

> **要点总结：**
>
> 1. `fgetc` 返回 `int` 而非 `char`，必须用 `int` 接收
> 2. `fgets` 保留换行符，`fputs` 不自动添加换行符（需手动处理）
> 3. `fgets` 最多读取 `n-1` 个字符，总是保证字符串以 `'\0'` 结尾
> 4. 永远不要使用 `gets()`，它无法限制读取长度，是缓冲区溢出的经典来源
> 5. 循环读取时要检查 `fgetc` 的返回值是否为 `EOF`，`fgets` 的返回值是否为 `NULL`
> 6. 对于大文件，逐字符读取效率较低，应使用缓冲读取（fgets 或 fread）
