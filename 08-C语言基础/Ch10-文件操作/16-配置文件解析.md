# 16 - 配置文件解析

## 一、配置文件格式

### INI格式

INI是最简单的配置文件格式之一：

```ini
# 应用程序配置
[database]
host = localhost
port = 3306
name = myapp_db
user = root
password = secret123

[server]
host = 0.0.0.0
port = 8080
max_connections = 100
timeout = 30.5
debug = true

[logging]
level = INFO
file = /var/log/myapp.log
max_size = 10485760
```

### INI格式规则

```
- 注释行以 # 或 ; 开头
- 节(section)用 [section_name] 标识
- 键值对格式: key = value 或 key: value
- 值可以包含空格，但首尾空格会被去除
- 节名和键名不区分大小写（通常）
```

## 二、INI配置文件解析器

### 数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define MAX_LINE_LEN    1024
#define MAX_KEY_LEN     128
#define MAX_VALUE_LEN   512
#define MAX_SECTIONS    64
#define MAX_ENTRIES     128

typedef struct {
    char key[MAX_KEY_LEN];
    char value[MAX_VALUE_LEN];
} ConfigEntry;

typedef struct {
    char name[MAX_KEY_LEN];
    ConfigEntry entries[MAX_ENTRIES];
    int entry_count;
} ConfigSection;

typedef struct {
    ConfigSection sections[MAX_SECTIONS];
    int section_count;
} Config;
```

### 解析函数

```c
// 去除字符串首尾空白
char *trim(char *str) {
    while (isspace((unsigned char)*str)) str++;
    if (*str == '\0') return str;

    char *end = str + strlen(str) - 1;
    while (end > str && isspace((unsigned char)*end)) end--;
    *(end + 1) = '\0';
    return str;
}

// 解析配置文件
int config_parse(Config *cfg, const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return -1;

    cfg->section_count = 0;
    ConfigSection *current = NULL;
    char line[MAX_LINE_LEN];

    while (fgets(line, sizeof(line), fp)) {
        char *p = trim(line);

        // 跳过空行和注释
        if (*p == '\0' || *p == '#' || *p == ';') continue;

        // 节 [section]
        if (*p == '[') {
            char *end = strchr(p, ']');
            if (end) {
                *end = '\0';
                current = &cfg->sections[cfg->section_count++];
                strncpy(current->name, p + 1, MAX_KEY_LEN - 1);
                current->entry_count = 0;
            }
            continue;
        }

        // 键值对 key = value
        char *eq = strchr(p, '=');
        if (!eq) eq = strchr(p, ':');

        if (eq && current && current->entry_count < MAX_ENTRIES) {
            *eq = '\0';
            char *key = trim(p);
            char *value = trim(eq + 1);

            ConfigEntry *entry = &current->entries[current->entry_count++];
            strncpy(entry->key, key, MAX_KEY_LEN - 1);
            strncpy(entry->value, value, MAX_VALUE_LEN - 1);
        }
    }

    fclose(fp);
    return 0;
}
```

### 查询函数

```c
// 查找节
ConfigSection *config_find_section(Config *cfg, const char *name) {
    for (int i = 0; i < cfg->section_count; i++) {
        if (strcasecmp(cfg->sections[i].name, name) == 0) {
            return &cfg->sections[i];
        }
    }
    return NULL;
}

// 获取字符串值
const char *config_get_string(Config *cfg, const char *section,
                               const char *key, const char *default_val) {
    ConfigSection *sec = config_find_section(cfg, section);
    if (!sec) return default_val;

    for (int i = 0; i < sec->entry_count; i++) {
        if (strcasecmp(sec->entries[i].key, key) == 0) {
            return sec->entries[i].value;
        }
    }
    return default_val;
}

// 获取整数值
int config_get_int(Config *cfg, const char *section,
                    const char *key, int default_val) {
    const char *val = config_get_string(cfg, section, key, NULL);
    return val ? atoi(val) : default_val;
}

// 获取浮点值
double config_get_double(Config *cfg, const char *section,
                          const char *key, double default_val) {
    const char *val = config_get_string(cfg, section, key, NULL);
    return val ? atof(val) : default_val;
}

// 获取布尔值
int config_get_bool(Config *cfg, const char *section,
                     const char *key, int default_val) {
    const char *val = config_get_string(cfg, section, key, NULL);
    if (!val) return default_val;

    if (strcasecmp(val, "true") == 0 || strcmp(val, "1") == 0 ||
        strcasecmp(val, "yes") == 0) return 1;
    if (strcasecmp(val, "false") == 0 || strcmp(val, "0") == 0 ||
        strcasecmp(val, "no") == 0) return 0;

    return default_val;
}
```

### 完整使用示例

```c
// 创建示例配置文件
void create_sample_config(const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (!fp) return;

    fprintf(fp, "# 应用程序配置\n");
    fprintf(fp, "[database]\n");
    fprintf(fp, "host = localhost\n");
    fprintf(fp, "port = 3306\n");
    fprintf(fp, "name = myapp_db\n\n");
    fprintf(fp, "[server]\n");
    fprintf(fp, "host = 0.0.0.0\n");
    fprintf(fp, "port = 8080\n");
    fprintf(fp, "timeout = 30.5\n");
    fprintf(fp, "debug = true\n");

    fclose(fp);
}

int main() {
    create_sample_config("app.conf");

    Config cfg;
    if (config_parse(&cfg, "app.conf") != 0) {
        perror("解析配置失败");
        return 1;
    }

    printf("数据库配置:\n");
    printf("  主机: %s\n", config_get_string(&cfg, "database", "host", "localhost"));
    printf("  端口: %d\n", config_get_int(&cfg, "database", "port", 3306));
    printf("  名称: %s\n", config_get_string(&cfg, "database", "name", ""));

    printf("\n服务器配置:\n");
    printf("  主机: %s\n", config_get_string(&cfg, "server", "host", "0.0.0.0"));
    printf("  端口: %d\n", config_get_int(&cfg, "server", "port", 80));
    printf("  超时: %.1f\n", config_get_double(&cfg, "server", "timeout", 10.0));
    printf("  调试: %s\n", config_get_bool(&cfg, "server", "debug", 0) ? "开启" : "关闭");

    return 0;
}
```

## 三、写入配置文件

```c
// 写入整个配置
int config_save(Config *cfg, const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (!fp) return -1;

    for (int i = 0; i < cfg->section_count; i++) {
        ConfigSection *sec = &cfg->sections[i];
        fprintf(fp, "[%s]\n", sec->name);
        for (int j = 0; j < sec->entry_count; j++) {
            fprintf(fp, "%s = %s\n",
                    sec->entries[j].key,
                    sec->entries[j].value);
        }
        fprintf(fp, "\n");
    }

    fclose(fp);
    return 0;
}

// 设置值
void config_set(Config *cfg, const char *section,
                const char *key, const char *value) {
    ConfigSection *sec = config_find_section(cfg, section);
    if (!sec) {
        sec = &cfg->sections[cfg->section_count++];
        strncpy(sec->name, section, MAX_KEY_LEN - 1);
        sec->entry_count = 0;
    }

    // 查找已有的键
    for (int i = 0; i < sec->entry_count; i++) {
        if (strcasecmp(sec->entries[i].key, key) == 0) {
            strncpy(sec->entries[i].value, value, MAX_VALUE_LEN - 1);
            return;
        }
    }

    // 添加新键
    ConfigEntry *entry = &sec->entries[sec->entry_count++];
    strncpy(entry->key, key, MAX_KEY_LEN - 1);
    strncpy(entry->value, value, MAX_VALUE_LEN - 1);
}
```

## 四、键值对格式

### 简单键值对（无节）

```c
// 读取简单的 key=value 文件
int read_key_value(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) return -1;

    char line[MAX_LINE_LEN];
    while (fgets(line, sizeof(line), fp)) {
        char *p = trim(line);
        if (*p == '#' || *p == '\0') continue;

        char *eq = strchr(p, '=');
        if (!eq) continue;

        *eq = '\0';
        char *key = trim(p);
        char *value = trim(eq + 1);

        printf("配置: %s = %s\n", key, value);
    }

    fclose(fp);
    return 0;
}
```

## 五、重点与注意事项

> **要点总结：**
>
> 1. 配置文件解析要注意处理注释、空行、首尾空白
> 2. **键名不区分大小写**，使用 `strcasecmp` 比较
> 3. 值中的特殊字符（如等号、分号）可能导致解析错误
> 4. 提供默认值是配置系统的重要特性，确保程序在配置缺失时仍能运行
> 5. 生产环境可使用成熟库（如 `libconfig`、`inih`）
> 6. 配置文件应有良好的文档说明每个选项的含义和取值范围
