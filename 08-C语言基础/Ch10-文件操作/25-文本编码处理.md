# 25 - 文本编码处理

## 一、字符编码基础

### ASCII

```
ASCII: 7位编码，128个字符
范围: 0x00 - 0x7F
包含: 英文字母、数字、标点、控制字符

'A' = 0x41 = 65
'a' = 0x61 = 97
'0' = 0x30 = 48
'\n' = 0x0A = 10
```

### Unicode

```
Unicode: 统一字符集，为每个字符分配唯一的码点
范围: U+0000 - U+10FFFF

中文 "你": U+4F60
中文 "好": U+597D
表情  "😀": U+1F600
```

### UTF-8 编码规则

```
码点范围              | 字节数 | 编码格式
U+0000 - U+007F      |   1    | 0xxxxxxx
U+0080 - U+07FF      |   2    | 110xxxxx 10xxxxxx
U+0800 - U+FFFF      |   3    | 1110xxxx 10xxxxxx 10xxxxxx
U+10000 - U+10FFFF   |   4    | 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx

UTF-8 特点：
- 完全兼容 ASCII（ASCII字符=1字节）
- 自同步（可以从任意位置开始解码）
- 无字节序问题
- 中文字符通常占3字节
```

### "你好" 的UTF-8编码

```
"你": U+4F60 → 1110-0100 10-111101 10-100000 → E4 BD A0
"好": U+597D → 1110-0101 10-010111 10-111101 → E5 A5 BD

"你好" 的UTF-8字节序列: E4 BD A0 E5 A5 BD (6字节)
```

## 二、UTF-8 文件读写

### 读取UTF-8文件

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// UTF-8字符长度查询
int utf8_char_len(unsigned char byte) {
    if ((byte & 0x80) == 0x00) return 1;  // 0xxxxxxx
    if ((byte & 0xE0) == 0xC0) return 2;  // 110xxxxx
    if ((byte & 0xF0) == 0xE0) return 3;  // 1110xxxx
    if ((byte & 0xF8) == 0xF0) return 4;  // 11110xxx
    return 1;  // 非法字节，当作单字节处理
}

// 逐字符读取UTF-8文本
void read_utf8_file(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) { perror("打开失败"); return; }

    unsigned char buf[8];
    int ch;
    long char_count = 0;
    long byte_count = 0;

    while ((ch = fgetc(fp)) != EOF) {
        int len = utf8_char_len((unsigned char)ch);
        buf[0] = ch;

        for (int i = 1; i < len; i++) {
            int next = fgetc(fp);
            if (next == EOF) break;
            buf[i] = next;
        }
        buf[len] = '\0';

        char_count++;
        byte_count += len;

        printf("字符 #%ld: %s (%d 字节)\n", char_count, buf, len);
    }

    printf("\n总计: %ld 个字符, %ld 字节\n", char_count, byte_count);
    fclose(fp);
}

int main() {
    // 创建UTF-8测试文件
    FILE *fp = fopen("utf8_test.txt", "wb");
    // 写入UTF-8 BOM（可选）
    // unsigned char bom[] = {0xEF, 0xBB, 0xBF};
    // fwrite(bom, 1, 3, fp);
    fprintf(fp, "Hello你好World世界");
    fclose(fp);

    read_utf8_file("utf8_test.txt");
    return 0;
}
```

## 三、BOM（Byte Order Mark）

### 什么是BOM

BOM是放在文件开头的特殊标记，用于标识文件的编码格式和字节序：

```
UTF-8 BOM:       EF BB BF
UTF-16 LE BOM:   FF FE
UTF-16 BE BOM:   FE FF
UTF-32 LE BOM:   FF FE 00 00
UTF-32 BE BOM:   00 00 FE FF
```

### 检测和处理BOM

```c
#include <stdio.h>
#include <string.h>

typedef enum {
    ENCODING_UNKNOWN,
    ENCODING_UTF8_BOM,
    ENCODING_UTF8,
    ENCODING_UTF16_LE,
    ENCODING_UTF16_BE,
    ENCODING_ASCII
} Encoding;

Encoding detect_encoding(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return ENCODING_UNKNOWN;

    unsigned char bom[4] = {0};
    size_t n = fread(bom, 1, 4, fp);
    fclose(fp);

    if (n >= 3 && bom[0] == 0xEF && bom[1] == 0xBB && bom[2] == 0xBF) {
        return ENCODING_UTF8_BOM;
    }
    if (n >= 2 && bom[0] == 0xFF && bom[1] == 0xFE) {
        return ENCODING_UTF16_LE;
    }
    if (n >= 2 && bom[0] == 0xFE && bom[1] == 0xFF) {
        return ENCODING_UTF16_BE;
    }

    // 检测是否为纯ASCII
    return ENCODING_UTF8;  // 假设为UTF-8
}

const char *encoding_name(Encoding enc) {
    switch (enc) {
        case ENCODING_UTF8_BOM: return "UTF-8 (with BOM)";
        case ENCODING_UTF8:     return "UTF-8";
        case ENCODING_UTF16_LE: return "UTF-16 Little Endian";
        case ENCODING_UTF16_BE: return "UTF-16 Big Endian";
        case ENCODING_ASCII:    return "ASCII";
        default:                return "Unknown";
    }
}

int main() {
    Encoding enc = detect_encoding("test.txt");
    printf("文件编码: %s\n", encoding_name(enc));
    return 0;
}
```

## 四、UTF-8 验证

```c
#include <stdio.h>

// 验证UTF-8字节序列是否合法
int validate_utf8(const unsigned char *data, size_t len) {
    size_t i = 0;

    while (i < len) {
        int expected;

        if ((data[i] & 0x80) == 0x00) {
            expected = 1;
        } else if ((data[i] & 0xE0) == 0xC0) {
            expected = 2;
            if (data[i] < 0xC2) return 0;  // 过长编码
        } else if ((data[i] & 0xF0) == 0xE0) {
            expected = 3;
        } else if ((data[i] & 0xF8) == 0xF0) {
            expected = 4;
            if (data[i] > 0xF4) return 0;  // 超出Unicode范围
        } else {
            return 0;  // 非法的起始字节
        }

        if (i + expected > len) return 0;  // 数据不完整

        for (int j = 1; j < expected; j++) {
            if ((data[i + j] & 0xC0) != 0x80) return 0;  // 非法的后续字节
        }

        i += expected;
    }

    return 1;  // 合法
}

int main() {
    unsigned char valid[]   = {0xE4, 0xBD, 0xA0, 0xE5, 0xA5, 0xBD};  // "你好"
    unsigned char invalid[] = {0xE4, 0xBD, 0xFF, 0xE5, 0xA5, 0xBD};  // 损坏

    printf("valid:   %s\n", validate_utf8(valid, 6) ? "合法" : "非法");
    printf("invalid: %s\n", validate_utf8(invalid, 6) ? "合法" : "非法");

    return 0;
}
```

## 五、读取UTF-8文件逐行处理

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// 读取UTF-8文件的行，正确处理多字节字符
int read_utf8_lines(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) { perror("打开失败"); return -1; }

    char *line = NULL;
    size_t line_cap = 0;
    int line_num = 0;

    // 使用getline（POSIX）或自定义实现
    while (1) {
        // 动态扩展行缓冲区
        if (line_cap == 0) {
            line_cap = 1024;
            line = malloc(line_cap);
        }

        int ch;
        size_t pos = 0;

        while ((ch = fgetc(fp)) != EOF) {
            if (pos + 2 >= line_cap) {
                line_cap *= 2;
                line = realloc(line, line_cap);
            }

            line[pos++] = ch;

            if (ch == '\n') break;
        }

        if (pos == 0 && ch == EOF) break;

        line[pos] = '\0';
        line_num++;

        // 统计这一行的UTF-8字符数
        int chars = 0;
        for (size_t i = 0; i < pos; i++) {
            if ((line[i] & 0xC0) != 0x80) chars++;
        }

        printf("第%d行 (%zu字节, %d字符): %s",
               line_num, pos, chars, line);
    }

    free(line);
    fclose(fp);
    return 0;
}
```

## 六、重点与注意事项

> **要点总结：**
>
> 1. **C语言没有内置Unicode支持**——需要手动处理多字节编码
> 2. **UTF-8是目前最通用的编码**——兼容ASCII，无字节序问题
> 3. **`char` 类型不足以表示一个UTF-8字符**——中文需要3个 `char`
> 4. **字符串长度 ≠ 字符数**——`strlen` 返回字节数，不是字符数
> 5. BOM不是必须的——许多工具不识别UTF-8 BOM
> 6. Windows的 `wchar_t` 和 `wcs` 函数使用UTF-16，POSIX使用UTF-8
> 7. 生产环境推荐使用 ICU（International Components for Unicode）库
