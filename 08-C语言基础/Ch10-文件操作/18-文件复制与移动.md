# 18 - 文件复制与移动

## 一、文件复制的基本原理

### 复制的本质

```
文件复制 = 创建新文件 + 将源文件内容逐块写入新文件

步骤：
1. 打开源文件（只读）
2. 创建目标文件（只写）
3. 循环读取源文件数据块，写入目标文件
4. 关闭两个文件
5. （可选）复制文件属性、权限、时间戳
```

## 二、基本文件复制

### 字符流复制

```c
#include <stdio.h>

int copy_file_basic(const char *src, const char *dst) {
    FILE *in = fopen(src, "rb");
    if (!in) {
        perror("打开源文件失败");
        return -1;
    }

    FILE *out = fopen(dst, "wb");
    if (!out) {
        perror("创建目标文件失败");
        fclose(in);
        return -1;
    }

    int ch;
    while ((ch = fgetc(in)) != EOF) {
        fputc(ch, out);
    }

    fclose(in);
    fclose(out);
    return 0;
}
```

### 缓冲区复制（推荐）

```c
#include <stdio.h>

int copy_file_buffered(const char *src, const char *dst) {
    FILE *in = fopen(src, "rb");
    if (!in) { perror("打开源文件失败"); return -1; }

    FILE *out = fopen(dst, "wb");
    if (!out) { perror("创建目标文件失败"); fclose(in); return -1; }

    char buffer[8192];  // 8KB 缓冲区
    size_t bytes;

    while ((bytes = fread(buffer, 1, sizeof(buffer), in)) > 0) {
        if (fwrite(buffer, 1, bytes, out) != bytes) {
            perror("写入失败");
            fclose(in);
            fclose(out);
            remove(dst);  // 删除不完整的目标文件
            return -1;
        }
    }

    // 检查读取是否因错误而停止
    if (ferror(in)) {
        perror("读取错误");
        fclose(in);
        fclose(out);
        remove(dst);
        return -1;
    }

    fclose(in);
    fclose(out);
    return 0;
}
```

## 三、保留文件属性的复制

### 使用 stat 复制权限和时间

```c
#include <stdio.h>
#include <sys/stat.h>
#include <utime.h>

int copy_file_preserve(const char *src, const char *dst) {
    struct stat st;
    if (stat(src, &st) == -1) {
        perror("获取源文件信息失败");
        return -1;
    }

    FILE *in = fopen(src, "rb");
    if (!in) { perror("打开源文件失败"); return -1; }

    FILE *out = fopen(dst, "wb");
    if (!out) { perror("创建目标文件失败"); fclose(in); return -1; }

    // 复制内容
    char buffer[8192];
    size_t bytes;
    while ((bytes = fread(buffer, 1, sizeof(buffer), in)) > 0) {
        fwrite(buffer, 1, bytes, out);
    }

    fclose(in);
    fclose(out);

    // 复制权限
    chmod(dst, st.st_mode & 0777);

    // 复制时间戳
    struct utimbuf times;
    times.actime  = st.st_atime;  // 访问时间
    times.modtime = st.st_mtime;  // 修改时间
    utime(dst, &times);

    return 0;
}
```

## 四、显示复制进度

```c
#include <stdio.h>
#include <sys/stat.h>

typedef void (*ProgressCallback)(long long current, long long total);

void default_progress(long long current, long long total) {
    int percent = (int)((current * 100) / total);
    int bars = percent / 2;
    printf("\r[");
    for (int i = 0; i < 50; i++) {
        printf(i < bars ? "#" : "-");
    }
    printf("] %d%% (%lld/%lld bytes)", percent, current, total);
    fflush(stdout);
}

int copy_file_progress(const char *src, const char *dst,
                       ProgressCallback callback) {
    struct stat st;
    if (stat(src, &st) == -1) return -1;
    long long total = st.st_size;

    FILE *in = fopen(src, "rb");
    if (!in) return -1;

    FILE *out = fopen(dst, "wb");
    if (!out) { fclose(in); return -1; }

    char buffer[65536];  // 64KB
    long long copied = 0;
    size_t bytes;

    while ((bytes = fread(buffer, 1, sizeof(buffer), in)) > 0) {
        fwrite(buffer, 1, bytes, out);
        copied += bytes;

        if (callback) {
            callback(copied, total);
        }
    }

    printf("\n");
    fclose(in);
    fclose(out);
    return 0;
}

int main() {
    printf("复制文件...\n");
    copy_file_progress("large_file.bin", "copy.bin", default_progress);
    printf("复制完成！\n");
    return 0;
}
```

## 五、文件移动（rename）

### 使用 rename 函数

```c
#include <stdio.h>

int move_file(const char *src, const char *dst) {
    // rename 是原子操作（在同一文件系统上）
    if (rename(src, dst) == 0) {
        return 0;  // 成功
    }

    // rename 可能因为跨文件系统而失败
    // 此时需要复制后删除
    perror("rename 失败");

    // 尝试复制+删除
    FILE *in = fopen(src, "rb");
    if (!in) return -1;

    FILE *out = fopen(dst, "wb");
    if (!out) { fclose(in); return -1; }

    char buf[8192];
    size_t n;
    while ((n = fread(buf, 1, sizeof(buf), in)) > 0) {
        fwrite(buf, 1, n, out);
    }

    fclose(in);
    fclose(out);
    remove(src);  // 删除源文件

    return 0;
}
```

### rename 的行为

```
在同一文件系统上：
  rename("a.txt", "b.txt") → 原子操作，只修改目录项
  - 非常快，不涉及数据复制
  - 即使程序崩溃也不会处于中间状态

跨文件系统：
  rename 可能失败（EXDEV）
  - 需要手动实现复制+删除
  - 不再是原子操作
```

## 六、大文件复制的优化

### 使用 sendfile（Linux）

```c
#ifdef __linux__
#include <sys/sendfile.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <unistd.h>

int copy_file_sendfile(const char *src, const char *dst) {
    int in_fd = open(src, O_RDONLY);
    if (in_fd == -1) { perror("打开源文件"); return -1; }

    struct stat st;
    fstat(in_fd, &st);

    int out_fd = open(dst, O_WRONLY | O_CREAT | O_TRUNC, st.st_mode);
    if (out_fd == -1) { perror("创建目标文件"); close(in_fd); return -1; }

    off_t offset = 0;
    ssize_t sent;
    while (offset < st.st_size) {
        sent = sendfile(out_fd, in_fd, &offset, st.st_size - offset);
        if (sent <= 0) break;
    }

    close(in_fd);
    close(out_fd);
    return (sent >= 0) ? 0 : -1;
}
#endif
```

## 七、重点与注意事项

> **要点总结：**
>
> 1. 使用 `"rb"` 和 `"wb"` 模式打开文件，避免换行符转换
> 2. **缓冲区大小影响性能**——通常 4KB-64KB 为宜
> 3. `rename` 是原子操作，优先使用而非复制+删除
> 4. 复制失败时应删除不完整的目标文件
> 5. 跨文件系统无法使用 `rename`，需要复制+删除
> 6. 大文件复制应显示进度，提升用户体验
> 7. 检查 `fread`/`fwrite` 的返回值，处理部分写入和错误情况
