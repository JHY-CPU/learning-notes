# 19 - 文件加密基础

## 一、加密基础概念

### 什么是加密

加密是将原始数据（明文）通过某种算法变换为不可读的形式（密文），只有拥有密钥的人才能将其还原。

```
明文 (Plaintext) → 加密算法 + 密钥 → 密文 (Ciphertext)
密文 → 解密算法 + 密钥 → 明文
```

### 对称加密与非对称加密

| 类型 | 特点 | 示例 |
|------|------|------|
| 对称加密 | 加密和解密使用相同密钥 | XOR, AES, DES |
| 非对称加密 | 公钥加密，私钥解密 | RSA, ECC |

> **注意**：本章仅介绍简单的教学示例，不适合用于真实的安全场景。生产环境请使用成熟的加密库（如 OpenSSL）。

## 二、XOR 异或加密

### 原理

XOR（异或）是最简单的对称加密方法：用密钥与每个字节进行异或运算。

```
明文字节:  01001000  ('H')
密钥字节:  01011011  ('K')
XOR结果:   00010011  (密文)

再次XOR:
密文字节:  00010011
密钥字节:  01011011
XOR结果:   01001000  ('H') ← 还原为明文
```

### 单字节密钥加密

```c
#include <stdio.h>
#include <stdlib.h>

void xor_encrypt_file(const char *input, const char *output, unsigned char key) {
    FILE *fin = fopen(input, "rb");
    FILE *fout = fopen(output, "wb");

    if (!fin || !fout) {
        perror("文件操作失败");
        if (fin) fclose(fin);
        if (fout) fclose(fout);
        return;
    }

    int ch;
    while ((ch = fgetc(fin)) != EOF) {
        fputc(ch ^ key, fout);
    }

    fclose(fin);
    fclose(fout);
}

int main(int argc, char *argv[]) {
    if (argc != 4) {
        printf("用法: %s <输入文件> <输出文件> <密钥(0-255)>\n", argv[0]);
        return 1;
    }

    unsigned char key = (unsigned char)atoi(argv[3]);
    xor_encrypt_file(argv[1], argv[2], key);

    // 加密和解密是同一操作
    // xor_encrypt_file("encrypted.dat", "decrypted.txt", key);
    // 即可还原

    printf("加密完成！使用相同密钥再次运行即可解密。\n");
    return 0;
}
```

### 多字节密钥加密（更安全）

```c
#include <stdio.h>
#include <string.h>

void xor_encrypt_stream(FILE *in, FILE *out,
                        const unsigned char *key, size_t key_len) {
    int ch;
    size_t key_index = 0;

    while ((ch = fgetc(in)) != EOF) {
        fputc(ch ^ key[key_index], out);
        key_index = (key_index + 1) % key_len;  // 循环使用密钥
    }
}

int main() {
    const char *password = "MySecretKey2024";
    size_t key_len = strlen(password);

    FILE *fin = fopen("plaintext.txt", "rb");
    FILE *fout = fopen("encrypted.dat", "wb");

    if (!fin || !fout) {
        perror("打开文件失败");
        return 1;
    }

    xor_encrypt_stream(fin, fout, (const unsigned char *)password, key_len);

    fclose(fin);
    fclose(fout);

    printf("加密完成！\n");
    return 0;
}
```

## 三、凯撒密码

### 原理

每个字母按照固定的偏移量移动：

```
偏移量=3:
A→D, B→E, C→F, ..., X→A, Y→B, Z→C

加密: HELLO → KHOOR
解密: KHOOR → HELLO（偏移-3）
```

### 实现

```c
#include <stdio.h>
#include <ctype.h>

void caesar_encrypt_file(const char *input, const char *output, int shift) {
    FILE *fin = fopen(input, "r");
    FILE *fout = fopen(output, "w");

    if (!fin || !fout) {
        perror("文件操作失败");
        return;
    }

    int ch;
    while ((ch = fgetc(fin)) != EOF) {
        if (isupper(ch)) {
            ch = 'A' + (ch - 'A' + shift + 26) % 26;
        } else if (islower(ch)) {
            ch = 'a' + (ch - 'a' + shift + 26) % 26;
        }
        fputc(ch, fout);
    }

    fclose(fin);
    fclose(fout);
}

int main() {
    caesar_encrypt_file("message.txt", "encrypted.txt", 3);
    // 解密：偏移-3
    caesar_encrypt_file("encrypted.txt", "decrypted.txt", -3);
    return 0;
}
```

## 四、文件哈希校验

### 简单校验和

```c
#include <stdio.h>
#include <stdint.h>

// 简单的 CRC32 实现
static uint32_t crc32_table[256];
static int table_initialized = 0;

void init_crc32_table(void) {
    if (table_initialized) return;
    for (uint32_t i = 0; i < 256; i++) {
        uint32_t crc = i;
        for (int j = 0; j < 8; j++) {
            crc = (crc >> 1) ^ (crc & 1 ? 0xEDB88320 : 0);
        }
        crc32_table[i] = crc;
    }
    table_initialized = 1;
}

uint32_t compute_file_crc32(const char *filename) {
    init_crc32_table();

    FILE *fp = fopen(filename, "rb");
    if (!fp) return 0;

    uint32_t crc = 0xFFFFFFFF;
    int ch;

    while ((ch = fgetc(fp)) != EOF) {
        crc = crc32_table[(crc ^ ch) & 0xFF] ^ (crc >> 8);
    }

    fclose(fp);
    return crc ^ 0xFFFFFFFF;
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        printf("用法: %s <文件名>\n", argv[0]);
        return 1;
    }

    uint32_t crc = compute_file_crc32(argv[1]);
    printf("文件: %s\n", argv[1]);
    printf("CRC32: %08X\n", crc);

    return 0;
}
```

## 五、完整加密/解密工具

```c
#include <stdio.h>
#include <string.h>
#include <stdint.h>

// 简单的密钥派生函数
void derive_key(const char *password, uint8_t *key, size_t key_len) {
    size_t pass_len = strlen(password);
    for (size_t i = 0; i < key_len; i++) {
        key[i] = password[i % pass_len] ^ (uint8_t)(i * 31);
        // 用质数增强扩散
    }
}

int encrypt_decrypt(const char *infile, const char *outfile,
                    const char *password) {
    uint8_t key[256];
    derive_key(password, key, sizeof(key));

    FILE *fin = fopen(infile, "rb");
    if (!fin) { perror("打开输入文件失败"); return -1; }

    FILE *fout = fopen(outfile, "wb");
    if (!fout) { perror("创建输出文件失败"); fclose(fin); return -1; }

    // 写入魔数标识（方便识别加密文件）
    const char *magic = "ENC1";
    if (strcmp(infile, outfile) != 0) {  // 非原地加密
        fwrite(magic, 1, 4, fout);
    }

    int ch;
    size_t key_idx = 0;
    while ((ch = fgetc(fin)) != EOF) {
        fputc(ch ^ key[key_idx], fout);
        key_idx = (key_idx + 1) % sizeof(key);
    }

    fclose(fin);
    fclose(fout);
    return 0;
}

int main(int argc, char *argv[]) {
    if (argc != 4) {
        printf("用法: %s <encrypt|decrypt> <输入文件> <密码>\n", argv[0]);
        return 1;
    }

    char output[512];
    if (strcmp(argv[1], "encrypt") == 0) {
        snprintf(output, sizeof(output), "%s.enc", argv[2]);
        encrypt_decrypt(argv[2], output, argv[3]);
        printf("已加密: %s → %s\n", argv[2], output);
    } else if (strcmp(argv[1], "decrypt") == 0) {
        snprintf(output, sizeof(output), "%s.dec", argv[2]);
        encrypt_decrypt(argv[2], output, argv[3]);
        printf("已解密: %s → %s\n", argv[2], output);
    }

    return 0;
}
```

## 六、重点与注意事项

> **要点总结：**
>
> 1. **XOR加密的教学性质**：单字节XOR极其脆弱，不应在真实场景使用
> 2. **密钥安全**：密码不应硬编码在程序中，应从安全来源获取
> 3. **生产加密**：使用 AES-256-GCM 等标准算法，配合 OpenSSL 或 libsodium
> 4. **哈希不等于加密**：CRC32/MD5 是校验工具，不是加密工具
> 5. **二进制模式**：加密文件必须使用 `"rb"`/`"wb"` 打开
> 6. **错误处理**：加密过程中的任何错误都应中止操作
