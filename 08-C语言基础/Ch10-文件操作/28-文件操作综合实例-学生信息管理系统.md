## 三、日志功能

```c
// 写入操作日志（追加模式）
void write_log(const char *operation, const char *detail) {
    FILE *log_fp = fopen(LOG_FILE, "a");
    if (!log_fp) return;

    time_t now = time(NULL);
    struct tm *t = localtime(&now);

    fprintf(log_fp, "[%04d-%02d-%02d %02d:%02d:%02d] %s: %s\n",
            t->tm_year + 1900, t->tm_mon + 1, t->tm_mday,
            t->tm_hour, t->tm_min, t->tm_sec,
            operation, detail);

    fclose(log_fp);
}
```

## 四、核心功能实现

### 4.1 添加学生

```c
int add_student(Student *stu) {
    if (!stu) return -1;

    // 计算平均分
    stu->average = (stu->scores[0] + stu->scores[1] + stu->scores[2]) / 3.0;
    stu->is_deleted = 0;

    // 检查学号是否已存在（遍历已有记录）
    FILE *fp = fopen(DATA_FILE, "rb");
    if (fp) {
        Student temp;
        while (fread(&temp, sizeof(Student), 1, fp) == 1) {
            if (!temp.is_deleted && strcmp(temp.id, stu->id) == 0) {
                fclose(fp);
                fprintf(stderr, "错误: 学号 %s 已存在\n", stu->id);
                return -1;
            }
        }
        fclose(fp);
    }

    // 以追加模式写入新记录
    fp = fopen(DATA_FILE, "ab");
    if (!fp) {
        perror("打开数据文件失败");
        return -1;
    }

    if (fwrite(stu, sizeof(Student), 1, fp) != 1) {
        fclose(fp);
        perror("写入失败");
        return -1;
    }

    fclose(fp);

    // 记录日志
    char detail[128];
    snprintf(detail, sizeof(detail), "添加学生: %s (%s)", stu->id, stu->name);
    write_log("ADD", detail);

    printf("成功添加学生: %s\n", stu->name);
    return 0;
}
```

### 4.2 查询学生

```c
// 按学号查找，返回匹配结果数组
Student *find_student(const char *id, int *count) {
    FILE *fp = fopen(DATA_FILE, "rb");
    if (!fp) {
        *count = 0;
        return NULL;
    }

    // 第一遍：统计匹配数量
    Student temp;
    *count = 0;
    while (fread(&temp, sizeof(Student), 1, fp) == 1) {
        if (!temp.is_deleted && strcmp(temp.id, id) == 0) {
            (*count)++;
        }
    }

    if (*count == 0) {
        fclose(fp);
        return NULL;
    }

    // 分配内存并读取
    Student *results = (Student *)malloc((*count) * sizeof(Student));
    if (!results) {
        fclose(fp);
        *count = 0;
        return NULL;
    }

    rewind(fp);  // 回到文件开头
    int idx = 0;
    while (fread(&temp, sizeof(Student), 1, fp) == 1) {
        if (!temp.is_deleted && strcmp(temp.id, id) == 0) {
            results[idx++] = temp;
        }
    }

    fclose(fp);
    return results;
}
```

### 4.4 删除学生（逻辑删除）

```c
int delete_student(const char *id) {
    FILE *fp = fopen(DATA_FILE, "r+b");
    if (!fp) {
        perror("打开数据文件失败");
        return -1;
    }

    Student stu;
    int found = 0;
    long pos;

    while (1) {
        pos = ftell(fp);  // 记录当前位置
        if (fread(&stu, sizeof(Student), 1, fp) != 1) break;

        if (!stu.is_deleted && strcmp(stu.id, id) == 0) {
            // 回到记录开头，设置删除标记，写回
            fseek(fp, pos, SEEK_SET);
            stu.is_deleted = 1;
            fwrite(&stu, sizeof(Student), 1, fp);
            found = 1;
            break;
        }
    }

    fclose(fp);

    if (found) {
        char detail[64];
        snprintf(detail, sizeof(detail), "删除学生: %s", id);
        write_log("DELETE", detail);
        printf("成功删除学生: %s\n", id);
        return 0;
    } else {
        fprintf(stderr, "未找到学号: %s\n", id);
        return -1;
    }
}
```

### 4.5 更新学生信息

```c
int update_student(const char *id, Student *new_info) {
    FILE *fp = fopen(DATA_FILE, "r+b");
    if (!fp) {
        perror("打开数据文件失败");
        return -1;
    }

    Student stu;
    int found = 0;
    long pos;

    while (1) {
        pos = ftell(fp);
        if (fread(&stu, sizeof(Student), 1, fp) != 1) break;

        if (!stu.is_deleted && strcmp(stu.id, id) == 0) {
            // 更新字段（保留学号）
            strncpy(stu.name, new_info->name, MAX_NAME_LEN - 1);
            stu.age = new_info->age;
            stu.gender = new_info->gender;
            strncpy(stu.major, new_info->major, 31);
            memcpy(stu.scores, new_info->scores, sizeof(stu.scores));
            stu.average = (stu.scores[0] + stu.scores[1] + stu.scores[2]) / 3.0;

            // 写回原位置
            fseek(fp, pos, SEEK_SET);
            fwrite(&stu, sizeof(Student), 1, fp);
            found = 1;
            break;
        }
    }

    fclose(fp);

    if (found) {
        char detail[128];
        snprintf(detail, sizeof(detail), "更新学生: %s -> %s", id, new_info->name);
        write_log("UPDATE", detail);
        printf("成功更新学生: %s\n", id);
        return 0;
    }

    fprintf(stderr, "未找到学号: %s\n", id);
    return -1;
}
```

## 五、统计与排序

```c
void show_statistics(void) {
    FILE *fp = fopen(DATA_FILE, "rb");
    if (!fp) { printf("暂无数据\n"); return; }

    Student stu;
    int count = 0;
    double total_avg = 0;
    double max_avg = 0;
    char top_name[MAX_NAME_LEN] = "";
    int pass_count = 0;

    while (fread(&stu, sizeof(Student), 1, fp) == 1) {
        if (stu.is_deleted) continue;
        count++;
        total_avg += stu.average;
        if (stu.average > max_avg) {
            max_avg = stu.average;
            strcpy(top_name, stu.name);
        }
        if (stu.average >= 60) pass_count++;
    }
    fclose(fp);

    if (count == 0) { printf("暂无有效数据\n"); return; }

    printf("\n=== 数据统计 ===\n");
    printf("学生总数:   %d\n", count);
    printf("平均成绩:   %.2f\n", total_avg / count);
    printf("最高平均分: %.1f (%s)\n", max_avg, top_name);
    printf("及格率:     %.1f%% (%d/%d)\n",
           100.0 * pass_count / count, pass_count, count);
}
```

```c
// 数据备份：缓冲区复制文件
int backup_data(void) {
    FILE *src = fopen(DATA_FILE, "rb");
    if (!src) { perror("打开源文件失败"); return -1; }

    FILE *dst = fopen(BACKUP_FILE, "wb");
    if (!dst) {
        perror("创建备份文件失败");
        fclose(src);
        return -1;
    }

    char buffer[4096];
    size_t bytes;
    while ((bytes = fread(buffer, 1, sizeof(buffer), src)) > 0) {
        if (fwrite(buffer, 1, bytes, dst) != bytes) {
            perror("写入备份文件失败");
            fclose(src);
            fclose(dst);
            return -1;
        }
    }

    fclose(src);
    fclose(dst);

    write_log("BACKUP", "数据备份成功");
    printf("数据备份完成: %s\n", BACKUP_FILE);
    return 0;
}

// 数据恢复
int restore_data(void) {
    FILE *src = fopen(BACKUP_FILE, "rb");
    if (!src) { perror("备份文件不存在"); return -1; }

    FILE *dst = fopen(DATA_FILE, "wb");
    if (!dst) {
        perror("创建数据文件失败");
        fclose(src);
        return -1;
    }

    char buffer[4096];
    size_t bytes;
    while ((bytes = fread(buffer, 1, sizeof(buffer), src)) > 0) {
        fwrite(buffer, 1, bytes, dst);
    }

    fclose(src);
    fclose(dst);

    write_log("RESTORE", "数据恢复成功");
    printf("数据恢复完成\n");
    return 0;
}
```

```c
int export_csv(const char *filename) {
    FILE *src = fopen(DATA_FILE, "rb");
    if (!src) { perror("打开数据文件失败"); return -1; }

    FILE *dst = fopen(filename, "w");
    if (!dst) { perror("创建CSV文件失败"); fclose(src); return -1; }

    fprintf(dst, "学号,姓名,性别,年龄,专业,语文,数学,英语,平均分\n");

    Student stu;
    while (fread(&stu, sizeof(Student), 1, src) == 1) {
        if (stu.is_deleted) continue;
        fprintf(dst, "%s,%s,%c,%d,%s,%.1f,%.1f,%.1f,%.1f\n",
                stu.id, stu.name, stu.gender, stu.age, stu.major,
                stu.scores[0], stu.scores[1], stu.scores[2], stu.average);
    }

    fclose(src);
    fclose(dst);

    write_log("EXPORT", filename);
    printf("已导出到 %s\n", filename);
    return 0;
}
```

### 关键设计原则

```
1. 二进制存储 + 文本导出：兼顾效率和可读性
2. 逻辑删除：安全可恢复，不破坏数据
3. fseek/ftell 原地更新：利用固定长度记录实现随机访问
4. "ab" 追加模式添加：不影响已有数据
5. 两遍遍历：第一遍统计数量，第二遍读取数据
6. 日志审计：所有关键操作记录日志
7. 缓冲区安全：scanf 使用宽度限制防止溢出
8. 错误处理：所有文件操作检查返回值
```

> **要点总结：**
>
> 1. 二进制文件存储效率高，文本CSV便于查看和跨程序交换
> 2. 逻辑删除比物理删除更安全——数据可以恢复
> 3. 固定长度记录使得 fseek 可以精确定位到任意记录
> 4. "r+b" 模式可以在文件中间修改数据而不影响其他记录
> 5. 备份文件使用缓冲区复制，适合大文件
> 6. 日志使用追加模式("a")，不会覆盖历史记录
> 7. 此系统为教学示例，生产环境需要更多安全和性能优化（如文件锁、索引）
