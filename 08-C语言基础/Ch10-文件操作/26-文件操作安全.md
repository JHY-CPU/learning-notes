# 26 - 文件操作安全

## 一、常见安全威胁

### 安全威胁概览

```
1. 路径遍历攻击 (Path Traversal)
2. 竞态条件 (TOCTOU)
3. 符号链接攻击 (Symlink Attack)
4. 缓冲区溢出 (Buffer Overflow)
5. 临时文件安全 (Temp File Security)
6. 权限提升 (Privilege Escalation)
```

## 二、路径遍历攻击

### 什么是路径遍历

```
攻击者通过构造包含 "../" 的路径，访问不应该被访问的文件：

用户输入: "../../../etc/passwd"
拼接后:   "/var/app/uploads/../../../etc/passwd"
解析后:   "/etc/passwd"  ← 窃取系统文件！

其他危险路径:
  "..\\..\\windows\\system32\\config\\sam"  (Windows)
  "/etc/shadow"
  "~/.ssh/id_rsa"
```

### 防御方法

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <limits.h>

#ifdef _WIN32
    #define PATH_SEP '\\'
#else
    #define PATH_SEP '/'
#endif

// 检查路径是否安全（防止路径遍历）
int is_safe_path(const char *base_dir, const char *user_path) {
    // 检查危险字符
    if (strstr(user_path, "..") != NULL) return 0;
    if (user_path[0] == '/' || user_path[0] == '\\') return 0;
    if (strstr(user_path, "~") != NULL) return 0;

    // 构建完整路径
    char fullpath[PATH_MAX];
    snprintf(fullpath, sizeof(fullpath), "%s%c%s", base_dir, PATH_SEP, user_path);

    // 使用realpath解析（POSIX）
#ifndef _WIN32
    char resolved[PATH_MAX];
    if (!realpath(fullpath, resolved)) return 0;

    // 确保解析后的路径仍在base_dir内
    size_t base_len = strlen(base_dir);
    if (strncmp(resolved, base_dir, base_len) != 0) return 0;
    if (resolved[base_len] != PATH_SEP && resolved[base_len] != '\0') return 0;
#endif

    return 1;
}

// 使用示例
int safe_read_file(const char *base_dir, const char *filename) {
    if (!is_safe_path(base_dir, filename)) {
        fprintf(stderr, "拒绝：不安全的文件路径 '%s'\n", filename);
        return -1;
    }

    char fullpath[PATH_MAX];
    snprintf(fullpath, sizeof(fullpath), "%s/%s", base_dir, filename);

    FILE *fp = fopen(fullpath, "r");
    if (!fp) { perror("打开失败"); return -1; }

    printf("安全读取: %s\n", fullpath);
    fclose(fp);
    return 0;
}
```

## 三、竞态条件（TOCTOU）

### 什么是TOCTOU

```
Time-of-Check to Time-of-Use（检查与使用之间的竞态）

检查阶段：if (access(file, W_OK) == 0)  ← 检查文件可写
         （攻击者在此时替换文件为符号链接）
使用阶段：fp = fopen(file, "w");          ← 实际打开（可能打开错误的文件！）
```

### 攻击示例

```
时间线：
  程序A: 检查 config.txt → 可以写入
  攻击者: 将 config.txt 替换为指向 /etc/shadow 的符号链接
  程序A: 打开 config.txt 写入 → 实际写入了 /etc/shadow！
```

### 防御方法

```c
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/stat.h>

// 安全的方式：使用O_NOFOLLOW防止符号链接攻击
int safe_open_no_follow(const char *path) {
    int fd = open(path, O_RDWR | O_NOFOLLOW);
    if (fd == -1) {
        perror("安全打开失败");
        return -1;
    }

    // 验证打开的确实是普通文件
    struct stat st;
    if (fstat(fd, &st) == -1) {
        close(fd);
        return -1;
    }

    if (!S_ISREG(st.st_mode)) {
        fprintf(stderr, "拒绝：不是普通文件\n");
        close(fd);
        return -1;
    }

    return fd;
}

// 原子创建：O_CREAT | O_EXCL 确保文件不存在时才创建
int safe_create(const char *path, mode_t mode) {
    int fd = open(path, O_WRONLY | O_CREAT | O_EXCL, mode);
    if (fd == -1) {
        if (errno == EEXIST) {
            fprintf(stderr, "文件已存在，拒绝覆盖\n");
        }
        return -1;
    }
    return fd;
}
```

## 四、符号链接攻击

### 检测符号链接

```c
#include <stdio.h>
#include <sys/stat.h>
#include <unistd.h>

// 检查文件是否为符号链接
int is_symlink(const char *path) {
    struct stat st;
    if (lstat(path, &st) == -1) return -1;
    return S_ISLNK(st.st_mode);
}

// 安全读取：拒绝符号链接
FILE *safe_open_regular(const char *path, const char *mode) {
    struct stat st;
    if (lstat(path, &st) == -1) {
        perror("获取文件信息失败");
        return NULL;
    }

    if (S_ISLNK(st.st_mode)) {
        fprintf(stderr, "拒绝：文件是符号链接\n");
        return NULL;
    }

    if (!S_ISREG(st.st_mode)) {
        fprintf(stderr, "拒绝：不是普通文件\n");
        return NULL;
    }

    return fopen(path, mode);
}
```

## 五、安全的文件写入模式

### 原子写入（先写临时文件再rename）

```c
#include <stdio.h>
#include <string.h>
#include <unistd.h>

// 安全的原子写入
int safe_write_file(const char *filename, const char *data, size_t len) {
    // 构建临时文件名
    char tmpname[PATH_MAX];
    snprintf(tmpname, sizeof(tmpname), "%s.tmp.XXXXXX", filename);

    // 创建临时文件
    int fd = mkstemp(tmpname);
    if (fd == -1) { perror("创建临时文件失败"); return -1; }

    FILE *fp = fdopen(fd, "w");
    if (!fp) { close(fd); unlink(tmpname); return -1; }

    // 写入数据
    if (fwrite(data, 1, len, fp) != len) {
        fclose(fp);
        unlink(tmpname);
        return -1;
    }

    // 确保数据写入磁盘
    fflush(fp);
    fclose(fp);

    // 原子重命名（在同一文件系统上是原子操作）
    if (rename(tmpname, filename) == -1) {
        perror("rename 失败");
        unlink(tmpname);
        return -1;
    }

    return 0;
}
```

## 六、安全的字符串处理

### 防止缓冲区溢出

```c
#include <stdio.h>
#include <string.h>

// ❌ 不安全
void unsafe_path(const char *dir, const char *file) {
    char path[256];
    strcpy(path, dir);     // 可能溢出
    strcat(path, "/");     // 可能溢出
    strcat(path, file);    // 可能溢出
}

// ✅ 安全
void safe_path(char *dest, size_t dest_size,
               const char *dir, const char *file) {
    int written = snprintf(dest, dest_size, "%s/%s", dir, file);
    if (written < 0 || (size_t)written >= dest_size) {
        fprintf(stderr, "路径拼接溢出！\n");
        dest[0] = '\0';
    }
}
```

## 七、安全检查清单

```c
/*
文件操作安全检查清单：

1. 路径验证
   □ 不直接使用用户输入作为文件路径
   □ 过滤或拒绝 ".."、"/"、"~" 等危险字符
   □ 使用 realpath 验证路径在允许范围内

2. 竞态条件
   □ 使用 open(O_NOFOLLOW) 而非 fopen
   □ 使用 O_CREAT | O_EXCL 原子创建文件
   □ 避免 access() + fopen() 的两步操作

3. 权限控制
   □ 创建文件时设置最小必要权限
   □ 使用 umask 限制默认权限
   □ 敏感文件存储在安全目录

4. 错误处理
   □ 检查所有文件操作的返回值
   □ 失败时清理已创建的临时文件
   □ 不在错误信息中泄露敏感路径

5. 临时文件
   □ 使用 mkstemp 而非 tmpnam
   □ 使用后立即 unlink
   □ 设置合理权限（0600）

6. 输入验证
   □ 限制文件名长度
   □ 验证文件大小在合理范围内
   □ 检查文件类型（MIME类型/扩展名）
*/
```

## 八、重点与注意事项

> **要点总结：**
>
> 1. **永远不要信任用户输入的文件路径**——必须验证和限制
> 2. **使用 `O_NOFOLLOW` 防止符号链接攻击**
> 3. **TOCTOU竞态条件**通过原子操作（O_EXCL、rename）避免
> 4. **先写临时文件再 rename** 是安全写入的标准模式
> 5. 使用 `realpath` 解析路径，确认在允许范围内
> 6. `snprintf` 替代 `strcpy`/`strcat` 防止缓冲区溢出
> 7. 生产环境建议使用 seccomp/AppArmor/SELinux 等额外安全机制
