# 17 - 日志系统实现

## 一、日志系统概述

### 为什么需要日志

```
日志的作用：
- 记录程序运行状态和行为
- 排查问题和调试的依据
- 安全审计和合规要求
- 性能监控和分析
- 用户行为追踪
```

### 日志级别

| 级别 | 名称 | 含义 | 用途 |
|------|------|------|------|
| 0 | TRACE | 最详细的信息 | 开发调试 |
| 1 | DEBUG | 调试信息 | 开发调试 |
| 2 | INFO | 正常运行信息 | 运行状态 |
| 3 | WARN | 警告信息 | 潜在问题 |
| 4 | ERROR | 错误信息 | 操作失败 |
| 5 | FATAL | 致命错误 | 程序无法继续 |

## 二、基础日志系统实现

### 数据结构与函数声明

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <stdarg.h>

typedef enum {
    LOG_TRACE,
    LOG_DEBUG,
    LOG_INFO,
    LOG_WARN,
    LOG_ERROR,
    LOG_FATAL
} LogLevel;

typedef struct {
    FILE *fp;           // 日志文件指针
    LogLevel min_level; // 最低输出级别
    int to_console;     // 是否同时输出到控制台
    int show_timestamp; // 是否显示时间戳
    char filename[256]; // 日志文件名
} Logger;

static Logger logger = {NULL, LOG_INFO, 1, 1, ""};
```

### 日志级别名称

```c
static const char *level_names[] = {
    "TRACE", "DEBUG", "INFO ", "WARN ", "ERROR", "FATAL"
};

static const char *level_colors[] = {
    "\033[90m",  // TRACE - 灰色
    "\033[36m",  // DEBUG - 青色
    "\033[32m",  // INFO  - 绿色
    "\033[33m",  // WARN  - 黄色
    "\033[31m",  // ERROR - 红色
    "\033[35m"   // FATAL - 紫色
};
```

### 初始化与关闭

```c
int log_init(const char *filename, LogLevel level, int to_console) {
    logger.min_level = level;
    logger.to_console = to_console;
    logger.show_timestamp = 1;

    if (filename && filename[0]) {
        strncpy(logger.filename, filename, sizeof(logger.filename) - 1);
        logger.fp = fopen(filename, "a");  // 追加模式
        if (!logger.fp) {
            fprintf(stderr, "无法打开日志文件: %s\n", filename);
            return -1;
        }
    }
    return 0;
}

void log_close(void) {
    if (logger.fp) {
        fclose(logger.fp);
        logger.fp = NULL;
    }
}
```

### 核心日志函数

```c
void log_write(LogLevel level, const char *file, int line,
               const char *fmt, ...) {
    if (level < logger.min_level) return;

    // 获取当前时间
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    char timebuf[32];
    strftime(timebuf, sizeof(timebuf), "%Y-%m-%d %H:%M:%S", tm_info);

    // 提取文件名（去掉路径）
    const char *basename = strrchr(file, '/');
#ifdef _WIN32
    if (!basename) basename = strrchr(file, '\\');
#endif
    basename = basename ? basename + 1 : file;

    // 格式化用户消息
    char msgbuf[1024];
    va_list args;
    va_start(args, fmt);
    vsnprintf(msgbuf, sizeof(msgbuf), fmt, args);
    va_end(args);

    // 构建完整日志行
    char logbuf[1536];
    snprintf(logbuf, sizeof(logbuf), "[%s] %-5s %s:%d - %s\n",
             timebuf, level_names[level], basename, line, msgbuf);

    // 输出到文件
    if (logger.fp) {
        fputs(logbuf, logger.fp);
        fflush(logger.fp);  // 确保立即写入
    }

    // 输出到控制台
    if (logger.to_console) {
        fprintf(stderr, "%s%s\033[0m",
                level_colors[level], logbuf);
    }
}

// 便利宏
#define log_trace(...) log_write(LOG_TRACE, __FILE__, __LINE__, __VA_ARGS__)
#define log_debug(...) log_write(LOG_DEBUG, __FILE__, __LINE__, __VA_ARGS__)
#define log_info(...)  log_write(LOG_INFO,  __FILE__, __LINE__, __VA_ARGS__)
#define log_warn(...)  log_write(LOG_WARN,  __FILE__, __LINE__, __VA_ARGS__)
#define log_error(...) log_write(LOG_ERROR, __FILE__, __LINE__, __VA_ARGS__)
#define log_fatal(...) log_write(LOG_FATAL, __FILE__, __LINE__, __VA_ARGS__)
```

## 三、使用示例

```c
int main() {
    // 初始化日志系统
    log_init("app.log", LOG_DEBUG, 1);

    log_info("程序启动");
    log_debug("配置加载完成，端口=%d", 8080);

    int connection_count = 42;
    log_info("当前连接数: %d", connection_count);

    if (connection_count > 100) {
        log_warn("连接数过高: %d", connection_count);
    }

    log_error("数据库连接失败: %s", "Connection refused");
    log_fatal("无法恢复的错误，程序退出");

    log_info("程序正常结束");
    log_close();

    return 0;
}
```

## 四、日志轮转（Log Rotation）

### 按大小轮转

```c
#include <sys/stat.h>

#define MAX_LOG_SIZE (10 * 1024 * 1024)  // 10MB

void log_rotate(void) {
    if (!logger.fp || !logger.filename[0]) return;

    // 检查文件大小
    struct stat st;
    if (fstat(fileno(logger.fp), &st) != 0) return;

    if (st.st_size < MAX_LOG_SIZE) return;

    fclose(logger.fp);

    // 重命名旧日志文件
    char oldname[300];
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);

    snprintf(oldname, sizeof(oldname), "%s.%04d%02d%02d_%02d%02d%02d",
             logger.filename,
             tm_info->tm_year + 1900, tm_info->tm_mon + 1, tm_info->tm_mday,
             tm_info->tm_hour, tm_info->tm_min, tm_info->tm_sec);

    rename(logger.filename, oldname);

    // 创建新的日志文件
    logger.fp = fopen(logger.filename, "a");
}
```

## 五、重点与注意事项

> **要点总结：**
>
> 1. **日志级别过滤**——低于设定级别的日志不会输出
> 2. **日志写入后应 fflush**——确保程序崩溃时日志不丢失
> 3. **日志应包含时间戳、级别、文件名、行号**——便于定位问题
> 4. **日志文件需要轮转**——防止文件无限增长占满磁盘
> 5. **多线程环境下需要加锁**保护日志写入
> 6. 生产环境日志级别通常设为 INFO 或 WARN
> 7. 使用宏封装可以减少性能开销（低于级别的日志不会执行格式化）
