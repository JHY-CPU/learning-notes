# 21 - 随机访问文件

## 一、随机访问概述

### 顺序访问 vs 随机访问

```
顺序访问：
  记录1 → 记录2 → 记录3 → 记录4 → ...
  必须按顺序依次读取，查找第N条需要O(N)时间

随机访问：
  可以直接跳到第N条记录读取，查找时间O(1)
  前提：记录长度固定
```

## 二、固定长度记录

### 数据结构

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int    id;          // 学号
    char   name[32];    // 姓名
    char   major[32];   // 专业
    double gpa;         // 绩点
    int    active;      // 是否有效
} StudentRecord;

#define REC_SIZE sizeof(StudentRecord)
```

### 基本操作

```c
// 写入指定位置
void write_at(FILE *fp, int index, const StudentRecord *rec) {
    fseek(fp, index * REC_SIZE, SEEK_SET);
    fwrite(rec, REC_SIZE, 1, fp);
}

// 读取指定位置
int read_at(FILE *fp, int index, StudentRecord *rec) {
    fseek(fp, index * REC_SIZE, SEEK_SET);
    return fread(rec, REC_SIZE, 1, fp) == 1;
}

// 获取记录数
int file_record_count(FILE *fp) {
    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fseek(fp, 0, SEEK_SET);
    return (int)(size / REC_SIZE);
}
```

## 三、索引文件

### 为什么要索引

```
问题：如何通过学号快速查找学生？

方案1：遍历所有记录 → O(N)
方案2：维护学号到位置的索引 → O(log N) 或 O(1)

索引文件结构：
  索引项 = { 学号, 记录在数据文件中的位置 }
  按学号排序后，可用二分查找
```

### 索引实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int id;
    long offset;  // 在数据文件中的字节偏移
} IndexEntry;

// 构建索引
int build_index(const char *data_file, const char *index_file) {
    FILE *dfp = fopen(data_file, "rb");
    if (!dfp) return -1;

    // 获取记录数
    fseek(dfp, 0, SEEK_END);
    int count = (int)(ftell(dfp) / REC_SIZE);
    fseek(dfp, 0, SEEK_SET);

    // 构建索引数组
    IndexEntry *index = malloc(count * sizeof(IndexEntry));
    StudentRecord rec;

    for (int i = 0; i < count; i++) {
        fread(&rec, REC_SIZE, 1, dfp);
        index[i].id = rec.id;
        index[i].offset = i * REC_SIZE;
    }
    fclose(dfp);

    // 按学号排序（简单冒泡排序）
    for (int i = 0; i < count - 1; i++) {
        for (int j = 0; j < count - 1 - i; j++) {
            if (index[j].id > index[j + 1].id) {
                IndexEntry tmp = index[j];
                index[j] = index[j + 1];
                index[j + 1] = tmp;
            }
        }
    }

    // 写入索引文件
    FILE *ifp = fopen(index_file, "wb");
    if (!ifp) { free(index); return -1; }

    fwrite(&count, sizeof(int), 1, ifp);
    fwrite(index, sizeof(IndexEntry), count, ifp);

    fclose(ifp);
    free(index);
    return count;
}

// 通过学号查找（二分查找）
long find_by_id_indexed(const char *data_file, const char *index_file,
                        int target_id, StudentRecord *result) {
    FILE *ifp = fopen(index_file, "rb");
    if (!ifp) return -1;

    int count;
    fread(&count, sizeof(int), 1, ifp);

    IndexEntry *index = malloc(count * sizeof(IndexEntry));
    fread(index, sizeof(IndexEntry), count, ifp);
    fclose(ifp);

    // 二分查找
    int low = 0, high = count - 1;
    long offset = -1;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (index[mid].id == target_id) {
            offset = index[mid].offset;
            break;
        } else if (index[mid].id < target_id) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    free(index);

    // 读取数据
    if (offset >= 0 && result) {
        FILE *dfp = fopen(data_file, "rb");
        fseek(dfp, offset, SEEK_SET);
        fread(result, REC_SIZE, 1, dfp);
        fclose(dfp);
    }

    return offset;
}
```

## 四、变长记录处理

### 变长记录的挑战

```
固定长度记录：
  [记录1: 80字节] [记录2: 80字节] [记录3: 80字节]
  → 可以直接 fseek 到任意位置

变长记录：
  [记录1: 52字节] [记录2: 128字节] [记录3: 35字节]
  → 无法直接跳到第N条，需要从头遍历
```

### 变长记录索引方案

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    long offset;  // 记录在文件中的偏移
    int  length;  // 记录的长度
} VarRecIndex;

// 写入变长记录
void write_var_record(FILE *data_fp, FILE *index_fp,
                      const char *data, int length) {
    // 获取当前数据文件位置
    fseek(data_fp, 0, SEEK_END);
    long offset = ftell(data_fp);

    // 写入数据
    fwrite(data, 1, length, data_fp);

    // 写入索引
    VarRecIndex idx = {offset, length};
    fwrite(&idx, sizeof(VarRecIndex), 1, index_fp);
}

// 读取变长记录
char *read_var_record(FILE *data_fp, FILE *index_fp, int index, int *length) {
    // 定位到索引项
    fseek(index_fp, sizeof(int) + index * sizeof(VarRecIndex), SEEK_SET);

    VarRecIndex idx;
    if (fread(&idx, sizeof(VarRecIndex), 1, index_fp) != 1) return NULL;

    // 定位到数据
    fseek(data_fp, idx.offset, SEEK_SET);

    char *data = malloc(idx.length + 1);
    fread(data, 1, idx.length, data_fp);
    data[idx.length] = '\0';

    if (length) *length = idx.length;
    return data;
}
```

## 五、完整示例：学生数据库

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int    id;
    char   name[32];
    double gpa;
} Student;

#define REC sizeof(Student)

void demo() {
    const char *db = "students.db";
    const char *idx = "students.idx";

    // 1. 创建数据
    FILE *dfp = fopen(db, "wb");
    Student students[] = {
        {2003, "王五", 3.8},
        {2001, "张三", 3.9},
        {2005, "赵六", 3.5},
        {2002, "李四", 3.7},
        {2004, "钱七", 3.6}
    };

    for (int i = 0; i < 5; i++) {
        fwrite(&students[i], REC, 1, dfp);
    }
    fclose(dfp);

    // 2. 构建索引
    int count = build_index(db, idx);
    printf("索引构建完成，共 %d 条记录\n\n", count);

    // 3. 快速查找
    int search_ids[] = {2003, 2001, 2005, 2099};
    Student found;

    for (int i = 0; i < 4; i++) {
        long offset = find_by_id_indexed(db, idx, search_ids[i], &found);
        if (offset >= 0) {
            printf("找到学号%d: %s (GPA: %.1f)\n",
                   found.id, found.name, found.gpa);
        } else {
            printf("学号%d 不存在\n", search_ids[i]);
        }
    }

    // 4. 随机访问演示
    printf("\n=== 随机访问第2条记录（索引1）===\n");
    dfp = fopen(db, "rb");
    Student s;
    read_at(dfp, 1, &s);  // 直接跳到第2条
    printf("ID:%d 名:%s GPA:%.1f\n", s.id, s.name, s.gpa);
    fclose(dfp);
}

int main() {
    demo();
    return 0;
}
```

## 六、重点与注意事项

> **要点总结：**
>
> 1. **随机访问要求固定长度记录**——变长记录需要额外的索引
> 2. **索引文件应与数据文件分离**——索引可以独立重建
> 3. **索引排序后使用二分查找**——O(log N) 查找效率
> 4. 删除记录可以用逻辑删除标记，避免移动后续记录
> 5. `fseek` 的第二个参数是**字节偏移**，不是记录号
> 6. 超大文件的索引也应考虑分页或B+树结构
> 7. 索引文件损坏时可从数据文件重建
