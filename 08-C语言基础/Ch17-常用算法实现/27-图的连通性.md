# 图的连通性

## 1. 连通性基础

- **无向图连通分量**：极大连通子图
- **有向图强连通分量（SCC）**：有向图中任意两点互相可达的极大子图
- **割点/割边**：删除后增加连通分量数的点/边
- **双连通分量**：不含割点的极大子图

## 2. Tarjan算法求强连通分量

```c
#include <stdio.h>
#include <string.h>

#define MAXN 10005
#define min(a, b) ((a) < (b) ? (a) : (b))

int head[MAXN], to[MAXN], nxt[MAXN], edge_cnt = 0;
int dfn[MAXN], low[MAXN], dfsClock = 0;
int stack[MAXN], top = -1, inStack[MAXN];
int scc[MAXN], sccCount = 0;  // scc[u]: u属于哪个SCC

void addEdge(int u, int v) {
    to[edge_cnt] = v;
    nxt[edge_cnt] = head[u];
    head[u] = edge_cnt++;
}

void tarjan(int u) {
    dfn[u] = low[u] = ++dfsClock;
    stack[++top] = u;
    inStack[u] = 1;

    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (!dfn[v]) {
            tarjan(v);
            low[u] = min(low[u], low[v]);
        } else if (inStack[v]) {
            low[u] = min(low[u], dfn[v]);
        }
    }

    // u是SCC的根
    if (low[u] == dfn[u]) {
        sccCount++;
        int v;
        do {
            v = stack[top--];
            inStack[v] = 0;
            scc[v] = sccCount;
        } while (v != u);
    }
}

int main() {
    memset(head, -1, sizeof(head));
    int n, m;
    // scanf("%d %d", &n, &m);
    // 添加边...
    for (int i = 0; i < n; i++)
        if (!dfn[i]) tarjan(i);
    printf("强连通分量数: %d\n", sccCount);
    return 0;
}
```

## 3. 割点判定

```c
int dfnV[MAXN], lowV[MAXN], visTime = 0;
int isCut[MAXN];  // 是否为割点

void findCutVertex(int u, int par) {
    dfnV[u] = lowV[u] = ++visTime;
    int children = 0;

    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (!dfnV[v]) {
            children++;
            findCutVertex(v, u);
            lowV[u] = min(lowV[u], lowV[v]);

            // u不是根节点，且v的子树不能回到u的祖先
            if (par != -1 && lowV[v] >= dfnV[u])
                isCut[u] = 1;
        } else if (v != par) {
            lowV[u] = min(lowV[u], dfnV[v]);
        }
    }
    // u是根节点，且有至少两个子树
    if (par == -1 && children >= 2)
        isCut[u] = 1;
}
```

## 4. 割边（桥）判定

```c
int dfnE[MAXN], lowE[MAXN], visTimeE = 0;

void findBridge(int u, int par) {
    dfnE[u] = lowE[u] = ++visTimeE;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == par) continue;  // 跳过父边
        if (!dfnE[v]) {
            findBridge(v, u);
            lowE[u] = min(lowE[u], lowE[v]);
            if (lowE[v] > dfnE[u])
                printf("桥: %d - %d\n", u, v);
        } else {
            lowE[u] = min(lowE[u], dfnE[v]);
        }
    }
}
```

## 5. 双连通分量

### 点双连通分量（不含割点）

```c
int dfnBCC[MAXN], lowBCC[MAXN], visTimeBCC = 0;
int stackEdge[MAXN * 2], topEdge = -1;  // 边栈

void findBCC(int u, int par) {
    dfnBCC[u] = lowBCC[u] = ++visTimeBCC;
    int childCount = 0;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (!dfnBCC[v]) {
            childCount++;
            stackEdge[++topEdge] = i;  // 入边栈
            findBCC(v, u);
            lowBCC[u] = min(lowBCC[u], lowBCC[v]);

            if (lowBCC[v] >= dfnBCC[u]) {
                // 找到一个点双连通分量
                printf("点双连通分量: ");
                int e;
                do {
                    e = stackEdge[topEdge--];
                    printf("%d-%d ", to[e ^ 1], to[e]);
                } while (e != i);
                printf("\n");
            }
        } else if (dfnBCC[v] < dfnBCC[u] && v != par) {
            stackEdge[++topEdge] = i;
            lowBCC[u] = min(lowBCC[u], dfnBCC[v]);
        }
    }
}
```

## 6. 2-SAT问题

```c
// 2-SAT：每个变量有两种选择，约束为析取式
// 建图：对于约束 (a || b)，添加边 ~a -> b, ~b -> a
// 解法：求SCC，若x和~x在同一SCC则无解

#include <stdio.h>
#include <string.h>

// 变量编号：x 对应 2x，~x 对应 2x+1
int varToNode(int x, int neg) { return x * 2 + neg; }

void addClause(int a, int negA, int b, int negB) {
    // (a || b) 等价于 ~a -> b 且 ~b -> a
    addEdge(varToNode(a, !negA), varToNode(b, negB));
    addEdge(varToNode(b, !negB), varToNode(a, negA));
}

int solve2SAT(int n) {
    // 求SCC
    for (int i = 0; i < 2 * n; i++)
        if (!dfn[i]) tarjan(i);
    // 检查矛盾
    for (int i = 0; i < n; i++)
        if (scc[2 * i] == scc[2 * i + 1])
            return 0;  // 无解
    return 1;  // 有解
}
```

## 7. 连通性算法总结

| 算法 | 适用场景 | 时间复杂度 |
|------|---------|-----------|
| DFS/BFS | 无向图连通分量 | O(V + E) |
| Tarjan SCC | 有向图强连通分量 | O(V + E) |
| 割点判定 | 无向图割点 | O(V + E) |
| 割边判定 | 无向图割边（桥） | O(V + E) |
| 点双连通分量 | 无向图点双 | O(V + E) |
| 边双连通分量 | 无向图边双 | O(V + E) |
| Kosaraju SCC | 有向图强连通分量 | O(V + E) |
| 2-SAT | 布尔可满足性 | O(V + E) |
