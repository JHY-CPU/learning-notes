# 前缀和与差分

## 1. 一维前缀和

前缀和是一种预处理技术，可以在O(1)时间内求出区间和。

```c
#include <stdio.h>

#define MAXN 100005

int arr[MAXN], prefix[MAXN], n;

// 构建前缀和数组
void buildPrefixSum() {
    prefix[0] = 0;
    for (int i = 1; i <= n; i++)
        prefix[i] = prefix[i - 1] + arr[i];
}

// 查询区间 [l, r] 的和（1-indexed）
int querySum(int l, int r) {
    return prefix[r] - prefix[l - 1];
}

int main() {
    n = 5;
    int a[] = {0, 1, 2, 3, 4, 5};  // arr[1..5]
    for (int i = 1; i <= n; i++) arr[i] = a[i];
    buildPrefixSum();

    printf("区间[2,4]的和: %d\n", querySum(2, 4));  // 2+3+4=9
    printf("区间[1,5]的和: %d\n", querySum(1, 5));  // 15
    return 0;
}
```

## 2. 二维前缀和

二维前缀和可以在O(1)时间内查询子矩阵的和。

```c
#include <stdio.h>

#define MAXN 1005

int matrix[MAXN][MAXN], prefix2D[MAXN][MAXN];
int n, m;  // 行数、列数

// 构建二维前缀和
void build2DPrefixSum() {
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            prefix2D[i][j] = matrix[i][j]
                            + prefix2D[i-1][j]
                            + prefix2D[i][j-1]
                            - prefix2D[i-1][j-1];
}

// 查询子矩阵 [(r1,c1), (r2,c2)] 的和
int query2DSum(int r1, int c1, int r2, int c2) {
    return prefix2D[r2][c2]
         - prefix2D[r1-1][c2]
         - prefix2D[r2][c1-1]
         + prefix2D[r1-1][c1-1];
}

int main() {
    n = 3; m = 3;
    int mat[4][4] = {
        {0, 0, 0, 0},
        {0, 1, 2, 3},
        {0, 4, 5, 6},
        {0, 7, 8, 9}
    };
    for (int i = 1; i <= 3; i++)
        for (int j = 1; j <= 3; j++)
            matrix[i][j] = mat[i][j];

    build2DPrefixSum();
    printf("子矩阵[(1,1),(2,2)]的和: %d\n", query2DSum(1, 1, 2, 2));  // 1+2+4+5=12
    return 0;
}
```

## 3. 一维差分

差分是前缀和的逆运算，用于高效地对区间进行批量加减操作。

```c
#include <stdio.h>

#define MAXN 100005

int diff[MAXN], n;  // 差分数组

// 初始化差分数组
void initDiff(int arr[]) {
    for (int i = 1; i <= n; i++)
        diff[i] = arr[i] - arr[i - 1];
}

// 对区间 [l, r] 的每个元素加上 val
// 只需O(1)操作
void addRange(int l, int r, int val) {
    diff[l] += val;
    diff[r + 1] -= val;
}

// 还原原数组（对差分数组求前缀和）
void restore(int arr[]) {
    arr[0] = 0;
    for (int i = 1; i <= n; i++)
        arr[i] = arr[i - 1] + diff[i];
}

int main() {
    n = 5;
    int arr[MAXN] = {0, 1, 2, 3, 4, 5};
    initDiff(arr);

    // 区间[2,4]每个元素加10
    addRange(2, 4, 10);
    restore(arr);

    printf("操作后: ");
    for (int i = 1; i <= n; i++) printf("%d ", arr[i]);
    printf("\n");  // 输出: 1 12 13 14 5

    return 0;
}
```

## 4. 二维差分

```c
#include <stdio.h>

#define MAXN 1005

int diff2D[MAXN][MAXN], n, m;

// 对子矩阵 [(r1,c1), (r2,c2)] 的每个元素加上 val
void addRange2D(int r1, int c1, int r2, int c2, int val) {
    diff2D[r1][c1] += val;
    diff2D[r1][c2 + 1] -= val;
    diff2D[r2 + 1][c1] -= val;
    diff2D[r2 + 1][c2 + 1] += val;
}

// 还原原矩阵
void restore2D(int mat[MAXN][MAXN]) {
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            mat[i][j] = diff2D[i][j]
                       + mat[i-1][j]
                       + mat[i][j-1]
                       - mat[i-1][j-1];
}
```

## 5. 前缀积

```c
#include <stdio.h>

#define MAXN 100005

// 前缀积（不含位置i的积）
int productExceptSelf(int arr[], int n, int result[]) {
    // 左侧前缀积
    int left[MAXN];
    left[0] = 1;
    for (int i = 1; i < n; i++)
        left[i] = left[i - 1] * arr[i - 1];

    // 右侧前缀积
    int right = 1;
    for (int i = n - 1; i >= 0; i--) {
        result[i] = left[i] * right;
        right *= arr[i];
    }
    return 0;
}
```

## 6. 前缀异或

```c
#include <stdio.h>

// 前缀异或
int prefixXOR[MAXN];

void buildPrefixXOR(int arr[], int n) {
    prefixXOR[0] = 0;
    for (int i = 1; i <= n; i++)
        prefixXOR[i] = prefixXOR[i - 1] ^ arr[i];
}

// 查询区间 [l, r] 的异或和
int queryXOR(int l, int r) {
    return prefixXOR[r] ^ prefixXOR[l - 1];
}

// 应用：子数组异或和为0的最长子数组
int longestSubarrayXORZero(int arr[], int n) {
    int prefixXOR[100005] = {0};
    int hash[200005];  // 存储某异或值首次出现的位置
    for (int i = 0; i < 200005; i++) hash[i] = -1;

    int maxLen = 0;
    hash[100000] = 0;  // prefixXOR[0] = 0 对应位置0
    for (int i = 1; i <= n; i++) {
        prefixXOR[i] = prefixXOR[i - 1] ^ arr[i];
        int idx = prefixXOR[i] + 100000;
        if (hash[idx] != -1)
            maxLen = (i - hash[idx] > maxLen) ? i - hash[idx] : maxLen;
        else
            hash[idx] = i;
    }
    return maxLen;
}
```

## 7. 应用场景总结

| 技巧 | 预处理 | 单次查询/修改 | 典型应用 |
|------|-------|-------------|---------|
| 一维前缀和 | O(n) | O(1) | 区间和查询 |
| 二维前缀和 | O(n*m) | O(1) | 子矩阵和查询 |
| 一维差分 | O(n) | O(1) | 区间加减 |
| 二维差分 | O(n*m) | O(1) | 子矩阵加减 |
| 前缀积 | O(n) | O(1) | 除自身外的积 |
| 前缀异或 | O(n) | O(1) | 区间异或查询 |

```c
// 经典应用：和为K的子数组个数
int subarraySumK(int arr[], int n, int k) {
    // 利用前缀和 + 哈希表
    int prefix = 0, count = 0;
    int hash[2000005] = {0};  // 前缀和 -> 出现次数
    hash[1000000] = 1;  // prefix = 0 出现1次

    for (int i = 0; i < n; i++) {
        prefix += arr[i];
        int target = prefix - k + 1000000;
        if (target >= 0 && target < 2000005)
            count += hash[target];
        if (prefix + 1000000 >= 0 && prefix + 1000000 < 2000005)
            hash[prefix + 1000000]++;
    }
    return count;
}
```
