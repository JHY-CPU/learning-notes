# 双指针技巧

## 1. 双指针概述

双指针是一种高效的数组/链表遍历技术，通过维护两个指针在序列上移动，将暴力的O(n^2)优化到O(n)。常见的双指针模式有：
- **快慢指针**：同向移动，速度不同
- **对撞指针**：从两端向中间移动
- **滑动窗口**：维护一个可变长度的窗口

## 2. 快慢指针

### 判断链表是否有环

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int val;
    struct Node *next;
} Node;

// 快慢指针判环
int hasCycle(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) return 1;  // 有环
    }
    return 0;
}

// 找环的入口
Node *detectCycle(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) {
            // 找入口：一个从head出发，一个从相遇点出发
            Node *ptr = head;
            while (ptr != slow) {
                ptr = ptr->next;
                slow = slow->next;
            }
            return ptr;
        }
    }
    return NULL;
}

// 找链表中点
Node *findMiddle(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;  // fast到末尾时slow在中点
}
```

### 删除有序数组的重复元素

```c
int removeDuplicates(int arr[], int n) {
    if (n == 0) return 0;
    int slow = 0;
    for (int fast = 1; fast < n; fast++) {
        if (arr[fast] != arr[slow]) {
            slow++;
            arr[slow] = arr[fast];
        }
    }
    return slow + 1;  // 新长度
}
```

## 3. 对撞指针（Two Pointer）

### 两数之和（有序数组）

```c
#include <stdio.h>

// 在有序数组中找两数之和等于target
int *twoSum(int arr[], int n, int target) {
    static int result[2];
    int left = 0, right = n - 1;
    while (left < right) {
        int sum = arr[left] + arr[right];
        if (sum == target) {
            result[0] = left; result[1] = right;
            return result;
        }
        if (sum < target) left++;
        else right--;
    }
    return NULL;  // 未找到
}
```

### 判断回文字符串

```c
#include <string.h>

int isPalindrome(const char *s) {
    int left = 0, right = strlen(s) - 1;
    while (left < right) {
        if (s[left] != s[right]) return 0;
        left++; right--;
    }
    return 1;
}
```

### 盛最多水的容器

```c
int maxArea(int height[], int n) {
    int left = 0, right = n - 1;
    int maxArea = 0;
    while (left < right) {
        int area = (right - left) *
                   (height[left] < height[right] ? height[left] : height[right]);
        if (area > maxArea) maxArea = area;
        if (height[left] < height[right]) left++;
        else right--;
    }
    return maxArea;
}
```

### 三数之和

```c
#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) { return *(int *)a - *(int *)b; }

void threeSum(int arr[], int n) {
    qsort(arr, n, sizeof(int), cmp);
    for (int i = 0; i < n - 2; i++) {
        if (i > 0 && arr[i] == arr[i - 1]) continue;  // 去重
        int left = i + 1, right = n - 1;
        int target = -arr[i];
        while (left < right) {
            int sum = arr[left] + arr[right];
            if (sum == target) {
                printf("[%d, %d, %d]\n", arr[i], arr[left], arr[right]);
                while (left < right && arr[left] == arr[left + 1]) left++;
                while (left < right && arr[right] == arr[right - 1]) right--;
                left++; right--;
            } else if (sum < target) left++;
            else right--;
        }
    }
}
```

## 4. 滑动窗口

### 最小覆盖子串

```c
#include <stdio.h>
#include <string.h>
#include <limits.h>

// 找包含t所有字符的最短子串
void minWindow(const char *s, const char *t) {
    int need[256] = {0}, window[256] = {0};
    int tlen = strlen(t);
    for (int i = 0; i < tlen; i++) need[(unsigned char)t[i]]++;

    int left = 0, right = 0, count = 0;
    int minLen = INT_MAX, start = 0;

    while (s[right]) {
        char c = s[right];
        if (need[(unsigned char)c] > 0) {
            window[(unsigned char)c]++;
            if (window[(unsigned char)c] <= need[(unsigned char)c])
                count++;
        }
        right++;

        // 收缩左边界
        while (count == tlen) {
            if (right - left < minLen) {
                minLen = right - left;
                start = left;
            }
            char d = s[left];
            if (need[(unsigned char)d] > 0) {
                window[(unsigned char)d]--;
                if (window[(unsigned char)d] < need[(unsigned char)d])
                    count--;
            }
            left++;
        }
    }
    if (minLen != INT_MAX)
        printf("最短子串: %.*s\n", minLen, s + start);
}
```

### 长度最小的子数组（和>=target）

```c
int minSubArrayLen(int target, int nums[], int n) {
    int left = 0, sum = 0, minLen = INT_MAX;
    for (int right = 0; right < n; right++) {
        sum += nums[right];
        while (sum >= target) {
            int len = right - left + 1;
            if (len < minLen) minLen = len;
            sum -= nums[left++];
        }
    }
    return minLen == INT_MAX ? 0 : minLen;
}
```

## 5. 滑动窗口最大值

```c
#include <stdio.h>

// 使用单调队列求滑动窗口最大值
void slidingWindowMax(int arr[], int n, int k) {
    int deque[100005];  // 存下标
    int front = 0, rear = 0;

    for (int i = 0; i < n; i++) {
        // 移除超出窗口的元素
        while (front < rear && deque[front] < i - k + 1) front++;
        // 维护单调递减
        while (front < rear && arr[deque[rear - 1]] < arr[i]) rear--;
        deque[rear++] = i;
        if (i >= k - 1) printf("%d ", arr[deque[front]]);
    }
    printf("\n");
}
```
