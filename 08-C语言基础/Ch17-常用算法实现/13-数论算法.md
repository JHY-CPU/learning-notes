# 数论算法

## 1. 素数判定与筛法

### 试除法判定素数 O(sqrt(n))

```c
#include <stdio.h>
#include <stdbool.h>
#include <math.h>

bool isPrime(int n) {
    if (n < 2) return false;
    if (n < 4) return true;
    if (n % 2 == 0 || n % 3 == 0) return false;
    for (int i = 5; i <= (int)sqrt(n); i += 6)
        if (n % i == 0 || n % (i + 2) == 0) return false;
    return true;
}
```

### 埃拉托斯特尼筛法 O(n log log n)

```c
#include <stdio.h>
#include <stdbool.h>
#include <string.h>

#define MAXN 10000005

bool notPrime[MAXN];
int primes[MAXN], primeCount;

void sieve(int n) {
    memset(notPrime, false, sizeof(notPrime));
    notPrime[0] = notPrime[1] = true;
    primeCount = 0;
    for (int i = 2; i <= n; i++) {
        if (!notPrime[i]) {
            primes[primeCount++] = i;
            if ((long long)i * i <= n)
                for (int j = i * i; j <= n; j += i)
                    notPrime[j] = true;
        }
    }
}
```

### 欧拉筛（线性筛）O(n)

```c
void eulerSieve(int n) {
    memset(notPrime, false, sizeof(notPrime));
    primeCount = 0;
    for (int i = 2; i <= n; i++) {
        if (!notPrime[i]) primes[primeCount++] = i;
        for (int j = 0; j < primeCount && (long long)i * primes[j] <= n; j++) {
            notPrime[i * primes[j]] = true;
            if (i % primes[j] == 0) break;  // 关键：保证每个数只被最小质因子筛掉
        }
    }
}
```

## 2. GCD与LCM

```c
// 辗转相除法（欧几里得算法）
int gcd(int a, int b) {
    return b == 0 ? a : gcd(b, a % b);
}

// 迭代版
int gcd_iter(int a, int b) {
    while (b) { int t = b; b = a % b; a = t; }
    return a;
}

int lcm(int a, int b) {
    return a / gcd(a, b) * b;  // 先除后乘防溢出
}

// 扩展欧几里得算法：求 ax + by = gcd(a, b) 的整数解
int exgcd(int a, int b, int *x, int *y) {
    if (b == 0) { *x = 1; *y = 0; return a; }
    int g = exgcd(b, a % b, y, x);
    *y -= (a / b) * (*x);
    return g;
}
```

## 3. 快速幂

```c
#include <stdio.h>

// 计算 base^exp % mod
long long fastPow(long long base, long long exp, long long mod) {
    long long result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1)
            result = result * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return result;
}

// 快速幂取模用于大数运算
int main() {
    // 计算 2^1000000000 % 1000000007
    printf("2^1000000000 mod 1e9+7 = %lld\n",
           fastPow(2, 1000000000LL, 1000000007LL));
    return 0;
}
```

## 4. 模运算

```c
#define MOD 1000000007

// 模加法
long long modAdd(long long a, long long b) {
    return ((a % MOD) + (b % MOD)) % MOD;
}

// 模乘法
long long modMul(long long a, long long b) {
    return ((a % MOD) * (b % MOD)) % MOD;
}

// 模减法
long long modSub(long long a, long long b) {
    return ((a % MOD) - (b % MOD) + MOD) % MOD;
}
```

## 5. 逆元

### 费马小定理求逆元（mod为素数）

```c
// 若p为素数，则 a^(p-1) ≡ 1 (mod p)
// 所以 a 的逆元为 a^(p-2) mod p
long long modInverse(long long a, long long mod) {
    return fastPow(a, mod - 2, mod);
}

// 除法取模: (a / b) % MOD = a * b^(-1) % MOD
long long modDiv(long long a, long long b) {
    return modMul(a, modInverse(b, MOD));
}
```

### 扩展欧几里得求逆元

```c
int modInverseExgcd(int a, int mod) {
    int x, y;
    int g = exgcd(a, mod, &x, &y);
    if (g != 1) return -1;  // 逆元不存在
    return (x % mod + mod) % mod;
}
```

## 6. 质因数分解

```c
#include <stdio.h>

typedef struct { int p, cnt; } Factor;

int factorize(int n, Factor factors[]) {
    int count = 0;
    for (int i = 2; (long long)i * i <= n; i++) {
        if (n % i == 0) {
            factors[count].p = i;
            factors[count].cnt = 0;
            while (n % i == 0) {
                factors[count].cnt++;
                n /= i;
            }
            count++;
        }
    }
    if (n > 1) {
        factors[count].p = n;
        factors[count].cnt = 1;
        count++;
    }
    return count;
}
```

## 7. 中国剩余定理（CRT）

```c
// 求解同余方程组:
// x ≡ r1 (mod m1)
// x ≡ r2 (mod m2)
// ...
long long chineseRemainder(int r[], int m[], int n) {
    long long M = 1, x = 0;
    for (int i = 0; i < n; i++) M *= m[i];
    for (int i = 0; i < n; i++) {
        long long Mi = M / m[i];
        int xi, yi;
        exgcd(Mi, m[i], &xi, &yi);
        x = (x + r[i] * Mi % M * xi % M) % M;
    }
    return (x % M + M) % M;
}
```

## 8. 欧拉函数

```c
// φ(n): 小于n且与n互质的正整数个数
int eulerPhi(int n) {
    int result = n;
    for (int i = 2; (long long)i * i <= n; i++) {
        if (n % i == 0) {
            result -= result / i;
            while (n % i == 0) n /= i;
        }
    }
    if (n > 1) result -= result / n;
    return result;
}

// 欧拉筛法线性求phi[1..n]
int phi[MAXN];
void eulerPhiSieve(int n) {
    for (int i = 1; i <= n; i++) phi[i] = i;
    for (int i = 2; i <= n; i++) {
        if (phi[i] == i) {  // i是素数
            for (int j = i; j <= n; j += i)
                phi[j] -= phi[j] / i;
        }
    }
}
```
