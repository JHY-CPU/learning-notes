# 二分查找变体

## 1. 标准二分查找

```c
#include <stdio.h>

// 在有序数组中查找目标值，返回下标（-1表示不存在）
int binarySearch(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}
```

## 2. 查找第一个等于目标的位置（左边界）

```c
int lowerBound(int arr[], int n, int target) {
    int left = 0, right = n;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] < target) left = mid + 1;
        else right = mid;
    }
    return left;  // 若不存在，返回插入位置
}
```

## 3. 查找最后一个等于目标的位置（右边界）

```c
int upperBound(int arr[], int n, int target) {
    int left = 0, right = n;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] <= target) left = mid + 1;
        else right = mid;
    }
    return left - 1;  // 最后一个<=target的位置
}

// 统计target出现次数
int countOccurrences(int arr[], int n, int target) {
    int left = lowerBound(arr, n, target);
    int right = upperBound(arr, n, target);
    if (left <= right) return right - left + 1;
    return 0;
}
```

## 4. 旋转排序数组搜索

### 搜索旋转数组中的目标值（无重复）

```c
int searchRotated(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target) return mid;
        // 左半部分有序
        if (arr[left] <= arr[mid]) {
            if (target >= arr[left] && target < arr[mid])
                right = mid - 1;
            else
                left = mid + 1;
        }
        // 右半部分有序
        else {
            if (target > arr[mid] && target <= arr[right])
                left = mid + 1;
            else
                right = mid - 1;
        }
    }
    return -1;
}
```

### 旋转数组中的最小值

```c
int findMin(int arr[], int n) {
    int left = 0, right = n - 1;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] > arr[right])
            left = mid + 1;  // 最小值在右半部分
        else
            right = mid;     // 最小值在左半部分（含mid）
    }
    return arr[left];
}
```

## 5. 二分答案

很多最大化最小值、最小化最大值的问题可以用二分答案法。

### 猜数字大小（猜答案）

```c
// 题目：将n个物品分成k组，使最大组的和最小
// 二分答案：二分组的最大和

int canSplit(int arr[], int n, int k, int maxSum) {
    int groups = 1, sum = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > maxSum) return 0;  // 单个元素超过上限
        if (sum + arr[i] > maxSum) {
            groups++;
            sum = arr[i];
        } else {
            sum += arr[i];
        }
    }
    return groups <= k;
}

int splitArray(int arr[], int n, int k) {
    int lo = 0, hi = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > lo) lo = arr[i];  // 最小答案至少是最大的单个元素
        hi += arr[i];                    // 最大答案是所有元素之和
    }
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (canSplit(arr, n, k, mid))
            hi = mid;
        else
            lo = mid + 1;
    }
    return lo;
}
```

### 木头切割问题

```c
// 切割木头：每段长度>=mid时最多能切多少段
int canCut(int arr[], int n, int mid) {
    int count = 0;
    for (int i = 0; i < n; i++)
        count += arr[i] / mid;
    return count;
}

// 找最大切割长度使得至少有k段
int maxCutLength(int arr[], int n, int k) {
    int lo = 1, hi = 0;
    for (int i = 0; i < n; i++)
        if (arr[i] > hi) hi = arr[i];

    while (lo < hi) {
        int mid = lo + (hi - lo + 1) / 2;  // 注意用上取整避免死循环
        if (canCut(arr, n, mid) >= k)
            lo = mid;
        else
            hi = mid - 1;
    }
    return lo;
}
```

## 6. 浮点数二分

```c
#include <stdio.h>
#include <math.h>

// 求立方根
double cubeRoot(double n) {
    double lo = 0, hi = fmax(1.0, fabs(n));
    while (hi - lo > 1e-9) {
        double mid = (lo + hi) / 2;
        if (mid * mid * mid < fabs(n))
            lo = mid;
        else
            hi = mid;
    }
    return n < 0 ? -lo : lo;
}
```

## 7. 二分查找注意事项

```c
// 1. 防止溢出
int mid = left + (right - left) / 2;  // 不要用 (left + right) / 2

// 2. 循环终止条件的选择
// while (left <= right)  ->  最终 left = right + 1
// while (left < right)   ->  最终 left = right

// 3. 左右边界更新
// 使用 left < right 时，注意 mid 是否会被排除
// lo = mid 时需要 mid = lo + (hi - lo + 1) / 2（上取整）避免死循环
// hi = mid 时使用 mid = lo + (hi - lo) / 2（下取整）

// 4. 二分答案时注意检查方向
// 最小化 -> check(mid)为真时 hi = mid
// 最大化 -> check(mid)为真时 lo = mid
```
