## 1. 图的存储方式

### 邻接矩阵


### 邻接表

```c
int head[MAXN], to[MAXM], w[MAXM], nxt[MAXM], edge_cnt = 0;
void addEdge(int u, int v, int weight) {
    to[edge_cnt] = v; w[edge_cnt] = weight;
    nxt[edge_cnt] = head[u]; head[u] = edge_cnt++;
}
```

## 2. Dijkstra最短路（单源）

适用于边权非负的有向/无向图，时间复杂度 O(V^2) 或 O(E log V)。

```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>
#include <limits.h>

#define MAXN 1005

int dist[MAXN];
bool visited[MAXN];
int graph[MAXN][MAXN], n;

// O(V^2) 基础版
void dijkstra(int src) {
    for (int i = 0; i < n; i++) { dist[i] = INT_MAX; visited[i] = false; }
    dist[src] = 0;

    for (int i = 0; i < n; i++) {
        // 找最近的未访问节点
        int u = -1, minDist = INT_MAX;
        for (int j = 0; j < n; j++)
            if (!visited[j] && dist[j] < minDist)
                { minDist = dist[j]; u = j; }
        if (u == -1) break;
        visited[u] = true;

        // 松弛操作
        for (int v = 0; v < n; v++) {
            if (!visited[v] && graph[u][v] != INT_MAX
                && dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }
}
```

## 3. Bellman-Ford算法

支持负权边，可检测负权环。时间复杂度 O(VE)。

```c
#include <stdio.h>
#include <limits.h>

typedef struct { int u, v, w; } Edge;
Edge edges[MAXM];
int dist[MAXN];
int n, m;  // 顶点数、边数

bool bellmanFord(int src) {
    for (int i = 0; i < n; i++) dist[i] = INT_MAX;
    dist[src] = 0;

    for (int i = 0; i < n - 1; i++) {
        bool updated = false;
        for (int j = 0; j < m; j++) {
            int u = edges[j].u, v = edges[j].v, w = edges[j].w;
            if (dist[u] != INT_MAX && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                updated = true;
            }
        }
        if (!updated) break;  // 优化：无更新则提前结束
    }

    // 检测负权环
    for (int j = 0; j < m; j++) {
        int u = edges[j].u, v = edges[j].v, w = edges[j].w;
        if (dist[u] != INT_MAX && dist[u] + w < dist[v])
            return false;  // 存在负权环
    }
    return true;
}
```

## 4. SPFA算法（队列优化的Bellman-Ford）

```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>
#include <limits.h>

#define MAXN 10005

int head[MAXN], to[MAXN], w[MAXN], nxt[MAXN], edge_cnt = 0;
int dist[MAXN], cnt[MAXN];
bool inQueue[MAXN];
int queue[MAXN * 10];

bool spfa(int src, int n) {
    for (int i = 0; i < n; i++) dist[i] = INT_MAX;
    int front = 0, rear = 0;
    dist[src] = 0;
    queue[rear++] = src;
    inQueue[src] = true;

    while (front < rear) {
        int u = queue[front++];
        inQueue[u] = false;
        for (int i = head[u]; i != -1; i = nxt[i]) {
            int v = to[i];
            if (dist[u] != INT_MAX && dist[u] + w[i] < dist[v]) {
                dist[v] = dist[u] + w[i];
                cnt[v] = cnt[u] + 1;
                if (cnt[v] >= n) return false;  // 负权环
                if (!inQueue[v]) {
                    queue[rear++] = v;
                    inQueue[v] = true;
                }
            }
        }
    }
    return true;
}
```

## 5. Floyd-Warshall算法（全源最短路）

```c
#include <stdio.h>
#include <string.h>

#define MAXN 505
#define INF 0x3f3f3f3f

int dist[MAXN][MAXN];
int n;

void floyd() {
    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (dist[i][k] + dist[k][j] < dist[i][j])
                    dist[i][j] = dist[i][k] + dist[k][j];
}
```

## 6. 最小生成树——Prim算法

```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>
#include <limits.h>

#define MAXN 5005

int graph[MAXN][MAXN], n;
int key[MAXN];
bool inMST[MAXN];

int prim() {
    for (int i = 0; i < n; i++) { key[i] = INT_MAX; inMST[i] = false; }
    key[0] = 0;
    int total = 0;

    for (int i = 0; i < n; i++) {
        int u = -1, minKey = INT_MAX;
        for (int j = 0; j < n; j++)
            if (!inMST[j] && key[j] < minKey)
                { minKey = key[j]; u = j; }
        if (u == -1) return -1;  // 不连通
        inMST[u] = true;
        total += key[u];

        for (int v = 0; v < n; v++)
            if (!inMST[v] && graph[u][v] < key[v])
                key[v] = graph[u][v];
    }
    return total;
}
```

## 7. 最大流

### Ford-Fulkerson / Edmonds-Karp

```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>
#include <limits.h>

#define MAXN 205

int capacity[MAXN][MAXN], flow[MAXN][MAXN];
int parent[MAXN];
int n;

int bfsMaxFlow(int s, int t) {
    memset(parent, -1, sizeof(parent));
    int queue[MAXN], front = 0, rear = 0;
    queue[rear++] = s;
    parent[s] = s;

    while (front < rear) {
        int u = queue[front++];
        for (int v = 0; v < n; v++) {
            if (parent[v] == -1 && capacity[u][v] - flow[u][v] > 0) {
                parent[v] = u;
                if (v == t) {
                    // 找到增广路径，计算瓶颈
                    int bottleneck = INT_MAX;
                    for (int cur = t; cur != s; cur = parent[cur])
                        if (capacity[parent[cur]][cur] - flow[parent[cur]][cur] < bottleneck)
                            bottleneck = capacity[parent[cur]][cur] - flow[parent[cur]][cur];
                    // 更新流量
                    for (int cur = t; cur != s; cur = parent[cur]) {
                        flow[parent[cur]][cur] += bottleneck;
                        flow[cur][parent[cur]] -= bottleneck;
                    }
                    return bottleneck;
                }
                queue[rear++] = v;
            }
        }
    }
    return 0;
}

int maxFlow(int s, int t) {
    memset(flow, 0, sizeof(flow));
    int total = 0, f;
    while ((f = bfsMaxFlow(s, t)) > 0)
        total += f;
    return total;
}
```
