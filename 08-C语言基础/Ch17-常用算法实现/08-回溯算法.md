# 回溯算法

## 1. 回溯思想

回溯法是一种通过探索所有可能的候选解来找出所有解或最优解的算法。当探索到某一步发现当前选择不能得到有效解时，就回退（回溯）到上一步，尝试其他选择。

### 回溯的核心框架

```c
void backtrack(路径, 选择列表) {
    if (满足结束条件) {
        记录结果;
        return;
    }
    for (选择 in 选择列表) {
        做出选择;           // 将选择加入路径
        backtrack(路径, 新的选择列表);
        撤销选择;           // 回溯，移除路径中最后的选择
    }
}
```

## 2. 全排列问题

生成数组的所有排列。

```c
#include <stdio.h>

int path[10], used[10], n;

void printPermutation(int depth) {
    if (depth == n) {
        for (int i = 0; i < n; i++)
            printf("%d ", path[i]);
        printf("\n");
        return;
    }
    for (int i = 0; i < n; i++) {
        if (!used[i]) {
            used[i] = 1;             // 标记已使用
            path[depth] = i + 1;
            printPermutation(depth + 1);
            used[i] = 0;             // 回溯：恢复未使用
        }
    }
}

int main() {
    n = 3;
    printPermutation(0);
    return 0;
}
```

### 有重复元素的全排列

```c
#include <stdio.h>
#include <stdlib.h>

int path2[10], used2[10];
int nums[10], n2;

int cmp(const void *a, const void *b) { return *(int *)a - *(int *)b; }

void permuteUnique(int depth) {
    if (depth == n2) {
        for (int i = 0; i < n2; i++) printf("%d ", path2[i]);
        printf("\n");
        return;
    }
    for (int i = 0; i < n2; i++) {
        if (used2[i]) continue;
        // 去重：相同元素只取第一个未使用的
        if (i > 0 && nums[i] == nums[i-1] && !used2[i-1]) continue;
        used2[i] = 1;
        path2[depth] = nums[i];
        permuteUnique(depth + 1);
        used2[i] = 0;
    }
}
```

## 3. 组合问题

从n个数中选k个的所有组合。

```c
#include <stdio.h>

int combPath[10], n, k;

void combine(int start, int depth) {
    if (depth == k) {
        for (int i = 0; i < k; i++)
            printf("%d ", combPath[i]);
        printf("\n");
        return;
    }
    for (int i = start; i <= n; i++) {
        combPath[depth] = i;
        combine(i + 1, depth + 1);  // 从i+1开始，避免重复
    }
}

int main() {
    n = 4; k = 2;
    combine(1, 0);
    return 0;
}
```

## 4. 子集生成

生成所有子集（幂集）。

```c
#include <stdio.h>

int subsetPath[10], subsetLen;

void generateSubsets(int arr[], int n, int start) {
    // 打印当前子集
    printf("{ ");
    for (int i = 0; i < subsetLen; i++)
        printf("%d ", subsetPath[i]);
    printf("}\n");

    for (int i = start; i < n; i++) {
        subsetPath[subsetLen++] = arr[i];
        generateSubsets(arr, n, i + 1);
        subsetLen--;  // 回溯
    }
}

int main() {
    int arr[] = {1, 2, 3};
    generateSubsets(arr, 3, 0);
    return 0;
}
```

## 5. N皇后问题

在n×n的棋盘上放置n个皇后，使其互不攻击。

```c
#include <stdio.h>
#include <stdlib.h>

int queenCol[20];  // queenCol[i] = j 表示第i行皇后放在第j列
int total = 0;

int isValid(int row, int col) {
    for (int i = 0; i < row; i++) {
        if (queenCol[i] == col) return 0;                // 同列
        if (abs(queenCol[i] - col) == abs(i - row)) return 0;  // 对角线
    }
    return 1;
}

void solveNQueens(int row, int n) {
    if (row == n) {
        total++;
        // 打印棋盘
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++)
                printf("%c ", j == queenCol[i] ? 'Q' : '.');
            printf("\n");
        }
        printf("\n");
        return;
    }
    for (int col = 0; col < n; col++) {
        if (isValid(row, col)) {
            queenCol[row] = col;           // 放置皇后
            solveNQueens(row + 1, n);
            queenCol[row] = -1;            // 回溯
        }
    }
}

int main() {
    int n = 8;
    solveNQueens(0, n);
    printf("%d皇后的解法总数: %d\n", n, total);
    return 0;
}
```

## 6. 括号生成

生成n对括号的所有有效组合。

```c
#include <stdio.h>

char path[20];

void generateParenthesis(int pos, int n, int open, int close) {
    if (pos == 2 * n) {
        path[pos] = '\0';
        printf("%s\n", path);
        return;
    }
    // 可以放 '(' 的条件：已放的 '(' 数量 < n
    if (open < n) {
        path[pos] = '(';
        generateParenthesis(pos + 1, n, open + 1, close);
    }
    // 可以放 ')' 的条件：已放的 ')' 数量 < '(' 数量
    if (close < open) {
        path[pos] = ')';
        generateParenthesis(pos + 1, n, open, close + 1);
    }
}

int main() {
    generateParenthesis(0, 3, 0, 0);
    return 0;
}
```

## 7. 回溯算法的通用模式

```c
// 通用回溯框架
int solution[MAXN];
int solutionSize;

void backtrack(int state) {
    if (isSolution(state)) {
        processSolution(solution, solutionSize);
        return;
    }
    int candidates[MAX_CANDIDATES];
    int numCandidates = getCandidates(state, candidates);

    for (int i = 0; i < numCandidates; i++) {
        // 做选择
        solution[solutionSize++] = candidates[i];
        int newState = applyChoice(state, candidates[i]);
        // 递归
        backtrack(newState);
        // 撤销选择
        solutionSize--;
    }
}
```
