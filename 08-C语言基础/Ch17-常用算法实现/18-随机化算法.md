# 随机化算法

## 1. 随机化算法概述

随机化算法在执行过程中引入随机因素，可分为两类：
- **蒙特卡罗算法**：可能给出错误结果，但出错概率可控
- **拉斯维加斯算法**：总是给出正确结果，但运行时间随机

## 2. 随机数生成

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// 初始化随机种子
void initRandom() {
    srand((unsigned)time(NULL));
}

// 生成 [a, b] 范围内的随机整数
int randInt(int a, int b) {
    return a + rand() % (b - a + 1);
}

// 生成 [0, 1) 范围内的随机浮点数
double randDouble() {
    return (double)rand() / ((double)RAND_MAX + 1.0);
}

// 生成 [a, b) 范围内的随机浮点数
double randRange(double a, double b) {
    return a + (b - a) * randDouble();
}
```

## 3. 随机快速排序

随机选择基准元素，避免最坏情况。

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void swap(int *a, int *b) {
    int t = *a; *a = *b; *b = t;
}

int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (arr[j] <= pivot) {
            i++;
            swap(&arr[i], &arr[j]);
        }
    }
    swap(&arr[i + 1], &arr[high]);
    return i + 1;
}

// 随机化分区
int randomizedPartition(int arr[], int low, int high) {
    int r = low + rand() % (high - low + 1);
    swap(&arr[r], &arr[high]);
    return partition(arr, low, high);
}

void randomizedQuickSort(int arr[], int low, int high) {
    if (low < high) {
        int pi = randomizedPartition(arr, low, high);
        randomizedQuickSort(arr, low, pi - 1);
        randomizedQuickSort(arr, pi + 1, high);
    }
}
```

## 4. 随机选择算法（第k小元素）

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// 随机选择第k小的元素（0-indexed）
int randomizedSelect(int arr[], int low, int high, int k) {
    if (low == high) return arr[low];

    int pi = randomizedPartition(arr, low, high);
    int rank = pi - low;  // pi在子数组中的排名

    if (k == rank)
        return arr[pi];
    else if (k < rank)
        return randomizedSelect(arr, low, pi - 1, k);
    else
        return randomizedSelect(arr, pi + 1, high, k - rank - 1);
}

int main() {
    srand(time(NULL));
    int arr[] = {3, 2, 1, 5, 4, 6};
    int n = 6, k = 2;  // 第3小
    printf("第%d小的元素: %d\n", k + 1,
           randomizedSelect(arr, 0, n - 1, k));
    return 0;
}
```

## 5. 蒙特卡罗方法求圆周率

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

double estimatePi(int samples) {
    int inside = 0;
    for (int i = 0; i < samples; i++) {
        double x = (double)rand() / RAND_MAX * 2.0 - 1.0;
        double y = (double)rand() / RAND_MAX * 2.0 - 1.0;
        if (x * x + y * y <= 1.0) inside++;
    }
    return 4.0 * inside / samples;
}

int main() {
    srand(time(NULL));
    int samples[] = {1000, 10000, 100000, 1000000, 10000000};
    for (int i = 0; i < 5; i++) {
        double pi = estimatePi(samples[i]);
        printf("样本数: %d0, 估算π: %.6f, 误差: %.6f\n",
               samples[i], pi, fabs(pi - 3.14159265358979));
    }
    return 0;
}
```

## 6. 蒙特卡罗求定积分

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

// 被积函数
double f(double x) {
    return x * x;  // 示例：计算 x^2 在 [0,1] 的积分
}

// 蒙特卡罗积分
double monteCarloIntegrate(double (*func)(double), double a, double b, int n) {
    double sum = 0;
    for (int i = 0; i < n; i++) {
        double x = a + (b - a) * rand() / RAND_MAX;
        sum += func(x);
    }
    return (b - a) * sum / n;
}

int main() {
    srand(time(NULL));
    double result = monteCarloIntegrate(f, 0.0, 1.0, 10000000);
    printf("积分结果: %.6f (精确值: 0.333333)\n", result);
    return 0;
}
```

## 7. 随机化素数测试（Miller-Rabin）

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

long long mulmod(long long a, long long b, long long mod) {
    long long res = 0;
    a %= mod;
    while (b > 0) {
        if (b & 1) res = (res + a) % mod;
        a = (a * a) % mod;
        b >>= 1;
    }
    return res;
}

long long power(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) res = mulmod(res, base, mod);
        base = mulmod(base, base, mod);
        exp >>= 1;
    }
    return res;
}

// Miller-Rabin素性测试
int millerTest(long long d, long long n) {
    long long a = 2 + rand() % (n - 4);
    long long x = power(a, d, n);
    if (x == 1 || x == n - 1) return 1;
    while (d != n - 1) {
        x = mulmod(x, x, n);
        d *= 2;
        if (x == 1) return 0;
        if (x == n - 1) return 1;
    }
    return 0;
}

int isPrime(long long n, int k) {
    if (n <= 1 || n == 4) return 0;
    if (n <= 3) return 1;
    long long d = n - 1;
    while (d % 2 == 0) d /= 2;
    for (int i = 0; i < k; i++)
        if (!millerTest(d, n)) return 0;
    return 1;
}
```

## 8. 随机化算法总结

| 算法 | 类型 | 时间复杂度 | 正确性 |
|------|------|-----------|--------|
| 随机快排 | 拉斯维加斯 | 平均O(n log n) | 始终正确 |
| 随机选择 | 拉斯维加斯 | 平均O(n) | 始终正确 |
| Miller-Rabin | 蒙特卡罗 | O(k log n) | 可能出错(概率极低) |
| 蒙特卡罗积分 | 蒙特卡罗 | O(n) | 估计值 |
