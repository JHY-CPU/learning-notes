# 网络流基础

## 1. 网络流基本概念

- **网络**：有向图，每条边有容量上限
- **流**：满足容量限制和流量守恒的边上的流量分配
- **最大流**：从源点到汇点的最大流量
- **最小割**：将源汇分开的最小容量边集

## 2. Ford-Fulkerson算法

通过不断寻找增广路径来增加流量，直到不存在增广路径。

```c
#include <stdio.h>
#include <string.h>
#include <limits.h>

#define MAXN 205
#define min(a, b) ((a) < (b) ? (a) : (b))

int cap[MAXN][MAXN];  // 容量
int flow[MAXN][MAXN]; // 流量
int parent[MAXN];
int n;  // 节点数

// BFS找增广路径（Edmonds-Karp）
int bfs(int s, int t) {
    memset(parent, -1, sizeof(parent));
    int queue[MAXN], front = 0, rear = 0;
    queue[rear++] = s;
    parent[s] = s;

    while (front < rear) {
        int u = queue[front++];
        for (int v = 0; v < n; v++) {
            if (parent[v] == -1 && cap[u][v] - flow[u][v] > 0) {
                parent[v] = u;
                if (v == t) {
                    // 计算增广路径的瓶颈容量
                    int bottleneck = INT_MAX;
                    for (int cur = t; cur != s; cur = parent[cur])
                        bottleneck = min(bottleneck, cap[parent[cur]][cur] - flow[parent[cur]][cur]);
                    // 更新流量
                    for (int cur = t; cur != s; cur = parent[cur]) {
                        flow[parent[cur]][cur] += bottleneck;
                        flow[cur][parent[cur]] -= bottleneck;
                    }
                    return bottleneck;
                }
                queue[rear++] = v;
            }
        }
    }
    return 0;
}

int maxFlow(int s, int t) {
    memset(flow, 0, sizeof(flow));
    int total = 0, f;
    while ((f = bfs(s, t)) > 0)
        total += f;
    return total;
}
```

## 3. Dinic算法

通过分层图和当前弧优化加速。

```c
#include <stdio.h>
#include <string.h>
#include <limits.h>

#define MAXN 10005
#define MAXM 200005

int headDinic[MAXN], toDinic[MAXM], capDinic[MAXM], nxtDinic[MAXM], edgeDinic = 0;
int level[MAXN], cur[MAXN];  // level: 分层, cur: 当前弧

void addEdgeDinic(int u, int v, int c) {
    toDinic[edgeDinic] = v; capDinic[edgeDinic] = c;
    nxtDinic[edgeDinic] = headDinic[u]; headDinic[u] = edgeDinic++;
    // 反向边
    toDinic[edgeDinic] = u; capDinic[edgeDinic] = 0;
    nxtDinic[edgeDinic] = headDinic[v]; headDinic[v] = edgeDinic++;
}

// BFS构建分层图
int bfsDinic(int s, int t) {
    memset(level, -1, sizeof(level));
    int queue[MAXN], front = 0, rear = 0;
    level[s] = 0;
    queue[rear++] = s;
    while (front < rear) {
        int u = queue[front++];
        for (int i = headDinic[u]; i != -1; i = nxtDinic[i]) {
            int v = toDinic[i];
            if (level[v] == -1 && capDinic[i] > 0) {
                level[v] = level[u] + 1;
                queue[rear++] = v;
            }
        }
    }
    return level[t] != -1;
}

// DFS找阻塞流
int dfsDinic(int u, int t, int minFlow) {
    if (u == t) return minFlow;
    for (int &i = cur[u]; i != -1; i = nxtDinic[i]) {
        int v = toDinic[i];
        if (level[v] == level[u] + 1 && capDinic[i] > 0) {
            int f = dfsDinic(v, t, min(minFlow, capDinic[i]));
            if (f > 0) {
                capDinic[i] -= f;
                capDinic[i ^ 1] += f;
                return f;
            }
        }
    }
    return 0;
}

int dinic(int s, int t) {
    int total = 0;
    while (bfsDinic(s, t)) {
        memcpy(cur, headDinic, sizeof(headDinic));
        int f;
        while ((f = dfsDinic(s, t, INT_MAX)) > 0)
            total += f;
    }
    return total;
}
```

## 4. 最大流最小割定理

最大流 = 最小割。在最大流之后，从源点出发在残量网络中BFS可达的点集为S，不可达的为T，割(S,T)即为最小割。

```c
// 求最小割的边集
void findMinCut(int s, int n) {
    int visited[MAXN] = {0};
    int queue[MAXN], front = 0, rear = 0;
    queue[rear++] = s;
    visited[s] = 1;

    while (front < rear) {
        int u = queue[front++];
        for (int i = headDinic[u]; i != -1; i = nxtDinic[i]) {
            int v = toDinic[i];
            if (!visited[v] && capDinic[i] > 0) {
                visited[v] = 1;
                queue[rear++] = v;
            }
        }
    }
    printf("最小割边: ");
    for (int u = 0; u < n; u++) {
        if (!visited[u]) continue;
        for (int i = headDinic[u]; i != -1; i = nxtDinic[i]) {
            int v = toDinic[i];
            if (!visited[v] && (i % 2 == 0))  // 正向边
                printf("%d->%d ", u, v);
        }
    }
    printf("\n");
}
```

## 5. 二分图匹配

### 匈牙利算法（基于DFS）

```c
#include <stdio.h>
#include <string.h>

#define MAXN 505

int graphMatch[MAXN][MAXN];  // 左侧到右侧的边
int matchR[MAXN];            // 右侧匹配的左侧节点
int visited[MAXN];           // DFS标记

int bpm(int u, int nR) {
    for (int v = 0; v < nR; v++) {
        if (graphMatch[u][v] && !visited[v]) {
            visited[v] = 1;
            if (matchR[v] == -1 || bpm(matchR[v], nR)) {
                matchR[v] = u;
                return 1;
            }
        }
    }
    return 0;
}

int hungarian(int nL, int nR) {
    memset(matchR, -1, sizeof(matchR));
    int result = 0;
    for (int u = 0; u < nL; u++) {
        memset(visited, 0, sizeof(visited));
        if (bpm(u, nR)) result++;
    }
    return result;
}
```

### 二分图最大匹配（网络流方法）

```c
// 建图：源点 -> 左侧节点(容量1)，左侧 -> 右侧(容量1)，右侧 -> 汇点(容量1)
// 最大流 = 最大匹配
int bipartiteMatch(int nL, int nR) {
    int s = nL + nR, t = s + 1;
    // 清空边...
    for (int i = 0; i < nL; i++)
        addEdgeDinic(s, i, 1);
    for (int i = 0; i < nR; i++)
        addEdgeDinic(nL + i, t, 1);
    // 添加匹配边：addEdgeDinic(u, nL + v, 1)
    return dinic(s, t);
}
```

## 6. 费用流（最小费用最大流）

```c
int costEdge[MAXN][MAXN];  // 费用
int dist[MAXN], inQueue[MAXN];

// SPFA找最短路（最小费用增广路）
int spfaFlow(int s, int t) {
    memset(dist, 0x3f, sizeof(dist));
    dist[s] = 0;
    int queue[MAXN * 10], front = 0, rear = 0;
    queue[rear++] = s; inQueue[s] = 1;
    memset(parent, -1, sizeof(parent));

    while (front < rear) {
        int u = queue[front++];
        inQueue[u] = 0;
        for (int v = 0; v < n; v++) {
            if (cap[u][v] - flow[u][v] > 0 && dist[u] + costEdge[u][v] < dist[v]) {
                dist[v] = dist[u] + costEdge[u][v];
                parent[v] = u;
                if (!inQueue[v]) { queue[rear++] = v; inQueue[v] = 1; }
            }
        }
    }
    return parent[t] != -1;
}

// 最小费用最大流
int minCostMaxFlow(int s, int t) {
    memset(flow, 0, sizeof(flow));
    int totalCost = 0, totalFlow = 0;
    while (spfaFlow(s, t)) {
        int f = INT_MAX;
        for (int cur = t; cur != s; cur = parent[cur])
            f = min(f, cap[parent[cur]][cur] - flow[parent[cur]][cur]);
        for (int cur = t; cur != s; cur = parent[cur]) {
            flow[parent[cur]][cur] += f;
            flow[cur][parent[cur]] -= f;
        }
        totalFlow += f;
        totalCost += f * dist[t];
    }
    return totalCost;
}
```

## 7. 网络流应用总结

| 问题 | 建图方式 |
|------|---------|
| 最大流 | 直接使用 |
| 最小割 | = 最大流 |
| 二分图最大匹配 | 源->左->右->汇 |
| 最小点覆盖 | = 最大匹配（König定理） |
| 最大独立集 | = V - 最大匹配 |
| 最小路径覆盖 | 拆点建图 |
| 多源多汇 | 添加超级源汇 |
