## 1. 快速幂

```c
long long fastPow(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) res = res * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return res;
}
```

## 2. 并查集

```c
int fa[100005], rk[100005];
void initDSU(int n) { for (int i = 0; i < n; i++) { fa[i] = i; rk[i] = 0; } }
int find(int x) { return fa[x] == x ? x : (fa[x] = find(fa[x])); }
int unite(int x, int y) {
    int px = find(x), py = find(y);
    if (px == py) return 0;
    if (rk[px] < rk[py]) { int t = px; px = py; py = t; }
    fa[py] = px;
    if (rk[px] == rk[py]) rk[px]++;
    return 1;
}
```

## 3. 线段树（区间和、单点修改）

```c
#define MAXN 100005
long long tree[4 * MAXN];

void build(int arr[], int node, int start, int end) {
    if (start == end) { tree[node] = arr[start]; return; }
    int mid = (start + end) / 2;
    build(arr, 2 * node, start, mid);
    build(arr, 2 * node + 1, mid + 1, end);
    tree[node] = tree[2 * node] + tree[2 * node + 1];
}

void update(int node, int start, int end, int idx, long long val) {
    if (start == end) { tree[node] = val; return; }
    int mid = (start + end) / 2;
    if (idx <= mid) update(2 * node, start, mid, idx, val);
    else update(2 * node + 1, mid + 1, end, idx, val);
    tree[node] = tree[2 * node] + tree[2 * node + 1];
}

long long query(int node, int start, int end, int l, int r) {
    if (r < start || end < l) return 0;
    if (l <= start && end <= r) return tree[node];
    int mid = (start + end) / 2;
    return query(2 * node, start, mid, l, r)
         + query(2 * node + 1, mid + 1, end, l, r);
}
```

## 4. 树状数组（BIT）

```c
int bit[MAXN], nBit;
int lowbit(int x) { return x & (-x); }
void bitUpdate(int i, int delta) {
    for (; i <= nBit; i += lowbit(i)) bit[i] += delta;
}
int bitQuery(int i) {
    int sum = 0;
    for (; i > 0; i -= lowbit(i)) sum += bit[i];
    return sum;
}
int bitRangeQuery(int l, int r) { return bitQuery(r) - bitQuery(l - 1); }
```

## 5. Dijkstra最短路

```c
#include <string.h>
int dist[MAXN], visited[MAXN];
int graphD[MAXN][MAXN];
void dijkstra(int src, int n) {
    for (int i = 0; i < n; i++) { dist[i] = INF; visited[i] = 0; }
    dist[src] = 0;
    for (int i = 0; i < n; i++) {
        int u = -1;
        for (int j = 0; j < n; j++)
            if (!visited[j] && (u == -1 || dist[j] < dist[u])) u = j;
        if (u == -1 || dist[u] == INF) break;
        visited[u] = 1;
        for (int v = 0; v < n; v++)
            if (graphD[u][v] != INF && dist[u] + graphD[u][v] < dist[v])
                dist[v] = dist[u] + graphD[u][v];
    }
}
```

## 6. KMP字符串匹配

```c
int nextArr[MAXN];
void computeNext(const char *p, int m) {
    nextArr[0] = -1;
    int j = -1;
    for (int i = 1; i < m; i++) {
        while (j >= 0 && p[j + 1] != p[i]) j = nextArr[j];
        if (p[j + 1] == p[i]) j++;
        nextArr[i] = j;
    }
}
int kmpMatch(const char *text, const char *pat) {
    int n = strlen(text), m = strlen(pat);
    computeNext(pat, m);
    int j = -1;
    for (int i = 0; i < n; i++) {
        while (j >= 0 && pat[j + 1] != text[i]) j = nextArr[j];
        if (pat[j + 1] == text[i]) j++;
        if (j == m - 1) return i - m + 1;
    }
    return -1;
}
```

## 7. 拓扑排序（BFS）

```c
int inDeg[MAXN], topoHead[MAXN], topoTo[MAXM], topoNxt[MAXM], topoEdge = 0;
int topoResult[MAXN], topoCnt;
void topoSort(int n) {
    int queue[MAXN], front = 0, rear = 0;
    for (int i = 0; i < n; i++) if (inDeg[i] == 0) queue[rear++] = i;
    topoCnt = 0;
    while (front < rear) {
        int u = queue[front++];
        topoResult[topoCnt++] = u;
        for (int i = topoHead[u]; i != -1; i = topoNxt[i])
            if (--inDeg[topoTo[i]] == 0) queue[rear++] = topoTo[i];
    }
}
```

## 9. 最大流（Dinic模板）

```c
int dHead[MAXN], dTo[MAXM], dCap[MAXM], dNxt[MAXM], dEdge = 0;
int dLevel[MAXN], dCur[MAXN];
void dAddEdge(int u, int v, int c) {
    dTo[dEdge] = v; dCap[dEdge] = c; dNxt[dEdge] = dHead[u]; dHead[u] = dEdge++;
    dTo[dEdge] = u; dCap[dEdge] = 0; dNxt[dEdge] = dHead[v]; dHead[v] = dEdge++;
}
int dBfs(int s, int t) {
    memset(dLevel, -1, sizeof(dLevel));
    int q[MAXN], f = 0, r = 0; dLevel[s] = 0; q[r++] = s;
    while (f < r) for (int u = q[f++], i = dHead[u]; i != -1; i = dNxt[i])
        if (dLevel[dTo[i]] == -1 && dCap[i] > 0) { dLevel[dTo[i]] = dLevel[u]+1; q[r++] = dTo[i]; }
    return dLevel[t] != -1;
}
int dDfs(int u, int t, int f) {
    if (u == t) return f;
    for (int &i = dCur[u]; i != -1; i = dNxt[i])
        if (dLevel[dTo[i]] == dLevel[u]+1 && dCap[i] > 0) {
            int d = dDfs(dTo[i], t, f < dCap[i] ? f : dCap[i]);
            if (d > 0) { dCap[i] -= d; dCap[i^1] += d; return d; }
        }
    return 0;
}
int dinicFlow(int s, int t) {
    int total = 0, f;
    while (dBfs(s, t)) { memcpy(dCur, dHead, sizeof(dHead)); while ((f = dDfs(s, t, INF)) > 0) total += f; }
    return total;
}
```
