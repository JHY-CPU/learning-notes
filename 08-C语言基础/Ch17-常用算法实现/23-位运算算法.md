# 位运算算法

## 1. 基本位运算操作

```c
#include <stdio.h>

// 常用位运算
// &  按位与：两位都为1时结果为1
// |  按位或：至少一位为1时结果为1
// ^  按位异或：两位不同时结果为1
// ~  按位取反
// << 左移：高位丢弃，低位补0
// >> 右移：低位丢弃，高位补符号位（有符号）或0（无符号）

// 常用技巧
int setBit(int n, int k)    { return n | (1 << k); }    // 设置第k位为1
int clearBit(int n, int k)  { return n & ~(1 << k); }   // 清除第k位
int toggleBit(int n, int k) { return n ^ (1 << k); }    // 翻转第k位
int getBit(int n, int k)    { return (n >> k) & 1; }     // 获取第k位的值

int lowbit(int n) { return n & (-n); }  // 最低位的1
int highbit(int n) {                   // 最高位的1
    n |= n >> 1; n |= n >> 2; n |= n >> 4;
    n |= n >> 8; n |= n >> 16;
    return (n + 1) >> 1;
}
```

## 2. 统计二进制中1的个数

```c
// 方法1：逐位检查
int countBits_v1(int n) {
    int count = 0;
    while (n) {
        count += n & 1;
        n >>= 1;
    }
    return count;
}

// 方法2：使用lowbit
int countBits_v2(int n) {
    int count = 0;
    while (n) {
        n -= n & (-n);  // 消去最低位的1
        count++;
    }
    return count;
}

// 方法3：Brian Kernighan算法
int countBits_v3(int n) {
    int count = 0;
    while (n) {
        n &= (n - 1);  // 消去最低位的1
        count++;
    }
    return count;
}

// GCC内置函数
int countBits_builtin(int n) {
    return __builtin_popcount(n);         // unsigned int
    // __builtin_popcountll(n)           // unsigned long long
}
```

## 3. 状态压缩

用二进制位表示集合状态，适用于n <= 20的情况。

```c
#include <stdio.h>

// 枚举所有子集
void enumerateSubsets(int n) {
    // n个元素，共2^n个子集
    for (int mask = 0; mask < (1 << n); mask++) {
        printf("子集%d: { ", mask);
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i))
                printf("%d ", i);
        }
        printf("}\n");
    }
}

// 枚举所有大小为k的子集
void enumerateSubsetsSizeK(int n, int k) {
    int mask = (1 << k) - 1;  // 初始: 低k位全1
    while (mask < (1 << n)) {
        printf("子集: { ");
        for (int i = 0; i < n; i++)
            if (mask & (1 << i)) printf("%d ", i);
        printf("}\n");

        // Gosper's Hack: 找下一个含k个1的数
        int c = mask & (-mask);
        int r = mask + c;
        mask = (((r ^ mask) >> 2) / c) | r;
    }
}
```

## 4. 位运算解决子集问题

### 子集和问题

```c
// 用位运算枚举所有子集，判断和是否为target
void subsetSum(int arr[], int n, int target) {
    for (int mask = 0; mask < (1 << n); mask++) {
        int sum = 0;
        for (int i = 0; i < n; i++)
            if (mask & (1 << i)) sum += arr[i];
        if (sum == target) {
            printf("找到子集: ");
            for (int i = 0; i < n; i++)
                if (mask & (1 << i)) printf("%d ", arr[i]);
            printf("\n");
        }
    }
}
```

### 分组问题

```c
// 将n个元素分成两组，使得两组和之差最小
int minDiffPartition(int arr[], int n) {
    int total = 0;
    for (int i = 0; i < n; i++) total += arr[i];

    int minDiff = total;
    for (int mask = 0; mask < (1 << n); mask++) {
        int sum = 0;
        for (int i = 0; i < n; i++)
            if (mask & (1 << i)) sum += arr[i];
        int diff = abs(total - 2 * sum);
        if (diff < minDiff) minDiff = diff;
    }
    return minDiff;
}
```

## 5. 位运算在算法竞赛中的应用

```c
// 判断n是否为2的幂
int isPowerOf2(int n) { return n > 0 && (n & (n - 1)) == 0; }

// 计算大于等于n的最小2的幂
int nextPowerOf2(int n) {
    n--;
    n |= n >> 1; n |= n >> 2; n |= n >> 4;
    n |= n >> 8; n |= n >> 16;
    return n + 1;
}

// 交换两个数（不使用临时变量）
void swap(int *a, int *b) {
    *a ^= *b;
    *b ^= *a;
    *a ^= *b;
}

// 二进制翻转（反转所有位）
int reverseBits(int n) {
    int result = 0;
    for (int i = 0; i < 32; i++) {
        result = (result << 1) | (n & 1);
        n >>= 1;
    }
    return result;
}

// Gray码生成
void grayCode(int n) {
    for (int i = 0; i < (1 << n); i++)
        printf("%d ", i ^ (i >> 1));
    printf("\n");
}
```

## 7. 位运算注意事项

```c
// 1. 优先级问题：位运算优先级低于比较运算符
if ((flags & FLAG) != 0)  // 正确
if (flags & FLAG != 0)    // 错误：!= 优先级高于 &

// 2. 有符号右移可能填充符号位
unsigned int n = 0x80000000;
n >> 1;  // 结果是 0x40000000（无符号）

// 3. 左移超过位数是未定义行为
1 << 32;  // 对于32位int是未定义行为

// 4. 使用 1LL << n 避免溢出
long long x = 1LL << 40;  // 正确
long long y = 1 << 40;    // 溢出
```
