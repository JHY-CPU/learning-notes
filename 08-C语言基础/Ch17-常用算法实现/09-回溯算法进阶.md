# 回溯算法进阶——剪枝与优化

## 1. 剪枝策略

剪枝是回溯算法优化的关键，通过提前排除不可能产生最优解的分支，大幅减少搜索空间。

### 剪枝类型
- **可行性剪枝**：当前状态已不满足约束条件，提前返回
- **最优性剪枝**：当前路径不可能比已知最优解更优，提前返回
- **对称性剪枝**：排除对称等价的搜索分支
- **顺序剪枝**：通过固定搜索顺序避免重复

## 2. 数独求解

```c
#include <stdio.h>
#include <stdbool.h>

int board[9][9];
bool row[9][10], col[9][10], box[3][3][10];

bool solveSudoku(int pos) {
    if (pos == 81) return true;  // 所有格子填完
    int r = pos / 9, c = pos % 9;
    if (board[r][c] != 0) return solveSudoku(pos + 1);  // 已有数字

    for (int num = 1; num <= 9; num++) {
        int b = (r / 3) * 3 + c / 3;
        if (!row[r][num] && !col[c][num] && !box[r/3][c/3][num]) {
            board[r][c] = num;
            row[r][num] = col[c][num] = box[r/3][c/3][num] = true;
            if (solveSudoku(pos + 1)) return true;
            board[r][c] = 0;
            row[r][num] = col[c][num] = box[r/3][c/3][num] = false;
        }
    }
    return false;
}

void initSudoku(int b[9][9]) {
    for (int i = 0; i < 9; i++)
        for (int j = 0; j < 9; j++) {
            board[i][j] = b[i][j];
            if (b[i][j]) {
                row[i][b[i][j]] = true;
                col[j][b[i][j]] = true;
                box[i/3][j/3][b[i][j]] = true;
            }
        }
}
```

## 3. 图的m着色问题

给定无向图，用m种颜色给每个顶点着色，使相邻顶点颜色不同。

```c
#include <stdio.h>
#include <stdbool.h>

#define MAXN 100

int graph[MAXN][MAXN], color[MAXN];
int n, m, count = 0;

bool isSafe(int v, int c) {
    for (int i = 0; i < n; i++)
        if (graph[v][i] && color[i] == c)
            return false;
    return true;
}

void graphColoring(int v) {
    if (v == n) {
        count++;
        printf("方案%d: ", count);
        for (int i = 0; i < n; i++) printf("%d ", color[i]);
        printf("\n");
        return;
    }
    for (int c = 1; c <= m; c++) {
        if (isSafe(v, c)) {
            color[v] = c;
            graphColoring(v + 1);
            color[v] = 0;  // 回溯
        }
    }
}
```

## 4. N皇后问题——进阶优化

使用位运算加速N皇后。

```c
#include <stdio.h>

int total = 0;

void solveNQueens(int n, int row, int cols, int diag1, int diag2) {
    if (row == n) {
        total++;
        return;
    }
    // 可放置的位置：1表示可放
    int available = ((1 << n) - 1) & ~(cols | diag1 | diag2);
    while (available) {
        // 取最低位的1
        int pos = available & (-available);
        available ^= pos;  // 移除这个位置
        solveNQueens(n, row + 1,
            cols | pos,
            (diag1 | pos) >> 1,
            (diag2 | pos) << 1);
    }
}

int main() {
    int n = 8;
    solveNQueens(n, 0, 0, 0, 0);
    printf("%d皇后的解的数量: %d\n", n, total);
    return 0;
}
```

## 5. 旅行商问题（TSP回溯法）

```c
#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

#define MAXN 20

int dist[MAXN][MAXN];
bool visited[MAXN];
int n, bestCost;
int path[MAXN], bestPath[MAXN];

void tsp(int pos, int depth, int cost) {
    if (depth == n) {
        cost += dist[pos][0];  // 回到起点
        if (cost < bestCost) {
            bestCost = cost;
            for (int i = 0; i < n; i++) bestPath[i] = path[i];
        }
        return;
    }
    // 剪枝：如果当前代价已经超过最优解，直接返回
    if (cost >= bestCost) return;

    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            visited[i] = true;
            path[depth] = i;
            tsp(i, depth + 1, cost + dist[pos][i]);
            visited[i] = false;
        }
    }
}
```

## 6. 子集和问题（带剪枝）

找出数组中和为目标值的所有子集。

```c
#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return *(int *)a - *(int *)b;
}

void subsetSum(int arr[], int n, int target, int *path, int depth, int start, int sum) {
    if (sum == target) {
        printf("{ ");
        for (int i = 0; i < depth; i++) printf("%d ", path[i]);
        printf("}\n");
        return;
    }
    for (int i = start; i < n; i++) {
        // 剪枝1: 和已超过目标
        if (sum + arr[i] > target) break;  // 数组已排序，后续更大
        // 剪枝2: 跳过重复元素
        if (i > start && arr[i] == arr[i - 1]) continue;
        path[depth] = arr[i];
        subsetSum(arr, n, target, path, depth + 1, i + 1, sum + arr[i]);
    }
}

int main() {
    int arr[] = {1, 2, 3, 4, 5, 6, 7};
    int n = 7, target = 10;
    qsort(arr, n, sizeof(int), cmp);
    int path[100];
    subsetSum(arr, n, target, path, 0, 0, 0);
    return 0;
}
```

## 7. 剪枝技巧总结

| 剪枝类型 | 说明 | 应用场景 |
|---------|------|---------|
| 可行性剪枝 | 不满足约束条件时提前终止 | 数独、N皇后 |
| 最优性剪枝 | 超过已知最优解时终止 | TSP、最优化问题 |
| 对称性剪枝 | 排除等价搜索分支 | 组合枚举 |
| 排序剪枝 | 排序后利用有序性剪枝 | 子集和 |
| 记忆化 | 记录已搜索状态 | 重复子问题 |

```c
// 通用剪枝框架
void backtrack_with_pruning(int state) {
    if (!feasible(state)) return;       // 可行性剪枝
    if (not_better(state)) return;      // 最优性剪枝
    if (is_solution(state)) {
        update_best(state);
        return;
    }
    for (each choice) {
        make_choice(state, choice);
        backtrack_with_pruning(state);
        undo_choice(state, choice);
    }
}
```
