# 动态规划经典问题

## 1. 0-1背包问题

有n个物品，每个物品有重量w和价值v，背包容量为W，求最大价值（每件物品最多选一次）。

```c
#include <stdio.h>

// dp[i][w]: 前i个物品、容量w时的最大价值
// dp[i][w] = max(dp[i-1][w], dp[i-1][w-w[i]] + v[i])
int knapsack(int W, int wt[], int val[], int n) {
    int dp[n + 1][W + 1];
    for (int i = 0; i <= n; i++) dp[i][0] = 0;
    for (int w = 0; w <= W; w++) dp[0][w] = 0;

    for (int i = 1; i <= n; i++) {
        for (int w = 0; w <= W; w++) {
            dp[i][w] = dp[i - 1][w];  // 不选第i个
            if (wt[i - 1] <= w) {
                int val_with = dp[i - 1][w - wt[i - 1]] + val[i - 1];
                if (val_with > dp[i][w])
                    dp[i][w] = val_with;
            }
        }
    }
    return dp[n][W];
}

// 空间优化版本：一维数组
int knapsack_optimized(int W, int wt[], int val[], int n) {
    int dp[W + 1];
    for (int w = 0; w <= W; w++) dp[w] = 0;

    for (int i = 0; i < n; i++) {
        // 必须逆序遍历，防止同一物品被选多次
        for (int w = W; w >= wt[i]; w--) {
            if (dp[w - wt[i]] + val[i] > dp[w])
                dp[w] = dp[w - wt[i]] + val[i];
        }
    }
    return dp[W];
}
```

## 2. 完全背包问题

每件物品可以选无限次。

```c
#include <stdio.h>

int unboundedKnapsack(int W, int wt[], int val[], int n) {
    int dp[W + 1];
    for (int w = 0; w <= W; w++) dp[w] = 0;

    for (int w = 0; w <= W; w++) {
        for (int i = 0; i < n; i++) {
            if (wt[i] <= w) {
                int v = dp[w - wt[i]] + val[i];
                if (v > dp[w]) dp[w] = v;
            }
        }
    }
    return dp[W];
}

// 零钱兑换II：凑出金额的方案数
int change(int amount, int coins[], int n) {
    int dp[amount + 1];
    for (int i = 0; i <= amount; i++) dp[i] = 0;
    dp[0] = 1;

    for (int i = 0; i < n; i++) {       // 遍历硬币
        for (int j = coins[i]; j <= amount; j++) {  // 正序遍历
            dp[j] += dp[j - coins[i]];
        }
    }
    return dp[amount];
}
```

## 3. 最长公共子序列（LCS）

找出两个序列的最长公共子序列长度。

```c
#include <stdio.h>
#include <string.h>

int lcs(char *X, char *Y, int m, int n) {
    int dp[m + 1][n + 1];
    for (int i = 0; i <= m; i++) dp[i][0] = 0;
    for (int j = 0; j <= n; j++) dp[0][j] = 0;

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (X[i - 1] == Y[j - 1])
                dp[i][j] = dp[i - 1][j - 1] + 1;
            else
                dp[i][j] = (dp[i - 1][j] > dp[i][j - 1]) ?
                           dp[i - 1][j] : dp[i][j - 1];
        }
    }

    // 回溯打印LCS
    int idx = dp[m][n];
    char lcs_str[idx + 1];
    lcs_str[idx] = '\0';
    int i = m, j = n;
    while (i > 0 && j > 0) {
        if (X[i - 1] == Y[j - 1]) {
            lcs_str[--idx] = X[i - 1];
            i--; j--;
        } else if (dp[i - 1][j] > dp[i][j - 1])
            i--;
        else
            j--;
    }
    printf("LCS: %s\n", lcs_str);
    return dp[m][n];
}
```

## 4. 最长递增子序列（LIS）

```c
#include <stdio.h>

// O(n^2) DP解法
int lis_dp(int arr[], int n) {
    int dp[n];
    for (int i = 0; i < n; i++) dp[i] = 1;
    int maxLen = 1;

    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (arr[j] < arr[i] && dp[j] + 1 > dp[i])
                dp[i] = dp[j] + 1;
        }
        if (dp[i] > maxLen) maxLen = dp[i];
    }
    return maxLen;
}

// O(n log n) 贪心+二分
int lower_bound(int tails[], int len, int key) {
    int lo = 0, hi = len;
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (tails[mid] < key) lo = mid + 1;
        else hi = mid;
    }
    return lo;
}

int lis_binary(int arr[], int n) {
    int tails[n], len = 0;
    for (int i = 0; i < n; i++) {
        int pos = lower_bound(tails, len, arr[i]);
        tails[pos] = arr[i];
        if (pos == len) len++;
    }
    return len;
}
```

## 5. 编辑距离

将字符串A变为字符串B的最少操作次数（插入、删除、替换）。

```c
#include <stdio.h>
#include <string.h>

int min3(int a, int b, int c) {
    int m = a < b ? a : b;
    return m < c ? m : c;
}

int editDistance(char *A, char *B) {
    int m = strlen(A), n = strlen(B);
    int dp[m + 1][n + 1];

    for (int i = 0; i <= m; i++) dp[i][0] = i; // 删除
    for (int j = 0; j <= n; j++) dp[0][j] = j; // 插入

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (A[i - 1] == B[j - 1])
                dp[i][j] = dp[i - 1][j - 1];  // 字符相同，无需操作
            else
                dp[i][j] = 1 + min3(
                    dp[i - 1][j],     // 删除
                    dp[i][j - 1],     // 插入
                    dp[i - 1][j - 1]  // 替换
                );
        }
    }
    return dp[m][n];
}
```

## 6. 最长回文子串

```c
#include <stdio.h>
#include <string.h>

void longestPalindrome(char *s) {
    int n = strlen(s);
    int dp[n][n];  // dp[i][j]: s[i..j]是否为回文
    int start = 0, maxLen = 1;

    for (int i = 0; i < n; i++) dp[i][i] = 1;         // 单字符
    for (int i = 0; i < n - 1; i++) {                   // 相邻字符
        dp[i][i + 1] = (s[i] == s[i + 1]);
        if (dp[i][i + 1]) { start = i; maxLen = 2; }
    }

    for (int len = 3; len <= n; len++) {
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            dp[i][j] = (s[i] == s[j]) && dp[i + 1][j - 1];
            if (dp[i][j] && len > maxLen) {
                start = i; maxLen = len;
            }
        }
    }
    printf("最长回文子串: ");
    for (int i = start; i < start + maxLen; i++) printf("%c", s[i]);
    printf(" (长度%d)\n", maxLen);
}
```
