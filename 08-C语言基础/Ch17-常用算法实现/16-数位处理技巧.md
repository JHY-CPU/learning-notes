# 数位处理技巧

## 1. 各位数字提取

数字的各位提取是基础操作，广泛用于数位DP、数字处理等场景。

```c
#include <stdio.h>

// 提取各位数字存入数组（低位在前）
int getDigits(int n, int digits[]) {
    int count = 0;
    if (n == 0) { digits[0] = 0; return 1; }
    while (n > 0) {
        digits[count++] = n % 10;
        n /= 10;
    }
    return count;
}

// 提取各位数字存入数组（高位在前）
int getDigitsHighFirst(int n, int digits[]) {
    int temp[20], count = 0;
    while (n > 0) {
        temp[count++] = n % 10;
        n /= 10;
    }
    for (int i = 0; i < count; i++)
        digits[i] = temp[count - 1 - i];
    return count;
}

// 获取第k位数字（从低位开始，k从0开始）
int getKthDigit(int n, int k) {
    for (int i = 0; i < k; i++) n /= 10;
    return n % 10;
}

// 获取数字位数
int countDigits(int n) {
    if (n == 0) return 1;
    int count = 0;
    while (n > 0) { count++; n /= 10; }
    return count;
}
```

## 2. 数字反转

```c
// 整数反转
int reverseInt(int n) {
    int result = 0;
    while (n != 0) {
        result = result * 10 + n % 10;
        n /= 10;
    }
    return result;
}

// 判断回文数
int isPalindrome(int n) {
    if (n < 0) return 0;
    return n == reverseInt(n);
}

// 不转字符串判断回文
int isPalindromeDirect(int n) {
    if (n < 0 || (n % 10 == 0 && n != 0)) return 0;
    int reversed = 0;
    while (n > reversed) {
        reversed = reversed * 10 + n % 10;
        n /= 10;
    }
    return n == reversed || n == reversed / 10;
}
```

## 3. 数字拆分与组合

```c
// 将数字拆分为若干个不重复的数字之和
// 例如 10 = 1+2+3+4
void splitIntoUnique(int n) {
    int parts[100], count = 0;
    int remaining = n, cur = 1;
    while (remaining > 0) {
        if (remaining <= 2 * cur) {
            parts[count++] = remaining;
            remaining = 0;
        } else {
            parts[count++] = cur;
            remaining -= cur;
            cur++;
        }
    }
    printf("%d = ", n);
    for (int i = 0; i < count; i++)
        printf("%d%s", parts[i], i < count - 1 ? "+" : "\n");
}

// 将数字拆分为2的幂之和
void splitIntoPowersOf2(int n) {
    printf("%d = ", n);
    int first = 1;
    for (int i = 0; i < 31; i++) {
        if (n & (1 << i)) {
            if (!first) printf(" + ");
            printf("%d", 1 << i);
            first = 0;
        }
    }
    printf("\n");
}
```

## 4. 数位统计问题

### 统计 [1, n] 中数字 k 出现的次数

```c
#include <stdio.h>

// 统计数字k在[1, n]中出现的次数
int countDigitK(int n, int k) {
    int count = 0;
    for (long long factor = 1; factor <= n; factor *= 10) {
        int higher = n / (factor * 10);    // 高位
        int cur = (n / factor) % 10;        // 当前位
        int lower = n % factor;             // 低位

        if (cur > k) {
            count += (higher + 1) * factor;
        } else if (cur == k) {
            count += higher * factor + lower + 1;
        } else {
            count += higher * factor;
        }
        if (k == 0) count -= factor;  // 前导零不算
    }
    return count;
}

int main() {
    // 统计 1~233 中数字 3 出现的次数
    printf("数字3在[1,233]中出现%d次\n", countDigitK(233, 3));
    return 0;
}
```

## 5. 数位DP基础

统计 [L, R] 中满足某些数位条件的数字个数。

### 统计不含相邻相同数字的数

```c
#include <stdio.h>
#include <string.h>

int digits[20], len;
int dp[20][10][2];  // pos, last_digit, tight

// 计算 [0, n] 中不含相邻相同数字的个数
int solve(int pos, int last, int tight) {
    if (pos == len) return 1;
    if (dp[pos][last][tight] != -1) return dp[pos][last][tight];

    int limit = tight ? digits[pos] : 9;
    int result = 0;
    for (int d = 0; d <= limit; d++) {
        if (pos == 0 || d != last) {  // 第一位或与前一位不同
            result += solve(pos + 1, d, tight && (d == limit));
        }
    }
    return dp[pos][last][tight] = result;
}

int countValid(int n) {
    len = 0;
    while (n > 0) { digits[len++] = n % 10; n /= 10; }
    // 反转
    for (int i = 0; i < len / 2; i++) {
        int t = digits[i]; digits[i] = digits[len-1-i]; digits[len-1-i] = t;
    }
    memset(dp, -1, sizeof(dp));
    return solve(0, 0, 1);
}
```

## 6. 进制转换

```c
#include <stdio.h>
#include <string.h>

// 十进制转任意进制
void decToBase(int n, int base, char *result) {
    char temp[64];
    int idx = 0;
    if (n == 0) { strcpy(result, "0"); return; }
    while (n > 0) {
        int rem = n % base;
        temp[idx++] = (rem < 10) ? '0' + rem : 'A' + rem - 10;
        n /= base;
    }
    // 反转
    for (int i = 0; i < idx; i++)
        result[i] = temp[idx - 1 - i];
    result[idx] = '\0';
}

// 任意进制转十进制
int baseToDec(const char *num, int base) {
    int result = 0;
    for (int i = 0; num[i]; i++) {
        int digit;
        if (num[i] >= '0' && num[i] <= '9') digit = num[i] - '0';
        else if (num[i] >= 'A' && num[i] <= 'Z') digit = num[i] - 'A' + 10;
        else digit = num[i] - 'a' + 10;
        result = result * base + digit;
    }
    return result;
}
```

## 7. 常用数位技巧汇总

```c
// 判断n是否为2的幂
int isPowerOf2(int n) { return n > 0 && (n & (n - 1)) == 0; }

// 判断n是否为完全平方数
int isPerfectSquare(int n) {
    int s = (int)sqrt(n);
    return s * s == n;
}

// 计算n的各位数字之和
int digitSum(int n) {
    int sum = 0;
    while (n > 0) { sum += n % 10; n /= 10; }
    return sum;
}

// 计算n的各位数字之积（跳过0）
int digitProduct(int n) {
    int prod = 1;
    while (n > 0) {
        if (n % 10 != 0) prod *= n % 10;
        n /= 10;
    }
    return prod;
}
```
