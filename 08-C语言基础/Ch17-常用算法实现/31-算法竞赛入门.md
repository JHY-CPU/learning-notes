# 算法竞赛入门

## 1. 竞赛编程概述

算法竞赛（如ACM-ICPC、Codeforces、LeetCode周赛）要求在有限时间内解决多个算法问题。掌握高效的编码技巧和常用数据结构至关重要。

## 2. 快速输入输出

竞赛中I/O优化可以显著提升速度。

```c
#include <stdio.h>
#include <stdlib.h>

// 快速读入整数（比scanf快很多）
int readInt() {
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    return x * f;
}

// 快速输出整数
void writeInt(int x) {
    if (x < 0) { putchar('-'); x = -x; }
    if (x > 9) writeInt(x / 10);
    putchar(x % 10 + '0');
}

// 更快的版本（使用fread）
#define BUF_SIZE 1 << 20
char buf[BUF_SIZE], *p1 = buf, *p2 = buf;
#define gc() (p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, BUF_SIZE, stdin), p1 == p2) ? EOF : *p1++)

int readIntFast() {
    int x = 0; char c = gc();
    while (c < '0' || c > '9') c = gc();
    while (c >= '0' && c <= '9') { x = x * 10 + c - '0'; c = gc(); }
    return x;
}
```

## 3. 常用C语言STL技巧

虽然C没有STL，但可以用宏和函数简化常用操作。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAXN 1000005
#define INF 0x3f3f3f3f
#define MOD 1000000007
#define max(a, b) ((a) > (b) ? (a) : (b))
#define min(a, b) ((a) < (b) ? (a) : (b))
#define abs(x) ((x) < 0 ? -(x) : (x))
#define swap(a, b) { typeof(a) t = a; a = b; b = t; }

// 通用排序比较函数
int cmpInt(const void *a, const void *b) { return *(int *)a - *(int *)b; }
int cmpIntDesc(const void *a, const void *b) { return *(int *)b - *(int *)a; }

// 最大最小值
int max3(int a, int b, int c) { return max(a, max(b, c)); }
int min3(int a, int b, int c) { return min(a, min(b, c)); }

// GCD/LCM
int gcd(int a, int b) { return b ? gcd(b, a % b) : a; }
int lcm(int a, int b) { return a / gcd(a, b) * b; }
```

## 4. 手写优先队列（堆）

```c
#define MAXHEAP 1000005

int heap[MAXHEAP], heapSize;

void heapSwap(int *a, int *b) { int t = *a; *a = *b; *b = t; }

void heapifyUp(int i) {
    while (i > 0) {
        int p = (i - 1) / 2;
        if (heap[p] < heap[i]) { heapSwap(&heap[p], &heap[i]); i = p; }
        else break;
    }
}

void heapifyDown(int i, int n) {
    while (2 * i + 1 < n) {
        int j = 2 * i + 1;
        if (j + 1 < n && heap[j + 1] > heap[j]) j++;
        if (heap[i] < heap[j]) { heapSwap(&heap[i], &heap[j]); i = j; }
        else break;
    }
}

void push(int val) { heap[heapSize] = val; heapifyUp(heapSize++); }
int top() { return heap[0]; }
void pop() { heap[0] = heap[--heapSize]; heapifyDown(0, heapSize); }
int empty() { return heapSize == 0; }
```

## 5. 手写双端队列

```c
#define MAXDQ 1000005
int deque[MAXDQ], dqFront = 0, dqBack = 0;

void pushBack(int x) { deque[dqBack++] = x; }
void pushFront(int x) { deque[--dqFront] = x; }
int front() { return deque[dqFront]; }
int back() { return deque[dqBack - 1]; }
void popFront() { dqFront++; }
void popBack() { dqBack--; }
int dqEmpty() { return dqFront == dqBack; }
int dqSize() { return dqBack - dqFront; }
```

## 6. 模运算常用操作

```c
#define MOD 1000000007

long long fastPow(long long a, long long b) {
    long long res = 1;
    a %= MOD;
    while (b > 0) {
        if (b & 1) res = res * a % MOD;
        a = a * a % MOD;
        b >>= 1;
    }
    return res;
}

long long modInv(long long a) { return fastPow(a, MOD - 2); }

long long add(long long a, long long b) { return ((a % MOD) + (b % MOD)) % MOD; }
long long sub(long long a, long long b) { return ((a % MOD) - (b % MOD) + MOD) % MOD; }
long long mul(long long a, long long b) { return ((a % MOD) * (b % MOD)) % MOD; }
long long div_(long long a, long long b) { return mul(a, modInv(b)); }
```

double ternarySearch(double (*f)(double), double lo, double hi) {
    for (int i = 0; i < 200; i++) {
        double m1 = lo + (hi - lo) / 3;
        double m2 = hi - (hi - lo) / 3;

## 8. 竞赛注意事项

1. **读题仔细**：注意数据范围，选择合适的数据类型
2. **边界处理**：特判 n=0, n=1 等边界情况
3. **溢出检查**：乘法可能溢出，使用 long long
4. **多组数据**：注意数组清空和变量重置
5. **时间限制**：O(n^2) 在 n>10^4 时超时，O(n log n) 在 n>10^6 时需注意
6. **空间限制**：大数组注意开在全局（栈空间有限）
