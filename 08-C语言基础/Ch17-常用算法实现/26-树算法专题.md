# 树算法专题

## 1. 树的基本表示

```c
#include <stdio.h>
#include <string.h>

#define MAXN 100005

// 邻接表表示
int head[MAXN], to[MAXN], nxt[MAXN], weight[MAXN], edge_cnt = 0;

void addEdge(int u, int v, int w) {
    to[edge_cnt] = v;
    weight[edge_cnt] = w;
    nxt[edge_cnt] = head[u];
    head[u] = edge_cnt++;
}

// 双向边
void addBidirectional(int u, int v, int w) {
    addEdge(u, v, w);
    addEdge(v, u, w);
}
```

## 2. 最近公共祖先（LCA）

### 倍增法 O(n log n)预处理，O(log n)查询

```c
#include <stdio.h>
#include <string.h>

#define MAXN 100005
#define LOGN 18

int parent[LOGN][MAXN];  // parent[k][u]: u的第2^k个祖先
int depth[MAXN];

void dfs(int u, int par, int d) {
    parent[0][u] = par;
    depth[u] = d;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v != par) dfs(v, u, d + 1);
    }
}

void preprocess(int n, int root) {
    memset(parent, -1, sizeof(parent));
    dfs(root, -1, 0);
    for (int k = 1; k < LOGN; k++)
        for (int i = 0; i < n; i++)
            if (parent[k - 1][i] != -1)
                parent[k][i] = parent[k - 1][parent[k - 1][i]];
}

int lca(int u, int v) {
    if (depth[u] < depth[v]) { int t = u; u = v; v = t; }
    // 提升u到同一深度
    int diff = depth[u] - depth[v];
    for (int k = 0; k < LOGN; k++)
        if ((diff >> k) & 1) u = parent[k][u];
    if (u == v) return u;
    // 一起向上跳
    for (int k = LOGN - 1; k >= 0; k--)
        if (parent[k][u] != parent[k][v]) {
            u = parent[k][u];
            v = parent[k][v];
        }
    return parent[0][u];
}
```

## 3. 树的直径

树的直径是树上最长路径，可用两次DFS或树形DP求解。

### 两次DFS法

```c
#include <stdio.h>
#include <string.h>

int dist[MAXN], farthestNode;

void dfsFarthest(int u, int par, int d) {
    dist[u] = d;
    if (d > dist[farthestNode]) farthestNode = u;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v != par) dfsFarthest(v, u, d + weight[i]);
    }
}

int treeDiameter() {
    // 第一次DFS：从任意点找到最远点A
    farthestNode = 0;
    dfsFarthest(0, -1, 0);
    int A = farthestNode;
    // 第二次DFS：从A找到最远点B，A-B即为直径
    dfsFarthest(A, -1, 0);
    return dist[farthestNode];
}
```

### 树形DP法

```c
int diameter = 0;

int dfsDiameter(int u, int par) {
    int max1 = 0, max2 = 0;  // 最长和次长子树路径
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == par) continue;
        int d = dfsDiameter(v, u) + weight[i];
        if (d > max1) { max2 = max1; max1 = d; }
        else if (d > max2) max2 = d;
    }
    if (max1 + max2 > diameter) diameter = max1 + max2;
    return max1;
}
```

## 4. 树的重心

树的重心是使删除该节点后最大子树最小的节点。

```c
int centroid, minMaxSubtree = MAXN;
int subtreeSize[MAXN];

void dfsCentroid(int u, int par, int total) {
    subtreeSize[u] = 1;
    int maxSub = 0;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == par) continue;
        dfsCentroid(v, u, total);
        subtreeSize[u] += subtreeSize[v];
        if (subtreeSize[v] > maxSub) maxSub = subtreeSize[v];
    }
    // 剩余部分
    int remain = total - subtreeSize[u];
    if (remain > maxSub) maxSub = remain;
    if (maxSub < minMaxSubtree) {
        minMaxSubtree = maxSub;
        centroid = u;
    }
}
```

## 5. 树上DP

### 以每个节点为根的子树大小

```c
int subtreeSize[MAXN];

void dfsSubtreeSize(int u, int par) {
    subtreeSize[u] = 1;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v != par) {
            dfsSubtreeSize(v, u);
            subtreeSize[u] += subtreeSize[v];
        }
    }
}
```

### 树上最大独立集

```c
int dp[MAXN][2];  // dp[u][0]: 不选u, dp[u][1]: 选u

void dfsIndependentSet(int u, int par) {
    dp[u][0] = 0; dp[u][1] = 1;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == par) continue;
        dfsIndependentSet(v, u);
        dp[u][0] += (dp[v][0] > dp[v][1]) ? dp[v][0] : dp[v][1];
        dp[u][1] += dp[v][0];
    }
}
```

## 6. 树链剖分（轻重链剖分）

将树分解为若干条链，使得树上路径查询变为链上查询。

```c
int size_[MAXN], son[MAXN], depth_[MAXN], parent_[MAXN];
int top[MAXN], dfn[MAXN], rnk[MAXN], dfsClock = 0;

void dfs1(int u, int par, int d) {
    size_[u] = 1; depth_[u] = d; parent_[u] = par; son[u] = -1;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == par) continue;
        dfs1(v, u, d + 1);
        size_[u] += size_[v];
        if (son[u] == -1 || size_[v] > size_[son[u]])
            son[u] = v;
    }
}

void dfs2(int u, int t) {
    top[u] = t;
    dfn[u] = ++dfsClock;
    rnk[dfsClock] = u;
    if (son[u] != -1) dfs2(son[u], t);  // 重儿子在同一条链
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == parent_[u] || v == son[u]) continue;
        dfs2(v, v);  // 轻儿子开新链
    }
}

// 求LCA（基于树链剖分）
int lcaHLD(int u, int v) {
    while (top[u] != top[v]) {
        if (depth_[top[u]] < depth_[top[v]]) { int t = u; u = v; v = t; }
        u = parent_[top[u]];
    }
    return depth_[u] < depth_[v] ? u : v;
}
```

## 7. 树的遍历

```c
// 前序遍历（先根遍历）
void preorder(int u, int par) {
    printf("%d ", u);
    for (int i = head[u]; i != -1; i = nxt[i])
        if (to[i] != par) preorder(to[i], u);
}

// 后序遍历（后根遍历）
void postorder(int u, int par) {
    for (int i = head[u]; i != -1; i = nxt[i])
        if (to[i] != par) postorder(to[i], u);
    printf("%d ", u);
}

// 欧拉序（进入和离开各记录一次）
int euler[MAXN * 2], eulerLen = 0;
void eulerTour(int u, int par) {
    euler[eulerLen++] = u;  // 进入
    for (int i = head[u]; i != -1; i = nxt[i])
        if (to[i] != par) eulerTour(to[i], u);
    euler[eulerLen++] = u;  // 离开
}
```
