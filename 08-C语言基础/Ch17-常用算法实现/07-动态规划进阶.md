# 动态规划进阶

## 1. 区间DP

区间DP以区间为状态，通过合并小区间得到大区间的最优解。通常定义 `dp[i][j]` 表示区间 `[i, j]` 的最优解。

### 石子合并问题

有n堆石子排成一排，每次合并相邻两堆，代价为两堆石子之和，求最小总代价。

```c
#include <stdio.h>
#include <string.h>
#include <limits.h>

#define MAXN 205

int stoneMerge(int stones[], int n) {
    int dp[MAXN][MAXN];
    int sum[MAXN];  // 前缀和
    sum[0] = 0;
    for (int i = 1; i <= n; i++) sum[i] = sum[i - 1] + stones[i - 1];

    // 区间长度从小到大
    for (int len = 2; len <= n; len++) {
        for (int i = 1; i + len - 1 <= n; i++) {
            int j = i + len - 1;
            dp[i][j] = INT_MAX;
            for (int k = i; k < j; k++) {
                int cost = dp[i][k] + dp[k + 1][j] + sum[j] - sum[i - 1];
                if (cost < dp[i][j]) dp[i][j] = cost;
            }
        }
    }
    return dp[1][n];
}

int main() {
    int stones[] = {3, 4, 3, 5, 1};
    int n = 5;
    printf("最小合并代价: %d\n", stoneMerge(stones, n));
    return 0;
}
```

## 2. 树形DP

在树结构上进行动态规划，通常以每个节点为根的子树作为子问题。

### 树的最大独立集

选择最多的节点使得任意两个选中的节点不相邻。

```c
#include <stdio.h>
#include <string.h>

#define MAXN 100005

int dp[MAXN][2];  // dp[u][0]: 不选u, dp[u][1]: 选u
int head[MAXN], to[MAXN], nxt[MAXN], edge_cnt = 0;

void addEdge(int u, int v) {
    to[edge_cnt] = v;
    nxt[edge_cnt] = head[u];
    head[u] = edge_cnt++;
}

void dfs(int u, int parent) {
    dp[u][0] = 0;
    dp[u][1] = 1;  // 选u，初始为1（包含u自己）
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == parent) continue;
        dfs(v, u);
        dp[u][0] += dp[v][1];  // 不选u，则v可选可不选，取最大
        dp[u][1] += dp[v][0];  // 选u，则v不能选
    }
    // 不选u时，v可以选也可以不选
    // 修正：dp[u][0] += max(dp[v][0], dp[v][1])
    dp[u][0] = 0;
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == parent) continue;
        dp[u][0] += (dp[v][0] > dp[v][1]) ? dp[v][0] : dp[v][1];
    }
}

int main() {
    memset(head, -1, sizeof(head));
    int n = 5;
    // 添加边: 1-2, 1-3, 2-4, 2-5
    addEdge(1, 2); addEdge(2, 1);
    addEdge(1, 3); addEdge(3, 1);
    addEdge(2, 4); addEdge(4, 2);
    addEdge(2, 5); addEdge(5, 2);
    dfs(1, -1);
    printf("最大独立集: %d\n",
        dp[1][0] > dp[1][1] ? dp[1][0] : dp[1][1]);
    return 0;
}
```

## 3. 状压DP（状态压缩DP）

用二进制位表示集合状态，适用于n较小（通常n<=20）的子集问题。

### 旅行商问题（TSP）

```c
#include <stdio.h>
#include <string.h>

#define INF 0x3f3f3f3f
#define MAXN 20

int dist[MAXN][MAXN];
int dp[1 << MAXN][MAXN];  // dp[S][i]: 已访问集合S，当前在i的最短路径

int tsp(int n) {
    int full = (1 << n) - 1;
    memset(dp, 0x3f, sizeof(dp));
    dp[1][0] = 0;  // 从城市0出发

    for (int S = 1; S <= full; S++) {
        for (int u = 0; u < n; u++) {
            if (!(S & (1 << u))) continue;  // u不在集合中
            if (dp[S][u] == INF) continue;
            for (int v = 0; v < n; v++) {
                if (S & (1 << v)) continue;  // v已在集合中
                int newS = S | (1 << v);
                int newCost = dp[S][u] + dist[u][v];
                if (newCost < dp[newS][v])
                    dp[newS][v] = newCost;
            }
        }
    }

    int ans = INF;
    for (int i = 0; i < n; i++)
        if (dp[full][i] + dist[i][0] < ans)
            ans = dp[full][i] + dist[i][0];
    return ans;
}
```

### 集合操作技巧

```c
// 常用位运算操作
int S = 0b10110;  // 集合{1,2,4}
S & (1 << i)      // 判断i是否在集合S中
S | (1 << i)      // 将i加入集合S
S & ~(1 << i)     // 将i从集合S中移除
S == (1 << n) - 1 // 判断S是否为全集
__builtin_popcount(S) // 计算S中元素个数
S & (S - 1)       // 移除S中最低位的1
```

## 4. 数位DP

统计满足某种条件的数字的个数，通常在区间 `[L, R]` 内。

### 不含数字4的数的个数

```c
#include <stdio.h>
#include <string.h>

int digits[20], len;
int dp[20][2];  // dp[pos][tight]: 当前位置、是否受上界约束

// 计算 [0, n] 中不含4的正整数个数
int solve(int n) {
    if (n <= 0) return 0;
    len = 0;
    while (n > 0) { digits[len++] = n % 10; n /= 10; }

    // 逐位枚举
    int memo[20][2];
    memset(memo, -1, sizeof(memo));

    // 简化版：直接计算
    // 将每一位看作9进制（去掉4）
    // 类似进制转换
    int result = 0;
    int power = 1;
    for (int i = len - 1; i >= 0; i--) {
        int d = digits[i];
        int contrib = (d > 4) ? d - 1 : d;  // 跳过4
        result += contrib * power;
        power *= 9;
        if (d == 4) return result;  // 包含4，后续都无效
    }
    return result;
}
```

## 5. 背包DP扩展

### 分组背包

```c
#include <stdio.h>

// 每组最多选一个物品
int groupKnapsack(int W, int groups[][], int weights[][],
                  int values[][], int groupSize[], int ng) {
    int dp[W + 1];
    for (int w = 0; w <= W; w++) dp[w] = 0;

    for (int g = 0; g < ng; g++) {
        // 注意：必须逆序更新dp
        for (int w = W; w >= 0; w--) {
            for (int k = 0; k < groupSize[g]; k++) {
                if (weights[g][k] <= w) {
                    int v = dp[w - weights[g][k]] + values[g][k];
                    if (v > dp[w]) dp[w] = v;
                }
            }
        }
    }
    return dp[W];
}
```

## 6. DP优化技巧

### 单调队列优化

```c
// 滑动窗口最大值用于DP优化
// 适用于形如 dp[i] = min/max(dp[j]) + cost 的转移
// 其中 j 在一个滑动窗口 [i-k, i-1] 内

#include <stdio.h>

#define MAXN 100005

int deque[MAXN];  // 双端队列存下标

int slidingWindowDP(int arr[], int n, int k) {
    int dp[MAXN];
    int front = 0, back = 0;

    for (int i = 0; i < n; i++) {
        // 移除超出窗口的元素
        while (front < back && deque[front] < i - k) front++;
        // dp转移
        dp[i] = arr[i] + (front < back ? dp[deque[front]] : 0);
        // 维护单调性
        while (front < back && dp[deque[back - 1]] <= dp[i]) back--;
        deque[back++] = i;
    }
    return dp[n - 1];
}
```
