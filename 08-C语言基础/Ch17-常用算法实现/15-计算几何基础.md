# 计算几何基础

## 1. 基本概念与数据结构

计算几何研究几何对象的算法，常用的数据结构是点和向量。

```c
#include <stdio.h>
#include <math.h>
#include <stdlib.h>

#define EPS 1e-9

typedef struct {
    double x, y;
} Point;

typedef Point Vector;

// 向量减法
Vector sub(Point a, Point b) { return (Vector){a.x - b.x, a.y - b.y}; }

// 向量加法
Vector add(Vector a, Vector b) { return (Vector){a.x + b.x, a.y + b.y}; }

// 向量数乘
Vector mul(Vector v, double k) { return (Vector){v.x * k, v.y * k}; }

// 向量长度
double length(Vector v) { return sqrt(v.x * v.x + v.y * v.y); }

// 两点距离
double dist(Point a, Point b) { return length(sub(a, b)); }
```

## 2. 叉积（Cross Product）

叉积是计算几何的核心，用于判断点的位置关系、线段相交等。

```c
// 叉积: a x b = a.x*b.y - a.y*b.x
// 结果 > 0: b在a的逆时针方向
// 结果 < 0: b在a的顺时针方向
// 结果 = 0: a和b共线
double cross(Vector a, Vector b) {
    return a.x * b.y - a.y * b.x;
}

// 点积
double dot(Vector a, Vector b) {
    return a.x * b.x + a.y * b.y;
}

// 判断点c在向量ab的哪一侧
int orientation(Point a, Point b, Point c) {
    double val = cross(sub(b, a), sub(c, a));
    if (val > EPS) return 1;   // 逆时针
    if (val < -EPS) return -1; // 顺时针
    return 0;                   // 共线
}
```

## 3. 线段相交判断

```c
// 判断点q是否在线段pr上
int onSegment(Point p, Point q, Point r) {
    return q.x <= fmax(p.x, r.x) + EPS && q.x >= fmin(p.x, r.x) - EPS &&
           q.y <= fmax(p.y, r.y) + EPS && q.y >= fmin(p.y, r.y) - EPS;
}

// 判断线段p1p2和p3p4是否相交
int segmentsIntersect(Point p1, Point p2, Point p3, Point p4) {
    int o1 = orientation(p1, p2, p3);
    int o2 = orientation(p1, p2, p4);
    int o3 = orientation(p3, p4, p1);
    int o4 = orientation(p3, p4, p2);

    // 一般情况：两线段互相跨越
    if (o1 != o2 && o3 != o4) return 1;

    // 特殊情况：共线且点在线段上
    if (o1 == 0 && onSegment(p1, p3, p2)) return 1;
    if (o2 == 0 && onSegment(p1, p4, p2)) return 1;
    if (o3 == 0 && onSegment(p3, p1, p4)) return 1;
    if (o4 == 0 && onSegment(p3, p2, p4)) return 1;

    return 0;
}
```

## 4. 点到线段距离

```c
double pointToSegmentDist(Point p, Point a, Point b) {
    Vector ap = sub(p, a), ab = sub(b, a);
    double t = dot(ap, ab) / dot(ab, ab);
    if (t < 0) return dist(p, a);
    if (t > 1) return dist(p, b);
    // 投影点
    Point proj = add(a, mul(ab, t));
    return dist(p, proj);
}
```

## 5. 凸包（Graham Scan）

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

typedef struct { double x, y; } Point;

Point pivot;  // 基准点

double crossO(Point o, Point a, Point b) {
    return (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x);
}

double distP(Point a, Point b) {
    return sqrt((a.x - b.x) * (a.x - b.x) + (a.y - b.y) * (a.y - b.y));
}

int cmpConvex(const void *a, const void *b) {
    Point *pa = (Point *)a, *pb = (Point *)b;
    double c = crossO(pivot, *pa, *pb);
    if (fabs(c) < 1e-9) return distP(pivot, *pa) < distP(pivot, *pb) ? -1 : 1;
    return c > 0 ? -1 : 1;
}

// Graham扫描法求凸包
int convexHull(Point pts[], int n, Point hull[]) {
    if (n < 3) return 0;

    // 找最左下方的点
    int minIdx = 0;
    for (int i = 1; i < n; i++)
        if (pts[i].y < pts[minIdx].y ||
            (pts[i].y == pts[minIdx].y && pts[i].x < pts[minIdx].x))
            minIdx = i;

    // 交换到首位
    Point temp = pts[0]; pts[0] = pts[minIdx]; pts[minIdx] = temp;
    pivot = pts[0];

    // 按极角排序
    qsort(pts + 1, n - 1, sizeof(Point), cmpConvex);

    // Graham扫描
    int top = 2;
    hull[0] = pts[0]; hull[1] = pts[1]; hull[2] = pts[2];
    for (int i = 3; i < n; i++) {
        while (top >= 1 && crossO(hull[top - 1], hull[top], pts[i]) <= 0)
            top--;
        hull[++top] = pts[i];
    }
    return top + 1;  // 凸包顶点数
}
```

## 6. 判断点在多边形内

```c
// 射线法：从点p向右发射射线，计算与多边形边的交点数
int pointInPolygon(Point p, Point poly[], int n) {
    int count = 0;
    for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        // 检查射线是否与边poly[i]-poly[j]相交
        if ((poly[i].y > p.y) != (poly[j].y > p.y)) {
            double xinters = (poly[j].y - p.y) * (poly[i].x - poly[j].x)
                           / (poly[i].y - poly[j].y) + poly[j].x;
            if (p.x < xinters) count++;
        }
    }
    return count % 2;  // 奇数次在内部
}
```

## 7. 两圆相交面积

```c
double circleIntersectionArea(double r1, double r2, double d) {
    if (d >= r1 + r2) return 0;       // 不相交
    if (d <= fabs(r1 - r2)) {         // 包含
        double r = r1 < r2 ? r1 : r2;
        return M_PI * r * r;
    }
    double a1 = 2 * acos((d * d + r1 * r1 - r2 * r2) / (2 * d * r1));
    double a2 = 2 * acos((d * d + r2 * r2 - r1 * r1) / (2 * d * r2));
    return 0.5 * (r1 * r1 * (a1 - sin(a1)) + r2 * r2 * (a2 - sin(a2)));
}
```

## 8. 凸包直径（旋转卡壳）

```c
// 求凸包的最远点对（直径）
double convexDiameter(Point hull[], int n) {
    if (n <= 1) return 0;
    double maxDist = 0;
    int j = 1;
    for (int i = 0; i < n; i++) {
        int ni = (i + 1) % n;
        while (fabs(crossO(hull[i], hull[ni], hull[(j + 1) % n])) >
               fabs(crossO(hull[i], hull[ni], hull[j])))
            j = (j + 1) % n;
        double d1 = distP(hull[i], hull[j]);
        double d2 = distP(hull[ni], hull[j]);
        if (d1 > maxDist) maxDist = d1;
        if (d2 > maxDist) maxDist = d2;
    }
    return maxDist;
}
```
