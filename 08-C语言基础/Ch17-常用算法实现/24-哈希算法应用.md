# 哈希算法应用

## 1. 哈希基础

哈希表通过哈希函数将键映射到数组下标，实现O(1)的平均查找时间。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define TABLE_SIZE 100003

typedef struct Node {
    int key, value;
    struct Node *next;
} Node;

Node *hashTable[TABLE_SIZE];

int hashFunc(int key) {
    return ((key % TABLE_SIZE) + TABLE_SIZE) % TABLE_SIZE;
}

void insert(int key, int value) {
    int idx = hashFunc(key);
    Node *node = (Node *)malloc(sizeof(Node));
    node->key = key; node->value = value;
    node->next = hashTable[idx];
    hashTable[idx] = node;
}

int search(int key) {
    int idx = hashFunc(key);
    for (Node *p = hashTable[idx]; p; p = p->next)
        if (p->key == key) return p->value;
    return -1;
}
```

## 2. 字符串哈希

将字符串映射为整数，用于快速比较字符串是否相等。

```c
#include <stdio.h>
#include <string.h>

#define BASE 131
#define MOD 1000000007
#define MAXN 1000005

unsigned long long h[MAXN], pw[MAXN];  // 哈希值和幂次

// 预处理字符串哈希
void buildHash(const char *s, int n) {
    pw[0] = 1;
    for (int i = 1; i <= n; i++) {
        h[i] = h[i - 1] * BASE + s[i - 1];
        pw[i] = pw[i - 1] * BASE;
    }
}

// 获取子串 [l, r] 的哈希值（1-indexed）
unsigned long long getHash(int l, int r) {
    return h[r] - h[l - 1] * pw[r - l + 1];
}

// 判断两个子串是否相等
int isEqual(int l1, int r1, int l2, int r2) {
    return getHash(l1, r1) == getHash(l2, r2);
}
```

## 3. 滚动哈希

在文本中动态滑动窗口时更新哈希值。

```c
#include <stdio.h>
#include <string.h>

#define BASE 257
#define MOD 1000000007

// Rabin-Karp字符串匹配
void rabinKarp(const char *text, const char *pattern) {
    int n = strlen(text), m = strlen(pattern);
    if (m > n) return;

    // 计算pattern的哈希值
    long long pHash = 0;
    for (int i = 0; i < m; i++)
        pHash = (pHash * BASE + pattern[i]) % MOD;

    // 计算第一个窗口的哈希值
    long long tHash = 0;
    long long baseM = 1;  // BASE^m
    for (int i = 0; i < m; i++) {
        tHash = (tHash * BASE + text[i]) % MOD;
        baseM = baseM * BASE % MOD;
    }

    // 滑动窗口
    for (int i = 0; i <= n - m; i++) {
        if (tHash == pHash) {
            // 哈希匹配，逐字符验证
            int match = 1;
            for (int j = 0; j < m; j++) {
                if (text[i + j] != pattern[j]) { match = 0; break; }
            }
            if (match) printf("匹配位置: %d\n", i);
        }
        // 滑动窗口
        if (i < n - m) {
            tHash = ((tHash - text[i] * baseM % MOD) * BASE + text[i + m]) % MOD;
            if (tHash < 0) tHash += MOD;
        }
    }
}
```

## 4. 双哈希（防冲突）

```c
#define BASE1 131
#define BASE2 233
#define MOD1 1000000007
#define MOD2 1000000009

typedef struct {
    long long h1, h2;
} HashPair;

HashPair h[MAXN];
HashPair pw[MAXN];

void buildDoubleHash(const char *s, int n) {
    pw[0].h1 = pw[0].h2 = 1;
    for (int i = 1; i <= n; i++) {
        h[i].h1 = (h[i-1].h1 * BASE1 + s[i-1]) % MOD1;
        h[i].h2 = (h[i-1].h2 * BASE2 + s[i-1]) % MOD2;
        pw[i].h1 = pw[i-1].h1 * BASE1 % MOD1;
        pw[i].h2 = pw[i-1].h2 * BASE2 % MOD2;
    }
}

int isSubstringEqual(int l1, int r1, int l2, int r2) {
    long long hash1_a = (h[r1].h1 - h[l1-1].h1 * pw[r1-l1+1].h1 % MOD1 + MOD1 * 2) % MOD1;
    long long hash2_a = (h[r2].h1 - h[l2-1].h1 * pw[r2-l2+1].h1 % MOD1 + MOD1 * 2) % MOD1;
    return hash1_a == hash2_a;  // 实际需要同时检查两组哈希
}
```

## 5. 一致性哈希（简介）

一致性哈希常用于分布式系统中，将数据均匀分布到多个节点。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define RING_SIZE 1000000

typedef struct {
    int hash;
    char name[50];
} VirtualNode;

// 简化版的一致性哈希环
VirtualNode ring[10000];
int ringSize = 0;

unsigned int murmurHash(const char *key) {
    unsigned int h = 0;
    while (*key) {
        h ^= *key++;
        h *= 0x5bd1e995;
        h ^= h >> 15;
    }
    return h;
}

// 根据key找到对应的节点
const char *getNode(const char *key) {
    unsigned int h = murmurHash(key) % RING_SIZE;
    // 在哈希环上顺时针查找第一个节点
    for (int i = 0; i < ringSize; i++) {
        if (ring[i].hash >= (int)h)
            return ring[i].name;
    }
    return ring[0].name;  // 绕回起点
}
```

## 6. 哈希表在算法中的应用

```c
// 两数之和：O(n) 解法
void twoSum(int arr[], int n, int target) {
    // 简化版哈希表（实际应用中应使用完整的哈希表实现）
    int map[2000005] = {0};  // value -> index
    for (int i = 0; i < n; i++) {
        int complement = target - arr[i] + 1000000;
        if (complement >= 0 && complement < 2000005 && map[complement]) {
            printf("索引: %d, %d\n", map[complement] - 1, i);
            return;
        }
        map[arr[i] + 1000000] = i + 1;
    }
}

// 字符串去重
int uniqueStrings(const char *strs[], int n) {
    int count = 0;
    unsigned long long seen[10000];
    for (int i = 0; i < n; i++) {
        unsigned long long hash = 0;
        for (int j = 0; strs[i][j]; j++)
            hash = hash * 131 + strs[i][j];
        int found = 0;
        for (int j = 0; j < count; j++)
            if (seen[j] == hash) { found = 1; break; }
        if (!found) seen[count++] = hash;
    }
    return count;
}
```

## 7. 哈希冲突处理方法

| 方法 | 原理 | 优缺点 |
|------|------|--------|
| 链地址法 | 每个桶用链表存储 | 实现简单，需要额外空间 |
| 开放地址法 | 冲突时探测下一个空位 | 空间利用率高，聚集现象 |
| 双重哈希 | 用第二个哈希函数确定步长 | 减少聚集，计算量大 |
| Robin Hood哈希 | 贫富均等：短距离的给长距离让位 | 方差小，性能稳定 |
