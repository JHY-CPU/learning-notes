# 字符串匹配算法详解

## 1. 暴力匹配

逐字符比较，时间复杂度 O(n*m)。

```c
#include <stdio.h>
#include <string.h>

int bruteForce(const char *text, const char *pattern) {
    int n = strlen(text), m = strlen(pattern);
    for (int i = 0; i <= n - m; i++) {
        int j;
        for (j = 0; j < m; j++) {
            if (text[i + j] != pattern[j]) break;
        }
        if (j == m) return i;  // 匹配成功
    }
    return -1;  // 未找到
}
```

## 2. KMP算法

KMP算法通过预处理模式串的**next数组**（前缀函数），避免重复比较，时间复杂度 O(n+m)。

### next数组的含义

`next[j]` 表示 `pattern[0..j-1]` 的最长相等前后缀长度。

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

void computeNext(const char *pattern, int *next, int m) {
    next[0] = -1;
    int j = -1;
    for (int i = 1; i < m; i++) {
        while (j >= 0 && pattern[j + 1] != pattern[i])
            j = next[j];
        if (pattern[j + 1] == pattern[i]) j++;
        next[i] = j;
    }
}

int kmp(const char *text, const char *pattern) {
    int n = strlen(text), m = strlen(pattern);
    if (m == 0) return 0;
    int *next = (int *)malloc(m * sizeof(int));
    computeNext(pattern, next, m);

    int j = -1;
    for (int i = 0; i < n; i++) {
        while (j >= 0 && pattern[j + 1] != text[i])
            j = next[j];
        if (pattern[j + 1] == text[i]) j++;
        if (j == m - 1) {
            free(next);
            return i - m + 1;  // 返回匹配位置
        }
    }
    free(next);
    return -1;
}

// 所有匹配位置
void kmpAll(const char *text, const char *pattern) {
    int n = strlen(text), m = strlen(pattern);
    int *next = (int *)malloc(m * sizeof(int));
    computeNext(pattern, next, m);

    int j = -1;
    for (int i = 0; i < n; i++) {
        while (j >= 0 && pattern[j + 1] != text[i])
            j = next[j];
        if (pattern[j + 1] == text[i]) j++;
        if (j == m - 1) {
            printf("匹配位置: %d\n", i - m + 1);
            j = next[j];  // 继续寻找下一个
        }
    }
    free(next);
}
```

## 3. BM算法（Boyer-Moore）

BM算法从右向左比较，利用**坏字符**和**好后缀**规则跳过不可能匹配的位置。

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define ALPHABET_SIZE 256

// 坏字符规则：记录每个字符在模式串中最右出现位置
void badCharHeuristic(const char *pattern, int m, int badChar[ALPHABET_SIZE]) {
    for (int i = 0; i < ALPHABET_SIZE; i++)
        badChar[i] = -1;
    for (int i = 0; i < m; i++)
        badChar[(unsigned char)pattern[i]] = i;
}

int bmSearch(const char *text, const char *pattern) {
    int n = strlen(text), m = strlen(pattern);
    int badChar[ALPHABET_SIZE];
    badCharHeuristic(pattern, m, badChar);

    int shift = 0;
    while (shift <= n - m) {
        int j = m - 1;
        // 从右向左比较
        while (j >= 0 && pattern[j] == text[shift + j])
            j--;
        if (j < 0) return shift;  // 匹配成功

        // 坏字符规则: shift至少移动1
        int badShift = j - badChar[(unsigned char)text[shift + j]];
        if (badShift < 1) badShift = 1;
        shift += badShift;
    }
    return -1;
}
```

## 4. Sunday算法

Sunday算法在不匹配时查看文本串中模式串末尾的下一个字符。

```c
#include <stdio.h>
#include <string.h>

#define ALPHABET_SIZE 256

int sundaySearch(const char *text, const char *pattern) {
    int n = strlen(text), m = strlen(pattern);
    int shift[ALPHABET_SIZE];

    // 预处理：每个字符到模式串末尾的距离+1
    for (int i = 0; i < ALPHABET_SIZE; i++)
        shift[i] = m + 1;
    for (int i = 0; i < m; i++)
        shift[(unsigned char)pattern[i]] = m - i;

    int i = 0;
    while (i <= n - m) {
        int j = 0;
        while (j < m && text[i + j] == pattern[j]) j++;
        if (j == m) return i;

        // 查看模式串末尾下一个字符
        if (i + m >= n) break;
        i += shift[(unsigned char)text[i + m]];
    }
    return -1;
}
```

## 5. AC自动机（Aho-Corasick）

AC自动机用于多模式串同时匹配。

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define ALPHABET_SIZE 26
#define MAXNODES 500005

typedef struct {
    int next[ALPHABET_SIZE];
    int fail;     // 失败指针
    int count;    // 匹配模式串数量
} Node;

Node trie[MAXNODES];
int nodeCount = 1;

void insert(const char *str) {
    int cur = 0;
    for (int i = 0; str[i]; i++) {
        int c = str[i] - 'a';
        if (!trie[cur].next[c])
            trie[cur].next[c] = nodeCount++;
        cur = trie[cur].next[c];
    }
    trie[cur].count++;
}

void buildAC() {
    int queue[MAXNODES], front = 0, rear = 0;
    for (int c = 0; c < ALPHABET_SIZE; c++) {
        if (trie[0].next[c]) {
            trie[trie[0].next[c]].fail = 0;
            queue[rear++] = trie[0].next[c];
        }
    }
    while (front < rear) {
        int cur = queue[front++];
        for (int c = 0; c < ALPHABET_SIZE; c++) {
            if (trie[cur].next[c]) {
                trie[trie[cur].next[c]].fail = trie[trie[cur].fail].next[c];
                queue[rear++] = trie[cur].next[c];
            } else {
                trie[cur].next[c] = trie[trie[cur].fail].next[c];
            }
        }
    }
}

int query(const char *text) {
    int cur = 0, result = 0;
    for (int i = 0; text[i]; i++) {
        cur = trie[cur].next[text[i] - 'a'];
        for (int t = cur; t && trie[t].count != -1; t = trie[t].fail) {
            result += trie[t].count;
            trie[t].count = -1;  // 防重复计数
        }
    }
    return result;
}
```

## 6. 算法对比

| 算法 | 预处理 | 匹配 | 适用场景 |
|------|-------|------|---------|
| 暴力 | O(1) | O(n*m) | 短模式串 |
| KMP | O(m) | O(n+m) | 单模式串 |
| BM | O(m+σ) | 平均O(n/m) | 长文本、字母表大 |
| Sunday | O(m+σ) | 平均O(n/m) | 实际中很快 |
| AC自动机 | O(Σm) | O(n+输出) | 多模式串 |
