# 数学公式与技巧

## 1. 排列组合

### 阶乘与排列数

```c
#include <stdio.h>

#define MOD 1000000007

// 阶乘（取模）
long long fact[1000005], invFact[1000005];

long long fastPow(long long a, long long b, long long mod) {
    long long res = 1;
    a %= mod;
    while (b > 0) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

// 预处理阶乘和逆元
void initComb(int n) {
    fact[0] = 1;
    for (int i = 1; i <= n; i++)
        fact[i] = fact[i - 1] * i % MOD;
    invFact[n] = fastPow(fact[n], MOD - 2, MOD);
    for (int i = n - 1; i >= 0; i--)
        invFact[i] = invFact[i + 1] * (i + 1) % MOD;
}

// 组合数 C(n, m) = n! / (m! * (n-m)!)
long long C(int n, int m) {
    if (m < 0 || m > n) return 0;
    return fact[n] % MOD * invFact[m] % MOD * invFact[n - m] % MOD;
}

// 排列数 A(n, m) = n! / (n-m)!
long long A(int n, int m) {
    if (m < 0 || m > n) return 0;
    return fact[n] % MOD * invFact[n - m] % MOD;
}
```

### 杨辉三角求组合数

```c
// C(n, m) 用杨辉三角预处理（适用于n较小）
long long comb[1005][1005];
void initPascal(int n) {
    for (int i = 0; i <= n; i++) {
        comb[i][0] = 1;
        for (int j = 1; j <= i; j++)
            comb[i][j] = (comb[i-1][j-1] + comb[i-1][j]) % MOD;
    }
}
```

## 2. 卡特兰数

卡特兰数广泛用于计数问题：括号匹配、二叉树计数、路径问题等。

```c
// 卡特兰数: C(n) = C(2n,n) / (n+1)
// 递推: C(n) = sum(C(i)*C(n-1-i)) for i=0..n-1

long long catalan[1005];

void initCatalan(int n) {
    catalan[0] = catalan[1] = 1;
    for (int i = 2; i <= n; i++) {
        catalan[i] = 0;
        for (int j = 0; j < i; j++)
            catalan[i] = (catalan[i] + catalan[j] * catalan[i-1-j]) % MOD;
    }
}

// 用组合数公式求卡特兰数
long long catalanByComb(int n) {
    return C(2 * n, n) % MOD * fastPow(n + 1, MOD - 2, MOD) % MOD;
}

// 应用场景：
// 1. n对括号的合法匹配方案数 = C(n)
// 2. n个节点的二叉树形态数 = C(n)
// 3. 不穿过对角线的网格路径数 = C(n)
// 4. 凸n+1边形的三角剖分数 = C(n-1)
// 5. n个元素的栈排列数 = C(n)
```

## 3. 斯特林数

### 第一类斯特林数

```c
// s(n, k): 将n个元素分成k个非空圆排列的方案数
// 递推: s(n, k) = s(n-1, k-1) + (n-1) * s(n-1, k)
long long stirling1[55][55];

void initStirling1(int n) {
    stirling1[0][0] = 1;
    for (int i = 1; i <= n; i++) {
        stirling1[i][0] = 0;
        for (int j = 1; j <= i; j++)
            stirling1[i][j] = (stirling1[i-1][j-1] + (i-1) * stirling1[i-1][j]) % MOD;
    }
}
```

### 第二类斯特林数

```c
// S(n, k): 将n个元素分成k个非空子集的方案数
// 递推: S(n, k) = S(n-1, k-1) + k * S(n-1, k)
long long stirling2[55][55];

void initStirling2(int n) {
    stirling2[0][0] = 1;
    for (int i = 1; i <= n; i++) {
        stirling2[i][0] = 0;
        for (int j = 1; j <= i; j++)
            stirling2[i][j] = (stirling2[i-1][j-1] + j * stirling2[i-1][j]) % MOD;
    }
}
```

## 4. 容斥原理

容斥原理用于求多个集合的并集大小。

```c
#include <stdio.h>

// 求 [1, n] 中能被 a1, a2, ..., ak 中任意一个整除的数的个数
long long inclusionExclusion(int n, int a[], int k) {
    long long result = 0;
    // 枚举所有非空子集
    for (int mask = 1; mask < (1 << k); mask++) {
        long long lcm = 1;
        int bits = 0;
        for (int i = 0; i < k; i++) {
            if (mask & (1 << i)) {
                bits++;
                long long g = gcd(lcm, a[i]);
                lcm = lcm / g * a[i];
                if (lcm > n) break;  // 超过n，贡献为0
            }
        }
        if (lcm > n) continue;
        if (bits % 2 == 1)
            result += n / lcm;
        else
            result -= n / lcm;
    }
    return result;
}

// 应用：求 [1, n] 中与m互质的数的个数
// = n - inclusionExclusion(n, prime_factors_of_m, count)
```

## 5. 斐波那契与卢卡斯数

```c
// 斐波那契数：F(0)=0, F(1)=1, F(n)=F(n-1)+F(n-2)
// 通项公式（矩阵快速幂）O(log n)
typedef struct { long long a[2][2]; } Matrix;

Matrix matMul(Matrix A, Matrix B, long long mod) {
    Matrix C = {{{0, 0}, {0, 0}}};
    for (int i = 0; i < 2; i++)
        for (int j = 0; j < 2; j++)
            for (int k = 0; k < 2; k++)
                C.a[i][j] = (C.a[i][j] + A.a[i][k] * B.a[k][j]) % mod;
    return C;
}

long long fibMatrix(long long n, long long mod) {
    if (n <= 1) return n;
    Matrix base = {{{1, 1}, {1, 0}}};
    Matrix result = {{{1, 0}, {0, 1}}};  // 单位矩阵
    n--;
    while (n > 0) {
        if (n & 1) result = matMul(result, base, mod);
        base = matMul(base, base, mod);
        n >>= 1;
    }
    return result.a[0][0];
}
```

## 6. 莫比乌斯函数

```c
#include <string.h>

int mu[1000005];
int primes[1000005], primeCount;
bool notPrime[1000005];

void mobiusSieve(int n) {
    memset(notPrime, false, sizeof(notPrime));
    mu[1] = 1;
    primeCount = 0;
    for (int i = 2; i <= n; i++) {
        if (!notPrime[i]) {
            primes[primeCount++] = i;
            mu[i] = -1;
        }
        for (int j = 0; j < primeCount && (long long)i * primes[j] <= n; j++) {
            notPrime[i * primes[j]] = true;
            if (i % primes[j] == 0) {
                mu[i * primes[j]] = 0;  // 含平方因子
                break;
            }
            mu[i * primes[j]] = -mu[i];
        }
    }
}
```
