# 23 - 函数模板思想

## 一、C 语言没有泛型

C++ 有模板（template），Java 有泛型（generics），但 C 语言**没有泛型机制**。在 C 中实现通用算法需要借助以下技术：

1. `void*` 通用指针
2. 函数指针
3. 宏（预处理期的"泛型"）
4. C11 的 `_Generic`

## 二、用 void* 和函数指针模拟泛型排序

### 2.1 通用排序函数的实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 通用冒排排序
void generic_sort(void *base, size_t nmemb, size_t size,
                  int (*cmp)(const void*, const void*)) {
    char *arr = (char*)base;

    for (size_t i = 0; i < nmemb - 1; i++) {
        for (size_t j = 0; j < nmemb - 1 - i; j++) {
            char *a = arr + j * size;
            char *b = arr + (j + 1) * size;
            if (cmp(a, b) > 0) {
                // 交换两个元素（逐字节交换）
                for (size_t k = 0; k < size; k++) {
                    char tmp = a[k];
                    a[k] = b[k];
                    b[k] = tmp;
                }
            }
        }
    }
}
```

### 2.2 排序整数

```c
int cmp_int(const void *a, const void *b) {
    return (*(const int*)a) - (*(const int*)b);
}

int main() {
    int arr[] = {5, 2, 8, 1, 9, 3};
    int n = 6;

    generic_sort(arr, n, sizeof(int), cmp_int);

    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);  // 1 2 3 5 8 9
    printf("\n");
    return 0;
}
```

### 2.3 排序浮点数

```c
int cmp_double(const void *a, const void *b) {
    double da = *(const double*)a;
    double db = *(const double*)b;
    if (da < db) return -1;
    if (da > db) return 1;
    return 0;
}

int main() {
    double arr[] = {3.14, 1.41, 2.72, 0.58};
    generic_sort(arr, 4, sizeof(double), cmp_double);
    for (int i = 0; i < 4; i++)
        printf("%.2f ", arr[i]);  // 0.58 1.41 2.72 3.14
    return 0;
}
```

### 2.4 排序字符串

```c
int cmp_string(const void *a, const void *b) {
    return strcmp(*(const char**)a, *(const char**)b);
}

int main() {
    const char *names[] = {"Charlie", "Alice", "Bob"};
    generic_sort(names, 3, sizeof(char*), cmp_string);

        printf("%s\n", names[i]);  // Alice, Bob, Charlie
```

## 三、通用搜索函数

```c
// 通用线性搜索
void* generic_find(const void *base, size_t nmemb, size_t size,
                   const void *target,
                   int (*cmp)(const void*, const void*)) {
    const char *arr = (const char*)base;
    for (size_t i = 0; i < nmemb; i++) {
        if (cmp(arr + i * size, target) == 0) {
            return (void*)(arr + i * size);
        }
    }
    return NULL;
}

// 通用二分查找
void* generic_bsearch(const void *key, const void *base,
                      size_t nmemb, size_t size,
                      int (*cmp)(const void*, const void*)) {
    const char *arr = (const char*)base;
    size_t lo = 0, hi = nmemb;

    while (lo < hi) {
        size_t mid = lo + (hi - lo) / 2;
        const char *elem = arr + mid * size;
        int result = cmp(key, elem);

        if (result == 0) return (void*)elem;
        if (result < 0) hi = mid;
        else lo = mid + 1;
    }
    return NULL;
}
```

## 四、通用数据结构

### 4.1 泛型动态数组

```c
#include <stdlib.h>
#include <string.h>

typedef struct {
    void *data;
    size_t elem_size;
    size_t length;
    size_t capacity;
} GenericArray;

void array_init(GenericArray *arr, size_t elem_size) {
    arr->data = malloc(elem_size * 4);
    arr->elem_size = elem_size;
    arr->length = 0;
    arr->capacity = 4;
}

void array_push(GenericArray *arr, const void *elem) {
    if (arr->length >= arr->capacity) {
        arr->capacity *= 2;
        arr->data = realloc(arr->data, arr->elem_size * arr->capacity);
    }
    memcpy((char*)arr->data + arr->length * arr->elem_size,
           elem, arr->elem_size);
    arr->length++;
}

void* array_get(GenericArray *arr, size_t index) {
    if (index >= arr->length) return NULL;
    return (char*)arr->data + index * arr->elem_size;
}

void array_free(GenericArray *arr) {
    free(arr->data);
    arr->data = NULL;
    arr->length = 0;
}

// 使用示例
int main() {
    GenericArray arr;
    array_init(&arr, sizeof(int));

    for (int i = 0; i < 10; i++)
        array_push(&arr, &i);

    for (size_t i = 0; i < arr.length; i++)
        printf("%d ", *(int*)array_get(&arr, i));
    printf("\n");

    array_free(&arr);
    return 0;
}
```

## 五、C11 的 _Generic 关键字

### 5.1 类型选择表达式

```c
#include <stdio.h>

// _Generic 类似于编译期的 switch
#define abs(x) _Generic((x), \
    int: abs_int, \
    double: abs_double, \
    float: abs_float \
)(x)

int abs_int(int x) { return x < 0 ? -x : x; }
double abs_double(double x) { return x < 0 ? -x : x; }
float abs_float(float x) { return x < 0 ? -x : x; }

int main() {
    printf("%d\n", abs(-5));         // 调用 abs_int
    printf("%.2f\n", abs(-3.14));    // 调用 abs_double
    printf("%.2f\n", abs(-2.71f));   // 调用 abs_float
    return 0;
}
```

### 5.2 类型名称宏

```c
#define type_name(x) _Generic((x), \
    int: "int", \
    long: "long", \
    double: "double", \
    float: "float", \
    char*: "char*", \
    default: "unknown" \
)

int main() {
    int a = 0;
    double b = 0;
    printf("a 的类型: %s\n", type_name(a));  // "int"
    printf("b 的类型: %s\n", type_name(b));  // "double"
    return 0;
}
```

## 六、要点总结

> **关键点：**
> 1. C 语言没有泛型，用 `void*` + 函数指针模拟泛型算法。
> 2. `void*` 作为通用指针，配合 `size_t` 元素大小和比较函数实现类型无关操作。
> 3. 标准库的 `qsort` 和 `bsearch` 就是这种模式的典型应用。
> 4. 通用数据结构（动态数组、链表等）通过 `void*` 存储任意类型数据。
> 5. C11 的 `_Generic` 提供了编译期类型选择的能力。
> 6. 宏可以在预处理期实现"泛型"，但有副作用风险。
> 7. `memcpy` 逐字节复制是 `void*` 泛型操作的基础。
