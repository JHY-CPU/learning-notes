# 16 - 命令行参数处理

## 一、argc/argv 基础解析

### 1.1 遍历所有参数

```c
#include <stdio.h>

int main(int argc, char *argv[]) {
    printf("程序名: %s\n", argv[0]);
    printf("共 %d 个参数:\n", argc - 1);

    for (int i = 1; i < argc; i++) {
        printf("  参数 %d: %s\n", i, argv[i]);
    }

    return 0;
}
```

### 1.2 参数验证

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <文件名> [-v]\n", argv[0]);
        return 1;
    }

    const char *filename = argv[1];
    int verbose = 0;

    // 检查可选参数
    for (int i = 2; i < argc; i++) {
        if (strcmp(argv[i], "-v") == 0) {
            verbose = 1;
        }
    }

    if (verbose) {
        printf("处理文件: %s\n", filename);
    }

    return 0;
}
```

## 二、手动解析选项

### 2.1 简单的选项处理

```c
#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[]) {
    int show_all = 0;
    int show_long = 0;
    const char *target = NULL;

    for (int i = 1; i < argc; i++) {
        if (strcmp(argv[i], "-a") == 0 || strcmp(argv[i], "--all") == 0) {
            show_all = 1;
        } else if (strcmp(argv[i], "-l") == 0 || strcmp(argv[i], "--long") == 0) {
            show_long = 1;
        } else if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
            printf("用法: %s [-a] [-l] [-h] [目标]\n", argv[0]);
            printf("  -a, --all    显示所有\n");
            printf("  -l, --long   长格式显示\n");
            printf("  -h, --help   显示帮助\n");
            return 0;
        } else if (argv[i][0] != '-') {
            target = argv[i];    // 非选项参数
        } else {
            fprintf(stderr, "未知选项: %s\n", argv[i]);
            return 1;
        }
    }

    printf("show_all=%d, show_long=%d, target=%s\n",
           show_all, show_long, target ? target : "(无)");
    return 0;
}
```

### 2.2 带参数的选项

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char *argv[]) {
    int port = 8080;
    const char *host = "localhost";
    const char *output = NULL;

    for (int i = 1; i < argc; i++) {
        if (strcmp(argv[i], "-p") == 0 || strcmp(argv[i], "--port") == 0) {
            if (++i >= argc) {
                fprintf(stderr, "-p 需要一个参数\n");
                return 1;
            }
            port = atoi(argv[i]);
        } else if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--host") == 0) {
            if (++i >= argc) {
                fprintf(stderr, "-h 需要一个参数\n");
                return 1;
            }
            host = argv[i];
        } else if (strcmp(argv[i], "-o") == 0 || strcmp(argv[i], "--output") == 0) {
            if (++i >= argc) {
                fprintf(stderr, "-o 需要一个参数\n");
                return 1;
            }
            output = argv[i];
        }
    }

    printf("Host: %s, Port: %d, Output: %s\n",
           host, port, output ? output : "(stdout)");
    return 0;
}
```

## 三、使用 getopt（POSIX）

### 3.1 getopt 函数

```c
#include <stdio.h>
#include <unistd.h>  // POSIX 标准头文件

int main(int argc, char *argv[]) {
    int opt;
    int verbose = 0;
    const char *output = NULL;

    // "vho:" 表示接受 -v, -h, -o（冒号表示 -o 需要参数）
    while ((opt = getopt(argc, argv, "vho:")) != -1) {
        switch (opt) {
            case 'v':
                verbose = 1;
                break;
            case 'h':
                printf("用法: %s [-v] [-h] [-o 文件]\n", argv[0]);
                return 0;
            case 'o':
                output = optarg;  // optarg 指向选项的参数
                break;
            default:
                fprintf(stderr, "用法: %s [-v] [-h] [-o 文件]\n", argv[0]);
                return 1;
        }
    }

    // optind 是第一个非选项参数的索引
    for (int i = optind; i < argc; i++) {
        printf("非选项参数: %s\n", argv[i]);
    }

    printf("verbose=%d, output=%s\n", verbose, output ? output : "(无)");
    return 0;
}
```

### 3.2 getopt 的选项字符串规则

```c
// "ab:c::d" 的含义：
//   -a     不需要参数
//   -b x   必须带参数（空格分隔或紧跟）
//   -c::   可选参数（必须紧跟，如 -cval）
//   -d     不需要参数

// 示例
getopt(argc, argv, "ab:c::d");

// 合法调用：
// ./prog -a -b hello -cworld -d file.txt
```

### 3.3 getopt_long（GNU 扩展）

```c
#include <stdio.h>
#include <getopt.h>

int main(int argc, char *argv[]) {
    int verbose = 0;
    const char *output = NULL;
    int port = 8080;

    static struct option long_options[] = {
        {"verbose", no_argument,       0, 'v'},
        {"help",    no_argument,       0, 'h'},
        {"output",  required_argument, 0, 'o'},
        {"port",    required_argument, 0, 'p'},
        {0, 0, 0, 0}  // 终止标记
    };

    int opt;
    int option_index = 0;

    while ((opt = getopt_long(argc, argv, "vho:p:",
                              long_options, &option_index)) != -1) {
        switch (opt) {
            case 'v': verbose = 1; break;
            case 'h':
                printf("用法: %s [选项] [文件...]\n", argv[0]);
                printf("  -v, --verbose    详细输出\n");
                printf("  -o, --output FILE 输出文件\n");
                printf("  -p, --port PORT   端口号\n");
                return 0;
            case 'o': output = optarg; break;
            case 'p': port = atoi(optarg); break;
            default: return 1;
        }
    }

    // 处理剩余的非选项参数
    for (int i = optind; i < argc; i++) {
        printf("文件: %s\n", argv[i]);
    }

    return 0;
}
```

## 四、子命令模式（类似 git）

```c
#include <stdio.h>
#include <string.h>

void cmd_add(int argc, char *argv[]);
void cmd_remove(int argc, char *argv[]);
void cmd_help(int argc, char *argv[]);

int main(int argc, char *argv[]) {
    if (argc < 2) {
        cmd_help(argc, argv);
        return 1;
    }

    const char *cmd = argv[1];

    if (strcmp(cmd, "add") == 0)        cmd_add(argc - 1, argv + 1);
    else if (strcmp(cmd, "remove") == 0) cmd_remove(argc - 1, argv + 1);
    else if (strcmp(cmd, "help") == 0)   cmd_help(argc - 1, argv + 1);
    else {
        fprintf(stderr, "未知命令: %s\n", cmd);
        cmd_help(argc, argv);
        return 1;
    }

    return 0;
}

void cmd_add(int argc, char *argv[]) {
    printf("执行 add 命令，参数个数: %d\n", argc);
    for (int i = 1; i < argc; i++)
        printf("  添加: %s\n", argv[i]);
}

void cmd_remove(int argc, char *argv[]) {
    printf("执行 remove 命令\n");
}

void cmd_help(int argc, char *argv[]) {
    printf("用法: mytool <命令> [选项]\n");
    printf("命令:\n");
    printf("  add     添加项目\n");
    printf("  remove  移除项目\n");
    printf("  help    显示帮助\n");
}
```

## 五、要点总结

> **关键点：**
> 1. 手动解析选项适合简单程序，`getopt`/`getopt_long` 适合复杂选项。
> 2. `getopt` 是 POSIX 标准，Windows 上需用替代方案。
> 3. `optarg` 指向当前选项的参数，`optind` 是第一个非选项参数的索引。
> 4. 带参数的选项：`-o file` 或 `-ofile` 都可以。
> 5. 选项解析完毕后，`argv[optind]` 开始是剩余的非选项参数。
> 6. 始终提供 `-h`/`--help` 选项，显示用法信息。
> 7. 子命令模式适合构建多功能命令行工具。
