# 22 - 类型大小与对齐

## 一、概述

**内存对齐（Memory Alignment）** 是指数据在内存中的存储地址必须满足一定的规则。对齐可以提高CPU的内存访问效率，某些架构甚至要求必须对齐否则会引发硬件异常。

## 二、基本类型的大小

```c
#include <stdio.h>

int main(void) {
    printf("char          = %zu\n", sizeof(char));          // 1
    printf("short         = %zu\n", sizeof(short));         // 通常2
    printf("int           = %zu\n", sizeof(int));           // 通常4
    printf("long          = %zu\n", sizeof(long));          // 4(LP32/LLP64)或8(LP64)
    printf("long long     = %zu\n", sizeof(long long));     // 通常8
    printf("float         = %zu\n", sizeof(float));         // 通常4
    printf("double        = %zu\n", sizeof(double));        // 通常8
    printf("long double   = %zu\n", sizeof(long double));   // 8/12/16（取决于实现）
    printf("void*         = %zu\n", sizeof(void*));         // 4(32位)或8(64位)
    printf("size_t        = %zu\n", sizeof(size_t));        // 同void*

    return 0;
}
```

### 常见数据模型

| 数据模型 | short | int | long | 指针 | 平台 |
|----------|-------|-----|------|------|------|
| ILP32 | 2 | 4 | 4 | 4 | 32位Windows/Linux |
| LLP64 | 2 | 4 | 4 | 8 | 64位Windows |
| LP64 | 2 | 4 | 8 | 8 | 64位Linux/macOS |

## 三、结构体对齐规则

### 3.1 对齐规则

1. **成员对齐**：每个成员的起始地址必须是该成员大小的整数倍。
2. **结构体对齐**：结构体的总大小必须是其最大成员对齐值的整数倍。
3. **填充字节（Padding）**：编译器在成员之间和结构体末尾插入填充字节以满足对齐要求。

### 3.2 示例

```c
#include <stdio.h>

struct Example1 {
    char  a;    // 1字节，偏移0
    // 3字节填充
    int   b;    // 4字节，偏移4（需要4字节对齐）
    char  c;    // 1字节，偏移8
    // 3字节填充
};
// 总大小：12字节（最大对齐是4，需要是4的倍数）

struct Example2 {
    char  a;    // 1字节，偏移0
    char  b;    // 1字节，偏移1
    short c;    // 2字节，偏移2（需要2字节对齐）
    int   d;    // 4字节，偏移4（需要4字节对齐）
};
// 总大小：8字节（紧凑排列，无浪费）

struct Example3 {
    int   a;    // 4字节，偏移0
    char  b;    // 1字节，偏移4
    double c;   // 8字节，偏移8（需要8字节对齐，3字节填充）
};
// 总大小：16字节

int main(void) {
    printf("sizeof(Example1) = %zu\n", sizeof(struct Example1));  // 12
    printf("sizeof(Example2) = %zu\n", sizeof(struct Example2));  // 8
    printf("sizeof(Example3) = %zu\n", sizeof(struct Example3));  // 16

    return 0;
}
```

### 3.3 减少填充的技巧

```c
// 好的排列：从大到小排列成员
struct Good {
    double a;   // 8字节，偏移0
    int    b;   // 4字节，偏移8
    short  c;   // 2字节，偏移12
    char   d;   // 1字节，偏移14
    // 1字节填充
};  // 总大小：16

// 差的排列：随意排列导致大量填充
struct Bad {
    char   a;   // 1字节，偏移0
    // 7字节填充
    double b;   // 8字节，偏移8
    char   c;   // 1字节，偏移16
    short  d;   // 2字节，偏移18（不需要填充，18是2的倍数）
    // 4字节填充
};  // 总大小：24
```

## 四、C11 对齐关键字

### 4.1 `_Alignof` —— 查询对齐要求

```c
#include <stdio.h>
#include <stdalign.h>  // 提供 alignof 宏

int main(void) {
    // _Alignof 是C11关键字，alignof 是<stdalign.h>中的宏
    printf("alignof(char)   = %zu\n", _Alignof(char));    // 1
    printf("alignof(short)  = %zu\n", _Alignof(short));   // 2
    printf("alignof(int)    = %zu\n", _Alignof(int));     // 4
    printf("alignof(double) = %zu\n", _Alignof(double));  // 8

    // 查询结构体的对齐要求
    struct S { char a; int b; };
    printf("alignof(S)      = %zu\n", _Alignof(struct S));  // 4（最大成员int的对齐）

    return 0;
}
```

### 4.2 `_Alignas` —— 指定对齐方式

```c
#include <stdio.h>
#include <stdalign.h>

int main(void) {
    // 默认对齐的变量
    int a;
    printf("a 的地址 % 16 = %zu\n", (size_t)(uintptr_t)&a % 16);  // 不一定是0

    // 指定16字节对齐
    _Alignas(16) int b;
    printf("b 的地址 % 16 = %zu\n", (size_t)(uintptr_t)&b % 16);  // 0

    // 也可以用于结构体
    struct _Alignas(32) AlignedStruct {
        int x;
        int y;
    };

    struct AlignedStruct s;
    printf("s 的地址 % 32 = %zu\n", (size_t)(uintptr_t)&s % 32);  // 0
    printf("sizeof(AlignedStruct) = %zu\n", sizeof(struct AlignedStruct));  // 32

    return 0;
}
```

## 五、`offsetof` 宏

`offsetof` 宏定义在 `<stddef.h>` 中，用于获取结构体成员相对于结构体起始地址的偏移量。

```c
#include <stdio.h>
#include <stddef.h>

struct Data {
    char  a;
    int   b;
    short c;
};

int main(void) {
    printf("offsetof(a) = %zu\n", offsetof(struct Data, a));  // 0
    printf("offsetof(b) = %zu\n", offsetof(struct Data, b));  // 4（3字节填充后）
    printf("offsetof(c) = %zu\n", offsetof(struct Data, c));  // 8

    // 利用offsetof分析对齐
    size_t padding_after_a = offsetof(struct Data, b) - offsetof(struct Data, a) - sizeof(char);
    printf("a后的填充字节 = %zu\n", padding_after_a);  // 3

    return 0;
}
```

## 六、`#pragma pack` 控制对齐

```c
#include <stdio.h>

// 默认对齐
struct Normal {
    char a;
    int  b;
};
// sizeof = 8

// 1字节对齐（紧凑排列）
#pragma pack(push, 1)
struct Packed {
    char a;
    int  b;
};
#pragma pack(pop)
// sizeof = 5

int main(void) {
    printf("sizeof(Normal) = %zu\n", sizeof(struct Normal));  // 8
    printf("sizeof(Packed) = %zu\n", sizeof(struct Packed));  // 5

    return 0;
}
```

### pack 的注意事项

- `#pragma pack(1)` 可能导致**未对齐访问**，在某些架构上会严重影响性能或引发硬件异常。
- 网络协议和文件格式中常用 `#pragma pack(1)` 来确保结构体布局与协议一致。
- 跨平台代码中应谨慎使用。

## 七、关键要点

1. 内存对齐可以提高CPU访问效率，避免未对齐访问的性能损失或硬件异常。
2. 结构体成员按声明顺序排列，每个成员的起始地址是其大小的整数倍。
3. 结构体总大小是其最大对齐值的整数倍。
4. 合理排列结构体成员（从大到小）可以减少填充字节。
5. C11提供了 `_Alignof`（查询对齐）和 `_Alignas`（指定对齐）关键字。
6. `offsetof` 可以获取结构体成员的偏移量。
7. `#pragma pack` 可以改变默认对齐，但要注意性能和可移植性影响。
8. 不同平台（32位/64位，Windows/Linux）的 `long` 和指针大小可能不同。
