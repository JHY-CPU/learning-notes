# 23 - 复数运算（C99）

## 一、概述

C99标准引入了复数类型和相关的数学运算支持，通过 `<complex.h>` 头文件提供。这使得C语言可以直接进行复数运算，而不需要自己实现复数结构体。

## 二、复数类型

C99提供三种复数类型：

| 类型 | 说明 | 组成 |
|------|------|------|
| `float _Complex` | 单精度复数 | 两个 `float` |
| `double _Complex` | 双精度复数（默认） | 两个 `double` |
| `long double _Complex` | 扩展精度复数 | 两个 `long double` |

`<complex.h>` 提供了简写宏：

| 宏 | 等价类型 |
|----|----------|
| `complex` | `_Complex` |
| `float complex` | `float _Complex` |
| `double complex` | `double _Complex` |
| `long double complex` | `long double _Complex` |

## 三、创建复数

### 3.1 使用宏

```c
#include <stdio.h>
#include <complex.h>

int main(void) {
    // 使用 CMPLX 宏创建复数（C11推荐）
    double complex z1 = CMPLX(3.0, 4.0);     // 3 + 4i
    double complex z2 = CMPLXF(1.0f, 2.0f);  // 1 + 2i (float complex)

    // 使用 I 宏
    double complex z3 = 3.0 + 4.0 * I;       // 3 + 4i

    // 直接赋值实部和虚部（某些编译器支持）
    double complex z4 = 3.0 + 4.0i;          // GCC扩展

    printf("z1 = %f + %fi\n", creal(z1), cimag(z1));
    printf("z2 = %f + %fi\n", creal(z2), cimag(z2));
    printf("z3 = %f + %fi\n", creal(z3), cimag(z3));

    return 0;
}
```

### 3.2 虚数类型

C99还提供了纯虚数类型（较少使用）：

```c
float imaginary       // 单精度纯虚数
double imaginary      // 双精度纯虚数
long double imaginary // 扩展精度纯虚数

// 简写宏
#define imaginary _Imaginary
```

## 四、复数运算

### 4.1 基本运算

```c
#include <stdio.h>
#include <complex.h>

int main(void) {
    double complex z1 = CMPLX(3.0, 4.0);   // 3 + 4i
    double complex z2 = CMPLX(1.0, 2.0);   // 1 + 2i

    // 基本算术运算
    double complex sum  = z1 + z2;          // (3+1) + (4+2)i = 4 + 6i
    double complex diff = z1 - z2;          // (3-1) + (4-2)i = 2 + 2i
    double complex prod = z1 * z2;          // (3*1-4*2) + (3*2+4*1)i = -5 + 10i
    double complex quot = z1 / z2;          // 复数除法

    printf("z1 + z2 = %f + %fi\n", creal(sum), cimag(sum));
    printf("z1 - z2 = %f + %fi\n", creal(diff), cimag(diff));
    printf("z1 * z2 = %f + %fi\n", creal(prod), cimag(prod));
    printf("z1 / z2 = %f + %fi\n", creal(quot), cimag(quot));

    return 0;
}
```

### 4.2 复数函数

`<complex.h>` 提供了丰富的复数运算函数：

```c
#include <stdio.h>
#include <complex.h>

int main(void) {
    double complex z = CMPLX(3.0, 4.0);  // 3 + 4i

    // 获取实部和虚部
    printf("实部 = %f\n", creal(z));     // 3.0
    printf("虚部 = %f\n", cimag(z));     // 4.0

    // 模（绝对值）和辐角
    printf("模 = %f\n", cabs(z));        // 5.0 (sqrt(3^2 + 4^2))
    printf("辐角 = %f\n", carg(z));      // 0.9273 (atan2(4, 3))

    // 共轭
    double complex conj_z = conj(z);     // 3 - 4i
    printf("共轭 = %f + %fi\n", creal(conj_z), cimag(conj_z));

    // 指数和对数
    double complex exp_z = cexp(z);      // e^(3+4i)
    double complex log_z = clog(z);      // ln(3+4i)
    printf("e^z = %f + %fi\n", creal(exp_z), cimag(exp_z));
    printf("ln(z) = %f + %fi\n", creal(log_z), cimag(log_z));

    // 幂运算
    double complex pow_z = cpow(z, 2);   // (3+4i)^2 = -7 + 24i
    printf("z^2 = %f + %fi\n", creal(pow_z), cimag(pow_z));

    // 平方根
    double complex sqrt_z = csqrt(z);    // sqrt(3+4i)
    printf("sqrt(z) = %f + %fi\n", creal(sqrt_z), cimag(sqrt_z));

    // 三角函数
    double complex sin_z = csin(z);
    double complex cos_z = ccos(z);
    printf("sin(z) = %f + %fi\n", creal(sin_z), cimag(sin_z));
    printf("cos(z) = %f + %fi\n", creal(cos_z), cimag(cos_z));

    return 0;
}
```

## 五、函数命名约定

`<complex.h>` 中的函数遵循统一的命名规则：

| 前缀 | 返回类型 | 示例 |
|------|----------|------|
| `c` | `double complex` | `csin()`, `cexp()` |
| `c` + `f` | `float complex` | `csinf()`, `cexpf()` |
| `c` + `l` | `long double complex` | `csinl()`, `cexpl()` |

```c
float complex zf = CMPLXF(1.0f, 2.0f);
float complex sin_zf = csinf(zf);        // float版本

long double complex zl = CMPLXL(1.0L, 2.0L);
long double complex sin_zl = csinl(zl);  // long double版本
```

## 六、实际应用示例

```c
#include <stdio.h>
#include <complex.h>
#include <math.h>

// 求解一元二次方程 ax^2 + bx + c = 0
void solve_quadratic(double a, double b, double c) {
    double discriminant = b * b - 4 * a * c;

    if (discriminant >= 0) {
        double x1 = (-b + sqrt(discriminant)) / (2 * a);
        double x2 = (-b - sqrt(discriminant)) / (2 * a);
        printf("实数根: x1 = %f, x2 = %f\n", x1, x2);
    } else {
        // 复数根
        double complex disc = CMPLX(discriminant, 0);
        double complex sqrt_disc = csqrt(disc);
        double complex x1 = (-b + sqrt_disc) / (2 * a);
        double complex x2 = (-b - sqrt_disc) / (2 * a);

        printf("复数根: x1 = %f + %fi\n", creal(x1), cimag(x1));
        printf("复数根: x2 = %f + %fi\n", creal(x2), cimag(x2));
    }
}

int main(void) {
    // x^2 + 2x + 5 = 0 的根
    solve_quadratic(1.0, 2.0, 5.0);
    // 输出: 复数根: x1 = -1.000000 + 2.000000i
    //       复数根: x2 = -1.000000 - 2.000000i

    return 0;
}
```

## 七、注意事项

1. C99对复数的支持是**可选的**，编译器可以定义 `__STDC_NO_COMPLEX__` 表示不支持。
2. `complex.h` 实际上是围绕复数运算的宏和函数封装。
3. 复数在内存中的布局是：**实部在前，虚部在后**。
4. 可以通过数组方式访问复数的实部和虚部（但不推荐，不保证可移植性）：
   ```c
   double complex z = CMPLX(3.0, 4.0);
   double *parts = (double*)&z;
   printf("实部 = %f\n", parts[0]);  // 可能正确，但不保证
   printf("虚部 = %f\n", parts[1]);
   ```

## 八、关键要点

1. C99引入了 `_Complex` 类型和 `<complex.h>` 头文件支持复数运算。
2. 使用 `CMPLX(real, imag)` 宏或 `real + imag * I` 创建复数。
3. `creal()` 和 `cimag()` 分别获取实部和虚部。
4. `<complex.h>` 提供了完整的复数数学函数（三角、指数、对数、幂等）。
5. 函数命名遵循 `c` + 函数名 + `f/l` 后缀的规则。
6. 复数支持是可选的，通过 `__STDC_NO_COMPLEX__` 检测。
7. 在信号处理、控制系统等领域，复数运算非常有用。
