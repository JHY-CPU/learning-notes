# 16 - 位运算应用技巧

## 一、概述

位运算是C语言中最高效的运算之一，在系统编程、嵌入式开发、算法竞赛中有广泛应用。本节介绍常见的位运算实用技巧。

## 二、判断奇偶

```c
// 用 & 1 判断最低位
// 奇数最低位为1，偶数最低位为0
if (n & 1) {
    printf("%d 是奇数\n", n);
} else {
    printf("%d 是偶数\n", n);
}

// 等价于 n % 2，但效率更高
```

## 三、乘除2的幂

```c
int n = 5;

// 左移1位 = 乘以2
printf("5 * 2 = %d\n", n << 1);   // 10

// 左移n位 = 乘以2^n
printf("5 * 8 = %d\n", n << 3);   // 40

// 右移1位 = 除以2
printf("20 / 2 = %d\n", 20 >> 1); // 10

// 右移n位 = 除以2^n
printf("40 / 8 = %d\n", 40 >> 3); // 5

// 注意：右移对于负数是算术右移，结果向负无穷取整
// 而整数除法向零取整，两者对负数结果不同
printf("-5 / 2 = %d\n", -5 / 2);    // -2（向零）
printf("-5 >> 1 = %d\n", -5 >> 1);  // -3（向负无穷）
```

## 四、交换两个变量（不使用临时变量）

```c
// 方法1：异或交换（仅适用于整数，且a和b不能是同一变量）
void swap_xor(int *a, int *b) {
    if (a != b) {  // 必须检查不是同一地址
        *a ^= *b;
        *b ^= *a;
        *a ^= *b;
    }
}

// 验证
int x = 10, y = 20;
swap_xor(&x, &y);
printf("x = %d, y = %d\n", x, y);  // x = 20, y = 10

// 推导过程：
// *a = a ^ b
// *b = (a ^ b) ^ b = a ^ (b ^ b) = a ^ 0 = a
// *a = (a ^ b) ^ a = (a ^ a) ^ b = 0 ^ b = b
```

## 五、取反与绝对值

```c
// 取绝对值（32位有符号整数）
int abs_bits(int n) {
    int mask = n >> 31;  // 全0（n>=0）或全1（n<0）
    return (n ^ mask) - mask;
    // n >= 0: mask = 0, (n ^ 0) - 0 = n
    // n < 0:  mask = -1, (n ^ -1) - (-1) = ~n + 1 = -n
}
```

## 六、取最大/最小值（不使用比较）

```c
// 获取int类型的最大值和最小值
int int_max = ~(1 << 31);         // 0x7FFFFFFF = 2147483647
int int_min = 1 << 31;            // 0x80000000 = -2147483648

// 获取unsigned int的最大值
unsigned int uint_max = ~0u;       // 0xFFFFFFFF = 4294967295

// 两个数中取较大值（不使用if）
int max_no_if(int a, int b) {
    int diff = a - b;
    int sign = (diff >> 31) & 1;   // 如果a < b，sign为1
    return a - diff * sign;
    // a < b: sign=1, result = a - (a-b)*1 = b
    // a >= b: sign=0, result = a - (a-b)*0 = a
}
```

## 七、向上取整到2的幂

```c
// 将n向上取整到最近的2的幂
unsigned int next_power_of_2(unsigned int n) {
    if (n == 0) return 1;
    n--;           // 处理n本身就是2的幂的情况
    n |= n >> 1;
    n |= n >> 2;
    n |= n >> 4;
    n |= n >> 8;
    n |= n >> 16;
    return n + 1;
}

// 示例
printf("%u\n", next_power_of_2(5));   // 8
printf("%u\n", next_power_of_2(8));   // 8
printf("%u\n", next_power_of_2(13));  // 16
```

## 八、统计二进制中1的个数

```c
// 方法1：逐位检查
int popcount_v1(unsigned int n) {
    int count = 0;
    while (n) {
        count += n & 1;
        n >>= 1;
    }
    return count;
}

// 方法2：Brian Kernighan算法（更快）
int popcount_v2(unsigned int n) {
    int count = 0;
    while (n) {
        n &= (n - 1);  // 清除最低位的1
        count++;
    }
    return count;
}

// 方法3：查表法（最快，适合大量计算）
static const unsigned char bit_count_table[256] = {
    0,1,1,2,1,2,2,3,1,2,2,3,2,3,3,4, /* ... 剩余240个值 */
};

int popcount_v3(unsigned int n) {
    return bit_count_table[n & 0xFF]
         + bit_count_table[(n >> 8) & 0xFF]
         + bit_count_table[(n >> 16) & 0xFF]
         + bit_count_table[(n >> 24) & 0xFF];
}
```

## 九、最低位的1

```c
// 提取最低位的1（lowbit）
unsigned int lowbit(unsigned int n) {
    return n & (-n);
    // -n 的补码：~n + 1
    // n    = ...01011000
    // ~n   = ...10100111
    // -n   = ...10101000
    // n&-n = ...00001000
}

// 清除最低位的1
unsigned int clear_lowest(unsigned int n) {
    return n & (n - 1);
}

// 判断是否是2的幂（只有一个1）
int is_power_of_2(unsigned int n) {
    return n != 0 && (n & (n - 1)) == 0;
}
```

## 十、位反转

```c
// 将整数的二进制位反转
unsigned int reverse_bits(unsigned int n) {
    n = ((n & 0xAAAAAAAA) >> 1) | ((n & 0x55555555) << 1);
    n = ((n & 0xCCCCCCCC) >> 2) | ((n & 0x33333333) << 2);
    n = ((n & 0xF0F0F0F0) >> 4) | ((n & 0x0F0F0F0F) << 4);
    n = ((n & 0xFF00FF00) >> 8) | ((n & 0x00FF00FF) << 8);
    n = (n >> 16) | (n << 16);
    return n;
}
```

## 十一、关键要点

1. `n & 1` 判断奇偶，`n << k` 实现乘以2^k，`n >> k` 实现除以2^k。
2. 异或交换要求两个变量**不能是同一个**（否则变成0）。
3. `n & (n-1)` 清除最低位的1，`n & (-n)` 提取最低位的1。
4. `n & (n-1) == 0` 可以判断n是否为2的幂（需排除n=0）。
5. 位运算技巧在算法竞赛和底层开发中非常有用。
6. 使用位运算前要考虑**可读性**，复杂技巧应添加注释。
7. 对于负数的右移，有符号数使用算术右移，与除法的向零取整行为不同。
