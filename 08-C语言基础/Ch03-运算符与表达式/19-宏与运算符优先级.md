# 19 - 宏与运算符优先级

## 一、概述

宏定义中的表达式在预处理阶段进行**文本替换**，不会考虑运算符优先级。如果宏参数不加括号，替换后可能因优先级问题产生错误结果。

## 二、基本问题

### 2.1 没有括号的宏

```c
#define SQUARE(x) x * x

int a = SQUARE(3);       // 3 * 3 = 9，正确
int b = SQUARE(1 + 2);   // 1 + 2 * 1 + 2 = 5，错误！期望是9
// 展开过程：1 + 2 * 1 + 2 = 1 + 2 + 2 = 5

int c = 2 * SQUARE(3);   // 2 * 3 * 3 = 18，正确
int d = 2 * SQUARE(1+2); // 2 * 1 + 2 * 1 + 2 = 6，错误！期望是18
```

### 2.2 参数加括号

```c
#define SQUARE(x) ((x) * (x))

int b = SQUARE(1 + 2);   // ((1 + 2) * (1 + 2)) = 9，正确！
int d = 2 * SQUARE(1+2); // 2 * ((1 + 2) * (1 + 2)) = 18，正确！
```

## 三、完整括号规则

**规则：宏定义中的每个参数和整个表达式都要加括号。**

```c
// 错误示范
#define ADD(a, b)        a + b
#define MUL(a, b)        a * b
#define MAX(a, b)        a > b ? a : b
#define ABS(x)           x > 0 ? x : -x

// 正确示范
#define ADD(a, b)        ((a) + (b))
#define MUL(a, b)        ((a) * (b))
#define MAX(a, b)        ((a) > (b) ? (a) : (b))
#define ABS(x)           ((x) > 0 ? (x) : -(x))
```

## 四、经典错误案例

### 4.1 MAX 宏

```c
// 错误版本
#define MAX(a, b) a > b ? a : b

int x = MAX(3, 5);           // 3 > 5 ? 3 : 5 = 5，正确
int y = 2 * MAX(3, 5);       // 2 * 3 > 5 ? 3 : 5 = 6 > 5 ? 3 : 5 = 3，错误！期望10
// 展开为：2 * 3 > 5 ? 3 : 5

// 正确版本
#define MAX(a, b) ((a) > (b) ? (a) : (b))

int y = 2 * MAX(3, 5);       // 2 * ((3) > (5) ? (3) : (5)) = 2 * 5 = 10，正确
```

### 4.2 宏中的副作用

```c
#define MAX(a, b) ((a) > (b) ? (a) : (b))

int i = 5, j = 3;
int m = MAX(i++, j++);
// 展开为：((i++) > (j++) ? (i++) : (j++))
// i++ > j++：5 > 3 为真，i变为6，j变为4
// 真分支：i++，值为6，i变为7
// 结果：m = 6，i = 7，j = 4
// i被自增了两次！

// 推荐使用内联函数替代
static inline int max_int(int a, int b) {
    return a > b ? a : b;
}
```

### 4.3 布尔运算宏

```c
// 错误
#define IS_EVEN(n)   n % 2 == 0
#define IS_ODD(n)    n % 2 != 0

if (!IS_EVEN(4)) { }    // !(4 % 2 == 0) = !0 = 1，正确
if (!IS_EVEN(3)) { }    // !(3 % 2 == 0) = !1 = 0，正确但混乱

// 正确
#define IS_EVEN(n)   ((n) % 2 == 0)
#define IS_ODD(n)    ((n) % 2 != 0)
```

### 4.4 多语句宏

```c
// 错误：多语句宏在if-else中会出问题
#define SWAP(a, b)  int tmp = a; a = b; b = tmp;

if (condition)
    SWAP(x, y);
else
    do_something();
// 展开为：
// if (condition)
//     int tmp = x; x = y; y = tmp;  // 只有int tmp = x在if内部！
// else
//     do_something();  // 编译错误！

// 正确：使用 do-while(0) 包装
#define SWAP(a, b) do { \
    int _tmp = (a); \
    (a) = (b); \
    (b) = _tmp; \
} while (0)
```

## 五、do-while(0) 技巧

`do-while(0)` 是C语言宏定义中一个重要的惯用法：

```c
#define LOG(msg) do { \
    printf("[LOG] %s\n", msg); \
    fflush(stdout); \
} while (0)

// 使用时和普通语句一样
if (error)
    LOG("发生错误");
else
    LOG("正常");

// do-while(0) 后面需要分号，与if-else完美配合
```

### 为什么用 do-while(0)

```c
// 如果用大括号
#define LOG(msg) { printf("[LOG] %s\n", msg); }

if (error)
    LOG("错误");     // 展开为 { printf("..."); }
else                 // else前面有多余的分号！编译错误
    printf("OK\n");

// do-while(0) 解决了这个问题
#define LOG(msg) do { printf("[LOG] %s\n", msg); } while(0)

if (error)
    LOG("错误");     // 展开为 do { printf("..."); } while(0);
else                 // 完美配合
    printf("OK\n");
```

## 六、安全的宏定义模板

```c
// 算术运算
#define ADD(a, b)      ((a) + (b))
#define SUB(a, b)      ((a) - (b))
#define MUL(a, b)      ((a) * (b))
#define DIV(a, b)      ((a) / (b))
#define MOD(a, b)      ((a) % (b))
#define MAX(a, b)      ((a) > (b) ? (a) : (b))
#define MIN(a, b)      ((a) < (b) ? (a) : (b))
#define ABS(x)         ((x) >= 0 ? (x) : -(x))
#define CLAMP(x, lo, hi) MAX(lo, MIN(x, hi))

// 位操作
#define BIT_SET(var, bit)    ((var) |= (1u << (bit)))
#define BIT_CLR(var, bit)    ((var) &= ~(1u << (bit)))
#define BIT_TOGGLE(var, bit) ((var) ^= (1u << (bit)))
#define BIT_TEST(var, bit)   (((var) >> (bit)) & 1u)

// 交换（使用临时变量的版本，需要GCC扩展或C11泛型）
```

## 七、宏 vs 内联函数

| 特性 | 宏 | 内联函数 |
|------|-----|---------|
| 类型检查 | 无 | 有 |
| 调试 | 困难 | 容易 |
| 副作用 | 有风险 | 安全 |
| 求值时机 | 编译前（文本替换） | 编译时 |
| 适用场景 | 简单表达式、泛型 | 类型安全的函数 |

```c
// 现代C推荐：优先使用内联函数
static inline int square(int x) {
    return x * x;
}

// 只在需要泛型或编译时常量时使用宏
#define SQUARE(x) ((x) * (x))  // 可以用于int、float、double等
```

## 八、关键要点

1. 宏是**文本替换**，必须为每个参数和整个表达式加括号。
2. 宏的参数如果有副作用（如 `i++`），可能导致多次求值。
3. 多语句宏应使用 `do { ... } while(0)` 包装。
4. `do-while(0)` 技巧确保宏在 `if-else` 等上下文中正常工作。
5. 现代C语言中，**内联函数**是比宏更安全的替代方案。
6. 启用编译器警告（`-Wall -Wmacro-not-expanded` 等）有助于发现问题。
7. 宏不能调试——预处理后的代码与源代码不同。
