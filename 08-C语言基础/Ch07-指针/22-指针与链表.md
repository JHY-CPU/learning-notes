# 指针与链表

## 一、链表概述

链表（Linked List）是一种通过指针连接的数据结构，每个节点包含数据和指向下一个节点的指针。

```c
struct Node {
    int data;
    struct Node *next;
};
```

```
head -> [10|*] -> [20|*] -> [30|*] -> NULL
```

## 二、链表的创建

### 节点的动态创建

```c
#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node *next;
};

struct Node* create_node(int data) {
    struct Node *node = (struct Node*)malloc(sizeof(struct Node));
    if (node == NULL) {
        fprintf(stderr, "内存分配失败\n");
        return NULL;
    }
    node->data = data;
    node->next = NULL;
    return node;
}
```

## 三、链表的插入操作

### 头部插入

```c
void insert_front(struct Node **head, int data) {
    struct Node *new_node = create_node(data);
    if (new_node == NULL) return;
    new_node->next = *head;
    *head = new_node;
}
```

### 尾部插入

```c
void insert_back(struct Node **head, int data) {
    struct Node *new_node = create_node(data);
    if (new_node == NULL) return;

    if (*head == NULL) {
        *head = new_node;
        return;
    }

    struct Node *current = *head;
    while (current->next != NULL) {
        current = current->next;
    }
    current->next = new_node;
}
```

### 按位置插入

```c
void insert_at(struct Node **head, int pos, int data) {
    if (pos == 0) {
        insert_front(head, data);
        return;
    }

    struct Node *current = *head;
    for (int i = 0; i < pos - 1 && current != NULL; i++) {
        current = current->next;
    }

    if (current == NULL) {
        fprintf(stderr, "位置超出范围\n");
        return;
    }

    struct Node *new_node = create_node(data);
    new_node->next = current->next;
    current->next = new_node;
}
```

## 四、链表的删除操作

### 按值删除

```c
void delete_value(struct Node **head, int value) {
    struct Node *current = *head;
    struct Node *prev = NULL;

    while (current != NULL && current->data != value) {
        prev = current;
        current = current->next;
    }

    if (current == NULL) return;  // 未找到

    if (prev == NULL) {
        *head = current->next;  // 删除头节点
    } else {
        prev->next = current->next;
    }
    free(current);
}
```

### 按位置删除

```c
void delete_at(struct Node **head, int pos) {
    if (*head == NULL) return;

    struct Node *temp = *head;

    if (pos == 0) {
        *head = temp->next;
        free(temp);
        return;
    }

    for (int i = 0; temp != NULL && i < pos - 1; i++) {
        temp = temp->next;
    }

    if (temp == NULL || temp->next == NULL) return;

    struct Node *to_delete = temp->next;
    temp->next = to_delete->next;
    free(to_delete);
}
```

## 五、链表的查找

```c
struct Node* find(struct Node *head, int value) {
    struct Node *current = head;
    while (current != NULL) {
        if (current->data == value) {
            return current;
        }
        current = current->next;
    }
    return NULL;
}

int contains(struct Node *head, int value) {
    return find(head, value) != NULL;
}
```

## 六、链表的修改

```c
int update(struct Node *head, int old_val, int new_val) {
    struct Node *node = find(head, old_val);
    if (node != NULL) {
        node->data = new_val;
        return 1;
    }
    return 0;
}
```

## 七、链表的遍历与打印

```c
void print_list(const struct Node *head) {
    const struct Node *current = head;
    while (current != NULL) {
        printf("%d", current->data);
        if (current->next != NULL) {
            printf(" -> ");
        }
        current = current->next;
    }
    printf(" -> NULL\n");
}

int list_length(const struct Node *head) {
    int count = 0;
    const struct Node *current = head;
    while (current != NULL) {
        count++;
        current = current->next;
    }
    return count;
}
```

## 八、链表的释放

```c
void free_list(struct Node **head) {
    struct Node *current = *head;
    while (current != NULL) {
        struct Node *temp = current;
        current = current->next;
        free(temp);
    }
    *head = NULL;
}
```

## 九、反转链表

```c
void reverse(struct Node **head) {
    struct Node *prev = NULL;
    struct Node *current = *head;
    struct Node *next = NULL;

    while (current != NULL) {
        next = current->next;
        current->next = prev;
        prev = current;
        current = next;
    }
    *head = prev;
}
```

## 十、完整使用示例

```c
int main() {
    struct Node *head = NULL;

    insert_back(&head, 10);
    insert_back(&head, 20);
    insert_back(&head, 30);
    insert_front(&head, 5);
    insert_at(&head, 2, 15);

    printf("链表: ");
    print_list(head);  // 5 -> 10 -> 15 -> 20 -> 30 -> NULL
    printf("长度: %d\n", list_length(head));

    delete_value(&head, 15);
    printf("删除 15 后: ");
    print_list(head);

    reverse(&head);
    printf("反转后: ");
    print_list(head);

    free_list(&head);
    return 0;
}
```

## 十一、关键要点总结

> **核心操作**
> - 插入：头插、尾插、按位置插
> - 删除：按值删、按位置删
> - 查找：按值查找
> - 遍历：逐个访问节点

> **关键技巧**
> - 使用二级指针 `Node**` 修改头指针
> - 删除节点时需要追踪前驱节点
> - 释放链表时逐个释放每个节点
> - 反转链表使用三指针法
