# 指针综合实例

## 一、概述

本节提供一个综合运用指针知识的完整示例项目——**学生成绩管理系统**，涵盖动态内存、结构体指针、链表、函数指针、字符串处理等多个知识点。

## 二、数据结构定义

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 学生成绩结构体（使用不透明指针模式）
typedef struct Student {
    char name[50];
    int id;
    float scores[3];  // 三门课程成绩
    struct Student *next;
} Student;
```

## 三、链表管理函数

### 创建学生节点

```c
Student* create_student(const char *name, int id,
                        float s1, float s2, float s3) {
    Student *stu = (Student*)malloc(sizeof(Student));
    if (stu == NULL) {
        fprintf(stderr, "内存分配失败\n");
        return NULL;
    }

    strncpy(stu->name, name, 49);
    stu->name[49] = '\0';
    stu->id = id;
    stu->scores[0] = s1;
    stu->scores[1] = s2;
    stu->scores[2] = s3;
    stu->next = NULL;

    return stu;
}
```

### 尾部插入

```c
void add_student(Student **head, Student *stu) {
    if (*head == NULL) {
        *head = stu;
        return;
    }

    Student *current = *head;
    while (current->next != NULL) {
        current = current->next;
    }
    current->next = stu;
}
```

### 按 ID 查找

```c
Student* find_by_id(Student *head, int id) {
    Student *current = head;
    while (current != NULL) {
        if (current->id == id) {
            return current;
        }
        current = current->next;
    }
    return NULL;
}
```

### 按姓名查找（返回所有匹配的数组）

```c
Student** find_by_name(Student *head, const char *name, int *count) {
    // 先统计匹配数量
    *count = 0;
    Student *current = head;
    while (current != NULL) {
        if (strstr(current->name, name) != NULL) {
            (*count)++;
        }
        current = current->next;
    }

    if (*count == 0) return NULL;

    // 分配指针数组
    Student **results = (Student**)malloc(*count * sizeof(Student*));
    if (results == NULL) return NULL;

    // 填充结果
    int idx = 0;
    current = head;
    while (current != NULL) {
        if (strstr(current->name, name) != NULL) {
            results[idx++] = current;
        }
        current = current->next;
    }

    return results;
}
```

### 删除学生

```c
int delete_student(Student **head, int id) {
    Student *current = *head;
    Student *prev = NULL;

    while (current != NULL && current->id != id) {
        prev = current;
        current = current->next;
    }

    if (current == NULL) return 0;  // 未找到

    if (prev == NULL) {
        *head = current->next;
    } else {
        prev->next = current->next;
    }
    free(current);
    return 1;
}
```

### 释放链表

```c
void free_all(Student **head) {
    Student *current = *head;
    while (current != NULL) {
        Student *temp = current;
        current = current->next;
        free(temp);
    }
    *head = NULL;
}
```

## 四、统计与分析函数

### 计算平均分

```c
float get_average(const Student *stu) {
    if (stu == NULL) return 0.0f;
    return (stu->scores[0] + stu->scores[1] + stu->scores[2]) / 3.0f;
}
```

### 查找最高分学生

```c
Student* find_top_student(Student *head) {
    if (head == NULL) return NULL;

    Student *top = head;
    Student *current = head->next;

    while (current != NULL) {
        if (get_average(current) > get_average(top)) {
            top = current;
        }
        current = current->next;
    }
    return top;
}
```

### 统计及格率

```c
typedef struct {
    int total;
    int passed;
    float pass_rate;
} CourseStats;

void calc_pass_rate(Student *head, int course_idx, CourseStats *stats) {
    stats->total = 0;
    stats->passed = 0;

    Student *current = head;
    while (current != NULL) {
        stats->total++;
        if (current->scores[course_idx] >= 60.0f) {
            stats->passed++;
        }
        current = current->next;
    }

    stats->pass_rate = stats->total > 0
        ? (float)stats->passed / stats->total * 100.0f
        : 0.0f;
}
```

## 五、排序功能（使用函数指针）

```c
typedef float (*ScoreFunc)(const Student*);
typedef int (*CompareFunc)(const Student*, const Student*);

// 比较函数
int compare_by_average(const Student *a, const Student *b) {
    float avg_a = get_average(a);
    float avg_b = get_average(b);
    if (avg_a < avg_b) return -1;
    if (avg_a > avg_b) return 1;
    return 0;
}

int compare_by_name(const Student *a, const Student *b) {
    return strcmp(a->name, b->name);
}

int compare_by_id(const Student *a, const Student *b) {
    return a->id - b->id;
}

// 链表排序（冒泡排序）
void sort_students(Student **head, CompareFunc cmp, int descending) {
    if (head == NULL || *head == NULL) return;

    int swapped;
    do {
        swapped = 0;
        Student **pp = head;

        while ((*pp)->next != NULL) {
            Student *a = *pp;
            Student *b = a->next;

            int result = cmp(a, b);
            if (descending) result = -result;

            if (result > 0) {
                // 交换 a 和 b
                a->next = b->next;
                b->next = a;
                *pp = b;
                swapped = 1;
            }
            pp = &(*pp)->next;
        }
    } while (swapped);
}
```

## 六、显示函数

```c
void print_student(const Student *stu) {
    if (stu == NULL) return;
    printf("ID: %d, 姓名: %-10s, 成绩: %.1f, %.1f, %.1f, 平均: %.1f\n",
           stu->id, stu->name,
           stu->scores[0], stu->scores[1], stu->scores[2],
           get_average(stu));
}

void print_all(const Student *head) {
    const Student *current = head;
    int count = 0;
    while (current != NULL) {
        print_student(current);
        current = current->next;
        count++;
    }
    printf("共 %d 名学生\n", count);
}
```

## 七、文件操作

```c
// 保存到文件（二进制格式）
int save_to_file(const Student *head, const char *filename) {
    FILE *fp = fopen(filename, "wb");
    if (fp == NULL) return 0;

    const Student *current = head;
    while (current != NULL) {
        // 写入除 next 之外的数据
        fwrite(current, offsetof(Student, next), 1, fp);
        current = current->next;
    }

    fclose(fp);
    return 1;
}

// 从文件加载
Student* load_from_file(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (fp == NULL) return NULL;

    Student *head = NULL;
    Student temp;

    while (fread(&temp, offsetof(Student, next), 1, fp) == 1) {
        Student *stu = create_student(temp.name, temp.id,
                                      temp.scores[0],
                                      temp.scores[1],
                                      temp.scores[2]);
        if (stu != NULL) {
            add_student(&head, stu);
        }
    }

    fclose(fp);
    return head;
}
```

## 八、主函数（菜单驱动）

```c
void print_menu() {
    printf("\n===== 学生成绩管理系统 =====\n");
    printf("1. 添加学生\n");
    printf("2. 显示所有学生\n");
    printf("3. 按 ID 查找\n");
    printf("4. 按姓名搜索\n");
    printf("5. 删除学生\n");
    printf("6. 排序（按平均分）\n");
    printf("7. 显示最高分学生\n");
    printf("8. 统计及格率\n");
    printf("9. 保存到文件\n");
    printf("10. 从文件加载\n");
    printf("0. 退出\n");
    printf("请选择: ");
}

int main() {
    Student *head = NULL;
    int choice;

    // 添加测试数据
    add_student(&head, create_student("张三", 1001, 85, 90, 78));
    add_student(&head, create_student("李四", 1002, 92, 88, 95));
    add_student(&head, create_student("王五", 1003, 70, 65, 58));
    add_student(&head, create_student("赵六", 1004, 55, 72, 68));

    do {
        print_menu();
        scanf("%d", &choice);

        switch (choice) {
            case 1: {
                char name[50];
                int id;
                float s1, s2, s3;
                printf("输入姓名: ");
                scanf("%s", name);
                printf("输入 ID: ");
                scanf("%d", &id);
                printf("输入三门成绩: ");
                scanf("%f %f %f", &s1, &s2, &s3);
                add_student(&head, create_student(name, id, s1, s2, s3));
                break;
            }
            case 2:
                print_all(head);
                break;
            case 3: {
                int id;
                printf("输入 ID: ");
                scanf("%d", &id);
                Student *found = find_by_id(head, id);
                if (found) print_student(found);
                else printf("未找到\n");
                break;
            }
            case 6:
                sort_students(&head, compare_by_average, 1);
                printf("已按平均分降序排列\n");
                break;
            case 7: {
                Student *top = find_top_student(head);
                if (top) {
                    printf("最高分学生: ");
                    print_student(top);
                }
                break;
            }
            case 8: {
                CourseStats stats;
                const char *courses[] = {"语文", "数学", "英语"};
                for (int i = 0; i < 3; i++) {
                    calc_pass_rate(head, i, &stats);
                    printf("%s: %d/%d (%.1f%%)\n",
                           courses[i], stats.passed,
                           stats.total, stats.pass_rate);
                }
                break;
            }
            case 9:
                save_to_file(head, "students.dat");
                printf("已保存\n");
                break;
            case 10:
                free_all(&head);
                head = load_from_file("students.dat");
                printf("已加载\n");
                break;
            case 0:
                break;
            default:
                printf("无效选择\n");
        }
    } while (choice != 0);

    free_all(&head);
    printf("程序结束\n");

    return 0;
}
```

## 九、涉及的指针知识点

这个综合示例涉及了以下指针知识点：

| 知识点 | 应用位置 |
|--------|---------|
| 结构体指针 | Student 节点 |
| 二级指针 | add_student、delete_student 修改头指针 |
| 函数指针 | sort_students 的比较函数参数 |
| 动态内存 malloc/free | 创建和释放节点 |
| 指针运算 | get_average、find_by_name |
| 字符串指针 | strncpy、strstr、strcmp |
| 指针数组 | find_by_name 返回结果 |
| const 指针 | print_student、get_average 参数 |
| 文件指针 | fopen/fclose/fread/fwrite |

## 十、关键要点总结

> **设计原则**
> - 使用二级指针修改链表头
> - 函数指针实现策略模式（排序比较）
> - const 保护不修改数据的参数

> **内存管理**
> - 每个 create 对应一个 free
> - free_all 释放整个链表
> - 删除节点时释放对应内存

> **扩展思路**
> - 可添加双向链表版本
> - 可实现更高效的排序算法
> - 可添加数据库持久化
> - 可用不透明指针模式封装接口
