# 指针高级技巧

## 一、指针运算优化

### 指针 vs 下标

```c
// 下标方式
for (int i = 0; i < n; i++) {
    arr[i] = arr[i] * 2;
}

// 指针方式（通常更快）
for (int *p = arr; p < arr + n; p++) {
    *p = *p * 2;
}
```

指针方式避免了每次循环计算 `base + i * sizeof(type)` 的开销。

### 预取优化

```c
void process(int *arr, int n) {
    int *p = arr;
    int *end = arr + n;

    while (p < end) {
        // 提前加载下一个元素到 cache
        __builtin_prefetch(p + 8, 0, 1);
        // 处理当前元素
        *p = *p * 2 + 1;
        p++;
    }
}
```

### 位操作指针

```c
// 通过指针和位操作设置/清除特定位
void set_bit(uint32_t *reg, int bit) {
    *reg |= (1u << bit);
}

void clear_bit(uint32_t *reg, int bit) {
    *reg &= ~(1u << bit);
}

int test_bit(uint32_t *reg, int bit) {
    return (*reg >> bit) & 1;
}

// 嵌入式应用：操作硬件寄存器
volatile uint32_t *GPIO_ODR = (volatile uint32_t*)0x40020C14;
set_bit((uint32_t*)GPIO_ODR, 5);   // 设置第 5 位
clear_bit((uint32_t*)GPIO_ODR, 3); // 清除第 3 位
```

## 三、函数跳转表

### 字节码解释器

```c
#include <stdio.h>
#include <stdint.h>

typedef void (*OpHandler)(int32_t *stack, int *sp);

void op_push(int32_t *stack, int *sp, int32_t val) {
    stack[++(*sp)] = val;
}

void op_add(int32_t *stack, int *sp) {
    int b = stack[(*sp)--];
    int a = stack[(*sp)--];
    stack[++(*sp)] = a + b;
}

void op_mul(int32_t *stack, int *sp) {
    int b = stack[(*sp)--];
    int a = stack[(*sp)--];
    stack[++(*sp)] = a * b;
}

void op_print(int32_t *stack, int *sp) {
    printf("Result: %d\n", stack[(*sp)--]);
}

// 操作码
enum { OP_PUSH = 0, OP_ADD, OP_MUL, OP_PRINT, OP_HALT };

void execute(const uint8_t *program) {
    int32_t stack[256];
    int sp = -1;
    int ip = 0;

    // 跳转表
    void (*handlers[])(int32_t*, int*) = {
        NULL,  // PUSH 需要特殊处理
        (void(*)(int32_t*,int*))op_add,
        (void(*)(int32_t*,int*))op_mul,
        (void(*)(int32_t*,int*))op_print,
        NULL   // HALT
    };

    while (program[ip] != OP_HALT) {
        uint8_t op = program[ip++];
        if (op == OP_PUSH) {
            int32_t val = *(int32_t*)&program[ip];
            ip += 4;
            op_push(stack, &sp, val);
        } else {
            handlers[op](stack, &sp);
        }
    }
}
```

## 四、基于指针的数据结构

### 异或链表（XOR Linked List）

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>

typedef struct XNode {
    int data;
    struct XNode *xor_ptr;  // prev XOR next
} XNode;

// XOR 两个指针
XNode* xor_ptrs(XNode *a, XNode *b) {
    return (XNode*)((uintptr_t)a ^ (uintptr_t)b);
}

void xlist_insert(XNode **head, int data) {
    XNode *new_node = malloc(sizeof(XNode));
    new_node->data = data;
    new_node->xor_ptr = xor_ptrs(NULL, *head);

    if (*head != NULL) {
        XNode *next = xor_ptrs(NULL, (*head)->xor_ptr);
        (*head)->xor_ptr = xor_ptrs(new_node, next);
    }
    *head = new_node;
}

void xlist_forward(XNode *head) {
    XNode *curr = head;
    XNode *prev = NULL;

    while (curr != NULL) {
        printf("%d ", curr->data);
        XNode *next = xor_ptrs(prev, curr->xor_ptr);
        prev = curr;
        curr = next;
    }
    printf("\n");
}
```

## 五、指针标记（Pointer Tagging）

利用指针低位为 0 的特性存储额外信息：

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>

// 假设所有分配的内存 8 字节对齐
// 指针低 3 位总是 0，可以存储标记

#define TAG_MASK 0x7
#define PTR_MASK ~((uintptr_t)TAG_MASK)

typedef struct Node {
    int data;
    uintptr_t next_tagged;  // 低 3 位存标记，高位存指针
} Node;

Node* get_next(Node *n) {
    return (Node*)(n->next_tagged & PTR_MASK);
}

int get_tag(Node *n) {
    return n->next_tagged & TAG_MASK;
}

void set_next(Node *n, Node *next, int tag) {
    n->next_tagged = (uintptr_t)next | (tag & TAG_MASK);
}
```

## 六、内存池（Memory Pool）

```c
#include <stdlib.h>
#include <string.h>

typedef struct Pool {
    char *memory;
    size_t size;
    size_t used;
} Pool;

Pool* pool_create(size_t size) {
    Pool *p = malloc(sizeof(Pool));
    p->memory = malloc(size);
    p->size = size;
    p->used = 0;
    return p;
}

void* pool_alloc(Pool *p, size_t size) {
    // 对齐到 8 字节
    size_t aligned = (size + 7) & ~7;
    if (p->used + aligned > p->size) return NULL;

    void *ptr = p->memory + p->used;
    p->used += aligned;
    return ptr;
}

void pool_reset(Pool *p) {
    p->used = 0;  // 一次释放所有内存
}

void pool_destroy(Pool *p) {
    free(p->memory);
    free(p);
}
```

## 七、函数组合（Function Composition）

```c
typedef int (*Transform)(int);

int compose(int x, Transform *funcs, int n) {
    int result = x;
    for (int i = 0; i < n; i++) {
        result = funcs[i](result);
    }
    return result;
}

int double_it(int x) { return x * 2; }
int add_one(int x)   { return x + 1; }
int negate(int x)    { return -x; }

// compose(5, {double_it, add_one, negate}, 3)
// = negate(add_one(double_it(5)))
// = negate(add_one(10))
// = negate(11)
// = -11
```

## 八、关键要点总结

> **性能优化**
> - 指针遍历通常比下标快
> - 减少指针运算中的重复计算
> - 利用 prefetch 预取数据

> **底层技巧**
> - 位域与指针配合操作硬件
> - 函数跳转表实现高效分支
> - 异或链表节省指针空间

> **内存管理**
> - 内存池减少 malloc/free 开销
> - 指针标记存储额外信息
> - 对齐保证数据访问效率
