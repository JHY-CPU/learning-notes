# 不透明指针（Opaque Pointer）

## 一、什么是不透明指针

不透明指针（Opaque Pointer），也称为**信息隐藏**（Information Hiding），是一种设计模式：**只在头文件中声明结构体指针类型，不暴露结构体的内部定义**。

使用者只知道有这样一个指针类型，但不知道它指向的结构体内部细节。

## 二、基本模式

### 头文件（公开接口）

```c
// student.h
#ifndef STUDENT_H
#define STUDENT_H

// 前向声明：只声明存在这样一个结构体
typedef struct Student Student;

// 公开接口
Student* student_create(const char *name, int age);
void student_destroy(Student *stu);
const char* student_get_name(const Student *stu);
int student_get_age(const Student *stu);
void student_set_age(Student *stu, int age);

#endif
```

### 实现文件（私有细节）

```c
// student.c
#include "student.h"
#include <stdlib.h>
#include <string.h>

// 结构体完整定义，只在 .c 文件中可见
struct Student {
    char name[50];
    int age;
    float gpa;
    int student_id;
};

Student* student_create(const char *name, int age) {
    Student *stu = (Student*)malloc(sizeof(Student));
    if (stu == NULL) return NULL;

    strncpy(stu->name, name, 49);
    stu->name[49] = '\0';
    stu->age = age;
    stu->gpa = 0.0f;
    stu->student_id = 0;

    return stu;
}

void student_destroy(Student *stu) {
    free(stu);
}

const char* student_get_name(const Student *stu) {
    return stu->name;
}

int student_get_age(const Student *stu) {
    return stu->age;
}

void student_set_age(Student *stu, int age) {
    stu->age = age;
}
```

### 使用方

```c
// main.c
#include "student.h"
#include <stdio.h>

int main() {
    // 使用者不知道 Student 的内部结构
    Student *stu = student_create("张三", 20);

    printf("姓名: %s\n", student_get_name(stu));
    printf("年龄: %d\n", student_get_age(stu));

    student_set_age(stu, 21);
    printf("修改后年龄: %d\n", student_get_age(stu));

    // 不能直接访问 stu->name 或 stu->gpa
    // stu->name  // 编译错误！

    student_destroy(stu);
    return 0;
}
```

## 三、为什么使用不透明指针

### 1. 信息隐藏

```c
// 使用者不知道结构体内部细节
// 可以自由修改内部实现而不影响使用方
// 添加字段、改变布局、优化实现都不需要修改头文件
```

### 2. 封装

```c
// 使用者只能通过接口操作数据
// 不能绕过接口直接访问内部数据
// 保证了数据的一致性和有效性
```

### 3. 二进制兼容性

```c
// 修改结构体内部不影响使用方的编译
// 使用方只需重新链接，无需重新编译
```

## 四、完整示例：栈实现

### stack.h

```c
#ifndef STACK_H
#define STACK_H

typedef struct Stack Stack;

Stack* stack_create(int capacity);
void stack_destroy(Stack *s);
int stack_push(Stack *s, int value);
int stack_pop(Stack *s, int *value);
int stack_peek(const Stack *s, int *value);
int stack_is_empty(const Stack *s);
int stack_size(const Stack *s);

#endif
```

### stack.c

```c
#include "stack.h"
#include <stdlib.h>

struct Stack {
    int *data;
    int capacity;
    int top;
};

Stack* stack_create(int capacity) {
    Stack *s = (Stack*)malloc(sizeof(Stack));
    if (s == NULL) return NULL;

    s->data = (int*)malloc(capacity * sizeof(int));
    if (s->data == NULL) {
        free(s);
        return NULL;
    }

    s->capacity = capacity;
    s->top = -1;
    return s;
}

void stack_destroy(Stack *s) {
    if (s != NULL) {
        free(s->data);
        free(s);
    }
}

int stack_push(Stack *s, int value) {
    if (s == NULL || s->top >= s->capacity - 1) return 0;
    s->data[++s->top] = value;
    return 1;
}

int stack_pop(Stack *s, int *value) {
    if (s == NULL || s->top < 0) return 0;
    *value = s->data[s->top--];
    return 1;
}

int stack_peek(const Stack *s, int *value) {
    if (s == NULL || s->top < 0) return 0;
    *value = s->data[s->top];
    return 1;
}

int stack_is_empty(const Stack *s) {
    return s == NULL || s->top < 0;
}

int stack_size(const Stack *s) {
    return s == NULL ? 0 : s->top + 1;
}
```

## 五、实际应用

### 标准库中的应用

```c
// FILE 类型就是不透明指针
FILE *fp = fopen("file.txt", "r");
// 我们不知道 FILE 结构体的内部定义
// 只能通过 fopen/fclose/fread/fwrite 等接口操作
```

### 图形库中的应用

```c
// OpenGL/Vulkan 中大量使用不透明指针
typedef struct GLFWwindow GLFWwindow;
GLFWwindow* glfwCreateWindow(int w, int h, const char *title, ...);
void glfwDestroyWindow(GLFWwindow *window);
```

### 网络库中的应用

```c
typedef struct Connection Connection;
Connection* connect(const char *host, int port);
int send_data(Connection *conn, const void *data, size_t len);
void disconnect(Connection *conn);
```

## 六、注意事项

```c
// 使用者不能
sizeof(Student)           // 编译错误：不完整类型
Student s;                // 编译错误：不能定义变量
stu->name                 // 编译错误：不知道内部结构

// 使用者只能
Student *stu;             // OK：可以定义指针
student_get_name(stu)     // OK：通过接口访问
```

## 七、关键要点总结

> **核心思想**
> - 头文件只声明结构体指针类型（前向声明）
> - 结构体完整定义放在 .c 文件中
> - 所有操作通过公开接口函数进行

> **设计优势**
> - 信息隐藏：外部不知道内部实现细节
> - 封装保护：防止直接访问内部数据
> - 灵活修改：内部变化不影响使用方
> - 二进制兼容：减少重新编译

> **应用场景**
> - 库的设计（标准库、第三方库）
> - 大型项目的模块化
> - 需要隐藏实现细节的组件
