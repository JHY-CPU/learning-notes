# 函数指针数组

## 一、概述

函数指针数组（Array of Function Pointers）是指**数组中每个元素都是一个函数指针**。它广泛应用于菜单驱动、状态机、跳转表等场景。

```c
// 定义函数指针类型
typedef int (*Operation)(int, int);

// 函数指针数组
Operation ops[4] = {add, sub, mul, divide};
```

## 二、基本定义与使用

```c
#include <stdio.h>

int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }
int divide(int a, int b) { return b ? a / b : 0; }

typedef int (*OpFunc)(int, int);

int main() {
    OpFunc ops[] = {add, sub, mul, divide};
    const char *names[] = {"加", "减", "乘", "除"};

    int a = 20, b = 4;
    for (int i = 0; i < 4; i++) {
        printf("%s: %d %c %d = %d\n",
               names[i], a,
               "+-*/"[i], b,
               ops[i](a, b));
    }

    return 0;
}
```

输出：
```
加: 20 + 4 = 24
减: 20 - 4 = 16
乘: 20 * 4 = 80
除: 20 / 4 = 5
```

## 三、菜单驱动系统

```c
#include <stdio.h>

typedef void (*MenuAction)(void);

void new_file()    { printf("新建文件\n"); }
void open_file()   { printf("打开文件\n"); }
void save_file()   { printf("保存文件\n"); }
void print_file()  { printf("打印文件\n"); }
void exit_app()    { printf("退出程序\n"); }

int main() {
    MenuAction actions[] = {
        new_file, open_file, save_file,
        print_file, exit_app
    };
    const char *menu[] = {
        "1. 新建", "2. 打开", "3. 保存",
        "4. 打印", "5. 退出"
    };
    int n = 5;

    int choice;
    while (1) {
        printf("\n=== 菜单 ===\n");
        for (int i = 0; i < n; i++) {
            printf("%s\n", menu[i]);
        }
        printf("请选择 (1-%d): ", n);
        scanf("%d", &choice);

        if (choice < 1 || choice > n) {
            printf("无效选择\n");
            continue;
        }

        actions[choice - 1]();

        if (choice == n) break;
    }

    return 0;
}
```

## 四、状态机实现

```c
#include <stdio.h>

// 状态枚举
typedef enum {
    STATE_IDLE,
    STATE_RUNNING,
    STATE_PAUSED,
    STATE_STOPPED,
    STATE_COUNT
} State;

// 状态处理函数类型
typedef State (*StateHandler)(int event);

// 各状态的处理函数
State idle_handler(int event) {
    printf("IDLE 状态，事件 %d\n", event);
    if (event == 1) return STATE_RUNNING;
    return STATE_IDLE;
}

State running_handler(int event) {
    printf("RUNNING 状态，事件 %d\n", event);
    if (event == 2) return STATE_PAUSED;
    if (event == 3) return STATE_STOPPED;
    return STATE_RUNNING;
}

State paused_handler(int event) {
    printf("PAUSED 状态，事件 %d\n", event);
    if (event == 1) return STATE_RUNNING;
    if (event == 3) return STATE_STOPPED;
    return STATE_PAUSED;
}

State stopped_handler(int event) {
    printf("STOPPED 状态，事件 %d\n", event);
    return STATE_STOPPED;
}

int main() {
    // 状态跳转表
    StateHandler handlers[STATE_COUNT] = {
        idle_handler,
        running_handler,
        paused_handler,
        stopped_handler
    };

    State current = STATE_IDLE;
    int events[] = {1, 2, 1, 3};

    for (int i = 0; i < 4; i++) {
        current = handlers[current](events[i]);
    }

    return 0;
}
```

## 五、跳转表（Jump Table）

跳转表用于高效的多路分支：

```c
#include <stdio.h>

typedef void (*Command)(const char*);

void cmd_help(const char *args)  { printf("帮助: %s\n", args); }
void cmd_list(const char *args)  { printf("列出: %s\n", args); }
void cmd_show(const char *args)  { printf("显示: %s\n", args); }
void cmd_quit(const char *args)  { printf("退出\n"); }
void cmd_unknown(const char *args) { printf("未知命令\n"); }

int main() {
    // 用第一个字符作为索引的跳转表
    Command jump[26] = {NULL};

    // 注册命令
    jump['h' - 'a'] = cmd_help;
    jump['l' - 'a'] = cmd_list;
    jump['s' - 'a'] = cmd_show;
    jump['q' - 'a'] = cmd_quit;

    const char *inputs[] = {"help", "list", "show", "quit", "xyz"};
    int n = 5;

    for (int i = 0; i < n; i++) {
        char key = inputs[i][0];
        if (key >= 'a' && key <= 'z' && jump[key - 'a'] != NULL) {
            jump[key - 'a'](inputs[i]);
        } else {
            cmd_unknown(inputs[i]);
        }
    }

    return 0;
}
```

## 六、回调注册系统

```c
#include <stdio.h>

#define MAX_CALLBACKS 10

typedef void (*Callback)(int event);

static Callback callbacks[MAX_CALLBACKS];
static int callback_count = 0;

int register_callback(Callback cb) {
    if (callback_count >= MAX_CALLBACKS) return -1;
    callbacks[callback_count++] = cb;
    return callback_count - 1;
}

void emit_event(int event) {
    for (int i = 0; i < callback_count; i++) {
        callbacks[i](event);
    }
}

void logger(int e) { printf("[LOG] 事件 %d\n", e); }
void alert(int e)  { if (e > 5) printf("[ALERT] 高优先级事件 %d\n", e); }

int main() {
    register_callback(logger);
    register_callback(alert);

    emit_event(3);
    emit_event(8);

    return 0;
}
```

## 七、关键要点总结

> **核心概念**
> - 函数指针数组：`ReturnType (*arr[N])(params)`
> - 每个元素可以指向不同但签名相同的函数
> - 通过索引调用，替代复杂的 switch/if-else

> **应用场景**
> - 菜单系统：根据用户选择调用对应函数
> - 状态机：每种状态对应一个处理函数
> - 跳转表：高效多路分支
> - 回调管理：事件驱动系统

> **最佳实践**
> - 使用 typedef 简化声明
> - 数组大小与实际函数数匹配
> - 处理 NULL 元素（未注册的入口）
