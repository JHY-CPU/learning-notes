# 指向指针函数的指针

## 一、什么是指针函数

指针函数（Pointer Function）是指**返回值为指针的函数**。

```c
// 指针函数：返回 int*
int* find_max(int *arr, int n) {
    int *max = &arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > *max) {
            max = &arr[i];
        }
    }
    return max;
}
```

## 二、指向指针函数的指针

指向指针函数的指针是指**可以存储返回指针的函数的地址**的指针。

```c
// 返回 int* 的函数
int* get_value(int index);

// 指向该函数的指针
int* (*fp)(int) = get_value;

// 通过指针调用
int *result = fp(0);
```

## 三、定义语法

```
返回类型* (*指针名)(参数列表);
```

```c
// 返回 int* 的函数
int* func(int, double);

// 指向它的指针
int* (*fp)(int, double) = func;
```

## 四、复杂声明示例

### 返回 char* 的函数指针

```c
char* (*format)(int, const char*);

// 使用
char* int_to_str(int n, const char *fmt);
format = int_to_str;
char *result = format(42, "%d");
```

### 返回 void* 的函数指针

```c
void* (*allocator)(size_t) = malloc;
void *p = allocator(100);
```

### 返回结构体指针的函数指针

```c
struct Node* (*create_func)(int) = create_node;
struct Node *node = create_func(42);
```

## 五、函数指针数组（返回指针）

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 返回 int* 的不同查找函数
int* find_first(int *arr, int n, int val) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == val) return &arr[i];
    }
    return NULL;
}

int* find_last(int *arr, int n, int val) {
    for (int i = n - 1; i >= 0; i--) {
        if (arr[i] == val) return &arr[i];
    }
    return NULL;
}

int* find_greater(int *arr, int n, int val) {
    for (int i = 0; i < n; i++) {
        if (arr[i] > val) return &arr[i];
    }
    return NULL;
}

int main() {
    typedef int* (*FindFunc)(int*, int, int);

    FindFunc finders[] = {find_first, find_last, find_greater};
    const char *names[] = {"首次出现", "最后出现", "第一个大于"};

    int arr[] = {3, 1, 4, 1, 5, 9, 2, 6};
    int n = sizeof(arr) / sizeof(arr[0]);

    for (int i = 0; i < 3; i++) {
        int *result = finders[i](arr, n, 1);
        if (result) {
            printf("%s: 值=%d, 位置=%ld\n",
                   names[i], *result, result - arr);
        } else {
            printf("%s: 未找到\n", names[i]);
        }
    }

    return 0;
}
```

## 六、作为回调的指针函数

```c
typedef char* (*Formatter)(int);

char* format_decimal(int n) {
    static char buf[32];
    snprintf(buf, 32, "%d", n);
    return buf;
}

char* format_hex(int n) {
    static char buf[32];
    snprintf(buf, 32, "0x%x", n);
    return buf;
}

char* format_binary(int n) {
    static char buf[33];
    buf[32] = '\0';
    for (int i = 31; i >= 0; i--) {
        buf[31 - i] = (n & (1 << i)) ? '1' : '0';
    }
    return buf;
}

void print_number(int n, Formatter fmt) {
    printf("%s\n", fmt(n));
}

int main() {
    int num = 255;
    print_number(num, format_decimal);  // 255
    print_number(num, format_hex);      // 0xff
    print_number(num, format_binary);   // 00000000000000000000000011111111
    return 0;
}
```

## 七、动态策略选择

```c
typedef void* (*Allocator)(size_t);
typedef void (*Deallocator)(void*);

typedef struct {
    Allocator alloc;
    Deallocator dealloc;
    const char *name;
} MemoryStrategy;

void* std_alloc(size_t s) { return malloc(s); }

void* calloc_wrapper(size_t s) {
    void *p = malloc(s);
    if (p) memset(p, 0, s);
    return p;
}

MemoryStrategy strategies[] = {
    {malloc, free, "标准分配"},
    {calloc_wrapper, free, "清零分配"}
};

void* allocate_with_strategy(int strategy_idx, size_t size) {
    return strategies[strategy_idx].alloc(size);
}
```

## 八、与返回函数指针的区别

```c
// 返回指针的函数的指针
int* (*fp1)(int);         // fp1 是指针，指向返回 int* 的函数

// 返回函数指针的函数（更复杂）
int* (*get_fp(void))(int);  // get_fp 返回 int*(*)(int)
```

## 九、使用 typedef 简化

```c
// 简化前
int* (*fp)(int, int);

// 使用 typedef
typedef int* (*IntFinder)(int*, int, int);
IntFinder finders[3] = {find_first, find_last, find_greater};
```

## 十、关键要点总结

> **核心概念**
> - 指针函数：返回值为指针的函数
> - 指向指针函数的指针：存储这种函数的地址
> - 声明格式：`返回类型* (*指针名)(参数)`

> **应用场景**
> - 多种查找策略的选择
> - 多种格式化方式的切换
> - 内存分配策略的设计

> **使用技巧**
> - 用 typedef 简化复杂声明
> - 函数指针数组实现策略模式
> - 回调机制中传递函数指针
