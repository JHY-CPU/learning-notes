# 日志库设计

## 概述

日志是软件调试和运维的重要工具。本节设计一个高性能、支持多级别和多输出的日志库。

---

## 1. 日志级别

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdarg.h>
#include <time.h>

#ifdef _WIN32
#include <windows.h>
#define THREAD_LOCAL __declspec(thread)
#else
#define THREAD_LOCAL __thread
#endif

/* 日志级别 */
typedef enum {
    LOG_TRACE = 0,
    LOG_DEBUG = 1,
    LOG_INFO  = 2,
    LOG_WARN  = 3,
    LOG_ERROR = 4,
    LOG_FATAL = 5,
    LOG_OFF   = 6
} LogLevel;

/* 级别名称 */
static const char *level_names[] = {
    "TRACE", "DEBUG", "INFO", "WARN", "ERROR", "FATAL"
};

/* 级别颜色（ANSI） */
static const char *level_colors[] = {
    "\033[90m",  // TRACE - 灰色
    "\033[36m",  // DEBUG - 青色
    "\033[32m",  // INFO  - 绿色
    "\033[33m",  // WARN  - 黄色
    "\033[31m",  // ERROR - 红色
    "\033[35m",  // FATAL - 紫色
};
```

## 2. 日志配置

```c
/* 输出目标 */
typedef enum {
    LOG_SINK_STDOUT = 1,    // 控制台输出
    LOG_SINK_FILE   = 2,    // 文件输出
    LOG_SINK_BOTH   = 3     // 同时输出
} LogSink;

/* 日志配置 */
typedef struct {
    LogLevel level;         // 最低日志级别
    LogSink sink;           // 输出目标
    FILE *file;             // 日志文件句柄
    int color_enabled;      // 是否启用颜色
    int timestamp_enabled;  // 是否显示时间戳
    char file_path[256];    // 日志文件路径
    long max_file_size;     // 文件最大大小（字节）
    int rotate_count;       // 轮转文件数量
} LoggerConfig;

/* 日志器 */
typedef struct {
    LoggerConfig config;
    int initialized;
} Logger;

static Logger logger = {
    .config = {
        .level = LOG_INFO,
        .sink = LOG_SINK_STDOUT,
        .file = NULL,
        .color_enabled = 1,
        .timestamp_enabled = 1,
        .max_file_size = 10 * 1024 * 1024,  // 10MB
        .rotate_count = 5
    },
    .initialized = 0
};
```

## 3. 核心功能

### 3.1 初始化

```c
/* 初始化日志器 */
void log_init(LogLevel level, const char *file_path) {
    logger.config.level = level;

    if (file_path) {
        strncpy(logger.config.file_path, file_path, sizeof(logger.config.file_path) - 1);
        logger.config.file = fopen(file_path, "a");
        logger.config.sink = LOG_SINK_BOTH;
    }

    logger.initialized = 1;
}

/* 设置日志级别 */
void log_set_level(LogLevel level) {
    logger.config.level = level;
}

/* 设置输出目标 */
void log_set_sink(LogSink sink) {
    logger.config.sink = sink;
}

/* 关闭日志器 */
void log_shutdown() {
    if (logger.config.file) {
        fclose(logger.config.file);
        logger.config.file = NULL;
    }
    logger.initialized = 0;
}
```

### 3.2 文件轮转

```c
/* 日志文件轮转 */
static void rotate_log_file() {
    if (!logger.config.file) return;

    // 获取当前文件大小
    fseek(logger.config.file, 0, SEEK_END);
    long size = ftell(logger.config.file);

    if (size < logger.config.max_file_size) {
        fseek(logger.config.file, 0, SEEK_END);
        return;
    }

    fclose(logger.config.file);

    // 轮转文件
    char old_path[300], new_path[300];
    for (int i = logger.config.rotate_count - 1; i > 0; i--) {
        snprintf(old_path, sizeof(old_path), "%s.%d",
                 logger.config.file_path, i - 1);
        snprintf(new_path, sizeof(new_path), "%s.%d",
                 logger.config.file_path, i);
        rename(old_path, new_path);
    }

    snprintf(new_path, sizeof(new_path), "%s.0", logger.config.file_path);
    rename(logger.config.file_path, new_path);

    logger.config.file = fopen(logger.config.file_path, "a");
}
```

### 3.3 格式化输出

```c
/* 获取当前时间字符串 */
static void get_timestamp(char *buf, int size) {
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    strftime(buf, size, "%Y-%m-%d %H:%M:%S", tm_info);
}

/* 核心日志函数 */
void log_log(LogLevel level, const char *file, int line,
             const char *fmt, ...) {
    if (!logger.initialized || level < logger.config.level) return;

    // 构建日志消息
    char message[4096];
    va_list args;
    va_start(args, fmt);
    vsnprintf(message, sizeof(message), fmt, args);
    va_end(args);

    // 时间戳
    char timestamp[32];
    if (logger.config.timestamp_enabled) {
        get_timestamp(timestamp, sizeof(timestamp));
    }

    // 提取文件名（去掉路径）
    const char *filename = file;
    const char *slash = strrchr(file, '/');
#ifdef _WIN32
    const char *bslash = strrchr(file, '\\');
    if (bslash && (!slash || bslash > slash)) slash = bslash;
#endif
    if (slash) filename = slash + 1;

    // 控制台输出
    if (logger.config.sink & LOG_SINK_STDOUT) {
        if (logger.config.color_enabled) {
            if (logger.config.timestamp_enabled) {
                fprintf(stderr, "%s [%s%-5s\033[0m] %s:%d - %s\n",
                        timestamp, level_colors[level], level_names[level],
                        filename, line, message);
            } else {
                fprintf(stderr, "[%s%-5s\033[0m] %s:%d - %s\n",
                        level_colors[level], level_names[level],
                        filename, line, message);
            }
        } else {
            if (logger.config.timestamp_enabled) {
                fprintf(stderr, "%s [%-5s] %s:%d - %s\n",
                        timestamp, level_names[level],
                        filename, line, message);
            } else {
                fprintf(stderr, "[%-5s] %s:%d - %s\n",
                        level_names[level], filename, line, message);
            }
        }
    }

    // 文件输出
    if ((logger.config.sink & LOG_SINK_FILE) && logger.config.file) {
        rotate_log_file();
        if (logger.config.file) {
            if (logger.config.timestamp_enabled) {
                fprintf(logger.config.file, "%s [%-5s] %s:%d - %s\n",
                        timestamp, level_names[level],
                        filename, line, message);
            }
            fflush(logger.config.file);
        }
    }
}
```

## 4. 便捷宏

```c
/* 便捷日志宏 */
#define log_trace(...) log_log(LOG_TRACE, __FILE__, __LINE__, __VA_ARGS__)
#define log_debug(...) log_log(LOG_DEBUG, __FILE__, __LINE__, __VA_ARGS__)
#define log_info(...)  log_log(LOG_INFO,  __FILE__, __LINE__, __VA_ARGS__)
#define log_warn(...)  log_log(LOG_WARN,  __FILE__, __LINE__, __VA_ARGS__)
#define log_error(...) log_log(LOG_ERROR, __FILE__, __LINE__, __VA_ARGS__)
#define log_fatal(...) log_log(LOG_FATAL, __FILE__, __LINE__, __VA_ARGS__)

/* 条件日志 */
#define log_if(cond, level, ...) do { \
    if (cond) log_log(level, __FILE__, __LINE__, __VA_ARGS__); \
} while(0)
```

## 5. 性能优化

### 5.1 异步日志

```c
/*
 * 异步日志思路：
 * 使用环形缓冲区 + 后台线程
 * 1. 日志消息写入环形缓冲区
 * 2. 后台线程从缓冲区读取并写入文件
 * 3. 避免文件I/O阻塞主线程
 */

typedef struct {
    char **messages;        // 消息数组（环形缓冲区）
    int capacity;           // 缓冲区容量
    int head;               // 写入位置
    int tail;               // 读取位置
    int count;              // 当前消息数
    int shutdown;           // 关闭标志
    // 实际实现需要互斥锁和条件变量
} AsyncLogBuffer;

/* 异步写入（简化版，不包含线程同步） */
void async_log_write(AsyncLogBuffer *buf, const char *msg) {
    if (buf->count >= buf->capacity) return;  // 缓冲区满

    int idx = buf->head;
    if (buf->messages[idx]) free(buf->messages[idx]);
    buf->messages[idx] = strdup(msg);
    buf->head = (buf->head + 1) % buf->capacity;
    buf->count++;
}
```

### 5.2 格式化缓存

```c
/* 线程本地格式化缓冲区，减少内存分配 */
static THREAD_LOCAL char fmt_buffer[4096];

const char *log_format(const char *fmt, ...) {
    va_list args;
    va_start(args, fmt);
    vsnprintf(fmt_buffer, sizeof(fmt_buffer), fmt, args);
    va_end(args);
    return fmt_buffer;
}
```

## 6. 测试

```c
int main() {
    // 初始化
    log_init(LOG_DEBUG, NULL);

    // 使用便捷宏
    log_trace("这是一条trace消息");
    log_debug("调试信息: x = %d", 42);
    log_info("服务器启动在端口 %d", 8080);
    log_warn("内存使用率: %.1f%%", 85.5);
    log_error("连接失败: %s", "timeout");
    log_fatal("程序崩溃，退出码: %d", -1);

    // 条件日志
    int error_code = 5;
    log_if(error_code != 0, LOG_ERROR, "检测到错误码: %d", error_code);

    // 关闭
    log_shutdown();

    return 0;
}
```

## 小结

- 日志库应支持多级别过滤，减少不必要的I/O开销
- 文件轮转防止单个日志文件无限增长
- 异步日志将I/O操作放到后台线程，避免阻塞主线程
- 使用 `__FILE__` 和 `__LINE__` 宏自动记录代码位置
- 实际生产中可参考log4c、spdlog等成熟库的设计
