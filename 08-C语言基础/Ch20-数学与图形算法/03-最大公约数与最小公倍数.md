# 最大公约数与最小公倍数

## 概述

最大公约数(GCD)和最小公倍数(LCM)是最基础的数论运算，在分数化简、周期问题、密码学等领域有广泛应用。

---

## 1. 欧几里得算法（辗转相除法）

### 1.1 基本原理

核心性质：`gcd(a, b) = gcd(b, a % b)`，当 `b == 0` 时，`a` 即为答案。

```c
#include <stdio.h>

/* 递归版本 */
int gcd_recursive(int a, int b) {
    if (b == 0) return a;
    return gcd_recursive(b, a % b);
}

/* 迭代版本（推荐，无栈溢出风险） */
int gcd(int a, int b) {
    while (b != 0) {
        int t = b;
        b = a % b;
        a = t;
    }
    return a;
}

/* 最小公倍数 */
int lcm(int a, int b) {
    return a / gcd(a, b) * b;  // 先除后乘，防止溢出
}

int main() {
    printf("gcd(12, 18) = %d\n", gcd(12, 18));   // 6
    printf("gcd(35, 14) = %d\n", gcd(35, 14));   // 7
    printf("lcm(12, 18) = %d\n", lcm(12, 18));   // 36
    printf("lcm(4, 6)   = %d\n", lcm(4, 6));     // 12
    return 0;
}
```

### 1.2 长整数版本

```c
long long gcd_ll(long long a, long long b) {
    while (b) {
        a %= b;
        // 交换 a 和 b
        long long temp = a;
        a = b;
        b = temp;
    }
    return a < 0 ? -a : a;
}

long long lcm_ll(long long a, long long b) {
    return a / gcd_ll(a, b) * b;
}
```

## 2. 扩展欧几里得算法（Extended GCD）

### 2.1 原理

扩展GCD不仅能求出 `gcd(a, b)`，还能求出满足 `ax + by = gcd(a, b)` 的整数解 `x` 和 `y`。

```c
#include <stdio.h>

/*
 * 扩展欧几里得算法
 * 返回 gcd(a, b)，同时 *x 和 *y 满足 ax + by = gcd(a, b)
 */
int extended_gcd(int a, int b, int *x, int *y) {
    if (b == 0) {
        *x = 1;
        *y = 0;
        return a;
    }

    int x1, y1;
    int g = extended_gcd(b, a % b, &x1, &y1);

    *x = y1;
    *y = x1 - (a / b) * y1;

    return g;
}

int main() {
    int x, y;
    int a = 35, b = 15;

    int g = extended_gcd(a, b, &x, &y);
    printf("gcd(%d, %d) = %d\n", a, b, g);
    printf("%d * %d + %d * %d = %d\n", a, x, b, y, a * x + b * y);

    return 0;
}
```

### 2.2 迭代版本扩展GCD

```c
int extended_gcd_iter(int a, int b, int *x, int *y) {
    int x0 = 1, y0 = 0;
    int x1 = 0, y1 = 1;

    while (b != 0) {
        int q = a / b;

        int temp_b = b;
        b = a % b;
        a = temp_b;

        int temp_x = x1;
        x1 = x0 - q * x1;
        x0 = temp_x;

        int temp_y = y1;
        y1 = y0 - q * y1;
        y0 = temp_y;
    }

    *x = x0;
    *y = y0;
    return a;
}
```

## 3. 应用

### 3.1 求模逆元

给定 `a` 和模数 `m`，求 `x` 使得 `ax ≡ 1 (mod m)`。

```c
/*
 * 求 a 在模 m 下的逆元
 * 要求 gcd(a, m) = 1
 * 返回逆元，若不存在返回 -1
 */
int mod_inverse(int a, int m) {
    int x, y;
    int g = extended_gcd(a, m, &x, &y);
    if (g != 1) return -1;  // 逆元不存在

    return (x % m + m) % m;  // 确保结果为正
}

int main() {
    // 3 * 7 = 21 ≡ 1 (mod 10)
    int inv = mod_inverse(3, 10);
    printf("3 在模 10 下的逆元: %d\n", inv);  // 7
    printf("验证: 3 * 7 %% 10 = %d\n", (3 * 7) % 10);

    return 0;
}
```

### 3.2 解线性同余方程

求解 `ax ≡ b (mod m)`。

```c
/*
 * 解 ax ≡ b (mod m)
 * 返回解的个数，最小正解存入 *x0，解的间距存入 *d
 */
int solve_congruence(int a, int b, int m, int *x0, int *d) {
    int x, y;
    int g = extended_gcd(a, m, &x, &y);

    if (b % g != 0) return 0;  // 无解

    // 特解
    *x0 = (long long)x * (b / g) % m;
    *x0 = (*x0 % (m / g) + m / g) % (m / g);
    *d = m / g;

    return g;  // 有 g 个解
}
```

### 3.3 分数运算

```c
#include <stdio.h>

typedef struct {
    long long num;   // 分子
    long long den;   // 分母（始终为正）
} Fraction;

/* 化简分数 */
Fraction simplify(Fraction f) {
    if (f.den < 0) {
        f.num = -f.num;
        f.den = -f.den;
    }
    long long g = gcd_ll(f.num < 0 ? -f.num : f.num, f.den);
    f.num /= g;
    f.den /= g;
    return f;
}

/* 分数加法 */
Fraction frac_add(Fraction a, Fraction b) {
    Fraction result;
    result.num = a.num * b.den + b.num * a.den;
    result.den = a.den * b.den;
    return simplify(result);
}

/* 分数乘法 */
Fraction frac_mul(Fraction a, Fraction b) {
    Fraction result;
    result.num = a.num * b.num;
    result.den = a.den * b.den;
    return simplify(result);
}

void print_frac(Fraction f) {
    if (f.den == 1) {
        printf("%lld\n", f.num);
    } else {
        printf("%lld/%lld\n", f.num, f.den);
    }
}

int main() {
    Fraction a = {1, 3};
    Fraction b = {1, 6};

    printf("1/3 + 1/6 = ");
    print_frac(frac_add(a, b));  // 1/2

    Fraction c = {2, 5};
    Fraction d = {3, 4};
    printf("2/5 * 3/4 = ");
    print_frac(frac_mul(c, d));  // 3/10

    return 0;
}
```

## 4. 多个数的GCD和LCM

```c
/* 多个数的GCD */
int gcd_array(int arr[], int n) {
    int result = arr[0];
    for (int i = 1; i < n; i++) {
        result = gcd(result, arr[i]);
    }
    return result;
}

/* 多个数的LCM */
int lcm_array(int arr[], int n) {
    long long result = arr[0];
    for (int i = 1; i < n; i++) {
        result = lcm_ll(result, arr[i]);
    }
    return (int)result;
}

int main() {
    int arr[] = {12, 18, 24, 30};
    int n = sizeof(arr) / sizeof(arr[0]);

    printf("GCD(12,18,24,30) = %d\n", gcd_array(arr, n));  // 6
    printf("LCM(12,18,24,30) = %d\n", lcm_array(arr, n));  // 360

    return 0;
}
```

## 5. Stein算法（二进制GCD）

对于大整数，用位运算替代取模操作更快。

```c
int gcd_stein(int a, int b) {
    if (a == 0) return b;
    if (b == 0) return a;

    // 找出2的最大公因子
    int shift = 0;
    while (((a | b) & 1) == 0) {
        a >>= 1;
        b >>= 1;
        shift++;
    }

    // 去除a中的因子2
    while ((a & 1) == 0) a >>= 1;

    do {
        // 去除b中的因子2
        while ((b & 1) == 0) b >>= 1;

        // 确保 a <= b
        if (a > b) {
            int temp = a;
            a = b;
            b = temp;
        }
        b = b - a;
    } while (b != 0);

    return a << shift;
}
```

---

## 小结

- 欧几里得算法是求GCD最经典高效的方法，时间复杂度 O(log min(a,b))
- 扩展GCD能求解 `ax + by = gcd(a,b)` 的整数解，是数论的基础工具
- LCM可以通过 GCD 快速计算：`lcm(a,b) = a / gcd(a,b) * b`
- 分数运算中先约分再计算可以有效避免溢出
