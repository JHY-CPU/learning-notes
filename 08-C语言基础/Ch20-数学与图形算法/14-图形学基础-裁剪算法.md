# 图形学基础 - 裁剪算法

## 概述

裁剪是确定图元哪些部分在窗口（裁剪区域）内的过程。本节介绍经典的Cohen-Sutherland和Liang-Barsky线段裁剪算法。

---

## 1. 裁剪区域编码（Cohen-Sutherland）

### 1.1 区域码定义

将平面分为9个区域，用4位二进制码表示点相对于裁剪窗口的位置。

```c
#include <stdio.h>
#include <math.h>

/* 裁剪窗口 */
typedef struct {
    double xmin, ymin, xmax, ymax;
} ClipWindow;

/* 区域码位定义 */
#define LEFT   1   // 0001 - 在左边界左侧
#define RIGHT  2   // 0010 - 在右边界右侧
#define BOTTOM 4   // 0100 - 在下边界下方
#define TOP    8   // 1000 - 在上边界上方

/*
 * 计算点的区域码
 * 根据点与裁剪窗口的位置关系设置4位编码
 */
int compute_code(ClipWindow *win, double x, double y) {
    int code = 0;
    if (x < win->xmin)      code |= LEFT;
    else if (x > win->xmax) code |= RIGHT;
    if (y < win->ymin)      code |= BOTTOM;
    else if (y > win->ymax) code |= TOP;
    return code;
}
```

### 1.2 Cohen-Sutherland 线段裁剪

```c
/*
 * Cohen-Sutherland 线段裁剪算法
 * 返回值: 0=完全在窗口外, 1=完全或部分在窗口内
 * 裁剪后的线段端点存回 (x1,y1)-(x2,y2)
 */
int cohen_sutherland_clip(ClipWindow *win,
                          double *x1, double *y1,
                          double *x2, double *y2) {
    int code1 = compute_code(win, *x1, *y1);
    int code2 = compute_code(win, *x2, *y2);
    int accept = 0;

    while (1) {
        if (!(code1 | code2)) {
            // 两个端点都在窗口内
            accept = 1;
            break;
        } else if (code1 & code2) {
            // 两个端点在同一外侧区域，完全在窗口外
            break;
        } else {
            // 需要裁剪：选一个在窗口外的端点
            int code_out = code1 ? code1 : code2;
            double x, y;

            // 求线段与边界的交点
            if (code_out & TOP) {
                x = *x1 + (*x2 - *x1) * (win->ymax - *y1) / (*y2 - *y1);
                y = win->ymax;
            } else if (code_out & BOTTOM) {
                x = *x1 + (*x2 - *x1) * (win->ymin - *y1) / (*y2 - *y1);
                y = win->ymin;
            } else if (code_out & RIGHT) {
                y = *y1 + (*y2 - *y1) * (win->xmax - *x1) / (*x2 - *x1);
                x = win->xmax;
            } else {  // LEFT
                y = *y1 + (*y2 - *y1) * (win->xmin - *x1) / (*x2 - *x1);
                x = win->xmin;
            }

            // 替换外部端点
            if (code_out == code1) {
                *x1 = x; *y1 = y;
                code1 = compute_code(win, x, y);
            } else {
                *x2 = x; *y2 = y;
                code2 = compute_code(win, x, y);
            }
        }
    }

    return accept;
}
```

## 2. Liang-Barsky 裁剪算法

### 2.1 算法原理

将线段参数化为 `P(t) = P1 + t(P2-P1)`，`t ∈ [0,1]`，通过不等式求交。

```c
/*
 * Liang-Barsky 线段裁剪算法
 * 使用参数化方法，效率比Cohen-Sutherland高
 * 返回值: 0=完全在窗口外, 1=部分/完全在窗口内
 */
int liang_barsky_clip(ClipWindow *win,
                      double *x1, double *y1,
                      double *x2, double *y2) {
    double dx = *x2 - *x1;
    double dy = *y2 - *y1;

    double p[4], q[4];
    p[0] = -dx;  q[0] = *x1 - win->xmin;  // 左
    p[1] =  dx;  q[1] = win->xmax - *x1;  // 右
    p[2] = -dy;  q[2] = *y1 - win->ymin;  // 下
    p[3] =  dy;  q[3] = win->ymax - *y1;  // 上

    double t0 = 0.0, t1 = 1.0;  // 参数范围

    for (int i = 0; i < 4; i++) {
        if (fabs(p[i]) < 1e-10) {
            // 线段平行于该边界
            if (q[i] < 0) return 0;  // 完全在窗口外
        } else {
            double t = q[i] / p[i];
            if (p[i] < 0) {
                // 进入点
                if (t > t1) return 0;
                if (t > t0) t0 = t;
            } else {
                // 离开点
                if (t < t0) return 0;
                if (t < t1) t1 = t;
            }
        }
    }

    // 计算裁剪后的端点
    double new_x1 = *x1 + t0 * dx;
    double new_y1 = *y1 + t0 * dy;
    double new_x2 = *x1 + t1 * dx;
    double new_y2 = *y1 + t1 * dy;

    *x1 = new_x1; *y1 = new_y1;
    *x2 = new_x2; *y2 = new_y2;

    return 1;
}
```

## 3. 多边形裁剪（Sutherland-Hodgman）

```c
#define MAX_POLY_VERTS 100

typedef struct { double x, y; } Point2D;

/*
 * 判断点是否在裁剪边的内侧
 * edge: 0=左, 1=右, 2=下, 3=上
 */
int is_inside(Point2D p, int edge, ClipWindow *win) {
    switch (edge) {
        case 0: return p.x >= win->xmin;
        case 1: return p.x <= win->xmax;
        case 2: return p.y >= win->ymin;
        case 3: return p.y <= win->ymax;
    }
    return 0;
}

/* 求线段与裁剪边的交点 */
Point2D intersect(Point2D p1, Point2D p2, int edge, ClipWindow *win) {
    Point2D result;
    double t;

    switch (edge) {
        case 0: // 左
            t = (win->xmin - p1.x) / (p2.x - p1.x);
            result.x = win->xmin;
            result.y = p1.y + t * (p2.y - p1.y);
            break;
        case 1: // 右
            t = (win->xmax - p1.x) / (p2.x - p1.x);
            result.x = win->xmax;
            result.y = p1.y + t * (p2.y - p1.y);
            break;
        case 2: // 下
            t = (win->ymin - p1.y) / (p2.y - p1.y);
            result.y = win->ymin;
            result.x = p1.x + t * (p2.x - p1.x);
            break;
        case 3: // 上
            t = (win->ymax - p1.y) / (p2.y - p1.y);
            result.y = win->ymax;
            result.x = p1.x + t * (p2.x - p1.x);
            break;
    }
    return result;
}

/*
 * Sutherland-Hodgman 多边形裁剪
 * 依次用4条边裁剪多边形
 */
int sutherland_hodgman_clip(Point2D *input, int n_in,
                            Point2D *output, int *n_out,
                            ClipWindow *win) {
    Point2D temp[MAX_POLY_VERTS * 2];

    // 复制输入
    for (int i = 0; i < n_in; i++) temp[i] = input[i];
    int count = n_in;

    // 依次对4条边进行裁剪
    for (int edge = 0; edge < 4; edge++) {
        Point2D in[MAX_POLY_VERTS * 2];
        int in_count = count;
        for (int i = 0; i < count; i++) in[i] = temp[i];
        count = 0;

        for (int i = 0; i < in_count; i++) {
            Point2D curr = in[i];
            Point2D prev = in[(i + in_count - 1) % in_count];

            int curr_in = is_inside(curr, edge, win);
            int prev_in = is_inside(prev, edge, win);

            if (curr_in) {
                if (!prev_in) {
                    // 从外到内：加交点
                    temp[count++] = intersect(prev, curr, edge, win);
                }
                temp[count++] = curr;
            } else if (prev_in) {
                // 从内到外：加交点
                temp[count++] = intersect(prev, curr, edge, win);
            }
        }
    }

    *n_out = count;
    for (int i = 0; i < count; i++) output[i] = temp[i];
    return count;
}
```

## 4. 测试示例

```c
int main() {
    ClipWindow win = {10, 10, 70, 30};

    // 测试 Cohen-Sutherland
    double x1 = 0, y1 = 15, x2 = 80, y2 = 25;
    printf("Cohen-Sutherland 裁剪:\n");
    printf("  原始线段: (%.1f,%.1f) -> (%.1f,%.1f)\n", x1, y1, x2, y2);
    if (cohen_sutherland_clip(&win, &x1, &y1, &x2, &y2)) {
        printf("  裁剪结果: (%.1f,%.1f) -> (%.1f,%.1f)\n", x1, y1, x2, y2);
    } else {
        printf("  线段完全在窗口外\n");
    }

    // 测试 Liang-Barsky
    x1 = 5; y1 = 5; x2 = 75; y2 = 35;
    printf("\nLiang-Barsky 裁剪:\n");
    printf("  原始线段: (%.1f,%.1f) -> (%.1f,%.1f)\n", x1, y1, x2, y2);
    if (liang_barsky_clip(&win, &x1, &y1, &x2, &y2)) {
        printf("  裁剪结果: (%.1f,%.1f) -> (%.1f,%.1f)\n", x1, y1, x2, y2);
    } else {
        printf("  线段完全在窗口外\n");
    }

    // 测试多边形裁剪
    Point2D polygon[] = {{0, 0}, {80, 0}, {80, 35}, {0, 35}};
    Point2D clipped[MAX_POLY_VERTS * 2];
    int n_out;

    sutherland_hodgman_clip(polygon, 4, clipped, &n_out, &win);
    printf("\n多边形裁剪结果（%d个顶点）:\n", n_out);
    for (int i = 0; i < n_out; i++) {
        printf("  (%.1f, %.1f)\n", clipped[i].x, clipped[i].y);
    }

    return 0;
}
```

## 小结

- Cohen-Sutherland通过区域码快速判断完全可见/完全不可见的情况，需多次迭代处理部分可见
- Liang-Barsky使用参数化方法，只需一次遍历，效率更高
- Sutherland-Hodgman算法依次用每条裁剪边裁剪多边形，适合凸多边形
- 裁剪是图形管线的基本操作，GPU中由硬件高效实现
