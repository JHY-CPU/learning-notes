# 测试框架实现

## 概述

单元测试是保证代码质量的重要手段。本节实现一个简洁的C语言单元测试框架，支持断言、测试组织和结果统计。

---

## 1. 基本数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <setjmp.h>
#include <signal.h>

/* 测试结果 */
typedef struct {
    int total;          // 总测试数
    int passed;         // 通过数
    int failed;         // 失败数
    int errors;         // 错误数（崩溃等）
    int skipped;        // 跳过数
} TestStats;

/* 测试用例 */
typedef struct {
    const char *name;
    void (*func)(void);
    int enabled;
} TestCase;

/* 测试套件 */
typedef struct {
    const char *name;
    TestCase *cases;
    int capacity;
    int count;
    void (*setup)(void);     // 测试前的准备函数
    void (*teardown)(void);  // 测试后的清理函数
} TestSuite;

/* 全局状态 */
static TestStats global_stats = {0};
static jmp_buf test_jmp_buf;
static int current_test_passed;
static char failure_message[4096];
static int failure_count;
```

## 2. 断言宏

```c
/* 记录失败 */
static void record_failure(const char *file, int line, const char *msg) {
    failure_count++;
    current_test_passed = 0;

    int len = (int)strlen(failure_message);
    if (len < sizeof(failure_message) - 100) {
        snprintf(failure_message + len, sizeof(failure_message) - len,
                 "  FAIL [%s:%d]: %s\n", file, line, msg);
    }
}

/* 基本断言 */
#define ASSERT_TRUE(expr) do { \
    if (!(expr)) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 TRUE: %s", #expr); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

#define ASSERT_FALSE(expr) do { \
    if ((expr)) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 FALSE: %s", #expr); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

/* 整数比较 */
#define ASSERT_INT_EQ(a, b) do { \
    long long _a = (long long)(a), _b = (long long)(b); \
    if (_a != _b) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 %s == %s, 实际 %lld != %lld", \
                 #a, #b, _a, _b); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

#define ASSERT_INT_NE(a, b) do { \
    long long _a = (long long)(a), _b = (long long)(b); \
    if (_a == _b) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 %s != %s, 实际均为 %lld", \
                 #a, #b, _a); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

#define ASSERT_INT_GT(a, b) do { \
    long long _a = (long long)(a), _b = (long long)(b); \
    if (_a <= _b) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 %s > %s, 实际 %lld <= %lld", \
                 #a, #b, _a, _b); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

/* 浮点比较 */
#define ASSERT_DOUBLE_EQ(a, b, eps) do { \
    double _a = (double)(a), _b = (double)(b); \
    if (fabs(_a - _b) > (eps)) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 %s ~= %s, 实际 %.10f != %.10f (误差 %.2e)", \
                 #a, #b, _a, _b, fabs(_a - _b)); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

/* 字符串比较 */
#define ASSERT_STR_EQ(a, b) do { \
    const char *_a = (const char *)(a), *_b = (const char *)(b); \
    if (strcmp(_a, _b) != 0) { \
        char _msg[512]; \
        snprintf(_msg, sizeof(_msg), "期望 %s == %s, 实际 \"%s\" != \"%s\"", \
                 #a, #b, _a, _b); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

/* 指针断言 */
#define ASSERT_NULL(ptr) do { \
    if ((ptr) != NULL) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 %s == NULL", #ptr); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

#define ASSERT_NOT_NULL(ptr) do { \
    if ((ptr) == NULL) { \
        char _msg[256]; \
        snprintf(_msg, sizeof(_msg), "期望 %s != NULL", #ptr); \
        record_failure(__FILE__, __LINE__, _msg); \
    } \
} while(0)

/* 数组内容比较 */
#define ASSERT_MEM_EQ(a, b, size) do { \
    if (memcmp((a), (b), (size)) != 0) { \
        record_failure(__FILE__, __LINE__, "内存内容不一致: " #a " vs " #b); \
    } \
} while(0)

/* 失败并终止当前测试 */
#define TEST_FAIL(msg) do { \
    record_failure(__FILE__, __LINE__, msg); \
    longjmp(test_jmp_buf, 1); \
} while(0)

/* 跳过当前测试 */
#define TEST_SKIP(msg) do { \
    printf("    SKIP: %s\n", msg); \
    global_stats.skipped++; \
    longjmp(test_jmp_buf, 2); \
} while(0)
```

## 3. 测试套件管理

```c
/* 创建测试套件 */
TestSuite suite_create(const char *name) {
    TestSuite suite;
    suite.name = name;
    suite.capacity = 64;
    suite.count = 0;
    suite.cases = (TestCase *)malloc(suite.capacity * sizeof(TestCase));
    suite.setup = NULL;
    suite.teardown = NULL;
    return suite;
}

/* 添加测试用例 */
void suite_add(TestSuite *suite, const char *name, void (*func)(void)) {
    if (suite->count >= suite->capacity) {
        suite->capacity *= 2;
        suite->cases = (TestCase *)realloc(suite->cases,
            suite->capacity * sizeof(TestCase));
    }
    suite->cases[suite->count].name = name;
    suite->cases[suite->count].func = func;
    suite->cases[suite->count].enabled = 1;
    suite->count++;
}

/* 设置setup/teardown */
void suite_set_setup(TestSuite *suite, void (*setup)(void)) {
    suite->setup = setup;
}

void suite_set_teardown(TestSuite *suite, void (*teardown)(void)) {
    suite->teardown = teardown;
}
```

## 4. 运行测试

```c
/* 运行单个测试用例 */
static int run_test_case(TestSuite *suite, TestCase *tc) {
    current_test_passed = 1;
    failure_count = 0;
    failure_message[0] = '\0';

    // setup
    if (suite->setup) suite->setup();

    // 运行测试（通过setjmp/longjmp处理崩溃）
    if (setjmp(test_jmp_buf) == 0) {
        tc->func();
    }

    // teardown
    if (suite->teardown) suite->teardown();

    // 统计结果
    global_stats.total++;
    if (current_test_passed && failure_count == 0) {
        global_stats.passed++;
        printf("    \033[32mPASS\033[0m: %s\n", tc->name);
    } else {
        global_stats.failed++;
        printf("    \033[31mFAIL\033[0m: %s\n", tc->name);
        if (failure_message[0]) {
            printf("%s", failure_message);
        }
    }

    return current_test_passed;
}

/* 运行整个测试套件 */
int suite_run(TestSuite *suite) {
    printf("\n\033[36m=== 套件: %s ===\033[0m (%d 个测试)\n",
           suite->name, suite->count);

    for (int i = 0; i < suite->count; i++) {
        if (suite->cases[i].enabled) {
            run_test_case(suite, &suite->cases[i]);
        }
    }

    return global_stats.failed == 0 ? 0 : -1;
}

/* 释放套件 */
void suite_destroy(TestSuite *suite) {
    free(suite->cases);
    suite->cases = NULL;
    suite->count = 0;
}
```

## 5. 结果报告

```c
/* 打印测试总结 */
void test_report() {
    printf("\n");
    printf("========================================\n");
    printf("  测试总结\n");
    printf("========================================\n");
    printf("  总数:   %d\n", global_stats.total);
    printf("  \033[32m通过:   %d\033[0m\n", global_stats.passed);

    if (global_stats.failed > 0)
        printf("  \033[31m失败:   %d\033[0m\n", global_stats.failed);
    if (global_stats.errors > 0)
        printf("  \033[33m错误:   %d\033[0m\n", global_stats.errors);
    if (global_stats.skipped > 0)
        printf("  跳过:   %d\n", global_stats.skipped);

    printf("========================================\n");

    if (global_stats.failed == 0 && global_stats.errors == 0) {
        printf("  \033[32m全部通过!\033[0m\n");
    } else {
        printf("  \033[31m存在失败!\033[0m\n");
    }
    printf("========================================\n");
}

/* 重置统计 */
void test_reset_stats() {
    memset(&global_stats, 0, sizeof(global_stats));
}
```

## 6. 测试示例

```c
/* 被测试的函数 */
int add(int a, int b) { return a + b; }
int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

/* 测试用例 */
void test_add() {
    ASSERT_INT_EQ(add(2, 3), 5);
    ASSERT_INT_EQ(add(-1, 1), 0);
    ASSERT_INT_EQ(add(0, 0), 0);
    ASSERT_INT_EQ(add(100, 200), 300);
}

void test_factorial() {
    ASSERT_INT_EQ(factorial(0), 1);
    ASSERT_INT_EQ(factorial(1), 1);
    ASSERT_INT_EQ(factorial(5), 120);
    ASSERT_INT_EQ(factorial(10), 3628800);
}

void test_double_compare() {
    ASSERT_DOUBLE_EQ(0.1 + 0.2, 0.3, 1e-9);
    ASSERT_DOUBLE_EQ(sin(0), 0.0, 1e-10);
    ASSERT_DOUBLE_EQ(sqrt(4.0), 2.0, 1e-10);
}

void test_string() {
    ASSERT_STR_EQ("hello", "hello");
    ASSERT_TRUE(strlen("abc") == 3);
    ASSERT_FALSE(strcmp("abc", "def") == 0);
}

void test_memory() {
    int a[] = {1, 2, 3};
    int b[] = {1, 2, 3};
    int c[] = {1, 2, 4};
    ASSERT_MEM_EQ(a, b, sizeof(a));
    // 下面这行应该失败
    // ASSERT_MEM_EQ(a, c, sizeof(a));
}

int main() {
    test_reset_stats();

    TestSuite suite = suite_create("数学函数测试");
    suite_add(&suite, "加法测试", test_add);
    suite_add(&suite, "阶乘测试", test_factorial);
    suite_add(&suite, "浮点比较", test_double_compare);
    suite_add(&suite, "字符串测试", test_string);
    suite_add(&suite, "内存比较", test_memory);

    suite_run(&suite);
    test_report();

    suite_destroy(&suite);

    return global_stats.failed > 0 ? 1 : 0;
}
```

## 小结

- 断言宏自动记录失败的文件名和行号，方便定位问题
- `setjmp/longjmp` 机制可捕获测试中的段错误等异常
- 测试套件支持setup/teardown，每个测试前后的环境独立
- 框架输出彩色结果，便于在终端中快速定位问题
- 生产环境可使用Check、Unity等成熟的C测试框架
