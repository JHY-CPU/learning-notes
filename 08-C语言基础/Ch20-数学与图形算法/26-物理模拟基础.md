# 物理模拟基础

## 概述

物理模拟用数学模型描述物体的运动和交互。本节实现弹簧系统、粒子系统和碰撞检测。

## 1. 二维向量运算

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>

typedef struct {
    double x, y;
} Vec2;

Vec2 vec2_add(Vec2 a, Vec2 b) { return (Vec2){a.x+b.x, a.y+b.y}; }
Vec2 vec2_sub(Vec2 a, Vec2 b) { return (Vec2){a.x-b.x, a.y-b.y}; }
Vec2 vec2_scale(Vec2 v, double s) { return (Vec2){v.x*s, v.y*s}; }
double vec2_length(Vec2 v) { return sqrt(v.x*v.x + v.y*v.y); }
Vec2 vec2_normalize(Vec2 v) {
    double len = vec2_length(v);
    if (len < 1e-10) return (Vec2){0, 0};
    return vec2_scale(v, 1.0/len);
}
double vec2_dot(Vec2 a, Vec2 b) { return a.x*b.x + a.y*b.y; }
```

## 2. 弹簧质点系统

### 2.1 数据结构

```c
/* 质点 */
typedef struct {
    Vec2 pos;       // 位置
    Vec2 vel;       // 速度
    Vec2 force;     // 合力
    double mass;    // 质量
    int pinned;     // 是否固定
} Mass;

/* 弹簧 */
typedef struct {
    int a, b;          // 连接的两个质点索引
    double rest_len;   // 自然长度
    double k;          // 弹簧常数（刚度）
    double damping;    // 阻尼系数
} Spring;

/* 弹簧系统 */
typedef struct {
    Mass masses[100];
    Spring springs[200];
    int mass_count;
    int spring_count;
    Vec2 gravity;       // 重力加速度
} SpringSystem;
```

### 2.2 系统操作

```c
/* 添加质点 */
int spring_system_add_mass(SpringSystem *sys, Vec2 pos, double mass, int pinned) {
    int idx = sys->mass_count++;
    sys->masses[idx].pos = pos;
    sys->masses[idx].vel = (Vec2){0, 0};
    sys->masses[idx].force = (Vec2){0, 0};
    sys->masses[idx].mass = mass;
    sys->masses[idx].pinned = pinned;
    return idx;
}

/* 添加弹簧 */
void spring_system_add_spring(SpringSystem *sys, int a, int b, double k, double damping) {
    int idx = sys->spring_count++;
    sys->springs[idx].a = a;
    sys->springs[idx].b = b;
    sys->springs[idx].k = k;
    sys->springs[idx].damping = damping;
    // 计算自然长度
    Vec2 diff = vec2_sub(sys->masses[b].pos, sys->masses[a].pos);
    sys->springs[idx].rest_len = vec2_length(diff);
}

/* 初始化系统 */
void spring_system_init(SpringSystem *sys) {
    sys->mass_count = 0;
    sys->spring_count = 0;
    sys->gravity = (Vec2){0, -9.8};
}
```

### 2.3 物理模拟

```c
/*
 * 计算弹簧力
 * 使用胡克定律: F = -k * (当前长度 - 自然长度) + 阻尼力
 */
void apply_spring_force(SpringSystem *sys, Spring *s) {
    Mass *ma = &sys->masses[s->a];
    Mass *mb = &sys->masses[s->b];

    Vec2 diff = vec2_sub(mb->pos, ma->pos);
    double dist = vec2_length(diff);

    if (dist < 1e-10) return;

    Vec2 dir = vec2_scale(diff, 1.0 / dist);

    // 弹簧力
    double stretch = dist - s->rest_len;
    double force_mag = s->k * stretch;

    // 阻尼力
    Vec2 rel_vel = vec2_sub(mb->vel, ma->vel);
    double damp_mag = s->damping * vec2_dot(rel_vel, dir);

    Vec2 force = vec2_scale(dir, force_mag + damp_mag);

    if (!ma->pinned) ma->force = vec2_add(ma->force, force);
    if (!mb->pinned) mb->force = vec2_sub(mb->force, force);
}

/*
 * 使用显式欧拉法更新系统
 * dt: 时间步长
 */
void spring_system_update(SpringSystem *sys, double dt) {
    // 清零力
    for (int i = 0; i < sys->mass_count; i++) {
        sys->masses[i].force = (Vec2){0, 0};
    }

    // 施加重力
    for (int i = 0; i < sys->mass_count; i++) {
        if (!sys->masses[i].pinned) {
            Vec2 g = vec2_scale(sys->gravity, sys->masses[i].mass);
            sys->masses[i].force = vec2_add(sys->masses[i].force, g);
        }
    }

    // 施加弹簧力
    for (int i = 0; i < sys->spring_count; i++) {
        apply_spring_force(sys, &sys->springs[i]);
    }

    // 更新速度和位置
    for (int i = 0; i < sys->mass_count; i++) {
        if (sys->masses[i].pinned) continue;

        Mass *m = &sys->masses[i];
        Vec2 accel = vec2_scale(m->force, 1.0 / m->mass);
        m->vel = vec2_add(m->vel, vec2_scale(accel, dt));
        m->pos = vec2_add(m->pos, vec2_scale(m->vel, dt));

        // 地面碰撞
        if (m->pos.y < 0) {
            m->pos.y = 0;
            m->vel.y = -m->vel.y * 0.8;  // 反弹系数
        }
    }
}
```

## 3. 粒子系统

### 3.1 粒子定义

```c
#define MAX_PARTICLES 1000

typedef struct {
    Vec2 pos;
    Vec2 vel;
    Vec2 accel;
    double life;        // 剩余生命（秒）
    double max_life;    // 最大生命值
    char symbol;        // 显示字符
    int active;
} Particle;

typedef struct {
    Particle particles[MAX_PARTICLES];
    int count;
    Vec2 emit_pos;      // 发射位置
    double emit_rate;   // 发射速率（每秒）
    double emit_timer;
    Vec2 gravity;
} ParticleEmitter;
```

### 3.2 粒子发射器

```c
void emitter_init(ParticleEmitter *em, Vec2 pos) {
    memset(em->particles, 0, sizeof(em->particles));
    em->count = 0;
    em->emit_pos = pos;
    em->emit_rate = 50;
    em->emit_timer = 0;
    em->gravity = (Vec2){0, -9.8};
}

/* 发射一个粒子 */
void emit_particle(ParticleEmitter *em) {
    for (int i = 0; i < MAX_PARTICLES; i++) {
        if (!em->particles[i].active) {
            Particle *p = &em->particles[i];
            p->pos = em->emit_pos;
            p->vel = (Vec2){
                ((rand() % 200) - 100) / 50.0,
                (rand() % 100) / 20.0 + 5.0
            };
            p->accel = em->gravity;
            p->life = 2.0 + (rand() % 100) / 100.0;
            p->max_life = p->life;
            p->symbol = ".oO*+#"[rand() % 6];
            p->active = 1;
            return;
        }
    }
}

void emitter_update(ParticleEmitter *em, double dt) {
    // 发射新粒子
    em->emit_timer += dt;
    while (em->emit_timer >= 1.0 / em->emit_rate) {
        emit_particle(em);
        em->emit_timer -= 1.0 / em->emit_rate;
    }

    // 更新现有粒子
    for (int i = 0; i < MAX_PARTICLES; i++) {
        if (!em->particles[i].active) continue;

        Particle *p = &em->particles[i];
        p->life -= dt;

        if (p->life <= 0) {
            p->active = 0;
            continue;
        }

        // Verlet积分
        p->vel = vec2_add(p->vel, vec2_scale(p->accel, dt));
        p->pos = vec2_add(p->pos, vec2_scale(p->vel, dt));

        // 地面碰撞
        if (p->pos.y < 0) {
            p->pos.y = 0;
            p->vel.y = -p->vel.y * 0.5;
        }
    }
}
```

## 4. 碰撞检测

### 4.1 圆形碰撞

```c
/* 圆形 */
typedef struct {
    Vec2 center;
    double radius;
} Circle;

/* AABB矩形 */
typedef struct {
    Vec2 min;  // 左下角
    Vec2 max;  // 右上角
} AABB;

/* 圆-圆碰撞检测 */
int circle_circle_collision(Circle a, Circle b) {
    Vec2 diff = vec2_sub(b.center, a.center);
    double dist = vec2_length(diff);
    return dist <= (a.radius + b.radius);
}

/* 圆-圆碰撞响应 */
void circle_circle_resolve(Circle *a, Circle *b,
                           Vec2 *va, Vec2 *vb,
                           double restitution) {
    Vec2 diff = vec2_sub(b->center, a->center);
    double dist = vec2_length(diff);
    double overlap = a->radius + b->radius - dist;

    if (overlap <= 0) return;

    Vec2 normal = vec2_normalize(diff);

    // 分离重叠
    double total_mass = 1.0;  // 简化：假设等质量
    a->center = vec2_sub(a->center, vec2_scale(normal, overlap * 0.5));
    b->center = vec2_add(b->center, vec2_scale(normal, overlap * 0.5));

    // 碰撞响应（弹性碰撞）
    Vec2 rel_vel = vec2_sub(*vb, *va);
    double vel_along_normal = vec2_dot(rel_vel, normal);

    if (vel_along_normal > 0) return;  // 已经在分离

    double impulse = -(1 + restitution) * vel_along_normal / 2.0;
    Vec2 impulse_vec = vec2_scale(normal, impulse);

    *va = vec2_sub(*va, impulse_vec);
    *vb = vec2_add(*vb, impulse_vec);
}
```

### 4.2 AABB碰撞

```c
/* AABB碰撞检测 */
int aabb_collision(AABB a, AABB b) {
    return (a.min.x <= b.max.x && a.max.x >= b.min.x &&
            a.min.y <= b.max.y && a.max.y >= b.min.y);
}

/* AABB碰撞响应（简单推离） */
void aabb_resolve(AABB *a, AABB *b, Vec2 *va, Vec2 *vb) {
    // 计算重叠量
    double overlap_x, overlap_y;

    if (a->max.x - b->min.x < b->max.x - a->min.x)
        overlap_x = a->max.x - b->min.x;
    else
        overlap_x = -(b->max.x - a->min.x);

    if (a->max.y - b->min.y < b->max.y - a->min.y)
        overlap_y = a->max.y - b->min.y;
    else
        overlap_y = -(b->max.y - a->min.y);

    // 沿最小重叠方向分离
    if (fabs(overlap_x) < fabs(overlap_y)) {
        a->min.x -= overlap_x * 0.5; a->max.x -= overlap_x * 0.5;
        b->min.x += overlap_x * 0.5; b->max.x += overlap_x * 0.5;
        va->x = -va->x * 0.8; vb->x = -vb->x * 0.8;
    } else {
        a->min.y -= overlap_y * 0.5; a->max.y -= overlap_y * 0.5;
        b->min.y += overlap_y * 0.5; b->max.y += overlap_y * 0.5;
        va->y = -va->y * 0.8; vb->y = -vb->y * 0.8;
    }
}
```

### 4.3 圆-线段碰撞

```c
/* 线段 */
typedef struct {
    Vec2 start, end;
} Segment;

/* 圆-线段碰撞检测 */
int circle_segment_collision(Circle c, Segment s, Vec2 *normal) {
    Vec2 seg_vec = vec2_sub(s.end, s.start);
    Vec2 circle_to_start = vec2_sub(c.center, s.start);

    double t = vec2_dot(circle_to_start, seg_vec) / vec2_dot(seg_vec, seg_vec);
    if (t < 0) t = 0;
    if (t > 1) t = 1;

    Vec2 closest = vec2_add(s.start, vec2_scale(seg_vec, t));
    Vec2 diff = vec2_sub(c.center, closest);
    double dist = vec2_length(diff);

    if (dist < c.radius) {
        if (normal) *normal = vec2_normalize(diff);
        return 1;
    }
    return 0;
}
```

```c
int main() {
    // 弹簧系统示例：悬挂的弹簧
    SpringSystem sys;
    spring_system_init(&sys);

    // 添加固定点和悬挂质点
    int pin = spring_system_add_mass(&sys, (Vec2){0, 5}, 1.0, 1);
    int m1  = spring_system_add_mass(&sys, (Vec2){0, 3}, 1.0, 0);
    int m2  = spring_system_add_mass(&sys, (Vec2){0, 1}, 1.0, 0);

    spring_system_add_spring(&sys, pin, m1, 100.0, 5.0);
    spring_system_add_spring(&sys, m1, m2, 100.0, 5.0);

    printf("弹簧质点系统模拟:\n");
    printf("%8s %10s %10s\n", "时间", "质点1.y", "质点2.y");

    double dt = 0.01;
    for (int step = 0; step <= 200; step++) {
        if (step % 20 == 0) {
            printf("%8.2f %10.4f %10.4f\n",
                   step * dt,
                   sys.masses[m1].pos.y,
                   sys.masses[m2].pos.y);
        }
        spring_system_update(&sys, dt);
    }

    return 0;
}
```

## 小结

- 弹簧系统使用胡克定律计算力，配合阻尼实现真实的弹性运动
- 粒子系统适合模拟烟雾、火焰、水流等效果
- 碰撞检测分为窄相（精确检测）和宽相（快速排除）两个阶段
- 物理模拟中时间步长的选择很重要，过大会导致不稳定
