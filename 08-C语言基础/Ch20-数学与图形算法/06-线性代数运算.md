# 线性代数运算

## 概述

线性代数在科学计算、图形学、机器学习等领域中不可或缺。本节用C语言实现矩阵运算、高斯消元和行列式计算。

## 1. 矩阵基本运算

### 1.1 矩阵数据结构与基本操作

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

#define MAXN 100
#define EPS 1e-9

typedef struct {
    int rows, cols;
    double data[MAXN][MAXN];
} Matrix;

/* 创建矩阵 */
Matrix mat_create(int rows, int cols) {
    Matrix m;
    m.rows = rows;
    m.cols = cols;
    memset(m.data, 0, sizeof(m.data));
    return m;
}

/* 打印矩阵 */
void mat_print(Matrix m) {
    for (int i = 0; i < m.rows; i++) {
        for (int j = 0; j < m.cols; j++) {
            printf("%10.4f ", m.data[i][j]);
        }
        printf("\n");
    }
}

/* 矩阵加法 */
Matrix mat_add(Matrix A, Matrix B) {
    Matrix C = mat_create(A.rows, A.cols);
    for (int i = 0; i < A.rows; i++)
        for (int j = 0; j < A.cols; j++)
            C.data[i][j] = A.data[i][j] + B.data[i][j];
    return C;
}

/* 矩阵减法 */
Matrix mat_sub(Matrix A, Matrix B) {
    Matrix C = mat_create(A.rows, A.cols);
    for (int i = 0; i < A.rows; i++)
        for (int j = 0; j < A.cols; j++)
            C.data[i][j] = A.data[i][j] - B.data[i][j];
    return C;
}

/* 矩阵乘法 */
Matrix mat_mul(Matrix A, Matrix B) {
    Matrix C = mat_create(A.rows, B.cols);
    for (int i = 0; i < A.rows; i++)
        for (int k = 0; k < A.cols; k++)
            if (fabs(A.data[i][k]) > EPS)
                for (int j = 0; j < B.cols; j++)
                    C.data[i][j] += A.data[i][k] * B.data[k][j];
    return C;
}

/* 标量乘法 */
Matrix mat_scale(Matrix A, double s) {
    Matrix C = mat_create(A.rows, A.cols);
    for (int i = 0; i < A.rows; i++)
        for (int j = 0; j < A.cols; j++)
            C.data[i][j] = A.data[i][j] * s;
    return C;
}

/* 转置 */
Matrix mat_transpose(Matrix A) {
    Matrix C = mat_create(A.cols, A.rows);
    for (int i = 0; i < A.rows; i++)
        for (int j = 0; j < A.cols; j++)
            C.data[j][i] = A.data[i][j];
    return C;
}
```

### 1.2 单位矩阵

```c
/* 创建n阶单位矩阵 */
Matrix mat_identity(int n) {
    Matrix I = mat_create(n, n);
    for (int i = 0; i < n; i++)
        I.data[i][i] = 1.0;
    return I;
```

## 2. 高斯消元法

### 2.1 求解线性方程组

求解 `Ax = b`，其中A是n×n矩阵，b是n维向量。

```c
/*
 * 高斯消元法求解线性方程组 Ax = b
 * 返回值: 0=唯一解, 1=无穷多解, -1=无解
 * 解存入x数组
 */
int gaussian_elimination(double A[][MAXN], double b[], double x[], int n) {
    // 构建增广矩阵
    double aug[MAXN][MAXN + 1];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++)
            aug[i][j] = A[i][j];
        aug[i][n] = b[i];
    }

    // 消元过程
    for (int col = 0; col < n; col++) {
        // 选主元（列主元消去法，提高数值稳定性）
        int max_row = col;
        for (int row = col + 1; row < n; row++) {
            if (fabs(aug[row][col]) > fabs(aug[max_row][col]))
                max_row = row;
        }

        // 交换行
        for (int j = 0; j <= n; j++) {
            double temp = aug[col][j];
            aug[col][j] = aug[max_row][j];
            aug[max_row][j] = temp;
        }

        // 主元为零，跳过
        if (fabs(aug[col][col]) < EPS) continue;

        // 消元
        for (int row = col + 1; row < n; row++) {
            double factor = aug[row][col] / aug[col][col];
            for (int j = col; j <= n; j++) {
                aug[row][j] -= factor * aug[col][j];
            }
        }
    }

    // 回代求解
    for (int i = n - 1; i >= 0; i--) {
        if (fabs(aug[i][i]) < EPS) {
            if (fabs(aug[i][n]) > EPS) return -1;  // 无解
            continue;
        }
        x[i] = aug[i][n];
        for (int j = i + 1; j < n; j++) {
            x[i] -= aug[i][j] * x[j];
        }
        x[i] /= aug[i][i];
    }

    return 0;
}

int main() {
    // 求解:
    // 2x + y - z = 8
    // -3x - y + 2z = -11
    // -2x + y + 2z = -3
    double A[MAXN][MAXN] = {
        {2,  1, -1},
        {-3, -1,  2},
        {-2,  1,  2}
    };
    double b[] = {8, -11, -3};
    double x[MAXN];
    int n = 3;

    int result = gaussian_elimination(A, b, x, n);
    if (result == 0) {
        printf("解:\n");
        for (int i = 0; i < n; i++) {
            printf("x[%d] = %.4f\n", i, x[i]);
        }
    }

    return 0;
}
```

### 2.2 矩阵求逆

```c
/*
 * 高斯-约旦消元法求矩阵的逆
 * 返回值: 0=成功, -1=不可逆
 */
int mat_inverse(double A[][MAXN], double inv[][MAXN], int n) {
    // 构建 [A | I]
    double aug[MAXN][2 * MAXN];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++)
            aug[i][j] = A[i][j];
        for (int j = 0; j < n; j++)
            aug[i][n + j] = (i == j) ? 1.0 : 0.0;
    }

    for (int col = 0; col < n; col++) {
        // 选主元
        int max_row = col;
        for (int row = col + 1; row < n; row++) {
            if (fabs(aug[row][col]) > fabs(aug[max_row][col]))
                max_row = row;
        }

        // 交换行
        for (int j = 0; j < 2 * n; j++) {
            double temp = aug[col][j];
            aug[col][j] = aug[max_row][j];
            aug[max_row][j] = temp;
        }

        if (fabs(aug[col][col]) < EPS) return -1;  // 不可逆

        // 将主元归一化
        double pivot = aug[col][col];
        for (int j = 0; j < 2 * n; j++)
            aug[col][j] /= pivot;

        // 消去其他行
        for (int row = 0; row < n; row++) {
            if (row == col) continue;
            double factor = aug[row][col];
            for (int j = 0; j < 2 * n; j++)
                aug[row][j] -= factor * aug[col][j];
        }
    }

    // 提取逆矩阵
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            inv[i][j] = aug[i][n + j];

    return 0;
}
```

## 3. 行列式计算

### 3.1 高斯消元法求行列式

```c
/*
 * 计算n阶方阵的行列式
 * 使用高斯消元，时间复杂度 O(n^3)
 */
double determinant(double A[][MAXN], int n) {
    double mat[MAXN][MAXN];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            mat[i][j] = A[i][j];

    double det = 1.0;

    for (int col = 0; col < n; col++) {
        int max_row = col;
        for (int row = col + 1; row < n; row++) {
            if (fabs(mat[row][col]) > fabs(mat[max_row][col]))
                max_row = row;
        }

        if (max_row != col) {
            for (int j = 0; j < n; j++) {
                double temp = mat[col][j];
                mat[col][j] = mat[max_row][j];
                mat[max_row][j] = temp;
            }
            det = -det;  // 行交换改变行列式符号
        }

        if (fabs(mat[col][col]) < EPS) return 0.0;

        det *= mat[col][col];

        for (int row = col + 1; row < n; row++) {
            double factor = mat[row][col] / mat[col][col];
            for (int j = col; j < n; j++) {
                mat[row][j] -= factor * mat[col][j];
            }
        }
    }

    return det;
}

int main() {
    double A[MAXN][MAXN] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 0}
    };

    printf("det(A) = %.4f\n", determinant(A, 3));  // 27

    return 0;
}
```

## 4. 矩阵的秩

```c
/* 计算矩阵的秩 */
int mat_rank(double A[][MAXN], int rows, int cols) {
    double mat[MAXN][MAXN];
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            mat[i][j] = A[i][j];

    int rank = 0;
    for (int col = 0; col < cols && rank < rows; col++) {
        int max_row = rank;
        for (int row = rank + 1; row < rows; row++) {
            if (fabs(mat[row][col]) > fabs(mat[max_row][col]))
                max_row = row;
        }

        if (fabs(mat[max_row][col]) < EPS) continue;

        for (int j = 0; j < cols; j++) {
            double temp = mat[rank][j];
            mat[rank][j] = mat[max_row][j];
            mat[max_row][j] = temp;
        }

        double pivot = mat[rank][col];
        for (int j = 0; j < cols; j++) mat[rank][j] /= pivot;

        for (int row = 0; row < rows; row++) {
            if (row == rank) continue;
            double factor = mat[row][col];
            for (int j = 0; j < cols; j++)
                mat[row][j] -= factor * mat[rank][j];
        }
        rank++;
    }
    return rank;
}
```

## 5. LU分解

```c
/*
 * LU分解: A = L * U
 * L为下三角矩阵，U为上三角矩阵
 */
int lu_decomposition(double A[][MAXN], double L[][MAXN], double U[][MAXN], int n) {
    memset(L, 0, sizeof(double) * MAXN * MAXN);
    memset(U, 0, sizeof(double) * MAXN * MAXN);

    for (int i = 0; i < n; i++) {
        // 计算U的第i行
        for (int j = i; j < n; j++) {
            double sum = 0;
            for (int k = 0; k < i; k++)
                sum += L[i][k] * U[k][j];
            U[i][j] = A[i][j] - sum;
        }

        // 计算L的第i列
        for (int j = i; j < n; j++) {
            if (i == j) {
                L[i][i] = 1.0;
            } else {
                double sum = 0;
                for (int k = 0; k < i; k++)
                    sum += L[j][k] * U[k][i];
                if (fabs(U[i][i]) < EPS) return -1;
                L[j][i] = (A[j][i] - sum) / U[i][i];
            }
        }
    }
    return 0;
}
```

## 小结

- 矩阵基本运算包括加法、乘法、转置等，时间复杂度通常是 O(n^3)
- 高斯消元法是求解线性方程组的核心算法，列主元消去提高数值稳定性
- 行列式可以通过高斯消元法高效计算
- LU分解将矩阵分解为下三角和上三角矩阵的乘积，便于求解多个方程组
