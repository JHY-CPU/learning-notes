# 图像处理基础

## 概述

数字图像处理是对图像进行数学运算以达到特定目的的技术。本节使用纯C语言实现灰度化、二值化、滤波和边缘检测等基础操作。

## 1. 图像数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

/* RGB像素 */
typedef struct {
    unsigned char r, g, b;
} Pixel;

/* 图像结构 */
typedef struct {
    int width, height;
    Pixel *data;  // 行优先存储
} Image;

/* 灰度图像 */
typedef struct {
    int width, height;
    unsigned char *data;
} GrayImage;

Image *image_create(int w, int h) {
    Image *img = (Image *)malloc(sizeof(Image));
    img->width = w;
    img->height = h;
    img->data = (Pixel *)calloc(w * h, sizeof(Pixel));
    return img;
}

GrayImage *gray_create(int w, int h) {
    GrayImage *img = (GrayImage *)malloc(sizeof(GrayImage));
    img->width = w;
    img->height = h;
    img->data = (unsigned char *)calloc(w * h, sizeof(unsigned char));
    return img;
}

void image_free(Image *img) {
    if (img) { free(img->data); free(img); }
}

void gray_free(GrayImage *img) {
    if (img) { free(img->data); free(img); }
}

Pixel image_get(Image *img, int x, int y) {
    return img->data[y * img->width + x];
}

void image_set(Image *img, int x, int y, Pixel p) {
    img->data[y * img->width + x] = p;
}

unsigned char gray_get(GrayImage *img, int x, int y) {
    return img->data[y * img->width + x];
}

void gray_set(GrayImage *img, int x, int y, unsigned char v) {
    img->data[y * img->width + x] = v;
}
```

## 2. 灰度化

### 2.1 加权灰度化

```c
/*
 * RGB转灰度图
 * 使用ITU-R BT.601标准权重
 * Gray = 0.299*R + 0.587*G + 0.114*B
 */
GrayImage *rgb_to_gray(Image *img) {
    GrayImage *gray = gray_create(img->width, img->height);

    for (int i = 0; i < img->width * img->height; i++) {
        Pixel p = img->data[i];
        unsigned char g = (unsigned char)(0.299 * p.r + 0.587 * p.g + 0.114 * p.b);
        gray->data[i] = g;
    }

    return gray;
}

/* 平均灰度化（简化版本） */
GrayImage *rgb_to_gray_avg(Image *img) {
    GrayImage *gray = gray_create(img->width, img->height);
    for (int i = 0; i < img->width * img->height; i++) {
        Pixel p = img->data[i];
        gray->data[i] = (unsigned char)((p.r + p.g + p.b) / 3);
    }
    return gray;
}
```

GrayImage *threshold_binary(GrayImage *img, unsigned char thresh) {

### 3.2 Otsu自动阈值

```c
/*
 * Otsu算法：自动寻找最佳二值化阈值
 * 使类间方差最大的阈值即为最佳阈值
 */
int otsu_threshold(GrayImage *img) {
    int histogram[256] = {0};
    int total = img->width * img->height;

    // 统计直方图
    for (int i = 0; i < total; i++) {
        histogram[img->data[i]]++;
    }

    double sum = 0;
    for (int i = 0; i < 256; i++) sum += i * histogram[i];

    double sumB = 0;
    int wB = 0;
    double max_variance = 0;
    int threshold = 0;

    for (int t = 0; t < 256; t++) {
        wB += histogram[t];
        if (wB == 0) continue;

        int wF = total - wB;
        if (wF == 0) break;

        sumB += t * histogram[t];
        double mB = sumB / wB;
        double mF = (sum - sumB) / wF;

        double variance = (double)wB * wF * (mB - mF) * (mB - mF);
        if (variance > max_variance) {
            max_variance = variance;
            threshold = t;
        }
    }

    return threshold;
}
```

## 4. 卷积与滤波

### 4.1 通用卷积操作

```c
/*
 * 二维卷积操作
 * kernel: 卷积核，ksize必须为奇数
 */
GrayImage *convolve(GrayImage *img, double *kernel, int ksize) {
    GrayImage *result = gray_create(img->width, img->height);
    int half = ksize / 2;

    // 计算卷积核的归一化因子
    double sum = 0;
    for (int i = 0; i < ksize * ksize; i++) sum += kernel[i];
    int normalize = (fabs(sum) > 1e-10);

    for (int y = half; y < img->height - half; y++) {
        for (int x = half; x < img->width - half; x++) {
            double val = 0;
            for (int ky = -half; ky <= half; ky++) {
                for (int kx = -half; kx <= half; kx++) {
                    int idx = (ky + half) * ksize + (kx + half);
                    val += kernel[idx] * gray_get(img, x + kx, y + ky);
                }
            }
            if (normalize && fabs(sum) > 1e-10) val /= sum;
            int v = (int)(val + 0.5);
            if (v < 0) v = 0;
            if (v > 255) v = 255;
            gray_set(result, x, y, (unsigned char)v);
        }
    }

    return result;
}
```

### 4.3 高斯滤波

```c
/* 生成高斯核 */
void generate_gaussian_kernel(double *kernel, int ksize, double sigma) {
    int half = ksize / 2;
    double sum = 0;

    for (int y = -half; y <= half; y++) {
        for (int x = -half; x <= half; x++) {
            double val = exp(-(x * x + y * y) / (2 * sigma * sigma));
            kernel[(y + half) * ksize + (x + half)] = val;
            sum += val;
        }
    }

    // 归一化
    for (int i = 0; i < ksize * ksize; i++) {
        kernel[i] /= sum;
    }
}

/* 高斯滤波 */
GrayImage *gaussian_filter(GrayImage *img, int ksize, double sigma) {
    double *kernel = (double *)malloc(ksize * ksize * sizeof(double));
    generate_gaussian_kernel(kernel, ksize, sigma);
    GrayImage *result = convolve(img, kernel, ksize);
    free(kernel);
    return result;
}
```

### 4.4 中值滤波（去椒盐噪声）

```c
/* 中值滤波 */
GrayImage *median_filter(GrayImage *img, int ksize) {
    GrayImage *result = gray_create(img->width, img->height);
    int half = ksize / 2;
    int *window = (int *)malloc(ksize * ksize * sizeof(int));

    for (int y = half; y < img->height - half; y++) {
        for (int x = half; x < img->width - half; x++) {
            int idx = 0;
            for (int ky = -half; ky <= half; ky++)
                for (int kx = -half; kx <= half; kx++)
                    window[idx++] = gray_get(img, x + kx, y + ky);

            // 排序取中值
            for (int i = 0; i < idx - 1; i++)
                for (int j = i + 1; j < idx; j++)
                    if (window[i] > window[j]) {
                        int t = window[i]; window[i] = window[j]; window[j] = t;
                    }

            gray_set(result, x, y, (unsigned char)window[idx / 2]);
        }
    }

    free(window);
    return result;
}
```

## 5. 边缘检测

### 5.1 Sobel算子

```c
/*
 * Sobel边缘检测
 * 计算水平和垂直方向的梯度，得到边缘强度
 */
GrayImage *sobel_edge(GrayImage *img) {
    GrayImage *result = gray_create(img->width, img->height);

    int gx_kernel[] = {-1, 0, 1, -2, 0, 2, -1, 0, 1};
    int gy_kernel[] = {-1, -2, -1, 0, 0, 0, 1, 2, 1};

    for (int y = 1; y < img->height - 1; y++) {
        for (int x = 1; x < img->width - 1; x++) {
            int gx = 0, gy = 0;
            for (int ky = -1; ky <= 1; ky++) {
                for (int kx = -1; kx <= 1; kx++) {
                    int idx = (ky + 1) * 3 + (kx + 1);
                    unsigned char val = gray_get(img, x + kx, y + ky);
                    gx += gx_kernel[idx] * val;
                    gy += gy_kernel[idx] * val;
                }
            }
            int mag = (int)sqrt(gx * gx + gy * gy);
            if (mag > 255) mag = 255;
            gray_set(result, x, y, (unsigned char)mag);
        }
    }

    return result;
}
```

/* Roberts交叉算子 */
GrayImage *roberts_edge(GrayImage *img) {
    GrayImage *result = gray_create(img->width, img->height);

    for (int y = 0; y < img->height - 1; y++) {
        for (int x = 0; x < img->width - 1; x++) {
            int g1 = gray_get(img, x, y) - gray_get(img, x + 1, y + 1);
            int g2 = gray_get(img, x + 1, y) - gray_get(img, x, y + 1);
            int mag = (int)sqrt(g1 * g1 + g2 * g2);
            if (mag > 255) mag = 255;
            gray_set(result, x, y, (unsigned char)mag);
        }
    }

    return result;
}

## 6. 直方图均衡化

```c
/* 直方图均衡化 — 增强对比度 */
GrayImage *histogram_equalization(GrayImage *img) {
    int histogram[256] = {0};
    int total = img->width * img->height;

    for (int i = 0; i < total; i++) {
        histogram[img->data[i]]++;
    }

    // 计算累积分布
    int cdf[256];
    cdf[0] = histogram[0];
    for (int i = 1; i < 256; i++) {
        cdf[i] = cdf[i - 1] + histogram[i];
    }

    // 找到cdf的最小非零值
    int cdf_min = 0;
    for (int i = 0; i < 256; i++) {
        if (cdf[i] > 0) { cdf_min = cdf[i]; break; }
    }

    // 映射
    unsigned char mapping[256];
    for (int i = 0; i < 256; i++) {
        mapping[i] = (unsigned char)(
            round((double)(cdf[i] - cdf_min) / (total - cdf_min) * 255));
    }

    GrayImage *result = gray_create(img->width, img->height);
    for (int i = 0; i < total; i++) {
        result->data[i] = mapping[img->data[i]];
    }

    return result;
}
```

## 7. 图像缩放（双线性插值）

```c
/* 双线性插值缩放 */
GrayImage *resize_bilinear(GrayImage *img, int new_w, int new_h) {
    GrayImage *result = gray_create(new_w, new_h);
    double sx = (double)img->width / new_w;
    double sy = (double)img->height / new_h;

    for (int y = 0; y < new_h; y++) {
        for (int x = 0; x < new_w; x++) {
            double src_x = x * sx;
            double src_y = y * sy;
            int x0 = (int)src_x, y0 = (int)src_y;
            int x1 = x0 + 1 < img->width ? x0 + 1 : x0;
            int y1 = y0 + 1 < img->height ? y0 + 1 : y0;

            double fx = src_x - x0, fy = src_y - y0;
            double val = (1 - fx) * (1 - fy) * gray_get(img, x0, y0) +
                         fx * (1 - fy) * gray_get(img, x1, y0) +
                         (1 - fx) * fy * gray_get(img, x0, y1) +
                         fx * fy * gray_get(img, x1, y1);

            gray_set(result, x, y, (unsigned char)(val + 0.5));
        }
    }

    return result;
}
```

## 小结

- 灰度化使用加权公式 `0.299R + 0.587G + 0.114B` 效果最好
- Otsu算法能自动找到最佳二值化阈值，无需人工设定
- 卷积是图像滤波的基础操作，不同核可实现不同效果
- Sobel算子同时提供边缘强度和方向信息
- 直方图均衡化能有效增强图像对比度
