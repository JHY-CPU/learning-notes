# 数学运算基础

## 概述

C语言提供了丰富的数学运算能力，从基本的整数运算到复杂的浮点运算。本节详细讲解整数运算、溢出处理和精度控制等核心内容。

---

## 1. 整数运算基础

### 1.1 基本算术运算

C语言支持加(`+`)、减(`-`)、乘(`*`)、除(`/`)、取模(`%`)五种基本运算。

```c
#include <stdio.h>

int main() {
    int a = 17, b = 5;

    printf("a + b = %d\n", a + b);   // 22
    printf("a - b = %d\n", a - b);   // 12
    printf("a * b = %d\n", a * b);   // 85
    printf("a / b = %d\n", a / b);   // 3  (整数除法截断)
    printf("a %% b = %d\n", a % b);  // 2  (取余数)

    return 0;
}
```

### 1.2 整数类型范围

不同整数类型的取值范围不同，了解这些范围对于防止溢出至关重要。

```c
#include <stdio.h>
#include <limits.h>

int main() {
    printf("char:      [%d, %d]\n", CHAR_MIN, CHAR_MAX);
    printf("short:     [%d, %d]\n", SHRT_MIN, SHRT_MAX);
    printf("int:       [%d, %d]\n", INT_MIN, INT_MAX);
    printf("long:      [%ld, %ld]\n", LONG_MIN, LONG_MAX);
    printf("long long: [%lld, %lld]\n", LLONG_MIN, LLONG_MAX);
    printf("unsigned int: [0, %u]\n", UINT_MAX);

    return 0;
}
```

### 1.3 类型转换

整数运算中的隐式类型转换需要特别注意。

```c
#include <stdio.h>

int main() {
    int a = -1;
    unsigned int b = 1;

    // 隐式转换：int 被转换为 unsigned int
    // -1 的补码表示为 0xFFFFFFFF，在无符号解释下非常大
    if (a < b) {
        printf("a < b\n");
    } else {
        printf("a >= b (unexpected!)\n"); // 实际输出这个
    }

    // 正确做法：显式类型转换
    if ((int)a < (int)b) {
        printf("a < b (correct)\n");
    }

    return 0;
}
```

## 2. 溢出处理

### 2.1 有符号整数溢出

有符号整数溢出在C语言中是**未定义行为(UB)**，必须主动检测。

```c
#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

/* 安全的有符号加法 */
bool safe_add(int a, int b, int *result) {
    if (b > 0 && a > INT_MAX - b) {
        return false;  // 正溢出
    }
    if (b < 0 && a < INT_MIN - b) {
        return false;  // 负溢出
    }
    *result = a + b;
    return true;
}

/* 安全的有符号乘法 */
bool safe_mul(int a, int b, int *result) {
    if (a == 0 || b == 0) {
        *result = 0;
        return true;
    }
    if (a > 0) {
        if (b > 0 && a > INT_MAX / b) return false;
        if (b < 0 && b < INT_MIN / a) return false;
    } else {
        if (b > 0 && a < INT_MIN / b) return false;
        if (b < 0 && a < INT_MAX / b) return false;
    }
    *result = a * b;
    return true;
}

int main() {
    int result;

    if (safe_add(INT_MAX, 1, &result)) {
        printf("结果: %d\n", result);
    } else {
        printf("加法溢出!\n");
    }

    if (safe_mul(100000, 100000, &result)) {
        printf("结果: %d\n", result);
    } else {
        printf("乘法溢出!\n");
    }

    return 0;
}
```

### 2.2 无符号整数溢出

无符号整数溢出定义明确：结果对 `UINT_MAX + 1` 取模。

```c
#include <stdio.h>

int main() {
    unsigned int x = UINT_MAX;
    unsigned int y = x + 1;  // 定义良好，结果为 0

    printf("UINT_MAX + 1 = %u\n", y);  // 输出 0

    unsigned int a = 0;
    unsigned int b = a - 1;  // 定义良好，结果为 UINT_MAX
    printf("0 - 1 = %u\n", b);  // 输出 4294967295

    return 0;
}
```

### 2.3 使用 GCC 内建溢出检测

GCC/Clang 提供了 `__builtin_add_overflow` 等内建函数。

```c
#include <stdio.h>

int main() {
    int a = 2000000000, b = 2000000000, result;

    if (__builtin_add_overflow(a, b, &result)) {
        printf("检测到溢出!\n");
    } else {
        printf("结果: %d\n", result);
    }

    // 乘法溢出检测
    long long x = 1000000000LL, y = 1000000000LL, res;
    if (__builtin_mul_overflow(x, y, &res)) {
        printf("乘法溢出!\n");
    } else {
        printf("结果: %lld\n", res);
    }

    return 0;
}
```

## 3. 精度控制

### 3.1 浮点数舍入控制

```c
#include <stdio.h>
#include <math.h>

int main() {
    double x = 3.141592653589793238;

    // 控制输出精度
    printf("默认:    %f\n", x);      // 3.141593
    printf("2位小数: %.2f\n", x);    // 3.14
    printf("10位小数: %.10f\n", x);  // 3.1415926536

    // 舍入函数
    printf("ceil(2.3)  = %.0f\n", ceil(2.3));    // 3
    printf("floor(2.7) = %.0f\n", floor(2.7));   // 2
    printf("round(2.5) = %.0f\n", round(2.5));   // 3
    printf("trunc(2.9) = %.0f\n", trunc(2.9));   // 2

    return 0;
}
```

### 3.2 定点数运算（模拟精度）

在需要精确计算的场景中，可以用整数模拟定点数。

```c
#include <stdio.h>

// 使用整数模拟3位小数精度
// 所有数值乘以1000存储
#define SCALE 1000

typedef struct {
    long long value;  // 实际值 = value / SCALE
} FixedPoint;

FixedPoint fp_from_int(int n) {
    return (FixedPoint){n * SCALE};
}

FixedPoint fp_from_double(double d) {
    return (FixedPoint){(long long)(d * SCALE + 0.5)};
}

double fp_to_double(FixedPoint fp) {
    return (double)fp.value / SCALE;
}

FixedPoint fp_add(FixedPoint a, FixedPoint b) {
    return (FixedPoint){a.value + b.value};
}

FixedPoint fp_sub(FixedPoint a, FixedPoint b) {
    return (FixedPoint){a.value - b.value};
}

FixedPoint fp_mul(FixedPoint a, FixedPoint b) {
    return (FixedPoint){(a.value * b.value) / SCALE};
}

FixedPoint fp_div(FixedPoint a, FixedPoint b) {
    return (FixedPoint){(a.value * SCALE) / b.value};
}

int main() {
    FixedPoint a = fp_from_double(1.234);
    FixedPoint b = fp_from_double(5.678);

    FixedPoint sum = fp_add(a, b);
    FixedPoint prod = fp_mul(a, b);

    printf("1.234 + 5.678 = %.3f\n", fp_to_double(sum));  // 6.912
    printf("1.234 * 5.678 = %.3f\n", fp_to_double(prod));  // 7.007

    // 演示定点数的精度优势
    double d1 = 0.1 + 0.2;
    FixedPoint f1 = fp_add(fp_from_double(0.1), fp_from_double(0.2));
    printf("浮点: 0.1 + 0.2 = %.17f\n", d1);   // 精度误差
    printf("定点: 0.1 + 0.2 = %.3f\n", fp_to_double(f1));  // 精确

    return 0;
}
```

## 4. 位运算技巧

### 4.1 常用位运算操作

```c
#include <stdio.h>

// 判断是否为2的幂
int is_power_of_2(unsigned int n) {
    return n != 0 && (n & (n - 1)) == 0;
}

// 计算最低位的1（lowbit）
int lowbit(int x) {
    return x & (-x);
}

// 统计二进制中1的个数
int popcount(unsigned int n) {
    int count = 0;
    while (n) {
        n &= (n - 1);  // 清除最低位的1
        count++;
    }
    return count;
}

// 不使用临时变量交换两数
void swap(int *a, int *b) {
    *a ^= *b;
    *b ^= *a;
    *a ^= *b;
}

int main() {
    printf("8是2的幂: %s\n", is_power_of_2(8) ? "是" : "否");
    printf("6是2的幂: %s\n", is_power_of_2(6) ? "是" : "否");
    printf("lowbit(12) = %d\n", lowbit(12));  // 4
    printf("popcount(7) = %d\n", popcount(7)); // 3

    int a = 5, b = 10;
    swap(&a, &b);
    printf("交换后: a=%d, b=%d\n", a, b);

    return 0;
}
```

## 小结

- 整数运算要注意**有符号溢出是未定义行为**，必须提前检测
- 使用 `limits.h` 中的宏了解类型的取值范围
- 浮点运算存在精度误差，对精度要求高的场景考虑定点数方案
- 位运算在性能优化中非常有用，掌握常用技巧可以写出更高效的代码
