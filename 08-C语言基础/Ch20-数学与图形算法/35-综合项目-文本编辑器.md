# 综合项目 - 终端文本编辑器

## 概述

本节实现一个简易的终端文本编辑器，综合运用C语言的文件I/O、字符串处理、终端控制等知识。编辑器支持基本的文本编辑、光标移动、文件读写等功能。

---

## 1. 编辑器架构

```
+-------------------+
|     主循环        |  - 读取按键 -> 执行命令 -> 刷新显示
+-------------------+
|   文档模型        |  - 行数组存储文本内容
+-------------------+
|   终端控制        |  - 光标定位、清屏、原始模式
+-------------------+
|   状态栏/提示行   |  - 显示文件名、位置、状态消息
+-------------------+
```

## 2. 数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
#include <stdarg.h>

#ifdef _WIN32
#include <windows.h>
#include <conio.h>
#else
#include <termios.h>
#include <unistd.h>
#include <sys/ioctl.h>
#endif

#define EDITOR_VERSION "0.1"
#define MAX_LINE_LEN 1024
#define TAB_SIZE 4
#define WELCOME_LINES 8

/* 文本行 */
typedef struct {
    char *text;         // 行内容
    int length;         // 当前长度
    int capacity;       // 分配容量
} Line;

/* 编辑器状态 */
typedef struct {
    Line *lines;            // 行数组
    int num_lines;          // 总行数
    int cursor_x;           // 光标列位置
    int cursor_y;           // 光标行位置
    int offset_x;           // 水平滚动偏移
    int offset_y;           // 垂直滚动偏移
    int screen_rows;        // 终端行数
    int screen_cols;        // 终端列数
    int dirty;              // 文件是否已修改
    char filename[256];     // 当前文件名
    char status_msg[256];   // 状态消息
    int running;            // 运行标志
} Editor;

static Editor editor;
```

## 3. 终端控制

### 3.1 原始模式

```c
#ifndef _WIN32
static struct termios orig_termios;

/* 禁用原始模式，恢复终端 */
void disable_raw_mode() {
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);
}

/* 启用原始模式 */
void enable_raw_mode() {
    tcgetattr(STDIN_FILENO, &orig_termios);
    atexit(disable_raw_mode);

    struct termios raw = orig_termios;
    raw.c_lflag &= ~(ECHO | ICANON | ISIG | IEXTEN);
    raw.c_iflag &= ~(IXON | ICRNL | BRKINT | INPCK | ISTRIP);
    raw.c_oflag &= ~(OPOST);
    raw.c_cflag |= CS8;
    raw.c_cc[VMIN] = 0;
    raw.c_cc[VTIME] = 1;  // 100ms超时

    tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
}
#endif
```

### 3.2 获取终端大小

```c
void get_terminal_size(int *rows, int *cols) {
#ifdef _WIN32
    CONSOLE_SCREEN_BUFFER_INFO csbi;
    GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
    *cols = csbi.srWindow.Right - csbi.srWindow.Left + 1;
    *rows = csbi.srWindow.Bottom - csbi.srWindow.Top + 1;
#else
    struct winsize ws;
    if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == -1) {
        *rows = 24;
        *cols = 80;
    } else {
        *rows = ws.ws_row;
        *cols = ws.ws_col;
    }
#endif
    editor.screen_rows = *rows;
    editor.screen_cols = *cols;
}
```

### 3.3 读取按键

```c
/* 按键定义 */
#define KEY_ARROW_UP    1000
#define KEY_ARROW_DOWN  1001
#define KEY_ARROW_LEFT  1002
#define KEY_ARROW_RIGHT 1003
#define KEY_PAGE_UP     1004
#define KEY_PAGE_DOWN   1005
#define KEY_HOME        1006
#define KEY_END         1007
#define KEY_DELETE      1008
#define KEY_BACKSPACE   127
#define KEY_ENTER       13
#define KEY_ESCAPE      27
#define KEY_CTRL_S      19
#define KEY_CTRL_Q      17
#define KEY_CTRL_F      6

/* 读取一个按键 */
int read_key() {
    int nread;
    char c;

    while ((nread = read(STDIN_FILENO, &c, 1)) != 1) {
        if (nread == -1) return -1;
    }

    if (c == KEY_ESCAPE) {
        char seq[3];
        if (read(STDIN_FILENO, &seq[0], 1) != 1) return KEY_ESCAPE;
        if (read(STDIN_FILENO, &seq[1], 1) != 1) return KEY_ESCAPE;

        if (seq[0] == '[') {
            if (seq[1] >= '0' && seq[1] <= '9') {
                if (read(STDIN_FILENO, &seq[2], 1) != 1) return KEY_ESCAPE;
                if (seq[2] == '~') {
                    switch (seq[1]) {
                        case '1': return KEY_HOME;
                        case '3': return KEY_DELETE;
                        case '4': return KEY_END;
                        case '5': return KEY_PAGE_UP;
                        case '6': return KEY_PAGE_DOWN;
                        case '7': return KEY_HOME;
                        case '8': return KEY_END;
                    }
                }
            } else {
                switch (seq[1]) {
                    case 'A': return KEY_ARROW_UP;
                    case 'B': return KEY_ARROW_DOWN;
                    case 'C': return KEY_ARROW_RIGHT;
                    case 'D': return KEY_ARROW_LEFT;
                    case 'H': return KEY_HOME;
                    case 'F': return KEY_END;
                }
            }
        }
        return KEY_ESCAPE;
    }

    return c;
}
```

## 4. 文档操作

### 4.1 行操作

```c
/* 初始化一行 */
Line line_create() {
    Line line;
    line.capacity = 64;
    line.text = (char *)malloc(line.capacity);
    line.text[0] = '\0';
    line.length = 0;
    return line;
}

/* 确保行有足够容量 */
void line_ensure_capacity(Line *line, int needed) {
    if (line->capacity >= needed) return;
    while (line->capacity < needed) line->capacity *= 2;
    line->text = (char *)realloc(line->text, line->capacity);
}

/* 在行中插入字符 */
void line_insert_char(Line *line, int pos, char c) {
    if (pos < 0 || pos > line->length) return;
    line_ensure_capacity(line, line->length + 2);

    memmove(line->text + pos + 1, line->text + pos, line->length - pos + 1);
    line->text[pos] = c;
    line->length++;
}

/* 从行中删除字符 */
void line_delete_char(Line *line, int pos) {
    if (pos < 0 || pos >= line->length) return;
    memmove(line->text + pos, line->text + pos + 1, line->length - pos);
    line->length--;
}

/* 在行尾追加字符串 */
void line_append(Line *line, const char *str, int len) {
    line_ensure_capacity(line, line->length + len + 1);
    memcpy(line->text + line->length, str, len);
    line->length += len;
    line->text[line->length] = '\0';
}
```

### 4.2 文档操作

```c
/* 初始化编辑器 */
void editor_init() {
    memset(&editor, 0, sizeof(editor));
    editor.num_lines = 1;
    editor.lines = (Line *)malloc(sizeof(Line));
    editor.lines[0] = line_create();
    strcpy(editor.filename, "[未命名]");
}

/* 在指定行后插入新行 */
void editor_insert_line(int after) {
    editor.lines = (Line *)realloc(editor.lines,
        (editor.num_lines + 1) * sizeof(Line));
    memmove(editor.lines + after + 2, editor.lines + after + 1,
            (editor.num_lines - after - 1) * sizeof(Line));
    editor.lines[after + 1] = line_create();
    editor.num_lines++;
}

/* 删除指定行 */
void editor_delete_line(int line_num) {
    if (line_num < 0 || line_num >= editor.num_lines) return;
    if (editor.num_lines <= 1) return;

    free(editor.lines[line_num].text);
    memmove(editor.lines + line_num, editor.lines + line_num + 1,
            (editor.num_lines - line_num - 1) * sizeof(Line));
    editor.num_lines--;
}

/* 插入字符 */
void editor_insert_char(int c) {
    Line *line = &editor.lines[editor.cursor_y];
    line_insert_char(line, editor.cursor_x, c);
    editor.cursor_x++;
    editor.dirty = 1;
}

/* 插入新行（回车键） */
void editor_insert_newline() {
    Line *current = &editor.lines[editor.cursor_y];

    // 将当前行光标后的部分移到新行
    Line new_line = line_create();
    if (editor.cursor_x < current->length) {
        line_append(&new_line, current->text + editor.cursor_x,
                    current->length - editor.cursor_x);
        current->length = editor.cursor_x;
        current->text[current->length] = '\0';
    }

    editor.lines = (Line *)realloc(editor.lines,
        (editor.num_lines + 1) * sizeof(Line));
    memmove(editor.lines + editor.cursor_y + 2,
            editor.lines + editor.cursor_y + 1,
            (editor.num_lines - editor.cursor_y - 1) * sizeof(Line));
    editor.lines[editor.cursor_y + 1] = new_line;
    editor.num_lines++;

    editor.cursor_y++;
    editor.cursor_x = 0;
    editor.dirty = 1;
}

/* 删除字符（退格键） */
void editor_delete_backspace() {
    if (editor.cursor_x == 0 && editor.cursor_y == 0) return;

    if (editor.cursor_x > 0) {
        Line *line = &editor.lines[editor.cursor_y];
        line_delete_char(line, editor.cursor_x - 1);
        editor.cursor_x--;
    } else {
        // 将当前行拼接到上一行
        Line *prev = &editor.lines[editor.cursor_y - 1];
        Line *curr = &editor.lines[editor.cursor_y];
        editor.cursor_x = prev->length;
        line_append(prev, curr->text, curr->length);
        editor_delete_line(editor.cursor_y);
        editor.cursor_y--;
    }
    editor.dirty = 1;
}
```

## 5. 文件读写

```c
/* 打开文件 */
void editor_open(const char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) {
        editor_set_status("无法打开文件: %s", filename);
        return;
    }

    // 清空现有内容
    for (int i = 0; i < editor.num_lines; i++) {
        free(editor.lines[i].text);
    }
    free(editor.lines);

    // 逐行读取
    editor.num_lines = 0;
    editor.lines = NULL;

    char buf[MAX_LINE_LEN];
    while (fgets(buf, sizeof(buf), fp)) {
        editor.lines = (Line *)realloc(editor.lines,
            (editor.num_lines + 1) * sizeof(Line));

        int len = (int)strlen(buf);
        while (len > 0 && (buf[len-1] == '\n' || buf[len-1] == '\r'))
            buf[--len] = '\0';

        editor.lines[editor.num_lines] = line_create();
        line_append(&editor.lines[editor.num_lines], buf, len);
        editor.num_lines++;
    }

    if (editor.num_lines == 0) {
        editor.lines = (Line *)malloc(sizeof(Line));
        editor.lines[0] = line_create();
        editor.num_lines = 1;
    }

    fclose(fp);
    strncpy(editor.filename, filename, sizeof(editor.filename) - 1);
    editor.dirty = 0;
    editor.cursor_x = 0;
    editor.cursor_y = 0;
    editor_set_status("已打开: %s (%d 行)", filename, editor.num_lines);
}

/* 保存文件 */
void editor_save() {
    if (editor.filename[0] == '[') {
        editor_set_status("请先设置文件名");
        return;
    }

    FILE *fp = fopen(editor.filename, "w");
    if (!fp) {
        editor_set_status("无法保存文件: %s", editor.filename);
        return;
    }

    for (int i = 0; i < editor.num_lines; i++) {
        fprintf(fp, "%s\n", editor.lines[i].text);
    }

    fclose(fp);
    editor.dirty = 0;
    editor_set_status("已保存: %s (%d 行)", editor.filename, editor.num_lines);
}
```

## 6. 渲染

```c
/* 设置状态消息 */
void editor_set_status(const char *fmt, ...) {
    va_list args;
    va_start(args, fmt);
    vsnprintf(editor.status_msg, sizeof(editor.status_msg), fmt, args);
    va_end(args);
}

/* 绘制界面 */
void editor_draw() {
    // 清屏
    write(STDOUT_FILENO, "\033[2J", 4);
    write(STDOUT_FILENO, "\033[H", 3);

    int visible_rows = editor.screen_rows - 2;  // 留出状态栏和提示行

    for (int y = 0; y < visible_rows; y++) {
        int file_line = y + editor.offset_y;

        if (file_line >= editor.num_lines) {
            // 显示欢迎信息
            if (editor.num_lines == 1 && editor.lines[0].length == 0) {
                if (y == visible_rows / 3) {
                    char welcome[80];
                    int len = snprintf(welcome, sizeof(welcome),
                        "简易文本编辑器 v%s", EDITOR_VERSION);
                    int padding = (editor.screen_cols - len) / 2;
                    if (padding > 0) printf("~%*s", padding - 1, "");
                    printf("%s", welcome);
                } else if (y == visible_rows / 3 + 1) {
                    printf("~  按 Ctrl+Q 退出 | Ctrl+S 保存 | Ctrl+F 搜索");
                } else {
                    printf("~");
                }
            } else {
                printf("~");
            }
        } else {
            Line *line = &editor.lines[file_line];
            int start = editor.offset_x;
            int end = start + editor.screen_cols - 1;
            if (end > line->length) end = line->length;

            for (int x = start; x < end; x++) {
                char c = line->text[x];
                if (c == '\t') {
                    printf("    ");
                } else {
                    putchar(c);
                }
            }
        }

        printf("\033[K");  // 清除行尾
        printf("\r\n");
    }

    // 状态栏
    printf("\033[7m");  // 反色
    char status[256];
    int len = snprintf(status, sizeof(status),
        " %.20s %s | 行 %d/%d 列 %d",
        editor.filename,
        editor.dirty ? "(已修改)" : "",
        editor.cursor_y + 1, editor.num_lines,
        editor.cursor_x + 1);
    printf("%s", status);
    for (int i = len; i < editor.screen_cols; i++) putchar(' ');
    printf("\033[0m\r\n");

    // 提示行
    printf("%s\033[K", editor.status_msg);

    // 定位光标
    int screen_y = editor.cursor_y - editor.offset_y + 1;
    int screen_x = editor.cursor_x - editor.offset_x + 1;
    printf("\033[%d;%dH", screen_y, screen_x);
    fflush(stdout);
}
```

## 7. 光标移动与主循环

```c
/* 处理光标移动 */
void editor_move_cursor(int key) {
    Line *line = &editor.lines[editor.cursor_y];

    switch (key) {
        case KEY_ARROW_LEFT:
            if (editor.cursor_x > 0) editor.cursor_x--;
            else if (editor.cursor_y > 0) {
                editor.cursor_y--;
                editor.cursor_x = editor.lines[editor.cursor_y].length;
            }
            break;
        case KEY_ARROW_RIGHT:
            if (editor.cursor_x < line->length) editor.cursor_x++;
            else if (editor.cursor_y < editor.num_lines - 1) {
                editor.cursor_y++;
                editor.cursor_x = 0;
            }
            break;
        case KEY_ARROW_UP:
            if (editor.cursor_y > 0) editor.cursor_y--;
            break;
        case KEY_ARROW_DOWN:
            if (editor.cursor_y < editor.num_lines - 1) editor.cursor_y++;
            break;
    }

    // 调整光标位置（确保不超出当前行）
    line = &editor.lines[editor.cursor_y];
    if (editor.cursor_x > line->length) editor.cursor_x = line->length;

    // 调整滚动偏移
    if (editor.cursor_y < editor.offset_y)
        editor.offset_y = editor.cursor_y;
    if (editor.cursor_y >= editor.offset_y + editor.screen_rows - 2)
        editor.offset_y = editor.cursor_y - editor.screen_rows + 3;
    if (editor.cursor_x < editor.offset_x)
        editor.offset_x = editor.cursor_x;
    if (editor.cursor_x >= editor.offset_x + editor.screen_cols)
        editor.offset_x = editor.cursor_x - editor.screen_cols + 1;
}

/* 处理按键 */
void editor_process_key() {
    int key = read_key();
    if (key == -1) return;

    switch (key) {
        case KEY_CTRL_Q:
            if (editor.dirty) {
                editor_set_status("文件已修改，再按一次Ctrl+Q强制退出");
                editor.dirty = 0;  // hack: 第二次按就能退出
            } else {
                editor.running = 0;
            }
            break;

        case KEY_CTRL_S:
            editor_save();
            break;

        case KEY_ARROW_UP:
        case KEY_ARROW_DOWN:
        case KEY_ARROW_LEFT:
        case KEY_ARROW_RIGHT:
            editor_move_cursor(key);
            break;

        case KEY_HOME:
            editor.cursor_x = 0;
            break;

        case KEY_END:
            editor.cursor_x = editor.lines[editor.cursor_y].length;
            break;

        case KEY_ENTER:
            editor_insert_newline();
            break;

        case KEY_BACKSPACE:
            editor_delete_backspace();
            break;

        case KEY_DELETE:
            if (editor.cursor_x < editor.lines[editor.cursor_y].length) {
                editor.cursor_x++;
                editor_delete_backspace();
            }
            break;

        default:
            if (key >= 32 && key < 127) {
                editor_insert_char(key);
            }
            break;
    }
}

/* 主函数 */
int main(int argc, char *argv[]) {
    enable_raw_mode();
    get_terminal_size(&editor.screen_rows, &editor.screen_cols);
    editor_init();

    if (argc >= 2) {
        editor_open(argv[1]);
    }

    editor.running = 1;
    editor_set_status("Ctrl+Q: 退出 | Ctrl+S: 保存");

    while (editor.running) {
        editor_draw();
        editor_process_key();
    }

    // 清屏退出
    write(STDOUT_FILENO, "\033[2J", 4);
    write(STDOUT_FILENO, "\033[H", 3);

    printf("编辑器已退出\n");

    return 0;
}
```

## 小结

- 终端编辑器需要将终端设置为原始模式，直接处理每个按键
- 行数组是最简单的文档模型，适合小型编辑器
- 渲染时需要处理滚动偏移，确保光标始终可见
- 文件操作使用标准C的fopen/fprintf接口
- 可以扩展的功能：语法高亮、搜索替换、多标签、Undo/Redo
