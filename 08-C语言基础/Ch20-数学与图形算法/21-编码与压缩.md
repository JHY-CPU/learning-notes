# 编码与压缩

## 概述

数据编码和压缩是计算机科学的基础技术。本节实现Base64编码、哈夫曼压缩和LZ77压缩算法。

## 1. Base64编码

### 1.1 编码原理

Base64将每3个字节（24位）编码为4个可打印ASCII字符。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>

/* Base64编码表 */
static const char BASE64_TABLE[] =
    "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";

/*
 * Base64编码
 * input: 二进制数据
 * in_len: 输入长度
 * output: 编码后的字符串（需自行释放）
 */
char *base64_encode(const uint8_t *input, size_t in_len) {
    size_t out_len = ((in_len + 2) / 3) * 4 + 1;
    char *output = (char *)malloc(out_len);
    size_t i, j;

    for (i = 0, j = 0; i < in_len; i += 3) {
        uint32_t n = (uint32_t)input[i] << 16;
        if (i + 1 < in_len) n |= (uint32_t)input[i + 1] << 8;
        if (i + 2 < in_len) n |= input[i + 2];

        output[j++] = BASE64_TABLE[(n >> 18) & 0x3F];
        output[j++] = BASE64_TABLE[(n >> 12) & 0x3F];
        output[j++] = (i + 1 < in_len) ? BASE64_TABLE[(n >> 6) & 0x3F] : '=';
        output[j++] = (i + 2 < in_len) ? BASE64_TABLE[n & 0x3F] : '=';
    }

    output[j] = '\0';
    return output;
}

/* Base64解码表 */
static int base64_char_val(char c) {
    if (c >= 'A' && c <= 'Z') return c - 'A';
    if (c >= 'a' && c <= 'z') return c - 'a' + 26;
    if (c >= '0' && c <= '9') return c - '0' + 52;
    if (c == '+') return 62;
    if (c == '/') return 63;
    return -1;
}

/*
 * Base64解码
 * 返回解码后的二进制数据和长度
 */
uint8_t *base64_decode(const char *input, size_t *out_len) {
    size_t in_len = strlen(input);
    if (in_len % 4 != 0) return NULL;

    // 计算输出长度
    *out_len = in_len / 4 * 3;
    if (input[in_len - 1] == '=') (*out_len)--;
    if (input[in_len - 2] == '=') (*out_len)--;

    uint8_t *output = (uint8_t *)malloc(*out_len);
    size_t i, j;

    for (i = 0, j = 0; i < in_len; i += 4) {
        uint32_t n = ((uint32_t)base64_char_val(input[i]) << 18) |
                     ((uint32_t)base64_char_val(input[i+1]) << 12);

        if (input[i+2] != '=') n |= ((uint32_t)base64_char_val(input[i+2]) << 6);
        if (input[i+3] != '=') n |= base64_char_val(input[i+3]);

        if (j < *out_len) output[j++] = (n >> 16) & 0xFF;
        if (j < *out_len) output[j++] = (n >> 8) & 0xFF;
        if (j < *out_len) output[j++] = n & 0xFF;
    }

    return output;
}
```

## 2. 哈夫曼编码

### 2.1 数据结构

```c
/* 哈夫曼树节点 */
typedef struct HuffmanNode {
    uint8_t byte;
    int freq;
    struct HuffmanNode *left, *right;
} HuffmanNode;

/* 编码表 */
typedef struct {
    uint32_t code;   // 编码值
    int length;      // 编码长度（位）
} HuffmanCode;

/* 创建节点 */
HuffmanNode *huffman_create_node(uint8_t byte, int freq) {
    HuffmanNode *node = (HuffmanNode *)malloc(sizeof(HuffmanNode));
    node->byte = byte;
    node->freq = freq;
    node->left = node->right = NULL;
    return node;
}
```

### 2.2 最小堆

```c
/* 简易最小堆 */
#define MAX_HEAP 512

typedef struct {
    HuffmanNode *nodes[MAX_HEAP];
    int size;
} MinHeap;

void heap_init(MinHeap *h) { h->size = 0; }

void heap_swap(HuffmanNode **a, HuffmanNode **b) {
    HuffmanNode *t = *a; *a = *b; *b = t;
}

void heap_push(MinHeap *h, HuffmanNode *node) {
    int i = h->size++;
    h->nodes[i] = node;
    while (i > 0) {
        int parent = (i - 1) / 2;
        if (h->nodes[parent]->freq <= h->nodes[i]->freq) break;
        heap_swap(&h->nodes[parent], &h->nodes[i]);
        i = parent;
    }
}

HuffmanNode *heap_pop(MinHeap *h) {
    HuffmanNode *result = h->nodes[0];
    h->nodes[0] = h->nodes[--h->size];
    int i = 0;
    while (1) {
        int smallest = i;
        int left = 2 * i + 1, right = 2 * i + 2;
        if (left < h->size && h->nodes[left]->freq < h->nodes[smallest]->freq)
            smallest = left;
        if (right < h->size && h->nodes[right]->freq < h->nodes[smallest]->freq)
            smallest = right;
        if (smallest == i) break;
        heap_swap(&h->nodes[i], &h->nodes[smallest]);
        i = smallest;
    }
    return result;
}
```

### 2.3 构建哈夫曼树

```c
/* 构建哈夫曼树 */
HuffmanNode *build_huffman_tree(const uint8_t *data, size_t len) {
    int freq[256] = {0};
    for (size_t i = 0; i < len; i++) freq[data[i]]++;

    MinHeap heap;
    heap_init(&heap);

    for (int i = 0; i < 256; i++) {
        if (freq[i] > 0) {
            heap_push(&heap, huffman_create_node((uint8_t)i, freq[i]));
        }
    }

    // 特殊情况：只有一个不同字节
    if (heap.size == 1) {
        HuffmanNode *only = heap_pop(&heap);
        HuffmanNode *root = huffman_create_node(0, only->freq);
        root->left = only;
        return root;
    }

    while (heap.size > 1) {
        HuffmanNode *left = heap_pop(&heap);
        HuffmanNode *right = heap_pop(&heap);
        HuffmanNode *parent = huffman_create_node(0, left->freq + right->freq);
        parent->left = left;
        parent->right = right;
        heap_push(&heap, parent);
    }

    return heap_pop(&heap);
}
```

### 2.4 生成编码表

```c
/* 递归生成编码表 */
void generate_codes(HuffmanNode *root, HuffmanCode *table,
                    uint32_t code, int depth) {
    if (!root) return;

    if (!root->left && !root->right) {
        table[root->byte].code = code;
        table[root->byte].length = depth;
        return;
    }

    generate_codes(root->left, table, code << 1, depth + 1);
    generate_codes(root->right, table, (code << 1) | 1, depth + 1);
}
```

### 2.5 哈夫曼压缩与解压

```c
/*
 * 哈夫曼压缩
 * 返回压缩后的数据和长度
 */
uint8_t *huffman_compress(const uint8_t *input, size_t in_len,
                          HuffmanCode *table, size_t *out_len) {
    // 计算总位数
    size_t total_bits = 0;
    for (size_t i = 0; i < in_len; i++) {
        total_bits += table[input[i]].length;
    }

    *out_len = (total_bits + 7) / 8;
    uint8_t *output = (uint8_t *)calloc(*out_len, 1);

    size_t byte_pos = 0;
    int bit_pos = 0;

    for (size_t i = 0; i < in_len; i++) {
        uint32_t code = table[input[i]].code;
        int code_len = table[input[i]].length;

        for (int b = code_len - 1; b >= 0; b--) {
            uint8_t bit = (code >> b) & 1;
            output[byte_pos] |= (bit << (7 - bit_pos));
            bit_pos++;
            if (bit_pos == 8) {
                bit_pos = 0;
                byte_pos++;
            }
        }
    }

    return output;
}

/*
 * 哈夫曼解压
 */
uint8_t *huffman_decompress(const uint8_t *input, size_t in_len,
                            HuffmanNode *root, size_t total_bits,
                            size_t *out_len) {
    size_t capacity = total_bits;
    uint8_t *output = (uint8_t *)malloc(capacity);
    *out_len = 0;

    HuffmanNode *current = root;
    size_t bits_processed = 0;

    for (size_t i = 0; i < in_len && bits_processed < total_bits; i++) {
        for (int b = 7; b >= 0 && bits_processed < total_bits; b--) {
            int bit = (input[i] >> b) & 1;
            current = bit ? current->right : current->left;
            bits_processed++;

            if (!current->left && !current->right) {
                output[(*out_len)++] = current->byte;
                current = root;
            }
        }
    }

    return output;
}
```

## 3. LZ77 压缩算法

### 3.1 算法原理

LZ77通过滑动窗口查找重复的字符串序列，用（偏移量, 长度）对替代重复内容。

```c
#define WINDOW_SIZE 4096
#define MAX_MATCH 18
#define MIN_MATCH 3

/* LZ77编码单元 */
typedef struct {
    int offset;     // 匹配的偏移量（0表示无匹配）
    int length;     // 匹配长度（0表示无匹配）
    uint8_t next;   // 匹配后的下一个字节
} LZ77Token;

/*
 * LZ77编码
 * tokens: 输出的token数组
 * 返回token数量
 */
int lz77_encode(const uint8_t *input, size_t len, LZ77Token *tokens) {
    int token_count = 0;
    size_t pos = 0;

    while (pos < len) {
        int best_offset = 0, best_length = 0;

        // 在滑动窗口中搜索最长匹配
        int win_start = (pos > WINDOW_SIZE) ? (int)(pos - WINDOW_SIZE) : 0;

        for (int i = win_start; i < (int)pos; i++) {
            int match_len = 0;
            while (pos + match_len < len &&
                   input[i + match_len] == input[pos + match_len] &&
                   match_len < MAX_MATCH) {
                match_len++;
            }
            if (match_len > best_length) {
                best_length = match_len;
                best_offset = (int)(pos - i);
            }
        }

        LZ77Token token;
        if (best_length >= MIN_MATCH) {
            token.offset = best_offset;
            token.length = best_length;
            token.next = (pos + best_length < len) ? input[pos + best_length] : 0;
            pos += best_length + 1;
        } else {
            token.offset = 0;
            token.length = 0;
            token.next = input[pos];
            pos++;
        }

        tokens[token_count++] = token;
    }

    return token_count;
}

/*
 * LZ77解码
 */
uint8_t *lz77_decode(const LZ77Token *tokens, int token_count,
                     size_t *out_len) {
    size_t capacity = 65536;
    uint8_t *output = (uint8_t *)malloc(capacity);
    size_t pos = 0;

    for (int i = 0; i < token_count; i++) {
        if (tokens[i].length > 0) {
            // 从窗口中复制
            size_t src = pos - tokens[i].offset;
            for (int j = 0; j < tokens[i].length; j++) {
                if (pos >= capacity) {
                    capacity *= 2;
                    output = (uint8_t *)realloc(output, capacity);
                }
                output[pos++] = output[src + j];
            }
        }

        if (pos >= capacity) {
            capacity *= 2;
            output = (uint8_t *)realloc(output, capacity);
        }
        output[pos++] = tokens[i].next;
    }

    *out_len = pos;
    return output;
}
```

## 4. 测试示例

```c
int main() {
    // Base64测试
    const char *msg = "Hello, World!";
    char *encoded = base64_encode((uint8_t *)msg, strlen(msg));
    printf("Base64编码: %s\n", encoded);

    size_t decoded_len;
    uint8_t *decoded = base64_decode(encoded, &decoded_len);
    printf("Base64解码: %.*s\n", (int)decoded_len, decoded);
    free(encoded);
    free(decoded);

    // 哈夫曼测试
    const char *text = "this is an example for huffman encoding";
    size_t text_len = strlen(text);

    HuffmanNode *tree = build_huffman_tree((uint8_t *)text, text_len);
    HuffmanCode table[256] = {0};
    generate_codes(tree, table, 0, 0);

    printf("\n哈夫曼编码表:\n");
    for (int i = 0; i < 256; i++) {
        if (table[i].length > 0) {
            printf("  '%c': ", i);
            for (int b = table[i].length - 1; b >= 0; b--)
                printf("%d", (table[i].code >> b) & 1);
            printf("\n");
        }
    }

    size_t compressed_len;
    uint8_t *compressed = huffman_compress((uint8_t *)text, text_len,
                                           table, &compressed_len);
    printf("\n原文大小: %zu 字节\n", text_len);
    printf("压缩后: %zu 字节\n", compressed_len);

    free(compressed);

    return 0;
}
```

## 小结

- Base64将二进制数据编码为ASCII文本，常用于邮件和数据传输
- 哈夫曼编码是一种无损压缩算法，为频率高的字符分配短编码
- LZ77是一种字典压缩算法，利用重复模式进行压缩
- 实际压缩工具（如gzip）通常组合多种算法：LZ77+哈夫曼
