# 线程池实现

## 概述

线程池预先创建一组工作线程，避免频繁创建/销毁线程的开销。本节实现固定大小和动态线程池。

---

## 1. 基础概念

线程池的核心组件：
- **工作线程**：预先创建，等待任务
- **任务队列**：存储待执行的任务
- **互斥锁**：保护任务队列的并发访问
- **条件变量**：通知线程有新任务

## 2. 固定大小线程池

### 2.1 数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <pthread.h>

/* 任务结构 */
typedef struct Task {
    void (*function)(void *);   // 任务函数
    void *argument;             // 任务参数
    struct Task *next;          // 下一个任务
} Task;

/* 任务队列 */
typedef struct {
    Task *head;
    Task *tail;
    int count;
    pthread_mutex_t lock;
    pthread_cond_t cond;
} TaskQueue;

/* 线程池 */
typedef struct {
    pthread_t *threads;         // 工作线程数组
    int thread_count;           // 线程数量
    TaskQueue queue;            // 任务队列
    int shutdown;               // 关闭标志
    pthread_mutex_t lock;       // 线程池锁
} ThreadPool;
```

### 2.2 任务队列操作

```c
/* 初始化任务队列 */
void queue_init(TaskQueue *queue) {
    queue->head = NULL;
    queue->tail = NULL;
    queue->count = 0;
    pthread_mutex_init(&queue->lock, NULL);
    pthread_cond_init(&queue->cond, NULL);
}

/* 入队 */
void queue_push(TaskQueue *queue, void (*func)(void *), void *arg) {
    Task *task = (Task *)malloc(sizeof(Task));
    task->function = func;
    task->argument = arg;
    task->next = NULL;

    pthread_mutex_lock(&queue->lock);

    if (queue->tail) {
        queue->tail->next = task;
    } else {
        queue->head = task;
    }
    queue->tail = task;
    queue->count++;

    pthread_cond_signal(&queue->cond);
    pthread_mutex_unlock(&queue->lock);
}

/* 出队（阻塞） */
Task *queue_pop(TaskQueue *queue) {
    pthread_mutex_lock(&queue->lock);

    while (queue->count == 0) {
        pthread_cond_wait(&queue->cond, &queue->lock);
    }

    Task *task = queue->head;
    if (task) {
        queue->head = task->next;
        if (!queue->head) queue->tail = NULL;
        queue->count--;
    }

    pthread_mutex_unlock(&queue->lock);
    return task;
}

/* 尝试出队（非阻塞） */
Task *queue_try_pop(TaskQueue *queue) {
    if (pthread_mutex_trylock(&queue->lock) != 0) return NULL;

    Task *task = NULL;
    if (queue->count > 0) {
        task = queue->head;
        queue->head = task->next;
        if (!queue->head) queue->tail = NULL;
        queue->count--;
    }

    pthread_mutex_unlock(&queue->lock);
    return task;
}

/* 销毁队列 */
void queue_destroy(TaskQueue *queue) {
    pthread_mutex_destroy(&queue->lock);
    pthread_cond_destroy(&queue->cond);
}
```

### 2.3 线程池实现

```c
/* 工作线程函数 */
static void *worker_thread(void *arg) {
    ThreadPool *pool = (ThreadPool *)arg;

    while (1) {
        Task *task = queue_pop(&pool->queue);

        if (!task) continue;

        // 检查关闭信号
        if (pool->shutdown) {
            free(task);
            break;
        }

        // 执行任务
        task->function(task->argument);
        free(task);
    }

    return NULL;
}

/* 创建线程池 */
ThreadPool *threadpool_create(int num_threads) {
    ThreadPool *pool = (ThreadPool *)calloc(1, sizeof(ThreadPool));
    pool->thread_count = num_threads;
    pool->shutdown = 0;

    queue_init(&pool->queue);
    pthread_mutex_init(&pool->lock, NULL);

    // 创建工作线程
    pool->threads = (pthread_t *)malloc(num_threads * sizeof(pthread_t));
    for (int i = 0; i < num_threads; i++) {
        pthread_create(&pool->threads[i], NULL, worker_thread, pool);
    }

    printf("线程池创建完成: %d 个工作线程\n", num_threads);
    return pool;
}

/* 提交任务 */
void threadpool_submit(ThreadPool *pool, void (*func)(void *), void *arg) {
    if (pool->shutdown) return;
    queue_push(&pool->queue, func, arg);
}

/* 等待所有任务完成 */
void threadpool_wait(ThreadPool *pool) {
    while (1) {
        pthread_mutex_lock(&pool->queue.lock);
        int remaining = pool->queue.count;
        pthread_mutex_unlock(&pool->queue.lock);

        if (remaining == 0) break;
        usleep(1000);  // 1ms
    }
}

/* 关闭线程池 */
void threadpool_shutdown(ThreadPool *pool) {
    pool->shutdown = 1;

    // 唤醒所有等待的线程
    pthread_mutex_lock(&pool->queue.lock);
    pthread_cond_broadcast(&pool->queue.cond);
    pthread_mutex_unlock(&pool->queue.lock);

    // 等待所有线程结束
    for (int i = 0; i < pool->thread_count; i++) {
        pthread_join(pool->threads[i], NULL);
    }

    // 清理剩余任务
    Task *task;
    while ((task = queue_try_pop(&pool->queue)) != NULL) {
        free(task);
    }

    free(pool->threads);
    queue_destroy(&pool->queue);
    pthread_mutex_destroy(&pool->lock);
    free(pool);

    printf("线程池已关闭\n");
}

/* 获取队列中的任务数 */
int threadpool_pending(ThreadPool *pool) {
    pthread_mutex_lock(&pool->queue.lock);
    int count = pool->queue.count;
    pthread_mutex_unlock(&pool->queue.lock);
    return count;
}
```

## 3. 动态线程池

```c
/*
 * 动态线程池
 * 根据负载自动调整线程数量
 */
typedef struct {
    ThreadPool *base;
    int min_threads;            // 最小线程数
    int max_threads;            // 最大线程数
    int current_threads;        // 当前线程数
    int idle_threads;           // 空闲线程数
    int target_queue_size;      // 目标队列大小
    pthread_t monitor_thread;   // 监控线程
    int shutdown;
} DynamicThreadPool;

/* 监控线程：根据负载调整线程数 */
static void *monitor_thread_func(void *arg) {
    DynamicThreadPool *dp = (DynamicThreadPool *)arg;

    while (!dp->shutdown) {
        int pending = threadpool_pending(dp->base);

        if (pending > dp->target_queue_size && dp->current_threads < dp->max_threads) {
            // 负载高，增加线程
            // （简化版本，实际需要动态添加线程）
            printf("[监控] 负载高: %d个待处理任务\n", pending);
        } else if (pending == 0 && dp->current_threads > dp->min_threads) {
            // 负载低，减少线程
            printf("[监控] 负载低，可减少线程\n");
        }

        usleep(100000);  // 100ms检查一次
    }

    return NULL;
}
```

## 4. Windows版本（简化）

```c
#ifdef _WIN32
#include <windows.h>

/*
 * Windows线程池（使用临界区和事件）
 * 注意：Windows原生线程池API (CreateThreadpoolWork) 更推荐
 */

typedef struct WinTask {
    void (*func)(void *);
    void *arg;
    struct WinTask *next;
} WinTask;

typedef struct {
    HANDLE *threads;
    int thread_count;
    WinTask *task_head;
    WinTask *task_tail;
    int task_count;
    CRITICAL_SECTION cs;
    HANDLE event;
    int shutdown;
} WinThreadPool;

DWORD WINAPI win_worker(LPVOID param) {
    WinThreadPool *pool = (WinThreadPool *)param;

    while (1) {
        WaitForSingleObject(pool->event, INFINITE);

        EnterCriticalSection(&pool->cs);
        if (pool->shutdown) {
            LeaveCriticalSection(&pool->cs);
            break;
        }

        WinTask *task = pool->task_head;
        if (task) {
            pool->task_head = task->next;
            if (!pool->task_head) pool->task_tail = NULL;
            pool->task_count--;
        }
        LeaveCriticalSection(&pool->cs);

        if (task) {
            task->func(task->arg);
            free(task);
        }
    }

    return 0;
}
#endif
```

## 5. 测试

```c
/* 示例任务 */
void example_task(void *arg) {
    int id = *(int *)arg;
    printf("  任务 %d 开始执行 (线程: %lu)\n",
           id, (unsigned long)pthread_self());
    usleep((rand() % 500 + 100) * 1000);  // 模拟工作
    printf("  任务 %d 完成\n", id);
}

int main() {
    srand(time(NULL));

    // 创建4线程的线程池
    ThreadPool *pool = threadpool_create(4);

    // 提交20个任务
    int ids[20];
    printf("\n提交20个任务...\n");
    for (int i = 0; i < 20; i++) {
        ids[i] = i + 1;
        threadpool_submit(pool, example_task, &ids[i]);
    }

    // 等待所有任务完成
    printf("\n等待任务完成...\n");
    threadpool_wait(pool);

    printf("\n所有任务完成\n");

    // 关闭线程池
    threadpool_shutdown(pool);

    return 0;
}
```

## 小结

- 线程池通过复用线程减少创建/销毁开销
- 任务队列是线程池的核心数据结构，需要加锁保护
- 条件变量用于高效地等待新任务
- 动态线程池可根据负载自动增减线程数
- 生产环境推荐使用平台原生线程池API或第三方库
