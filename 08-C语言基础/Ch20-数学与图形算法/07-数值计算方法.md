# 数值计算方法

## 概述

数值计算方法是用计算机近似求解数学问题的科学。本节介绍二分法求根、牛顿法、数值积分等常用方法。

---

## 1. 二分法求根

### 1.1 基本原理

如果函数 `f(x)` 在区间 `[a, b]` 上连续且 `f(a) * f(b) < 0`，则至少存在一个根。

```c
#include <stdio.h>
#include <math.h>

#define EPS 1e-9

/*
 * 二分法求方程 f(x) = 0 在 [a, b] 上的根
 * 要求: f(a) * f(b) < 0
 */
double bisection(double (*f)(double), double a, double b, int max_iter) {
    double fa = f(a);
    double fb = f(b);

    if (fa * fb > 0) {
        printf("错误: f(a) 和 f(b) 同号\n");
        return NAN;
    }

    for (int i = 0; i < max_iter; i++) {
        double mid = (a + b) / 2.0;
        double fm = f(mid);

        if (fabs(fm) < EPS || (b - a) / 2.0 < EPS) {
            return mid;
        }

        if (fa * fm < 0) {
            b = mid;
            fb = fm;
        } else {
            a = mid;
            fa = fm;
        }
    }
    return (a + b) / 2.0;
}

/* 示例函数: f(x) = x^3 - x - 2 */
double f1(double x) {
    return x * x * x - x - 2;
}

int main() {
    double root = bisection(f1, 1.0, 2.0, 100);
    printf("x^3 - x - 2 = 0 的根: %.10f\n", root);
    printf("验证: f(%.10f) = %.2e\n", root, f1(root));
    return 0;
}
```

## 2. 牛顿法（切线法）

### 2.1 基本原理

迭代公式：`x_{n+1} = x_n - f(x_n) / f'(x_n)`

收敛速度通常比二分法快，但需要导数且可能发散。

```c
#include <stdio.h>
#include <math.h>

/*
 * 牛顿法求根
 * f: 目标函数
 * df: 导函数
 * x0: 初始猜测值
 */
double newton(double (*f)(double), double (*df)(double),
              double x0, int max_iter) {
    double x = x0;

    for (int i = 0; i < max_iter; i++) {
        double fx = f(x);
        double dfx = df(x);

        if (fabs(dfx) < 1e-15) {
            printf("导数接近零，无法继续迭代\n");
            return x;
        }

        double x_new = x - fx / dfx;

        if (fabs(x_new - x) < EPS) {
            return x_new;
        }
        x = x_new;
    }
    return x;
}

/* f(x) = x^3 - x - 2 */
double f_newton(double x) {
    return x * x * x - x - 2;
}

/* f'(x) = 3x^2 - 1 */
double df_newton(double x) {
    return 3 * x * x - 1;
}

/* 示例: 求平方根 a^(1/2) 使用牛顿法 */
double sqrt_newton(double a) {
    if (a < 0) return NAN;
    if (a == 0) return 0;

    double x = a / 2.0;  // 初始猜测
    for (int i = 0; i < 100; i++) {
        double x_new = (x + a / x) / 2.0;  // 牛顿迭代
        if (fabs(x_new - x) < EPS) return x_new;
        x = x_new;
    }
    return x;
}

int main() {
    double root = newton(f_newton, df_newton, 1.5, 100);
    printf("牛顿法求根: %.10f\n", root);
    printf("验证: f(%.10f) = %.2e\n", root, f_newton(root));

    printf("\n牛顿法求平方根:\n");
    printf("sqrt(2) = %.15f\n", sqrt_newton(2));
    printf("sqrt(3) = %.15f\n", sqrt_newton(3));

    return 0;
}
```

## 3. 弦截法

不需要计算导数，用差商近似导数。

```c
/*
 * 弦截法求根
 * 用前两个点的割线斜率代替导数
 */
double secant_method(double (*f)(double), double x0, double x1, int max_iter) {
    for (int i = 0; i < max_iter; i++) {
        double f0 = f(x0);
        double f1 = f(x1);

        if (fabs(f1 - f0) < 1e-15) {
            return x1;
        }

        double x2 = x1 - f1 * (x1 - x0) / (f1 - f0);

        if (fabs(x2 - x1) < EPS) {
            return x2;
        }

        x0 = x1;
        x1 = x2;
    }
    return x1;
}
```

## 4. 数值积分

### 4.1 梯形法则

```c
/*
 * 梯形法则数值积分
 * 将区间 [a, b] 等分为 n 份
 */
double trapezoidal(double (*f)(double), double a, double b, int n) {
    double h = (b - a) / n;
    double sum = (f(a) + f(b)) / 2.0;

    for (int i = 1; i < n; i++) {
        double x = a + i * h;
        sum += f(x);
    }

    return sum * h;
}
```

### 4.2 辛普森法则

```c
/*
 * 辛普森法则（复合辛普森公式）
 * 精度比梯形法则高
 */
double simpson(double (*f)(double), double a, double b, int n) {
    if (n % 2 != 0) n++;  // n 必须是偶数
    double h = (b - a) / n;
    double sum = f(a) + f(b);

    for (int i = 1; i < n; i++) {
        double x = a + i * h;
        if (i % 2 == 0)
            sum += 2 * f(x);
        else
            sum += 4 * f(x);
    }

    return sum * h / 3.0;
}
```

### 4.3 自适应辛普森

```c
/* 自适应辛普森积分，自动控制精度 */
double adaptive_simpson_helper(double (*f)(double), double a, double b,
                                double eps, double whole) {
    double mid = (a + b) / 2.0;
    double left = simpson(f, a, mid, 2);
    double right = simpson(f, mid, b, 2);

    if (fabs(left + right - whole) < 15 * eps) {
        return left + right;
    }

    return adaptive_simpson_helper(f, a, mid, eps / 2, left) +
           adaptive_simpson_helper(f, mid, b, eps / 2, right);
}

double adaptive_simpson(double (*f)(double), double a, double b, double eps) {
    double whole = simpson(f, a, b, 100);
    return adaptive_simpson_helper(f, a, b, eps, whole);
}

/* 测试函数 */
double sin_func(double x) {
    return sin(x);
}

double gaussian(double x) {
    return exp(-x * x);
}

int main() {
    // sin(x) 在 [0, pi] 上的积分 = 2
    printf("sin(x) dx [0,pi]:\n");
    printf("  梯形法则: %.10f\n", trapezoidal(sin_func, 0, M_PI, 1000));
    printf("  辛普森法则: %.10f\n", simpson(sin_func, 0, M_PI, 1000));
    printf("  精确值:   2.0000000000\n");

    // 高斯积分 e^(-x^2) 在 [0, 1] 上
    printf("\ne^(-x^2) dx [0,1]:\n");
    printf("  辛普森法则: %.10f\n", simpson(gaussian, 0, 1, 1000));

    return 0;
}
```

## 5. 常微分方程数值解

### 5.1 欧拉法

```c
/*
 * 欧拉法求解 dy/dx = f(x, y), y(x0) = y0
 * 简单但精度低
 */
void euler(double (*f)(double, double), double x0, double y0,
           double h, int steps) {
    double x = x0, y = y0;

    printf("欧拉法求解:\n");
    printf("  x = %8.4f, y = %8.4f\n", x, y);

    for (int i = 0; i < steps; i++) {
        y = y + h * f(x, y);
        x = x + h;
        if (i < 5 || i >= steps - 3)
            printf("  x = %8.4f, y = %8.4f\n", x, y);
        else if (i == 5)
            printf("  ...\n");
    }
}
```

### 5.2 四阶龙格-库塔法

```c
/*
 * 经典四阶龙格-库塔法 (RK4)
 * 精度高，是工程中最常用的ODE求解方法
 */
void rk4(double (*f)(double, double), double x0, double y0,
         double h, int steps) {
    double x = x0, y = y0;

    printf("RK4求解:\n");
    printf("  x = %8.4f, y = %8.4f\n", x, y);

    for (int i = 0; i < steps; i++) {
        double k1 = h * f(x, y);
        double k2 = h * f(x + h / 2, y + k1 / 2);
        double k3 = h * f(x + h / 2, y + k2 / 2);
        double k4 = h * f(x + h, y + k3);

        y = y + (k1 + 2 * k2 + 2 * k3 + k4) / 6.0;
        x = x + h;

        if (i < 5 || i >= steps - 3)
            printf("  x = %8.4f, y = %8.4f\n", x, y);
        else if (i == 5)
            printf("  ...\n");
    }
}

/* 示例: dy/dx = -2xy, 精确解 y = e^(-x^2) */
double test_ode(double x, double y) {
    return -2 * x * y;
}

int main() {
    printf("求解 dy/dx = -2xy, y(0) = 1\n");
    printf("精确解: y = e^(-x^2)\n\n");

    euler(test_ode, 0, 1, 0.1, 10);
    printf("\n");
    rk4(test_ode, 0, 1, 0.1, 10);

    printf("\n精确值: y(1) = %.10f\n", exp(-1.0));

    return 0;
}
```

## 小结

- 二分法稳定但收敛慢，适合有确定区间的情况
- 牛顿法收敛快但需要导数，且对初始值敏感
- 辛普森法则在数值积分中精度高，自适应版本可自动控制误差
- 四阶龙格-库塔法是求解常微分方程的标准方法
