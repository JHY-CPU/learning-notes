# 网络协议解析

## 概述

理解网络协议的底层结构是网络编程的基础。本节用C语言解析IP、TCP、UDP数据包头，并简要介绍抓包原理。

## 1. 字节序处理

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>

/*
 * 网络字节序（大端序）与主机字节序转换
 */
uint16_t ntohs_custom(uint16_t net_short) {
    return (net_short >> 8) | (net_short << 8);
}

uint32_t ntohl_custom(uint32_t net_long) {
    return ((net_long >> 24) & 0xFF) |
           ((net_long >> 8)  & 0xFF00) |
           ((net_long << 8)  & 0xFF0000) |
           ((net_long << 24) & 0xFF000000);
}

#define HTONS ntohs_custom
#define HTONL ntohl_custom
```

## 2. 以太网帧头

```c
#pragma pack(push, 1)

/* 以太网帧头 (14字节) */
typedef struct {
    uint8_t  dst_mac[6];    // 目标MAC地址
    uint8_t  src_mac[6];    // 源MAC地址
    uint16_t ether_type;    // 上层协议类型
} EthernetHeader;

#define ETHERTYPE_IP  0x0800
#define ETHERTYPE_ARP 0x0806

/* 打印MAC地址 */
void print_mac(const uint8_t *mac) {
    printf("%02x:%02x:%02x:%02x:%02x:%02x",
           mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
}

/* 解析以太网帧头 */
void parse_ethernet(const uint8_t *data) {
    EthernetHeader *eth = (EthernetHeader *)data;

    printf("=== 以太网帧 ===\n");
    printf("目标MAC: "); print_mac(eth->dst_mac); printf("\n");
    printf("源MAC:   "); print_mac(eth->src_mac); printf("\n");
    printf("类型:    0x%04x", ntohs(eth->ether_type));

    switch (ntohs(eth->ether_type)) {
        case ETHERTYPE_IP:  printf(" (IPv4)"); break;
        case ETHERTYPE_ARP: printf(" (ARP)"); break;
    }
    printf("\n\n");
}
```

## 3. IP数据包头

```c
/* IPv4头部 (20字节基础) */
typedef struct {
    uint8_t  version_ihl;       // 版本(4位) + 头长度(4位)
    uint8_t  tos;               // 服务类型
    uint16_t total_length;      // 总长度
    uint16_t identification;    // 标识
    uint16_t flags_fragment;    // 标志(3位) + 片偏移(13位)
    uint8_t  ttl;               // 生存时间
    uint8_t  protocol;          // 协议类型
    uint16_t checksum;          // 头部校验和
    uint32_t src_ip;            // 源IP地址
    uint32_t dst_ip;            // 目标IP地址
} IPv4Header;

#define IP_PROTOCOL_ICMP 1
#define IP_PROTOCOL_TCP  6
#define IP_PROTOCOL_UDP  17

/* IP地址转字符串 */
void ip_to_string(uint32_t ip, char *buf) {
    sprintf(buf, "%d.%d.%d.%d",
            (ip >> 24) & 0xFF, (ip >> 16) & 0xFF,
            (ip >> 8) & 0xFF, ip & 0xFF);
}

/* 解析IP数据包 */
void parse_ipv4(const uint8_t *data) {
    IPv4Header *ip = (IPv4Header *)data;

    int version = (ip->version_ihl >> 4) & 0x0F;
    int ihl = (ip->version_ihl & 0x0F) * 4;  // 头部长度（字节）

    char src_ip[16], dst_ip[16];
    ip_to_string(ntohl(ip->src_ip), src_ip);
    ip_to_string(ntohl(ip->dst_ip), dst_ip);

    printf("=== IPv4数据包 ===\n");
    printf("版本:     %d\n", version);
    printf("头长度:   %d 字节\n", ihl);
    printf("总长度:   %d 字节\n", ntohs(ip->total_length));
    printf("TTL:      %d\n", ip->ttl);
    printf("协议:     %d", ip->protocol);

    switch (ip->protocol) {
        case IP_PROTOCOL_ICMP: printf(" (ICMP)"); break;
        case IP_PROTOCOL_TCP:  printf(" (TCP)"); break;
        case IP_PROTOCOL_UDP:  printf(" (UDP)"); break;
    }
    printf("\n");

    printf("校验和:   0x%04x\n", ntohs(ip->checksum));
    printf("源IP:     %s\n", src_ip);
    printf("目标IP:   %s\n\n", dst_ip);
}

/* 计算IP校验和 */
uint16_t ip_checksum(const uint8_t *data, int len) {
    uint32_t sum = 0;
    uint16_t *ptr = (uint16_t *)data;

    while (len > 1) {
        sum += *ptr++;
        len -= 2;
    }
    if (len == 1) {
        sum += *(uint8_t *)ptr;
    }

    while (sum >> 16) {
        sum = (sum & 0xFFFF) + (sum >> 16);
    }

    return ~((uint16_t)sum);
}
```

## 4. TCP数据段头

```c
/* TCP头部 (20字节基础) */
typedef struct {
    uint16_t src_port;          // 源端口
    uint16_t dst_port;          // 目标端口
    uint32_t seq_number;        // 序列号
    uint32_t ack_number;        // 确认号
    uint16_t offset_flags;      // 数据偏移(4位) + 保留(6位) + 标志(6位)
    uint16_t window;            // 窗口大小
    uint16_t checksum;          // 校验和
    uint16_t urgent_pointer;    // 紧急指针
} TCPHeader;

/* TCP标志位 */
#define TCP_FIN  0x01
#define TCP_SYN  0x02
#define TCP_RST  0x04
#define TCP_PSH  0x08
#define TCP_ACK  0x10
#define TCP_URG  0x20

/* 解析TCP数据段 */
void parse_tcp(const uint8_t *data) {
    TCPHeader *tcp = (TCPHeader *)data;

    int offset = ((ntohs(tcp->offset_flags) >> 12) & 0x0F) * 4;
    uint8_t flags = ntohs(tcp->offset_flags) & 0x3F;

    printf("=== TCP数据段 ===\n");
    printf("源端口:   %d\n", ntohs(tcp->src_port));
    printf("目标端口: %d\n", ntohs(tcp->dst_port));
    printf("序列号:   %u\n", ntohl(tcp->seq_number));
    printf("确认号:   %u\n", ntohl(tcp->ack_number));
    printf("头长度:   %d 字节\n", offset);
    printf("窗口:     %d\n", ntohs(tcp->window));

    printf("标志:     ");
    if (flags & TCP_SYN) printf("SYN ");
    if (flags & TCP_ACK) printf("ACK ");
    if (flags & TCP_FIN) printf("FIN ");
    if (flags & TCP_RST) printf("RST ");
    if (flags & TCP_PSH) printf("PSH ");
    if (flags & TCP_URG) printf("URG ");
    printf("\n\n");
}

/* 构造TCP SYN数据包（用于演示） */
void build_tcp_syn(uint8_t *buffer, uint16_t src_port, uint16_t dst_port,
                   uint32_t seq) {
    TCPHeader *tcp = (TCPHeader *)buffer;
    memset(tcp, 0, sizeof(TCPHeader));

    tcp->src_port = htons(src_port);
    tcp->dst_port = htons(dst_port);
    tcp->seq_number = htonl(seq);
    tcp->ack_number = 0;
    tcp->offset_flags = htons((5 << 12) | TCP_SYN);  // 头长20字节 + SYN
    tcp->window = htons(65535);
    tcp->checksum = 0;
}
```

## 6. TCP状态机

```c
/* TCP连接状态 */
typedef enum {
    TCP_CLOSED,
    TCP_LISTEN,
    TCP_SYN_SENT,
    TCP_SYN_RECEIVED,
    TCP_ESTABLISHED,
    TCP_FIN_WAIT_1,
    TCP_FIN_WAIT_2,
    TCP_CLOSE_WAIT,
    TCP_CLOSING,
    TCP_LAST_ACK,
    TCP_TIME_WAIT
} TCPState;

const char *tcp_state_name(TCPState state) {
    const char *names[] = {
        "CLOSED", "LISTEN", "SYN_SENT", "SYN_RECEIVED",
        "ESTABLISHED", "FIN_WAIT_1", "FIN_WAIT_2",
        "CLOSE_WAIT", "CLOSING", "LAST_ACK", "TIME_WAIT"
    };
    return names[state];
}

/* TCP三次握手模拟 */
void simulate_three_way_handshake() {
    printf("=== TCP三次握手模拟 ===\n\n");

    TCPState client = TCP_CLOSED;
    TCPState server = TCP_LISTEN;

    // 第一次握手: 客户端发送SYN
    printf("1. 客户端 -> 服务端: SYN (seq=1000)\n");
    client = TCP_SYN_SENT;
    printf("   客户端状态: %s\n", tcp_state_name(client));

    // 第二次握手: 服务端回复SYN+ACK
    printf("2. 服务端 -> 客户端: SYN+ACK (seq=2000, ack=1001)\n");
    server = TCP_SYN_RECEIVED;
    printf("   服务端状态: %s\n", tcp_state_name(server));

    // 第三次握手: 客户端发送ACK
    printf("3. 客户端 -> 服务端: ACK (ack=2001)\n");
    client = TCP_ESTABLISHED;
    server = TCP_ESTABLISHED;
    printf("   客户端状态: %s\n", tcp_state_name(client));
    printf("   服务端状态: %s\n", tcp_state_name(server));
    printf("\n连接建立完成!\n");
}
```

## 小结

- 网络协议采用分层结构：以太网帧 > IP数据包 > TCP/UDP段
- 所有协议头部使用网络字节序（大端序），需注意字节序转换
- TCP通过三次握手建立连接，四次挥手断开连接
- 抓包工具（如Wireshark）通过原始套接字或专用库捕获网络数据
- 校验和通过反码求和算法计算，用于检测数据完整性
