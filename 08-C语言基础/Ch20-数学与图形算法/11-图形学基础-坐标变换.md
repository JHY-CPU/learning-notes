# 图形学基础 - 坐标变换

## 概述

坐标变换是计算机图形学的核心基础，包括平移、旋转、缩放等操作。本节介绍2D和3D坐标变换的原理与C语言实现。

## 1. 二维向量与矩阵

### 1.1 二维向量

```c
#include <stdio.h>
#include <math.h>

typedef struct {
    double x, y;
} Vec2;

Vec2 vec2_create(double x, double y) {
    return (Vec2){x, y};
}

Vec2 vec2_add(Vec2 a, Vec2 b) {
    return (Vec2){a.x + b.x, a.y + b.y};
}

Vec2 vec2_sub(Vec2 a, Vec2 b) {
    return (Vec2){a.x - b.x, a.y - b.y};
}

Vec2 vec2_scale(Vec2 v, double s) {
    return (Vec2){v.x * s, v.y * s};
}

double vec2_dot(Vec2 a, Vec2 b) {
    return a.x * b.x + a.y * b.y;
}

double vec2_length(Vec2 v) {
    return sqrt(v.x * v.x + v.y * v.y);
}

Vec2 vec2_normalize(Vec2 v) {
    double len = vec2_length(v);
    if (len < 1e-10) return (Vec2){0, 0};
    return (Vec2){v.x / len, v.y / len};
}

void vec2_print(Vec2 v) {
    printf("(%.4f, %.4f)", v.x, v.y);
}
```

### 1.2 二维变换矩阵（3x3齐次坐标）

```c
/*
 * 3x3矩阵用于2D齐次坐标变换
 * 使用齐次坐标可以统一表示所有仿射变换
 */
typedef struct {
    double m[3][3];
} Mat3;

Mat3 mat3_identity() {
    Mat3 mat = {{{1,0,0},{0,1,0},{0,0,1}}};
    return mat;
}

Mat3 mat3_multiply(Mat3 a, Mat3 b) {
    Mat3 result = {{{0}}};
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            for (int k = 0; k < 3; k++)
                result.m[i][j] += a.m[i][k] * b.m[k][j];
    return result;
}

/* 用矩阵变换点 (x, y, 1) */
Vec2 mat3_transform_point(Mat3 mat, Vec2 p) {
    double x = mat.m[0][0] * p.x + mat.m[0][1] * p.y + mat.m[0][2];
    double y = mat.m[1][0] * p.x + mat.m[1][1] * p.y + mat.m[1][2];
    double w = mat.m[2][0] * p.x + mat.m[2][1] * p.y + mat.m[2][2];
    if (fabs(w) > 1e-10) {
        x /= w; y /= w;
    }
    return (Vec2){x, y};
}

void mat3_print(Mat3 mat) {
    for (int i = 0; i < 3; i++) {
        printf("[ ");
        for (int j = 0; j < 3; j++) {
            printf("%8.4f ", mat.m[i][j]);
        }
        printf("]\n");
    }
}
```

## 2. 二维变换

### 2.1 平移变换

```c
/* 平移变换矩阵 */
Mat3 mat3_translation(double tx, double ty) {
    Mat3 mat = mat3_identity();
    mat.m[0][2] = tx;
    mat.m[1][2] = ty;
    return mat;
}
```

### 2.2 旋转变换

```c
/* 绕原点旋转 angle 弧度（逆时针） */
Mat3 mat3_rotation(double angle) {
    Mat3 mat = mat3_identity();
    double c = cos(angle);
    double s = sin(angle);
    mat.m[0][0] = c;  mat.m[0][1] = -s;
    mat.m[1][0] = s;  mat.m[1][1] = c;
    return mat;
}

/* 绕任意点 (cx, cy) 旋转 */
Mat3 mat3_rotation_around(double cx, double cy, double angle) {
    Mat3 t1 = mat3_translation(-cx, -cy);  // 平移到原点
    Mat3 r  = mat3_rotation(angle);         // 旋转
    Mat3 t2 = mat3_translation(cx, cy);    // 平移回去
    return mat3_multiply(t2, mat3_multiply(r, t1));
}
```

### 2.3 缩放变换

```c
/* 缩放变换矩阵 */
Mat3 mat3_scaling(double sx, double sy) {
    Mat3 mat = mat3_identity();
    mat.m[0][0] = sx;
    mat.m[1][1] = sy;
    return mat;
}

/* 相对于点 (cx, cy) 缩放 */
Mat3 mat3_scaling_around(double cx, double cy, double sx, double sy) {
    Mat3 t1 = mat3_translation(-cx, -cy);
    Mat3 s  = mat3_scaling(sx, sy);
    Mat3 t2 = mat3_translation(cx, cy);
    return mat3_multiply(t2, mat3_multiply(s, t1));
}
```

/* 水平错切: x' = x + shx * y */
Mat3 mat3_shear_x(double shx) {

## 3. 三维向量与变换

### 3.1 三维向量

```c
typedef struct {
    double x, y, z;
} Vec3;

Vec3 vec3_create(double x, double y, double z) {
    return (Vec3){x, y, z};
}

Vec3 vec3_add(Vec3 a, Vec3 b) {
    return (Vec3){a.x + b.x, a.y + b.y, a.z + b.z};
}

Vec3 vec3_sub(Vec3 a, Vec3 b) {
    return (Vec3){a.x - b.x, a.y - b.y, a.z - b.z};
}

Vec3 vec3_scale(Vec3 v, double s) {
    return (Vec3){v.x * s, v.y * s, v.z * s};
}

double vec3_dot(Vec3 a, Vec3 b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 vec3_cross(Vec3 a, Vec3 b) {
    return (Vec3){
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double vec3_length(Vec3 v) {
    return sqrt(v.x * v.x + v.y * v.y + v.z * v.z);
}

Vec3 vec3_normalize(Vec3 v) {
    double len = vec3_length(v);
    if (len < 1e-10) return (Vec3){0, 0, 0};
    return vec3_scale(v, 1.0 / len);
}
```

### 3.2 4x4变换矩阵

```c
typedef struct {
    double m[4][4];
} Mat4;

Mat4 mat4_identity() {
    Mat4 mat = {{{0}}};
    for (int i = 0; i < 4; i++) mat.m[i][i] = 1.0;
    return mat;
}

Mat4 mat4_multiply(Mat4 a, Mat4 b) {
    Mat4 result = {{{0}}};
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 4; j++)
            for (int k = 0; k < 4; k++)
                result.m[i][j] += a.m[i][k] * b.m[k][j];
    return result;
}

Vec3 mat4_transform_point(Mat4 mat, Vec3 p) {
    double x = mat.m[0][0]*p.x + mat.m[0][1]*p.y + mat.m[0][2]*p.z + mat.m[0][3];
    double y = mat.m[1][0]*p.x + mat.m[1][1]*p.y + mat.m[1][2]*p.z + mat.m[1][3];
    double z = mat.m[2][0]*p.x + mat.m[2][1]*p.y + mat.m[2][2]*p.z + mat.m[2][3];
    double w = mat.m[3][0]*p.x + mat.m[3][1]*p.y + mat.m[3][2]*p.z + mat.m[3][3];
    if (fabs(w) > 1e-10) { x /= w; y /= w; z /= w; }
    return (Vec3){x, y, z};
}

/* 3D平移 */
Mat4 mat4_translation(double tx, double ty, double tz) {
    Mat4 mat = mat4_identity();
    mat.m[0][3] = tx;
    mat.m[1][3] = ty;
    mat.m[2][3] = tz;
    return mat;
}

/* 3D绕X轴旋转 */
Mat4 mat4_rotation_x(double angle) {
    Mat4 mat = mat4_identity();
    double c = cos(angle), s = sin(angle);
    mat.m[1][1] = c;  mat.m[1][2] = -s;
    mat.m[2][1] = s;  mat.m[2][2] = c;
    return mat;
}

/* 3D绕Y轴旋转 */
Mat4 mat4_rotation_y(double angle) {
    Mat4 mat = mat4_identity();
    double c = cos(angle), s = sin(angle);
    mat.m[0][0] = c;  mat.m[0][2] = s;
    mat.m[2][0] = -s; mat.m[2][2] = c;
    return mat;
}

/* 3D绕Z轴旋转 */
Mat4 mat4_rotation_z(double angle) {
    Mat4 mat = mat4_identity();
    double c = cos(angle), s = sin(angle);
    mat.m[0][0] = c;  mat.m[0][1] = -s;
    mat.m[1][0] = s;  mat.m[1][1] = c;
    return mat;
}

/* 3D缩放 */
Mat4 mat4_scaling(double sx, double sy, double sz) {
    Mat4 mat = mat4_identity();
    mat.m[0][0] = sx;
    mat.m[1][1] = sy;
    mat.m[2][2] = sz;
    return mat;
}

/* 绕任意轴旋转（罗德里格斯公式） */
Mat4 mat4_rotation_axis(Vec3 axis, double angle) {
    axis = vec3_normalize(axis);
    Mat4 mat = mat4_identity();
    double c = cos(angle), s = sin(angle), t = 1 - c;
    double x = axis.x, y = axis.y, z = axis.z;

    mat.m[0][0] = t*x*x + c;    mat.m[0][1] = t*x*y - s*z;  mat.m[0][2] = t*x*z + s*y;
    mat.m[1][0] = t*x*y + s*z;  mat.m[1][1] = t*y*y + c;    mat.m[1][2] = t*y*z - s*x;
    mat.m[2][0] = t*x*z - s*y;  mat.m[2][1] = t*y*z + s*x;  mat.m[2][2] = t*z*z + c;

    return mat;
}
```

## 4. 透视投影

```c
/*
 * 透视投影矩阵
 * fov: 垂直视场角（弧度）
 * aspect: 宽高比
 * near, far: 近远裁剪面
 */
Mat4 mat4_perspective(double fov, double aspect, double near, double far) {
    Mat4 mat = {{{0}}};
    double tan_half = tan(fov / 2.0);
    mat.m[0][0] = 1.0 / (aspect * tan_half);
    mat.m[1][1] = 1.0 / tan_half;
    mat.m[2][2] = -(far + near) / (far - near);
    mat.m[2][3] = -2.0 * far * near / (far - near);
    mat.m[3][2] = -1.0;
    return mat;
}
```

## 5. 综合示例

```c
int main() {
    // 2D变换示例
    Vec2 p = vec2_create(1, 0);
    printf("原始点: "); vec2_print(p); printf("\n");

    // 先旋转90度，再平移(2, 3)
    Mat3 rot = mat3_rotation(M_PI / 2);
    Mat3 trans = mat3_translation(2, 3);
    Mat3 combined = mat3_multiply(trans, rot);

    Vec2 result = mat3_transform_point(combined, p);
    printf("变换后: "); vec2_print(result); printf("\n");

    // 3D变换示例
    Vec3 p3 = vec3_create(1, 0, 0);
    Mat4 rot_z = mat4_rotation_z(M_PI / 4);
    Vec3 r3 = mat4_transform_point(rot_z, p3);
    printf("\n3D旋转: (%.4f, %.4f, %.4f)\n", r3.x, r3.y, r3.z);

    return 0;
}
```

## 小结

- 齐次坐标使平移可以用矩阵乘法统一表示
- 2D使用3x3矩阵，3D使用4x4矩阵
- 变换矩阵的组合通过矩阵乘法实现，注意乘法顺序（从右向左应用）
- 绕任意轴旋转可以用罗德里格斯公式实现
