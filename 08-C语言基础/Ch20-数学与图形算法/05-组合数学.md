# 组合数学

## 概述

组合数学研究计数、排列和组合问题。本节涵盖排列组合、卡特兰数、卢卡斯定理等核心内容。

## 1. 排列与组合

### 1.1 阶乘与排列数

```c
#include <stdio.h>

/* 计算阶乘 n! */
long long factorial(int n) {
    long long result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

/* 排列数 P(n, k) = n! / (n-k)! */
long long permutation(int n, int k) {
    long long result = 1;
    for (int i = 0; i < k; i++) {
        result *= (n - i);
    }
    return result;
}

int main() {
    printf("5! = %lld\n", factorial(5));         // 120
    printf("P(5,3) = %lld\n", permutation(5, 3)); // 60
    printf("P(10,2) = %lld\n", permutation(10, 2)); // 90
    return 0;
}
```

### 1.2 组合数

```c
/*
 * 组合数 C(n, k) 递归实现
 * 利用帕斯卡公式: C(n,k) = C(n-1,k-1) + C(n-1,k)
 */
long long comb_pascal(int n, int k) {
    if (k == 0 || k == n) return 1;
    if (k > n) return 0;
    return comb_pascal(n - 1, k - 1) + comb_pascal(n - 1, k);
}

/* 组合数 C(n, k) 迭代实现，利用公式 C(n,k) = n! / (k! * (n-k)!) */
long long comb(int n, int k) {
    if (k > n - k) k = n - k;  // 对称性优化
    long long result = 1;
    for (int i = 0; i < k; i++) {
        result = result * (n - i) / (i + 1);
    }
    return result;
}

/* 打印杨辉三角（帕斯卡三角） */
void print_pascal_triangle(int rows) {
    for (int i = 0; i < rows; i++) {
        // 打印前导空格
        for (int j = 0; j < rows - i - 1; j++) {
            printf("  ");
        }
        for (int j = 0; j <= i; j++) {
            printf("%4lld ", comb(i, j));
        }
        printf("\n");
    }
}

int main() {
    printf("C(5,2) = %lld\n", comb(5, 2));   // 10
    printf("C(10,3) = %lld\n", comb(10, 3));  // 120

    printf("\n杨辉三角(8行):\n");
    print_pascal_triangle(8);

    return 0;
}
```

### 1.3 取模组合数

```c
#define MOD 1000000007LL
#define MAXN 100001

long long fact[MAXN], inv_fact[MAXN];

long long power_mod(long long base, long long exp, long long mod) {
    long long result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) result = result * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return result;
}

void precompute(int n) {
    fact[0] = 1;
    for (int i = 1; i <= n; i++)
        fact[i] = fact[i - 1] * i % MOD;
    inv_fact[n] = power_mod(fact[n], MOD - 2, MOD);
    for (int i = n - 1; i >= 0; i--)
        inv_fact[i] = inv_fact[i + 1] * (i + 1) % MOD;
}

long long comb_mod(int n, int k) {
    if (k < 0 || k > n) return 0;
    return fact[n] * inv_fact[k] % MOD * inv_fact[n - k] % MOD;
}
```

## 2. 卡特兰数（Catalan Numbers）

### 2.1 定义与公式

卡特兰数满足递推关系：

```
C(0) = 1
C(n) = sum(C(i) * C(n-1-i)), i = 0..n-1
C(n) = C(2n,n) / (n+1)
```

前几项：1, 1, 2, 5, 14, 42, 132, 429, 1430, 4862...

### 2.2 C语言实现

```c
#include <stdio.h>

/* 卡特兰数: 递推法 */
long long catalan_dp(int n) {
    long long cat[n + 1];
    cat[0] = 1;
    for (int i = 1; i <= n; i++) {
        cat[i] = 0;
        for (int j = 0; j < i; j++) {
            cat[i] += cat[j] * cat[i - 1 - j];
        }
    }
    return cat[n];
}

/* 卡特兰数: 公式法 C(2n,n)/(n+1) */
long long catalan_formula(int n) {
    return comb(2 * n, n) / (n + 1);
}

/* 卡特兰数: 取模版本 */
long long catalan_mod(int n) {
    return comb_mod(2 * n, n) * power_mod(n + 1, MOD - 2, MOD) % MOD;
}

int main() {
    printf("前15个卡特兰数:\n");
    for (int i = 0; i < 15; i++) {
        printf("C(%2d) = %lld\n", i, catalan_dp(i));
    }
    return 0;
}
```

### 2.3 卡特兰数的应用

```c
/*
 * 应用1: n对括号的合法排列数
 * 应用2: n个节点的二叉搜索树个数
 * 应用3: 凸n+2边形的三角剖分数
 * 应用4: 从(0,0)到(n,n)不越过对角线的路径数
 */

/* 生成所有合法括号组合（DFS回溯） */
#include <string.h>

void generate_parenthesis(int n, int open, int close, char *buffer, int pos) {
    if (pos == 2 * n) {
        buffer[pos] = '\0';
        printf("%s\n", buffer);
        return;
    }
    if (open < n) {
        buffer[pos] = '(';
        generate_parenthesis(n, open + 1, close, buffer, pos + 1);
    }
    if (close < open) {
        buffer[pos] = ')';
        generate_parenthesis(n, open, close + 1, buffer, pos + 1);
    }
}

int main() {
    int n = 3;
    char buffer[100];
    printf("%d对括号的所有合法排列:\n", n);
    generate_parenthesis(n, 0, 0, buffer, 0);
    printf("共 C(%d) = %lld 种\n", n, catalan_dp(n));
    return 0;
}
```

## 3. 卢卡斯定理（Lucas Theorem）

### 3.1 原理

当 `n` 和 `k` 很大但模数 `p` 较小（且为素数）时：

```
C(n, k) mod p = C(n%p, k%p) * C(n/p, k/p) mod p
```

### 3.2 C语言实现

```c
#include <stdio.h>

long long C_lucas(long long n, long long k, long long p) {
    if (k > n) return 0;
    if (k > n - k) k = n - k;

    long long result = 1;
    for (long long i = 0; i < k; i++) {
        result = result * (n - i) % p;
        result = result * power_mod(i + 1, p - 2, p) % p;
    }
    return result;
}

/* 卢卡斯定理 */
long long lucas(long long n, long long k, long long p) {
    if (k == 0) return 1;
    return lucas(n / p, k / p, p) * C_lucas(n % p, k % p, p) % p;
}

int main() {
    long long n = 1000000, k = 500000, p = 1000000007LL;
    printf("C(%lld, %lld) mod %lld = %lld\n",
           n, k, p, lucas(n, k, p));
    return 0;
}
```

## 4. 多重集排列与组合

```c
/* 多重集排列: n个物品中有k种，每种有n1,n2,...,nk个 */
/* 答案: n! / (n1! * n2! * ... * nk!) */
long long multinomial(int n, int counts[], int k) {
    long long result = factorial(n);
    for (int i = 0; i < k; i++) {
        result /= factorial(counts[i]);
    }
    return result;
}

/* 隔板法: n个相同物品分给k个人，每人至少0个 */
/* 答案: C(n+k-1, k-1) */
long long stars_and_bars(int n, int k) {
    return comb(n + k - 1, k - 1);
}
```

## 5. 容斥原理

```c
/*
 * 容斥原理: |A∪B| = |A| + |B| - |A∩B|
 * 推广到多个集合
 */

/* 例: 1到n中能被集合primes中任意一个整除的数的个数 */
long long inclusion_exclusion(int n, int primes[], int k) {
    long long result = 0;

    // 枚举所有非空子集
    for (int mask = 1; mask < (1 << k); mask++) {
        long long lcm_val = 1;
        int bit_count = 0;

        for (int i = 0; i < k; i++) {
            if (mask & (1 << i)) {
                lcm_val = lcm_ll(lcm_val, primes[i]);
                bit_count++;
                if (lcm_val > n) break;  // 超过范围，不用继续
            }
        }

        if (lcm_val > n) continue;

        if (bit_count % 2 == 1) {
            result += n / lcm_val;
        } else {
            result -= n / lcm_val;
        }
    }
    return result;
}

int main() {
    // 1到100中能被2、3、5整除的数的个数
    int primes[] = {2, 3, 5};
    int n = 100;
    long long count = inclusion_exclusion(n, primes, 3);
    printf("1到%d中能被2、3或5整除的数: %lld个\n", n, count);

    return 0;
}
```

## 小结

- 组合数的核心公式是 `C(n,k) = n! / (k! * (n-k)!)`，注意取模时使用逆元
- 卡特兰数的典型应用包括括号匹配、二叉树计数等
- 卢卡斯定理解决了大数组合数对小素数取模的问题
- 容斥原理是处理"至少"类计数问题的有力工具
