# 贪吃蛇游戏实现

## 概述

贪吃蛇是经典控制台游戏。本节使用C语言实现完整的贪吃蛇游戏，包含移动控制、食物生成、碰撞检测和得分系统。

---

## 1. 完整代码

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#ifdef _WIN32
#include <windows.h>
#include <conio.h>
#define CLEAR system("cls")
#define SLEEP(ms) Sleep(ms)
#define KBHIT _kbhit()
#define GETCH _getch()
#else
#include <unistd.h>
#define CLEAR system("clear")
#define SLEEP(ms) usleep((ms) * 1000)
// Linux下的kbhit实现省略，参考上一节
#endif

/* 游戏地图大小 */
#define MAP_W 40
#define MAP_H 20
#define MAX_SNAKE (MAP_W * MAP_H)

/* 方向 */
enum { DIR_UP, DIR_DOWN, DIR_LEFT, DIR_RIGHT };

/* 蛇身节点 */
typedef struct { int x, y; } Point;

/* 游戏状态 */
typedef struct {
    Point snake[MAX_SNAKE];  // 蛇身（snake[0]是头部）
    int length;              // 蛇的长度
    int direction;           // 当前方向
    Point food;              // 食物位置
    int score;               // 得分
    int game_over;           // 游戏结束标志
    int speed;               // 速度（毫秒/帧）
} GameState;

/* 游戏地图缓冲区 */
static char map_buf[MAP_H][MAP_W + 1];
```

## 2. 游戏初始化

```c
/* 初始化游戏 */
void game_init(GameState *game) {
    // 蛇初始位置（地图中央）
    game->length = 3;
    game->snake[0] = (Point){MAP_W / 2, MAP_H / 2};      // 头
    game->snake[1] = (Point){MAP_W / 2 - 1, MAP_H / 2};
    game->snake[2] = (Point){MAP_W / 2 - 2, MAP_H / 2};  // 尾

    game->direction = DIR_RIGHT;
    game->score = 0;
    game->game_over = 0;
    game->speed = 150;

    srand((unsigned int)time(NULL));

    // 生成第一个食物
    generate_food(game);
}

/* 在空白位置生成食物 */
void generate_food(GameState *game) {
    int valid = 0;
    while (!valid) {
        game->food.x = rand() % MAP_W;
        game->food.y = rand() % MAP_H;

        // 确保食物不在蛇身上
        valid = 1;
        for (int i = 0; i < game->length; i++) {
            if (game->food.x == game->snake[i].x &&
                game->food.y == game->snake[i].y) {
                valid = 0;
                break;
            }
        }
    }
}
```

## 3. 输入处理

```c
/* 处理键盘输入 */
void handle_input(GameState *game) {
    if (!KBHIT) return;

    int ch = GETCH;

#ifdef _WIN32
    if (ch == 0 || ch == 224) {
        ch = GETCH;
        switch (ch) {
            case 72: if (game->direction != DIR_DOWN)  game->direction = DIR_UP;    break;
            case 80: if (game->direction != DIR_UP)    game->direction = DIR_DOWN;  break;
            case 75: if (game->direction != DIR_RIGHT) game->direction = DIR_LEFT;  break;
            case 77: if (game->direction != DIR_LEFT)  game->direction = DIR_RIGHT; break;
        }
    }
#else
    if (ch == 27) {
        GETCH;  // '['
        ch = GETCH;
        switch (ch) {
            case 'A': if (game->direction != DIR_DOWN)  game->direction = DIR_UP;    break;
            case 'B': if (game->direction != DIR_UP)    game->direction = DIR_DOWN;  break;
            case 'D': if (game->direction != DIR_RIGHT) game->direction = DIR_LEFT;  break;
            case 'C': if (game->direction != DIR_LEFT)  game->direction = DIR_RIGHT; break;
        }
    }
#endif

    // WASD控制
    switch (ch) {
        case 'w': case 'W':
            if (game->direction != DIR_DOWN) game->direction = DIR_UP; break;
        case 's': case 'S':
            if (game->direction != DIR_UP) game->direction = DIR_DOWN; break;
        case 'a': case 'A':
            if (game->direction != DIR_RIGHT) game->direction = DIR_LEFT; break;
        case 'd': case 'D':
            if (game->direction != DIR_LEFT) game->direction = DIR_RIGHT; break;
        case 'q': case 'Q': case 27:
            game->game_over = 1; break;
        case 'p': case 'P':
            // 暂停
            printf("\n游戏暂停，按任意键继续...\n");
            GETCH;
            break;
    }
}
```

## 4. 游戏逻辑更新

```c
/* 检测碰撞 */
int check_collision(GameState *game, Point head) {
    // 撞墙
    if (head.x < 0 || head.x >= MAP_W || head.y < 0 || head.y >= MAP_H) {
        return 1;
    }

    // 撞自身
    for (int i = 0; i < game->length; i++) {
        if (head.x == game->snake[i].x && head.y == game->snake[i].y) {
            return 1;
        }
    }

    return 0;
}

/* 更新游戏状态 */
void game_update(GameState *game) {
    if (game->game_over) return;

    // 计算新头部位置
    Point new_head = game->snake[0];
    switch (game->direction) {
        case DIR_UP:    new_head.y--; break;
        case DIR_DOWN:  new_head.y++; break;
        case DIR_LEFT:  new_head.x--; break;
        case DIR_RIGHT: new_head.x++; break;
    }

    // 检测碰撞
    if (check_collision(game, new_head)) {
        game->game_over = 1;
        return;
    }

    // 移动蛇身（从尾部向头部逐个移动）
    for (int i = game->length - 1; i > 0; i--) {
        game->snake[i] = game->snake[i - 1];
    }
    game->snake[0] = new_head;

    // 检测是否吃到食物
    if (new_head.x == game->food.x && new_head.y == game->food.y) {
        // 蛇变长
        if (game->length < MAX_SNAKE) {
            game->snake[game->length] = game->snake[game->length - 1];
            game->length++;
        }

        game->score += 10;

        // 提速
        if (game->speed > 50) {
            game->speed -= 2;
        }

        // 生成新食物
        generate_food(game);
    }
}
```

## 5. 渲染

```c
/* 渲染游戏画面 */
void game_render(GameState *game) {
    // 清空地图
    for (int y = 0; y < MAP_H; y++) {
        memset(map_buf[y], ' ', MAP_W);
        map_buf[y][MAP_W] = '\0';
    }

    // 绘制食物
    if (game->food.x >= 0 && game->food.x < MAP_W &&
        game->food.y >= 0 && game->food.y < MAP_H) {
        map_buf[game->food.y][game->food.x] = '*';
    }

    // 绘制蛇
    for (int i = 0; i < game->length; i++) {
        int x = game->snake[i].x;
        int y = game->snake[i].y;
        if (x >= 0 && x < MAP_W && y >= 0 && y < MAP_H) {
            if (i == 0) {
                // 蛇头
                switch (game->direction) {
                    case DIR_UP:    map_buf[y][x] = '^'; break;
                    case DIR_DOWN:  map_buf[y][x] = 'v'; break;
                    case DIR_LEFT:  map_buf[y][x] = '<'; break;
                    case DIR_RIGHT: map_buf[y][x] = '>'; break;
                }
            } else {
                map_buf[y][x] = '#';
            }
        }
    }

    // 输出画面
    CLEAR;
    printf("+");
    for (int i = 0; i < MAP_W; i++) printf("-");
    printf("+\n");

    for (int y = 0; y < MAP_H; y++) {
        printf("|%s|\n", map_buf[y]);
    }

    printf("+");
    for (int i = 0; i < MAP_W; i++) printf("-");
    printf("+\n");

    printf("得分: %d | 长度: %d | 方向: %s | 按Q退出, P暂停\n",
           game->score, game->length,
           (const char*[]){"上","下","左","右"}[game->direction]);
}
```

## 6. 主函数

```c
int main() {
    GameState game;
    game_init(&game);

    CLEAR;
    printf("=== 贪吃蛇游戏 ===\n");
    printf("使用 WASD 或方向键控制方向\n");
    printf("按任意键开始...\n");
    GETCH;

    while (!game.game_over) {
        handle_input(&game);
        game_update(&game);
        game_render(&game);

        if (!game.game_over) {
            SLEEP(game.speed);
        }
    }

    // 游戏结束
    printf("\n");
    printf("游戏结束!\n");
    printf("最终得分: %d\n", game.score);
    printf("蛇身长度: %d\n", game.length);

    return 0;
}
```

## 7. 增强功能

### 7.1 障碍物

```c
/* 障碍物 */
typedef struct {
    Point obstacles[100];
    int count;
} Obstacles;

void obstacles_init(Obstacles *obs) { obs->count = 0; }

void obstacles_add(Obstacles *obs, int x, int y) {
    if (obs->count < 100) {
        obs->obstacles[obs->count++] = (Point){x, y};
    }
}

int obstacles_check(Obstacles *obs, int x, int y) {
    for (int i = 0; i < obs->count; i++) {
        if (obs->obstacles[i].x == x && obs->obstacles[i].y == y)
            return 1;
    }
    return 0;
}
```

### 7.2 高分记录

```c
#define HIGHSCORE_FILE "snake_scores.dat"

/* 保存高分 */
void save_highscore(int score) {
    FILE *fp = fopen(HIGHSCORE_FILE, "a");
    if (fp) {
        fprintf(fp, "%d\n", score);
        fclose(fp);
    }
}

/* 读取最高分 */
int load_highscore() {
    FILE *fp = fopen(HIGHSCORE_FILE, "r");
    if (!fp) return 0;

    int max_score = 0, score;
    while (fscanf(fp, "%d", &score) == 1) {
        if (score > max_score) max_score = score;
    }
    fclose(fp);
    return max_score;
}
```

## 小结

- 贪吃蛇的核心数据结构是蛇身坐标数组，用数组模拟队列
- 移动时从尾部向头部逐个复制位置，头部根据方向计算新位置
- 不能直接反向移动（如向右时不能直接向左），需额外检测
- 碰撞检测包括撞墙和撞自身两种情况
- 控制台游戏需要跨平台处理键盘输入差异
