# 正则表达式引擎

## 概述

正则表达式是一种强大的文本模式匹配工具。本节从零实现一个简易的正则表达式引擎，支持基本的模式匹配功能。

---

## 1. 支持的语法

| 语法 | 含义 |
|------|------|
| `c` | 匹配字符c |
| `.` | 匹配任意单个字符 |
| `^` | 匹配行首 |
| `$` | 匹配行尾 |
| `*` | 前一个字符出现0次或多次 |
| `+` | 前一个字符出现1次或多次 |
| `?` | 前一个字符出现0次或1次 |
| `[abc]` | 字符类，匹配a、b或c |
| `[^abc]` | 否定字符类 |
| `\d` | 匹配数字 |
| `\w` | 匹配字母数字下划线 |

## 2. 简易递归回溯引擎

### 2.1 核心匹配函数

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

/*
 * 检查单个字符是否匹配
 * pattern: 模式中的字符（支持转义）
 */
static int match_char(char pattern, char c) {
    switch (pattern) {
        case '.': return 1;
        case '\\': return 0;  // 后面单独处理
        default: return pattern == c;
    }
}

/*
 * 检查字符是否在字符类中匹配
 * char_class: 字符类内容，如 "abc" 或 "^abc"
 */
static int match_char_class(const char *char_class, int len, char c) {
    int negate = 0;
    int start = 0;

    if (char_class[0] == '^') {
        negate = 1;
        start = 1;
    }

    int matched = 0;
    for (int i = start; i < len; i++) {
        if (i + 2 < len && char_class[i + 1] == '-') {
            // 范围匹配，如 a-z
            if (c >= char_class[i] && c <= char_class[i + 2]) {
                matched = 1;
                break;
            }
            i += 2;
        } else if (char_class[i] == '\\' && i + 1 < len) {
            // 转义字符
            switch (char_class[i + 1]) {
                case 'd': if (isdigit(c)) matched = 1; break;
                case 'w': if (isalnum(c) || c == '_') matched = 1; break;
                case 's': if (isspace(c)) matched = 1; break;
                default: if (char_class[i + 1] == c) matched = 1; break;
            }
            i++;
        } else {
            if (char_class[i] == c) matched = 1;
        }
    }

    return negate ? !matched : matched;
}
```

### 2.2 递归匹配

```c
/*
 * 前向声明
 */
static int match_here(const char *pattern, const char *text);

/*
 * 处理重复量词 (*, +, ?)
 * 匹配 pattern 重复 min 到 max 次
 */
static int match_repeat(const char *pattern, const char *text,
                         int min_count, int max_count) {
    int count = 0;

    // 先贪心匹配
    while (count < max_count && *text != '\0') {
        if (*pattern == '\\' && *(pattern + 1)) {
            int matched = 0;
            switch (*(pattern + 1)) {
                case 'd': matched = isdigit(*text); break;
                case 'w': matched = isalnum(*text) || *text == '_'; break;
                case 's': matched = isspace(*text); break;
                default: matched = (*text == *(pattern + 1)); break;
            }
            if (!matched) break;
        } else if (*pattern == '.') {
            // 匹配任意字符
        } else if (*pattern != *text) {
            break;
        }
        text++;
        count++;
    }

    // 回溯
    while (count >= min_count) {
        if (match_here(pattern + (*pattern == '\\' ? 2 : 1), text)) {
            return 1;
        }
        if (count == 0) break;
        count--;
        text--;
    }

    return 0;
}

/*
 * 从当前位置开始匹配
 * 返回：1=匹配成功，0=匹配失败
 */
static int match_here(const char *pattern, const char *text) {
    // 模式为空，匹配成功
    if (*pattern == '\0') return 1;

    // 处理 $ (行尾)
    if (*pattern == '$' && *(pattern + 1) == '\0') {
        return *text == '\0';
    }

    // 处理重复量词
    if (*(pattern + 1) == '*' || *(pattern + 1) == '+' || *(pattern + 1) == '?') {
        int min_count, max_count;

        switch (*(pattern + 1)) {
            case '*': min_count = 0; max_count = INT_MAX; break;
            case '+': min_count = 1; max_count = INT_MAX; break;
            case '?': min_count = 0; max_count = 1; break;
        }

        // 处理转义字符
        if (*pattern == '\\') {
            return match_repeat(pattern, text, min_count, max_count);
        }
        return match_repeat(pattern, text, min_count, max_count);
    }

    // 处理字符类 [...]
    if (*pattern == '[') {
        const char *end = strchr(pattern, ']');
        if (!end) return 0;

        int class_len = (int)(end - pattern - 1);
        if (*text != '\0' && match_char_class(pattern + 1, class_len, *text)) {
            return match_here(end + 1, text + 1);
        }
        return 0;
    }

    // 处理转义字符
    if (*pattern == '\\' && *(pattern + 1)) {
        int matched = 0;
        switch (*(pattern + 1)) {
            case 'd': matched = isdigit(*text); break;
            case 'w': matched = isalnum(*text) || *text == '_'; break;
            case 's': matched = isspace(*text); break;
            case 'D': matched = !isdigit(*text); break;
            case 'W': matched = !isalnum(*text) && *text != '_'; break;
            case 'S': matched = !isspace(*text); break;
            default: matched = (*text == *(pattern + 1)); break;
        }
        if (*text != '\0' && matched) {
            return match_here(pattern + 2, text + 1);
        }
        return 0;
    }

    // 普通字符或 . 匹配
    if (*text != '\0' && (*pattern == '.' || *pattern == *text)) {
        return match_here(pattern + 1, text + 1);
    }

    return 0;
}
```

### 2.3 入口函数

```c
/*
 * 正则表达式匹配
 * pattern: 正则表达式
 * text: 待匹配文本
 * 返回：1=匹配成功，0=匹配失败
 */
int regex_match(const char *pattern, const char *text) {
    // 处理 ^ (行首)
    if (*pattern == '^') {
        return match_here(pattern + 1, text);
    }

    // 尝试文本中每个位置
    do {
        if (match_here(pattern, text)) {
            return 1;
        }
    } while (*text++ != '\0');

    return 0;
}

/*
 * 搜索：找到第一个匹配的位置
 * 返回匹配起始位置的索引，未找到返回-1
 */
int regex_search(const char *pattern, const char *text) {
    if (*pattern == '^') {
        return match_here(pattern + 1, text) ? 0 : -1;
    }

    for (int i = 0; text[i] != '\0'; i++) {
        if (match_here(pattern, text + i)) {
            return i;
        }
    }
    return -1;
}

/*
 * 全局搜索：找到所有匹配位置
 * 返回匹配次数
 */
int regex_search_all(const char *pattern, const char *text,
                     int *positions, int max_matches) {
    int count = 0;
    int start = 0;

    while (text[start] != '\0' && count < max_matches) {
        int pos = -1;
        for (int i = start; text[i] != '\0'; i++) {
            if (match_here(pattern, text + i)) {
                pos = i;
                break;
            }
        }
        if (pos == -1) break;
        positions[count++] = pos;
        start = pos + 1;
    }

    return count;
}
```

## 3. 简易NFA引擎

```c
/*
 * NFA状态类型
 */
typedef enum {
    STATE_CHAR,     // 匹配具体字符
    STATE_ANY,      // 匹配任意字符 (.)
    STATE_SPLIT,    // 分支
    STATE_MATCH     // 匹配成功
} StateType;

typedef struct State {
    StateType type;
    char c;
    struct State *out;      // 转移目标
    struct State *out1;     // 分支转移目标
    int lastlist;           // 用于去重
} State;

/* NFA片段 */
typedef struct {
    State *start;
    State **out;  // 未连接的出口列表
    int out_count;
} Frag;

/* 状态池 */
static State state_pool[10000];
static int state_count = 0;

State *create_state(StateType type, char c) {
    State *s = &state_pool[state_count++];
    s->type = type;
    s->c = c;
    s->out = s->out1 = NULL;
    s->lastlist = 0;
    return s;
}

/*
 * 将正则表达式编译为NFA
 * 简化版本，支持 . * 和基本字符
 */
State *compile_nfa(const char *pattern) {
    state_count = 0;

    State *start = create_state(STATE_SPLIT, 0);
    State *current = start;

    for (const char *p = pattern; *p; p++) {
        State *s = NULL;

        switch (*p) {
            case '.':
                s = create_state(STATE_ANY, 0);
                break;
            case '*':
                if (current) {
                    State *split = create_state(STATE_SPLIT, 0);
                    split->out = current;
                    current = split;
                }
                continue;
            default:
                s = create_state(STATE_CHAR, *p);
                break;
        }

        if (s) {
            current->out = s;
            current = s;
        }
    }

    State *match_state = create_state(STATE_MATCH, 0);
    current->out = match_state;

    return start;
}
```

## 4. 测试

```c
int main() {
    printf("=== 简易正则表达式引擎测试 ===\n\n");

    // 测试用例
    struct { const char *pattern; const char *text; int expected; } tests[] = {
        {"hello",       "hello world",      1},
        {"hello",       "goodbye",          0},
        {"h.llo",       "hello",            1},
        {"h.llo",       "hallo",            1},
        {"^hello",      "hello world",      1},
        {"^world",      "hello world",      0},
        {"world$",      "hello world",      1},
        {"hello$",      "hello world",      0},
        {"he*lo",       "helo",             1},
        {"he*lo",       "heeeeelo",         1},
        {"he+lo",       "helo",             1},
        {"he+lo",       "hlo",              0},
        {"[abc]",       "apple",            1},
        {"[abc]",       "dog",              0},
        {"[a-z]+",      "hello123",         1},
        {"\\d+",        "abc123def",        1},
        {"\\w+",        "hello_world",      1},
    };

    int num_tests = sizeof(tests) / sizeof(tests[0]);
    int passed = 0;

    for (int i = 0; i < num_tests; i++) {
        int result = regex_match(tests[i].pattern, tests[i].text);
        int ok = (result == tests[i].expected);
        printf("[%s] pattern='%-12s' text='%-15s' result=%d expected=%d\n",
               ok ? "PASS" : "FAIL",
               tests[i].pattern, tests[i].text,
               result, tests[i].expected);
        if (ok) passed++;
    }

    printf("\n通过: %d/%d\n", passed, num_tests);

    // 搜索测试
    printf("\n=== 搜索测试 ===\n");
    const char *text = "The quick brown fox jumps over 3 lazy dogs 42 times";
    int positions[10];
    int count = regex_search_all("\\d+", text, positions, 10);
    printf("在 '%s' 中搜索 \\d+:\n", text);
    for (int i = 0; i < count; i++) {
        printf("  位置 %d\n", positions[i]);
    }

    return 0;
}
```

## 小结

- 递归回溯引擎实现简单，支持丰富的语法，但可能遇到灾难性回溯
- NFA引擎使用状态机思想，理论上更高效
- 实际正则引擎（如PCRE）结合了DFA和NFA的优点
- 生产环境应使用标准库（如POSIX regex、PCRE）
