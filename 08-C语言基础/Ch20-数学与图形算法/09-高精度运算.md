# 高精度运算

## 概述

C语言中标准整数类型有范围限制（`long long` 最大约 9.2e18）。当需要处理更大整数时，需要手动实现高精度运算。

---

## 1. 高精度整数数据结构

### 1.1 基本定义

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_DIGITS 1001  // 最大位数

typedef struct {
    int digits[MAX_DIGITS];  // digits[0] 是最低位
    int length;              // 有效位数
    int sign;                // 1=正, -1=负
} BigInt;

/* 初始化为0 */
BigInt bigint_zero() {
    BigInt n;
    memset(n.digits, 0, sizeof(n.digits));
    n.length = 1;
    n.sign = 1;
    return n;
}

/* 从字符串解析 */
BigInt bigint_from_string(const char *s) {
    BigInt n = bigint_zero();
    int len = strlen(s);
    int start = 0;

    if (s[0] == '-') {
        n.sign = -1;
        start = 1;
    } else {
        n.sign = 1;
    }

    n.length = len - start;
    for (int i = 0; i < n.length; i++) {
        n.digits[i] = s[len - 1 - i] - '0';
    }

    return n;
}

/* 从整数创建 */
BigInt bigint_from_int(long long val) {
    BigInt n = bigint_zero();
    if (val < 0) {
        n.sign = -1;
        val = -val;
    }
    if (val == 0) return n;

    n.length = 0;
    while (val > 0) {
        n.digits[n.length++] = val % 10;
        val /= 10;
    }
    return n;
}

/* 打印大整数 */
void bigint_print(const BigInt *n) {
    if (n->sign == -1) printf("-");
    for (int i = n->length - 1; i >= 0; i--) {
        printf("%d", n->digits[i]);
    }
}

/* 去除前导零 */
void bigint_normalize(BigInt *n) {
    while (n->length > 1 && n->digits[n->length - 1] == 0) {
        n->length--;
    }
    if (n->length == 1 && n->digits[0] == 0) {
        n->sign = 1;  // 0的符号为正
    }
}
```

## 2. 高精度加法

### 2.1 同号加法

```c
/* 绝对值加法（假设 a, b >= 0） */
BigInt bigint_add_abs(const BigInt *a, const BigInt *b) {
    BigInt c = bigint_zero();
    int carry = 0;
    int max_len = a->length > b->length ? a->length : b->length;

    for (int i = 0; i < max_len || carry; i++) {
        int sum = carry;
        if (i < a->length) sum += a->digits[i];
        if (i < b->length) sum += b->digits[i];
        c.digits[i] = sum % 10;
        carry = sum / 10;
    }

    c.length = max_len;
    if (carry) c.digits[c.length++] = carry;

    return c;
}
```

### 2.2 完整加法（处理符号）

```c
/* 比较绝对值大小: 返回 1(|a|>|b|), 0(|a|=|b|), -1(|a|<|b|) */
int bigint_compare_abs(const BigInt *a, const BigInt *b) {
    if (a->length != b->length) {
        return a->length > b->length ? 1 : -1;
    }
    for (int i = a->length - 1; i >= 0; i--) {
        if (a->digits[i] != b->digits[i]) {
            return a->digits[i] > b->digits[i] ? 1 : -1;
        }
    }
    return 0;
}

/* 绝对值减法（假设 |a| >= |b|） */
BigInt bigint_sub_abs(const BigInt *a, const BigInt *b) {
    BigInt c = bigint_zero();
    int borrow = 0;

    for (int i = 0; i < a->length; i++) {
        int diff = a->digits[i] - borrow;
        if (i < b->length) diff -= b->digits[i];

        if (diff < 0) {
            diff += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        c.digits[i] = diff;
    }
    c.length = a->length;
    bigint_normalize(&c);
    return c;
}

/* 高精度加法 */
BigInt bigint_add(const BigInt *a, const BigInt *b) {
    BigInt result;

    if (a->sign == b->sign) {
        result = bigint_add_abs(a, b);
        result.sign = a->sign;
    } else {
        int cmp = bigint_compare_abs(a, b);
        if (cmp >= 0) {
            result = bigint_sub_abs(a, b);
            result.sign = a->sign;
        } else {
            result = bigint_sub_abs(b, a);
            result.sign = b->sign;
        }
    }

    bigint_normalize(&result);
    return result;
}
```

## 3. 高精度乘法

### 3.1 基本乘法

```c
/* 高精度乘法 O(n*m) */
BigInt bigint_mul(const BigInt *a, const BigInt *b) {
    BigInt c = bigint_zero();
    c.length = a->length + b->length;

    for (int i = 0; i < a->length; i++) {
        int carry = 0;
        for (int j = 0; j < b->length || carry; j++) {
            long long cur = c.digits[i + j] +
                           (long long)a->digits[i] * (j < b->length ? b->digits[j] : 0) +
                           carry;
            c.digits[i + j] = cur % 10;
            carry = cur / 10;
        }
    }

    c.sign = a->sign * b->sign;
    bigint_normalize(&c);
    return c;
}
```

### 3.2 Karatsuba快速乘法

```c
/*
 * Karatsuba算法 O(n^1.585)
 * 对于非常大的整数比基本乘法快
 */
BigInt bigint_mul_karatsuba(const BigInt *a, const BigInt *b) {
    // 小规模时使用基本乘法
    if (a->length <= 20 || b->length <= 20) {
        return bigint_mul(a, b);
    }

    int m = a->length > b->length ? a->length : b->length;
    int half = (m + 1) / 2;

    // 将 a 拆分为 a1 * 10^half + a0
    // 将 b 拆分为 b1 * 10^half + b0
    BigInt a0 = bigint_zero(), a1 = bigint_zero();
    BigInt b0 = bigint_zero(), b1 = bigint_zero();

    a0.length = (half < a->length) ? half : a->length;
    for (int i = 0; i < a0.length; i++) a0.digits[i] = a->digits[i];

    a1.length = (a->length > half) ? a->length - half : 0;
    for (int i = 0; i < a1.length; i++) a1.digits[i] = a->digits[i + half];

    b0.length = (half < b->length) ? half : b->length;
    for (int i = 0; i < b0.length; i++) b0.digits[i] = b->digits[i];

    b1.length = (b->length > half) ? b->length - half : 0;
    for (int i = 0; i < b1.length; i++) b1.digits[i] = b->digits[i + half];

    bigint_normalize(&a0); bigint_normalize(&a1);
    bigint_normalize(&b0); bigint_normalize(&b1);

    // 递归: z0 = a0*b0, z2 = a1*b1, z1 = (a0+a1)(b0+b1) - z0 - z2
    BigInt z0 = bigint_mul_karatsuba(&a0, &b0);
    BigInt z2 = bigint_mul_karatsuba(&a1, &b1);

    BigInt a0a1 = bigint_add_abs(&a0, &a1);
    BigInt b0b1 = bigint_add_abs(&b0, &b1);
    BigInt z1_temp = bigint_mul_karatsuba(&a0a1, &b0b1);
    BigInt z1 = bigint_sub_abs(&z1_temp, &z0);
    z1 = bigint_sub_abs(&z1, &z2);

    // result = z2 * 10^(2*half) + z1 * 10^half + z0
    BigInt result = bigint_zero();
    result.length = z2.length + 2 * half;
    for (int i = 0; i < z2.length; i++)
        result.digits[i + 2 * half] = z2.digits[i];
    for (int i = 0; i < z1.length; i++)
        result.digits[i + half] += z1.digits[i];

    // 处理进位
    int carry = 0;
    for (int i = 0; i < result.length; i++) {
        result.digits[i] += carry;
        carry = result.digits[i] / 10;
        result.digits[i] %= 10;
    }

    // 加上z0
    for (int i = 0; i < z0.length; i++) {
        result.digits[i] += z0.digits[i];
    }
    carry = 0;
    for (int i = 0; i < result.length; i++) {
        result.digits[i] += carry;
        carry = result.digits[i] / 10;
        result.digits[i] %= 10;
    }
    if (carry) result.digits[result.length++] = carry;

    result.sign = a->sign * b->sign;
    bigint_normalize(&result);
    return result;
}
```

## 4. 高精度除法

```c
/* 高精度除以单精度整数 */
BigInt bigint_div_int(const BigInt *a, int b, int *remainder) {
    BigInt c = bigint_zero();
    long long rem = 0;

    for (int i = a->length - 1; i >= 0; i--) {
        rem = rem * 10 + a->digits[i];
        c.digits[i] = rem / b;
        rem %= b;
    }

    c.length = a->length;
    c.sign = a->sign * (b > 0 ? 1 : -1);
    bigint_normalize(&c);

    if (remainder) *remainder = (int)rem;
    return c;
}

/* 高精度除法（二分搜索法） */
BigInt bigint_div(const BigInt *a, const BigInt *b, BigInt *remainder) {
    // 简化版本: 使用二分搜索求商
    BigInt lo = bigint_from_int(0);
    BigInt hi = bigint_from_int(1);

    // 找到上界
    while (bigint_compare_abs(&hi, a) <= 0) {
        BigInt temp = bigint_mul(&hi, b);
        if (bigint_compare_abs(&temp, a) > 0) break;
        hi = bigint_add_abs(&hi, &hi);  // hi *= 2
    }

    // 二分搜索
    BigInt one = bigint_from_int(1);
    while (bigint_compare_abs(&lo, &hi) < 0) {
        BigInt mid = bigint_add_abs(&lo, &hi);
        // mid = (lo + hi + 1) / 2
        int carry = 1;
        for (int i = 0; i < mid.length; i++) {
            mid.digits[i] += carry;
            carry = mid.digits[i] / 10;
            mid.digits[i] %= 10;
        }
        if (carry) mid.digits[mid.length++] = carry;

        // 除以2
        int rem = 0;
        for (int i = mid.length - 1; i >= 0; i--) {
            rem = rem * 10 + mid.digits[i];
            mid.digits[i] = rem / 2;
            rem %= 2;
        }
        bigint_normalize(&mid);

        BigInt prod = bigint_mul(&mid, b);
        if (bigint_compare_abs(&prod, a) <= 0) {
            lo = mid;
        } else {
            hi = bigint_sub_abs(&mid, &one);
        }
    }

    // 计算余数
    if (remainder) {
        BigInt prod = bigint_mul(&lo, b);
        *remainder = bigint_sub_abs(a, &prod);
    }

    lo.sign = a->sign * b->sign;
    bigint_normalize(&lo);
    return lo;
}
```

## 5. 测试与使用

```c
int main() {
    BigInt a = bigint_from_string("12345678901234567890");
    BigInt b = bigint_from_string("98765432109876543210");

    printf("a = "); bigint_print(&a); printf("\n");
    printf("b = "); bigint_print(&b); printf("\n");

    BigInt sum = bigint_add(&a, &b);
    printf("a + b = "); bigint_print(&sum); printf("\n");

    BigInt prod = bigint_mul(&a, &b);
    printf("a * b = "); bigint_print(&prod); printf("\n");

    // 计算阶乘
    printf("\n50! = ");
    BigInt fact = bigint_from_int(1);
    for (int i = 2; i <= 50; i++) {
        BigInt temp = bigint_from_int(i);
        fact = bigint_mul(&fact, &temp);
    }
    bigint_print(&fact); printf("\n");

    return 0;
}
```

---

## 小结

- 高精度运算的核心是将大数按位存储在数组中，逐位运算
- 加法和乘法的实现相对简单，除法较为复杂
- Karatsuba算法在大规模乘法中比朴素方法快
- 实际工程中可使用GMP等成熟库处理高精度运算
