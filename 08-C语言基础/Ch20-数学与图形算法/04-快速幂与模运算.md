# 快速幂与模运算

## 概述

快速幂是一种高效计算幂运算的算法，将时间复杂度从 O(n) 降低到 O(log n)。模运算是数论和密码学的基础。本节讲解快速幂、矩阵快速幂和模逆元。

---

## 1. 快速幂

### 1.1 基本原理

将指数按二进制分解：`a^n = a^(b0*1 + b1*2 + b2*4 + ...)`。

例如 `a^13 = a^(1101_2) = a^8 * a^4 * a^1`。

```c
#include <stdio.h>

/* 快速幂: 计算 base^exp */
long long fast_power(long long base, long long exp) {
    long long result = 1;
    while (exp > 0) {
        if (exp & 1) {        // 当前位为1
            result *= base;
        }
        base *= base;         // 底数平方
        exp >>= 1;            // 指数右移
    }
    return result;
}

int main() {
    printf("2^10 = %lld\n", fast_power(2, 10));    // 1024
    printf("3^5  = %lld\n", fast_power(3, 5));     // 243
    printf("5^0  = %lld\n", fast_power(5, 0));     // 1
    return 0;
}
```

### 1.2 快速幂取模

在计算 `(base^exp) % mod` 时，利用模运算性质防止溢出。

```c
#include <stdio.h>

/*
 * 快速幂取模
 * 计算 (base^exp) % mod
 * 使用 __int128 防止乘法溢出
 */
long long power_mod(long long base, long long exp, long long mod) {
    long long result = 1;
    base %= mod;

    while (exp > 0) {
        if (exp & 1) {
            result = (__int128)result * base % mod;
        }
        base = (__int128)base * base % mod;
        exp >>= 1;
    }
    return result;
}

/* 不支持 __int128 时的溢出安全乘法 */
long long mul_mod(long long a, long long b, long long mod) {
    long long result = 0;
    a %= mod;
    while (b > 0) {
        if (b & 1) {
            result = (result + a) % mod;
        }
        a = (a + a) % mod;
        b >>= 1;
    }
    return result;
}

long long power_mod_safe(long long base, long long exp, long long mod) {
    long long result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) {
            result = mul_mod(result, base, mod);
        }
        base = mul_mod(base, base, mod);
        exp >>= 1;
    }
    return result;
}

int main() {
    long long mod = 1000000007LL;

    // 计算 2^1000000000 mod 10^9+7
    printf("2^1000000000 mod 10^9+7 = %lld\n",
           power_mod(2, 1000000000LL, mod));

    // 验证费马小定理: a^(p-1) ≡ 1 (mod p), p为素数
    long long a = 123456789;
    printf("a^(p-1) mod p = %lld\n", power_mod(a, mod - 1, mod)); // 1

    return 0;
}
```

## 2. 矩阵快速幂

### 2.1 矩阵乘法

```c
#include <stdio.h>
#include <string.h>

#define MOD 1000000007LL
#define MAXN 4

typedef long long Matrix[MAXN][MAXN];

/* 矩阵乘法 C = A * B */
void mat_mul(Matrix A, Matrix B, Matrix C, int n) {
    Matrix temp;
    memset(temp, 0, sizeof(temp));

    for (int i = 0; i < n; i++) {
        for (int k = 0; k < n; k++) {
            if (A[i][k] == 0) continue;
            for (int j = 0; j < n; j++) {
                temp[i][j] = (temp[i][j] + A[i][k] * B[k][j]) % MOD;
            }
        }
    }
    memcpy(C, temp, sizeof(Matrix));
}

/* 矩阵快速幂: 计算 A^exp */
void mat_power(Matrix A, long long exp, Matrix result, int n) {
    // 初始化为单位矩阵
    memset(result, 0, sizeof(Matrix));
    for (int i = 0; i < n; i++) {
        result[i][i] = 1;
    }

    Matrix base;
    memcpy(base, A, sizeof(Matrix));

    while (exp > 0) {
        if (exp & 1) {
            mat_mul(result, base, result, n);
        }
        mat_mul(base, base, base, n);
        exp >>= 1;
    }
}

/* 打印矩阵 */
void print_matrix(Matrix M, int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            printf("%lld ", M[i][j]);
        }
        printf("\n");
    }
}
```

### 2.2 用矩阵快速幂求斐波那契数

```c
/*
 * 求第n个斐波那契数 mod 10^9+7
 * 利用矩阵:
 * | F(n+1) |   | 1 1 |^n   | 1 |
 * | F(n)   | = | 1 0 |   * | 0 |
 */
long long fibonacci(long long n) {
    if (n <= 0) return 0;
    if (n == 1) return 1;

    Matrix A = {{1, 1}, {1, 0}};
    Matrix result;

    mat_power(A, n - 1, result, 2);
    return result[0][0];
}

int main() {
    printf("F(10) = %lld\n", fibonacci(10));       // 55
    printf("F(50) = %lld\n", fibonacci(50));       // 12586269025 mod MOD
    printf("F(100) = %lld\n", fibonacci(100));     // 非常大的数

    // 用矩阵快速幂可以在 O(log n) 时间内计算
    printf("F(1000000000) = %lld\n", fibonacci(1000000000LL));

    return 0;
}
```

### 2.3 矩阵快速幂解递推关系

对于任意线性递推关系 `f(n) = a1*f(n-1) + a2*f(n-2) + ...`，都可以构造转移矩阵。

```c
/*
 * 求解 f(n) = 3*f(n-1) + 2*f(n-2)
 * 初始: f(0) = 0, f(1) = 1
 *
 * 转移矩阵:
 * | f(n+1) |   | 3 2 |^n   | f(1) |
 * | f(n)   | = | 1 0 |   * | f(0) |
 */
long long solve_recurrence(long long n) {
    if (n == 0) return 0;
    if (n == 1) return 1;

    Matrix A = {{3, 2}, {1, 0}};
    Matrix result;
    mat_power(A, n - 1, result, 2);

    // result[0][0] * f(1) + result[0][1] * f(0)
    return result[0][0];
}
```

## 3. 模逆元

### 3.1 费马小定理求逆元

当 `mod` 是素数时：`a^(-1) ≡ a^(mod-2) (mod mod)`。

```c
/* 用费马小定理求模逆元（mod必须是素数） */
long long mod_inverse_fermat(long long a, long long mod) {
    return power_mod(a, mod - 2, mod);
}
```

### 3.2 扩展GCD求逆元

适用于 `mod` 不是素数的情况。

```c
long long exgcd(long long a, long long b, long long *x, long long *y) {
    if (b == 0) {
        *x = 1;
        *y = 0;
        return a;
    }
    long long x1, y1;
    long long g = exgcd(b, a % b, &x1, &y1);
    *x = y1;
    *y = x1 - (a / b) * y1;
    return g;
}

/* 用扩展GCD求模逆元 */
long long mod_inverse_exgcd(long long a, long long mod) {
    long long x, y;
    long long g = exgcd(a, mod, &x, &y);
    if (g != 1) return -1;  // 逆元不存在
    return (x % mod + mod) % mod;
}
```

### 3.3 线性求逆元（批量）

在O(n)时间内求1到n所有数的逆元。

```c
#define MAXN 1000001

long long inv[MAXN];

/* 线性预处理逆元: inv[i] 为 i 在模 mod 下的逆元 */
void precompute_inv(int n, long long mod) {
    inv[1] = 1;
    for (int i = 2; i <= n; i++) {
        inv[i] = (mod - mod / i) * inv[mod % i] % mod;
    }
}

int main() {
    long long mod = 1000000007LL;
    precompute_inv(1000000, mod);

    // 验证
    printf("inv(3) = %lld\n", inv[3]);  // 333333336
    printf("验证: 3 * %lld mod 10^9+7 = %lld\n",
           inv[3], (3 * inv[3]) % mod);  // 应为1

    return 0;
}
```

## 4. 综合应用：组合数取模

```c
#define MAXN 200001
long long fact[MAXN], inv_fact[MAXN];
long long MOD = 1000000007LL;

/* 预处理阶乘和阶乘逆元 */
void precompute_fact(int n) {
    fact[0] = 1;
    for (int i = 1; i <= n; i++) {
        fact[i] = fact[i - 1] * i % MOD;
    }
    inv_fact[n] = power_mod(fact[n], MOD - 2, MOD);
    for (int i = n - 1; i >= 0; i--) {
        inv_fact[i] = inv_fact[i + 1] * (i + 1) % MOD;
    }
}

/* 组合数 C(n, k) mod MOD */
long long comb(int n, int k) {
    if (k < 0 || k > n) return 0;
    return fact[n] % MOD * inv_fact[k] % MOD * inv_fact[n - k] % MOD;
}

int main() {
    precompute_fact(200000);

    printf("C(10,3) mod 10^9+7 = %lld\n", comb(10, 3));       // 120
    printf("C(100,50) mod 10^9+7 = %lld\n", comb(100, 50));

    return 0;
}
```

---

## 小结

- 快速幂将幂运算的时间复杂度从 O(n) 降到 O(log n)
- 矩阵快速幂可以高效求解线性递推关系
- 模逆元的三种求法：费马小定理(需素数模)、扩展GCD(通用)、线性递推(批量)
- 快速幂取模是密码学和竞赛编程的基础工具
