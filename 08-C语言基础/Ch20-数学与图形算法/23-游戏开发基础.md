# 游戏开发基础

## 概述

使用C语言开发控制台游戏需要处理屏幕输出、键盘输入和游戏循环。本节构建一个通用的控制台游戏框架。

## 1. 控制台屏幕管理

### 1.1 屏幕缓冲区

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#ifdef _WIN32
#include <windows.h>
#include <conio.h>
#define CLEAR_SCREEN system("cls")
#define SLEEP_MS(ms) Sleep(ms)
#define KBHIT _kbhit()
#define GETCH _getch()
#else
#include <unistd.h>
#include <termios.h>
#include <fcntl.h>
#define CLEAR_SCREEN system("clear")
#define SLEEP_MS(ms) usleep((ms) * 1000)

int kbhit(void) {
    struct termios oldt, newt;
    int ch, oldf;
    tcgetattr(STDIN_FILENO, &oldt);
    newt = oldt;
    newt.c_lflag &= ~(ICANON | ECHO);
    tcsetattr(STDIN_FILENO, TCSANOW, &newt);
    oldf = fcntl(STDIN_FILENO, F_GETFL, 0);
    fcntl(STDIN_FILENO, F_SETFL, oldf | O_NONBLOCK);
    ch = getchar();
    tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
    fcntl(STDIN_FILENO, F_SETFL, oldf);
    if (ch != EOF) { ungetch(ch, stdin); return 1; }
    return 0;
}
#define KBHIT kbhit()
#define GETCH getchar()
#endif

/* 屏幕配置 */
#define SCREEN_W 80
#define SCREEN_H 25

/* 屏幕缓冲区 */
static char screen_buf[SCREEN_H][SCREEN_W + 1];
static char screen_prev[SCREEN_H][SCREEN_W + 1];

/* 初始化屏幕 */
void screen_init() {
    for (int y = 0; y < SCREEN_H; y++) {
        memset(screen_buf[y], ' ', SCREEN_W);
        screen_buf[y][SCREEN_W] = '\0';
        memset(screen_prev[y], '\0', SCREEN_W + 1);
    }
}

/* 清空缓冲区 */
void screen_clear() {
    for (int y = 0; y < SCREEN_H; y++)
        memset(screen_buf[y], ' ', SCREEN_W);
}

/* 设置字符 */
void screen_put(int x, int y, char ch) {
    if (x >= 0 && x < SCREEN_W && y >= 0 && y < SCREEN_H)
        screen_buf[y][x] = ch;
}

/* 绘制字符串 */
void screen_print(int x, int y, const char *str) {
    for (int i = 0; str[i] && x + i < SCREEN_W; i++) {
        screen_put(x + i, y, str[i]);
    }
}

/* 刷新屏幕（双缓冲，只更新变化的部分） */
void screen_flush() {
    for (int y = 0; y < SCREEN_H; y++) {
        if (memcmp(screen_buf[y], screen_prev[y], SCREEN_W) != 0) {
            // 定位光标到行首
            printf("\033[%d;1H", y + 1);
            printf("%s", screen_buf[y]);
            memcpy(screen_prev[y], screen_buf[y], SCREEN_W);
        }
    }
    fflush(stdout);
}
```

## 2. 键盘输入处理

```c
/* 键码定义 */
enum KeyCode {
    KEY_NONE = 0,
    KEY_UP = 256,
    KEY_DOWN,
    KEY_LEFT,
    KEY_RIGHT,
    KEY_SPACE,
    KEY_ENTER,
    KEY_ESCAPE,
    KEY_W, KEY_A, KEY_S, KEY_D,
    KEY_Q
};

/* 读取按键（非阻塞） */
int get_key() {
    if (!KBHIT) return KEY_NONE;

    int ch = GETCH;

#ifdef _WIN32
    if (ch == 0 || ch == 224) {
        int ext = GETCH;
        switch (ext) {
            case 72: return KEY_UP;
            case 80: return KEY_DOWN;
            case 75: return KEY_LEFT;
            case 77: return KEY_RIGHT;
        }
    }
#else
    if (ch == 27) {  // ESC序列
        if (KBHIT) {
            GETCH;  // 跳过 '['
            int dir = GETCH;
            switch (dir) {
                case 'A': return KEY_UP;
                case 'B': return KEY_DOWN;
                case 'D': return KEY_LEFT;
                case 'C': return KEY_RIGHT;
            }
        }
        return KEY_ESCAPE;
    }
#endif

    switch (ch) {
        case 'w': case 'W': return KEY_W;
        case 'a': case 'A': return KEY_A;
        case 's': case 'S': return KEY_S;
        case 'd': case 'D': return KEY_D;
        case 'q': case 'Q': return KEY_Q;
        case ' ': return KEY_SPACE;
        case 13: case 10: return KEY_ENTER;
        case 27: return KEY_ESCAPE;
        default: return ch;
    }
}
```

## 3. 游戏循环框架

### 3.1 帧率控制

```c
/* 帧率控制 */
typedef struct {
    int target_fps;
    clock_t frame_start;
    double delta_time;
    int fps;
    int frame_count;
    clock_t fps_timer;
} GameTimer;

void timer_init(GameTimer *timer, int fps) {
    timer->target_fps = fps;
    timer->frame_start = clock();
    timer->delta_time = 0;
    timer->fps = 0;
    timer->frame_count = 0;
    timer->fps_timer = clock();
}

void timer_begin_frame(GameTimer *timer) {
    timer->frame_start = clock();
}

void timer_end_frame(GameTimer *timer) {
    clock_t frame_end = clock();
    double elapsed = (double)(frame_end - timer->frame_start) / CLOCKS_PER_SEC;
    double target = 1.0 / timer->target_fps;

    if (elapsed < target) {
        SLEEP_MS((int)((target - elapsed) * 1000));
    }

    timer->delta_time = elapsed;
    timer->frame_count++;

    // 每秒更新FPS
    if ((double)(frame_end - timer->fps_timer) / CLOCKS_PER_SEC >= 1.0) {
        timer->fps = timer->frame_count;
        timer->frame_count = 0;
        timer->fps_timer = frame_end;
    }
}
```

### 3.2 游戏主循环模板

```c
/*
 * 游戏主循环模板
 * void (*init)(void):     初始化回调
 * void (*update)(double): 更新回调，参数为delta_time
 * void (*render)(void):   渲染回调
 * int  (*running)(void):  运行状态检查回调
 */
typedef struct {
    void (*init)(void);
    void (*update)(double dt);
    void (*render)(void);
    int  (*running)(void);
    int fps;
} GameConfig;

void game_run(GameConfig *config) {
    screen_init();
    CLEAR_SCREEN;
    printf("\033[?25l");  // 隐藏光标

    GameTimer timer;
    timer_init(&timer, config->fps);

    if (config->init) config->init();

    while (!config->running || config->running()) {
        timer_begin_frame(&timer);

        if (config->update) config->update(timer.delta_time);
        if (config->render) {
            config->render();
            screen_flush();
        }

        timer_end_frame(&timer);
    }

    printf("\033[?25h");  // 显示光标
    CLEAR_SCREEN;
}
```

## 4. 精灵与动画

```c
/* 精灵结构 */
typedef struct {
    int x, y;
    int width, height;
    const char **frames;   // 多帧动画
    int frame_count;
    int current_frame;
    double frame_timer;
    double frame_duration;
    int visible;
} Sprite;

/* 创建精灵 */
Sprite sprite_create(int x, int y, const char **frames,
                     int frame_count, double fps) {
    Sprite s;
    s.x = x; s.y = y;
    s.frames = frames;
    s.frame_count = frame_count;
    s.current_frame = 0;
    s.frame_timer = 0;
    s.frame_duration = 1.0 / fps;
    s.visible = 1;

    // 计算宽高
    s.width = (int)strlen(frames[0]);
    s.height = 1;
    for (int i = 0; frames[0][i]; i++) {
        if (frames[0][i] == '\n') s.height++;
    }

    return s;
}

/* 更新精灵动画 */
void sprite_update(Sprite *s, double dt) {
    if (!s->visible || s->frame_count <= 1) return;

    s->frame_timer += dt;
    if (s->frame_timer >= s->frame_duration) {
        s->frame_timer -= s->frame_duration;
        s->current_frame = (s->current_frame + 1) % s->frame_count;
    }
}

/* 绘制精灵到屏幕 */
void sprite_draw(Sprite *s) {
    if (!s->visible) return;

    const char *frame = s->frames[s->current_frame];
    int x = s->x, y = s->y;

    for (int i = 0; frame[i]; i++) {
        if (frame[i] == '\n') {
            x = s->x;
            y++;
        } else {
            if (frame[i] != ' ') screen_put(x, y, frame[i]);
            x++;
        }
    }
}

/* 碰撞检测（矩形） */
int sprite_collide(Sprite *a, Sprite *b) {
    return (a->x < b->x + b->width &&
            a->x + a->width > b->x &&
            a->y < b->b->height &&
            a->y + a->height > b->y);
}
```

## 5. 粒子系统

```c
#define MAX_PARTICLES 100

typedef struct {
    float x, y;
    float vx, vy;
    char ch;
    int life;
    int active;
} Particle;

typedef struct {
    Particle particles[MAX_PARTICLES];
    int count;
} ParticleSystem;

void particle_system_init(ParticleSystem *ps) {
    memset(ps->particles, 0, sizeof(ps->particles));
    ps->count = 0;
}

void particle_emit(ParticleSystem *ps, float x, float y, char ch, int life) {
    for (int i = 0; i < MAX_PARTICLES; i++) {
        if (!ps->particles[i].active) {
            ps->particles[i].x = x;
            ps->particles[i].y = y;
            ps->particles[i].vx = ((rand() % 100) - 50) / 50.0f;
            ps->particles[i].vy = ((rand() % 100) - 50) / 50.0f;
            ps->particles[i].ch = ch;
            ps->particles[i].life = life;
            ps->particles[i].active = 1;
            break;
        }
    }
}

void particle_update(ParticleSystem *ps, double dt) {
    for (int i = 0; i < MAX_PARTICLES; i++) {
        if (!ps->particles[i].active) continue;

        Particle *p = &ps->particles[i];
        p->x += p->vx * dt * 20;
        p->y += p->vy * dt * 20;
        p->life--;

        if (p->life <= 0) p->active = 0;
    }
}

void particle_draw(ParticleSystem *ps) {
    for (int i = 0; i < MAX_PARTICLES; i++) {
        if (ps->particles[i].active) {
            screen_put((int)ps->particles[i].x,
                       (int)ps->particles[i].y,
                       ps->particles[i].ch);
        }
    }
}
```

## 6. 测试：弹跳小球

```c
static int ball_x = 40, ball_y = 12;
static int ball_dx = 1, ball_dy = 1;
static int game_quit = 0;

void game_init() { game_quit = 0; }

int game_is_running() { return !game_quit; }

void game_update(double dt) {
    int key = get_key();
    if (key == KEY_ESCAPE || key == KEY_Q) game_quit = 1;

    ball_x += ball_dx;
    ball_y += ball_dy;

    if (ball_x <= 0 || ball_x >= SCREEN_W - 1) ball_dx = -ball_dx;
    if (ball_y <= 0 || ball_y >= SCREEN_H - 1) ball_dy = -ball_dy;
}

void game_render() {
    screen_clear();
    screen_put(ball_x, ball_y, 'O');

    char info[64];
    snprintf(info, sizeof(info), "Press Q or ESC to quit | Pos: (%d,%d)",
             ball_x, ball_y);
    screen_print(0, 0, info);
}

int main() {
    GameConfig config = {
        .init = game_init,
        .update = game_update,
        .render = game_render,
        .running = game_is_running,
        .fps = 30
    };

    game_run(&config);
    printf("游戏结束\n");

    return 0;
}
```

## 小结

- 双缓冲技术避免屏幕闪烁：先在缓冲区绘制，再一次性刷新
- 帧率控制通过计算每帧耗时并sleep差值来实现
- 游戏循环的三要素：输入处理、状态更新、渲染绘制
- 跨平台需要分别处理Windows和Unix的控制台API差异
