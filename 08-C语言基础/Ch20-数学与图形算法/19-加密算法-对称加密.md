# 加密算法 - 对称加密

## 概述

对称加密使用同一密钥进行加密和解密。本节实现AES和DES的核心逻辑，展示分组密码的工作原理。

---

## 1. 基本概念

对称加密分为：
- **分组密码**：将明文分成固定大小的块，逐块加密（AES、DES）
- **流密码**：逐字节/位加密（RC4、ChaCha20）

常用工作模式：ECB、CBC、CTR、CFB、OFB

### 2.2 位操作辅助函数

```c
/* 将字节数组转换为64位 */
static uint64_t bytes_to_u64(const uint8_t *bytes) {
    uint64_t result = 0;
    for (int i = 0; i < 8; i++) {
        result = (result << 8) | bytes[i];
    }
    return result;
}

/* 将64位转换为字节数组 */
static void u64_to_bytes(uint64_t val, uint8_t *bytes) {
    for (int i = 7; i >= 0; i--) {
        bytes[i] = val & 0xFF;
        val >>= 8;
    }
}

/* 置换操作 */
static uint64_t permute(uint64_t input, const int *table, int n) {
    uint64_t result = 0;
    for (int i = 0; i < n; i++) {
        int bit_pos = table[i];
        uint64_t bit = (input >> (64 - bit_pos)) & 1;
        result = (result << 1) | bit;
    }
    return result;
}
```

### 2.3 DES加密核心

```c
/*
 * DES加密一个64位块
 * plaintext: 8字节明文
 * key: 8字节密钥
 * ciphertext: 8字节密文输出
 */
void des_encrypt_block(const uint8_t *plaintext, const uint8_t *key,
                       uint8_t *ciphertext) {
    uint64_t data = bytes_to_u64(plaintext);
    uint64_t k = bytes_to_u64(key);

    // 初始置换IP
    data = permute(data, IP, 64);

    // 拆分为左右各32位
    uint32_t L = (uint32_t)(data >> 32);
    uint32_t R = (uint32_t)(data & 0xFFFFFFFF);

    // 16轮Feistel结构
    for (int round = 0; round < 16; round++) {
        uint32_t old_R = R;

        // 扩展置换 E: 32位 -> 48位
        uint64_t expanded = 0;
        for (int i = 0; i < 48; i++) {
            int bit_pos = E[i];
            uint64_t bit = ((uint64_t)R >> (32 - bit_pos)) & 1;
            expanded = (expanded << 1) | bit;
        }

        // 与子密钥异或（简化版：使用原始密钥的移位）
        uint64_t subkey = (k >> round) | (k << (64 - round));
        subkey &= 0x0000FFFFFFFFFFFF;  // 48位
        expanded ^= subkey;

        // S盒替换: 48位 -> 32位
        uint32_t sbox_out = 0;
        for (int i = 0; i < 8; i++) {
            int chunk = (expanded >> (42 - i * 6)) & 0x3F;
            int row = ((chunk >> 5) & 1) | ((chunk & 1) << 1);
            int col = (chunk >> 1) & 0xF;
            sbox_out = (sbox_out << 4) | S_BOX[i][row][col];
        }

        // P盒置换
        uint32_t p_out = 0;
        for (int i = 0; i < 32; i++) {
            int bit_pos = P[i];
            uint32_t bit = (sbox_out >> (32 - bit_pos)) & 1;
            p_out = (p_out << 1) | bit;
        }

        // L和R交换
        R = L ^ p_out;
        L = old_R;
    }

    // 最终合并（注意最后一轮不交换）
    uint64_t combined = ((uint64_t)R << 32) | L;

    // 逆初始置换
    combined = permute(combined, IP_INV, 64);

    u64_to_bytes(combined, ciphertext);
}

/* DES解密（加密的逆过程） */
void des_decrypt_block(const uint8_t *ciphertext, const uint8_t *key,
                       uint8_t *plaintext) {
    // DES解密与加密相同，只是子密钥顺序相反
    // 这里简化实现
    des_encrypt_block(ciphertext, key, plaintext);
}
```

```c
/* AES轮常量 */
static const uint8_t RCON[10] = {
    0x01,0x02,0x04,0x08,0x10,0x20,0x40,0x80,0x1b,0x36
};

/* GF(2^8)上的乘法 */
static uint8_t gf_mul(uint8_t a, uint8_t b) {
    uint8_t result = 0;
    while (b) {
        if (b & 1) result ^= a;
        a = (a << 1) ^ (a & 0x80 ? 0x1B : 0);
        b >>= 1;
    }
    return result;
}

/* SubBytes: 用S盒替换每个字节 */
static void sub_bytes(uint8_t state[4][4]) {
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 4; j++)
            state[i][j] = AES_SBOX[state[i][j]];
}

/* ShiftRows: 行移位 */
static void shift_rows(uint8_t state[4][4]) {
    uint8_t temp;
    // 第2行左移1位
    temp = state[1][0];
    state[1][0] = state[1][1]; state[1][1] = state[1][2];
    state[1][2] = state[1][3]; state[1][3] = temp;
    // 第3行左移2位
    temp = state[2][0]; state[2][0] = state[2][2]; state[2][2] = temp;
    temp = state[2][1]; state[2][1] = state[2][3]; state[2][3] = temp;
    // 第4行左移3位
    temp = state[3][3];
    state[3][3] = state[3][2]; state[3][2] = state[3][1];
    state[3][1] = state[3][0]; state[3][0] = temp;
}

/* MixColumns: 列混合 */
static void mix_columns(uint8_t state[4][4]) {
    for (int j = 0; j < 4; j++) {
        uint8_t a0 = state[0][j], a1 = state[1][j];
        uint8_t a2 = state[2][j], a3 = state[3][j];
        state[0][j] = gf_mul(a0,2)^gf_mul(a1,3)^a2^a3;
        state[1][j] = a0^gf_mul(a1,2)^gf_mul(a2,3)^a3;
        state[2][j] = a0^a1^gf_mul(a2,2)^gf_mul(a3,3);
        state[3][j] = gf_mul(a0,3)^a1^a2^gf_mul(a3,2);
    }
}

/* AddRoundKey: 轮密钥加 */
static void add_round_key(uint8_t state[4][4], const uint8_t *round_key) {
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 4; j++)
            state[j][i] ^= round_key[i * 4 + j];
}
```

### 3.2 AES密钥扩展与加密

```c
/* AES-128密钥扩展 */
void aes128_key_expansion(const uint8_t key[16], uint8_t round_keys[176]) {
    // 前16字节就是原始密钥
    memcpy(round_keys, key, 16);

    for (int i = 4; i < 44; i++) {
        uint8_t temp[4];
        memcpy(temp, &round_keys[(i - 1) * 4], 4);

        if (i % 4 == 0) {
            // RotWord
            uint8_t t = temp[0];
            temp[0] = temp[1]; temp[1] = temp[2];
            temp[2] = temp[3]; temp[3] = t;
            // SubWord
            for (int j = 0; j < 4; j++) temp[j] = AES_SBOX[temp[j]];
            temp[0] ^= RCON[i / 4 - 1];
        }

        for (int j = 0; j < 4; j++) {
            round_keys[i * 4 + j] = round_keys[(i - 4) * 4 + j] ^ temp[j];
        }
    }
}

/*
 * AES-128加密一个16字节块
 * plaintext: 16字节明文
 * key: 16字节密钥
 * ciphertext: 16字节密文输出
 */
void aes128_encrypt_block(const uint8_t *plaintext, const uint8_t *key,
                          uint8_t *ciphertext) {
    uint8_t round_keys[176];
    aes128_key_expansion(key, round_keys);

    uint8_t state[4][4];
    // 将明文按列优先填入状态矩阵
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 4; j++)
            state[j][i] = plaintext[i * 4 + j];

    // 初始轮密钥加
    add_round_key(state, round_keys);

    // 9个完整轮
    for (int round = 1; round <= 9; round++) {
        sub_bytes(state);
        shift_rows(state);
        mix_columns(state);
        add_round_key(state, round_keys + round * 16);
    }

    // 最后一轮（无MixColumns）
    sub_bytes(state);
    shift_rows(state);
    add_round_key(state, round_keys + 160);

    // 输出密文
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 4; j++)
            ciphertext[i * 4 + j] = state[j][i];
}
```

## 4. CBC工作模式

```c
/*
 * AES-128 CBC模式加密
 * 需要初始化向量IV（16字节）
 */
void aes128_cbc_encrypt(const uint8_t *plaintext, int len,
                        const uint8_t *key, const uint8_t *iv,
                        uint8_t *ciphertext) {
    uint8_t prev[16];
    memcpy(prev, iv, 16);

    for (int i = 0; i < len; i += 16) {
        uint8_t block[16];
        // 明文与前一个密文块（或IV）异或
        for (int j = 0; j < 16; j++) {
            block[j] = plaintext[i + j] ^ prev[j];
        }
        aes128_encrypt_block(block, key, ciphertext + i);
        memcpy(prev, ciphertext + i, 16);
    }
}

/* AES-128 CBC模式解密 */
void aes128_cbc_decrypt(const uint8_t *ciphertext, int len,
                        const uint8_t *key, const uint8_t *iv,
                        uint8_t *plaintext) {
    uint8_t prev[16];
    memcpy(prev, iv, 16);

    for (int i = 0; i < len; i += 16) {
        uint8_t block[16];
        aes128_decrypt_block(ciphertext + i, key, block);
        for (int j = 0; j < 16; j++) {
            plaintext[i + j] = block[j] ^ prev[j];
        }
        memcpy(prev, ciphertext + i, 16);
    }
}
```

## 小结

- DES使用56位密钥（64位中8位奇偶校验），16轮Feistel结构
- AES-128使用128位密钥，10轮替换-置换网络
- ECB模式简单但不安全（相同明文产生相同密文），推荐使用CBC或CTR模式
- 实际生产中应使用成熟的加密库（如OpenSSL），自行实现仅供学习
