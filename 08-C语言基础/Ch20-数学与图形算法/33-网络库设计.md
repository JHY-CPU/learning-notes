# 网络库设计

## 概述

网络通信库封装底层socket操作，提供简洁的API。本节设计一个简易的TCP网络通信库，支持服务器和客户端模式。

## 1. 跨平台Socket封装

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#ifdef _WIN32
#include <winsock2.h>
#include <ws2tcpip.h>
#pragma comment(lib, "ws2_32.lib")
typedef SOCKET socket_t;
#define INVALID_SOCK INVALID_SOCKET
#define SOCK_ERROR SOCKET_ERROR
#define CLOSE_SOCKET closesocket
static int net_initialized = 0;

int net_init() {
    if (net_initialized) return 0;
    WSADATA wsa;
    if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0) return -1;
    net_initialized = 1;
    return 0;
}

void net_cleanup() {
    if (net_initialized) {
        WSACleanup();
        net_initialized = 0;
    }
}
#else
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <errno.h>
#include <fcntl.h>
typedef int socket_t;
#define INVALID_SOCK (-1)
#define SOCK_ERROR (-1)
#define CLOSE_SOCKET close
#define net_init() 0
#define net_cleanup()
#endif

/* 关闭连接 */
void net_close(socket_t sock) {
    if (sock != INVALID_SOCK) {
        CLOSE_SOCKET(sock);
    }
}
```

## 2. 服务器端API

### 2.1 服务器创建

```c
/* 服务器结构 */
typedef struct {
    socket_t listen_sock;
    int port;
    int running;
} NetServer;

/*
 * 创建并启动服务器
 * 返回: 0=成功, -1=失败
 */
NetServer *server_create(int port, int backlog) {
    net_init();

    NetServer *server = (NetServer *)calloc(1, sizeof(NetServer));
    server->port = port;

    // 创建socket
    server->listen_sock = socket(AF_INET, SOCK_STREAM, 0);
    if (server->listen_sock == INVALID_SOCK) {
        fprintf(stderr, "创建socket失败\n");
        free(server);
        return NULL;
    }

    // 设置SO_REUSEADDR
    int opt = 1;
    setsockopt(server->listen_sock, SOL_SOCKET, SO_REUSEADDR,
               (const char *)&opt, sizeof(opt));

    // 绑定地址
    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);

    if (bind(server->listen_sock, (struct sockaddr *)&addr, sizeof(addr)) == SOCK_ERROR) {
        fprintf(stderr, "绑定端口 %d 失败\n", port);
        net_close(server->listen_sock);
        free(server);
        return NULL;
    }

    // 开始监听
    if (listen(server->listen_sock, backlog) == SOCK_ERROR) {
        fprintf(stderr, "监听失败\n");
        net_close(server->listen_sock);
        free(server);
        return NULL;
    }

    server->running = 1;
    printf("服务器启动: 端口 %d\n", port);
    return server;
}
```

### 2.2 接受连接

```c
/*
 * 接受一个客户端连接（阻塞）
 * client_ip: 输出客户端IP地址
 * 返回: 客户端socket
 */
socket_t server_accept(NetServer *server, char *client_ip, int ip_len) {
    struct sockaddr_in client_addr;
    socklen_t addr_len = sizeof(client_addr);

    socket_t client = accept(server->listen_sock,
                             (struct sockaddr *)&client_addr, &addr_len);
    if (client == INVALID_SOCK) {
        return INVALID_SOCK;
    }

    if (client_ip) {
        strncpy(client_ip, inet_ntoa(client_addr.sin_addr), ip_len - 1);
    }

    printf("新连接: %s\n", inet_ntoa(client_addr.sin_addr));
    return client;
}

/* 关闭服务器 */
void server_destroy(NetServer *server) {
    if (server) {
        server->running = 0;
        net_close(server->listen_sock);
        free(server);
    }
}
```

## 3. 客户端API

```c
/* 客户端结构 */
typedef struct {
    socket_t sock;
    char host[256];
    int port;
    int connected;
} NetClient;

/*
 * 连接到服务器
 * 返回: 0=成功, -1=失败
 */
NetClient *client_connect(const char *host, int port) {
    net_init();

    NetClient *client = (NetClient *)calloc(1, sizeof(NetClient));
    strncpy(client->host, host, sizeof(client->host) - 1);
    client->port = port;

    client->sock = socket(AF_INET, SOCK_STREAM, 0);
    if (client->sock == INVALID_SOCK) {
        free(client);
        return NULL;
    }

    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_port = htons(port);
    inet_pton(AF_INET, host, &addr.sin_addr);

    if (connect(client->sock, (struct sockaddr *)&addr, sizeof(addr)) == SOCK_ERROR) {
        fprintf(stderr, "连接 %s:%d 失败\n", host, port);
        net_close(client->sock);
        free(client);
        return NULL;
    }

    client->connected = 1;
    printf("已连接到 %s:%d\n", host, port);
    return client;
}

/* 断开连接 */
void client_disconnect(NetClient *client) {
    if (client) {
        client->connected = 0;
        net_close(client->sock);
        free(client);
    }
}
```

## 4. 数据收发

### 4.1 基本收发

```c
/*
 * 发送数据
 * 返回: 发送的字节数, -1=错误
 */
int net_send(socket_t sock, const void *data, int len) {
    int total = 0;
    const char *ptr = (const char *)data;

    while (total < len) {
        int sent = send(sock, ptr + total, len - total, 0);
        if (sent == SOCK_ERROR) return -1;
        total += sent;
    }

    return total;
}

/*
 * 接收数据
 * 返回: 接收的字节数, 0=连接关闭, -1=错误
 */
int net_recv(socket_t sock, void *buffer, int len) {
    return recv(sock, (char *)buffer, len, 0);
}

/*
 * 接收指定长度的数据
 * 返回: 实际接收的字节数
 */
int net_recv_exact(socket_t sock, void *buffer, int len) {
    int total = 0;
    char *ptr = (char *)buffer;

    while (total < len) {
        int received = recv(sock, ptr + total, len - total, 0);
        if (received <= 0) return received;
        total += received;
    }

    return total;
}
```

### 4.2 消息帧协议

```c
/*
 * 消息帧格式: [4字节长度][数据]
 * 长度字段为网络字节序
 */

/* 发送一个消息帧 */
int net_send_msg(socket_t sock, const void *data, uint32_t len) {
    uint32_t net_len = htonl(len);
    if (net_send(sock, &net_len, 4) != 4) return -1;
    if (len > 0 && net_send(sock, data, len) != (int)len) return -1;
    return 0;
}

/*
 * 接收一个消息帧
 * buffer: 输出缓冲区（自动分配，调用者负责释放）
 * len: 输出数据长度
 * 返回: 0=成功, -1=错误, 1=连接关闭
 */
int net_recv_msg(socket_t sock, void **buffer, uint32_t *len) {
    uint32_t net_len;
    int ret = net_recv_exact(sock, &net_len, 4);
    if (ret == 0) return 1;   // 连接关闭
    if (ret < 0) return -1;

    *len = ntohl(net_len);
    if (*len == 0) {
        *buffer = NULL;
        return 0;
    }

    *buffer = malloc(*len);
    if (net_recv_exact(sock, *buffer, *len) != (int)*len) {
        free(*buffer);
        *buffer = NULL;
        return -1;
    }

    return 0;
}
```

### 4.3 发送字符串

```c
/* 发送字符串 */
int net_send_string(socket_t sock, const char *str) {
    return net_send_msg(sock, str, (uint32_t)strlen(str));
}

/* 接收字符串 */
char *net_recv_string(socket_t sock) {
    void *buffer;
    uint32_t len;

    if (net_recv_msg(sock, &buffer, &len) != 0) return NULL;

    char *str = (char *)malloc(len + 1);
    memcpy(str, buffer, len);
    str[len] = '\0';
    free(buffer);

    return str;
}
```

## 5. 非阻塞IO与超时

```c
/* 设置非阻塞模式 */
int net_set_nonblocking(socket_t sock) {
#ifdef _WIN32
    u_long mode = 1;
    return ioctlsocket(sock, FIONBIO, &mode);
#else
    int flags = fcntl(sock, F_GETFL, 0);
    return fcntl(sock, F_SETFL, flags | O_NONBLOCK);
#endif
}

/* 设置接收超时 */
int net_set_recv_timeout(socket_t sock, int timeout_ms) {
#ifdef _WIN32
    DWORD timeout = timeout_ms;
#else
    struct timeval timeout;
    timeout.tv_sec = timeout_ms / 1000;
    timeout.tv_usec = (timeout_ms % 1000) * 1000;
#endif
    return setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO,
                      (const char *)&timeout, sizeof(timeout));
}

/* 设置发送超时 */
int net_set_send_timeout(socket_t sock, int timeout_ms) {
#ifdef _WIN32
    DWORD timeout = timeout_ms;
#else
    struct timeval timeout;
    timeout.tv_sec = timeout_ms / 1000;
    timeout.tv_usec = (timeout_ms % 1000) * 1000;
#endif
    return setsockopt(sock, SOL_SOCKET, SO_SNDTIMEO,
                      (const char *)&timeout, sizeof(timeout));
}
```

## 6. 简易Echo服务器示例

```c
void echo_server_example() {
    NetServer *server = server_create(8888, 10);
    if (!server) return;

    printf("Echo服务器运行中... 按Ctrl+C停止\n");

    while (server->running) {
        char client_ip[64];
        socket_t client = server_accept(server, client_ip, sizeof(client_ip));
        if (client == INVALID_SOCK) continue;

        printf("客户端连接: %s\n", client_ip);

        // Echo循环
        char buffer[1024];
        while (1) {
            int received = net_recv(client, buffer, sizeof(buffer) - 1);
            if (received <= 0) break;

            buffer[received] = '\0';
            printf("收到: %s", buffer);
            net_send(client, buffer, received);
        }

        printf("客户端断开: %s\n", client_ip);
        net_close(client);
    }

    server_destroy(server);
}

void echo_client_example() {
    NetClient *client = client_connect("127.0.0.1", 8888);
    if (!client) return;

    printf("输入消息 (输入quit退出):\n");

    char buffer[1024];
    while (fgets(buffer, sizeof(buffer), stdin)) {
        if (strncmp(buffer, "quit", 4) == 0) break;

        net_send_string(client->sock, buffer);

        char *response = net_recv_string(client->sock);
        if (response) {
            printf("服务器回显: %s", response);
            free(response);
        }
    }

    client_disconnect(client);
}
```

## 7. 地址工具

```c
/* 获取本地IP地址 */
int net_get_local_ip(char *ip_buf, int buf_len) {
    char hostname[256];
    gethostname(hostname, sizeof(hostname));

    struct hostent *host = gethostbyname(hostname);
    if (!host) return -1;

    struct in_addr **addr_list = (struct in_addr **)host->h_addr_list;
    if (addr_list[0]) {
        strncpy(ip_buf, inet_ntoa(*addr_list[0]), buf_len - 1);
        return 0;
    }
    return -1;
}
```

## 小结

- 网络库封装了socket的基本操作：创建、绑定、监听、连接、收发
- 消息帧协议（长度前缀）解决了TCP粘包问题
- 跨平台封装隐藏了Windows和Linux的API差异
- 非阻塞IO和超时设置避免程序在I/O操作上无限等待
- 生产环境推荐使用libevent、libuv等成熟的网络库
