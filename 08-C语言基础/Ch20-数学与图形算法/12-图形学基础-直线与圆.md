# 图形学基础 - 直线与圆

## 概述

光栅化是将几何图元转换为像素的过程。Bresenham算法是经典的光栅化算法，使用纯整数运算，效率极高。

---

## 1. Bresenham 画线算法

### 1.1 基本原理

通过判断误差项的正负来决定下一个像素的位置，避免浮点运算。

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define WIDTH 80
#define HEIGHT 40

char canvas[HEIGHT][WIDTH];

/* 初始化画布 */
void init_canvas() {
    for (int y = 0; y < HEIGHT; y++)
        for (int x = 0; x < WIDTH; x++)
            canvas[y][x] = ' ';
}

/* 设置像素 */
void set_pixel(int x, int y, char ch) {
    if (x >= 0 && x < WIDTH && y >= 0 && y < HEIGHT) {
        canvas[y][x] = ch;
    }
}

/* 显示画布 */
void display_canvas() {
    for (int y = HEIGHT - 1; y >= 0; y--) {
        for (int x = 0; x < WIDTH; x++) {
            printf("%c", canvas[y][x]);
        }
        printf("\n");
    }
}
```

### 1.2 八方向Bresenham算法

```c
/*
 * Bresenham画线算法
 * 支持所有斜率的直线
 */
void bresenham_line(int x0, int y0, int x1, int y1, char ch) {
    int dx = abs(x1 - x0);
    int dy = abs(y1 - y0);
    int sx = (x0 < x1) ? 1 : -1;
    int sy = (y0 < y1) ? 1 : -1;
    int err = dx - dy;

    while (1) {
        set_pixel(x0, y0, ch);

        if (x0 == x1 && y0 == y1) break;

        int e2 = 2 * err;
        if (e2 > -dy) {
            err -= dy;
            x0 += sx;
        }
        if (e2 < dx) {
            err += dx;
            y0 += sy;
        }
    }
}
```

### 1.3 数值微分法（DDA）

```c
/* DDA算法 — 原理简单但需要浮点运算 */
void dda_line(int x0, int y0, int x1, int y1, char ch) {
    int dx = x1 - x0;
    int dy = y1 - y0;
    int steps = abs(dx) > abs(dy) ? abs(dx) : abs(dy);

    float x_inc = (float)dx / steps;
    float y_inc = (float)dy / steps;

    float x = x0, y = y0;
    for (int i = 0; i <= steps; i++) {
        set_pixel((int)round(x), (int)round(y), ch);
        x += x_inc;
        y += y_inc;
    }
}
```

## 2. 中点画圆算法

### 2.1 算法原理

利用圆的八对称性，只需计算八分之一圆弧，其余通过对称得到。

```c
/*
 * 中点画圆算法
 * 利用圆的对称性，只需判断1/8圆弧
 */
void mid_point_circle(int cx, int cy, int r, char ch) {
    int x = 0, y = r;
    int d = 1 - r;  // 初始决策参数

    while (x <= y) {
        // 八对称绘制
        set_pixel(cx + x, cy + y, ch);
        set_pixel(cx - x, cy + y, ch);
        set_pixel(cx + x, cy - y, ch);
        set_pixel(cx - x, cy - y, ch);
        set_pixel(cx + y, cy + x, ch);
        set_pixel(cx - y, cy + x, ch);
        set_pixel(cx + y, cy - x, ch);
        set_pixel(cx - y, cy - x, ch);

        if (d < 0) {
            d += 2 * x + 3;
        } else {
            d += 2 * (x - y) + 5;
            y--;
        }
        x++;
    }
}
```

### 2.2 Bresenham画圆

```c
/*
 * Bresenham画圆算法（整数运算版本）
 */
void bresenham_circle(int cx, int cy, int r, char ch) {
    int x = 0, y = r;
    int d = 3 - 2 * r;

    while (x <= y) {
        set_pixel(cx + x, cy + y, ch);
        set_pixel(cx - x, cy + y, ch);
        set_pixel(cx + x, cy - y, ch);
        set_pixel(cx - x, cy - y, ch);
        set_pixel(cx + y, cy + x, ch);
        set_pixel(cx - y, cy + x, ch);
        set_pixel(cx + y, cy - x, ch);
        set_pixel(cx - y, cy - x, ch);

        if (d < 0) {
            d += 4 * x + 6;
        } else {
            d += 4 * (x - y) + 10;
            y--;
        }
        x++;
    }
}
```

### 2.3 填充圆

```c
/* 使用扫描线填充圆 */
void filled_circle(int cx, int cy, int r, char ch) {
    for (int y = -r; y <= r; y++) {
        int half_width = (int)sqrt(r * r - y * y);
        for (int x = -half_width; x <= half_width; x++) {
            set_pixel(cx + x, cy + y, ch);
        }
    }
}

/* 使用Bresenham方式填充（更高效） */
void filled_circle_bresenham(int cx, int cy, int r, char ch) {
    int x = 0, y = r;
    int d = 3 - 2 * r;

    while (x <= y) {
        // 画水平扫描线
        for (int i = cx - x; i <= cx + x; i++) {
            set_pixel(i, cy + y, ch);
            set_pixel(i, cy - y, ch);
        }
        for (int i = cx - y; i <= cx + y; i++) {
            set_pixel(i, cy + x, ch);
            set_pixel(i, cy - x, ch);
        }

        if (d < 0) {
            d += 4 * x + 6;
        } else {
            d += 4 * (x - y) + 10;
            y--;
        }
        x++;
    }
}
```

## 3. 画椭圆

```c
/*
 * 中点画椭圆算法
 * 使用四对称性
 */
void mid_point_ellipse(int cx, int cy, int a, int b, char ch) {
    int x = 0, y = b;
    double d1 = b * b - a * a * b + 0.25 * a * a;
    double dx = b * b * (2 * x + 1);
    double dy = a * a * (2 * y - 1);

    // 区域1: 斜率绝对值 < 1
    while (dx < dy) {
        set_pixel(cx + x, cy + y, ch);
        set_pixel(cx - x, cy + y, ch);
        set_pixel(cx + x, cy - y, ch);
        set_pixel(cx - x, cy - y, ch);

        if (d1 < 0) {
            x++;
            dx += 2 * b * b;
            d1 += dx + b * b;
        } else {
            x++;
            y--;
            dx += 2 * b * b;
            dy -= 2 * a * a;
            d1 += dx - dy + b * b;
        }
    }

    // 区域2: 斜率绝对值 >= 1
    double d2 = b * b * (x + 0.5) * (x + 0.5) +
                a * a * (y - 1) * (y - 1) - a * a * b * b;

    while (y >= 0) {
        set_pixel(cx + x, cy + y, ch);
        set_pixel(cx - x, cy + y, ch);
        set_pixel(cx + x, cy - y, ch);
        set_pixel(cx - x, cy - y, ch);

        if (d2 > 0) {
            y--;
            dy -= 2 * a * a;
            d2 += a * a - dy;
        } else {
            y--;
            x++;
            dx += 2 * b * b;
            dy -= 2 * a * a;
            d2 += dx - dy + a * a;
        }
    }
}
```

## 4. 抗锯齿画线

```c
/*
 * Wu's画线算法 — 简单的抗锯齿
 * 在主要像素两侧用不同亮度的像素实现平滑效果
 */
void wu_line(int x0, int y0, int x1, int y1) {
    int steep = abs(y1 - y0) > abs(x1 - x0);
    if (steep) { int t; t=x0; x0=y0; y0=t; t=x1; x1=y1; y1=t; }
    if (x0 > x1) { int t; t=x0; x0=x1; x1=t; t=y0; y0=y1; y1=t; }

    int dx = x1 - x0;
    int dy = y1 - y0;
    float gradient = (dx == 0) ? 1.0f : (float)dy / dx;

    // 起点
    float y = y0 + gradient;
    if (steep) {
        set_pixel(y0, x0, '#');  // 全亮度
        set_pixel(y0 + 1, x0, '.'); // 半亮度
    } else {
        set_pixel(x0, y0, '#');
        set_pixel(x0, y0 + 1, '.');
    }

    // 主循环
    for (int x = x0 + 1; x < x1; x++) {
        int iy = (int)y;
        float frac = y - iy;
        char bright = (frac < 0.5) ? '#' : '.';
        char dim = (frac < 0.5) ? '.' : '#';

        if (steep) {
            set_pixel(iy, x, bright);
            set_pixel(iy + (frac < 0.5 ? -1 : 1), x, dim);
        } else {
            set_pixel(x, iy, bright);
            set_pixel(x, iy + (frac < 0.5 ? -1 : 1), dim);
        }
        y += gradient;
    }

    // 终点
    if (steep) {
        set_pixel(y1, x1, '#');
    } else {
        set_pixel(x1, y1, '#');
    }
}
```

## 5. 测试示例

```c
int main() {
    init_canvas();

    // 画多条不同方向的直线
    bresenham_line(5, 5, 70, 35, '*');
    bresenham_line(5, 35, 70, 5, '+');
    bresenham_line(5, 20, 70, 20, '-');
    bresenham_line(40, 2, 40, 38, '|');

    // 画圆
    bresenham_circle(40, 20, 12, 'O');

    display_canvas();

    printf("\n--- 填充圆示例 ---\n");
    init_canvas();
    filled_circle_bresenham(40, 20, 15, '@');
    bresenham_circle(40, 20, 15, '#');
    display_canvas();

    return 0;
}
```

## 小结

- Bresenham算法使用纯整数运算，是硬件光栅化的基础
- 画圆时利用对称性只需计算1/8弧段
- 中点画圆通过决策参数判断下一像素位置
- 抗锯齿（如Wu算法）通过混合相邻像素亮度实现平滑效果
