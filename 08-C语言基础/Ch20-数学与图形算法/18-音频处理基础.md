# 音频处理基础

## 概述

WAV是最常见的无损音频格式之一，结构简单，适合学习数字音频的基本概念。本节讲解WAV格式解析和PCM数据处理。

---

## 1. WAV文件格式

### 1.1 文件结构

```
+------------------+
| RIFF Header      | - "RIFF"标识 + 文件大小 + "WAVE"
+------------------+
| fmt  Chunk       | - 音频格式信息（采样率、位深、声道数等）
+------------------+
| data Chunk       | - 实际PCM音频数据
+------------------+
```

### 1.2 数据结构定义

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>
#include <math.h>

#pragma pack(push, 1)

/* RIFF文件头 */
typedef struct {
    char riff_id[4];       // "RIFF"
    uint32_t file_size;    // 文件大小 - 8
    char wave_id[4];       // "WAVE"
} WAVHeader;

/* fmt块 */
typedef struct {
    char fmt_id[4];         // "fmt "
    uint32_t fmt_size;      // fmt块大小（通常16或18）
    uint16_t audio_format;  // 1=PCM
    uint16_t num_channels;  // 声道数
    uint32_t sample_rate;   // 采样率（如44100）
    uint32_t byte_rate;     // 字节率 = sample_rate * num_channels * bits_per_sample/8
    uint16_t block_align;   // 块对齐 = num_channels * bits_per_sample/8
    uint16_t bits_per_sample; // 位深（8/16/24/32）
} WAVFmtChunk;

/* data块头 */
typedef struct {
    char data_id[4];        // "data"
    uint32_t data_size;     // 数据大小
} WAVDataChunk;

#pragma pack(pop)

/* WAV音频结构 */
typedef struct {
    WAVHeader header;
    WAVFmtChunk fmt;
    WAVDataChunk data_chunk;
    uint8_t *raw_data;      // 原始PCM数据
    int num_samples;        // 总样本数（含所有声道）
} WAVAudio;
```

## 2. WAV文件读取

```c
/*
 * 读取WAV文件
 * 支持8/16位PCM格式
 */
WAVAudio *wav_read(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) {
        fprintf(stderr, "无法打开文件: %s\n", filename);
        return NULL;
    }

    WAVAudio *wav = (WAVAudio *)calloc(1, sizeof(WAVAudio));

    // 读取RIFF头
    fread(&wav->header, sizeof(WAVHeader), 1, fp);
    if (strncmp(wav->header.riff_id, "RIFF", 4) != 0 ||
        strncmp(wav->header.wave_id, "WAVE", 4) != 0) {
        fprintf(stderr, "不是有效的WAV文件\n");
        free(wav);
        fclose(fp);
        return NULL;
    }

    // 读取fmt块
    fread(&wav->fmt, sizeof(WAVFmtChunk), 1, fp);
    if (strncmp(wav->fmt.fmt_id, "fmt ", 4) != 0) {
        fprintf(stderr, "格式错误: 找不到fmt块\n");
        free(wav);
        fclose(fp);
        return NULL;
    }

    // 跳过额外的fmt数据
    if (wav->fmt.fmt_size > 16) {
        fseek(fp, wav->fmt.fmt_size - 16, SEEK_CUR);
    }

    // 查找data块
    while (1) {
        fread(&wav->data_chunk, sizeof(WAVDataChunk), 1, fp);
        if (strncmp(wav->data_chunk.data_id, "data", 4) == 0) {
            break;
        }
        // 跳过未知块
        fseek(fp, wav->data_chunk.data_size, SEEK_CUR);
    }

    // 读取PCM数据
    wav->raw_data = (uint8_t *)malloc(wav->data_chunk.data_size);
    fread(wav->raw_data, wav->data_chunk.data_size, 1, fp);

    wav->num_samples = wav->data_chunk.data_size /
                       (wav->fmt.bits_per_sample / 8);

    fclose(fp);

    printf("WAV信息:\n");
    printf("  声道数: %d\n", wav->fmt.num_channels);
    printf("  采样率: %d Hz\n", wav->fmt.sample_rate);
    printf("  位深: %d bit\n", wav->fmt.bits_per_sample);
    printf("  样本数: %d\n", wav->num_samples);
    printf("  时长: %.2f 秒\n",
           (double)wav->num_samples / wav->fmt.num_channels / wav->fmt.sample_rate);

    return wav;
}
```

## 3. WAV文件写入

```c
/*
 * 写入16位PCM WAV文件
 * samples: 16位有符号整数数组
 */
int wav_write_16(const char *filename, int sample_rate, int num_channels,
                 int16_t *samples, int num_samples) {
    FILE *fp = fopen(filename, "wb");
    if (!fp) {
        fprintf(stderr, "无法创建文件: %s\n", filename);
        return -1;
    }

    int data_size = num_samples * sizeof(int16_t);

    WAVHeader header;
    memcpy(header.riff_id, "RIFF", 4);
    header.file_size = 36 + data_size;
    memcpy(header.wave_id, "WAVE", 4);

    WAVFmtChunk fmt;
    memcpy(fmt.fmt_id, "fmt ", 4);
    fmt.fmt_size = 16;
    fmt.audio_format = 1;  // PCM
    fmt.num_channels = num_channels;
    fmt.sample_rate = sample_rate;
    fmt.bits_per_sample = 16;
    fmt.block_align = num_channels * 2;
    fmt.byte_rate = sample_rate * fmt.block_align;

    WAVDataChunk data_chunk;
    memcpy(data_chunk.data_id, "data", 4);
    data_chunk.data_size = data_size;

    fwrite(&header, sizeof(WAVHeader), 1, fp);
    fwrite(&fmt, sizeof(WAVFmtChunk), 1, fp);
    fwrite(&data_chunk, sizeof(WAVDataChunk), 1, fp);
    fwrite(samples, data_size, 1, fp);

    fclose(fp);
    printf("成功写入WAV: %s\n", filename);
    return 0;
}
```

## 4. PCM数据处理

### 4.1 获取单声道样本

```c
/*
 * 从16位立体声中提取指定声道
 * channel: 0=左声道, 1=右声道
 */
int16_t *extract_channel(WAVAudio *wav, int channel) {
    if (wav->fmt.bits_per_sample != 16 || wav->fmt.num_channels < 2) {
        fprintf(stderr, "仅支持16位立体声\n");
        return NULL;
    }

    int16_t *data = (int16_t *)wav->raw_data;
    int total_samples = wav->num_samples / wav->fmt.num_channels;
    int16_t *result = (int16_t *)malloc(total_samples * sizeof(int16_t));

    for (int i = 0; i < total_samples; i++) {
        result[i] = data[i * wav->fmt.num_channels + channel];
    }

    return result;
}
```

### 4.2 立体声转单声道

```c
/* 立体声转单声道（取平均） */
int16_t *stereo_to_mono(WAVAudio *wav) {
    if (wav->fmt.num_channels != 2) {
        fprintf(stderr, "输入不是立体声\n");
        return NULL;
    }

    int16_t *data = (int16_t *)wav->raw_data;
    int total_samples = wav->num_samples / 2;
    int16_t *result = (int16_t *)malloc(total_samples * sizeof(int16_t));

    for (int i = 0; i < total_samples; i++) {
        result[i] = (data[i * 2] + data[i * 2 + 1]) / 2;
    }

    return result;
}
```

### 4.3 音量调节

```c
/* 调节音量 */
void adjust_volume(int16_t *samples, int num_samples, double factor) {
    for (int i = 0; i < num_samples; i++) {
        int32_t val = (int32_t)(samples[i] * factor);
        if (val > 32767) val = 32767;
        if (val < -32768) val = -32768;
        samples[i] = (int16_t)val;
    }
}
```

### 4.4 淡入淡出

```c
/* 淡入效果 */
void fade_in(int16_t *samples, int num_samples, int fade_samples) {
    for (int i = 0; i < fade_samples && i < num_samples; i++) {
        double factor = (double)i / fade_samples;
        samples[i] = (int16_t)(samples[i] * factor);
    }
}

/* 淡出效果 */
void fade_out(int16_t *samples, int num_samples, int fade_samples) {
    int start = num_samples - fade_samples;
    if (start < 0) start = 0;
    for (int i = start; i < num_samples; i++) {
        double factor = (double)(num_samples - i) / fade_samples;
        samples[i] = (int16_t)(samples[i] * factor);
    }
}
```

## 5. 正弦波生成

```c
/*
 * 生成正弦波
 * freq: 频率(Hz), duration: 时长(秒), sample_rate: 采样率
 */
int16_t *generate_sine(double freq, double duration, int sample_rate, int *num_samples) {
    *num_samples = (int)(duration * sample_rate);
    int16_t *samples = (int16_t *)malloc(*num_samples * sizeof(int16_t));

    for (int i = 0; i < *num_samples; i++) {
        double t = (double)i / sample_rate;
        double value = sin(2.0 * M_PI * freq * t);
        samples[i] = (int16_t)(value * 32767 * 0.8);  // 80%音量
    }

    return samples;
}

/*
 * 生成DTMF音调
 * row_freq, col_freq: 行频和列频
 */
int16_t *generate_dtmf(double row_freq, double col_freq,
                        double duration, int sample_rate, int *num_samples) {
    *num_samples = (int)(duration * sample_rate);
    int16_t *samples = (int16_t *)malloc(*num_samples * sizeof(int16_t));

    for (int i = 0; i < *num_samples; i++) {
        double t = (double)i / sample_rate;
        double value = sin(2.0 * M_PI * row_freq * t) +
                       sin(2.0 * M_PI * col_freq * t);
        samples[i] = (int16_t)(value * 32767 * 0.4);  // 两路信号各40%
    }

    return samples;
}
```

## 6. 简单音频效果

### 6.1 回声效果

```c
/* 添加回声 */
void add_echo(int16_t *samples, int num_samples,
              int delay_samples, double decay) {
    for (int i = delay_samples; i < num_samples; i++) {
        int32_t val = samples[i] + (int32_t)(samples[i - delay_samples] * decay);
        if (val > 32767) val = 32767;
        if (val < -32768) val = -32768;
        samples[i] = (int16_t)val;
    }
}
```

### 6.2 反转音频

```c
/* 反转音频 */
void reverse_audio(int16_t *samples, int num_samples) {
    for (int i = 0; i < num_samples / 2; i++) {
        int16_t temp = samples[i];
        samples[i] = samples[num_samples - 1 - i];
        samples[num_samples - 1 - i] = temp;
    }
}
```

### 6.3 速度变化

```c
/* 变速（2倍速）— 简单抽取 */
int16_t *speed_up_2x(int16_t *samples, int num_samples, int *new_num) {
    *new_num = num_samples / 2;

    for (int i = 0; i < *new_num; i++) {
```

## 7. 测试示例

```c
int main() {
    // 生成440Hz A4音，持续2秒
    int num_samples;
    int16_t *sine = generate_sine(440.0, 2.0, 44100, &num_samples);

    // 添加淡入淡出
    fade_in(sine, num_samples, 4410);   // 0.1秒淡入
    fade_out(sine, num_samples, 4410);  // 0.1秒淡出

    // 写入WAV
    wav_write_16("sine_440hz.wav", 44100, 1, sine, num_samples);
    free(sine);

    // 生成DTMF "5" (770Hz + 1336Hz)
    int dtmf_samples;
    int16_t *dtmf = generate_dtmf(770, 1336, 0.3, 44100, &dtmf_samples);
    wav_write_16("dtmf_5.wav", 44100, 1, dtmf, dtmf_samples);
    free(dtmf);

    printf("音频文件已生成\n");

    return 0;
}
```

## 小结

- WAV格式结构简单，包含RIFF头、fmt块和data块
- PCM数据按样本交错存储：LRLRLR...（立体声）
- 16位PCM使用有符号整数表示，范围 -32768 ~ 32767
- 音频效果处理（如回声、淡入淡出）本质上是对PCM数据的数学运算
- 生成波形时注意控制幅度防止溢出
