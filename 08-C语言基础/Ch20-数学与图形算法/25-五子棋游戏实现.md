# 五子棋游戏实现

## 概述

五子棋是一种双人对弈棋类游戏。本节实现带AI的五子棋程序，使用极大极小搜索（Minimax）算法结合Alpha-Beta剪枝。

---

## 1. 棋盘与基本数据

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define BOARD_SIZE 15    // 15x15棋盘
#define EMPTY 0
#define BLACK 1          // 黑棋（玩家）
#define WHITE 2          // 白棋（AI）
#define MAX_DEPTH 3      // 搜索深度

/* 棋盘 */
static int board[BOARD_SIZE][BOARD_SIZE];

/* 方向：水平、垂直、两个对角线 */
static int dirs[4][2] = {{1,0}, {0,1}, {1,1}, {1,-1}};

/* 初始化棋盘 */
void board_init() {
    memset(board, 0, sizeof(board));
}

/* 打印棋盘 */
void board_print() {
    printf("  ");
    for (int i = 0; i < BOARD_SIZE; i++) printf("%2d", i);
    printf("\n");

    for (int y = 0; y < BOARD_SIZE; y++) {
        printf("%2d ", y);
        for (int x = 0; x < BOARD_SIZE; x++) {
            switch (board[y][x]) {
                case EMPTY: printf(". "); break;
                case BLACK: printf("X "); break;
                case WHITE: printf("O "); break;
            }
        }
        printf("\n");
    }
}
```

## 2. 胜负判断

```c
/*
 * 检查从(x,y)出发沿某个方向是否有连成五子
 * 返回连续同色棋子的数量
 */
int count_line(int x, int y, int dx, int dy, int player) {
    int count = 0;
    int nx = x + dx, ny = y + dy;

    while (nx >= 0 && nx < BOARD_SIZE && ny >= 0 && ny < BOARD_SIZE &&
           board[ny][nx] == player) {
        count++;
        nx += dx;
        ny += dy;
    }
    return count;
}

/* 检查某位置放子后是否获胜 */
int check_win(int x, int y, int player) {
    for (int d = 0; d < 4; d++) {
        int dx = dirs[d][0], dy = dirs[d][1];
        int count = 1;
        count += count_line(x, y, dx, dy, player);
        count += count_line(x, y, -dx, -dy, player);
        if (count >= 5) return 1;
    }
    return 0;
}

/* 检查棋盘是否已满 */
int board_full() {
    for (int y = 0; y < BOARD_SIZE; y++)
        for (int x = 0; x < BOARD_SIZE; x++)
            if (board[y][x] == EMPTY) return 0;
    return 1;
}
```

## 3. 评估函数

```c
/* 评分表 */
#define SCORE_FIVE    10000000   // 五连
#define SCORE_LIVE4   100000     // 活四
#define SCORE_RUSH4   10000      // 冲四
#define SCORE_LIVE3   1000       // 活三
#define SCORE_SLEEP3  100        // 眠三
#define SCORE_LIVE2   100        // 活二
#define SCORE_SLEEP2  10         // 眠二
#define SCORE_LIVE1   10         // 活一

/*
 * 评估某个方向上一段连续棋型的分数
 * 返回该方向上player的棋型得分
 */
int evaluate_line(int x, int y, int dx, int dy, int player) {
    int count = 1;        // 连续棋子数
    int space = 0;        // 一端空格数
    int block = 0;        // 被封堵的端数

    // 正方向
    int nx = x + dx, ny = y + dy;
    while (nx >= 0 && nx < BOARD_SIZE && ny >= 0 && ny < BOARD_SIZE) {
        if (board[ny][nx] == player) { count++; nx += dx; ny += dy; }
        else if (board[ny][nx] == EMPTY) { space++; break; }
        else { block++; break; }
    }

    // 反方向
    nx = x - dx; ny = y - dy;
    while (nx >= 0 && nx < BOARD_SIZE && ny >= 0 && ny < BOARD_SIZE) {
        if (board[ny][nx] == player) { count++; nx -= dx; ny -= dy; }
        else if (board[ny][nx] == EMPTY) { space++; break; }
        else { block++; break; }
    }

    // 根据连续数、空格数和封堵数评分
    if (count >= 5) return SCORE_FIVE;
    if (block == 2) return 0;  // 两端都被堵

    switch (count) {
        case 4:
            if (space == 2) return SCORE_LIVE4;
            if (space == 1) return SCORE_RUSH4;
            return 0;
        case 3:
            if (space == 2) return SCORE_LIVE3;
            if (space == 1) return SCORE_SLEEP3;
            return 0;
        case 2:
            if (space == 2) return SCORE_LIVE2;
            if (space == 1) return SCORE_SLEEP2;
            return 0;
        case 1:
            if (space == 2) return SCORE_LIVE1;
            return 0;
    }
    return 0;
}

/* 评估某位置对指定玩家的得分 */
int evaluate_point(int x, int y, int player) {
    int score = 0;
    for (int d = 0; d < 4; d++) {
        score += evaluate_line(x, y, dirs[d][0], dirs[d][1], player);
    }
    return score;
}

/* 评估整个棋盘的分数（AI视角） */
int evaluate_board() {
    int ai_score = 0, human_score = 0;

    for (int y = 0; y < BOARD_SIZE; y++) {
        for (int x = 0; x < BOARD_SIZE; x++) {
            if (board[y][x] == WHITE) {
                ai_score += evaluate_point(x, y, WHITE);
            } else if (board[y][x] == BLACK) {
                human_score += evaluate_point(x, y, BLACK);
            }
        }
    }

    return ai_score - human_score;
}
```

## 4. 极大极小搜索与Alpha-Beta剪枝

```c
/*
 * 生成候选落子位置
 * 只考虑已有棋子周围的位置，减少搜索空间
 */
typedef struct { int x, y; } Move;

int generate_moves(Move *moves, int max_moves) {
    int count = 0;
    int visited[BOARD_SIZE][BOARD_SIZE];
    memset(visited, 0, sizeof(visited));

    // 找已有棋子周围两格内的空位
    for (int y = 0; y < BOARD_SIZE; y++) {
        for (int x = 0; x < BOARD_SIZE; x++) {
            if (board[y][x] != EMPTY) {
                for (int dy = -2; dy <= 2; dy++) {
                    for (int dx = -2; dx <= 2; dx++) {
                        int nx = x + dx, ny = y + dy;
                        if (nx >= 0 && nx < BOARD_SIZE &&
                            ny >= 0 && ny < BOARD_SIZE &&
                            board[ny][nx] == EMPTY &&
                            !visited[ny][nx]) {
                            visited[ny][nx] = 1;
                            if (count < max_moves)
                                moves[count++] = (Move){nx, ny};
                        }
                    }
                }
            }
        }
    }

    // 按评估分数排序（启发式剪枝）
    for (int i = 0; i < count - 1; i++) {
        for (int j = i + 1; j < count; j++) {
            int si = evaluate_point(moves[i].x, moves[i].y, WHITE) +
                     evaluate_point(moves[i].x, moves[i].y, BLACK);
            int sj = evaluate_point(moves[j].x, moves[j].y, WHITE) +
                     evaluate_point(moves[j].x, moves[j].y, BLACK);
            if (sj > si) {
                Move tmp = moves[i]; moves[i] = moves[j]; moves[j] = tmp;
            }
        }
    }

    // 只取前N个最优候选
    if (count > 20) count = 20;
    return count;
}

/*
 * Alpha-Beta剪枝搜索
 * is_max: 是否为极大层（AI回合）
 * alpha: 极大值下界
 * beta: 极小值上界
 */
int alpha_beta(int depth, int alpha, int beta, int is_max) {
    // 检查是否有人获胜
    for (int y = 0; y < BOARD_SIZE; y++) {
        for (int x = 0; x < BOARD_SIZE; x++) {
            if (board[y][x] != EMPTY && check_win(x, y, board[y][x])) {
                return (board[y][x] == WHITE) ? (10000000 + depth) :
                                                (-10000000 - depth);
            }
        }
    }

    // 达到搜索深度，返回评估值
    if (depth == 0) return evaluate_board();

    Move moves[BOARD_SIZE * BOARD_SIZE];
    int move_count = generate_moves(moves, 20);

    if (move_count == 0) return 0;  // 无子可下

    if (is_max) {
        int max_eval = -20000000;
        for (int i = 0; i < move_count; i++) {
            board[moves[i].y][moves[i].x] = WHITE;
            int eval = alpha_beta(depth - 1, alpha, beta, 0);
            board[moves[i].y][moves[i].x] = EMPTY;

            if (eval > max_eval) max_eval = eval;
            if (eval > alpha) alpha = eval;
            if (alpha >= beta) break;  // Beta剪枝
        }
        return max_eval;
    } else {
        int min_eval = 20000000;
        for (int i = 0; i < move_count; i++) {
            board[moves[i].y][moves[i].x] = BLACK;
            int eval = alpha_beta(depth - 1, alpha, beta, 1);
            board[moves[i].y][moves[i].x] = EMPTY;

            if (eval < min_eval) min_eval = eval;
            if (eval < beta) beta = eval;
            if (alpha >= beta) break;  // Alpha剪枝
        }
        return min_eval;
    }
}

/*
 * AI选择最佳落子位置
 */
Move ai_best_move() {
    Move best = {-1, -1};
    int best_score = -20000000;

    Move moves[BOARD_SIZE * BOARD_SIZE];
    int move_count = generate_moves(moves, 20);

    for (int i = 0; i < move_count; i++) {
        board[moves[i].y][moves[i].x] = WHITE;
        int score = alpha_beta(MAX_DEPTH - 1, -20000000, 20000000, 0);
        board[moves[i].y][moves[i].x] = EMPTY;

        if (score > best_score) {
            best_score = score;
            best = moves[i];
        }
    }

    return best;
}
```

## 5. 游戏主循环

```c
int main() {
    board_init();

    printf("=== 五子棋 ===\n");
    printf("你执黑(X)，AI执白(O)\n");
    printf("输入坐标(x y)落子，例如: 7 7\n\n");

    int turn = BLACK;  // 黑棋先手

    while (1) {
        board_print();

        if (turn == BLACK) {
            // 玩家回合
            printf("\n你的回合 (输入 x y): ");
            int x, y;
            if (scanf("%d %d", &x, &y) != 2) {
                while (getchar() != '\n');
                printf("输入无效，请重新输入\n");
                continue;
            }

            if (x < 0 || x >= BOARD_SIZE || y < 0 || y >= BOARD_SIZE) {
                printf("坐标超出范围\n");
                continue;
            }
            if (board[y][x] != EMPTY) {
                printf("该位置已有棋子\n");
                continue;
            }

            board[y][x] = BLACK;

            if (check_win(x, y, BLACK)) {
                board_print();
                printf("\n恭喜你获胜!\n");
                break;
            }

            turn = WHITE;
        } else {
            // AI回合
            printf("\nAI思考中...\n");

            Move ai_move = ai_best_move();
            if (ai_move.x == -1) {
                printf("AI无处可下，平局!\n");
                break;
            }

            board[ai_move.y][ai_move.x] = WHITE;
            printf("AI落子: (%d, %d)\n", ai_move.x, ai_move.y);

            if (check_win(ai_move.x, ai_move.y, WHITE)) {
                board_print();
                printf("\nAI获胜!\n");
                break;
            }

            turn = BLACK;
        }

        if (board_full()) {
            board_print();
            printf("\n棋盘已满，平局!\n");
            break;
        }
    }

    return 0;
}
```

## 小结

- 五子棋AI的核心是评估函数和搜索算法
- 评估函数需要识别各种棋型（活四、冲四、活三等）并赋予相应分数
- 极大极小搜索交替最大化和最小化得分
- Alpha-Beta剪枝通过维护上下界来跳过不必要的搜索分支
- 只搜索已有棋子附近的候选位置可以大幅减少搜索空间
