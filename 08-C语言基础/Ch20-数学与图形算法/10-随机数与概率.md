# 随机数与概率

## 概述

随机数在模拟、游戏、密码学和统计学中有广泛应用。本节讲解伪随机数生成、分布模拟和蒙特卡罗方法。

---

## 1. 伪随机数生成器

### 1.1 标准库随机数

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    // 使用当前时间作为种子
    srand((unsigned int)time(NULL));

    printf("rand()的范围: [0, %d]\n", RAND_MAX);

    // 生成10个随机数
    printf("10个随机数:\n");
    for (int i = 0; i < 10; i++) {
        printf("%d ", rand());
    }
    printf("\n");

    // 生成 [a, b] 范围内的随机整数
    int a = 1, b = 100;
    printf("\n[%d, %d]范围内的随机数:\n", a, b);
    for (int i = 0; i < 10; i++) {
        int r = a + rand() % (b - a + 1);
        printf("%d ", r);
    }
    printf("\n");

    // 生成 [0, 1] 之间的随机浮点数
    printf("\n[0, 1]随机浮点数:\n");
    for (int i = 0; i < 5; i++) {
        double f = (double)rand() / RAND_MAX;
        printf("%.6f ", f);
    }
    printf("\n");

    return 0;
}
```

### 1.2 线性同余生成器（LCG）

```c
/*
 * 线性同余生成器
 * X_{n+1} = (a * X_n + c) % m
 * 经典参数来自 Knuth, Numerical Recipes
 */
typedef struct {
    unsigned long long state;
    unsigned long long a;  // 乘数
    unsigned long long c;  // 增量
    unsigned long long m;  // 模数
} LCG;

LCG lcg_create(unsigned long long seed) {
    LCG gen;
    gen.state = seed;
    gen.a = 6364136223846793005ULL;
    gen.c = 1442695040888963407ULL;
    gen.m = 0;  // 使用2^64自然溢出
    return gen;
}

unsigned long long lcg_next(LCG *gen) {
    gen->state = gen->a * gen->state + gen->c;
    return gen->state;
}

/* 生成 [0, max) 范围内的随机数 */
unsigned long lcg_range(LCG *gen, unsigned long max) {
    return lcg_next(gen) % max;
}
```

### 1.3 Xorshift 算法

```c
/*
 * Xorshift128+ — 高质量、快速的伪随机数生成器
 */
typedef struct {
    unsigned long long s[2];
} Xorshift128plus;

Xorshift128plus xorshift_create(unsigned long long seed) {
    Xorshift128plus gen;
    // 用splitmix64初始化
    gen.s[0] = seed;
    gen.s[1] = seed ^ 0x9E3779B97F4A7C15ULL;
    return gen;
}

unsigned long long xorshift_next(Xorshift128plus *gen) {
    unsigned long long s1 = gen->s[0];
    unsigned long long s0 = gen->s[1];
    unsigned long long result = s0 + s1;
    gen->s[0] = s0;
    s1 ^= s1 << 23;
    gen->s[1] = s1 ^ s0 ^ (s1 >> 18) ^ (s0 >> 5);
    return result;
}

/* 生成 [0, 1) 之间的双精度浮点数 */
double xorshift_double(Xorshift128plus *gen) {
    return (xorshift_next(gen) >> 11) * (1.0 / (1ULL << 53));
}
```

## 2. 概率分布模拟

### 2.1 均匀分布

```c
/* [a, b] 上的均匀分布 */
double uniform(double a, double b, Xorshift128plus *gen) {
    return a + (b - a) * xorshift_double(gen);
}
```

### 2.2 正态分布（Box-Muller变换）

```c
#include <math.h>

/*
 * Box-Muller变换: 将两个均匀分布随机数转换为两个正态分布随机数
 * 返回均值为 mu、标准差为 sigma 的正态分布随机数
 */
double normal(double mu, double sigma, Xorshift128plus *gen) {
    double u1 = xorshift_double(gen);
    double u2 = xorshift_double(gen);

    // 避免 log(0)
    if (u1 < 1e-10) u1 = 1e-10;

    double z0 = sqrt(-2.0 * log(u1)) * cos(2.0 * M_PI * u2);
    return mu + sigma * z0;
}

/* 生成n个正态分布样本 */
void generate_normal_samples(double *samples, int n,
                             double mu, double sigma,
                             Xorshift128plus *gen) {
    for (int i = 0; i < n; i++) {
        samples[i] = normal(mu, sigma, gen);
    }
}
```

### 2.3 指数分布

```c
/* 指数分布: 参数 lambda（率参数） */
double exponential(double lambda, Xorshift128plus *gen) {
    double u = xorshift_double(gen);
    if (u < 1e-10) u = 1e-10;
    return -log(u) / lambda;
}
```

### 2.4 泊松分布

```c
/* 泊松分布: Knuth算法 */
int poisson_knuth(double lambda, Xorshift128plus *gen) {
    double L = exp(-lambda);
    int k = 0;
    double p = 1.0;

    do {
        k++;
        p *= xorshift_double(gen);
    } while (p > L);

    return k - 1;
}

/* 泊松分布: 对大lambda使用正态近似 */
int poisson(double lambda, Xorshift128plus *gen) {
    if (lambda < 30) {
        return poisson_knuth(lambda, gen);
    }
    // 正态近似
    int result = (int)round(normal(lambda, sqrt(lambda), gen));
    return result < 0 ? 0 : result;
}
```

## 3. 蒙特卡罗方法

### 3.1 估算圆周率

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

/*
 * 蒙特卡罗方法估算 pi
 * 在单位正方形内随机撒点，统计落入单位圆内的比例
 * pi / 4 ≈ 落入圆内的点数 / 总点数
 */
double estimate_pi(int num_samples, Xorshift128plus *gen) {
    int inside = 0;

    for (int i = 0; i < num_samples; i++) {
        double x = xorshift_double(gen);
        double y = xorshift_double(gen);

        if (x * x + y * y <= 1.0) {
            inside++;
        }
    }

    return 4.0 * inside / num_samples;
}

int main() {
    Xorshift128plus gen = xorshift_create(time(NULL));

    int samples[] = {100, 1000, 10000, 100000, 1000000, 10000000};

    printf("蒙特卡罗方法估算pi:\n");
    printf("%12s %15s %15s\n", "样本数", "估算值", "误差");

    for (int i = 0; i < 6; i++) {
        double pi_est = estimate_pi(samples[i], &gen);
        printf("%12d %15.10f %15.2e\n",
               samples[i], pi_est, fabs(pi_est - M_PI));
    }

    return 0;
}
```

### 3.2 蒙特卡罗积分

```c
/*
 * 蒙特卡罗方法计算定积分
 * 积分 a..b f(x) dx ≈ (b-a)/N * sum(f(xi))
 * 其中 xi 是 [a,b] 上的均匀随机点
 */
double monte_carlo_integral(double (*f)(double),
                             double a, double b,
                             int num_samples,
                             Xorshift128plus *gen) {
    double sum = 0.0;

    for (int i = 0; i < num_samples; i++) {
        double x = uniform(a, b, gen);
        sum += f(x);
    }

    return (b - a) * sum / num_samples;
}

/* 测试: 积分 e^(-x^2) 从 0 到 1 */
double gaussian_integrand(double x) {
    return exp(-x * x);
}

int main() {
    Xorshift128plus gen = xorshift_create(42);

    double result = monte_carlo_integral(gaussian_integrand, 0, 1, 10000000, &gen);
    printf("积分 e^(-x^2) dx [0,1]:\n");
    printf("蒙特卡罗: %.10f\n", result);
    // 精确值 ≈ 0.7468241328

    return 0;
}
```

### 3.3 蒙特卡罗求面积

```c
/*
 * 蒙特卡罗方法计算不规则图形面积
 * 示例: 计算心脏线 r = 1 - sin(theta) 的面积
 * 精确值 = 3*pi/2
 */
double heart_area_mc(int num_samples, Xorshift128plus *gen) {
    int inside = 0;
    // 外接正方形 [-2, 0] x [-2, 2]
    for (int i = 0; i < num_samples; i++) {
        double x = uniform(-2, 0, gen);
        double y = uniform(-2, 2, gen);

        // 转换为极坐标
        double r = sqrt(x * x + y * y);
        double theta = atan2(y, x);

        // 心脏线方程: r <= 1 - sin(theta)
        if (r <= 1.0 - sin(theta)) {
            inside++;
        }
    }

    double rect_area = 4.0 * 4.0;  // 正方形面积
    return rect_area * inside / num_samples;
}
```

## 4. 重采样与Bootstrap

```c
/*
 * Bootstrap方法估计统计量的置信区间
 * 示例: 估计均值的95%置信区间
 */
void bootstrap_mean(double *data, int n, int num_bootstraps,
                    double *ci_lower, double *ci_upper,
                    Xorshift128plus *gen) {
    double *means = (double *)malloc(num_bootstraps * sizeof(double));

    for (int b = 0; b < num_bootstraps; b++) {
        // 有放回抽样
        double sum = 0.0;
        for (int i = 0; i < n; i++) {
            int idx = (int)(xorshift_double(gen) * n);
            sum += data[idx];
        }
        means[b] = sum / n;
    }

    // 排序
    qsort(means, num_bootstraps, sizeof(double),
          (int (*)(const void *, const void *))double_compare);

    // 取2.5%和97.5%分位数
    *ci_lower = means[(int)(0.025 * num_bootstraps)];
    *ci_upper = means[(int)(0.975 * num_bootstraps)];

    free(means);
}
```

## 5. 洗牌算法

```c
/*
 * Fisher-Yates洗牌算法
 * 生成 [0, n-1] 的随机排列
 */
void shuffle(int *arr, int n, Xorshift128plus *gen) {
    for (int i = n - 1; i > 0; i--) {
        int j = (int)(xorshift_double(gen) * (i + 1));
        // 交换 arr[i] 和 arr[j]
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

/* 带权随机采样 */
int weighted_random(double *weights, int n, Xorshift128plus *gen) {
    double total = 0;
    for (int i = 0; i < n; i++) total += weights[i];

    double r = xorshift_double(gen) * total;
    double cumsum = 0;
    for (int i = 0; i < n; i++) {
        cumsum += weights[i];
        if (r <= cumsum) return i;
    }
    return n - 1;
}
```

## 小结

- `rand()` 函数质量较差且范围有限，高质量场景应使用更好的PRNG
- Xorshift128+ 速度快且质量高，适合一般用途
- Box-Muller变换是从均匀分布生成正态分布的经典方法
- 蒙特卡罗方法通过大量随机采样近似求解，样本越多精度越高
- Fisher-Yates算法可以生成等概率的随机排列
