# 内存池实现

## 概述

频繁调用 `malloc/free` 会产生内存碎片且性能不佳。内存池通过预分配大块内存并管理小块分配来解决这些问题。

---

## 1. 固定大小内存池

### 1.1 原理

预分配大块内存，分割为等大小的块，用空闲链表管理。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>

/*
 * 固定大小内存池
 * 适用于频繁分配/释放相同大小对象的场景
 */
typedef struct PoolBlock {
    struct PoolBlock *next;  // 下一个内存块
    char data[];             // 柔性数组（实际数据区域）
} PoolBlock;

typedef struct {
    PoolBlock *free_list;    // 空闲链表
    char *pool;              // 内存池首地址
    size_t block_size;       // 每个块的大小（含管理开销）
    size_t pool_size;        // 内存池总大小
    size_t block_count;      // 总块数
    size_t free_count;       // 空闲块数
} FixedPool;

/* 创建固定大小内存池 */
FixedPool *fixed_pool_create(size_t block_size, size_t block_count) {
    // 确保块大小至少能容纳链表指针
    if (block_size < sizeof(PoolBlock *)) {
        block_size = sizeof(PoolBlock *);
    }

    FixedPool *pool = (FixedPool *)malloc(sizeof(FixedPool));
    pool->block_size = block_size;
    pool->block_count = block_count;
    pool->free_count = block_count;
    pool->pool_size = block_size * block_count;

    // 分配大块内存
    pool->pool = (char *)malloc(pool->pool_size);
    if (!pool->pool) {
        free(pool);
        return NULL;
    }

    // 初始化空闲链表：将所有块串起来
    pool->free_list = NULL;
    for (size_t i = 0; i < block_count; i++) {
        PoolBlock *block = (PoolBlock *)(pool->pool + i * block_size);
        block->next = pool->free_list;
        pool->free_list = block;
    }

    return pool;
}

/* 从内存池分配一个块 */
void *fixed_pool_alloc(FixedPool *pool) {
    if (!pool || !pool->free_list) return NULL;

    PoolBlock *block = pool->free_list;
    pool->free_list = block->next;
    pool->free_count--;

    return block;
}

/* 释放一个块回内存池 */
void fixed_pool_free(FixedPool *pool, void *ptr) {
    if (!pool || !ptr) return;

    PoolBlock *block = (PoolBlock *)ptr;
    block->next = pool->free_list;
    pool->free_list = block;
    pool->free_count++;
}

/* 销毁内存池 */
void fixed_pool_destroy(FixedPool *pool) {
    if (pool) {
        free(pool->pool);
        free(pool);
    }
}

/* 获取使用情况 */
void fixed_pool_stats(FixedPool *pool) {
    printf("内存池统计:\n");
    printf("  块大小: %zu 字节\n", pool->block_size);
    printf("  总块数: %zu\n", pool->block_count);
    printf("  空闲块: %zu\n", pool->free_count);
    printf("  已用块: %zu\n", pool->block_count - pool->free_count);
    printf("  利用率: %.1f%%\n",
           100.0 * (pool->block_count - pool->free_count) / pool->block_count);
}
```

## 2. 通用内存池（伙伴系统简化版）

### 2.1 数据结构

```c
#define POOL_TOTAL_SIZE (1024 * 1024)  // 1MB
#define MIN_BLOCK_SIZE 32               // 最小块32字节
#define MAX_BLOCK_SIZE POOL_TOTAL_SIZE  // 最大块1MB

/*
 * 通用内存池
 * 支持不同大小的分配，使用首次适配策略
 */
typedef struct MemBlock {
    size_t size;              // 数据区大小
    int free;                 // 是否空闲
    struct MemBlock *next;    // 下一个块
    struct MemBlock *prev;    // 上一个块
    uint32_t magic;           // 魔数，用于检测溢出
} MemBlock;

#define MAGIC_ALLOC 0xDEADBEEF
#define MAGIC_FREE  0xCAFEBABE

typedef struct {
    char *pool;               // 内存池首地址
    MemBlock *head;           // 第一个块
    size_t total_size;        // 总大小
    size_t used_size;         // 已使用大小
    size_t alloc_count;       // 分配次数
    size_t free_count;        // 释放次数
} MemPool;
```

### 2.2 创建与销毁

```c
/* 创建通用内存池 */
MemPool *mem_pool_create(size_t size) {
    MemPool *pool = (MemPool *)malloc(sizeof(MemPool));
    pool->total_size = size;
    pool->used_size = 0;
    pool->alloc_count = 0;
    pool->free_count = 0;

    pool->pool = (char *)malloc(size);
    if (!pool->pool) {
        free(pool);
        return NULL;
    }

    // 初始化为一个大的空闲块
    pool->head = (MemBlock *)pool->pool;
    pool->head->size = size - sizeof(MemBlock);
    pool->head->free = 1;
    pool->head->next = NULL;
    pool->head->prev = NULL;
    pool->head->magic = MAGIC_FREE;

    return pool;
}

/* 销毁内存池 */
void mem_pool_destroy(MemPool *pool) {
    if (pool) {
        free(pool->pool);
        free(pool);
    }
}
```

### 2.3 分配

```c
/* 内存对齐 */
static size_t align_size(size_t size) {
    return (size + 7) & ~7;  // 8字节对齐
}

/* 分配内存 */
void *mem_pool_alloc(MemPool *pool, size_t size) {
    if (!pool || size == 0) return NULL;

    size = align_size(size);

    // 首次适配：找到第一个足够大的空闲块
    MemBlock *block = pool->head;
    while (block) {
        if (block->free && block->size >= size) {
            // 如果块比需要的大很多，拆分它
            if (block->size >= size + sizeof(MemBlock) + MIN_BLOCK_SIZE) {
                MemBlock *new_block = (MemBlock *)(
                    (char *)block + sizeof(MemBlock) + size);
                new_block->size = block->size - size - sizeof(MemBlock);
                new_block->free = 1;
                new_block->magic = MAGIC_FREE;
                new_block->next = block->next;
                new_block->prev = block;

                if (block->next) block->next->prev = new_block;
                block->next = new_block;
                block->size = size;
            }

            block->free = 0;
            block->magic = MAGIC_ALLOC;
            pool->used_size += block->size;
            pool->alloc_count++;

            return (void *)((char *)block + sizeof(MemBlock));
        }
        block = block->next;
    }

    return NULL;  // 没有足够空间
}
```

### 2.4 释放

```c
/* 释放内存 */
void mem_pool_free(MemPool *pool, void *ptr) {
    if (!pool || !ptr) return;

    MemBlock *block = (MemBlock *)((char *)ptr - sizeof(MemBlock));

    // 检查魔数
    if (block->magic != MAGIC_ALLOC) {
        fprintf(stderr, "内存池错误: 释放无效或重复释放的内存!\n");
        return;
    }

    block->free = 1;
    block->magic = MAGIC_FREE;
    pool->used_size -= block->size;
    pool->free_count++;

    // 合并相邻的空闲块
    // 与后面的块合并
    if (block->next && block->next->free) {
        block->size += sizeof(MemBlock) + block->next->size;
        MemBlock *to_remove = block->next;
        block->next = to_remove->next;
        if (to_remove->next) to_remove->next->prev = block;
    }

    // 与前面的块合并
    if (block->prev && block->prev->free) {
        block->prev->size += sizeof(MemBlock) + block->size;
        block->prev->next = block->next;
        if (block->next) block->next->prev = block->prev;
    }
}
```

### 2.5 重新分配

```c
/* 重新分配内存（类似realloc） */
void *mem_pool_realloc(MemPool *pool, void *ptr, size_t new_size) {
    if (!ptr) return mem_pool_alloc(pool, new_size);
    if (new_size == 0) { mem_pool_free(pool, ptr); return NULL; }

    MemBlock *block = (MemBlock *)((char *)ptr - sizeof(MemBlock));

    // 如果当前块够大，直接返回
    if (block->size >= align_size(new_size)) {
        return ptr;
    }

    // 分配新块，复制数据，释放旧块
    void *new_ptr = mem_pool_alloc(pool, new_size);
    if (new_ptr) {
        memcpy(new_ptr, ptr, block->size);
        mem_pool_free(pool, ptr);
    }

    return new_ptr;
}
```

### 2.6 统计信息

```c
/* 打印内存池统计 */
void mem_pool_stats(MemPool *pool) {
    size_t free_blocks = 0, used_blocks = 0;
    size_t free_size = 0;

    MemBlock *block = pool->head;
    while (block) {
        if (block->free) {
            free_blocks++;
            free_size += block->size;
        } else {
            used_blocks++;
        }
        block = block->next;
    }

    printf("=== 内存池统计 ===\n");
    printf("总大小:     %zu 字节\n", pool->total_size);
    printf("已使用:     %zu 字节 (%.1f%%)\n",
           pool->used_size, 100.0 * pool->used_size / pool->total_size);
    printf("空闲:       %zu 字节\n", free_size);
    printf("使用块数:   %zu\n", used_blocks);
    printf("空闲块数:   %zu\n", free_blocks);
    printf("分配次数:   %zu\n", pool->alloc_count);
    printf("释放次数:   %zu\n", pool->free_count);
}
```

## 3. 测试

```c
int main() {
    // 固定大小内存池测试
    printf("=== 固定大小内存池 ===\n\n");
    FixedPool *fp = fixed_pool_create(64, 1000);

    void *ptrs[100];
    for (int i = 0; i < 100; i++) {
        ptrs[i] = fixed_pool_alloc(fp);
    }
    fixed_pool_stats(fp);

    for (int i = 0; i < 50; i++) {
        fixed_pool_free(fp, ptrs[i]);
    }
    fixed_pool_stats(fp);

    fixed_pool_destroy(fp);

    // 通用内存池测试
    printf("\n=== 通用内存池 ===\n\n");
    MemPool *mp = mem_pool_create(1024 * 1024);

    int *arr = (int *)mem_pool_alloc(mp, 1000 * sizeof(int));
    char *str = (char *)mem_pool_alloc(mp, 256);
    double *dbl = (double *)mem_pool_alloc(mp, 100 * sizeof(double));

    for (int i = 0; i < 1000; i++) arr[i] = i;
    strcpy(str, "Hello, Memory Pool!");
    for (int i = 0; i < 100; i++) dbl[i] = i * 0.1;

    printf("arr[999] = %d\n", arr[999]);
    printf("str = %s\n", str);
    printf("dbl[99] = %.1f\n", dbl[99]);

    mem_pool_stats(mp);

    mem_pool_free(mp, arr);
    mem_pool_free(mp, str);
    mem_pool_free(mp, dbl);

    printf("\n释放后:\n");
    mem_pool_stats(mp);

    mem_pool_destroy(mp);

    return 0;
}
```

## 小结

- 固定大小内存池适合频繁分配/释放相同大小对象的场景（如链表节点）
- 通用内存池通过拆分/合并空闲块支持不同大小的分配
- 内存对齐（通常8或16字节）对CPU访问效率很重要
- 魔数机制可检测内存越界和重复释放
- 内存池销毁时一次性释放所有内存，避免碎片
