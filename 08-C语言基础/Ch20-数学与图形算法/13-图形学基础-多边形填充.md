# 图形学基础 - 多边形填充

## 概述

多边形填充是将多边形内部区域用颜色或图案覆盖的过程。本节介绍扫描线填充和种子填充两种经典算法。

---

## 1. 数据结构

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

#define WIDTH 80
#define HEIGHT 40

char canvas[HEIGHT][WIDTH];

void init_canvas() {
    memset(canvas, ' ', sizeof(canvas));
}

void set_pixel(int x, int y, char ch) {
    if (x >= 0 && x < WIDTH && y >= 0 && y < HEIGHT)
        canvas[y][x] = ch;
}

char get_pixel(int x, int y) {
    if (x >= 0 && x < WIDTH && y >= 0 && y < HEIGHT)
        return canvas[y][x];
    return ' ';
}

void display_canvas() {
    for (int y = HEIGHT - 1; y >= 0; y--) {
        for (int x = 0; x < WIDTH; x++)
            printf("%c", canvas[y][x]);
        printf("\n");
    }
}

/* 顶点 */
typedef struct { int x, y; } Point;

/* 边表节点 */
typedef struct EdgeNode {
    int ymax;         // 边的最大y值
    double x;         // 当前扫描线与边的交点x坐标
    double inv_slope; // 1/k（x的增量）
    struct EdgeNode *next;
} EdgeNode;
```

## 2. 扫描线填充算法

### 2.1 算法原理

扫描线算法逐行处理，维护一个活性边表(AET)，每次处理完一行后更新交点并删除不再相交的边。

```c
/*
 * 扫描线填充算法
 * 基于活性边表(AET)实现
 */
void scanline_fill(Point *polygon, int n, char fill_ch) {
    if (n < 3) return;

    // 找y的范围
    int ymin = polygon[0].y, ymax = polygon[0].y;
    for (int i = 1; i < n; i++) {
        if (polygon[i].y < ymin) ymin = polygon[i].y;
        if (polygon[i].y > ymax) ymax = polygon[i].y;
    }

    // 构建边表(ET)
    EdgeNode **edge_table = (EdgeNode **)calloc(ymax - ymin + 1, sizeof(EdgeNode *));

    for (int i = 0; i < n; i++) {
        Point p1 = polygon[i];
        Point p2 = polygon[(i + 1) % n];

        // 跳过水平边
        if (p1.y == p2.y) continue;

        // 确保 p1.y < p2.y
        if (p1.y > p2.y) {
            Point temp = p1; p1 = p2; p2 = temp;
        }

        EdgeNode *edge = (EdgeNode *)malloc(sizeof(EdgeNode));
        edge->ymax = p2.y;
        edge->x = p1.x;
        edge->inv_slope = (double)(p2.x - p1.x) / (p2.y - p1.y);

        int idx = p1.y - ymin;
        edge->next = edge_table[idx];
        edge_table[idx] = edge;
    }

    // 活性边表
    EdgeNode *aet = NULL;

    for (int y = ymin; y <= ymax; y++) {
        // 将当前扫描线对应的边加入AET
        EdgeNode *edge = edge_table[y - ymin];
        while (edge) {
            EdgeNode *next = edge->next;
            edge->next = aet;
            aet = edge;
            edge = next;
        }

        // 从AET中移除ymax == y的边
        EdgeNode **curr = &aet;
        while (*curr) {
            if ((*curr)->ymax == y) {
                EdgeNode *temp = *curr;
                *curr = temp->next;
                free(temp);
            } else {
                curr = &((*curr)->next);
            }
        }

        // 收集交点
        int intersections[WIDTH];
        int count = 0;
        for (EdgeNode *e = aet; e; e = e->next) {
            intersections[count++] = (int)round(e->x);
        }

        // 排序交点
        for (int i = 0; i < count - 1; i++) {
            for (int j = i + 1; j < count; j++) {
                if (intersections[i] > intersections[j]) {
                    int t = intersections[i];
                    intersections[i] = intersections[j];
                    intersections[j] = t;
                }
            }
        }

        // 两两配对填充
        for (int i = 0; i < count - 1; i += 2) {
            for (int x = intersections[i]; x <= intersections[i + 1]; x++) {
                set_pixel(x, y, fill_ch);
            }
        }

        // 更新AET中边的x坐标
        for (EdgeNode *e = aet; e; e = e->next) {
            e->x += e->inv_slope;
        }
    }

    // 释放边表
    free(edge_table);
}
```

### 2.2 改进的扫描线填充

```c
/*
 * 处理奇点（共享顶点）的扫描线填充
 * 通过调整边的计数处理顶点共享问题
 */
typedef struct {
    int ymax;
    double x;
    double inv_slope;
    int ydir;  // 边的方向: +1 上到下, -1 下到上
} Edge;

int edge_compare(const void *a, const void *b) {
    const Edge *ea = (const Edge *)a;
    const Edge *eb = (const Edge *)b;
    if (fabs(ea->x - eb->x) < 1e-6) return 0;
    return ea->x < eb->x ? -1 : 1;
}

void scanline_fill_improved(Point *polygon, int n, char fill_ch) {
    if (n < 3) return;

    int ymin = polygon[0].y, ymax = polygon[0].y;
    for (int i = 1; i < n; i++) {
        if (polygon[i].y < ymin) ymin = polygon[i].y;
        if (polygon[i].y > ymax) ymax = polygon[i].y;
    }

    // 活性边表（动态数组）
    Edge *aet = (Edge *)malloc(n * sizeof(Edge));
    int aet_count = 0;

    // 所有边
    Edge *all_edges = (Edge *)malloc(n * sizeof(Edge));
    int edge_count = 0;

    for (int i = 0; i < n; i++) {
        Point p1 = polygon[i];
        Point p2 = polygon[(i + 1) % n];
        if (p1.y == p2.y) continue;

        Edge e;
        e.ymax = p1.y > p2.y ? p1.y : p2.y;
        e.x = p1.y < p2.y ? p1.x : p2.x;
        e.inv_slope = (double)(p2.x - p1.x) / (p2.y - p1.y);
        e.ydir = (p2.y > p1.y) ? 1 : -1;

        // 按ymin存储边
        all_edges[edge_count++] = e;
    }

    for (int y = ymin; y <= ymax; y++) {
        // 添加新边
        for (int i = 0; i < edge_count; i++) {
            int ymin_edge = all_edges[i].ymax - abs(
                (int)(1.0 / all_edges[i].inv_slope *
                (all_edges[i].ymax - all_edges[i].x)));
            // 简化处理
        }

        // 排序、配对、填充
        qsort(aet, aet_count, sizeof(Edge), edge_compare);

        for (int i = 0; i < aet_count - 1; i += 2) {
            int x0 = (int)ceil(aet[i].x);
            int x1 = (int)floor(aet[i + 1].x);
            for (int x = x0; x <= x1; x++) {
                set_pixel(x, y, fill_ch);
            }
        }

        // 更新x
        for (int i = 0; i < aet_count; i++) {
            aet[i].x += aet[i].inv_slope;
        }
    }

    free(aet);
    free(all_edges);
}
```

## 3. 种子填充算法

### 3.1 递归种子填充

```c
/*
 * 递归四连通种子填充
 * 从种子点出发，向四个方向扩展填充
 * 注意: 深度较大时可能栈溢出
 */
void seed_fill_recursive(int x, int y, char boundary, char fill_ch) {
    if (x < 0 || x >= WIDTH || y < 0 || y >= HEIGHT) return;
    if (get_pixel(x, y) == boundary) return;
    if (get_pixel(x, y) == fill_ch) return;

    set_pixel(x, y, fill_ch);

    seed_fill_recursive(x + 1, y, boundary, fill_ch);
    seed_fill_recursive(x - 1, y, boundary, fill_ch);
    seed_fill_recursive(x, y + 1, boundary, fill_ch);
    seed_fill_recursive(x, y - 1, boundary, fill_ch);
}
```

### 3.2 扫描线种子填充（推荐）

```c
/*
 * 扫描线种子填充算法
 * 使用栈代替递归，避免栈溢出
 */
typedef struct {
    int x, y;
} Seed;

void scanline_seed_fill(int seed_x, int seed_y, char boundary, char fill_ch) {
    Seed *stack = (Seed *)malloc(WIDTH * HEIGHT * sizeof(Seed));
    int top = 0;

    // 初始种子入栈
    stack[top++] = (Seed){seed_x, seed_y};

    while (top > 0) {
        Seed s = stack[--top];
        int x = s.x, y = s.y;

        if (get_pixel(x, y) == boundary || get_pixel(x, y) == fill_ch)
            continue;

        // 向右填充
        int xl = x;
        while (xl < WIDTH && get_pixel(xl, y) != boundary &&
               get_pixel(xl, y) != fill_ch) {
            set_pixel(xl, y, fill_ch);
            xl++;
        }
        int xr = xl - 1;

        // 向左填充
        xl = x - 1;
        while (xl >= 0 && get_pixel(xl, y) != boundary &&
               get_pixel(xl, y) != fill_ch) {
            set_pixel(xl, y, fill_ch);
            xl--;
        }
        xl++;

        // 检查上一行和下一行
        for (int dy = -1; dy <= 1; dy += 2) {
            int ny = y + dy;
            if (ny < 0 || ny >= HEIGHT) continue;

            int in_span = 0;
            for (int nx = xl; nx <= xr; nx++) {
                char c = get_pixel(nx, ny);
                if (c != boundary && c != fill_ch) {
                    if (!in_span) {
                        if (top < WIDTH * HEIGHT)
                            stack[top++] = (Seed){nx, ny};
                        in_span = 1;
                    }
                } else {
                    in_span = 0;
                }
            }
        }
    }

    free(stack);
}
```

```c
void bresenham_line(int x0, int y0, int x1, int y1, char ch) {
    int dx = abs(x1 - x0), dy = abs(y1 - y0);
    int sx = x0 < x1 ? 1 : -1, sy = y0 < y1 ? 1 : -1;
    int err = dx - dy;
    while (1) {
        set_pixel(x0, y0, ch);
        if (x0 == x1 && y0 == y1) break;
        int e2 = 2 * err;
        if (e2 > -dy) { err -= dy; x0 += sx; }
        if (e2 < dx)  { err += dx; y0 += sy; }
    }
}

int main() {
    // 示例1: 扫描线填充三角形
    init_canvas();
    Point triangle[] = {{10, 5}, {50, 35}, {70, 10}};
    scanline_fill(triangle, 3, '.');
    draw_polygon(triangle, 3, '#');
    printf("扫描线填充三角形:\n");
    display_canvas();

    // 示例2: 种子填充矩形

    draw_polygon(rect, 4, '*');
```

## 小结

- 扫描线填充适合凸/凹多边形，效率高，需要维护边表和活性边表
- 种子填充实现简单，但递归版本可能栈溢出，推荐使用扫描线种子填充
- 处理多边形顶点共享（奇点）时需要注意交点计数的正确性
- 填充前应先画出边框以定义填充边界
