# 浮点数精度处理

## 概述

浮点数在计算机中以IEEE 754标准存储，存在固有的精度限制。理解浮点数的本质对于编写正确的数值程序至关重要。

---

## 1. 浮点数的本质

### 1.1 IEEE 754 格式

```c
#include <stdio.h>
#include <stdint.h>
#include <string.h>

/* 查看浮点数的二进制表示 */
void print_float_bits(float f) {
    uint32_t bits;
    memcpy(&bits, &f, sizeof(f));

    int sign = (bits >> 31) & 1;
    int exponent = (bits >> 23) & 0xFF;
    uint32_t mantissa = bits & 0x7FFFFF;

    printf("浮点数: %f\n", f);
    printf("符号位: %d\n", sign);
    printf("指数位: %d (实际指数: %d)\n", exponent, exponent - 127);
    printf("尾数位: 0x%06X\n", mantissa);
    printf("二进制: ");
    for (int i = 31; i >= 0; i--) {
        printf("%d", (bits >> i) & 1);
        if (i == 31 || i == 23) printf(" ");
    }
    printf("\n\n");
}

int main() {
    print_float_bits(0.1f);
    print_float_bits(1.0f);
    print_float_bits(-3.14f);

    return 0;
}
```

### 1.2 精度问题演示

```c
#include <stdio.h>

int main() {
    // 经典的浮点精度问题
    double a = 0.1 + 0.2;
    printf("0.1 + 0.2 = %.20f\n", a);       // 不等于 0.3
    printf("0.1 + 0.2 == 0.3 ? %s\n", a == 0.3 ? "true" : "false");

    // 累积误差
    double sum = 0.0;
    for (int i = 0; i < 1000000; i++) {
        sum += 0.000001;
    }
    printf("累加100万次 0.000001:\n");
    printf("  理论值: 1.0\n");
    printf("  实际值: %.15f\n", sum);
    printf("  误差:   %.2e\n", fabs(sum - 1.0));

    // 大数吃小数
    double big = 1e16;
    double small = 1.0;
    printf("\n大数吃小数:\n");
    printf("  1e16 + 1.0 = %.1f\n", big + small);  // 仍然是 1e16
    printf("  1e16 + 1.0 == 1e16 ? %s\n",
           (big + small == big) ? "true" : "false");

    return 0;
}
```

## 2. 浮点数比较方法

### 2.1 相对误差与绝对误差比较

```c
#include <stdio.h>
#include <math.h>
#include <float.h>

#define EPS_ABS 1e-9
#define EPS_REL 1e-9

/*
 * 安全的浮点数比较
 * 结合绝对误差和相对误差
 */
int double_equal(double a, double b) {
    double diff = fabs(a - b);

    // 绝对误差检查（处理接近零的情况）
    if (diff < EPS_ABS) return 1;

    // 相对误差检查
    double max_val = fabs(a) > fabs(b) ? fabs(a) : fabs(b);
    if (diff / max_val < EPS_REL) return 1;

    return 0;
}

/*
 * 带有自定义精度的比较
 */
int double_equal_eps(double a, double b, double eps) {
    double diff = fabs(a - b);
    if (diff < eps) return 1;

    double max_val = fabs(a) > fabs(b) ? fabs(a) : fabs(b);
    if (max_val > eps && diff / max_val < eps) return 1;

    return 0;
}

/* 比较函数，用于排序 */
int double_compare(const void *a, const void *b) {
    double da = *(const double *)a;
    double db = *(const double *)b;
    if (fabs(da - db) < EPS_ABS) return 0;
    return (da > db) ? 1 : -1;
}

int main() {
    printf("0.1+0.2 == 0.3? %s\n",
           double_equal(0.1 + 0.2, 0.3) ? "是" : "否");

    printf("1.0/3.0*3.0 == 1.0? %s\n",
           double_equal(1.0 / 3.0 * 3.0, 1.0) ? "是" : "否");

    // 排序示例
    double arr[] = {3.0, 1.0000000001, 2.0, 1.0, 1.0000000002};
    int n = sizeof(arr) / sizeof(arr[0]);
    qsort(arr, n, sizeof(double), double_compare);

    printf("\n排序后: ");
    for (int i = 0; i < n; i++) printf("%.10f ", arr[i]);
    printf("\n");

    return 0;
}
```

### 2.2 特殊值处理

```c
#include <math.h>
#include <stdio.h>

/* 检查浮点数是否有效 */
int is_valid_double(double x) {
    return !isnan(x) && !isinf(x);
}

/* 安全比较（处理NaN和Inf） */
int safe_equal(double a, double b) {
    if (isnan(a) || isnan(b)) return 0;  // NaN != NaN
    if (isinf(a) && isinf(b)) {
        // +Inf == +Inf, -Inf == -Inf
        return (a > 0) == (b > 0);
    }
    return double_equal(a, b);
}

int main() {
    double nan_val = 0.0 / 0.0;
    double inf_val = 1.0 / 0.0;

    printf("NaN == NaN: %d\n", nan_val == nan_val);     // false
    printf("NaN != NaN: %d\n", nan_val != nan_val);     // true!
    printf("is_valid(NaN): %d\n", is_valid_double(nan_val));
    printf("is_valid(Inf): %d\n", is_valid_double(inf_val));

    return 0;
}
```

## 3. 高精度模拟浮点数

### 3.1 用整数模拟定点数

```c
#include <stdio.h>

/*
 * 定点数: 用 long long 存储，最后 DECIMAL_DIGITS 位为小数部分
 * 例如 DECIMAL_DIGITS=6 时，存储值1234567表示1.234567
 */
#define DECIMAL_DIGITS 6
#define SCALE 1000000LL

typedef long long FixedPoint;

FixedPoint fp_from_double(double x) {
    return (long long)(x * SCALE + (x >= 0 ? 0.5 : -0.5));
}

FixedPoint fp_from_int(int n) {
    return (long long)n * SCALE;
}

double fp_to_double(FixedPoint fp) {
    return (double)fp / SCALE;
}

FixedPoint fp_add(FixedPoint a, FixedPoint b) {
    return a + b;
}

FixedPoint fp_sub(FixedPoint a, FixedPoint b) {
    return a - b;
}

FixedPoint fp_mul(FixedPoint a, FixedPoint b) {
    return (a * b) / SCALE;
}

FixedPoint fp_div(FixedPoint a, FixedPoint b) {
    return (a * SCALE) / b;
}

void fp_print(FixedPoint fp) {
    long long int_part = fp / SCALE;
    long long frac_part = fp % SCALE;
    if (frac_part < 0) frac_part = -frac_part;
    printf("%lld.%06lld", int_part, frac_part);
}

int main() {
    FixedPoint a = fp_from_double(1.5);
    FixedPoint b = fp_from_double(0.3);

    printf("a = "); fp_print(a); printf("\n");
    printf("b = "); fp_print(b); printf("\n");

    FixedPoint sum = fp_add(a, b);
    printf("a + b = "); fp_print(sum); printf("\n");

    FixedPoint prod = fp_mul(a, b);
    printf("a * b = "); fp_print(prod); printf("\n");

    // 验证: 浮点 vs 定点
    printf("\n精度对比:\n");
    printf("浮点: 0.1 + 0.2 = %.15f\n", 0.1 + 0.2);
    FixedPoint fps = fp_add(fp_from_double(0.1), fp_from_double(0.2));
    printf("定点: 0.1 + 0.2 = "); fp_print(fps); printf("\n");

    return 0;
}
```

## 4. 数值稳定的算法

### 4.1 防止下溢和上溢

```c
#include <math.h>
#include <stdio.h>

/* 安全的softmax计算（减去最大值防止溢出） */
void softmax(double *x, int n, double *result) {
    // 找最大值
    double max_val = x[0];
    for (int i = 1; i < n; i++) {
        if (x[i] > max_val) max_val = x[i];
    }

    // 减去最大值后计算exp
    double sum = 0.0;
    for (int i = 0; i < n; i++) {
        result[i] = exp(x[i] - max_val);
        sum += result[i];
    }

    // 归一化
    for (int i = 0; i < n; i++) {
        result[i] /= sum;
    }
}

/* Kahan求和算法：减少浮点累加误差 */
double kahan_sum(double *arr, int n) {
    double sum = 0.0;
    double compensation = 0.0;  // 补偿项

    for (int i = 0; i < n; i++) {
        double y = arr[i] - compensation;
        double t = sum + y;
        compensation = (t - sum) - y;  // 计算丢失的低阶位
        sum = t;
    }
    return sum;
}

int main() {
    // Kahan求和 vs 普通求和
    double arr[1000000];
    for (int i = 0; i < 1000000; i++) {
        arr[i] = 0.000001;
    }

    // 普通求和
    double naive_sum = 0.0;
    for (int i = 0; i < 1000000; i++) {
        naive_sum += arr[i];
    }

    // Kahan求和
    double kahan = kahan_sum(arr, 1000000);

    printf("普通求和: %.15f\n", naive_sum);
    printf("Kahan求和: %.15f\n", kahan);
    printf("理论值:   1.000000000000000\n");

    return 0;
}
```

### 4.2 稳定的方差计算

```c
/*
 * Welford在线算法计算方差
 * 数值稳定，只需一次遍历
 */
typedef struct {
    int n;
    double mean;
    double m2;  // 累积的平方偏差
} WelfordState;

void welford_init(WelfordState *s) {
    s->n = 0;
    s->mean = 0.0;
    s->m2 = 0.0;
}

void welford_update(WelfordState *s, double x) {
    s->n++;
    double delta = x - s->mean;
    s->mean += delta / s->n;
    double delta2 = x - s->mean;
    s->m2 += delta * delta2;
}

double welford_variance(WelfordState *s) {
    if (s->n < 2) return 0.0;
    return s->m2 / (s->n - 1);  // 样本方差
}

double welford_mean(WelfordState *s) {
    return s->mean;
}
```

## 小结

- 浮点数比较永远不要用 `==`，应使用带容差的比较函数
- 大数加小数会丢失精度，尽量避免相差悬殊的数值运算
- Kahan求和算法可以显著减少累加误差
- 定点数适用于需要精确小数表示的场景（如金融计算）
- Welford算法提供了数值稳定的在线统计量计算方法
