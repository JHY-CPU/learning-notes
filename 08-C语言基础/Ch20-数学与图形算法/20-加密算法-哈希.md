# 加密算法 - 哈希

## 概述

哈希函数将任意长度的数据映射为固定长度的摘要。好的哈希函数具有单向性、抗碰撞性等特性。本节实现MD5和SHA-256。

---

## 1. 哈希函数基本概念

| 特性 | 说明 |
|------|------|
| 确定性 | 相同输入总是产生相同输出 |
| 快速计算 | 计算效率高 |
| 单向性 | 从摘要无法反推原文 |
| 抗碰撞性 | 难以找到两个不同输入产生相同摘要 |
| 雪崩效应 | 输入微小变化导致输出大幅变化 |

### 2.2 MD5核心算法

```c
/* 左循环移位 */
static uint32_t left_rotate(uint32_t x, int n) {
    return (x << n) | (x >> (32 - n));
}

/* MD5辅助函数 */
#define F(x,y,z) (((x) & (y)) | ((~x) & (z)))
#define G(x,y,z) (((x) & (z)) | ((y) & (~z)))
#define H(x,y,z) ((x) ^ (y) ^ (z))
#define I(x,y,z) ((y) ^ ((x) | (~z)))

/*
 * MD5计算
 * input: 输入数据
 * len: 输入长度（字节）
 * output: 16字节哈希输出
 */
void md5(const uint8_t *input, size_t len, uint8_t output[16]) {
    // 初始状态
    uint32_t a0 = 0x67452301;
    uint32_t b0 = 0xefcdab89;
    uint32_t c0 = 0x98badcfe;
    uint32_t d0 = 0x10325476;

    // 消息填充：使长度 ≡ 448 (mod 512) 位
    size_t padded_len = ((len + 8) / 64 + 1) * 64;
    uint8_t *padded = (uint8_t *)calloc(padded_len, 1);
    memcpy(padded, input, len);
    padded[len] = 0x80;  // 填充10000000

    // 填充原始长度（64位，小端序）
    uint64_t bit_len = (uint64_t)len * 8;
    memcpy(padded + padded_len - 8, &bit_len, 8);

    // 处理每个512位（64字节）块
    for (size_t offset = 0; offset < padded_len; offset += 64) {
        uint32_t M[16];
        for (int i = 0; i < 16; i++) {
            memcpy(&M[i], padded + offset + i * 4, 4);
        }

        uint32_t A = a0, B = b0, C = c0, D = d0;

        for (int i = 0; i < 64; i++) {
            uint32_t F_val, g;
            if (i < 16) {
                F_val = F(B, C, D); g = i;
            } else if (i < 32) {
                F_val = G(B, C, D); g = (5 * i + 1) % 16;
            } else if (i < 48) {
                F_val = H(B, C, D); g = (3 * i + 5) % 16;
            } else {
                F_val = I(B, C, D); g = (7 * i) % 16;
            }

            F_val = F_val + A + T[i] + M[g];
            A = D;
            D = C;
            C = B;
            B = B + left_rotate(F_val, S[i]);
        }

        a0 += A; b0 += B; c0 += C; d0 += D;
    }

    free(padded);

    // 输出（小端序）
    memcpy(output + 0, &a0, 4);
    memcpy(output + 4, &b0, 4);
    memcpy(output + 8, &c0, 4);
    memcpy(output + 12, &d0, 4);
}
```

### 3.2 SHA-256核心算法

```c
/* SHA-256辅助函数 */
#define CH(x,y,z) (((x) & (y)) ^ (~(x) & (z)))
#define MAJ(x,y,z) (((x) & (y)) ^ ((x) & (z)) ^ ((y) & (z)))
#define EP0(x) (right_rotate(x,2) ^ right_rotate(x,13) ^ right_rotate(x,22))
#define EP1(x) (right_rotate(x,6) ^ right_rotate(x,11) ^ right_rotate(x,25))
#define SIG0(x) (right_rotate(x,7) ^ right_rotate(x,18) ^ ((x) >> 3))
#define SIG1(x) (right_rotate(x,17) ^ right_rotate(x,19) ^ ((x) >> 10))

static uint32_t right_rotate(uint32_t x, int n) {
    return (x >> n) | (x << (32 - n));
}

/*
 * SHA-256计算
 * output: 32字节哈希输出
 */
void sha256(const uint8_t *input, size_t len, uint8_t output[32]) {
    // 初始哈希值
    uint32_t h[8] = {
        0x6a09e667, 0xbb67ae85, 0x3c6ef372, 0xa54ff53a,
        0x510e527f, 0x9b05688c, 0x1f83d9ab, 0x5be0cd19
    };

    // 消息填充
    size_t padded_len = ((len + 8) / 64 + 1) * 64;
    uint8_t *padded = (uint8_t *)calloc(padded_len, 1);
    memcpy(padded, input, len);
    padded[len] = 0x80;

    // 长度（大端序）
    uint64_t bit_len = (uint64_t)len * 8;
    for (int i = 0; i < 8; i++) {
        padded[padded_len - 1 - i] = (uint8_t)(bit_len >> (i * 8));
    }

    // 处理每个512位块
    for (size_t offset = 0; offset < padded_len; offset += 64) {
        uint32_t W[64];

        // 前16个字
        for (int i = 0; i < 16; i++) {
            W[i] = ((uint32_t)padded[offset + i*4] << 24) |
                   ((uint32_t)padded[offset + i*4 + 1] << 16) |
                   ((uint32_t)padded[offset + i*4 + 2] << 8) |
                   ((uint32_t)padded[offset + i*4 + 3]);
        }

        // 扩展到64个字
        for (int i = 16; i < 64; i++) {
            W[i] = SIG1(W[i-2]) + W[i-7] + SIG0(W[i-15]) + W[i-16];
        }

        uint32_t a = h[0], b = h[1], c = h[2], d = h[3];
        uint32_t e = h[4], f = h[5], g = h[6], hh = h[7];

        for (int i = 0; i < 64; i++) {
            uint32_t T1 = hh + EP1(e) + CH(e,f,g) + K256[i] + W[i];
            uint32_t T2 = EP0(a) + MAJ(a,b,c);
            hh = g; g = f; f = e; e = d + T1;
            d = c; c = b; b = a; a = T1 + T2;
        }

        h[0] += a; h[1] += b; h[2] += c; h[3] += d;
        h[4] += e; h[5] += f; h[6] += g; h[7] += hh;
    }

    free(padded);

    // 输出（大端序）
    for (int i = 0; i < 8; i++) {
        output[i*4]   = (h[i] >> 24) & 0xFF;
        output[i*4+1] = (h[i] >> 16) & 0xFF;
        output[i*4+2] = (h[i] >> 8) & 0xFF;
        output[i*4+3] = h[i] & 0xFF;
    }
}
```

## 4. 使用示例

```c
void print_hash(const char *label, const uint8_t *hash, int len) {
    printf("%s: ", label);
    for (int i = 0; i < len; i++) printf("%02x", hash[i]);
    printf("\n");
}

int main() {
    // MD5测试
    const char *msg1 = "";
    const char *msg2 = "abc";
    const char *msg3 = "hello world";

    uint8_t md5_result[16];

    md5((uint8_t *)msg1, strlen(msg1), md5_result);
    print_hash("MD5(\"\")", md5_result, 16);
    // 预期: d41d8cd98f00b204e9800998ecf8427e

    md5((uint8_t *)msg2, strlen(msg2), md5_result);
    print_hash("MD5(\"abc\")", md5_result, 16);
    // 预期: 900150983cd24fb0d6963f7d28e17f72

    md5((uint8_t *)msg3, strlen(msg3), md5_result);
    print_hash("MD5(\"hello world\")", md5_result, 16);

    // SHA-256测试
    uint8_t sha_result[32];

    sha256((uint8_t *)msg2, strlen(msg2), sha_result);
    print_hash("SHA256(\"abc\")", sha_result, 32);
    // 预期: ba7816bf8f01cfea414140de5dae2223b00361a396177a9cb410ff61f20015ad

    sha256((uint8_t *)msg3, strlen(msg3), sha_result);
    print_hash("SHA256(\"hello world\")", sha_result, 32);

    return 0;
}
```

## 5. 哈希应用

### 5.1 文件完整性校验

```c
/* 计算文件的SHA-256摘要 */
int sha256_file(const char *filename, uint8_t output[32]) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return -1;

    // 分块读取并更新哈希
    // 实际实现应使用增量哈希
    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fseek(fp, 0, SEEK_SET);

    uint8_t *buffer = (uint8_t *)malloc(size);
    fread(buffer, 1, size, fp);
    fclose(fp);

    sha256(buffer, size, output);
    free(buffer);

    return 0;
}
```

### 5.2 密码存储

```c
/* 使用哈希存储密码（简化版，生产中应加盐+慢哈希） */
void hash_password(const char *password, const char *salt,
                   uint8_t output[32]) {
    char combined[256];
    snprintf(combined, sizeof(combined), "%s%s", salt, password);
    sha256((uint8_t *)combined, strlen(combined), output);
}
```

## 小结

- MD5输出128位，SHA-256输出256位，后者更安全
- MD5已被证明不抗碰撞，不应用于安全场景
- SHA-256是当前推荐的哈希算法之一
- 哈希函数的核心操作：位运算、模加、循环移位
- 密码存储应使用bcrypt/scrypt/Argon2等慢哈希算法
