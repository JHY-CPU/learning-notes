# 迭代器模式思想

## 1. 迭代器概述

**迭代器（Iterator）** 是一种设计模式，用于顺序访问集合中的元素，而不暴露集合的内部结构。C语言没有内置的迭代器语法，但我们可以用指针、函数指针和回调函数来模拟迭代器的行为。

## 2. C 语言中的迭代器基础：指针

### 2.1 数组的天然迭代器

C语言中的数组天然支持迭代——指针就是最简单的迭代器：

```c
#include <stdio.h>

int main(void) {
    int arr[] = {10, 20, 30, 40, 50};
    int n = sizeof(arr) / sizeof(arr[0]);

    // 指针作为迭代器
    for (int *p = arr; p < arr + n; p++) {
        printf("%d ", *p);
    }
    printf("\n");

    // 等价的下标写法
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    return 0;
}
```

### 2.2 字符串遍历

```c
#include <stdio.h>

int main(void) {
    char str[] = "Hello";

    // 用指针迭代字符串
    for (char *p = str; *p != '\0'; p++) {
        printf("'%c' (ASCII: %d)\n", *p, *p);
    }

    return 0;
}
```

## 3. 链表的迭代

### 3.1 链表结构

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;
```

### 3.2 手动迭代链表

```c
// 遍历链表打印
void print_list(Node *head) {
    Node *current = head;          // current 就是迭代器

    while (current != NULL) {
        printf("%d -> ", current->data);
        current = current->next;   // 移动迭代器
    }
    printf("NULL\n");
}

// 计算链表长度
int list_length(Node *head) {
    int count = 0;
    for (Node *p = head; p != NULL; p = p->next) {
        count++;
    }
    return count;
}

// 查找元素
Node *find_node(Node *head, int target) {
    for (Node *p = head; p != NULL; p = p->next) {
        if (p->data == target) {
            return p;
        }
    }
    return NULL;
}
```

### 3.3 统一的遍历模式

链表遍历的通用模式：

```c
for (Node *iter = head; iter != NULL; iter = iter->next) {
    // 处理 iter->data
}
```

这与数组遍历的模式一致：
```c
for (int *iter = arr; iter < arr + n; iter++) {
    // 处理 *iter
}
```

## 4. 回调函数遍历

### 4.1 使用函数指针实现回调

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

// 回调函数类型：对每个元素执行的操作
typedef void (*VisitFunc)(int);

// 遍历链表，对每个元素调用回调函数
void foreach_list(Node *head, VisitFunc func) {
    for (Node *p = head; p != NULL; p = p->next) {
        func(p->data);
    }
}

// 各种回调函数
void print_int(int x) {
    printf("%d ", x);
}

void print_if_even(int x) {
    if (x % 2 == 0) {
        printf("%d ", x);
    }
}

void print_square(int x) {
    printf("%d ", x * x);
}

int main(void) {
    // 创建链表 1 -> 2 -> 3 -> 4 -> 5
    Node *head = NULL;
    for (int i = 5; i >= 1; i--) {
        Node *new_node = malloc(sizeof(Node));
        new_node->data = i;
        new_node->next = head;
        head = new_node;
    }

    printf("所有元素: ");
    foreach_list(head, print_int);
    printf("\n");

    printf("偶数: ");
    foreach_list(head, print_if_even);
    printf("\n");

    printf("平方: ");
    foreach_list(head, print_square);
    printf("\n");

    return 0;
}
```

### 4.2 带上下文的回调

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

// 带上下文参数的回调类型
typedef void (*VisitFuncCtx)(int, void *context);

void foreach_with_context(Node *head, VisitFuncCtx func, void *context) {
    for (Node *p = head; p != NULL; p = p->next) {
        func(p->data, context);
    }
}

// 累加器回调
void accumulator(int value, void *context) {
    int *sum = (int *)context;
    *sum += value;
}

// 查找最大值回调
void find_max(int value, void *context) {
    int *max = (int *)context;
    if (value > *max) {
        *max = value;
    }
}

int main(void) {
    Node *head = NULL;
    for (int i = 5; i >= 1; i--) {
        Node *node = malloc(sizeof(Node));
        node->data = i * 10;
        node->next = head;
        head = node;
    }

    int sum = 0;
    foreach_with_context(head, accumulator, &sum);
    printf("总和: %d\n", sum);

    int max = 0;
    foreach_with_context(head, find_max, &max);
    printf("最大值: %d\n", max);

    return 0;
}
```

## 5. 函数式风格的迭代

### 5.1 Map：对每个元素进行变换

```c
#include <stdio.h>
#include <stdlib.h>

// 对数组每个元素应用函数，返回新数组
int *map(int *arr, int n, int (*func)(int)) {
    int *result = malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        result[i] = func(arr[i]);
    }
    return result;
}

int double_value(int x) { return x * 2; }
int square(int x) { return x * x; }

int main(void) {
    int arr[] = {1, 2, 3, 4, 5};
    int n = 5;

    int *doubled = map(arr, n, double_value);
    int *squared = map(arr, n, square);

    printf("原始:  ");
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");

    printf("翻倍:  ");
    for (int i = 0; i < n; i++) printf("%d ", doubled[i]);
    printf("\n");

    printf("平方:  ");
    for (int i = 0; i < n; i++) printf("%d ", squared[i]);
    printf("\n");

    free(doubled);
    free(squared);
    return 0;
}
```

### 5.2 Filter：筛选满足条件的元素

```c
#include <stdio.h>
#include <stdlib.h>

// 筛选满足条件的元素
int *filter(int *arr, int n, int (*pred)(int), int *out_count) {
    int *result = malloc(n * sizeof(int));
    int count = 0;

    for (int i = 0; i < n; i++) {
        if (pred(arr[i])) {
            result[count++] = arr[i];
        }
    }

    *out_count = count;
    return result;
}

int is_even(int x) { return x % 2 == 0; }
int is_positive(int x) { return x > 0; }

int main(void) {
    int arr[] = {-3, -1, 0, 1, 2, 3, 4, 5};
    int n = 8;

    int count;
    int *evens = filter(arr, n, is_even, &count);

    printf("偶数: ");
    for (int i = 0; i < count; i++) printf("%d ", evens[i]);
    printf("\n");

    free(evens);
    return 0;
}
```

### 5.3 Reduce：累积计算

```c
// 累积操作
int reduce(int *arr, int n, int (*func)(int, int), int initial) {
    int acc = initial;
    for (int i = 0; i < n; i++) {
        acc = func(acc, arr[i]);
    }
    return acc;
}

int add(int a, int b) { return a + b; }
int multiply(int a, int b) { return a * b; }
int maximum(int a, int b) { return a > b ? a : b; }

int main(void) {
    int arr[] = {1, 2, 3, 4, 5};
    int n = 5;

    printf("求和: %d\n", reduce(arr, n, add, 0));         // 15
    printf("乘积: %d\n", reduce(arr, n, multiply, 1));     // 120
    printf("最大: %d\n", reduce(arr, n, maximum, arr[0])); // 5

    return 0;
}
```

## 6. 要点总结

1. C语言中的**指针**就是天然的迭代器
2. 链表遍历使用 `current = current->next` 模式
3. **回调函数**（函数指针）可以实现通用的遍历逻辑
4. Map、Filter、Reduce 是函数式编程的三大基本操作，可以用回调函数在C中实现
5. 回调函数可以带上下文参数（void *context）来传递额外信息
6. 这些模式使代码更模块化、更可复用
