# 状态机实现

## 1. 什么是有限状态机

**有限状态机（Finite State Machine, FSM）** 是一种计算模型，由有限个状态、状态之间的转移和触发转移的事件组成。状态机在某一时刻处于一个状态，根据输入事件决定转移到哪个状态。

状态机广泛应用于：词法分析、协议解析、游戏AI、UI交互、编译器设计等。

## 2. 状态机的组成

| 组成部分 | 说明 |
|----------|------|
| 状态（State） | 系统可能处于的有限个状态 |
| 事件/输入（Event/Input） | 触发状态转移的外部信号 |
| 转移（Transition） | 从一个状态到另一个状态的规则 |
| 动作（Action） | 状态转移时执行的操作 |

## 3. 用 switch 实现状态机

### 3.1 基本模式

```c
State current_state = INITIAL_STATE;

while (has_more_input()) {
    Event event = get_next_event();

    switch (current_state) {
        case STATE_A:
            switch (event) {
                case EVENT_1:
                    // 执行动作
                    current_state = STATE_B;
                    break;
                case EVENT_2:
                    current_state = STATE_C;
                    break;
            }
            break;

        case STATE_B:
            // ...
            break;
    }
}
```

### 3.2 简单的自动售货机

```c
#include <stdio.h>

typedef enum {
    STATE_IDLE,        // 空闲，等待投币
    STATE_HAS_COIN,    // 已投币，等待选择
    STATE_DISPENSING   // 出货中
} State;

typedef enum {
    EVENT_COIN,        // 投币
    EVENT_SELECT,      // 选择商品
    EVENT_TIMEOUT,     // 超时
    EVENT_DONE         // 出货完成
} Event;

void state_machine(Event events[], int n) {
    State state = STATE_IDLE;

    for (int i = 0; i < n; i++) {
        Event event = events[i];
        printf("当前状态: %d, 事件: %d → ", state, event);

        switch (state) {
            case STATE_IDLE:
                if (event == EVENT_COIN) {
                    printf("接收硬币\n");
                    state = STATE_HAS_COIN;
                } else {
                    printf("忽略事件\n");
                }
                break;

            case STATE_HAS_COIN:
                if (event == EVENT_SELECT) {
                    printf("开始出货\n");
                    state = STATE_DISPENSING;
                } else if (event == EVENT_TIMEOUT) {
                    printf("超时退币\n");
                    state = STATE_IDLE;
                } else {
                    printf("已投币，请选择\n");
                }
                break;

            case STATE_DISPENSING:
                if (event == EVENT_DONE) {
                    printf("出货完成\n");
                    state = STATE_IDLE;
                } else {
                    printf("正在出货，请等待\n");
                }
                break;
        }
    }
}

int main(void) {
    Event events[] = {
        EVENT_SELECT,   // 没投币就选择
        EVENT_COIN,     // 投币
        EVENT_SELECT,   // 选择商品
        EVENT_DONE,     // 出货完成
        EVENT_COIN,     // 再次投币
        EVENT_TIMEOUT   // 超时
    };

    int n = sizeof(events) / sizeof(events[0]);
    state_machine(events, n);

    return 0;
}
```

输出：
```
当前状态: 0, 事件: 1 → 忽略事件
当前状态: 0, 事件: 0 → 接收硬币
当前状态: 1, 事件: 1 → 开始出货
当前状态: 2, 事件: 3 → 出货完成
当前状态: 0, 事件: 0 → 接收硬币
当前状态: 1, 事件: 2 → 超时退币
```

## 4. 状态转移表

用表格清晰地描述状态转移规则：

```
            │ 投币(COIN)  │ 选择(SELECT) │ 超时(TIMEOUT) │ 完成(DONE)
────────────┼─────────────┼──────────────┼───────────────┼───────────
空闲(IDLE)  │ →已投币     │ 忽略         │ 忽略          │ 忽略
已投币      │ 提示已投币  │ →出货中      │ →空闲(退币)   │ 忽略
出货中      │ 等待        │ 等待         │ 等待          │ →空闲
```

### 4.1 用转移表实现

```c
#include <stdio.h>

#define NUM_STATES 3
#define NUM_EVENTS 4

typedef enum { S_IDLE, S_HAS_COIN, S_DISPENSING } State;
typedef enum { E_COIN, E_SELECT, E_TIMEOUT, E_DONE } Event;

// 转移表：next_state[current_state][event]
State transition_table[NUM_STATES][NUM_EVENTS] = {
    /* IDLE      */ { S_HAS_COIN, S_IDLE,        S_IDLE,        S_IDLE },
    /* HAS_COIN  */ { S_HAS_COIN, S_DISPENSING,  S_IDLE,        S_HAS_COIN },
    /* DISPENSING*/ { S_DISPENSING, S_DISPENSING, S_DISPENSING,  S_IDLE }
};

const char *state_name[] = { "空闲", "已投币", "出货中" };
const char *event_name[] = { "投币", "选择", "超时", "完成" };

void run_fsm(Event events[], int n) {
    State state = S_IDLE;

    for (int i = 0; i < n; i++) {
        Event e = events[i];
        State next = transition_table[state][e];
        printf("[%s] + %s → [%s]\n",
               state_name[state], event_name[e], state_name[next]);
        state = next;
    }
}

int main(void) {
    Event events[] = { E_COIN, E_SELECT, E_DONE, E_COIN, E_TIMEOUT };
    int n = sizeof(events) / sizeof(events[0]);
    run_fsm(events, n);
    return 0;
}
```

## 5. 用 if 实现状态机

### 5.1 词法分析器（识别整数和标识符）

```c
#include <stdio.h>
#include <ctype.h>

typedef enum {
    S_START,
    S_IN_NUMBER,
    S_IN_WORD,
    S_DONE
} State;

void tokenize(const char *input) {
    State state = S_START;
    int i = 0;

    while (input[i] != '\0') {
        char ch = input[i];

        switch (state) {
            case S_START:
                if (isdigit(ch)) {
                    printf("数字开始: %c", ch);
                    state = S_IN_NUMBER;
                } else if (isalpha(ch) || ch == '_') {
                    printf("标识符开始: %c", ch);
                    state = S_IN_WORD;
                } else if (ch == ' ' || ch == '\t') {
                    // 忽略空白
                } else {
                    printf("符号: %c\n", ch);
                }
                break;

            case S_IN_NUMBER:
                if (isdigit(ch)) {
                    printf("%c", ch);
                } else {
                    printf(" (结束)\n");
                    state = S_START;
                    i--;  // 重新处理当前字符
                }
                break;

            case S_IN_WORD:
                if (isalnum(ch) || ch == '_') {
                    printf("%c", ch);
                } else {
                    printf(" (结束)\n");
                    state = S_START;
                    i--;
                }
                break;

            default:
                break;
        }
        i++;
    }

    if (state == S_IN_NUMBER || state == S_IN_WORD) {
        printf(" (结束)\n");
    }
}

int main(void) {
    printf("词法分析:\n");
    tokenize("abc123 456 def _var 789");
    return 0;
}
```

输出：
```
词法分析:
标识符开始: abc123 (结束)
数字开始: 456 (结束)
标识符开始: def (结束)
标识符开始: _var (结束)
数字开始: 789 (结束)
```

## 6. 带动作的状态机

在状态转移时执行特定动作：

```c
#include <stdio.h>

typedef enum { S_IDLE, S_HEATING, S_COOLING } State;
typedef enum { E_TEMP_LOW, E_TEMP_HIGH, E_SHUTDOWN } Event;

void on_start_heating() { printf("  [动作] 开始加热\n"); }
void on_stop_heating()  { printf("  [动作] 停止加热\n"); }
void on_start_cooling() { printf("  [动作] 开始降温\n"); }
void on_stop_cooling()  { printf("  [动作] 停止降温\n"); }
void on_shutdown()      { printf("  [动作] 关闭系统\n"); }

void thermostat(Event events[], int n) {
    State state = S_IDLE;

    for (int i = 0; i < n; i++) {
        Event e = events[i];
        printf("状态=%d, 事件=%d\n", state, e);

        if (state == S_IDLE && e == E_TEMP_LOW) {
            on_start_heating();
            state = S_HEATING;
        } else if (state == S_IDLE && e == E_TEMP_HIGH) {
            on_start_cooling();
            state = S_COOLING;
        } else if (state == S_HEATING && e == E_TEMP_HIGH) {
            on_stop_heating();
            state = S_IDLE;
        } else if (state == S_COOLING && e == E_TEMP_LOW) {
            on_stop_cooling();
            state = S_IDLE;
        } else if (e == E_SHUTDOWN) {
            if (state == S_HEATING) on_stop_heating();
            if (state == S_COOLING) on_stop_cooling();
            on_shutdown();
            state = S_IDLE;
            break;
        }
    }
}

int main(void) {
    Event events[] = { E_TEMP_LOW, E_TEMP_HIGH, E_TEMP_LOW, E_SHUTDOWN };
    thermostat(events, 4);
    return 0;
}
```

## 7. 要点总结

1. 有限状态机由**状态、事件、转移、动作**四要素组成
2. switch-case 是实现状态机最自然的方式
3. **状态转移表**可以清晰地描述状态转移规则，适合状态较多的情况
4. 状态机在编译器、协议解析、游戏开发中广泛应用
5. 注意处理非法状态/事件的组合
6. 实际工程中，状态机常常需要结合动作（函数调用）一起使用
