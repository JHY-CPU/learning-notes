# 循环控制技巧

## 1. 概述

本节介绍循环编程中常用的设计模式和控制技巧，包括哨兵值、标志变量、循环计数器等模式，帮助你写出更高效、更清晰的循环代码。

## 2. 哨兵值模式（Sentinel Value）

哨兵值是一个特殊的标记值，用于表示"数据结束"或"终止条件"。

### 2.1 基本概念

```c
#include <stdio.h>

int main(void) {
    int score;
    int total = 0;
    int count = 0;

    printf("请输入分数（输入-1结束）:\n");
    scanf("%d", &score);

    while (score != -1) {      // -1 是哨兵值
        total += score;
        count++;
        scanf("%d", &score);
    }

    if (count > 0) {
        printf("平均分: %.2f\n", (double)total / count);
    }

    return 0;
}
```

### 2.2 哨兵值的选择原则

1. **不在正常数据范围内**：如温度用 -999，分数用 -1
2. **易于输入和识别**：用户能方便地输入
3. **文档化**：明确告知用户什么值表示结束

```c
// 好的哨兵值选择
while (scanf("%d", &num) == 1)  // EOF 作为哨兵
while (strcmp(line, "quit") != 0)  // "quit" 作为哨兵
while (ch != '\n')  // 换行符作为哨兵
```

### 2.3 综合示例

```c
#include <stdio.h>

int main(void) {
    double price;
    double total = 0.0;
    int item_count = 0;

    printf("购物车计算系统\n");
    printf("请输入商品价格（输入0结账）:\n");

    while (1) {
        printf("商品%d: ¥", item_count + 1);
        scanf("%lf", &price);

        if (price == 0) {         // 0 作为哨兵值
            break;
        }

        if (price < 0) {
            printf("价格无效，请重新输入\n");
            continue;
        }

        total += price;
        item_count++;
    }

    printf("\n===== 账单 =====\n");
    printf("商品数量: %d\n", item_count);
    printf("总金额: ¥%.2f\n", total);

    return 0;
}
```

## 3. 标志变量模式（Flag Variable）

标志变量是一个布尔类型的变量，用于表示某种状态或条件。

### 3.1 基本用法

```c
#include <stdio.h>
#include <stdbool.h>

int main(void) {
    // 判断一个数是否为素数
    int n = 29;
    bool is_prime = true;     // 标志变量

    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            is_prime = false; // 发现因子，修改标志
            break;
        }
    }

    if (is_prime) {
        printf("%d 是素数\n", n);
    } else {
        printf("%d 不是素数\n", n);
    }

    return 0;
}
```

### 3.2 控制循环退出

```c
#include <stdio.h>
#include <stdbool.h>

int main(void) {
    int data[] = {3, 7, 1, 9, 2, 8, 5};
    int n = 7;
    int target = 9;
    bool found = false;       // 标志变量
    int index = -1;

    for (int i = 0; i < n && !found; i++) {  // 标志作为循环条件
        if (data[i] == target) {
            found = true;
            index = i;
        }
    }

    if (found) {
        printf("找到 %d 在索引 %d\n", target, index);
    }

    return 0;
}
```

### 3.3 多标志组合

```c
#include <stdio.h>
#include <stdbool.h>

int main(void) {
    // 验证密码：需要同时满足多个条件
    char password[] = "Abc123!";
    bool has_upper = false;
    bool has_lower = false;
    bool has_digit = false;

    for (int i = 0; password[i] != '\0'; i++) {
        if (password[i] >= 'A' && password[i] <= 'Z') {
            has_upper = true;
        }
        if (password[i] >= 'a' && password[i] <= 'z') {
            has_lower = true;
        }
        if (password[i] >= '0' && password[i] <= '9') {
            has_digit = true;
        }
    }

    if (has_upper && has_lower && has_digit) {
        printf("密码强度合格\n");
    } else {
        printf("密码强度不合格\n");
        if (!has_upper) printf("  缺少大写字母\n");
        if (!has_lower) printf("  缺少小写字母\n");
        if (!has_digit) printf("  缺少数字\n");
    }

    return 0;
}
```

## 4. 循环计数器模式

### 4.1 计数器控制的循环

```c
// 固定次数循环
for (int i = 0; i < N; i++) {
    // 执行N次
}

// 累计操作
int count = 0;
while (condition) {
    count++;      // 统计满足条件的次数
}
```

### 4.2 多计数器

```c
#include <stdio.h>

int main(void) {
    char str[] = "Hello, World! 123";
    int letters = 0, digits = 0, spaces = 0, others = 0;

    for (int i = 0; str[i] != '\0'; i++) {
        if ((str[i] >= 'a' && str[i] <= 'z') ||
            (str[i] >= 'A' && str[i] <= 'Z')) {
            letters++;
        } else if (str[i] >= '0' && str[i] <= '9') {
            digits++;
        } else if (str[i] == ' ') {
            spaces++;
        } else {
            others++;
        }
    }

    printf("字母: %d\n数字: %d\n空格: %d\n其他: %d\n",
           letters, digits, spaces, others);

    return 0;
}
```

### 4.3 累加器与计数器结合

```c
#include <stdio.h>

int main(void) {
    int numbers[] = {12, 45, 78, 34, 56, 89, 23, 67};
    int n = 8;

    int sum = 0;          // 累加器
    int count = 0;        // 计数器
    int max = numbers[0]; // 最大值追踪

    for (int i = 0; i < n; i++) {
        sum += numbers[i];
        count++;
        if (numbers[i] > max) {
            max = numbers[i];
        }
    }

    printf("总和: %d\n", sum);
    printf("个数: %d\n", count);
    printf("平均: %.2f\n", (double)sum / count);
    printf("最大: %d\n", max);

    return 0;
}
```

## 5. 提前退出模式

### 5.1 查找-退出

```c
// 找到第一个满足条件的就退出
int find_first_negative(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        if (arr[i] < 0) {
            return i;     // 找到即返回
        }
    }
    return -1;
}
```

### 5.2 全部满足检查

```c
#include <stdbool.h>

// 判断是否所有元素都满足条件
bool all_positive(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        if (arr[i] <= 0) {
            return false; // 发现一个不满足就返回
        }
    }
    return true;
}

// 判断是否存在满足条件的元素
bool any_positive(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) {
            return true;  // 发现一个满足就返回
        }
    }
    return false;
}
```

## 6. 状态累积模式

### 6.1 字符串构建

```c
#include <stdio.h>

int main(void) {
    char src[] = "Hello, World!";
    char dst[100];
    int j = 0;

    // 只复制字母字符
    for (int i = 0; src[i] != '\0'; i++) {
        if ((src[i] >= 'a' && src[i] <= 'z') ||
            (src[i] >= 'A' && src[i] <= 'Z')) {
            dst[j++] = src[i];
        }
    }
    dst[j] = '\0';

    printf("只含字母: %s\n", dst);
    // 输出: 只含字母: HelloWorld

    return 0;
}
```

### 6.2 统计模式

```c
#include <stdio.h>

int main(void) {
    // 统计最长连续相同字符的长度
    char str[] = "aaabbbccccddeeee";
    int max_len = 1;
    int cur_len = 1;

    for (int i = 1; str[i] != '\0'; i++) {
        if (str[i] == str[i - 1]) {
            cur_len++;
            if (cur_len > max_len) {
                max_len = cur_len;
            }
        } else {
            cur_len = 1;
        }
    }

    printf("最长连续相同字符长度: %d\n", max_len);
    // 输出: 最长连续相同字符长度: 4

    return 0;
}
```

## 7. 要点总结

1. **哨兵值**：用特殊值标记数据结束，避免预知数据量
2. **标志变量**：用布尔变量追踪状态，控制循环和条件判断
3. **循环计数器**：统计次数、追踪最大最小值、累加求和
4. **提前退出**：找到结果立即返回，避免不必要的计算
5. **状态累积**：在循环中逐步构建结果（字符串、统计等）
6. 选择合适的模式能让代码更清晰、更高效
