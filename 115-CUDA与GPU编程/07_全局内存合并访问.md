# 全局内存合并访问

## 合并访问（Coalesced Access）原理

全局内存的带宽虽然很高（A100约2TB/s），但每次访问都有~400周期的延迟。GPU通过**合并访问**来最大化带宽利用率：同一warp内32个线程的内存请求被合并为少数几个内存事务。

```
理想的合并访问（128字节事务，对齐到128字节边界）：

线程:    T0  T1  T2  T3 ... T31
访问:    [0] [1] [2] [3]...[31]  ← 连续地址
         ┌──────────────────────┐
事务:    │    1个128字节事务      │  → 完美合并
         └──────────────────────┘

非合并访问（随机访问）：

线程:    T0  T1  T2  T3 ... T31
访问:    [100][3][88][7]...[201]  ← 分散地址
         ┌──┐┌──┐┌──┐    ┌──┐
事务:    │  ││  ││  │....│  │  → 32个独立事务！
         └──┘└──┘└──┘    └──┘
```

## 合并 vs 非合并访问对比

```cpp
// 合并访问：相邻线程访问相邻内存地址
__global__ void coalesced_read(const float* input, float* output, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        // T0读[0], T1读[1], T2读[2]... → 合并
        output[idx] = input[idx] * 2.0f;
    }
}

// 非合并访问：相邻线程访问不相邻的地址
__global__ void strided_read(const float* input, float* output,
                              int N, int stride) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N / stride) {
        // T0读[0], T1读[stride], T2读[2*stride]...
        // stride > 1 时产生非合并访问
        output[idx] = input[idx * stride] * 2.0f;
    }
}

// 随机访问：最差情况
__global__ void random_read(const float* input, float* output,
                             const int* indices, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        // 每个线程访问完全随机的地址 → 无法合并
        output[idx] = input[indices[idx]] * 2.0f;
    }
}
```

## 内存对齐

全局内存访问应该对齐到32字节或128字节边界：

```cpp
// 未对齐访问示例
__global__ void misaligned_access(const float* input, float* output, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N - 1) {
        // input + 1 导致所有线程的访问都偏移4字节
        // 整个warp的访问不再对齐到128字节边界
        output[idx] = input[idx + 1] * 2.0f;
    }
}

// 对齐的访问
__global__ void aligned_access(const float* input, float* output, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        output[idx] = input[idx] * 2.0f;  // 从边界开始，完美对齐
    }
}

// 确保分配的内存对齐
void allocate_aligned(float** d_ptr, int N) {
    // cudaMalloc保证返回256字节对齐的内存
    cudaMalloc(d_ptr, N * sizeof(float));
}
```

## 结构体的内存布局

结构体数组（AoS）vs 数组结构体（SoA）对合并访问影响巨大：

```cpp
// AoS (Array of Structures) - 不利于GPU
struct Particle_AoS {
    float x, y, z;     // 12字节位置
    float vx, vy, vz;  // 12字节速度
    float mass;         // 4字节质量
    // 总计28字节，warp内线程访问x时，间隔28字节
};

__global__ void update_aos(Particle_AoS* particles, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        // 线程0读x在offset 0, 线程1读x在offset 28
        // 28不是32的倍数，无法良好合并
        particles[idx].x += particles[idx].vx;
        particles[idx].y += particles[idx].vy;
        particles[idx].z += particles[idx].vz;
    }
}

// SoA (Structure of Arrays) - 利于GPU
struct Particles_SoA {
    float* x;   // 所有x连续存储
    float* y;   // 所有y连续存储
    float* z;   // 所有z连续存储
    float* vx;
    float* vy;
    float* vz;
    float* mass;
};

__global__ void update_soa(Particles_SoA particles, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        // 线程0读x[0], 线程1读x[1]... → 完美合并
        particles.x[idx] += particles.vx[idx];
        particles.y[idx] += particles.vy[idx];
        particles.z[idx] += particles.vz[idx];
    }
}
```

## 分块读取优化

当数据布局不适合直接合并访问时，重新排列读取方式：

```cpp
// 矩阵按列读取 - 非合并访问
__global__ void column_read_bad(const float* matrix, float* output,
                                 int rows, int cols) {
    int col = threadIdx.x + blockIdx.x * blockDim.x;
    if (col < cols) {
        for (int row = 0; row < rows; row++) {
            // 同一warp内线程读取 matrix[0*cols+col], matrix[1*cols+col]...
            // 相邻线程读取间隔cols个float（4*cols字节）→ 非合并
            output[row * cols + col] = matrix[row * cols + col];
        }
    }
}

// 优化：使用共享内存先按行读取再转置
__global__ void column_read_shared(const float* matrix, float* output,
                                    int rows, int cols) {
    __shared__ float tile[32][32 + 1];  // padding避免bank conflict

    for (int base = 0; base < rows; base += 32) {
        int r = base + threadIdx.y;
        int c = blockIdx.x * 32 + threadIdx.x;

        // 先按行合并读取到共享内存
        if (r < rows && c < cols) {
            tile[threadIdx.y][threadIdx.x] = matrix[r * cols + c];
        }
        __syncthreads();

        // 从共享内存按列写出（转置访问）
        int out_r = base + threadIdx.x;
        int out_c = blockIdx.x * 32 + threadIdx.y;
        if (out_r < rows && out_c < cols) {
            output[out_r * cols + out_c] = tile[threadIdx.x][threadIdx.y];
        }
        __syncthreads();
    }
}
```

## Texture内存利用空间局部性

对于非合并访问模式（如双线性插值），texture内存可以提供缓存加速：

```cpp
// 使用L2持久化缓存提示（CUDA 11.0+）
__global__ void cached_access(const float* __restrict__ input,
                               float* output, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        // __restrict__告诉编译器指针无别名，可以更激进优化
        // 编译器可能使用L2缓存策略
        output[idx] = input[idx] * input[idx];
    }
}
```

## L2缓存持久化（CUDA 11.0+）

```cpp
// 使用cudaAccessPolicyWindow设置L2持久化策略
void setup_l2_persistent(float* d_data, size_t data_size) {
    // 将数据保留在L2缓存中
    cudaMemPool_t mempool;
    cudaDeviceGetDefaultMemPool(&mempool, 0);

    // L2缓存持久化窗口
    cudaMemPoolProps props = {};
    props.allocType = cudaMemAllocationTypePinned;
    props.location.type = cudaMemLocationTypeDevice;
    props.location.id = 0;

    // 或者使用访问策略
    cudaMemAccessDesc desc = {};
    desc.location.type = cudaMemLocationTypeDevice;
    desc.location.id = 0;
    desc.flags = cudaMemAccessFlagsProtReadWrite;

    cudaMemPoolSetAccess(mempool, &desc, 1);
}
```

## 性能测量与诊断

```cpp
#include <cstdio>
#include <cuda_runtime.h>

// 测量不同访问模式的带宽
__global__ void bandwidth_test(float* src, float* dst, int N, int stride) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N / stride) {
        dst[idx] = src[idx * stride];
    }
}

void measure_bandwidth() {
    int N = 1024 * 1024;
    float *d_src, *d_dst;
    cudaMalloc(&d_src, N * sizeof(float));
    cudaMalloc(&d_dst, N * sizeof(float));

    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);

    int strides[] = {1, 2, 4, 8, 16, 32};
    for (int s = 0; s < 6; s++) {
        int stride = strides[s];
        int threads = 256;
        int blocks = (N / stride + threads - 1) / threads;

        // Warm up
        bandwidth_test<<<blocks, threads>>>(d_src, d_dst, N, stride);

        cudaEventRecord(start);
        for (int i = 0; i < 100; i++) {
            bandwidth_test<<<blocks, threads>>>(d_src, d_dst, N, stride);
        }
        cudaEventRecord(stop);
        cudaEventSynchronize(stop);

        float ms;
        cudaEventElapsedTime(&ms, start, stop);

        float bytes = (float)(N / stride) * sizeof(float) * 100;
        float gb_s = bytes / (ms / 1000.0f) / 1e9;

        printf("Stride %2d: %.2f GB/s\n", stride, gb_s);
    }

    cudaEventDestroy(start);
    cudaEventDestroy(stop);
    cudaFree(d_src);
    cudaFree(d_dst);
}
```

## 优化检查清单

| 检查项 | 好的做法 | 坏的做法 |
|--------|----------|----------|
| 访问模式 | 相邻线程访问相邻地址 | 随机/跨步访问 |
| 数据布局 | SoA（结构体数组） | AoS（数组结构体） |
| 对齐 | 从128字节边界开始 | 偏移几个字节 |
| 数组维度 | 行优先读取 | 列优先读取（stride=cols） |
| 指针别名 | 使用`__restrict__` | 多个指针可能重叠 |

## 小结

1. 合并访问是全局内存优化的基础：相邻线程访问相邻地址
2. 内存对齐到32/128字节边界避免额外的内存事务
3. SoA布局优于AoS布局，因为同一字段的访问天然合并
4. 使用共享内存作为中介可以将非合并访问转化为合并访问
5. 编译器提示（`__restrict__`）和L2缓存策略可以进一步优化
