# 统一内存与托管内存

## 统一内存概述

CUDA统一内存（Unified Memory）创建一个CPU和GPU都能访问的托管内存池，简化了内存管理。驱动程序自动在CPU和GPU之间迁移数据。

```
传统模型：
CPU内存 ←──cudaMemcpy──→ GPU内存
  Host          HostToDevice/DeviceToHost         Device

统一内存模型：
┌─────────────────────────────────────────────┐
│           统一内存（Managed Memory）           │
│  ┌─────────────┐     ┌─────────────┐        │
│  │  CPU访问     │     │  GPU访问     │        │
│  └─────────────┘     └─────────────┘        │
│        ↕ 自动页面迁移 ↕                      │
│  ┌─────────────┐     ┌─────────────┐        │
│  │ CPU物理页面  │     │ GPU物理页面  │        │
│  └─────────────┘     └─────────────┘        │
└─────────────────────────────────────────────┘
```

## 基本使用

```cpp
#include <cstdio>
#include <cuda_runtime.h>

__global__ void add_one(float* data, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        data[idx] += 1.0f;
    }
}

int main() {
    int N = 1024 * 1024;

    // 分配统一内存：CPU和GPU都可以直接访问
    float* data;
    cudaMallocManaged(&data, N * sizeof(float));

    // CPU端初始化
    for (int i = 0; i < N; i++) {
        data[i] = (float)i;
    }

    // GPU端修改 - 无需cudaMemcpy
    add_one<<<(N + 255) / 256, 256>>>(data, N);
    cudaDeviceSynchronize();

    // CPU端读取结果 - 无需cudaMemcpy
    float sum = 0.0f;
    for (int i = 0; i < N; i++) {
        sum += data[i];
    }
    printf("Sum: %f\n", sum);

    // 释放统一内存
    cudaFree(data);
    return 0;
}
```

## 页面迁移机制

统一内存使用按需页面迁移（demand paging）：

```cpp
// 演示页面迁移
__global__ void gpu_touch(float* data, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        data[idx] *= 2.0f;
    }
}

void demonstrate_page_migration() {
    int N = 1024 * 1024;
    float* data;
    cudaMallocManaged(&data, N * sizeof(float));

    // 此时数据在CPU上（或尚未分配物理页面）

    // 首次GPU访问：触发页面迁移到GPU
    gpu_touch<<<(N + 255) / 256, 256>>>(data, N);
    cudaDeviceSynchronize();
    // 数据现在在GPU上

    // CPU访问：触发页面迁回CPU
    float val = data[0];
    // 数据现在在CPU上

    printf("First element: %f\n", val);
    cudaFree(data);
}
```

## 预取优化

`cudaMemPrefetchAsync`可以提前将数据迁移到目标设备，避免页面错误延迟：

```cpp
void prefetch_example() {
    int N = 1024 * 1024;
    float* data;
    cudaMallocManaged(&data, N * sizeof(float));

    cudaStream_t stream;
    cudaStreamCreate(&stream);

    int deviceId;
    cudaGetDevice(&deviceId);

    // CPU初始化
    for (int i = 0; i < N; i++) {
        data[i] = (float)i;
    }

    // 预取到GPU（异步）
    cudaMemPrefetchAsync(data, N * sizeof(float), deviceId, stream);

    // GPU计算
    gpu_touch<<<(N + 255) / 256, 256, 0, stream>>>(data, N);

    // 预取回CPU
    cudaMemPrefetchAsync(data, N * sizeof(float), cudaCpuDeviceId, stream);

    cudaStreamSynchronize(stream);

    // CPU现在可以直接访问，没有页面错误
    printf("After prefetch: data[0] = %f\n", data[0]);

    cudaStreamDestroy(stream);
    cudaFree(data);
}
```

## 内存访问建议

`cudaMemAdvise`向驱动程序提供内存访问模式的提示：

```cpp
void memory_advice_example() {
    int N = 1024 * 1024;
    float* data;
    cudaMallocManaged(&data, N * sizeof(float));

    int deviceId;
    cudaGetDevice(&deviceId);

    // 提示：数据将被GPU频繁读取
    cudaMemAdvise(data, N * sizeof(float), cudaMemAdviseSetReadMostly, deviceId);

    // 提示：数据将被特定GPU访问
    cudaMemAdvise(data, N * sizeof(float), cudaMemAdviseSetAccessedBy, deviceId);

    // 提示：数据将被首选在某个设备上
    cudaMemAdvise(data, N * sizeof(float), cudaMemAdviseSetPreferredLocation, deviceId);

    // GPU读取 - 因为ReadMostly提示，每个GPU会得到自己的副本
    gpu_touch<<<(N + 255) / 256, 256>>>(data, N);

    // 取消ReadMostly提示
    cudaMemAdvise(data, N * sizeof(float), cudaMemAdviseUnsetReadMostly, deviceId);

    cudaFree(data);
}
```

## 内存访问建议总结

| 建议 | 含义 | 使用场景 |
|------|------|----------|
| `cudaMemAdviseSetReadMostly` | 数据主要是只读的 | GPU会缓存副本 |
| `cudaMemAdviseSetPreferredLocation` | 首选保留在某设备 | 已知数据归属 |
| `cudaMemAdviseSetAccessedBy` | 将被某设备访问 | 提前映射页表 |

## 多GPU统一内存

```cpp
__global__ void process_on_gpu(float* data, int N, float scale) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        data[idx] *= scale;
    }
}

void multi_gpu_unified() {
    int N = 1024 * 1024;
    float* data;
    cudaMallocManaged(&data, N * sizeof(float));

    int numDevices;
    cudaGetDeviceCount(&numDevices);

    // CPU初始化
    for (int i = 0; i < N; i++) {
        data[i] = 1.0f;
    }

    // 每个GPU处理一部分数据
    int chunkSize = N / numDevices;
    for (int dev = 0; dev < numDevices; dev++) {
        cudaSetDevice(dev);

        cudaStream_t stream;
        cudaStreamCreate(&stream);

        // 预取到当前GPU
        cudaMemPrefetchAsync(data + dev * chunkSize, chunkSize * sizeof(float),
                              dev, stream);

        // GPU处理
        process_on_gpu<<<chunkSize / 256, 256, 0, stream>>>(
            data + dev * chunkSize, chunkSize, (float)(dev + 1));

        cudaStreamDestroy(stream);
    }

    // 同步所有GPU
    cudaDeviceSynchronize();

    // 预取回CPU
    cudaMemPrefetchAsync(data, N * sizeof(float), cudaCpuDeviceId, 0);
    cudaDeviceSynchronize();

    // 验证
    printf("data[0] = %f (GPU 0)\n", data[0]);
    printf("data[%d] = %f (GPU 1)\n", chunkSize, data[chunkSize]);

    cudaFree(data);
}
```

```cpp
void check_unified_memory_support() {
    int deviceCount;
    cudaGetDeviceCount(&deviceCount);

    for (int i = 0; i < deviceCount; i++) {
        cudaDeviceProp prop;
        cudaGetDeviceProperties(&prop, i);

        printf("Device %d: %s\n", i, prop.name);
        printf("  Unified Memory: %s\n", prop.unifiedAddressing ? "Yes" : "No");
        printf("  Managed Memory: %s\n", prop.managedMemory ? "Yes" : "No");
        printf("  Concurrent Managed Access: %s\n",
               prop.concurrentManagedAccess ? "Yes" : "No");
    }
}
```

要求：
- GPU Compute Capability 3.0+
- 64位操作系统
- Pascal及以上架构（CC 6.0+）支持真正的按需页面迁移

## 性能考虑

```cpp
// 统一内存 vs 显式传输的性能比较
void compare_performance() {
    int N = 1024 * 1024 * 100;  // 400MB
    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);

    // 方案1：显式传输
    {
        float *h_data, *d_data;
        cudaMallocHost(&h_data, N * sizeof(float));
        cudaMalloc(&d_data, N * sizeof(float));

        cudaEventRecord(start);
        cudaMemcpy(d_data, h_data, N * sizeof(float), cudaMemcpyHostToDevice);
        gpu_touch<<<N / 256, 256>>>(d_data, N);
        cudaMemcpy(h_data, d_data, N * sizeof(float), cudaMemcpyDeviceToHost);
        cudaEventRecord(stop);
        cudaEventSynchronize(stop);

        float ms;
        cudaEventElapsedTime(&ms, start, stop);
        printf("Explicit transfer: %.3f ms\n", ms);

        cudaFreeHost(h_data);
        cudaFree(d_data);
    }

    // 方案2：统一内存 + 预取
    {
        float* data;
        cudaMallocManaged(&data, N * sizeof(float));
        int deviceId;
        cudaGetDevice(&deviceId);

        cudaEventRecord(start);
        cudaMemPrefetchAsync(data, N * sizeof(float), deviceId, 0);
        gpu_touch<<<N / 256, 256>>>(data, N);
        cudaMemPrefetchAsync(data, N * sizeof(float), cudaCpuDeviceId, 0);
        cudaEventRecord(stop);
        cudaEventSynchronize(stop);

        float ms;
        cudaEventElapsedTime(&ms, start, stop);
        printf("Unified memory + prefetch: %.3f ms\n", ms);

        cudaFree(data);
    }

    cudaEventDestroy(start);
    cudaEventDestroy(stop);
}
```

## 小结

1. 统一内存通过`cudaMallocManaged`分配，CPU和GPU都能直接访问
2. 页面迁移由驱动程序自动管理，但有性能开销
3. `cudaMemPrefetchAsync`提前迁移数据，避免页面错误延迟
4. `cudaMemAdvise`提供访问模式提示，帮助驱动优化
5. 统一内存大幅简化编程，但在性能关键场景需要仔细优化
6. 建议在原型开发阶段使用统一内存，在性能优化阶段切换到显式管理
