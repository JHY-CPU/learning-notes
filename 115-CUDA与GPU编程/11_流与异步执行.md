# 流与异步执行

## CUDA流的概念

CUDA流（Stream）是一系列按顺序执行的GPU操作。不同流之间的操作可以并发执行，这是实现计算与通信重叠的关键。

```
单流（串行）：
Host: [Launch K1] → [Launch K2] → [Memcpy D→H]
GPU:  [K1][K2][Memcpy]

多流（并发）：
Host: [Launch K1][Launch K2][Launch K3][Launch K4]
Stream 0: [K1]            [Memcpy H→D(3)]
Stream 1:      [K2]            [K4]
Stream 2:           [Memcpy H→D(2)]
```

## 流的基本操作

```cpp
#include <cstdio>
#include <cuda_runtime.h>

__global__ void kernel(float* data, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        data[idx] = data[idx] * 2.0f + 1.0f;
    }
}

int main() {
    int N = 1024 * 1024;
    size_t bytes = N * sizeof(float);

    // 创建流
    cudaStream_t stream0, stream1;
    cudaStreamCreate(&stream0);
    cudaStreamCreate(&stream1);

    float *d_data0, *d_data1;
    cudaMalloc(&d_data0, bytes);
    cudaMalloc(&d_data1, bytes);

    // 在不同流上启动kernel
    kernel<<<4096, 256, 0, stream0>>>(d_data0, N);
    kernel<<<4096, 256, 0, stream1>>>(d_data1, N);

    // 同步特定流
    cudaStreamSynchronize(stream0);
    cudaStreamSynchronize(stream1);

    // 销毁流
    cudaStreamDestroy(stream0);
    cudaStreamDestroy(stream1);
    cudaFree(d_data0);
    cudaFree(d_data1);

    return 0;
}
```

## 计算与通信重叠

这是CUDA流最重要的应用场景：在GPU计算的同时进行数据传输。

```cpp
__global__ void compute_kernel(float* data, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        float x = data[idx];
        for (int i = 0; i < 100; i++) {
            x = x * 1.001f + 0.001f;
        }
        data[idx] = x;
    }
}

void overlap_compute_transfer() {
    int N = 1024 * 1024;
    int numChunks = 4;
    int chunkSize = N / numChunks;
    size_t chunkBytes = chunkSize * sizeof(float);

    float *h_data;
    cudaMallocHost(&h_data, N * sizeof(float));  // 锁页内存（必须的！）

    float *d_data;
    cudaMalloc(&d_data, N * sizeof(float));

    // 创建多个流
    cudaStream_t streams[4];
    for (int i = 0; i < numChunks; i++) {
        cudaStreamCreate(&streams[i]);
    }

    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);
    cudaEventRecord(start);

    // 流水线：计算chunk i的同时传输chunk i+1
    for (int i = 0; i < numChunks; i++) {
        int offset = i * chunkSize;

        // 异步传输：Host → Device
        cudaMemcpyAsync(d_data + offset, h_data + offset,
                        chunkBytes, cudaMemcpyHostToDevice, streams[i]);

        // 在同一个流上启动kernel（会等待传输完成再执行）
        compute_kernel<<<chunkSize / 256, 256, 0, streams[i]>>>(
            d_data + offset, chunkSize);

        // 异步传输：Device → Host
        cudaMemcpyAsync(h_data + offset, d_data + offset,
                        chunkBytes, cudaMemcpyDeviceToHost, streams[i]);
    }

    // 等待所有流完成
    for (int i = 0; i < numChunks; i++) {
        cudaStreamSynchronize(streams[i]);
    }

    cudaEventRecord(stop);
    cudaEventSynchronize(stop);

    float ms;
    cudaEventElapsedTime(&ms, start, stop);
    printf("Overlapped execution: %.3f ms\n", ms);

    // 清理
    for (int i = 0; i < numChunks; i++) {
        cudaStreamDestroy(streams[i]);
    }
    cudaEventDestroy(start);
    cudaEventDestroy(stop);
    cudaFreeHost(h_data);
    cudaFree(d_data);
}
```

## 锁页内存（Pinned Memory）

异步数据传输**必须**使用锁页内存（cudaMallocHost分配的内存）：

```cpp
// 分配锁页内存
float* h_pinned;
cudaMallocHost(&h_pinned, N * sizeof(float));  // 物理上锁定，不被换出
// 使用完毕后
cudaFreeHost(h_pinned);

// 对比普通malloc
float* h_normal = (float*)malloc(N * sizeof(float));
// cudaMemcpyAsync 不能用于 h_normal！只能用 cudaMemcpy
free(h_normal);
```

锁页内存的优势：
- 支持异步传输（cudaMemcpyAsync）
- 传输带宽更高（DMA直接访问）
- 缺点：分配和释放较慢，占用物理内存

## CUDA事件与流同步

```cpp
void event_based_sync() {
    int N = 1024 * 1024;
    float *d_data;
    cudaMalloc(&d_data, N * sizeof(float));

    cudaStream_t stream0, stream1;
    cudaStreamCreate(&stream0);
    cudaStreamCreate(&stream1);

    cudaEvent_t event0_done;
    cudaEventCreate(&event0_done);

    // 流0：执行kernel
    kernel<<<4096, 256, 0, stream0>>>(d_data, N);

    // 在流0中记录事件
    cudaEventRecord(event0_done, stream0);

    // 流1等待流0的事件完成
    cudaStreamWaitEvent(stream1, event0_done, 0);

    // 流1中的kernel会在流0完成后才执行
    kernel<<<4096, 256, 0, stream1>>>(d_data, N);

    cudaStreamSynchronize(stream1);

    cudaEventDestroy(event0_done);
    cudaStreamDestroy(stream0);
    cudaStreamDestroy(stream1);
    cudaFree(d_data);
}
```

## 流优先级（CUDA 8.0+）

```cpp
void stream_priority() {
    int leastPriority, greatestPriority;
    cudaDeviceGetStreamPriorityRange(&leastPriority, &greatestPriority);
    printf("Priority range: %d to %d\n", greatestPriority, leastPriority);

    cudaStream_t highPri, lowPri;
    // 创建高优先级流
    cudaStreamCreateWithPriority(&highPri, cudaStreamNonBlocking, greatestPriority);
    // 创建低优先级流
    cudaStreamCreateWithPriority(&lowPri, cudaStreamNonBlocking, leastPriority);

    // 高优先级流上的操作会优先调度
    kernel<<<4096, 256, 0, highPri>>>(d_data, N);  // 优先执行
    kernel<<<4096, 256, 0, lowPri>>>(d_data, N);    // 排队等待

    cudaStreamDestroy(highPri);
    cudaStreamDestroy(lowPri);
}
```

## 回调函数

CUDA流支持回调函数，在流中操作完成后在主机线程执行：

```cpp
void CUDART_CB myCallback(cudaStream_t stream, cudaError_t status, void* userData) {
    int* chunkId = (int*)userData;
    printf("Stream completed for chunk %d, status: %s\n",
           *chunkId, cudaGetErrorString(status));
}

void stream_callbacks() {
    int N = 1024 * 1024;
    int numStreams = 4;
    int chunkSize = N / numStreams;

    float *d_data;
    cudaMalloc(&d_data, N * sizeof(float));

    cudaStream_t streams[4];
    int chunkIds[4];

    for (int i = 0; i < numStreams; i++) {
        cudaStreamCreate(&streams[i]);
        chunkIds[i] = i;

        kernel<<<chunkSize / 256, 256, 0, streams[i]>>>(d_data + i * chunkSize, chunkSize);

        // 添加回调：kernel完成后调用
        cudaStreamAddCallback(streams[i], myCallback, &chunkIds[i], 0);
    }

    cudaDeviceSynchronize();

    for (int i = 0; i < numStreams; i++) {
        cudaStreamDestroy(streams[i]);
    }
    cudaFree(d_data);
}
```

## 完整的流水线示例

```cpp
// 数据处理流水线：H2D → Compute → D2H 完全重叠
void pipeline_processing() {
    const int N = 1024 * 1024;
    const int numStreams = 4;
    const int chunkSize = N / numStreams;
    const size_t chunkBytes = chunkSize * sizeof(float);

    // 分配锁页主机内存
    float *h_input, *h_output;
    cudaMallocHost(&h_input, N * sizeof(float));
    cudaMallocHost(&h_output, N * sizeof(float));

    // 初始化
    for (int i = 0; i < N; i++) h_input[i] = (float)i;

    // 分配设备内存
    float *d_input, *d_output;
    cudaMalloc(&d_input, N * sizeof(float));
    cudaMalloc(&d_output, N * sizeof(float));

    // 创建流和事件
    cudaStream_t streams[numStreams];
    cudaEvent_t events[numStreams][3];  // [i][0]=H2D done, [i][1]=compute done, [i][2]=D2H done

    for (int i = 0; i < numStreams; i++) {
        cudaStreamCreateWithFlags(&streams[i], cudaStreamNonBlocking);
        for (int j = 0; j < 3; j++) {
            cudaEventCreate(&events[i][j]);
        }
    }

    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);
    cudaEventRecord(start);

    // 流水线处理
    for (int i = 0; i < numStreams; i++) {
        int offset = i * chunkSize;

        // H2D传输
        cudaMemcpyAsync(d_input + offset, h_input + offset,
                        chunkBytes, cudaMemcpyHostToDevice, streams[i]);
        cudaEventRecord(events[i][0], streams[i]);

        // 计算（等待H2D完成）
        kernel<<<chunkSize / 256, 256, 0, streams[i]>>>(d_input + offset, chunkSize);
        cudaEventRecord(events[i][1], streams[i]);

        // D2H传输（等待计算完成）
        cudaMemcpyAsync(h_output + offset, d_input + offset,
                        chunkBytes, cudaMemcpyDeviceToHost, streams[i]);
        cudaEventRecord(events[i][2], streams[i]);
    }

    // 等待所有流完成
    cudaDeviceSynchronize();

    cudaEventRecord(stop);
    cudaEventSynchronize(stop);

    float ms;
    cudaEventElapsedTime(&ms, start, stop);
    printf("Pipeline time: %.3f ms\n", ms);

    // 打印各阶段时间（用于分析重叠效果）
    for (int i = 0; i < numStreams; i++) {
        float h2d_ms, compute_ms, d2h_ms;
        cudaEventElapsedTime(&h2d_ms, events[i][0], events[i][1]);
        cudaEventElapsedTime(&compute_ms, events[i][1], events[i][2]);
        printf("Stream %d: H2D=%.3fms Compute=%.3fms D2H=%.3fms\n",
               i, h2d_ms, compute_ms, d2h_ms);
    }

    // 清理
    for (int i = 0; i < numStreams; i++) {
        cudaStreamDestroy(streams[i]);
        for (int j = 0; j < 3; j++) cudaEventDestroy(events[i][j]);
    }
    cudaEventDestroy(start);
    cudaEventDestroy(stop);
    cudaFreeHost(h_input);
    cudaFreeHost(h_output);
    cudaFree(d_input);
    cudaFree(d_output);
}
```

## 小结

1. CUDA流提供操作的并行执行能力，不同流之间的kernel和传输可以并发
2. 异步数据传输（cudaMemcpyAsync）必须使用锁页内存（cudaMallocHost）
3. 计算与通信重叠是流的主要应用场景，可以隐藏数据传输延迟
4. CUDA事件用于流间同步和精确计时
5. 流水线技术将大数据分成小块，在不同流上交替进行传输和计算
