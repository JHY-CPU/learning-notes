# 归约算法

## 归约问题概述

归约（Reduction）是将一组数据合并为单个值的操作，如求和、求最大值、求最小值。这是并行计算中最基本也最重要的算法模式之一。

```
归约求和示例（8个元素）:
[1] [2] [3] [4] [5] [6] [7] [8]
 \   /     \   /     \   /     \   /
 [3]       [7]       [11]      [15]
   \         /         \         /
   [10]                 [26]
     \                   /
          [36]
```

## 朴素归约（Naive Reduction）

```cpp
// 方案1：所有线程做原子操作 - 效率最低
__global__ void reduce_naive_atomic(const float* input, float* output, int N) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx < N) {
        atomicAdd(output, input[idx]);  // 每个元素一次全局内存原子操作
    }
}

// 方案2：每个Block归约到一个值，再原子加
__global__ void reduce_block(const float* input, float* output, int N) {
    extern __shared__ float s_data[];

    int tid = threadIdx.x;
    int idx = threadIdx.x + blockIdx.x * blockDim.x;

    // 加载到共享内存
    s_data[tid] = (idx < N) ? input[idx] : 0.0f;
    __syncthreads();

    // 朴素的归约 - 有bank conflict和线程发散问题
    for (int s = 1; s < blockDim.x; s *= 2) {
        if (tid % (2 * s) == 0) {  // 问题：同一warp内线程走不同分支
            s_data[tid] += s_data[tid + s];
        }
        __syncthreads();
    }

    if (tid == 0) {
        atomicAdd(output, s_data[0]);
    }
}
```

## 优化1：消除线程发散

```cpp
// 避免warp内分支发散
__global__ void reduce_no_divergence(const float* input, float* output, int N) {
    extern __shared__ float s_data[];

    int tid = threadIdx.x;
    int idx = threadIdx.x + blockIdx.x * blockDim.x;

    s_data[tid] = (idx < N) ? input[idx] : 0.0f;
    __syncthreads();

    // 改变循环方式：让连续线程参与
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {  // 前半部分线程参与
            s_data[tid] += s_data[tid + s];
        }
        __syncthreads();
    }

    if (tid == 0) {
        atomicAdd(output, s_data[0]);
    }
}
```

## 优化2：消除Bank Conflict

```cpp
// 使用交错归约避免bank conflict
__global__ void reduce_no_bank_conflict(const float* input, float* output, int N) {
    extern __shared__ float s_data[];

    int tid = threadIdx.x;
    int idx = threadIdx.x + blockIdx.x * blockDim.x;

    s_data[tid] = (idx < N) ? input[idx] : 0.0f;
    __syncthreads();

    // 从后往前归约
    for (int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) {
            s_data[tid] += s_data[tid + s];
        }
        __syncthreads();
    }

    // 最后一个warp内不需要__syncthreads()
    // warp内线程天然同步
    if (tid < 32) {
        volatile float* vs_data = s_data;  // volatile确保每次写入都到达共享内存
        vs_data[tid] += vs_data[tid + 32];
        vs_data[tid] += vs_data[tid + 16];
        vs_data[tid] += vs_data[tid + 8];
        vs_data[tid] += vs_data[tid + 4];
        vs_data[tid] += vs_data[tid + 2];
        vs_data[tid] += vs_data[tid + 1];
    }

    if (tid == 0) {
        atomicAdd(output, s_data[0]);
    }
}
```

## 优化3：一个线程处理多个元素

```cpp
// 每个线程处理多个元素，增加算术强度
__global__ void reduce_multi_element(const float* input, float* output, int N) {
    extern __shared__ float s_data[];

    int tid = threadIdx.x;
    int idx = threadIdx.x + blockIdx.x * blockDim.x * 2;  // 跳跃索引

    // 每个线程累加两个元素（grid大小减半）
    float sum = 0.0f;
    if (idx < N) sum += input[idx];
    if (idx + blockDim.x < N) sum += input[idx + blockDim.x];

    s_data[tid] = sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) {
            s_data[tid] += s_data[tid + s];
        }
        __syncthreads();
    }

    if (tid < 32) {
        volatile float* vs_data = s_data;
        vs_data[tid] += vs_data[tid + 32];
        vs_data[tid] += vs_data[tid + 16];
        vs_data[tid] += vs_data[tid + 8];
        vs_data[tid] += vs_data[tid + 4];
        vs_data[tid] += vs_data[tid + 2];
        vs_data[tid] += vs_data[tid + 1];
    }

    if (tid == 0) {
        output[blockIdx.x] = s_data[0];  // 每个Block输出一个值
    }
}
```

## Warp Shuffle归约

Warp shuffle允许warp内线程直接交换寄存器值，不经过共享内存：

```cpp
__device__ float warp_reduce_sum(float val) {
    // 使用warp shuffle进行归约
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xFFFFFFFF, val, offset);
    }
    return val;  // 返回warp内第一个线程的值
}

__global__ void reduce_warp_shuffle(const float* input, float* output, int N) {
    extern __shared__ float s_data[];

    int tid = threadIdx.x;
    int idx = threadIdx.x + blockIdx.x * blockDim.x * 2;

    // 每个线程累加多个元素
    float sum = 0.0f;
    int gridSize = blockDim.x * 2 * gridDim.x;
    for (int i = idx; i < N; i += gridSize) {
        sum += input[i];
        if (i + blockDim.x < N) sum += input[i + blockDim.x];
    }

    // warp内归约
    sum = warp_reduce_sum(sum);

    // 每个warp的第一个线程写入共享内存
    if (threadIdx.x % 32 == 0) {
        s_data[threadIdx.x / 32] = sum;
    }
    __syncthreads();

    // 第一个warp归约所有warp的结果
    if (threadIdx.x < 32) {
        sum = (threadIdx.x < (blockDim.x + 31) / 32) ? s_data[threadIdx.x] : 0.0f;
        sum = warp_reduce_sum(sum);
    }

    if (threadIdx.x == 0) {
        output[blockIdx.x] = sum;
    }
}
```

## 完整的归约框架

```cpp
#include <cstdio>
#include <cuda_runtime.h>

#define BLOCK_SIZE 256

// warp归约
__device__ float warp_reduce(float val) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xFFFFFFFF, val, offset);
    }
    return val;
}

// Block归约
__device__ float block_reduce(float val) {
    __shared__ float s_data[32];
    int lane = threadIdx.x % 32;
    int warp_id = threadIdx.x / 32;

    // warp内归约
    val = warp_reduce(val);

    // warp结果写入共享内存
    if (lane == 0) s_data[warp_id] = val;
    __syncthreads();

    // 第一个warp归约所有warp
    if (warp_id == 0) {
        val = (lane < (blockDim.x + 31) / 32) ? s_data[lane] : 0.0f;
        val = warp_reduce(val);
    }
    return val;
}

// 归约kernel
__global__ void reduce_kernel(const float* input, float* output, int N) {
    float sum = 0.0f;
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    int stride = blockDim.x * gridDim.x;

    // Grid-stride loop：每个线程累加多个元素
    for (int i = idx; i < N; i += stride) {
        sum += input[i];
    }

    sum = block_reduce(sum);

    if (threadIdx.x == 0) {
        output[blockIdx.x] = sum;
    }
}

// 多级归约
float reduce(const float* d_input, int N) {
    float *d_temp, *d_result;
    int numBlocks = (N + BLOCK_SIZE - 1) / BLOCK_SIZE;

    cudaMalloc(&d_temp, numBlocks * sizeof(float));
    cudaMalloc(&d_result, sizeof(float));

    // 第一级归约
    reduce_kernel<<<numBlocks, BLOCK_SIZE>>>(d_input, d_temp, N);

    // 如果Block数量仍然很多，再做一级归约
    int remaining = numBlocks;
    while (remaining > 1) {
        int blocks = (remaining + BLOCK_SIZE - 1) / BLOCK_SIZE;
        reduce_kernel<<<blocks, BLOCK_SIZE>>>(d_temp, d_temp, remaining);
        remaining = blocks;
    }

    float result;
    cudaMemcpy(&result, d_temp, sizeof(float), cudaMemcpyDeviceToHost);

    cudaFree(d_temp);
    cudaFree(d_result);
    return result;
}

// 测试
int main() {
    int N = 1024 * 1024;
    float* h_input = new float[N];
    for (int i = 0; i < N; i++) {
        h_input[i] = 1.0f;  // 期望结果：N
    }

    float* d_input;
    cudaMalloc(&d_input, N * sizeof(float));
    cudaMemcpy(d_input, h_input, N * sizeof(float), cudaMemcpyHostToDevice);

    float gpu_sum = reduce(d_input, N);
    float cpu_sum = 0.0f;
    for (int i = 0; i < N; i++) cpu_sum += h_input[i];

    printf("GPU sum: %f\n", gpu_sum);
    printf("CPU sum: %f\n", cpu_sum);
    printf("Relative error: %e\n", fabsf(gpu_sum - cpu_sum) / cpu_sum);

    delete[] h_input;
    cudaFree(d_input);
    return 0;
}
```

## 其他归约操作

```cpp
// 归约求最大值
__device__ float warp_reduce_max(float val) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xFFFFFFFF, val, offset);
        val = fmaxf(val, other);
    }
    return val;
}

// 归约求最小值
__device__ float warp_reduce_min(float val) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xFFFFFFFF, val, offset);
        val = fminf(val, other);
    }
    return val;
}

// argmax：返回最大值及其索引
struct ValIdx {
    float val;
    int idx;
};

__device__ ValIdx warp_reduce_argmax(ValIdx vi) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other_val = __shfl_down_sync(0xFFFFFFFF, vi.val, offset);
        int other_idx = __shfl_down_sync(0xFFFFFFFF, vi.idx, offset);
        if (other_val > vi.val) {
            vi.val = other_val;
            vi.idx = other_idx;
        }
    }
    return vi;
}
```

## 小结

1. 归约是从N个值到1个值的并行计算基本操作
2. 优化关键：消除线程发散、消除bank conflict、增加每个线程的工作量
3. Warp shuffle提供比共享内存更高效的warp内归约
4. 多级归约处理大规模数据：Block内归约 → 跨Block归约
5. Grid-stride loop让kernel灵活处理任意大小的输入
