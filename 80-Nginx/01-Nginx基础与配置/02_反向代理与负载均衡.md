# 反向代理与负载均衡


## 反向代理与负载均衡


反向代理负载均衡upstream


反向代理将客户端请求转发到后端服务器，负载均衡在多个后端服务器之间分配请求，提高系统的可用性和吞吐量。


## proxy_pass 配置


```
基本代理配置：
  location /api {
      proxy_pass http://backend_server;
      # 请求 http://nginx/api/users → 转发到 backend_server/api/users
  }

proxy_pass 路径规则：
  # 规则1：proxy_pass 有 URI（含路径）
  location /api {
      proxy_pass http://backend/app;
  }
  # /api/users → /app/users（/api 被替换为 /app）

  # 规则2：proxy_pass 无 URI
  location /api {
      proxy_pass http://backend;
  }
  # /api/users → /api/users（原样传递）

  # 规则3：proxy_pass 带 / 结尾
  location /api/ {
      proxy_pass http://backend/;
  }
  # /api/users → /users（/api/ 被替换为 /）

  # 规则4：proxy_pass 无 / 结尾
  location /api/ {
      proxy_pass http://backend;
  }
  # /api/users → /api/users（原样传递）

关键代理头部：
  location /api {
      proxy_pass http://backend;

      # 传递原始请求信息
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header X-Forwarded-Proto $scheme;
      proxy_set_header X-Forwarded-Host $host;
      proxy_set_header X-Forwarded-Port $server_port;

      # 连接设置
      proxy_http_version 1.1;
      proxy_set_header Connection "";
      proxy_connect_timeout 5s;
      proxy_send_timeout 60s;
      proxy_read_timeout 60s;

      # 缓冲设置
      proxy_buffering on;
      proxy_buffer_size 4k;
      proxy_buffers 8 4k;
  }

代理 WebSocket：
  location /ws {
      proxy_pass http://backend:8080;
      proxy_http_version 1.1;
      proxy_set_header Upgrade $http_upgrade;
      proxy_set_header Connection "upgrade";
      proxy_read_timeout 3600s;
  }

代理上传文件：
  location /upload {
      proxy_pass http://backend;
      client_max_body_size 100m;     # 最大上传大小
      proxy_request_buffering off;   # 不缓冲请求体
  }
```


## upstream 后端服务器组


```
基本配置：
  upstream backend {
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
      server 192.168.1.12:8080;
  }

  server {
      location / {
          proxy_pass http://backend;
      }
  }

服务器参数：
  upstream backend {
      server 192.168.1.10:8080 weight=5;        # 权重（默认1）
      server 192.168.1.11:8080 max_fails=3 fail_timeout=30s;
      # max_fails: 最大失败次数
      # fail_timeout: 失败后暂停时间

      server 192.168.1.12:8080 backup;          # 备用服务器
      server 192.168.1.13:8080 down;            # 永久下线
      server 192.168.1.14:8080 max_conns=100;   # 最大连接数
  }

服务器状态：
  # 标记服务器状态
  server 192.168.1.10:8080 down;       # 永久下线
  server 192.168.1.11:8080 backup;     # 备用（其他都不可用时才使用）
  server 192.168.1.12:8080 weight=0;   # 权重为0（不分配请求）

Nginx Plus 商业版健康检查：
  upstream backend {
      zone backend 64k;  # 共享内存区域

      server 192.168.1.10:8080;
      server 192.168.1.11:8080;

      # 主动健康检查
      health_check interval=5s fails=3 passes=2
                    match=health_check_response;
  }

  # 健康检查响应匹配规则
  match health_check_response {
      status 200-399;
      header Content-Type ~ "text/html";
      body ~ "OK";
  }
```


## 负载均衡算法


```
1. 轮询（Round Robin）— 默认
  upstream backend {
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
      server 192.168.1.12:8080;
  }
  // 请求依次分配到各服务器
  // Request1 → S1, Request2 → S2, Request3 → S3, Request4 → S1...

2. 加权轮询（Weighted Round Robin）
  upstream backend {
      server 192.168.1.10:8080 weight=5;
      server 192.168.1.11:8080 weight=3;
      server 192.168.1.12:8080 weight=1;
  }
  // 权重越高分配越多请求
  // 适用于服务器性能不同的场景
  // 每9个请求：S1(5个), S2(3个), S3(1个)

3. IP 哈希（ip_hash）
  upstream backend {
      ip_hash;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }
  // 同一客户端IP始终访问同一服务器
  // hash(client_ip) % server_count
  // 实现会话保持（粘性会话）
  // 问题：NAT环境下多个用户同一IP

4. 最少连接（least_conn）— Nginx Plus
  upstream backend {
      least_conn;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }
  // 优先分配给连接数最少的服务器
  // 适用于请求处理时间差异大的场景

5. 一致性哈希（hash）— Nginx Plus
  upstream backend {
      hash $request_uri consistent;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }
  // 相同URI始终路由到同一服务器
  // 适合缓存场景

6. 最短响应时间（least_time）— Nginx Plus
  upstream backend {
      least_time header;  # 或 last_byte
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }
  // 优先分配给响应最快的服务器

算法对比：
  ┌──────────────┬────────────────┬──────────────┐
  │ 算法         │ 适用场景       │ 会话保持     │
  ├──────────────┼────────────────┼──────────────┤
  │ 轮询         │ 服务器性能相近 │ 否           │
  │ 加权轮询     │ 服务器性能不同 │ 否           │
  │ ip_hash      │ 需要会话保持   │ 是（按IP）   │
  │ least_conn   │ 请求时间差异大 │ 否           │
  │ hash $var    │ 缓存亲和性     │ 是（按变量） │
  │ least_time   │ 追求最优性能   │ 否           │
  └──────────────┴────────────────┴──────────────┘
```


## 健康检查与会话保持


```
被动健康检查（开源版）：
  upstream backend {
      server 192.168.1.10:8080 max_fails=3 fail_timeout=30s;
      server 192.168.1.11:8080 max_fails=3 fail_timeout=30s;
  }

  // 30秒内失败3次 → 标记为不可用 → 暂停30秒
  // 30秒后恢复 → 重新分配请求
  // 失败定义：proxy_next_upstream 定义的条件

proxy_next_upstream（定义失败条件）：
  location /api {
      proxy_pass http://backend;
      proxy_next_upstream error timeout http_500 http_502 http_503;
      proxy_next_upstream_timeout 10s;
      proxy_next_upstream_tries 3;
  }
  // 遇到错误/超时/5xx → 自动尝试下一个服务器
  // 最多尝试3次，总超时10秒

主动健康检查（Nginx Plus）：
  upstream backend {
      zone backend 64k;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;

      # 每5秒检查一次
      # 连续3次失败 → 标记不可用
      # 连续2次成功 → 恢复可用
      health_check interval=5s fails=3 passes=2;
  }

  # 健康检查端点
  server {
      location /health {
          access_log off;
          return 200 "OK";
      }
  }

会话保持方案：

方案1：ip_hash
  upstream backend {
      ip_hash;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }

方案2：Cookie 哈希（Nginx Plus）
  upstream backend {
      hash $cookie_jsessionid;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }

方案3：sticky cookie（Nginx Plus）
  upstream backend {
      sticky cookie srv_id expires=1h path=/;
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }
  // Nginx 自动设置 cookie 标识后端服务器

方案4：应用层会话（推荐）
  // 使用 Redis 共享会话
  // Nginx 不需要做会话保持
  // 后端任意服务器都可处理
  // 最适合无状态微服务架构

  upstream backend {
      least_conn;  # 无需会话保持
      server 192.168.1.10:8080;
      server 192.168.1.11:8080;
  }
```


> **Note:** Nginx 反向代理通过 proxy_pass 实现，关键配置包括头部传递（X-Real-IP/X-Forwarded-For）和超时设置。负载均衡算法包括轮询（默认）、加权轮询、ip_hash、least_conn 等。健康检查分为被动（max_fails/fail_timeout）和主动（Nginx Plus）。会话保持建议使用应用层 Redis 共享会话而非 Nginx 层粘性会话。
