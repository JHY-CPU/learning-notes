# 缓存与压缩


## Nginx 缓存与压缩


代理缓存gzip/brotliFastCGI


Nginx 提供多层缓存机制和压缩功能，能大幅减少后端压力和网络传输量，是性能优化的核心手段。


## 代理缓存（Proxy Cache）


```
缓存配置：
  # 定义缓存区域
  proxy_cache_path /var/cache/nginx
      levels=1:2                    # 目录层级
      keys_zone=my_cache:10m        # 共享内存区（10MB）
      max_size=1g                   # 最大磁盘空间
      inactive=60m                  # 60分钟未访问则删除
      use_temp_path=off;            # 直接写入缓存目录

  server {
      location /api {
          proxy_pass http://backend;

          # 启用缓存
          proxy_cache my_cache;
          proxy_cache_valid 200 302 10m;   # 200/302 缓存10分钟
          proxy_cache_valid 404 1m;        # 404 缓存1分钟
          proxy_cache_valid any 5m;        # 其他状态缓存5分钟

          # 缓存键
          proxy_cache_key $scheme$host$request_uri;

          # 条件缓存
          proxy_cache_methods GET HEAD;    # 只缓存 GET/HEAD

          # 响应头中添加缓存状态
          add_header X-Cache-Status $upstream_cache_status;
      }
  }

缓存状态（X-Cache-Status）：
  HIT      - 缓存命中
  MISS     - 缓存未命中（从后端获取）
  EXPIRED  - 缓存过期（重新从后端获取）
  STALE    - 缓存过期但后端不可用（使用旧缓存）
  BYPASS   - 跳过缓存
  UPDATING - 缓存正在更新
  REVALIDATED - 缓存已验证有效

控制缓存行为：
  # 禁用缓存
  location /api/private {
      proxy_pass http://backend;
      proxy_cache off;
      add_header Cache-Control "no-store";
  }

  # 根据响应头控制缓存
  proxy_cache_valid any 5m;               # 默认缓存5分钟
  proxy_no_cache $http_pragma;            # Pragma: no-cache 时不缓存
  proxy_cache_bypass $http_pragma;        # 跳过缓存检查

  # 根据请求方法和头部控制
  proxy_cache_bypass $cookie_nocache $arg_nocache;
  proxy_no_cache $cookie_nocache $arg_nocache;

  # 后端控制缓存
  # 后端返回 Cache-Control: no-cache → Nginx 不缓存
  proxy_ignore_headers Cache-Control Expires;
  # 忽略后端缓存头，强制使用 Nginx 缓存策略

缓存目录结构：
  /var/cache/nginx/
  ├── 1/                   # level1（1个字符）
  │   ├── a/               # level2（2个字符）
  │   │   ├── a3f2b1...    # 缓存文件
  │   │   └── b4c8d2...
  │   └── b/
  │       └── e5f9a1...
  └── 2/
      └── c/
          └── d7e3f4...

  文件名 = MD5(proxy_cache_key)
  缓存文件包含头部和body
```


## Gzip 与 Brotli 压缩


```
Gzip 配置：
  # 开启 gzip 压缩
  gzip on;
  gzip_vary on;                    # 响应头加 Vary: Accept-Encoding
  gzip_proxied any;                # 对代理请求也压缩
  gzip_comp_level 6;               # 压缩级别（1-9，推荐5-6）
  gzip_min_length 256;             # 最小压缩长度（字节）
  gzip_buffers 16 8k;              # 压缩缓冲区
  gzip_http_version 1.1;           # HTTP/1.1 以上才压缩
  gzip_disable "msie6";            # IE6 不压缩

  # 压缩的 MIME 类型
  gzip_types
      text/plain
      text/css
      text/javascript
      text/xml
      application/json
      application/javascript
      application/xml
      application/xml+rss
      image/svg+xml;

Gzip 原理：
  客户端请求：
    Accept-Encoding: gzip, deflate, br
  服务端响应：
    Content-Encoding: gzip
    Vary: Accept-Encoding

  压缩效果：
  ┌──────────────────┬───────────┬───────────┬────────┐
  │ 文件类型         │ 原始大小  │ 压缩后    │ 比率   │
  ├──────────────────┼───────────┼───────────┼────────┤
  │ HTML             │ 100KB     │ 20KB      │ 80%    │
  │ CSS              │ 50KB      │ 8KB       │ 84%    │
  │ JavaScript       │ 200KB     │ 50KB      │ 75%    │
  │ JSON             │ 100KB     │ 15KB      │ 85%    │
  │ 图片(JPG/PNG)    │ -         │ -         │ 不压缩 │
  └──────────────────┴───────────┴───────────┴────────┘

Brotli 压缩（更高效）：
  # 需要 Nginx 编译 brotli 模块
  # 或使用 ngx_brotli 第三方模块

  brotli on;
  brotli_comp_level 6;             # 压缩级别（0-11）
  brotli_min_length 256;
  brotli_types
      text/plain
      text/css
      application/javascript
      application/json
      application/xml;

Gzip vs Brotli：
  ┌────────────┬──────────┬──────────┬──────────┐
  │            │ Gzip     │ Brotli   │ 差异     │
  ├────────────┼──────────┼──────────┼──────────┤
  │ 压缩率     │ 好       │ 更好     │ +15-25%  │
  │ 压缩速度   │ 快       │ 较慢     │ 服务端   │
  │ 解压速度   │ 快       │ 快       │ 相当     │
  │ 浏览器支持 │ 全部     │ 现代     │ IE不支持 │
  └────────────┴──────────┴──────────┴──────────┘

  // 推荐：同时支持两种，按浏览器能力选择
  Accept-Encoding: br, gzip, deflate

哪些不要压缩：
  # 已压缩的格式不要再压缩
  # 图片（JPG/PNG/GIF）已经是压缩格式
  # 压缩反而增大体积
  # 视频/音频同理
```


## Expires 头与静态资源缓存


```
Expires 配置：
  # 设置过期时间
  location ~* \.(jpg|jpeg|png|gif|ico|svg)$ {
      expires 30d;
      add_header Cache-Control "public, immutable";
  }

  location ~* \.(css|js)$ {
      expires 1y;
      add_header Cache-Control "public, immutable";
  }

  location ~* \.(woff|woff2|ttf|eot)$ {
      expires 1y;
      add_header Cache-Control "public, immutable";
  }

  # 动态内容不缓存
  location /api {
      expires -1;
      add_header Cache-Control "no-cache, no-store, must-revalidate";
  }

Cache-Control 优先级：
  Cache-Control: no-cache       → 每次验证（可用缓存）
  Cache-Control: no-store       → 不缓存
  Cache-Control: public         → 中间代理可缓存
  Cache-Control: private        → 只有浏览器可缓存
  Cache-Control: max-age=31536000 → 缓存1年
  Cache-Control: immutable      → 永不重新验证

静态资源版本控制：
  # 文件名加哈希
  # app.abc123.js → 永久缓存
  # 内容变化 → 哈希变化 → 新URL → 重新获取

  # Nginx 配置
  location ~* \.\w+\.\w+\.(js|css)$ {
      # 带哈希的文件名 → 长期缓存
      expires 1y;
      add_header Cache-Control "public, immutable";
  }

  location ~* \.(js|css)$ {
      # 不带哈希的文件 → 短期缓存
      expires 1h;
      add_header Cache-Control "public";
  }

静态文件服务优化：
  server {
      # 启用 sendfile（零拷贝）
      sendfile on;

      # 合并数据包
      tcp_nopush on;

      # 禁用 Nagle 算法（小文件即时发送）
      tcp_nodelay on;

      # 文件读取缓存
      open_file_cache max=1000 inactive=20s;
      open_file_cache_valid 30s;
      open_file_cache_min_uses 2;
      open_file_cache_errors on;

      # 目录索引缓存
      location / {
          root /var/www/html;
          index index.html;
          try_files $uri $uri/ /index.html;
      }
  }
```


## FastCGI 缓存与缓存清理


```
FastCGI 缓存（PHP/Python 等动态应用）：
  # 定义缓存路径
  fastcgi_cache_path /var/cache/fastcgi
      levels=1:2
      keys_zone=php_cache:10m
      max_size=100m
      inactive=60m;

  # 定义缓存键
  fastcgi_cache_key "$scheme$request_method$host$request_uri";

  server {
      location ~ \.php$ {
          fastcgi_pass unix:/var/run/php-fpm.sock;
          fastcgi_index index.php;
          include fastcgi_params;

          # 启用 FastCGI 缓存
          fastcgi_cache php_cache;
          fastcgi_cache_valid 200 10m;
          fastcgi_cache_valid 404 1m;

          # 缓存状态头
          add_header X-FastCGI-Cache $upstream_cache_status;

          # 跳过缓存的条件
          fastcgi_cache_bypass $cookie_logged_in;
          fastcgi_no_cache $cookie_logged_in;
      }
  }

缓存清理方法：

方法1：手动清理缓存目录
  # 删除所有缓存
  rm -rf /var/cache/nginx/*

  # 删除特定URL缓存
  # 需要计算缓存文件名（MD5）
  find /var/cache/nginx -type f -delete

方法2：通过 PURGE 请求清理（需要模块支持）
  # 安装 ngx_cache_purge 模块
  location /purge {
      allow 127.0.0.1;
      deny all;
      proxy_cache_purge my_cache $scheme$host$request_uri;
  }

  # 使用
  curl -X PURGE http://localhost/api/users

方法3：使缓存失效的变通方法
  # 在 proxy_cache_key 中加入版本号
  proxy_cache_key $scheme$host$request_uri$v;
  # 改变 $v 的值 → 不同的缓存键 → 旧缓存自然过期

  # 通过 map 从请求头获取版本
  map $http_x_cache_version $v {
      default "";
      ~.+     $http_x_cache_version;
  }

缓存预热：
  # 定时预热热门页面
  #!/bin/bash
  for url in $(cat popular_urls.txt); do
      curl -s -o /dev/null "http://localhost$url"
  done
  # 提前将热门内容缓存，减少首次访问延迟

缓存监控：
  # 查看缓存命中率
  log_format cache '$remote_addr - $upstream_cache_status '
                   '"$request" $status $body_bytes_sent';

  access_log /var/log/nginx/cache.log cache;

  # 统计缓存命中率
  # HIT / (HIT + MISS) × 100%
```


> **Note:** Nginx 缓存体系包括代理缓存（proxy_cache）、FastCGI 缓存和浏览器缓存（Expires/Cache-Control）。Gzip 压缩可减少 70-85% 的文本传输量，Brotli 压缩率更高但需额外模块。静态资源建议使用文件名哈希 + 长期缓存策略。缓存清理可通过删除目录、PURGE 请求或变更缓存键实现。
