# 安全与限流


## Nginx 安全与限流


SSL/TLS限流WAF


Nginx 提供 SSL/TLS 加密、请求限流（limit_req）、连接限制（limit_conn）和访问控制等安全功能，是 Web 安全的第一道防线。


## SSL/TLS 配置


```
基本 HTTPS 配置：
  server {
      listen 443 ssl http2;
      server_name www.example.com;

      # 证书和密钥
      ssl_certificate     /etc/nginx/ssl/fullchain.pem;
      ssl_certificate_key /etc/nginx/ssl/privkey.pem;

      # 协议和加密套件
      ssl_protocols TLSv1.2 TLSv1.3;
      ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
      ssl_prefer_server_ciphers on;

      # 会话缓存
      ssl_session_cache shared:SSL:10m;
      ssl_session_timeout 1d;
      ssl_session_tickets off;

      # OCSP Stapling
      ssl_stapling on;
      ssl_stapling_verify on;
      ssl_trusted_certificate /etc/nginx/ssl/chain.pem;
      resolver 8.8.8.8 8.8.4.4 valid=300s;

      root /var/www/html;
  }

  # HTTP 重定向到 HTTPS
  server {
      listen 80;
      server_name www.example.com;
      return 301 https://$host$request_uri;
  }

安全头部配置：
  # HSTS - 强制 HTTPS
  add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

  # 防止 MIME 嗅探
  add_header X-Content-Type-Options "nosniff" always;

  # 防止点击劫持
  add_header X-Frame-Options "SAMEORIGIN" always;

  # XSS 过滤
  add_header X-XSS-Protection "1; mode=block" always;

  # CSP - 内容安全策略
  add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline'" always;

  # Referrer 策略
  add_header Referrer-Policy "strict-origin-when-cross-origin" always;

  # 权限策略
  add_header Permissions-Policy "camera=(), microphone=()" always;

SSL 证书管理：
  # Let's Encrypt 自动证书
  # 安装 certbot
  apt install certbot python3-certbot-nginx

  # 获取证书
  certbot --nginx -d www.example.com -d example.com

  # 自动续期
  certbot renew --dry-run

  # Nginx 配置（certbot 自动配置）
  ssl_certificate /etc/letsencrypt/live/example.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/example.com/privkey.pem;

HTTPS 优化：
  # 0-RTT（TLS 1.3）
  ssl_early_data on;

  # 会话票据
  ssl_session_tickets on;

  # OCSP Stapling 减少客户端验证延迟
  ssl_stapling on;
  ssl_stapling_verify on;

  # 吊销列表
  ssl_crl /etc/nginx/ssl/crl.pem;
```


## 请求限流（limit_req）


```
limit_req 基本配置：
  # 定义限流区域
  # $binary_remote_addr = 客户端IP（二进制，节省内存）
  limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;

  server {
      location /api {
          limit_req zone=api_limit burst=20 nodelay;
          proxy_pass http://backend;
      }
  }

参数说明：
  limit_req_zone
      $binary_remote_addr    # 限流键（按IP）
      zone=api_limit:10m     # 共享内存区（10MB ≈ 16万IP）
      rate=10r/s             # 速率限制（每秒10个请求）

  limit_req
      zone=api_limit         # 使用哪个限流区
      burst=20               # 突发容量（允许超出20个请求排队）
      nodelay                # 突发请求立即处理（不延迟）

  // burst=20 nodelay 的含义：
  正常速率：10r/s
  突发允许：20个额外请求
  处理方式：突发请求立即处理，不排队
  后续请求：超过突发容量则拒绝

限流粒度：
  # 按 IP 限流
  limit_req_zone $binary_remote_addr zone=per_ip:10m rate=5r/s;

  # 按 Server 限流
  limit_req_zone $server_name zone=per_server:10m rate=1000r/s;

  # 按 URI 限流
  limit_req_zone $uri zone=per_uri:10m rate=100r/s;

  # 组合限流（IP + URI）
  limit_req_zone "$binary_remote_addr$uri" zone=per_ip_uri:10m rate=5r/s;

  # 按 User 限流（需要认证后获取用户ID）
  limit_req_zone $http_authorization zone=per_user:10m rate=20r/s;

限流响应：
  # 默认返回 503
  limit_req_status 429;

  # 自定义响应
  limit_req_status 429;

  error_page 429 /rate_limit.html;
  location = /rate_limit.html {
      internal;
      return 429 '{"error": "Rate limit exceeded"}';
  }

  # 响应头
  add_header X-RateLimit-Limit 10 always;
  add_header X-RateLimit-Remaining $limit_req_status always;

多级限流：
  # 全局宽松限流
  limit_req_zone $binary_remote_addr zone=global:10m rate=100r/s;

  # API 严格限流
  limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;

  # 登录接口极严格限流
  limit_req_zone $binary_remote_addr zone=login:10m rate=1r/s;

  server {
      # 全局
      limit_req zone=global burst=50;

      location /api {
          limit_req zone=api burst=20 nodelay;
      }

      location /login {
          limit_req zone=login burst=3;
      }
  }
```


## 连接限流与访问控制


```
连接限流（limit_conn）：
  # 每个IP最大10个并发连接
  limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

  server {
      limit_conn conn_limit 10;
      limit_conn_status 429;

      # 连接级别限流
      location /download {
          limit_conn conn_limit 2;  # 下载最多2个并发
      }
  }

  # 结合带宽限制
  limit_rate 512k;          # 每连接限速 512KB/s
  limit_rate_after 1m;      # 前1MB不限速

访问控制（allow/deny）：
  # 允许指定 IP
  allow 192.168.1.0/24;
  allow 10.0.0.1;
  deny all;

  # 限制管理后台
  location /admin {
      allow 192.168.1.0/24;    # 内网可访问
      allow 10.0.0.1;           # 特定IP
      deny all;                 # 其他都拒绝

      proxy_pass http://backend;
  }

  # 限制文件类型访问
  location ~ /\. {
      deny all;                 # 隐藏文件（.htaccess, .git）
  }

  location ~* \.(sql|bak|log)$ {
      deny all;                 # 敏感文件
  }

HTTP 方法限制：
  # 只允许 GET 和 POST
  if ($request_method !~ ^(GET|POST)$) {
      return 405;
  }

  # limit_except 更优雅
  limit_except GET {
      allow 192.168.1.0/24;    # 内网可以任意方法
      deny all;                 # 外网只允许GET
      proxy_pass http://backend;
  }

基本认证：
  # 生成密码文件
  # htpasswd -c /etc/nginx/.htpasswd admin
  # htpasswd /etc/nginx/.htpasswd user1

  location /admin {
      auth_basic "Restricted Area";
      auth_basic_user_file /etc/nginx/.htpasswd;
      proxy_pass http://backend;
  }

  # 白名单 + 认证
  satisfy any;                 # 满足任一条件即可
  allow 192.168.1.0/24;       # 内网免认证
  deny all;
  auth_basic "Login";
  auth_basic_user_file /etc/nginx/.htpasswd;
```

总结：Nginx 安全体系包括 SSL/TLS 配置（TLS 1.2+、HSTS、OCSP Stapling）、limit_req 请求限流（按IP/URI/用户多级限流）、
    limit_conn 连接限流、allow/deny 访问控制和 basic_auth 基本认证。
    WAF 可通过 ModSecurity + OWASP CRS 实现 SQL 注入/XSS/路径遍历等攻击防护。
    安全头部：HSTS、X-Content-Type-Options、X-Frame-Options、CSP。隐藏 server_tokens，禁用敏感文件访问。
