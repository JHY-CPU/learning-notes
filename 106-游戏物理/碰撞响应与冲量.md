# 碰撞响应与冲量

## 核心概念

碰撞检测后需要计算碰撞响应，使物体正确反弹、滑动或停止。基于冲量的方法直接修改速度，比逐步施加力更高效。冲量是力在时间上的积分，可以直接改变物体的速度。

### 碰撞数据

- **碰撞法线 n**：从碰撞体A指向B的单位向量（沿最小穿透方向）
- **碰撞点 p**：两物体的接触点
- **穿透深度 d**：两物体重叠的距离
- **恢复系数 e**：控制反弹程度，0=完全非弹性，1=完全弹性

### 相对速度

碰撞点处的相对速度（包含线性和角速度贡献）：

```
v_rel = (vA + ωA × rA) - (vB + ωB × rB)
其中 rA = p - posA, rB = p - posB
```

### 冲量公式推导

沿法线方向的冲量标量：

```
j = -(1 + e) * v_rel · n / [1/mA + 1/mB + ((rA × n)^T * I_A^(-1) * (rA × n)) + ((rB × n)^T * I_B^(-1) * (rB × n))]
```

其中 e 是恢复系数。分母可以理解为碰撞的"等效质量"。

## 完整碰撞响应实现

```cpp
#include <cmath>
#include <vector>
#include <algorithm>

struct Vec3 {
    float x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(float s) const { return {x*s, y*s, z*s}; }
    Vec3 operator-() const { return {-x, -y, -z}; }
    Vec3& operator+=(const Vec3& o) { x+=o.x; y+=o.y; z+=o.z; return *this; }
    Vec3& operator-=(const Vec3& o) { x-=o.x; y-=o.y; z-=o.z; return *this; }
    float dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z-z*o.y, z*o.x-x*o.z, x*o.y-y*o.x};
    }
    float lengthSq() const { return x*x+y*y+z*z; }
    float length() const { return sqrt(lengthSq()); }
    Vec3 normalized() const { float l=length(); return l>1e-6f?Vec3{x/l,y/l,z/l}:Vec3{0,0,0}; }
};

struct RigidBody {
    Vec3 position, velocity, angularVelocity;
    float mass, inverseMass;
    // 惯性张量（简化：假设各向同性）
    float inertia, inverseInertia;

    bool isStatic() const { return inverseMass == 0; }
};

struct CollisionInfo {
    RigidBody* bodyA;
    RigidBody* bodyB;
    Vec3 normal;        // 从A指向B
    Vec3 contactPoint;
    float penetration;
};

// ========== 单碰撞点冲量求解 ==========
void resolveCollision(CollisionInfo& info, float restitution = 0.5f) {
    RigidBody* A = info.bodyA;
    RigidBody* B = info.bodyB;
    Vec3 n = info.normal;
    Vec3 p = info.contactPoint;

    Vec3 rA = p - A->position;
    Vec3 rB = p - B->position;

    // 碰撞点的相对速度（包含旋转贡献）
    Vec3 velA = A->velocity + A->angularVelocity.cross(rA);
    Vec3 velB = B->velocity + B->angularVelocity.cross(rB);
    Vec3 relVel = velA - velB;

    // 沿法线方向的相对速度
    float velAlongNormal = relVel.dot(n);

    // 如果物体正在分离，不需要响应
    if (velAlongNormal > 0) return;

    // 计算冲量分母（等效质量的倒数）
    Vec3 rAxN = rA.cross(n);
    Vec3 rBxN = rB.cross(n);

    float denom = A->inverseMass + B->inverseMass
                + rAxN.dot(rAxN) * A->inverseInertia
                + rBxN.dot(rBxN) * B->inverseInertia;

    // 法线冲量大小
    float j = -(1 + restitution) * velAlongNormal / denom;

    // 应用法线冲量
    Vec3 impulse = n * j;
    A->velocity += impulse * A->inverseMass;
    B->velocity -= impulse * B->inverseMass;
    A->angularVelocity += rA.cross(impulse) * A->inverseInertia;
    B->angularVelocity -= rB.cross(impulse) * B->inverseInertia;
}

// ========== 带摩擦的碰撞响应 ==========
void resolveCollisionWithFriction(CollisionInfo& info,
    float restitution = 0.5f, float friction = 0.4f) {

    RigidBody* A = info.bodyA;
    RigidBody* B = info.bodyB;
    Vec3 n = info.normal;
    Vec3 p = info.contactPoint;

    Vec3 rA = p - A->position;
    Vec3 rB = p - B->position;

    Vec3 velA = A->velocity + A->angularVelocity.cross(rA);
    Vec3 velB = B->velocity + B->angularVelocity.cross(rB);
    Vec3 relVel = velA - velB;

    float velAlongNormal = relVel.dot(n);
    if (velAlongNormal > 0) return;

    // ========== 法线冲量 ==========
    Vec3 rAxN = rA.cross(n);
    Vec3 rBxN = rB.cross(n);
    float denom = A->inverseMass + B->inverseMass
                + rAxN.dot(rAxN) * A->inverseInertia
                + rBxN.dot(rBxN) * B->inverseInertia;

    float jn = -(1 + restitution) * velAlongNormal / denom;
    Vec3 normalImpulse = n * jn;

    A->velocity += normalImpulse * A->inverseMass;
    B->velocity -= normalImpulse * B->inverseMass;
    A->angularVelocity += rA.cross(normalImpulse) * A->inverseInertia;
    B->angularVelocity -= rB.cross(normalImpulse) * B->inverseInertia;

    // ========== 切向摩擦冲量 ==========
    // 重新计算碰撞后的相对速度
    velA = A->velocity + A->angularVelocity.cross(rA);
    velB = B->velocity + B->angularVelocity.cross(rB);
    relVel = velA - velB;

    // 提取切向分量
    Vec3 tangent = relVel - n * relVel.dot(n);
    float tangentSpeed = tangent.length();

    if (tangentSpeed > 0.001f) {
        tangent = tangent * (1.0f / tangentSpeed);

        // 切向分母
        Vec3 rAxT = rA.cross(tangent);
        Vec3 rBxT = rB.cross(tangent);
        float tangentDenom = A->inverseMass + B->inverseMass
                           + rAxT.dot(rAxT) * A->inverseInertia
                           + rBxT.dot(rBxT) * B->inverseInertia;

        // 切向冲量
        float jt = -relVel.dot(tangent) / tangentDenom;

        // 库仑摩擦定律：切向冲量限制在摩擦锥内
        // |jt| <= μ * |jn|
        if (abs(jt) > abs(jn) * friction) {
            jt = copysign(abs(jn) * friction, jt);
            // 滑动摩擦
        }
        // 否则为静摩擦

        Vec3 frictionImpulse = tangent * jt;
        A->velocity += frictionImpulse * A->inverseMass;
        B->velocity -= frictionImpulse * B->inverseMass;
        A->angularVelocity += rA.cross(frictionImpulse) * A->inverseInertia;
        B->angularVelocity -= rB.cross(frictionImpulse) * B->inverseInertia;
    }
}

// ========== 位置修正（避免持续穿透） ==========
void positionalCorrection(CollisionInfo& info,
    float percent = 0.4f, float slop = 0.01f) {

    // 修正量 = (穿透 - 容差) / 总逆质量 * 比例
    float correctionMag = std::max(info.penetration - slop, 0.0f)
                        / (info.bodyA->inverseMass + info.bodyB->inverseMass)
                        * percent;

    Vec3 correction = info.normal * correctionMag;

    // 按逆质量比例分配修正
    info.bodyA->position -= correction * info.bodyA->inverseMass;
    info.bodyB->position += correction * info.bodyB->inverseMass;
}
```

## 多碰撞点顺序冲量求解器

```cpp
// ========== 顺序冲量求解器 ==========
class SequentialImpulseSolver {
    struct ContactConstraint {
        CollisionInfo info;
        float normalImpulseAccum;   // 法线冲量累积
        float tangentImpulseAccum;  // 切向冲量累积
        float restitution;
        float friction;

        // 预计算的分母
        float normalMass;    // 法线方向等效质量
        float tangentMass;   // 切向方向等效质量
        Vec3 tangent;        // 切向方向
    };

    std::vector<ContactConstraint> constraints;
    int iterationCount = 10;

public:
    void addContact(const CollisionInfo& info, float restitution, float friction) {
        ContactConstraint cc;
        cc.info = info;
        cc.normalImpulseAccum = 0;
        cc.tangentImpulseAccum = 0;
        cc.restitution = restitution;
        cc.friction = friction;

        // 预计算分母
        Vec3 rA = info.contactPoint - info.bodyA->position;
        Vec3 rB = info.contactPoint - info.bodyB->position;
        Vec3 rAxN = rA.cross(info.normal);
        Vec3 rBxN = rB.cross(info.normal);

        cc.normalMass = 1.0f / (info.bodyA->inverseMass + info.bodyB->inverseMass
                       + rAxN.dot(rAxN) * info.bodyA->inverseInertia
                       + rBxN.dot(rBxN) * info.bodyB->inverseInertia);

        constraints.push_back(cc);
    }

    void solve(float dt) {
        // 预求解：计算切向方向
        for (auto& c : constraints) {
            Vec3 rA = c.info.contactPoint - c.info.bodyA->position;
            Vec3 rB = c.info.contactPoint - c.info.bodyB->position;
            Vec3 velA = c.info.bodyA->velocity + c.info.bodyA->angularVelocity.cross(rA);
            Vec3 velB = c.info.bodyB->velocity + c.info.bodyB->angularVelocity.cross(rB);
            Vec3 relVel = velA - velB;

            c.tangent = relVel - c.info.normal * relVel.dot(c.info.normal);
            float tLen = c.tangent.length();
            if (tLen > 0.001f) c.tangent = c.tangent * (1.0f / tLen);

            Vec3 rAxT = rA.cross(c.tangent);
            Vec3 rBxT = rB.cross(c.tangent);
            c.tangentMass = 1.0f / (c.info.bodyA->inverseMass + c.info.bodyB->inverseMass
                          + rAxT.dot(rAxT) * c.info.bodyA->inverseInertia
                          + rBxT.dot(rBxT) * c.info.bodyB->inverseInertia);
        }

        // 迭代求解（Gauss-Seidel风格）
        for (int iter = 0; iter < iterationCount; iter++) {
            for (auto& c : constraints) {
                solveOneConstraint(c);
            }
        }

        // 位置修正
        for (auto& c : constraints) {
            positionalCorrection(c.info);
        }
    }

private:
    void solveOneConstraint(ContactConstraint& c) {
        RigidBody* A = c.info.bodyA;
        RigidBody* B = c.info.bodyB;
        Vec3 rA = c.info.contactPoint - A->position;
        Vec3 rB = c.info.contactPoint - B->position;

        // ========== 法线冲量 ==========
        Vec3 velA = A->velocity + A->angularVelocity.cross(rA);
        Vec3 velB = B->velocity + B->angularVelocity.cross(rB);
        Vec3 relVel = velA - velB;
        float velAlongNormal = relVel.dot(c.info.normal);

        // Baumgarte稳定项（防止漂移）
        float bias = 0.0f;
        if (c.info.penetration > 0.01f) {
            bias = -(0.2f / 0.016f) * std::max(c.info.penetration - 0.01f, 0.0f);
        }

        float lambda = -(velAlongNormal + bias) * c.normalMass;

        // 累积并钳制冲量（只能推不能拉）
        float oldImpulse = c.normalImpulseAccum;
        c.normalImpulseAccum = std::max(oldImpulse + lambda, 0.0f);
        lambda = c.normalImpulseAccum - oldImpulse;

        Vec3 normalImpulse = c.info.normal * lambda;
        A->velocity += normalImpulse * A->inverseMass;
        B->velocity -= normalImpulse * B->inverseMass;
        A->angularVelocity += rA.cross(normalImpulse) * A->inverseInertia;
        B->angularVelocity -= rB.cross(normalImpulse) * B->inverseInertia;

        // ========== 切向摩擦冲量 ==========
        velA = A->velocity + A->angularVelocity.cross(rA);
        velB = B->velocity + B->angularVelocity.cross(rB);
        relVel = velA - velB;
        float velAlongTangent = relVel.dot(c.tangent);

        float tangentLambda = -velAlongTangent * c.tangentMass;

        // 库仑摩擦锥钳制
        float maxFriction = c.friction * c.normalImpulseAccum;
        float oldTangentImpulse = c.tangentImpulseAccum;
        c.tangentImpulseAccum = std::max(-maxFriction,
            std::min(oldTangentImpulse + tangentLambda, maxFriction));
        tangentLambda = c.tangentImpulseAccum - oldTangentImpulse;

        Vec3 frictionImpulse = c.tangent * tangentLambda;
        A->velocity += frictionImpulse * A->inverseMass;
        B->velocity -= frictionImpulse * B->inverseMass;
        A->angularVelocity += rA.cross(frictionImpulse) * A->inverseInertia;
        B->angularVelocity -= rB.cross(frictionImpulse) * B->inverseInertia;
    }
};
```

## 恢复系数物理含义

```
e = 0 (完全非弹性碰撞):
  碰撞后两物体粘在一起，无反弹
  v_after = (m1*v1 + m2*v2) / (m1+m2)  // 动量守恒，动能损失

e = 1 (完全弹性碰撞):
  碰撞后动能完全守恒，完全反弹
  v1_after = (m1-m2)/(m1+m2)*v1 + 2*m2/(m1+m2)*v2
  v2_after = 2*m1/(m1+m2)*v1 + (m2-m1)/(m1+m2)*v2

e = 0.5 (典型现实物体):
  玻璃球 ~0.8, 高尔夫球 ~0.83, 足球 ~0.7
  金属球 ~0.3-0.6, 木材 ~0.5-0.6
```

## Unity/C#碰撞响应

```csharp
using UnityEngine;

public class CollisionResponseDemo : MonoBehaviour
{
    private Rigidbody rb;

    void OnCollisionEnter(Collision collision)
    {
        ContactPoint contact = collision.GetContact(0);
        Vector3 normal = contact.normal;
        Vector3 point = contact.point;

        // Unity已内置碰撞响应，但可以自定义补充
        // 例如：基于碰撞强度播放不同音效

        float impactForce = collision.impulse.magnitude / Time.fixedDeltaTime;
        float relativeSpeed = collision.relativeVelocity.magnitude;

        if (relativeSpeed > 10f)
        {
            // 强碰撞
            PlayHeavyImpactSound(point);
            SpawnImpactEffect(point, normal);
        }
        else if (relativeSpeed > 3f)
        {
            // 中等碰撞
            PlayMediumImpactSound(point);
        }
    }

    // 自定义弹性碰撞计算（用于特殊效果）
    Vector3 CalculateBounce(Vector3 velocity, Vector3 normal, float restitution)
    {
        float velDotN = Vector3.Dot(velocity, normal);
        if (velDotN > 0) return velocity; // 正在分离

        // v' = v - (1+e) * (v·n) * n
        return velocity - (1 + restitution) * velDotN * normal;
    }

    // 自定义摩擦
    Vector3 ApplyFriction(Vector3 velocity, Vector3 normal,
        float frictionCoeff, float dt)
    {
        Vector3 normalVel = Vector3.Dot(velocity, normal) * normal;
        Vector3 tangentVel = velocity - normalVel;

        if (tangentVel.magnitude < 0.001f) return normalVel;

        // 摩擦减速度
        float frictionDecel = frictionCoeff * Physics.gravity.magnitude;
        float newSpeed = Mathf.Max(0, tangentVel.magnitude - frictionDecel * dt);

        return normalVel + tangentVel.normalized * newSpeed;
    }

    void PlayHeavyImpactSound(Vector3 pos) { }
    void PlayMediumImpactSound(Vector3 pos) { }
    void SpawnImpactEffect(Vector3 pos, Vector3 normal) { }
}
```

## 游戏应用场景

1. **弹球游戏**：完全弹性碰撞（e接近1）实现反弹
2. **车辆碰撞**：部分弹性碰撞（e约0.3）模拟金属变形
3. **保龄球**：摩擦力使球产生旋转和滑动
4. **积木堆叠**：位置修正确保积木不穿透，迭代求解器处理多接触点
5. **台球**：精确的弹性碰撞和旋转传递
6. **碰撞音效**：根据碰撞强度和材质选择音效

## 常见陷阱

- **过冲修正**：位置修正可能导致物体被"弹飞"，需限制修正比例（通常0.2-0.8）
- **多接触点**：单独求解每个接触点会导致违反其他约束，需迭代求解
- **静摩擦**：切向冲量需限制在摩擦锥内（库仑定律），否则物体会"粘"住
- **小质量物体**：碰撞小质量物体时可能产生极大的速度，需限制最大冲量
- **法线方向**：确保法线方向一致（总是从A指向B或相反），否则冲量方向错误
- **穿透修正时机**：位置修正应在速度求解之后进行
- **迭代次数**：太少导致关节"拉伸"，太多浪费性能。通常5-20次
