# 约束求解器

## 核心概念

约束限制物体之间的相对运动，是实现关节、绳索、布娃娃等效果的核心。约束求解器通过迭代调整速度或位置来满足所有约束条件。游戏中的约束求解主要有两种方法：基于速度的Sequential Impulse（顺序冲量）和基于位置的PBD（Position Based Dynamics）。

### 关节约束类型

- **距离约束**：两点间距离保持不变（绳索、连杆）
- **球窝关节（Ball-Socket）**：允许所有方向旋转，限制两物体某点重合
- **铰链关节（Hinge）**：限制为单一旋转轴（门、车轮）
- **棱柱关节（Prismatic）**：允许沿单一轴滑动（活塞）
- **固定关节（Fixed）**：完全约束所有相对运动
- **角度约束**：限制两物体间的最大/最小角度

### 约束数学表示

约束可表示为一个等式或不等式：

```
等式约束: C(x) = 0
不等式约束: C(x) >= 0

其中 x 是物体的状态（位置和旋转）

约束的时间导数（速度级别）:
C'(x) = J * v = 0
其中 J 是约束的雅可比矩阵，v 是速度向量 [v1, ω1, v2, ω2]
```

## Position Based Dynamics（PBD）

基于位置而非力/速度求解，由Mueller等人在2007年提出：

```
PBD算法步骤:
1. 施加重力等外力，预测新位置: x* = x + v*dt
2. 对每个约束迭代求解位置修正: Δx
3. 从修正后的位置计算新速度: v = (x_new - x_old) / dt

优点: 直观、稳定、易于实现
缺点: 刚度有物理上限，迭代次数影响精度
```

### PBD完整实现

```cpp
#include <vector>
#include <cmath>
#include <algorithm>

struct Vec3 {
    float x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(float s) const { return {x*s, y*s, z*s}; }
    Vec3 operator-() const { return {-x, -y, -z}; }
    Vec3& operator+=(const Vec3& o) { x+=o.x; y+=o.y; z+=o.z; return *this; }
    Vec3& operator-=(const Vec3& o) { x-=o.x; y-=o.y; z-=o.z; return *this; }
    float dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z-z*o.y, z*o.x-x*o.z, x*o.y-y*o.x};
    }
    float lengthSq() const { return x*x+y*y+z*z; }
    float length() const { return sqrt(lengthSq()); }
    Vec3 normalized() const { float l=length(); return l>1e-6f?Vec3{x/l,y/l,z/l}:Vec3{0,0,0}; }
};

struct Quaternion {
    float w, x, y, z;
    Quaternion() : w(1), x(0), y(0), z(0) {}
    Quaternion(float w, float x, float y, float z) : w(w), x(x), y(y), z(z) {}
    Quaternion operator*(const Quaternion& q) const {
        return {w*q.w-x*q.x-y*q.y-z*q.z, w*q.x+x*q.w+y*q.z-z*q.y,
                w*q.y-x*q.z+y*q.w+z*q.x, w*q.z+x*q.y-y*q.x+z*q.w};
    }
    Quaternion operator+(const Quaternion& q) const { return {w+q.w,x+q.x,y+q.y,z+q.z}; }
    Quaternion operator*(float s) const { return {w*s,x*s,y*s,z*s}; }
    Quaternion conjugate() const { return {w,-x,-y,-z}; }
    Quaternion normalized() const {
        float l=sqrt(w*w+x*x+y*y+z*z);
        return l>1e-6f?Quaternion{w/l,x/l,y/l,z/l}:Quaternion{1,0,0,0};
    }
    Vec3 rotate(const Vec3& v) const {
        Quaternion qv={0,v.x,v.y,v.z};
        Quaternion r=(*this)*qv*conjugate();
        return {r.x,r.y,r.z};
    }
    static Quaternion fromAxisAngle(Vec3 axis, float angle) {
        float h = angle*0.5f, s = sin(h);
        Vec3 a = axis.normalized();
        return {cos(h), a.x*s, a.y*s, a.z*s};
    }
};

// ========== PBD粒子 ==========
struct PBDEntity {
    Vec3 position;
    Vec3 prevPosition;
    Vec3 velocity;
    float inverseMass;
    Quaternion orientation;
    Vec3 prevOrientation; // 简化：使用角速度表示
    float inverseInertia;

    bool isStatic() const { return inverseMass == 0; }
};

// ========== PBD距离约束 ==========
struct DistanceConstraint {
    int bodyA, bodyB;
    float restLength;
    float stiffness; // 刚度 [0, 1]，1=硬约束

    void solve(std::vector<PBDEntity>& entities) {
        PBDEntity& a = entities[bodyA];
        PBDEntity& b = entities[bodyB];

        Vec3 delta = b.position - a.position;
        float dist = delta.length();
        if (dist < 0.0001f) return;

        float wSum = a.inverseMass + b.inverseMass;
        if (wSum == 0) return;

        // 约束误差
        float C = dist - restLength;

        // 位置修正
        Vec3 correction = delta * (C / dist);
        float k = stiffness; // 可以根据需要调整

        a.position += correction * (a.inverseMass / wSum) * k;
        b.position -= correction * (b.inverseMass / wSum) * k;
    }
};

// ========== PBD球窝关节 ==========
struct BallSocketConstraint {
    int bodyA, bodyB;
    Vec3 localAnchorA; // A的局部空间锚点
    Vec3 localAnchorB; // B的局部空间锚点
    float stiffness;

    void solve(std::vector<PBDEntity>& entities) {
        PBDEntity& a = entities[bodyA];
        PBDEntity& b = entities[bodyB];

        // 将局部锚点变换到世界空间
        Vec3 worldA = a.position + a.orientation.rotate(localAnchorA);
        Vec3 worldB = b.position + b.orientation.rotate(localAnchorB);

        Vec3 diff = worldB - worldA;
        float dist = diff.length();
        if (dist < 0.0001f) return;

        float wSum = a.inverseMass + b.inverseMass;
        if (wSum == 0) return;

        Vec3 correction = diff * (dist / dist); // 方向

        // 考虑旋转的修正臂
        Vec3 rA = worldA - a.position;
        Vec3 rB = worldB - b.position;
        float wA = a.inverseMass + rA.cross(correction).lengthSq() * a.inverseInertia;
        float wB = b.inverseMass + rB.cross(correction).lengthSq() * b.inverseInertia;
        wSum = wA + wB;

        a.position += correction * (wA / wSum) * stiffness;
        b.position -= correction * (wB / wSum) * stiffness;

        // 旋转修正
        if (a.inverseInertia > 0) {
            Vec3 rotAxis = rA.cross(correction);
            float rotAngle = rotAxis.length() * a.inverseInertia * stiffness;
            if (rotAngle > 0.001f) {
                a.orientation = Quaternion::fromAxisAngle(rotAxis.normalized(), rotAngle)
                              * a.orientation;
                a.orientation = a.orientation.normalized();
            }
        }
    }
};

// ========== PBD主求解器 ==========
class PBDSolver {
    std::vector<PBDEntity> entities;
    std::vector<DistanceConstraint> distanceConstraints;
    std::vector<BallSocketConstraint> ballSocketConstraints;

    int solverIterations = 10;
    Vec3 gravity = {0, -9.81f, 0};
    float damping = 0.99f; // 速度阻尼

public:
    int addEntity(Vec3 pos, float invMass, float invInertia = 1.0f) {
        PBDEntity e;
        e.position = pos;
        e.prevPosition = pos;
        e.velocity = {0, 0, 0};
        e.inverseMass = invMass;
        e.inverseInertia = invInertia;
        entities.push_back(e);
        return (int)entities.size() - 1;
    }

    void addDistanceConstraint(int a, int b, float restLen, float stiffness = 1.0f) {
        distanceConstraints.push_back({a, b, restLen, stiffness});
    }

    void addBallSocket(int a, int b, Vec3 localA, Vec3 localB, float stiffness = 1.0f) {
        ballSocketConstraints.push_back({a, b, localA, localB, stiffness});
    }

    void step(float dt) {
        // ========== 1. 施加外力，预测位置 ==========
        for (auto& entity : entities) {
            if (entity.isStatic()) continue;

            // 施加重力
            entity.velocity += gravity * dt;

            // 阻尼
            entity.velocity = entity.velocity * damping;

            // 保存当前位置
            entity.prevPosition = entity.position;

            // 预测新位置
            entity.position += entity.velocity * dt;
        }

        // ========== 2. 迭代求解约束 ==========
        for (int iter = 0; iter < solverIterations; iter++) {
            for (auto& c : distanceConstraints)
                c.solve(entities);

            for (auto& c : ballSocketConstraints)
                c.solve(entities);
        }

        // ========== 3. 从位置变化计算新速度 ==========
        for (auto& entity : entities) {
            if (entity.isStatic()) continue;
            entity.velocity = (entity.position - entity.prevPosition) / dt;
        }
    }

    const std::vector<PBDEntity>& getEntities() const { return entities; }
};
```

struct Constraint {
    int bodyA, bodyB;
    virtual void solve(float dt) = 0;
    virtual ~Constraint() {}
};

struct VelocityDistanceConstraint : Constraint {
    Vec3 localAnchorA, localAnchorB;
    float restLength;
    float stiffness;
    float accumulatedImpulse;

    VelocityDistanceConstraint(int a, int b, Vec3 anchorA, Vec3 anchorB,
        float restLen, float stiff = 1.0f)
    {
        bodyA = a; bodyB = b;
        localAnchorA = anchorA; localAnchorB = anchorB;
        restLength = restLen; stiffness = stiff;
        accumulatedImpulse = 0;
    }

    void solve(float dt) override {
        // 假设有全局bodies数组
        // 计算约束雅可比矩阵 J
        // C = |pB - pA| - restLength
        // J = [n, rA×n, -n, -rB×n] 其中 n = (pB-pA)/|pB-pA|

        // 计算当前约束值
        // 应用拉格朗日乘子 λ = -C' / (J * M^(-1) * J^T)
        // 累积并钳制冲量
        // 应用冲量到两个物体
    }
};

// ========== 铰链约束 ==========
struct HingeConstraint : Constraint {
    Vec3 pivot;         // 铰链点（世界空间）
    Vec3 axis;          // 旋转轴（世界空间）
    float minAngle;     // 最小角度限制
    float maxAngle;     // 最大角度限制

    HingeConstraint(int a, int b, Vec3 pivotPoint, Vec3 hingeAxis,
        float minAng = -1e6f, float maxAng = 1e6f)
    {
        bodyA = a; bodyB = b;
        pivot = pivotPoint;
        axis = hingeAxis.normalized();
        minAngle = minAng; maxAngle = maxAng;
    }

    void solve(float dt) override {
        // 1. 点约束：两物体的铰链点必须重合
        // 2. 轴约束：两物体的旋转轴必须对齐
        // 3. 角度限制：限制相对旋转角度在[minAngle, maxAngle]内

        // 点约束求解（类似球窝关节）
        solvePointConstraint(dt);

        // 轴对齐约束
        solveAxisConstraint(dt);

        // 角度限制
        solveAngleLimit(dt);
    }

private:
    void solvePointConstraint(float dt) {
        // 两物体在铰链点的相对速度应为零
        // J_point = [I, -rA×, -I, rB×]
    }

    void solveAxisConstraint(float dt) {
        // 两物体绕铰链轴以外的相对角速度应为零
    }

    void solveAngleLimit(float dt) {
        // 如果当前角度超出限制，施加修正角速度
    }
};

## 弹簧约束

用于布娃娃的"软"约束和车辆悬挂：

```cpp
struct SpringConstraint {
    int bodyA, bodyB;
    float stiffness;    // 弹簧刚度 (N/m)
    float damping;      // 阻尼系数 (N*s/m)
    float restLength;
    float maxLength;    // 最大拉伸长度
    float minLength;    // 最小压缩长度

    void applyForce(RigidBody& a, RigidBody& b) {
        Vec3 diff = b.position - a.position;
        float dist = diff.length();
        if (dist < 0.0001f) return;

        Vec3 dir = diff / dist;
        float displacement = dist - restLength;

        // 弹簧力（胡克定律）: F = -k * x
        Vec3 springForce = dir * (stiffness * displacement);

        // 阻尼力: F = -c * v_relative
        Vec3 relVel = b.velocity - a.velocity;
        float dampingForce = damping * relVel.dot(dir);
        Vec3 dampVec = dir * dampingForce;

        Vec3 totalForce = springForce + dampVec;

        a.applyForce(totalForce);
        b.applyForce(-totalForce);
    }

    // PBD版本的弹簧约束
    void solvePBD(std::vector<Vec3>& positions, std::vector<float>& invMasses) {
        Vec3 pA = positions[bodyA];
        Vec3 pB = positions[bodyB];
        Vec3 diff = pB - pA;
        float dist = diff.length();
        if (dist < 0.0001f) return;

        float wA = invMasses[bodyA];
        float wB = invMasses[bodyB];
        float wSum = wA + wB;
        if (wSum == 0) return;

        // 柔性约束：使用刚度参数控制修正幅度
        float error = dist - restLength;
        Vec3 correction = diff * (error / dist) * stiffness;

        positions[bodyA] += correction * (wA / wSum);
        positions[bodyB] -= correction * (wB / wSum);
    }
};

// ========== 车辆悬挂系统 ==========
class VehicleSuspension {
    SpringConstraint springs[4]; // 四轮悬挂
    float wheelRadius;

    void update(float dt, RigidBody& chassis) {
        for (int i = 0; i < 4; i++) {
            // 获取轮子的世界位置
            Vec3 wheelPos = chassis.position + chassis.orientation.rotate(wheelPositions[i]);

            // Raycast向下检测地面
            float groundDist;
            Vec3 groundNormal;
            if (raycastDown(wheelPos, groundDist, groundNormal)) {
                float compression = wheelRadius - groundDist;
                if (compression > 0) {
                    // 弹簧压缩，施加向上的力
                    float springForce = springs[i].stiffness * compression;
                    float dampingForce = springs[i].damping * chassis.velocity.y;
                    float totalForce = springForce - dampingForce;

                    chassis.applyForceAtPoint(
                        groundNormal * std::max(0.0f, totalForce), wheelPos);
                }
            }
        }
    }
};
```

## 绳索与布料模拟

```cpp
// ========== PBD绳索 ==========
class PBDRope {
    struct RopeParticle {
        Vec3 position;
        Vec3 prevPosition;
        Vec3 velocity;
        float inverseMass;
    };

    std::vector<RopeParticle> particles;
    std::vector<DistanceConstraint> constraints;
    float segmentLength;

public:
    void create(Vec3 start, Vec3 end, int numSegments, float totalMass) {
        segmentLength = (end - start).length() / numSegments;
        float massPerParticle = totalMass / numSegments;

        for (int i = 0; i <= numSegments; i++) {
            float t = (float)i / numSegments;
            RopeParticle p;
            p.position = start + (end - start) * t;
            p.prevPosition = p.position;
            p.velocity = {0, 0, 0};
            p.inverseMass = (i == 0) ? 0 : 1.0f / massPerParticle; // 固定端点
            particles.push_back(p);
        }

        for (int i = 0; i < numSegments; i++) {
            constraints.push_back({i, i + 1, segmentLength, 1.0f});
        }
    }

    void step(float dt, Vec3 gravity = {0, -9.81f, 0}) {
        // 1. 施加外力
        for (auto& p : particles) {
            if (p.inverseMass == 0) continue;
            p.velocity += gravity * dt;
            p.prevPosition = p.position;
            p.position += p.velocity * dt;
        }

        // 2. 求解距离约束（多次迭代）
        for (int iter = 0; iter < 10; iter++) {
            for (auto& c : constraints) {
                auto& a = particles[c.bodyA];
                auto& b = particles[c.bodyB];
                Vec3 delta = b.position - a.position;
                float dist = delta.length();
                if (dist < 0.0001f) continue;
                float wSum = a.inverseMass + b.inverseMass;
                if (wSum == 0) continue;

                Vec3 correction = delta * ((dist - c.restLength) / dist);
                a.position += correction * (a.inverseMass / wSum);
                b.position -= correction * (b.inverseMass / wSum);
            }
        }

        // 3. 更新速度
        for (auto& p : particles) {
            if (p.inverseMass == 0) continue;
            p.velocity = (p.position - p.prevPosition) / dt;
        }
    }
};

// ========== PBD布料（质点-弹簧模型） ==========
class PBDCloth {
    int width, height;
    std::vector<Vec3> positions;
    std::vector<Vec3> prevPositions;
    std::vector<float> inverseMasses;
    std::vector<DistanceConstraint> constraints;

public:
    void create(int w, int h, float spacing, Vec3 origin) {
        width = w; height = h;

        for (int y = 0; y < h; y++) {
            for (int x = 0; x < w; x++) {
                positions.push_back(origin + Vec3(x * spacing, 0, y * spacing));
                prevPositions.push_back(positions.back());
                inverseMasses.push_back(1.0f); // 默认可移动
            }
        }

        // 固定顶部两端
        inverseMasses[0] = 0;
        inverseMasses[w - 1] = 0;

        // 结构弹簧（水平和垂直）
        for (int y = 0; y < h; y++) {
            for (int x = 0; x < w; x++) {
                int idx = y * w + x;
                if (x < w - 1) addConstraint(idx, idx + 1, spacing);
                if (y < h - 1) addConstraint(idx, idx + w, spacing);
            }
        }

        // 剪切弹簧（对角线）
        for (int y = 0; y < h - 1; y++) {
            for (int x = 0; x < w - 1; x++) {
                int idx = y * w + x;
                addConstraint(idx, idx + w + 1, spacing * 1.414f);
                addConstraint(idx + 1, idx + w, spacing * 1.414f);
            }
        }
    }

    void step(float dt) {
        Vec3 gravity = {0, -9.81f, 0};
        for (size_t i = 0; i < positions.size(); i++) {
            if (inverseMasses[i] == 0) continue;
            Vec3 vel = (positions[i] - prevPositions[i]) / dt;
            vel += gravity * dt;
            prevPositions[i] = positions[i];
            positions[i] += vel * dt;
        }

        for (int iter = 0; iter < 10; iter++) {
            for (auto& c : constraints) {
                Vec3 delta = positions[c.bodyB] - positions[c.bodyA];
                float dist = delta.length();
                if (dist < 0.0001f) continue;
                float wSum = inverseMasses[c.bodyA] + inverseMasses[c.bodyB];
                if (wSum == 0) continue;
                Vec3 corr = delta * ((dist - c.restLength) / dist);
                positions[c.bodyA] += corr * (inverseMasses[c.bodyA] / wSum);
                positions[c.bodyB] -= corr * (inverseMasses[c.bodyB] / wSum);
            }
        }
    }

private:
    void addConstraint(int a, int b, float restLen) {
        constraints.push_back({a, b, restLen, 1.0f});
    }
};

## 游戏应用场景

1. **布娃娃系统**：骨骼间用球窝/铰链关节约束连接
2. **绳索/链条**：一系列距离约束连接的质点
3. **门/杠杆**：铰链约束实现可交互机关
4. **载具悬挂**：弹簧+阻尼约束模拟悬挂系统
5. **桥梁**：多个距离约束构成可通行桥梁
6. **旗帜/布料**：PBD质点-弹簧模型
7. **软体**：四面体约束的体网格

## 实际游戏案例

### 案例：《Garry's Mod》的物理工具

Garry's Mod使用Source引擎的物理系统（基于Bullet变体），玩家可以创建各种约束：
- 焊接（Weld）：固定关节，完全约束两个物体的相对运动
- 弹簧（Spring）：弹性连接，可设置刚度和阻尼
- 轴（Axis）：铰链关节，允许绕轴旋转
- 绳索（Rope）：距离约束链
- 滑轮（Pulley）：距离约束的变体

### 案例：《The Last of Us》的布娃娃

Naughty Dog的布娃娃系统使用多层约束：
1. 球窝关节连接骨骼段
2. 角度限制约束防止关节反折
3. 弹性约束在剧烈碰撞时让关节"断裂"
4. 主动肌肉力让角色在死亡时做出自然反应

        // ========== 可配置关节（最灵活） ==========
        ConfigurableJoint cj = gameObject.AddComponent<ConfigurableJoint>();
        cj.connectedBody = otherRigidbody;
        cj.xMotion = ConfigurableJointMotion.Limited;
        cj.yMotion = ConfigurableJointMotion.Locked;
        cj.zMotion = ConfigurableJointMotion.Locked;

        // 线性限制
        SoftJointLimit linearLimit = new SoftJointLimit();
        linearLimit.limit = 1f;
        cj.linearLimit = linearLimit;

        // 线性弹簧
        JointDrive drive = new JointDrive();
        drive.positionSpring = 100f;
        drive.positionDamper = 10f;
        drive.maximumForce = 1000f;
        cj.xDrive = drive;
    }
}
```

## 常见陷阱

- **迭代次数不足**：约束求解不充分导致关节"拉伸"或"断裂"。通常需要5-20次迭代
- **奇异构型**：完全伸展的关节链可能导致求解震荡。使用更小的时间步或增加迭代
- **stiffness过大会导致不稳定**：PBD的刚度参数有物理上限。超过后需要更多迭代或更小dt
- **质量比悬殊**：轻质物体与重质物体的约束可能不稳定。限制最大质量比或使用堆叠处理
- **约束顺序敏感**：Gauss-Seidel迭代的结果依赖约束的处理顺序
- **暖启动**：未使用暖启动的约束求解需要更多迭代才能收敛
- **关节断裂阈值**：设置过低的断裂力会导致正常碰撞就断裂
