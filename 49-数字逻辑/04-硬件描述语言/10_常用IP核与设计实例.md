# 常用IP核与设计实例

## FIFO（先进先出队列）

FIFO 是数字系统中最常用的数据缓冲模块，用于在不同速率的模块之间传递数据。FIFO 分为同步 FIFO（读写共用同一时钟）和异步 FIFO（读写使用不同时钟）。

```verilog
// 同步 FIFO 设计
module sync_fifo #(
    parameter DATA_WIDTH = 16,
    parameter FIFO_DEPTH = 16,
    parameter ADDR_WIDTH = 4   // log2(FIFO_DEPTH)
) (
    input  wire                  clk,
    input  wire                  rst_n,
    // 写端口
    input  wire                  wr_en,
    input  wire [DATA_WIDTH-1:0] wr_data,
    output wire                  full,
    // 读端口
    input  wire                  rd_en,
    output reg  [DATA_WIDTH-1:0] rd_data,
    output wire                  empty,
    // 状态
    output wire [ADDR_WIDTH:0]   data_count
);
    // 存储阵列
    reg [DATA_WIDTH-1:0] mem [0:FIFO_DEPTH-1];

    // 读写指针
    reg [ADDR_WIDTH:0] wr_ptr, rd_ptr;

    // 空满标志
    assign full       = (wr_ptr[ADDR_WIDTH] != rd_ptr[ADDR_WIDTH]) &&
                        (wr_ptr[ADDR_WIDTH-1:0] == rd_ptr[ADDR_WIDTH-1:0]);
    assign empty      = (wr_ptr == rd_ptr);
    assign data_count = wr_ptr - rd_ptr;

    // 写逻辑
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            wr_ptr <= 0;
        else if (wr_en && !full) begin
            mem[wr_ptr[ADDR_WIDTH-1:0]] <= wr_data;
            wr_ptr <= wr_ptr + 1;
        end
    end

    // 读逻辑
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            rd_ptr <= 0;
            rd_data <= 0;
        end else if (rd_en && !empty) begin
            rd_data <= mem[rd_ptr[ADDR_WIDTH-1:0]];
            rd_ptr  <= rd_ptr + 1;
        end
    end
endmodule
```

## RAM（随机存取存储器）

FPGA 中的 RAM 可以通过 Block RAM（BRAM）或分布式 RAM 实现。根据读写端口可分为单端口 RAM、简单双端口 RAM 和真双端口 RAM。

```verilog
// 单端口同步 RAM
module single_port_ram #(
    parameter DATA_WIDTH = 8,
    parameter ADDR_WIDTH = 10,
    parameter DEPTH      = (1 << ADDR_WIDTH)
) (
    input  wire                  clk,
    input  wire                  we,
    input  wire [ADDR_WIDTH-1:0] addr,
    input  wire [DATA_WIDTH-1:0] din,
    output reg  [DATA_WIDTH-1:0] dout
);
    // 推断 BRAM
    reg [DATA_WIDTH-1:0] ram [0:DEPTH-1];

    always @(posedge clk) begin
        if (we)
            ram[addr] <= din;
        dout <= ram[addr];
    end
endmodule

// 简单双端口 RAM（一个写端口，一个读端口）
module simple_dual_port_ram #(
    parameter DATA_WIDTH = 32,
    parameter ADDR_WIDTH = 10,
    parameter DEPTH      = (1 << ADDR_WIDTH)
) (
    input  wire                  clk,
    // 写端口
    input  wire                  wr_en,
    input  wire [ADDR_WIDTH-1:0] wr_addr,
    input  wire [DATA_WIDTH-1:0] wr_data,
    // 读端口
    input  wire [ADDR_WIDTH-1:0] rd_addr,
    output reg  [DATA_WIDTH-1:0] rd_data
);
    reg [DATA_WIDTH-1:0] ram [0:DEPTH-1];

    always @(posedge clk) begin
        if (wr_en)
            ram[wr_addr] <= wr_data;
    end

    always @(posedge clk) begin
        rd_data <= ram[rd_addr];
    end
endmodule
```

## PLL（锁相环）

PLL 用于产生不同时钟频率和相位的时钟信号。不同 FPGA 厂商的 PLL IP 核使用方式不同，以下以 Xilinx MMCM 为例说明其 Verilog 例化方式。

```verilog
// Xilinx MMCM 时钟管理模块例化
// 在 Vivado 中通过 IP Catalog 生成
module clk_wiz_example (
    input  wire clk_in1,      // 输入时钟 100MHz
    input  wire reset,
    output wire clk_out1,     // 输出时钟 200MHz
    output wire clk_out2,     // 输出时钟 50MHz
    output wire clk_out3,     // 输出时钟 25MHz
    output wire locked        // PLL 锁定信号
);
    // 以下为 Vivado Clocking Wizard 生成的例化模板
    clk_wiz_0 u_clk_wiz (
        .clk_in1  (clk_in1),
        .reset    (reset),
        .clk_out1 (clk_out1),
        .clk_out2 (clk_out2),
        .clk_out3 (clk_out3),
        .locked   (locked)
    );
endmodule

// 使用 PLL 输出的时钟
module top_with_pll (
    input  wire sys_clk,     // 板载 100MHz 晶振
    input  wire rst_n,
    input  wire [7:0] data_in,
    output wire [7:0] data_out
);
    wire clk_200m, clk_50m, clk_25m;
    wire pll_locked;
    wire sys_rst_n = rst_n & pll_locked;  // PLL 锁定后才释放复位

    clk_wiz_example u_pll (
        .clk_in1  (sys_clk),
        .reset    (~rst_n),
        .clk_out1 (clk_200m),
        .clk_out2 (clk_50m),
        .clk_out3 (clk_25m),
        .locked   (pll_locked)
    );

    // 使用不同频率的时钟
    fast_module u_fast (.clk(clk_200m), .rst_n(sys_rst_n), .din(data_in), .dout(data_out));
endmodule
```

## UART 控制器

UART（通用异步收发传输器）是最基本的串行通信接口，常用于 FPGA 与 PC 或其他设备的通信。

```verilog
// UART 发送模块
module uart_tx #(
    parameter CLK_FREQ  = 100_000_000,  // 系统时钟频率
    parameter BAUD_RATE = 115200         // 波特率
) (
    input  wire       clk,
    input  wire       rst_n,
    input  wire [7:0] tx_data,
    input  wire       tx_valid,
    output reg        tx_ready,
    output reg        txd               // 串行输出
);
    localparam CLKS_PER_BIT = CLK_FREQ / BAUD_RATE;

    reg [15:0] clk_count;
    reg [3:0]  bit_index;
    reg [9:0]  tx_shift;  // {stop, data[7:0], start}
    reg        active;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            txd        <= 1'b1;
            tx_ready   <= 1'b1;
            clk_count  <= 0;
            bit_index  <= 0;
            tx_shift   <= 10'h3FF;
            active     <= 1'b0;
        end else if (tx_valid && tx_ready) begin
            // 加载数据帧：起始位(0) + 8位数据 + 停止位(1)
            tx_shift   <= {1'b1, tx_data, 1'b0};
            active     <= 1'b1;
            tx_ready   <= 1'b0;
            clk_count  <= 0;
            bit_index  <= 0;
        end else if (active) begin
            if (clk_count == CLKS_PER_BIT - 1) begin
                clk_count <= 0;
                txd       <= tx_shift[bit_index];
                if (bit_index == 9) begin
                    active    <= 1'b0;
                    tx_ready  <= 1'b1;
                    bit_index <= 0;
                end else begin
                    bit_index <= bit_index + 1;
                end
            end else begin
                clk_count <= clk_count + 1;
            end
        end
    end
endmodule

// UART 接收模块
module uart_rx #(
    parameter CLK_FREQ  = 100_000_000,
    parameter BAUD_RATE = 115200
) (
    input  wire       clk,
    input  wire       rst_n,
    input  wire       rxd,          // 串行输入
    output reg  [7:0] rx_data,
    output reg        rx_valid      // 数据有效标志
);
    localparam CLKS_PER_BIT = CLK_FREQ / BAUD_RATE;
    localparam HALF_BIT     = CLKS_PER_BIT / 2;

    reg [15:0] clk_count;
    reg [3:0]  bit_index;
    reg [7:0]  rx_shift;
    reg        active;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            rx_data   <= 8'h00;
            rx_valid  <= 1'b0;
            clk_count <= 0;
            bit_index <= 0;
            rx_shift  <= 8'h00;
            active    <= 1'b0;
        end else begin
            rx_valid <= 1'b0;
            if (!active && !rxd) begin
                // 检测到起始位
                active    <= 1'b1;
                clk_count <= 0;
                bit_index <= 0;
            end else if (active) begin
                if (clk_count == HALF_BIT) begin
                    // 采样起始位中心
                    clk_count <= clk_count + 1;
                end else if (clk_count == CLKS_PER_BIT - 1) begin
                    clk_count <= 0;
                    if (bit_index < 8) begin
                        rx_shift[bit_index] <= rxd;
                        bit_index <= bit_index + 1;
                    end else begin
                        // 停止位
                        rx_data  <= rx_shift;
                        rx_valid <= 1'b1;
                        active   <= 1'b0;
                    end
                end else begin
                    clk_count <= clk_count + 1;
                end
            end
        end
    end
endmodule
```

## SPI 控制器

SPI（Serial Peripheral Interface）是同步串行通信协议，支持全双工通信。SPI 有四种工作模式，由 CPOL（时钟极性）和 CPHA（时钟相位）决定。

```verilog
// SPI Master 控制器
module spi_master #(
    parameter CLK_FREQ  = 100_000_000,
    parameter SPI_FREQ  = 10_000_000,
    parameter DATA_BITS = 8
) (
    input  wire                    clk,
    input  wire                    rst_n,
    // 用户接口
    input  wire [DATA_BITS-1:0]    tx_data,
    input  wire                    tx_valid,
    output reg  [DATA_BITS-1:0]    rx_data,
    output reg                     rx_valid,
    output wire                    busy,
    // SPI 接口
    output reg                     sclk,
    output reg                     mosi,
    input  wire                    miso,
    output reg                     cs_n
);
    localparam CLKS_PER_HALF = CLK_FREQ / (2 * SPI_FREQ);

    reg [15:0]            clk_count;
    reg [DATA_BITS-1:0]   tx_shift;
    reg [DATA_BITS-1:0]   rx_shift;
    reg [3:0]             bit_count;
    reg                   active;

    assign busy = active;

    // 时钟分频产生 SCLK
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            sclk      <= 1'b0;
            clk_count <= 0;
        end else if (active) begin
            if (clk_count == CLKS_PER_HALF - 1) begin
                sclk      <= ~sclk;
                clk_count <= 0;
            end else begin
                clk_count <= clk_count + 1;
            end
        end else begin
            sclk      <= 1'b0;
            clk_count <= 0;
        end
    end

    // 数据移位
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            tx_shift  <= 0;
            rx_shift  <= 0;
            bit_count <= 0;
            active    <= 1'b0;
            cs_n      <= 1'b1;
            mosi      <= 1'b0;
            rx_valid  <= 1'b0;
        end else begin
            rx_valid <= 1'b0;
            if (tx_valid && !active) begin
                tx_shift  <= tx_data;
                rx_shift  <= 0;
                bit_count <= DATA_BITS;
                active    <= 1'b1;
                cs_n      <= 1'b0;
                mosi      <= tx_data[DATA_BITS-1];
            end else if (active && clk_count == CLKS_PER_HALF - 1 && sclk) begin
                // SCLK 下降沿：更新 MOSI
                tx_shift <= {tx_shift[DATA_BITS-2:0], 1'b0};
                mosi     <= tx_shift[DATA_BITS-2];
                // SCLK 上升沿采样（在下一个半周期）
                rx_shift <= {rx_shift[DATA_BITS-2:0], miso};
                if (bit_count == 1) begin
                    active   <= 1'b0;
                    cs_n     <= 1'b1;
                    rx_data  <= {rx_shift[DATA_BITS-2:0], miso};
                    rx_valid <= 1'b1;
                end else begin
                    bit_count <= bit_count - 1;
                end
            end
        end
    end
endmodule
```

IP 核使用建议：
1. 优先使用厂商提供的验证过的 IP 核（如 Xilinx 的 Block Memory Generator、Clocking Wizard）
2. 自行设计的 IP 核应经过充分的仿真验证
3. FIFO 的深度应根据实际数据突发长度确定
4. UART/SPI 等通信模块需注意波特率/时钟频率的参数化设计
5. 时钟管理模块的 locked 信号应作为后续模块的复位条件
