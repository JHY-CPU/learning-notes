// ========== 生产者-消费者 ========== // 最基础的并发模式 type ProducerConsumer struct { bufferSize int } func (pc *ProducerConsumer) Run() { // 缓冲 channel 解耦生产和消费 jobs := make(chan int, pc.bufferSize) done := make(chan struct{}) // 生产者 (多个) go func() { for i := 0; i < 20; i++ { jobs <- i fmt.Println("生产:", i) } close(jobs) }() // 消费者 (多个) for w := 0; w < 3; w++ { go func(id int) { for job := range jobs { fmt.Printf("消费者 %d 消费: %d\n", id, job) } done <- struct{}{} }(w) } // 等待所有消费者完成 for w := 0; w < 3; w++ { <-done } } // ========== 发布-订阅 (Pub/Sub) ========== type PubSub struct { mu sync.RWMutex subscribers map[string][]chan interface{} } func NewPubSub() *PubSub { return &PubSub{ subscribers: make(map[string][]chan interface{}), } } func (ps *PubSub) Subscribe(topic string) <-chan interface{} { ps.mu.Lock() defer ps.mu.Unlock() ch := make(chan interface{}, 1) ps.subscribers[topic] = append(ps.subscribers[topic], ch) return ch } func (ps *PubSub) Publish(topic string, msg interface{}) { ps.mu.RLock() defer ps.mu.RUnlock() for _, ch := range ps.subscribers[topic] { // 非阻塞发送 (如果订阅者慢则丢弃) select { case ch <- msg: default: // 丢弃消息 (或记录日志) } } } func (ps *PubSub) Close() { ps.mu.Lock() defer ps.mu.Unlock() for _, subs := range ps.subscribers { for _, ch := range subs { close(ch) } } } // 使用: // ps := NewPubSub() // ch1 := ps.Subscribe("events") // go func() { // for msg := range ch1 { fmt.Println(msg) } // }() // ps.Publish("events", "hello")

// ========== 单例模式 4 种实现 ========== // 1. 包级别变量 (最简单的单例) var GlobalConfig = loadConfig() func loadConfig() *Config { return &Config{Host: "localhost", Port: 8080} } // 2. sync.Once (懒加载, 线程安全) var ( config *Config configOnce sync.Once ) func GetConfig() *Config { configOnce.Do(func() { config = loadConfig() }) return config } // 3. init() 函数 (程序启动时初始化) var dbPool *sql.DB func init() { var err error dbPool, err = sql.Open("postgres", dsn) if err != nil { log.Fatal(err) } } // 4. 原子指针 (懒加载 + 无锁) var configPtr atomic.Pointer[Config] func GetConfigAtomic() *Config { if c := configPtr.Load(); c != nil { return c } // 慢路径: 创建并原子存储 c := loadConfig() configPtr.Store(c) return c } // ========== 并行处理模式 ========== // 1. 分而治之 func divideAndConquer(items []int, fn func(int) int) []int { if len(items) < 100 { // 小数据直接处理 result := make([]int, len(items)) for i, item := range items { result[i] = fn(item) } return result } // 大数据分半处理 mid := len(items) / 2 var wg sync.WaitGroup var result1, result2 []int wg.Add(2) go func() { result1 = divideAndConquer(items[:mid], fn) wg.Done() }() go func() { result2 = divideAndConquer(items[mid:], fn) wg.Done() }() wg.Wait() return append(result1, result2...) } // 2. MapReduce (简化版) func MapReduce[T, U, V any]( items []T, mapper func(T) U, reducer func(V, U) V, initial V, workers int, ) V { input := make(chan T, len(items)) output := make(chan U, len(items)) // Map var mapWg sync.WaitGroup for i := 0; i < workers; i++ { mapWg.Add(1) go func() { defer mapWg.Done() for item := range input { output <- mapper(item) } }() } // 发送输入 for _, item := range items { input <- item } close(input) // 等待 Map 完成, 关闭 output go func() { mapWg.Wait() close(output) }() // Reduce (单线程) result := initial for v := range output { result = reducer(result, v) } return result } // 使用: // sum := MapReduce([]int{1,2,3,4,5}, // func(v int) int { return v * 2 }, // map // func(acc, v int) int { return acc + v }, // reduce // 0, 3)

并发模式选择指南

// ========== 并发模式决策树 ========== // // 1. 数据流向 // 单向 (A→B) → Pipeline / 生产者-消费者 // 双向 (A↔B) → channel 通信 // 广播 (1→N) → Pub/Sub // 聚合 (N→1) → Fan-In // // 2. 任务特征 // 独立任务 → Fan-Out (多个 worker) // 分阶段 → Pipeline // 需合并 → MapReduce // 需降级 → 熔断器 // 需限速 → 令牌桶 // // 3. 协调需求 // 同步等待 → WaitGroup // 取消信号 → Context // 单次初始化 → Once // 并发限制 → Semaphore (channel) // 条件等待 → sync.Cond / channel // ========== 并发模型的权衡 ========== // channel: 通信 + 同步, 但增加了 goroutine 管理开销 // mutex: 直接, 高效, 但容易出错 (死锁) // atomic: 最快, 但只适用于简单状态 // 经验法则: // 1. 优先使用 channel 通信 (Go 哲学) // 2. 共享状态用 mutex 保护 // 3. 简单计数器用 atomic // 4. 有超时/取消用 context // 5. 控制并发数用 semaphore // ========== 并发安全指南 ========== // 1. 谁创建, 谁负责 // - 创建 goroutine 的函数应负责其退出 // 2. 明确 ownership // - channel 的创建者负责关闭 // - mutex 的持有者负责解锁 // 3. 不要猜并发安全性 // - 默认不是并发安全的 // - 文档说明并发安全性 // 4. 数据竞争用 -race 检测 // go run -race main.go // go test -race ./... // ========== 完整代码: 并发版 Web 爬虫 ========== type Crawler struct { urls chan string done chan struct{} wg sync.WaitGroup result sync.Map // URL → 状态 } func NewCrawler(workers int) *Crawler { c := &Crawler{ urls: make(chan string, 100), done: make(chan struct{}), } // 启动 workers for i := 0; i < workers; i++ { c.wg.Add(1) go func() { defer c.wg.Done() for { select { case url, ok := <-c.urls: if !ok { return } status := c.fetch(url) c.result.Store(url, status) case <-c.done: return } } }() } return c } func (c *Crawler) fetch(url string) int { resp, err := http.Get(url) if err != nil { return 0 } defer resp.Body.Close() return resp.StatusCode } func (c *Crawler) Add(url string) { c.urls <- url } func (c *Crawler) Close() { close(c.done) c.wg.Wait() }
