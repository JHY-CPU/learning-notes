中间件模式

// ========== 中间件类型 ========== type Middleware func(http.Handler) http.Handler // ========== 链式组合 ========== func Chain(handler http.Handler, middlewares ...Middleware) http.Handler { for i := len(middlewares) - 1; i >= 0; i-- { handler = middlewares[i](handler) } return handler } // 使用: // handler := Chain(mux, Recovery, Logger, CORS, Auth) // ========== 标准中间件模式 ========== // 每个中间件包装 http.Handler // 1. 日志中间件 func Logger(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { start := time.Now() // 包装 ResponseWriter 捕获状态码 lrw := &logResponseWriter{ResponseWriter: w, statusCode: 200} next.ServeHTTP(lrw, r) log.Printf("%s %s %d %v", r.Method, r.URL.Path, lrw.statusCode, time.Since(start)) }) } type logResponseWriter struct { http.ResponseWriter statusCode int } func (lrw *logResponseWriter) WriteHeader(code int) { lrw.statusCode = code lrw.ResponseWriter.WriteHeader(code) } // 2. 恢复中间件 (防止 panic 崩溃) func Recovery(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { defer func() { if err := recover(); err != nil { log.Printf("panic: %v, path: %s", err, r.URL.Path) http.Error(w, `{"error":"internal error"}`, 500) } }() next.ServeHTTP(w, r) }) } // 3. CORS 中间件 func CORS(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { w.Header().Set("Access-Control-Allow-Origin", "*") w.Header().Set("Access-Control-Allow-Methods", "GET,POST,PUT,DELETE,OPTIONS") w.Header().Set("Access-Control-Allow-Headers", "Content-Type,Authorization") if r.Method == "OPTIONS" { w.WriteHeader(204) return } next.ServeHTTP(w, r) }) } // 4. 认证中间件 func Auth(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { token := r.Header.Get("Authorization") if token == "" || !isValid(token) { w.Header().Set("Content-Type", "application/json") w.WriteHeader(401) json.NewEncoder(w).Encode(map[string]string{"error": "unauthorized"}) return } next.ServeHTTP(w, r) }) }

请求上下文传值

// ========== Context 传递用户信息 ========== // 认证成功后, 将用户信息存入 context // 1. 安全的 key 类型 type contextKey string const ( UserKey contextKey = "user" RolesKey contextKey = "roles" RequestIDKey contextKey = "request_id" ) // 2. 类型安全的 getter/setter func WithUser(ctx context.Context, user *User) context.Context { return context.WithValue(ctx, UserKey, user) } func UserFromContext(ctx context.Context) *User { if user, ok := ctx.Value(UserKey).(*User); ok { return user } return nil } func WithRequestID(ctx context.Context, id string) context.Context { return context.WithValue(ctx, RequestIDKey, id) } func RequestIDFromContext(ctx context.Context) string { if id, ok := ctx.Value(RequestIDKey).(string); ok { return id } return "" } // ========== 认证中间件 (带 Context) ========== func AuthWithContext(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { token := r.Header.Get("Authorization") if token == "" { http.Error(w, `{"error":"unauthorized"}`, 401) return } user, err := validateToken(token) if err != nil { http.Error(w, `{"error":"invalid token"}`, 401) return } // 用户信息存入 context ctx := WithUser(r.Context(), user) ctx = WithRequestID(ctx, generateID()) next.ServeHTTP(w, r.WithContext(ctx)) }) } // Handler 中获取用户 func userHandler(w http.ResponseWriter, r *http.Request) { user := UserFromContext(r.Context()) if user == nil { http.Error(w, "unauthorized", 401) return } json.NewEncoder(w).Encode(user) } // ========== 请求 ID 中间件 ========== func RequestID(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { id := r.Header.Get("X-Request-ID") if id == "" { id = uuid.New().String() } ctx := WithRequestID(r.Context(), id) w.Header().Set("X-Request-ID", id) next.ServeHTTP(w, r.WithContext(ctx)) }) } // ========== 速率限制中间件 ========== func RateLimit(rps int) Middleware { limiter := NewTokenBucket(float64(rps), float64(rps)) return func(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { if !limiter.Allow() { w.Header().Set("Retry-After", "1") http.Error(w, `{"error":"rate limited"}`, 429) return } next.ServeHTTP(w, r) }) } } // ========== 超时中间件 ========== func Timeout(duration time.Duration) Middleware { return func(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { ctx, cancel := context.WithTimeout(r.Context(), duration) defer cancel() next.ServeHTTP(w, r.WithContext(ctx)) }) } } // ========== 组合使用 ========== func setupRouter() http.Handler { mux := http.NewServeMux() mux.HandleFunc("/api/users", userHandler) mux.HandleFunc("/api/ping", pingHandler) // 全栈中间件 handler := Chain(mux, Recovery, // 1. panic 恢复 RequestID, // 2. 请求 ID Logger, // 3. 日志 CORS, // 4. CORS Timeout(30*time.Second), // 5. 超时 RateLimit(100), // 6. 限流 AuthWithContext, // 7. 认证 ) return handler }

中间件实践

// ========== 第三方中间件库 ========== // 常用中间件 (go-chi/chi, gorilla/mux, alice) // 1. chi 路由 + 中间件 // import "github.com/go-chi/chi/v5" // // r := chi.NewRouter() // r.Use(middleware.Logger) // r.Use(middleware.Recoverer) // r.Use(middleware.Timeout(30 * time.Second)) // // r.Get("/api/users", listUsers) // r.With(authMiddleware).Get("/api/admin", adminHandler) // 2. alice 链式工具 // import "github.com/justinas/alice" // // c := alice.New(Recovery, Logger, CORS) // handler := c.Then(mux) // 3. gorilla/mux // import "github.com/gorilla/mux" // // r := mux.NewRouter() // r.Use(loggingMiddleware) // r.HandleFunc("/users/{id}", getUser).Methods("GET") // ========== 中间件路径匹配 ========== // 某些中间件只对特定路径生效 func PathMiddleware(path string, mw Middleware) Middleware { return func(next http.Handler) http.Handler { return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) { if strings.HasPrefix(r.URL.Path, path) { mw(next).ServeHTTP(w, r) return } next.ServeHTTP(w, r) }) } } // 使用: adminAuth := PathMiddleware("/admin", AuthWithContext) // ========== 中间件顺序 ========== // 执行顺序从左到右, 但包装从右到左 // Recovery(Logger(Handler)) // 执行: Recovery → Logger → Handler → Logger' → Recovery' // 通常顺序: // 1. Recovery (最外层, 捕获所有 panic) // 2. RequestID / Trace // 3. Logger // 4. CORS // 5. Timeout // 6. RateLimit // 7. Auth (最内层, handler 之前) // ========== 中间件性能 ========== // 每个中间件增加 ~1μs 开销 // 10 个中间件 ~10μs // 避免在中间件中做重操作 (DB 查询等) // ========== 简洁实现: ResponseWriter 包装 ========== type wrappedWriter struct { http.ResponseWriter status int written int64 } func wrapWriter(w http.ResponseWriter) *wrappedWriter { return &wrappedWriter{ResponseWriter: w, status: 200} } func (w *wrappedWriter) WriteHeader(status int) { w.status = status w.ResponseWriter.WriteHeader(status) } func (w *wrappedWriter) Write(b []byte) (int, error) { n, err := w.ResponseWriter.Write(b) w.written += int64(n) return n, err}
💡 中间件要点: type Middleware func(http.Handler) http.Handler; 链式组合 Chain(handler, mws...); 包装 ResponseWriter 捕获状态码; context.WithValue 传递用户/请求ID; 安全 key 用自定义类型; 中间件顺序: Recovery → Logger → CORS → Timeout → RateLimit → Auth; PathMiddleware 路径匹配; chi/alice/gorilla 第三方中间件; 避免中间件中做 DB 查询; 执行顺序像洋葱: 从外到内再到外。
