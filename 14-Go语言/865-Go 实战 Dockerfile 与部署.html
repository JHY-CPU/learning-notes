Go 多阶段 Dockerfile

// ========== 最优 Go Dockerfile ========== // Dockerfile // === 阶段 1: 编译 === FROM golang:1.22-alpine AS builder RUN apk add --no-cache git ca-certificates WORKDIR /app // 利用 Docker 层缓存 COPY go.mod go.sum ./ RUN go mod download COPY . . // 静态编译, 禁用 CGO, 去除调试信息 RUN CGO_ENABLED=0 GOOS=linux GOARCH=amd64 \ go build -ldflags="-w -s" -o /app/server . // === 阶段 2: 运行 === FROM alpine:3.19 RUN apk add --no-cache ca-certificates tzdata // 非 root 用户运行 RUN adduser -D -g '' appuser COPY --from=builder /app/server /app/server COPY --from=builder /etc/ssl/certs /etc/ssl/certs USER appuser WORKDIR /app EXPOSE 8080 HEALTHCHECK --interval=30s --timeout=3s --retries=3 \ CMD wget --no-verbose --tries=1 --spider http://localhost:8080/health || exit 1 ENTRYPOINT ["/app/server"] // ========== 更小的镜像: distroless ========== // FROM golang:1.22-alpine AS builder // ... // // FROM gcr.io/distroless/static-debian12:nonroot // COPY --from=builder /app/server /server // EXPOSE 8080 // ENTRYPOINT ["/server"] // distroless 只有应用 + 运行时依赖 // 没有 shell, 没有包管理器, 攻击面最小 // ========== 更小的镜像: scratch ========== // FROM scratch // COPY --from=builder /app/server /server // COPY --from=builder /etc/ssl/certs /etc/ssl/certs // EXPOSE 8080 // ENTRYPOINT ["/server"] // scratch 完全空镜像, 最小可能 // 注意: 没有 shell, 调试困难 // ========== Docker 构建命令 ========== // docker build -t myapp:latest . // docker build --platform linux/amd64 -t myapp:latest . // docker build --build-arg VERSION=1.0.0 -t myapp:latest . // 使用 buildx 多架构: // docker buildx build \ // --platform linux/amd64,linux/arm64 \ // -t myapp:latest --push .

// ========== 12-Factor App 配置 ========== // 1. 环境变量存储配置 // 2. 不硬编码敏感信息 // 3. 默认值可本地开发 func loadConfig() *Config { return &Config{ Port: getEnv("PORT", ":8080"), DSN: getEnv("DATABASE_DSN", ""), Redis: getEnv("REDIS_ADDR", "localhost:6379"), Secret: getEnv("JWT_SECRET", "dev-secret"), } } // ========== 健康检查端点 ========== func healthHandler(c *gin.Context) { status := "ok" code := 200 // 检查数据库 if err := db.Ping(); err != nil { status = "degraded" code = 503 } c.JSON(code, gin.H{ "status": status, "version": Version, "time": time.Now().Unix(), }) } // ========== 优雅关闭 ========== func gracefulShutdown(srv *http.Server) { quit := make(chan os.Signal, 1) signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM) <-quit log.Println("正在关闭服务器...") ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second) defer cancel() if err := srv.Shutdown(ctx); err != nil { log.Fatalf("强制关闭: %v", err) } log.Println("服务器已退出") } // main 函数整合: func main() { cfg := loadConfig() db := initDB(cfg.DSN) r := setupRouter(db) srv := &http.Server{ Addr: cfg.Port, Handler: r, ReadTimeout: 10 * time.Second, WriteTimeout: 15 * time.Second, IdleTimeout: 60 * time.Second, } go func() { log.Printf("服务启动在 %s", cfg.Port) if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed { log.Fatalf("启动失败: %v", err) } }() gracefulShutdown(srv) } // ========== Nginx 反向代理配置 ========== // /etc/nginx/sites-available/myapp // server { // listen 80; CI/CD 部署流水线

// ========== GitHub Actions ========== // .github/workflows/deploy.yml // name: Build and Deploy // // on: // push: // branches: [main] // // jobs: // test: // runs-on: ubuntu-latest // steps: // - uses: actions/checkout@v4 // - uses: actions/setup-go@v5 // with: // go-version: '1.22' // - run: go mod download // - run: go vet ./... // - run: go test -race -coverprofile=coverage.out ./... // - uses: codecov/codecov-action@v3 // // build-and-deploy: // needs: test // runs-on: ubuntu-latest // steps: // - uses: actions/checkout@v4 // // - name: Docker Buildx // uses: docker/setup-buildx-action@v3 // // - name: Login to Registry // uses: docker/login-action@v3 // with: // registry: ghcr.io // username: ${{ github.actor }} // password: ${{ secrets.GITHUB_TOKEN }} // // - name: Build and Push // uses: docker/build-push-action@v5 // with: // context: . // platforms: linux/amd64,linux/arm64 // push: true // tags: | // ghcr.io/${{ github.repository }}:latest // ghcr.io/${{ github.repository }}:${{ github.sha }} // ========== 部署到服务器 ========== // 方式 1: SSH + Docker // ssh user@host "cd /app && docker compose pull && docker compose up -d" // 方式 2: Ansible Playbook // - name: Deploy application // hosts: app-servers // tasks: // - name: Pull latest image // docker_image: // name: ghcr.io/myorg/myapp // source: pull // - name: Restart container // docker_container: // name: myapp // image: ghcr.io/myorg/myapp:latest // state: started // restart: yes // ========== 部署检查清单 ========== // [ ] 多阶段 Dockerfile (builder + distroless/alpine) // [ ] 非 root 用户运行 // [ ] 健康检查端点 + Docker HEALTHCHECK // [ ] 优雅关闭 (signal.Notify + Shutdown) // [ ] 环境变量配置 (12-Factor) // [ ] Docker Compose 编排 (app + db + redis) // [ ] 资源限制 (CPU/内存) // [ ] 日志驱动配置 // [ ] SSL/TLS (Nginx + Let's Encrypt) // [ ] CI/CD 自动化 // [ ] 数据库迁移自动化 // [ ] 备份策略
💡 部署要点: 多阶段构建 (builder → alpine/distroless/scratch); 最小镜像 (~10MB); 非 root 用户 USER appuser; HEALTHCHECK 健康检查; 环境变量配置 (12-Factor); Docker Compose 编排多服务; depends_on + condition: service_healthy; 优雅关闭 Shutdown; Nginx 反向代理; CI/CD GitHub Actions Build + Push; 生产资源限制; 跨平台构建 buildx。
