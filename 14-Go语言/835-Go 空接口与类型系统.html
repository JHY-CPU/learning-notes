空接口 (interface{} / any)

// ========== interface{} 与 any ========== // Go 1.18+ any 是 interface{} 的别名 // type any = interface{} func anyDemo() { var i any // 等价于 var i interface{} i = 42 i = "hello" i = struct{ Name string }{"Alice"} i = []int{1, 2, 3} i = func() { fmt.Println("function") } fmt.Println(i) } // ========== 空接口的常见场景 ========== // 1. 打印任意值 func PrintAny(v any) { fmt.Printf("值: %v, 类型: %T\n", v, v) } // 2. 异构容器 func heteroContainer() { var items []any items = append(items, 1, "hello", true, 3.14) for _, item := range items { switch v := item.(type) { case int: fmt.Println("整数:", v*2) case string: fmt.Println("字符串:", len(v)) } } } // 3. JSON 反序列化 (不确定的结构) func dynamicJSON() { data := `{"name":"Alice","age":30,"tags":["go","web"]}` var result map[string]any json.Unmarshal([]byte(data), &result) for k, v := range result { switch val := v.(type) { case string: fmt.Printf("%s = %s\n", k, val) case float64: fmt.Printf("%s = %f\n", k, val) case []any: fmt.Printf("%s = %v\n", k, val) } } } // 4. 泛型之前的通用数据结构 // (泛型后可替代) type StackOld struct { items []any } func (s *StackOld) Push(item any) { s.items = append(s.items, item) } func (s *StackOld) Pop() (any, bool) { if len(s.items) == 0 { return nil, false } item := s.items[len(s.items)-1] s.items = s.items[:len(s.items)-1] return item, true } // 使用: stack.Push("hello"); v, _ := stack.Pop(); s := v.(string)

类型约束与泛型

// ========== 类型约束 (Type Constraints) ========== // 限制类型参数可接受的具体类型 // 1. any 约束 (无限制) func Print[T any](v T) { fmt.Println(v) } // 2. 内置约束 func Sum[T ~int | ~float64](vals []T) T { var total T for _, v := range vals { total += v } return total } // 3. 接口约束 type Stringer interface { String() string } func PrintWithString[T Stringer](v T) { fmt.Println(v.String()) } // 4. comparable (内置约束, 可比较) func Contains[T comparable](slice []T, target T) bool { for _, v := range slice { if v == target { return true } } return false } // 5. ~ 近似约束 (接受底层类型) type MyInt int // Sum 用 ~int 则 Sum[MyInt] 合法 // ========== 自定义类型约束 ========== // Go 1.18+ 类型集 (Type Set) // 数字类型约束 type Number interface { ~int | ~int8 | ~int16 | ~int32 | ~int64 | ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64 | ~float32 | ~float64 } // 可排序类型 type Ordered interface { ~int | ~int8 | ~int16 | ~int32 | ~int64 | ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64 | ~float32 | ~float64 | ~string } // 泛型函数 func Min[T Ordered](a, b T) T { if a < b { return a } return b } func Max[T Ordered](a, b T) T { if a > b { return a } return b } // ========== 接口 vs 泛型选择 ========== // 接口: 运行时多态 (动态分派) // 泛型: 编译期多态 (生成专用代码) // 接口风格 (动态) type Sorter interface { Sort([]int) } // 泛型风格 (静态) func Sort[T Ordered](s []T) { sort.Slice(s, func(i, j int) bool { return s[i] < s[j] }) } // 选择准则: // - 不同类型但相同行为 → 接口 // - 不同行为但类型参数 → 泛型 // - 混合: 泛型函数 + 接口约束

隐式接口满足

// ========== Go 隐式接口满足 ========== // 类型自动满足接口, 无需 implements 声明 // 优点: 松耦合, 易适配 // 定义接口 type Logger interface { Log(message string) Close() error } // 隐式实现: 无需 `implements Logger` type FileLogger struct { file *os.File } func (l *FileLogger) Log(msg string) { l.file.WriteString(msg + "\n") } func (l *FileLogger) Close() error { return l.file.Close() } // 另一个隐式实现 type ConsoleLogger struct{} func (l ConsoleLogger) Log(msg string) { fmt.Println(msg) } func (l ConsoleLogger) Close() error { return nil } // ========== 隐式接口的优势 ========== // 1. 松耦合: 接口定义和使用方解耦 // 包定义接口, 用户代码无需导入 type Database interface { Query(sql string) ([]Row, error) } // 适配任何满足接口的类型 type MySQLDB struct{ /* ... */ } func (db *MySQLDB) Query(sql string) ([]Row, error) { // 实现... return nil, nil } // 2. 接口定义在消费方 (而非实现方) // 标准库示例: // io.Reader / io.Writer 定义在 io 包 // 很多包的类型隐式满足它们 // 3. 接口可后加 — 一个类型可无痛满足新接口 // 这是 Go 接口设计的核心哲学 // ========== 空接口没有方法 ========== // interface{} / any 没有方法, 所有类型都满足 // ========== 接口 = 协议 ========== // Go 的接口是隐式协议, 类似 Python 的鸭子类型 // "如果它走起来像鸭子, 叫起来像鸭子, 那它就是鸭子" // 示例: 接受任何有 ServeHTTP 方法的类型 type HTTPServer interface { ServeHTTP(w http.ResponseWriter, r *http.Request) } // http.Handler 就是这样一个接口 // 任何有 ServeHTTP 方法的类型都可用 // ========== 显式检查 ========== // 有时需要确保类型实现接口 (编译期检查) var _ Logger = (*FileLogger)(nil) // 编译期检查 var _ Logger = ConsoleLogger{} // 编译期检查 // 如果 FileLogger 没实现 Logger, 编译报错 // 这种模式常用于包级别声明

类型安全设计

// ========== Go 类型安全哲学 ========== // 1. 静态类型: 编译期检查 // 2. 隐式接口: 运行时灵活 // 3. 零值: 初始化安全性 // 4. 无类型继承: 避免复杂层次 // ========== 常见安全模式 ========== // 1. 用新类型增强语义 type UserID int64 type OrderID string func ProcessUser(id UserID) { /* 明确类型 */ } func ProcessOrder(id OrderID) { /* 明确类型 */ } // ProcessUser(OrderID("xxx")) — 编译错误! // 2. 枚举用 const + iota (带类型) type Status int const ( StatusPending Status = iota StatusActive StatusInactive StatusDeleted ) // Status(5) 不合法 (但不会被 Go 编译拦截, 运行时校验) // 3. 可选值用指针 (nil = 未设置) type UpdateRequest struct { Name *string `json:"name,omitempty"` // nil = 不更新 Age *int `json:"age,omitempty"` } // 4. 私有结构体 + 公开接口 type Repository interface { FindByID(id int) (*User, error) Save(user *User) error } // 只有 NewRepository 返回接口, 具体实现私有 type postgresRepo struct{ db *sql.DB } func (r *postgresRepo) FindByID(id int) (*User, error) { /* ... */ } func (r *postgresRepo) Save(user *User) error { /* ... */ } func NewRepository(db *sql.DB) Repository { return &postgresRepo{db: db} } // ========== 类型安全 vs 泛型 ========== // 泛型之前: interface{} + 断言 (运行时安全) // 泛型之后: 类型参数 (编译期安全) // 旧方式: interface{} + 类型断言 func FilterOld(items []interface{}, fn func(interface{}) bool) []interface{} { var result []interface{} for _, item := range items { if fn(item) { result = append(result, item) } } return result } // 新方式: 泛型 func FilterNew[T any](items []T, fn func(T) bool) []T { var result []T // 类型安全! 零值初始化 for _, item := range items { if fn(item) { result = append(result, item) } } return result } // FilterNew([]int{1,2,3}, func(v int) bool { return v > 1 })
💡 类型系统要点: any = interface{} (Go 1.18+); 空接口通用容器 + JSON 动态解析; 泛型约束 (any/comparable/Ordered/自定义); ~ 表示接受底层类型; 隐式接口满足: 松耦合, 无需 implements; 接口定义在消费方, 非实现方; var _ Interface = (*T)(nil) 编译期检查接口实现; 新类型 (type UserID int64) 增强类型安全; 泛型替代 interface{} + 断言, 编译期类型安全。
