# 路由导航与参数

React Router 提供了声明式和命令式两种导航方式，以及丰富的参数管理 API。

---

## 一、声明式导航

### 1.1 Link

```jsx
import { Link } from 'react-router-dom';

// 基础用法
<Link to="/about">关于</Link>

// 绝对路径
<Link to="/users/123">用户详情</Link>

// 相对路径（相对于当前路由）
<Link to="settings">设置</Link>     // 当前在 /dashboard → /dashboard/settings
<Link to="../profile">个人资料</Link> // 当前在 /dashboard → /profile
<Link to="./edit">编辑</Link>        // 当前在 /dashboard → /dashboard/edit

// 带查询参数
<Link to="/search?q=react&page=2">搜索</Link>

// 带 hash
<Link to="/docs#installation">安装说明</Link>

// 带 state（不会出现在 URL 中）
<Link to="/dashboard" state={{ from: location.pathname }}>
  Dashboard
</Link>

// 替换历史记录（不能后退）
<Link to="/login" replace>登录</Link>
```

### 1.2 NavLink

```jsx
import { NavLink } from 'react-router-dom';

// className 函数
<NavLink
  to="/dashboard"
  className={({ isActive, isPending }) =>
    isActive ? 'nav-active' : isPending ? 'nav-pending' : ''
  }
>
  仪表盘
</NavLink>

// style 函数
<NavLink
  to="/settings"
  style={({ isActive }) => ({
    color: isActive ? '#fff' : '#333',
    background: isActive ? '#007bff' : 'transparent',
  })}
>
  设置
</NavLink>

// end 精确匹配
<NavLink to="/" end>首页</NavLink>
// / 时 active ✓
// /about 时 不 active ✓
// 不加 end：/about 时也 active ✗

// 区分 isActive 和 isPending
<NavLink
  to="/about"
  className={({ isActive, isPending, isTransitioning }) =>
    [
      isPending ? "pending" : "",
      isActive ? "active" : "",
      isTransitioning ? "transitioning" : "",
    ].join(" ")
  }
>
  关于
</NavLink>
```

## 二、命令式导航

### 2.1 useNavigate

```jsx
import { useNavigate } from 'react-router-dom';

function LoginPage() {
  const navigate = useNavigate();

  const handleLogin = async (credentials) => {
    await apiLogin(credentials);
    navigate('/dashboard');  // 跳转到仪表盘
  };

  return <form onSubmit={handleLogin}>...</form>;
}
```

### 2.2 navigate 的参数

```jsx
const navigate = useNavigate();

// 基础跳转
navigate('/about');

// 相对路径
navigate('..');           // 上一级
navigate('../sibling');   // 兄弟路由
navigate(-1);            // 后退（等同于浏览器后退按钮）
navigate(-2);            // 后退两步

// 带 state
navigate('/dashboard', {
  state: { from: '/login', message: '登录成功' },
});

// 替换当前记录（不可后退）
navigate('/home', { replace: true });

// 带查询参数
navigate({
  pathname: '/search',
  search: '?q=react&page=1',
});

// 或使用 URLSearchParams
const params = new URLSearchParams({ q: 'react', page: '1' });
navigate(`/search?${params.toString()}`);
```

### 2.3 push vs replace

```jsx
// push（默认）：在历史记录栈中添加新记录
navigate('/page2');
// 历史栈: [page1] → [page1, page2]
// 用户可以后退到 page1

// replace：替换当前记录
navigate('/page2', { replace: true });
// 历史栈: [page1] → [page2]
// 用户不能后退到 page1

// 典型使用场景：
// 登录后跳转 → replace（不希望用户后退到登录页）
// 表单提交成功后 → replace
// 正常页面跳转 → push（默认）
```

## 三、useLocation

获取当前 URL 的详细信息。

```jsx
import { useLocation } from 'react-router-dom';

function DebugInfo() {
  const location = useLocation();
  // location 对象结构：
  // {
  //   pathname: '/users/123',
  //   search: '?tab=profile',
  //   hash: '#comments',
  //   state: { from: '/home' },
  //   key: 'abc123'  // 唯一标识
  // }

  return (
    <div>
      <p>路径: {location.pathname}</p>
      <p>查询: {location.search}</p>
      <p>Hash: {location.hash}</p>
      <p>来源: {location.state?.from}</p>
    </div>
  );
}

// 典型用途
function useQueryParams() {
  const { search } = useLocation();
  return useMemo(() => new URLSearchParams(search), [search]);
}

// 使用
function SearchPage() {
  const params = useQueryParams();
  const query = params.get('q');
  const page = params.get('page');
  // ...
}
```

## 四、路径模式

### 4.1 路径匹配规则

```jsx
// 静态路径
<Route path="/about" />        // 匹配 /about

// 动态参数
<Route path="/users/:id" />    // 匹配 /users/123, /users/abc
// useParams() → { id: "123" }

// 可选参数
<Route path="/posts/:id?" />   // 匹配 /posts 和 /posts/123

// 通配符
<Route path="*" />             // 匹配任何路径
<Route path="/files/*" />      // 匹配 /files/ 及其所有子路径
// useParams() → { "*": "documents/report.pdf" }

// 多个参数
<Route path="/users/:userId/posts/:postId" />
// /users/1/posts/42 → { userId: "1", postId: "42" }
```

### 4.2 相对路径 vs 绝对路径

```jsx
// 在 /dashboard 路由的组件中

// 绝对路径（以 / 开头）
<Link to="/settings">设置</Link>  // → /settings

// 相对路径（不以 / 开头）
<Link to="settings">设置</Link>   // → /dashboard/settings
<Link to="./settings">设置</Link> // → /dashboard/settings
<Link to="../profile">资料</Link> // → /profile
<Link to="..">返回</Link>        // → /
```

### 4.3 useMatch

检测当前 URL 是否匹配某个路径模式。

```jsx
import { useMatch } from 'react-router-dom';

function Navigation() {
  const isHome = useMatch('/');
  const isUserPage = useMatch('/users/:id');

  return (
    <nav>
      <span>当前在首页: {isHome ? '是' : '否'}</span>
      <span>用户页面 ID: {isUserPage?.params.id || 'N/A'}</span>
    </nav>
  );
}
```

## 五、编程式导航进阶

### 5.1 表单提交后跳转

```jsx
function CreatePost() {
  const navigate = useNavigate();
  const location = useLocation();

  const handleSubmit = async (data) => {
    const newPost = await api.createPost(data);
    // 跳转到新创建的帖子

      state: { message: '帖子创建成功' },
    });
  };

  return <PostForm onSubmit={handleSubmit} />;

```

### 5.2 保存和恢复滚动位置

```jsx
function ProductList() {
  const navigate = useNavigate();
  const location = useLocation();

  const handleProductClick = (productId) => {
    // 保存当前滚动位置到 state
    navigate(`/products/${productId}`, {
      state: { scrollPosition: window.scrollY },
    });
  };

  // 返回时恢复滚动位置
  useEffect(() => {
    if (location.state?.scrollPosition) {
      window.scrollTo(0, location.state.scrollPosition);
    }
  }, [location.state]);

  return products.map(p => (
    <div key={p.id} onClick={() => handleProductClick(p.id)}>
      {p.name}
    </div>
  ));
}
```

### 5.3 确认导航（防离开）

```jsx
function usePrompt(message, when) {
  const navigate = useNavigate();

  useEffect(() => {
    if (!when) return;

    const handleBeforeUnload = (e) => {
      e.preventDefault();
      e.returnValue = message;
    };

    window.addEventListener('beforeunload', handleBeforeUnload);
    return () => window.removeEventListener('beforeunload', handleBeforeUnload);
  }, [message, when]);
}

// 使用
function EditForm() {
  const [dirty, setDirty] = useState(false);
  usePrompt('表单有未保存的更改，确定要离开吗？', dirty);

  return (
    <form>
      <input onChange={() => setDirty(true)} />
    </form>
  );
}
```

## 六、useSearchParams 进阶

### 6.1 复杂查询参数管理

```jsx
import { useSearchParams } from 'react-router-dom';

function ProductFilters() {
  const [searchParams, setSearchParams] = useSearchParams();

  // 读取当前值
  const filters = {
    category: searchParams.get('category') || 'all',
    minPrice: searchParams.get('minPrice') || '',
    maxPrice: searchParams.get('maxPrice') || '',
    sort: searchParams.get('sort') || 'name',
    page: parseInt(searchParams.get('page') || '1'),
  };

  // 更新单个参数
  const setFilter = (key, value) => {
    setSearchParams(prev => {
      if (value) {
        prev.set(key, value);
      } else {
        prev.delete(key);
      }
      prev.set('page', '1');  // 切换筛选时重置分页
      return prev;
    }, { replace: true });  // replace 避免产生大量历史记录
  };

  // 批量更新
  const resetFilters = () => {
    setSearchParams({}, { replace: true });
  };

  return (
    <div>
      <select
        value={filters.category}
        onChange={(e) => setFilter('category', e.target.value)}
      >
        <option value="all">全部分类</option>
        <option value="electronics">电子产品</option>
      </select>

      <input
        type="number"
        value={filters.minPrice}
        onChange={(e) => setFilter('minPrice', e.target.value)}
        placeholder="最低价格"
      />

      <button onClick={resetFilters}>重置筛选</button>
    </div>
  );
}
```

### 6.2 URL 与状态同步

```jsx
function useUrlState(key, defaultValue) {
  const [searchParams, setSearchParams] = useSearchParams();

  const value = searchParams.get(key) || defaultValue;

  const setValue = (newValue) => {
    setSearchParams(prev => {
      if (newValue === defaultValue || !newValue) {
        prev.delete(key);
      } else {
        prev.set(key, String(newValue));
      }
      return prev;
    }, { replace: true });
  };

  return [value, setValue];
}

// 使用
function SearchPage() {
  const [query, setQuery] = useUrlState('q', '');
  const [page, setPage] = useUrlState('page', '1');

  // URL 和 state 自动同步
  // 分享 URL 时可以恢复完整的筛选状态
}
```
