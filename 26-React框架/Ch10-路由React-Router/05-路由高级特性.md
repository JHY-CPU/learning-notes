# 路由高级特性

React Router 6.4+ 引入了数据加载（loader）、数据变更（action）、deferred 响应等高级特性，使路由具备了全栈框架的数据管理能力。

---

## 一、Data Loading (loader)

### 1.1 基础 Loader

```jsx
import { createBrowserRouter, useLoaderData } from 'react-router-dom';

const router = createBrowserRouter([
  {
    path: '/products',
    element: <Products />,
    // loader 在渲染前执行，返回的数据供组件使用
    loader: async () => {
      const response = await fetch('/api/products');
      if (!response.ok) throw new Response('加载失败', { status: 500 });
      return response.json();
    },
  },
]);

function Products() {
  const products = useLoaderData();  // 获取 loader 返回的数据

  return (
    <ul>
      {products.map(p => <li key={p.id}>{p.name}</li>)}
    </ul>
  );
}
```

### 1.2 Loader 参数

```jsx
const router = createBrowserRouter([
  {
    path: '/products/:id',
    element: <ProductDetail />,
    loader: async ({ params, request }) => {
      // params: 路径参数 → { id: '123' }
      // request: Request 对象，包含 URL、headers 等
      const { id } = params;
      const url = new URL(request.url);
      const includeReviews = url.searchParams.get('reviews') === 'true';

      const product = await fetchProduct(id);
      return { product };
    },
  },
]);

function ProductDetail() {
  const { product } = useLoaderData();
  return <h1>{product.name}</h1>;
}
```

### 1.3 嵌套路由 Loader

子路由的 loader 在父路由的 loader 之后执行。可以并行加载提升性能。

```jsx
const router = createBrowserRouter([
  {
    path: '/dashboard',
    element: <DashboardLayout />,
    loader: async () => fetchDashboardData(),
    children: [
      {
        index: true,
        element: <DashboardHome />,
        // 子 loader 与父 loader 并行执行
        loader: async () => fetchRecentActivity(),
      },
      {
        path: 'analytics',
        element: <Analytics />,
        loader: async () => fetchAnalyticsData(),
      },
    ],
  },
]);

// 子路由获取父 loader 数据
function Analytics() {
  const analytics = useLoaderData();
  const dashboard = useRouteLoaderData('dashboard');  // 获取指定 ID 路由的数据
  // ...
}
```

### 1.4 Loader 缓存

React Router 自动缓存 loader 数据。用户通过后退/前进导航时不会重新执行 loader。

```jsx
// 自定义缓存策略
loader: async ({ params }) => {
  // 检查缓存
  const cached = cache.get(`product-${params.id}`);
  if (cached && Date.now() - cached.timestamp < 60000) {
    return cached.data;
  }

  const data = await fetchProduct(params.id);
  cache.set(`product-${params.id}`, { data, timestamp: Date.now() });
  return data;
}
```

## 二、Data Mutation (action)

### 2.1 基础 Action

Action 处理表单提交等数据变更操作。

```jsx
import { createBrowserRouter, Form, useActionData } from 'react-router-dom';

const router = createBrowserRouter([
  {
    path: '/contact',
    element: <Contact />,
    action: async ({ request }) => {
      // 获取表单数据
      const formData = await request.formData();
      const name = formData.get('name');
      const email = formData.get('email');

      // 验证
      const errors = {};
      if (!name) errors.name = '姓名不能为空';
      if (!email.includes('@')) errors.email = '邮箱格式不正确';

      if (Object.keys(errors).length > 0) {
        return { errors };  // 返回错误给组件
      }

      // 提交
      await submitContact({ name, email });
      return { success: true };
    },
  },
]);

function Contact() {
  const actionData = useActionData();
  const navigation = useNavigation();
  const isSubmitting = navigation.state === 'submitting';

  return (
    <Form method="post">
      <input name="name" />
      {actionData?.errors?.name && <span>{actionData.errors.name}</span>}

      <input name="email" type="email" />
      {actionData?.errors?.email && <span>{actionData.errors.email}</span>}

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? '提交中...' : '提交'}
      </button>

      {actionData?.success && <p>提交成功！</p>}
    </Form>
  );
}
```

### 2.2 Form 组件

React Router 的 `Form` 组件在不启用 JavaScript 时也能工作（渐进增强）。

```jsx
import { Form } from 'react-router-dom';

// 普通表单提交
<Form method="post">
  <input name="title" />
  <button type="submit">保存</button>
</Form>

// 带 URL 参数
<Form action="/products/123" method="delete">
  <button type="submit">删除</button>
</Form>

// 相对 action
<Form action="../settings" method="patch">
  <input name="theme" />
  <button type="submit">更新</button>
</Form>

// 用 fetcher 而不导航（见下文）
```

### 2.3 带参数的 Action

```jsx
const router = createBrowserRouter([
  {
    path: 'products/:id',
    element: <ProductDetail />,
    loader: async ({ params }) => fetchProduct(params.id),
    action: async ({ params, request }) => {
      const formData = await request.formData();

      switch (request.method) {
        case 'PUT':
          await updateProduct(params.id, Object.fromEntries(formData));
          return { success: true };
        case 'DELETE':
          await deleteProduct(params.id);
          return redirect('/products');
      }
    },
  },
]);
```

## 三、useFetcher

`useFetcher` 允许在不触发导航的情况下与 loaders 和 actions 交互。

### 3.1 静默提交

```jsx
import { useFetcher } from 'react-router-dom';

function AddToCartButton({ productId }) {
  const fetcher = useFetcher();

  const isAdding = fetcher.state !== 'idle';

  return (
    <fetcher.Form method="post" action={`/cart/add`}>
      <input type="hidden" name="productId" value={productId} />
      <button type="submit" disabled={isAdding}>
        {isAdding ? '添加中...' : '加入购物车'}
      </button>
    </fetcher.Form>
  );
}

// 路由配置
{
  path: 'cart/add',
  action: async ({ request }) => {
    const formData = await request.formData();
    await addToCart(formData.get('productId'));
    return { success: true };
  },
}
```

### 3.2 内联编辑

```jsx
function InlineEdit({ todo }) {
  const fetcher = useFetcher();
  const isSaving = fetcher.state !== 'idle';

  return (
    <fetcher.Form method="patch" action={`/todos/${todo.id}`}>
      <input
        name="text"
        defaultValue={todo.text}
        onBlur={(e) => {
          if (e.target.value !== todo.text) {
            fetcher.submit(
              { text: e.target.value },
              { method: 'patch', action: `/todos/${todo.id}` }
            );
          }
        }}
      />
      {isSaving && <span>保存中...</span>}
    </fetcher.Form>
  );
}
```

### 3.3 乐观更新

```jsx
import { useFetcher, useFetchers } from 'react-router-dom';

function TodoList({ todos }) {
  const fetcher = useFetcher();

  // 检查是否有正在进行的 "TOGGLE" 操作
  const optimisticTodos = todos.map(todo => {
    const toggleAction = useFetchers().find(
      f => f.formData?.get('todoId') === String(todo.id)
          && f.formMethod === 'PATCH'
    );
    if (toggleAction) {
      return { ...todo, done: !todo.done };  // 乐观更新
    }
    return todo;
  });

  return (
    <ul>
      {optimisticTodos.map(todo => (
        <li key={todo.id}>
          <fetcher.Form method="patch">
            <input type="hidden" name="todoId" value={todo.id} />
            <button type="submit" style={{
              textDecoration: todo.done ? 'line-through' : 'none',
            }}>
              {todo.text}
            </button>
          </fetcher.Form>
        </li>
      ))}
    </ul>
  );
}
```

## 四、defer + Await

`defer` 允许在数据还未完全加载时就开始渲染页面，对慢数据实现流式加载。

### 4.1 基础用法

```jsx
import { defer, Await, useLoaderData } from 'react-router-dom';
import { Suspense } from 'react';

const router = createBrowserRouter([
  {
    path: '/dashboard',
    element: <Dashboard />,
    loader: async () => {
      // 关键数据：必须等待
      const user = await fetchUser();

      // 非关键数据：可以延迟
      const analytics = fetchAnalytics();  // 注意：不 await，返回 Promise

      return defer({
        user,           // 立即可用的数据
        analytics,      // Promise，稍后可用
      });
    },
  },
]);

function Dashboard() {
  const { user, analytics } = useLoaderData();

  return (
    <div>
      {/* 立即可用的数据 */}
      <h1>欢迎, {user.name}</h1>

      {/* 延迟加载的数据 */}
      <Suspense fallback={<AnalyticsSkeleton />}>
        <Await resolve={analytics} errorElement={<AnalyticsError />}>
          {(resolvedAnalytics) => (
            <AnalyticsChart data={resolvedAnalytics} />
          )}
        </Await>
      </Suspense>
    </div>
  );
}
```

### 4.2 多个延迟数据

```jsx
loader: async () => {
  const user = await fetchUser();

  return defer({
    user,
    orders: fetchOrders(),          // 延迟
    notifications: fetchNotifications(),  // 延迟
    recommendations: fetchRecommendations(),  // 延迟
  });
},

function Dashboard() {
  const { user, orders, notifications, recommendations } = useLoaderData();

  return (
    <div>
      <h1>欢迎, {user.name}</h1>

      <div className="dashboard-grid">
        <Suspense fallback={<OrdersSkeleton />}>
          <Await resolve={orders}>
            {(data) => <OrderList orders={data} />}
          </Await>
        </Suspense>

        <Suspense fallback={<NotificationsSkeleton />}>
          <Await resolve={notifications}>
            {(data) => <NotificationList items={data} />}
          </Await>
        </Suspense>

        <Suspense fallback={<RecommendationsSkeleton />}>
          <Await resolve={recommendations}>
            {(data) => <RecommendationList items={data} />}
          </Await>
        </Suspense>
      </div>
    </div>
  );
}
```

## 五、错误处理

### 5.1 Error Element

```jsx
const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    errorElement: <RootError />,  // 顶级错误边界
    children: [
      {
        path: 'products',
        element: <Products />,
        loader: async () => {
          const res = await fetch('/api/products');
          if (!res.ok) throw new Response('产品加载失败', { status: 500 });
          return res.json();
        },
        errorElement: <ProductsError />,  // 路由级错误边界
      },
    ],
  },
]);
```

### 5.2 错误处理组件

```jsx
import { useRouteError, isRouteErrorResponse } from 'react-router-dom';

function RootError() {
  const error = useRouteError();

  if (isRouteErrorResponse(error)) {
    // HTTP 错误响应
    return (
      <div>
        <h1>{error.status} {error.statusText}</h1>
        <p>{error.data}</p>
      </div>
    );
  }

  if (error instanceof Error) {
    // JavaScript 错误
    return (
      <div>
        <h1>出错了</h1>
        <p>{error.message}</p>
        <pre>{error.stack}</pre>
      </div>
    );
  }

  return <h1>未知错误</h1>;
}

function ProductsError() {
  const error = useRouteError();

  return (
    <div className="error-boundary">
      <h2>产品页面加载失败</h2>
      <p>{error.message}</p>
      <Link to="/">返回首页</Link>
    </div>
  );
}
```

### 5.3 Loader 中抛出重定向

```jsx

  }
```

## 六、Scroll 恢复

```jsx
const router = createBrowserRouter([
  {
    path: '/',
    element: <App />,
    // React Router 6.4+ 自动处理 scroll restoration
  },
]);

// 自定义滚动行为
function ScrollRestoration() {
  const { pathname } = useLocation();

  useEffect(() => {
    window.scrollTo(0, 0);
  }, [pathname]);

  return null;
}
```

## 七、Route Handle

`handle` 允许为路由附加任意元数据，供 `useMatches` 等 Hook 读取。

```jsx
const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    handle: {
      breadcrumb: '首页',
      title: '我的应用',
    },
    children: [
      {
        path: 'products',
        element: <Products />,
        handle: {
          breadcrumb: '产品',
          title: '产品列表',
          crumb: (match) => ({
            label: '产品',
            path: match.pathname,
          }),
        },
      },
    ],
  },
]);

// 读取 handle 数据
function PageTitle() {
  const matches = useMatches();
  const currentMatch = matches[matches.length - 1];
  const title = currentMatch?.handle?.title || '默认标题';

  useEffect(() => {
    document.title = title;
  }, [title]);

  return null;
}
```

## 八、完整示例：带数据加载的 CRUD

```jsx
const router = createBrowserRouter([
  {
    path: '/',
    element: <MainLayout />,
    errorElement: <ErrorPage />,
    children: [
      {
        path: 'posts',
        element: <PostList />,
        loader: async () => fetchPosts(),
        action: async ({ request }) => {
          const formData = await request.formData();
          const post = await createPost(Object.fromEntries(formData));
          return redirect(`/posts/${post.id}`);
        },
      },
      {
        path: 'posts/:id',
        element: <PostDetail />,
        loader: async ({ params }) => {
          const post = await fetchPost(params.id);
          if (!post) throw new Response('文章未找到', { status: 404 });
          return { post };
        },
        action: async ({ params, request }) => {
          if (request.method === 'DELETE') {
            await deletePost(params.id);
            return redirect('/posts');
          }
          const formData = await request.formData();
          await updatePost(params.id, Object.fromEntries(formData));
          return { success: true };
        },
      },
    ],
  },
]);

// PostList.jsx
function PostList() {
  const posts = useLoaderData();
  const fetcher = useFetcher();

  return (
    <div>
      <h1>文章列表</h1>
      <ul>
        {posts.map(post => (
          <li key={post.id}>
            <Link to={`${post.id}`}>{post.title}</Link>
          </li>
        ))}
      </ul>

      {/* 新建表单 */}
      <fetcher.Form method="post">
        <input name="title" placeholder="标题" required />
        <textarea name="content" placeholder="内容" required />
        <button type="submit">发布</button>
      </fetcher.Form>
    </div>
  );
}

// PostDetail.jsx
function PostDetail() {
  const { post } = useLoaderData();
  const actionData = useActionData();
  const submit = useSubmit();
  const navigate = useNavigate();

  const handleDelete = () => {
    if (window.confirm('确定删除？')) {
      submit(null, { method: 'delete' });
    }
  };

  return (
    <article>
      <h1>{post.title}</h1>
      <p>{post.content}</p>
      {actionData?.success && <p>更新成功</p>}
      <button onClick={handleDelete}>删除</button>
    </article>
  );
}
```
