# 嵌套路由与布局

嵌套路由是 React Router 的核心特性之一，允许在父路由中渲染子路由，实现复杂的页面布局系统。

---

## 一、嵌套路由基础

### 1.1 基本结构

```jsx
import { Routes, Route, Outlet, Link } from 'react-router-dom';

function App() {
  return (
    <Routes>
      <Route path="/" element={<RootLayout />}>
        <Route index element={<Home />} />
        <Route path="products" element={<ProductsLayout />}>
          <Route index element={<ProductList />} />
          <Route path=":id" element={<ProductDetail />} />
          <Route path=":id/reviews" element={<ProductReviews />} />
        </Route>
        <Route path="about" element={<About />} />
        <Route path="*" element={<NotFound />} />
      </Route>
    </Routes>
  );
}
```

对应 URL 结构：

```
/                    → RootLayout > Home
/products            → RootLayout > ProductsLayout > ProductList
/products/123        → RootLayout > ProductsLayout > ProductDetail
/products/123/reviews → RootLayout > ProductsLayout > ProductReviews
/about               → RootLayout > About
/anything            → RootLayout > NotFound
```

## 二、Outlet 组件

`Outlet` 是嵌套路由的核心，它告诉 React Router "在这里渲染子路由的内容"。

### 2.1 基础用法

```jsx
function RootLayout() {
  return (
    <div className="app">
      <header>
        <nav>
          <Link to="/">首页</Link>
          <Link to="/products">产品</Link>
        </nav>
      </header>
      <main>
        <Outlet />  {/* 子路由内容渲染在这里 */}
      </main>
      <footer>页脚</footer>
    </div>
  );
}
```

### 2.2 多层嵌套

```jsx
function App() {
  return (
    <Routes>
      <Route path="/" element={<RootLayout />}>
        <Route path="dashboard" element={<DashboardLayout />}>
          <Route index element={<DashboardHome />} />
          <Route path="analytics" element={<Analytics />} />
          <Route path="settings" element={<DashboardSettings />} />
        </Route>
      </Route>
    </Routes>
  );
}

// RootLayout
function RootLayout() {
  return (
    <div>
      <TopNav />
      <div className="main-wrapper">
        <Outlet />  {/* 渲染 DashboardLayout */}
      </div>
    </div>
  );
}

// DashboardLayout
function DashboardLayout() {
  return (
    <div className="dashboard">
      <aside>
        <nav>
          <NavLink to="/dashboard">概览</NavLink>
          <NavLink to="/dashboard/analytics">分析</NavLink>
          <NavLink to="/dashboard/settings">设置</NavLink>
        </nav>
      </aside>
      <section className="dashboard-content">
        <Outlet />  {/* 渲染 DashboardHome / Analytics / Settings */}
      </section>
    </div>
  );
}

// /dashboard/analytics 的渲染结果：
// <RootLayout>
//   <TopNav />
//   <DashboardLayout>
//     <aside>...</aside>
//     <Analytics />    ← 最终内容
//   </DashboardLayout>
// </RootLayout>
```

## 三、布局路由

### 3.1 无路径布局

布局路由不匹配任何路径，仅用于提供共享 UI。

```jsx
function App() {
  return (
    <Routes>
      {/* 布局路由：没有 path，仅提供布局 */}
      <Route element={<MainLayout />}>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
        <Route path="/contact" element={<Contact />} />
      </Route>

      {/* 另一个布局 */}
      <Route element={<AuthLayout />}>
        <Route path="/login" element={<Login />} />
        <Route path="/register" element={<Register />} />
      </Route>
    </Routes>
  );
}

// MainLayout：带导航栏的布局
function MainLayout() {
  return (
    <div>
      <Navbar />
      <Outlet />
      <Footer />
    </div>
  );
}

// AuthLayout：居中的简洁布局
function AuthLayout() {
  return (
    <div className="auth-wrapper">
      <div className="auth-container">
        <Outlet />
      </div>
    </div>
  );
}
```

### 3.2 共享导航

```jsx
function MainLayout() {
  return (
    <div className="app">
      <nav className="sidebar">
        <div className="logo">
          <img src="/logo.png" alt="Logo" />
        </div>

        </header>
        <Outlet />
      </main>
    </div>
  );
}
```

## 四、路由级代码分割

### 4.1 配合 React.lazy

```jsx
import { lazy, Suspense } from 'react';
import { Routes, Route } from 'react-router-dom';

// 每个路由对应的组件按需加载
const Home = lazy(() => import('./pages/Home'));
const Products = lazy(() => import('./pages/Products'));
const ProductDetail = lazy(() => import('./pages/ProductDetail'));
const Dashboard = lazy(() => import('./pages/Dashboard'));

function App() {
  return (
    <Suspense fallback={<PageSpinner />}>
      <Routes>
        <Route element={<MainLayout />}>
          <Route path="/" element={<Home />} />
          <Route path="/products" element={<Products />} />
          <Route path="/products/:id" element={<ProductDetail />} />
          <Route path="/dashboard" element={<Dashboard />} />
        </Route>
      </Routes>
    </Suspense>
  );
}
```

### 4.2 布局组件不懒加载

```jsx
// 布局组件应该静态导入（首屏需要）
import MainLayout from './layouts/MainLayout';
import AuthLayout from './layouts/AuthLayout';

// 页面组件可以懒加载
const Home = lazy(() => import('./pages/Home'));
const Login = lazy(() => import('./pages/Login'));

function App() {
  return (
    <Suspense fallback={<PageSpinner />}>
      <Routes>
        <Route element={<MainLayout />}>
          <Route path="/" element={<Home />} />
        </Route>
        <Route element={<AuthLayout />}>
          <Route path="/login" element={<Login />} />
        </Route>
      </Routes>
    </Suspense>
  );
}
```

## 五、面包屑导航

### 5.1 使用 useMatches

```jsx
import { useMatches } from 'react-router-dom';

// 路由配置中添加 handle
const routes = [
  {
    path: '/',
    element: <RootLayout />,
    handle: { breadcrumb: '首页' },
    children: [
      {
        path: 'products',
        element: <ProductsLayout />,
        handle: { breadcrumb: '产品' },
        children: [
          {
            path: ':id',
            element: <ProductDetail />,
            handle: { breadcrumb: '产品详情' },
          },
        ],
      },
    ],
  },
];

// 面包屑组件
function Breadcrumbs() {
  const matches = useMatches();

  const crumbs = matches
    .filter((match) => match.handle?.breadcrumb)
    .map((match) => ({
      path: match.pathname,
      breadcrumb: match.handle.breadcrumb,
    }));

  return (
    <nav aria-label="面包屑">
      <ol>
        {crumbs.map((crumb, index) => (
          <li key={crumb.path}>
            {index < crumbs.length - 1 ? (
              <Link to={crumb.path}>{crumb.breadcrumb}</Link>
            ) : (
              <span>{crumb.breadcrumb}</span>
            )}
          </li>
        ))}
      </ol>
    </nav>
  );
}
```

### 5.2 动态面包屑

```jsx
// 使用 loader 数据动态生成面包屑
const routes = [
  {
    path: 'products/:id',
    element: <ProductDetail />,
    loader: async ({ params }) => {
      const product = await fetchProduct(params.id);
      return { product };
    },
    handle: {
      // handle 可以是函数，从 match 中获取 loader 数据
      breadcrumb: (match) => match.data?.product?.name || '产品详情',
    },
  },
];

function Breadcrumbs() {
  const matches = useMatches();

  const crumbs = matches
    .filter((match) => match.handle?.breadcrumb)
    .map((match) => ({
      path: match.pathname,
      breadcrumb:
        typeof match.handle.breadcrumb === 'function'
          ? match.handle.breadcrumb(match)
          : match.handle.breadcrumb,
    }));

  return (
    <nav>
      {crumbs.map((crumb, i) => (
        <span key={crumb.path}>
          {i > 0 && ' / '}
          <Link to={crumb.path}>{crumb.breadcrumb}</Link>
        </span>
      ))}
    </nav>
  );
}
```

## 六、嵌套 URL 层级

### 6.1 URL 结构设计

```
/                              首页
/products                      产品列表
/products/:id                  产品详情
/products/:id/edit             编辑产品
/products/:id/reviews          产品评论
/products/:id/reviews/:rid     评论详情

/dashboard                     仪表盘
/dashboard/analytics           分析
/dashboard/settings            设置
/dashboard/users               用户管理
/dashboard/users/:id           用户详情
```

### 6.2 对应路由配置

```jsx
<Routes>
  <Route path="/" element={<MainLayout />}>
    <Route index element={<Home />} />

    <Route path="products">
      <Route index element={<ProductList />} />
      <Route path=":id" element={<ProductDetail />} />
      <Route path=":id/edit" element={<EditProduct />} />
      <Route path=":id/reviews" element={<ReviewsLayout />}>
        <Route index element={<ReviewList />} />
        <Route path=":rid" element={<ReviewDetail />} />
      </Route>
    </Route>

    <Route path="dashboard" element={<DashboardLayout />}>
      <Route index element={<DashboardHome />} />
      <Route path="analytics" element={<Analytics />} />
      <Route path="settings" element={<Settings />} />
      <Route path="users">
        <Route index element={<UserList />} />
        <Route path=":id" element={<UserDetail />} />
      </Route>
    </Route>

    <Route path="*" element={<NotFound />} />
  </Route>
</Routes>
```
