## 1. React.memo

### 1.1 基本用法

`React.memo` 是一个高阶组件，用于跳过 props 没有变化的组件的重新渲染：

```jsx
import { memo } from "react";

// 没有 memo：父组件每次渲染，ExpensiveChild 都会重新渲染
function ExpensiveChild({ name, age }) {
  console.log("ExpensiveChild 渲染");
  return (
    <div>
      {name}: {age}
    </div>
  );
}

// 有 memo：只在 props 变化时重新渲染
const MemoizedChild = memo(function ExpensiveChild({ name, age }) {
  console.log("ExpensiveChild 渲染");
  return (
    <div>
      {name}: {age}
    </div>
  );
});

function Parent() {
  const [count, setCount] = useState(0);
  const [name, setName] = useState("Alice");

  return (
    <div>
      {/* count 变化时，MemoizedChild 不会重新渲染（因为 name 没变） */}
      <MemoizedChild name={name} age={25} />
      <button onClick={() => setCount((c) => c + 1)}>Count: {count}</button>
    </div>
  );
}
```

### 1.2 自定义比较函数

```jsx
// 默认：浅比较所有 props
const Component = memo(function MyComponent(props) { ... });

// 自定义比较函数
const Component = memo(
  function MyComponent({ user, options }) {
    return <div>{user.name}</div>;
  },
  (prevProps, nextProps) => {
    // 返回 true 表示"props 相等，不需要重新渲染"
    // 返回 false 表示"props 不同，需要重新渲染"
    return prevProps.user.id === nextProps.user.id;
  }
);
```

### 1.3 React.memo 的浅比较规则

```jsx
const MemoizedChild = memo(function Child({ user, items, onClick }) {
  // ...
});

function Parent() {
  // ✅ 字符串和数字 - 浅比较能正确检测变化
  const [name, setName] = useState("Alice");
  <MemoizedChild name={name} />

  // ❌ 新对象每次都是不同的引用
  const user = { id: 1, name: "Alice" };
  <MemoizedChild user={user} />  // 每次渲染 user 都是新对象 → memo 失效

  // ✅ 使用 useMemo 稳定对象引用
  const user = useMemo(() => ({ id: 1, name: "Alice" }), []);

  // ❌ 内联函数每次创建新引用
  <MemoizedChild onClick={() => handleClick()} />  // memo 失效

  // ✅ 使用 useCallback 稳定函数引用
  const handleClick = useCallback(() => { ... }, []);
  <MemoizedChild onClick={handleClick} />

  return <div />;
}
```

### 1.4 memo 的陷阱

```jsx
// ⚠️ 陷阱一：不必要的 memo
// 如果子组件很轻量，memo 的比较开销可能大于重新渲染的开销
const TrivialChild = memo(function ({ text }) {
  return <span>{text}</span>; // 渲染成本极低，不需要 memo
});

// ⚠️ 陷阱二：memo 传递的 props 不稳定
const MemoizedChild = memo(Child);
function Parent() {
  // 每次渲染都创建新对象，memo 完全无效
  return <MemoizedChild style={{ color: "red" }} onClick={() => {}} />;
}

// ✅ 正确做法
function Parent() {
  const style = useMemo(() => ({ color: "red" }), []);
  const handleClick = useCallback(() => {}, []);
  return <MemoizedChild style={style} onClick={handleClick} />;
}
```

## 2. useMemo

### 2.1 基本用法

`useMemo` 缓存计算结果，只在依赖变化时重新计算：

```jsx
import { useMemo } from "react";

function ExpensiveComponent({ items, filter }) {
  // ❌ 每次渲染都重新计算（即使 items 和 filter 没变）
  const filtered = items.filter((item) => item.type === filter);
  const sorted = [...filtered].sort((a, b) => b.score - a.score);
  const stats = {
    total: sorted.length,
    avgScore: sorted.reduce((sum, i) => sum + i.score, 0) / sorted.length,
  };

  // ✅ 只在 items 或 filter 变化时重新计算
  const stats = useMemo(() => {
    console.log("计算 stats");
    const filtered = items.filter((item) => item.type === filter);
    const sorted = [...filtered].sort((a, b) => b.score - a.score);
    return {
      total: sorted.length,
      avgScore: sorted.reduce((sum, i) => sum + i.score, 0) / sorted.length,
    };
  }, [items, filter]); // 依赖数组

  return <div>平均分: {stats.avgScore}</div>;
}
```

### 2.2 稳定对象引用

```jsx
function Parent() {
  const [count, setCount] = useState(0);

  // ❌ 每次渲染都创建新对象 → Child（即使被 memo）也会重新渲染
  const config = { width: 100, height: 200 };

  // ✅ 引用稳定，只在依赖变化时创建新对象
  const config = useMemo(() => ({ width: 100, height: 200 }), []);

  return (
    <div>
      <MemoizedChild config={config} />
      <button onClick={() => setCount((c) => c + 1)}>{count}</button>
    </div>
  );
}
```

### 2.3 useMemo 的使用场景

```jsx
// ✅ 场景一：昂贵的计算
const result = useMemo(() => {
  return heavyComputation(largeData);
}, [largeData]);

// ✅ 场景二：保持对象引用稳定（传给 memo 子组件）
const style = useMemo(() => ({ color: theme.primary }), [theme.primary]);

// ✅ 场景三：避免重复创建复杂对象
const chartData = useMemo(() => {
  return rawData.map((d) => ({
    x: d.timestamp,
    y: d.value,
    label: formatLabel(d),
  }));
}, [rawData]);

// ❌ 不需要 useMemo 的场景
function Component() {
  // 简单计算，不需要 useMemo
  const doubled = count * 2;

  // 简单对象，不需要 useMemo（除非传给 memo 子组件）
  const className = `btn btn-${variant}`;

  return <button className={className}>{doubled}</button>;
}
```

## 3. useCallback

### 3.1 基本用法

`useCallback` 返回一个稳定的函数引用：

```jsx
import { useCallback } from "react";

function Parent() {
  const [count, setCount] = useState(0);

  // ❌ 每次渲染都创建新函数 → MemoizedChild 重新渲染
  const handleClick = () => {
    console.log("clicked");
  };

  // ✅ 引用稳定（除非依赖变化）
  const handleClick = useCallback(() => {
    console.log("clicked");
  }, []); // 空依赖 = 永远稳定

  return (
    <div>
      <MemoizedChild onClick={handleClick} />
      <button onClick={() => setCount((c) => c + 1)}>{count}</button>
    </div>
  );
}
```

### 3.2 带依赖的 useCallback

```jsx
function Parent() {
  const [userId, setUserId] = useState(1);

  // 当 userId 变化时，handleDelete 也会变化
  const handleDelete = useCallback(() => {
    deleteUser(userId);
  }, [userId]);

  // 如果 userId 没变，handleDelete 引用不变
  // MemoizedChild 不会重新渲染
  return <MemoizedChild onDelete={handleDelete} />;
}
```

### 3.3 useCallback vs useMemo

```jsx
// useCallback(fn, deps) 等价于：
useMemo(() => fn, deps);

// 区别在于语义：
const memoizedFn = useCallback(() => doSomething(a, b), [a, b]);
// 等价于：
const memoizedFn = useMemo(() => () => doSomething(a, b), [a, b]);

// useCallback：缓存函数本身
// useMemo：缓存函数的执行结果
```

### 3.4 useCallback 的使用场景

```jsx
// ✅ 场景一：传递给 memo 子组件
const handleClick = useCallback(() => { ... }, []);
<MemoizedChild onClick={handleClick} />

// ✅ 场景二：作为其他 Hook 的依赖
useEffect(() => {
  fetchData(userId);
}, [userId]); // 如果 fetchData 用了 useCallback，依赖稳定

// ✅ 场景三：传递给自定义 Hook
useDebouncedCallback(handleSearch, 300);

// ❌ 不需要的场景
function Component() {
  // 函数没有传递给其他组件
  const handleClick = useCallback(() => {
    setCount((c) => c + 1);
  }, []);
  // 直接写内联函数即可，useCallback 的开销不值得
  return <button onClick={() => setCount((c) => c + 1)}>+1</button>;
}
```

## 4. 组件拆分与懒加载

### 4.1 按功能拆分组件

```jsx
// ❌ 巨型组件 - 整个组件树都会重新渲染
function Dashboard() {
  const [users, setUsers] = useState([]);
  const [posts, setPosts] = useState([]);
  const [comments, setComments] = useState([]);
  const [filter, setFilter] = useState("");

  // 大量数据处理逻辑...

  return (
    <div>
      <UserList users={users} filter={filter} />
      <PostList posts={posts} />
      <CommentList comments={comments} />
      <FilterInput onChange={setFilter} />
    </div>
  );
}

// ✅ 拆分组件 - 只有相关部分重新渲染
function Dashboard() {
  return (
    <div>
      <UserSection />
      <PostSection />
      <CommentSection />
    </div>
  );
}

// 每个子组件管理自己的状态
function UserSection() {
  const [users, setUsers] = useState([]);
  const [filter, setFilter] = useState("");
  // filter 变化时，只有 UserSection 重新渲染

    <div>
      <FilterInput onChange={setFilter} />
      <UserList users={users} filter={filter} />
    </div>
  );
}
```

## 5. 虚拟化 (Windowing)

### 5.1 什么是虚拟化

虚拟化只渲染可见区域内的列表项，极大减少 DOM 节点数量：

```
10,000 条数据的列表：

无虚拟化：                     有虚拟化：
渲染 10,000 个 DOM 节点         只渲染可见的 ~20 个 DOM 节点
性能差、内存占用高              性能好、内存占用低

|=========================|    |=========================|
| Item 1                  |    | Item 1 (DOM)            |
| Item 2                  |    | Item 2 (DOM)            |
| ...                     |    | ...                     |
| Item 10,000             |    | Item 20 (DOM)           |
|=========================|    | Item 21~10,000 (占位)   |
                                |=========================|
```

### 5.2 使用 react-window

```jsx
import { FixedSizeList, VariableSizeList } from "react-window";

// 固定高度列表
function UserList({ users }) {
  return (
    <FixedSizeList
      height={600}        // 容器高度
      itemCount={users.length} // 总条目数
      itemSize={50}       // 每项高度
      width="100%"        // 容器宽度
    >
      {({ index, style }) => (
        // style 包含了 position 和 height，必须应用
        <div style={style}>
          {users[index].name}
        </div>
      )}
    </FixedSizeList>
  );
}

// 可变高度列表
function DynamicList({ items }) {
  const getItemSize = (index) => {
    return items[index].isExpanded ? 200 : 50;
  };

  return (
    <VariableSizeList
      height={600}
      itemCount={items.length}
      itemSize={getItemSize}
      width="100%"
    >
      {({ index, style }) => (
        <div style={style}>
          <ItemContent item={items[index]} />
        </div>
      )}
    </VariableSizeList>
  );
}
```

### 5.3 使用 @tanstack/react-virtual

```jsx
import { useVirtualizer } from "@tanstack/react-virtual";

function VirtualList({ items }) {
  const parentRef = useRef(null);

  const virtualizer = useVirtualizer({

            }}

            {items[virtualItem.index].name}
          </div>

      </div>
    </div>
  );
}
```

## 6. 避免 Props 中的内联对象/函数

### 6.1 问题

```jsx
function Parent() {
  // 每次渲染都创建新对象/函数
  return (
    <div>
      <MemoizedChild
        style={{ color: "red" }}           // 新对象
        config={{ width: 100 }}            // 新对象
        onClick={() => handleClick()}      // 新函数
        items={[1, 2, 3]}                  // 新数组
      />
    </div>
  );
}
// 结果：MemoizedChild 每次都重新渲染，memo 完全失效
```

### 6.2 解决方案

```jsx
function Parent() {
  // ✅ 使用 useMemo / useCallback 稳定引用
  const style = useMemo(() => ({ color: "red" }), []);
  const config = useMemo(() => ({ width: 100 }), []);
  const items = useMemo(() => [1, 2, 3], []);
  const handleClick = useCallback(() => { ... }, []);

  return (
    <MemoizedChild
      style={style}
      config={config}
      items={items}
      onClick={handleClick}
    />
  );
}

// ✅ 或者将常量移到组件外部
const STYLE = { color: "red" };
const ITEMS = [1, 2, 3];

function Parent() {
  return <MemoizedChild style={STYLE} items={ITEMS} />;
}
```

## 7. State Colocation (状态就近原则)

### 7.1 原则

将状态放到离使用它的组件最近的位置，而不是都放在顶层：

```jsx
// ❌ 状态放在太高的位置
function App() {
  const [searchQuery, setSearchQuery] = useState("");
  const [filter, setFilter] = useState("all");
  const [sortBy, setSortBy] = useState("date");
  const [selectedIds, setSelectedIds] = useState([]);
  // ... 更多状态

  // 每个状态变化都会导致整个 App 重新渲染
  return (
    <div>
      <SearchBar query={searchQuery} onChange={setSearchQuery} />
      <FilterBar filter={filter} onChange={setFilter} />
      <SortBar sortBy={sortBy} onChange={setSortBy} />
      <ItemList />
    </div>
  );
}

// ✅ 状态放在使用它的组件中
function App() {
  return (
    <div>
      <SearchBar />
      <FilterBar />
      <SortBar />
      <ItemList />
    </div>
  );
}

function SearchBar() {
  const [query, setQuery] = useState("");
  // query 变化时只有 SearchBar 重新渲染
  return <input value={query} onChange={(e) => setQuery(e.target.value)} />;
}
```

### 7.2 找到状态放置的最佳位置

```
判断状态位置的步骤：
1. 找到使用该状态的所有组件
2. 找到它们的共同父组件
3. 将状态放在共同父组件中，或使用状态管理库提升

示例：
  ComponentA uses count
  ComponentB uses count
  └── 共同父组件 Parent → 将 count 放在 Parent 中
      ├── ComponentA
      └── ComponentB

如果共同父组件层级太高，考虑：
  - 使用 Context 局部共享
  - 组件组合 (Component Composition)
  - 状态管理库 (Zustand, Jotai 等)
```

## 8. 综合优化示例

### 8.1 优化前

```jsx
function ProductList({ products, onAddToCart }) {
  const [sortBy, setSortBy] = useState("name");
  const [filter, setFilter] = useState("");

  // 每次渲染都排序和过滤
  const sorted = [...products].sort((a, b) =>
    a[sortBy].localeCompare(b[sortBy])
  );
  const filtered = sorted.filter((p) =>
    p.name.toLowerCase().includes(filter.toLowerCase())
  );

  return (
    <div>
      <input
        value={filter}
        onChange={(e) => setFilter(e.target.value)}
        style={{ padding: "8px" }}  // 内联对象
      />
      <select onChange={(e) => setSortBy(e.target.value)}>
        <option value="name">按名称</option>
        <option value="price">按价格</option>
      </select>
      <ul>
        {filtered.map((product) => (
          <li key={product.id} style={{ color: "blue" }}>
            {product.name} - ¥{product.price}
            <button onClick={() => onAddToCart(product)}>加入购物车</button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

### 8.2 优化后

```jsx
import { memo, useMemo, useCallback } from "react";

// 1. 子组件使用 memo
const ProductItem = memo(function ProductItem({ product, onAdd }) {
  const handleClick = useCallback(() => {
    onAdd(product);
  }, [onAdd, product]);

  return (
    <li style={ITEM_STYLE}>
      {product.name} - ¥{product.price}
      <button onClick={handleClick}>加入购物车</button>
    </li>
  );
});

// 2. 常量移到组件外部
const ITEM_STYLE = { color: "blue" };
const INPUT_STYLE = { padding: "8px" };

// 3. 主组件
function ProductList({ products, onAddToCart }) {
  const [sortBy, setSortBy] = useState("name");
  const [filter, setFilter] = useState("");

  // 4. 使用 useMemo 缓存计算结果
  const filteredProducts = useMemo(() => {
    const sorted = [...products].sort((a, b) =>
      a[sortBy].localeCompare(b[sortBy])
    );
    return sorted.filter((p) =>
      p.name.toLowerCase().includes(filter.toLowerCase())
    );
  }, [products, sortBy, filter]);

  // 5. 使用 useCallback 稳定回调
  const handleAddToCart = useCallback((product) => {
    onAddToCart(product);
  }, [onAddToCart]);

  const handleFilterChange = useCallback((e) => {
    setFilter(e.target.value);
  }, []);

  const handleSortChange = useCallback((e) => {
    setSortBy(e.target.value);
  }, []);

  return (
    <div>
      <input
        value={filter}
        onChange={handleFilterChange}
        style={INPUT_STYLE}
      />
      <select onChange={handleSortChange}>
        <option value="name">按名称</option>
        <option value="price">按价格</option>
      </select>
      <ul>
        {filteredProducts.map((product) => (
          <ProductItem
            key={product.id}
            product={product}
            onAdd={handleAddToCart}
          />
        ))}
      </ul>
    </div>
  );
}
```

## 小结

- React.memo 用于避免 props 没变的子组件重新渲染，但需要配合稳定的 props 引用
- useMemo 缓存昂贵的计算结果和对象引用
- useCallback 缓存函数引用，确保传递给 memo 子组件的回调稳定
- 虚拟化技术只渲染可见区域的列表项，适合大数据列表
- 状态就近原则：将状态放到离使用它的组件最近的位置
- 优化要基于实际性能数据，避免过度优化
