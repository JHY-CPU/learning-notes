## 1. "Cannot update during render"

### 1.1 错误描述

```
Warning: Cannot update a component (`ParentComponent`) while rendering
a different component (`ChildComponent`). To locate the bad setState
call inside `ChildComponent`, follow the stack trace...
```

### 1.2 原因

在渲染阶段（组件函数执行期间）调用另一个组件的 setState：

```jsx
// ❌ 错误写法
function Parent({ onReady }) {
  const [ready, setReady] = useState(false);

  // 在子组件渲染期间调用父组件的 setState
  useEffect(() => {
    onReady(() => setReady(true));
  }, [onReady]);

  return <div>{ready ? "Ready" : "Not Ready"}</div>;
}

function App() {
  // ❌ 每次渲染都创建新函数 → 触发 Parent 的 effect → setState
  return <Parent onReady={(cb) => cb()} />;
}

// ❌ 另一个常见场景
function Child({ onChange }) {
  const [value, setValue] = useState("");

  // 在渲染期间调用 onChange
  onChange(value); // ← 这里触发了父组件的 setState

  return <input value={value} onChange={(e) => setValue(e.target.value)} />;
}
```

### 1.3 解决方案

```jsx
// ✅ 方案一：将副作用放到 useEffect 中
function Child({ onChange, value }) {
  useEffect(() => {
    onChange(value);
  }, [onChange, value]);

  return <div>{value}</div>;
}

// ✅ 方案二：稳定回调引用
function App() {
  const handleReady = useCallback((cb) => cb(), []);
  return <Parent onReady={handleReady} />;
}

// ✅ 方案三：使用 ref 代替回调
function Parent() {
  const [ready, setReady] = useState(false);
  const readyCallback = useRef(null);

  useEffect(() => {
    readyCallback.current = () => setReady(true);
  }, []);

  return <Child triggerReady={() => readyCallback.current?.()} />;
}
```

## 2. "Too many re-renders"

### 2.1 错误描述

```
Error: Too many re-renders. React limits the number of renders to
prevent an infinite loop.
```

### 2.2 常见原因

```jsx
// ❌ 原因一：在渲染期间直接调用 setState
function Counter() {
  const [count, setCount] = useState(0);
  setCount(count + 1); // 每次渲染都调用 → 无限循环
  return <div>{count}</div>;
}

// ❌ 原因二：事件处理器直接调用（没有用函数包裹）
function Counter() {
  const [count, setCount] = useState(0);
  return (
    // onClick={setCount(count + 1)} 直接调用了函数
    <button onClick={setCount(count + 1)}>+1</button>
  );
}

// ❌ 原因三：useEffect 依赖导致无限循环
function Component() {
  const [data, setData] = useState([]);

  useEffect(() => {
    fetchData().then(setData); // setData 创建新数组
  }, [data]); // ← data 变化 → 重新执行 → 无限循环

  return <div>{data.length}</div>;
}

// ❌ 原因四：对象/数组作为依赖
function Component() {
  useEffect(() => {
    // ...
  }, [{ id: 1 }]); // ← 每次都是新对象 → 每次都执行
}
```

### 2.3 解决方案

```jsx
// ✅ 解决方案一：正确的事件处理
function Counter() {
  const [count, setCount] = useState(0);
  return (
    <button onClick={() => setCount(count + 1)}>+1</button>
  );
}

// ✅ 解决方案二：正确的 useEffect 依赖
function Component() {
  const [data, setData] = useState([]);

  useEffect(() => {
    fetchData().then(setData);
  }, []); // 空依赖，只执行一次

  return <div>{data.length}</div>;
}

// ✅ 解决方案三：使用 useRef 避免不必要的依赖
function Component() {
  const [data, setData] = useState([]);
  const isInitialMount = useRef(true);

  useEffect(() => {
    if (isInitialMount.current) {
      isInitialMount.current = false;
      fetchData().then(setData);
    }
  }, []);

  return <div>{data.length}</div>;
}
```

## 3. "Objects are not valid as React child"

### 3.1 错误描述

```
Error: Objects are not valid as a React child (found: object with keys
{id, name, email}). If you meant to render a collection of children,
use an array instead.
```

### 3.2 常见原因

```jsx
// ❌ 原因一：直接渲染对象
function UserProfile({ user }) {
  return <div>{user}</div>; // user 是对象，不能直接渲染
}

// ❌ 原因二：直接渲染 Promise
async function fetchUser() { return { id: 1, name: "Alice" }; }
function Profile() {
  const user = fetchUser(); // 返回 Promise（对象）
  return <div>{user}</div>;
}

// ❌ 原因三：日期对象
function EventCard({ event }) {
  return <div>{event.date}</div>; // event.date 是 Date 对象
}

// ❌ 原因四：state 是对象但直接渲染
function App() {
  const [config, setConfig] = useState({ theme: "dark", lang: "zh" });
  return <div>{config}</div>; // ❌ 不能渲染对象
}
```

### 3.3 解决方案

```jsx
// ✅ 解决方案一：渲染对象的具体属性
function UserProfile({ user }) {
  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}

// ✅ 解决方案二：正确处理异步数据
function Profile() {
  const [user, setUser] = useState(null);

  useEffect(() => {
    fetchUser().then(setUser);
  }, []);

  if (!user) return <div>Loading...</div>;
  return <div>{user.name}</div>;
}

// ✅ 解决方案三：格式化日期
function EventCard({ event }) {
  return <div>{event.date.toLocaleDateString()}</div>;
}

// ✅ 解决方案四：序列化对象用于调试
function DebugInfo({ data }) {
  return <pre>{JSON.stringify(data, null, 2)}</pre>;
}
```

## 4. "Each child needs unique key"

### 4.1 错误描述

```
Warning: Each child in a list should have a unique "key" prop.
```

### 4.2 常见原因

```jsx
// ❌ 缺少 key
function List({ items }) {
  return (
    <ul>
      {items.map((item) => (
        <li>{item.name}</li>
      ))}
    </ul>
  );
}

// ❌ 使用索引作为 key（数据可能被排序/过滤）
function List({ items }) {
  return (
    <ul>
      {items.map((item, index) => (
        <li key={index}>{item.name}</li>
      ))}
    </ul>
  );
}

// ❌ key 放在了错误的位置
function List({ items }) {
  return (
    <ul>
      {items.map((item) => (
        <Wrapper key={item.id}>  {/* key 应该在最外层元素 */}
          <li>{item.name}</li>
        </Wrapper>
      ))}
    </ul>
  );
}

// ❌ 在 Fragment 上使用 key 时用了短语法
function List({ items }) {
  return (
    <ul>
      {items.map((item) => (
        <>  {/* 短语法 <> 不支持 key */}
          <li>{item.name}</li>
          <li>{item.description}</li>
        </>
      ))}
    </ul>
  );
}
```

### 4.3 解决方案

```jsx
// ✅ 使用唯一且稳定的 id
function List({ items }) {
  return (
    <ul>
      {items.map((item) => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
}

// ✅ 需要 Fragment 时使用完整语法
function List({ items }) {
  return (
    <ul>
      {items.map((item) => (
        <React.Fragment key={item.id}>
          <li>{item.name}</li>
          <li>{item.description}</li>
        </React.Fragment>
      ))}
    </ul>
  );
}

// ✅ 嵌套列表使用复合 key
function NestedList({ categories }) {
  return (
    <div>
      {categories.map((category) => (
        <div key={category.id}>
          <h2>{category.name}</h2>
          <ul>
            {category.items.map((item) => (
              <li key={`${category.id}-${item.id}`}>{item.name}</li>
            ))}
          </ul>
        </div>
      ))}
    </div>
  );
}
```

## 5. "Can't perform state update on unmounted component"

### 5.1 错误描述

```
Warning: Can't perform a React state update on an unmounted component.
This is a no-op, but it indicates a memory leak in your application.
```

> 注意：React 18 已移除此警告，但问题本身仍然存在，可能导致 bug。

### 5.2 常见原因

```jsx
// ❌ 异步操作完成后组件已卸载
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    fetch(`/api/users/${userId}`)
      .then((res) => res.json())
      .then((data) => setUser(data)); // 组件可能已经卸载了
  }, [userId]);

  return user ? <div>{user.name}</div> : <div>Loading...</div>;
}

// ❌ 定时器未清理
function Timer() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    setInterval(() => {
      setCount((c) => c + 1); // 组件卸载后仍在执行
    }, 1000);
  }, []);

  return <div>{count}</div>;
}

// ❌ 事件监听未移除
function WindowSize() {
  const [size, setSize] = useState({ width: 0, height: 0 });

  useEffect(() => {
    const handleResize = () => {
      setSize({ width: window.innerWidth, height: window.innerHeight });
    };
    window.addEventListener("resize", handleResize);
    // 缺少 cleanup
  }, []);

  return <div>{size.width} x {size.height}</div>;
}
```

### 5.3 解决方案

```jsx
// ✅ 方案一：AbortController
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    const controller = new AbortController();

    fetch(`/api/users/${userId}`, { signal: controller.signal })
      .then((res) => res.json())
      .then((data) => setUser(data))
      .catch((err) => {
        if (err.name !== "AbortError") {
          console.error(err);
        }
      });

    return () => controller.abort();
  }, [userId]);

  return user ? <div>{user.name}</div> : <div>Loading...</div>;
}

// ✅ 方案二：清除定时器
function Timer() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    const id = setInterval(() => {
      setCount((c) => c + 1);
    }, 1000);

    return () => clearInterval(id);
  }, []);

  return <div>{count}</div>;
}

// ✅ 方案三：移除事件监听
function WindowSize() {
  const [size, setSize] = useState({ width: 0, height: 0 });

  useEffect(() => {
    const handleResize = () => {
      setSize({ width: window.innerWidth, height: window.innerHeight });
    };
    window.addEventListener("resize", handleResize);

    return () => window.removeEventListener("resize", handleResize);
  }, []);

  return <div>{size.width} x {size.height}</div>;
}

// ✅ 方案四：使用 mounted flag
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    let mounted = true;

    fetch(`/api/users/${userId}`)
      .then((res) => res.json())
      .then((data) => {
        if (mounted) {
          setUser(data);
        }
      });

    return () => { mounted = false; };
  }, [userId]);

  return user ? <div>{user.name}</div> : <div>Loading...</div>;
}
```

## 6. 内存泄漏

### 6.1 常见泄漏来源

```jsx
// ❌ 泄漏一：未清理的订阅
function LiveData() {
  const [data, setData] = useState(null);

  useEffect(() => {
    const subscription = dataService.subscribe((newData) => {
      setData(newData); // 组件卸载后仍在更新
    });
    // 缺少 cleanup
  }, []);

  return <div>{data}</div>;
}

// ❌ 泄漏二：未清理的 Observer
function InfiniteScroll() {
  const ref = useRef();

  useEffect(() => {
    const observer = new IntersectionObserver(callback);
    observer.observe(ref.current);
    // 缺少 disconnect
  }, []);

  return <div ref={ref}>...</div>;
}

// ❌ 泄漏三：闭包持有大对象引用
function BigDataComponent() {
  const [data] = useState(() => generateLargeData());

  useEffect(() => {
    const handleEvent = () => {
      // 闭包持有 data 的引用，即使 data 不再使用
      console.log(data.length);
    };
    window.addEventListener("click", handleEvent);
    // 缺少 cleanup → data 永远不会被 GC
  }, [data]);

  return <div>{data.length}</div>;
}
```

### 6.2 解决方案

```jsx
// ✅ 订阅清理
function LiveData() {
  const [data, setData] = useState(null);

  useEffect(() => {
    const subscription = dataService.subscribe(setData);
    return () => subscription.unsubscribe();
  }, []);

  return <div>{data}</div>;
}

// ✅ Observer 清理
function InfiniteScroll() {
  const ref = useRef();

  useEffect(() => {
    const observer = new IntersectionObserver(callback);
    if (ref.current) {
      observer.observe(ref.current);
    }
    return () => observer.disconnect();
  }, []);

  return <div ref={ref}>...</div>;
}

// ✅ 使用 WeakRef 减少闭包引用
function BigDataComponent() {
  const [data] = useState(() => generateLargeData());
  const dataRef = useRef(data);

  useEffect(() => {
    dataRef.current = data; // 更新 ref
    const handleEvent = () => {
      console.log(dataRef.current.length);
    };
    window.addEventListener("click", handleEvent);
    return () => window.removeEventListener("click", handleEvent);
  }, [data]);

  return <div>{data.length}</div>;
}
```

## 7. 无限循环

### 7.1 常见场景

```jsx
// ❌ 场景一：useEffect 中 setState 导致依赖变化
function InfiniteLoop() {
  const [items, setItems] = useState([]);

  useEffect(() => {
    // items 变化 → 重新执行 → 又创建新数组 → 无限循环
    setItems((prev) => [...prev, { id: Date.now() }]);
  }, [items]);

  return <div>{items.length}</div>;
}

// ❌ 场景二：对象依赖
function InfiniteLoop2() {
  useEffect(() => {
    // ...
  }, [{ id: 1 }]); // 每次都是新对象

  return <div />;
}

// ❌ 场景三：inline 函数依赖
function InfiniteLoop3({ onChange }) {
  const [value, setValue] = useState("");

  useEffect(() => {
    onChange(value);
  }, [onChange, value]); // onChange 每次都是新函数

  return <input value={value} onChange={(e) => setValue(e.target.value)} />;
}
```

### 7.2 解决方案

```jsx
// ✅ 方案一：正确的依赖
function Fixed() {
  const [items, setItems] = useState([]);

  useEffect(() => {
    // 只在挂载时执行一次
    setItems([{ id: 1 }, { id: 2 }]);
  }, []);

  return <div>{items.length}</div>;
}

// ✅ 方案二：稳定依赖
function Fixed2({ onChange }) {
  const [value, setValue] = useState("");
  const onChangeRef = useRef(onChange);

  useEffect(() => {
    onChangeRef.current = onChange;
  }, [onChange]);

  useEffect(() => {
    onChangeRef.current(value);
  }, [value]);

  return <input value={value} onChange={(e) => setValue(e.target.value)} />;
}

// ✅ 方案三：父组件使用 useCallback
function Parent() {
  const handleChange = useCallback((value) => {
    console.log(value);
  }, []);

  return <Child onChange={handleChange} />;
}
```

## 8. Hydration Mismatches

### 8.1 什么是 Hydration

Hydration（水合）是 SSR 中的过程：服务端生成 HTML → 客户端 React "接管"这些静态 HTML，使其变为可交互的。

```
服务端：生成 HTML 字符串
  → <div>Count: 0</div>

客户端：React 尝试匹配服务端 HTML
  → 如果客户端渲染结果不同 → Hydration Mismatch
```

### 8.2 常见的 Mismatch

```jsx
// ❌ 错误一：使用随机值
function RandomGreeting() {
  const messages = ["Hello", "Hi", "Hey"];
  const greeting = messages[Math.floor(Math.random() * messages.length)];
  // 服务端和客户端生成不同的随机值
  return <div>{greeting}</div>;
}

// ❌ 错误二：使用时间
function CurrentTime() {
  return <div>{new Date().toLocaleString()}</div>;
  // 服务端和客户端时间不同
}

// ❌ 错误三：使用浏览器 API
function WindowWidth() {
  const [width, setWidth] = useState(window.innerWidth);
  // 服务端没有 window 对象
  return <div>{width}</div>;
}

// ❌ 错误四：条件渲染依赖环境
function ConditionalRender() {
  const isMobile = typeof window !== "undefined" && window.innerWidth < 768;
  // 服务端和客户端判断可能不同
  return isMobile ? <MobileView /> : <DesktopView />;
}
```

### 8.3 解决方案

```jsx
// ✅ 方案一：客户端渲染动态内容
function CurrentTime() {
  const [time, setTime] = useState("");

  useEffect(() => {
    // 只在客户端更新
    setTime(new Date().toLocaleString());
    const timer = setInterval(() => {
      setTime(new Date().toLocaleString());
    }, 1000);
    return () => clearInterval(timer);
  }, []);

  return <div>{time || "加载中..."}</div>;
}

// ✅ 方案二：使用 suppressHydrationWarning
function CurrentDate() {
  return (
    <div suppressHydrationWarning>
      {new Date().toLocaleDateString()}
    </div>
  );
}
// suppressHydrationWarning 只适用于一层文本内容

// ✅ 方案三：延迟渲染到客户端
function WindowWidth() {
  const [width, setWidth] = useState(0); // 服务端和客户端初始值一致
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    setWidth(window.innerWidth);
    setMounted(true);
  }, []);

  if (!mounted) return <div>加载中...</div>;
  return <div>{width}px</div>;
}

// ✅ 方案四：动态导入避免 SSR
import dynamic from "next/dynamic";

const ClientOnlyComponent = dynamic(
  () => import("./ClientOnlyComponent"),
  { ssr: false } // 不在服务端渲染
);

// ✅ 方案五：使用 useId 生成稳定 ID
function FormField() {
  const id = useId(); // 服务端和客户端生成相同的 ID
  return (
    <div>
      <label htmlFor={id}>Name</label>
      <input id={id} />
    </div>
  );
}
```

### 8.4 Next.js 中的 Hydration

```jsx
// app/page.tsx

// ✅ 使用 useSyncExternalStore 获取客户端状态
import { useSyncExternalStore } from "react";

function useMediaQuery(query) {
  return useSyncExternalStore(
    (callback) => {
      const mediaQuery = window.matchMedia(query);
      mediaQuery.addEventListener("change", callback);
      return () => mediaQuery.removeEventListener("change", callback);
    },
    () => window.matchMedia(query).matches,
    () => false // 服务端返回默认值
  );
}

function Page() {
  const isMobile = useMediaQuery("(max-width: 768px)");
  return isMobile ? <MobileView /> : <DesktopView />;
}
```

## 小结

- "Cannot update during render" — 将 setState 调用放到 useEffect 或事件处理器中
- "Too many re-renders" — 检查是否有渲染期间的 setState、正确的依赖数组
- "Objects are not valid as React child" — 渲染对象的具体属性，而非对象本身
- "Each child needs unique key" — 使用稳定且唯一的 id，避免索引
- 状态更新在卸载组件上 — 使用 AbortController、清理定时器和事件监听
- 内存泄漏 — 清理所有订阅、Observer、事件监听
- 无限循环 — 确保 useEffect 依赖稳定，避免创建新对象/函数
- Hydration Mismatches — 动态内容应在客户端渲染，保持服务端和客户端初始值一致
