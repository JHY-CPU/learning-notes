## 1. 分层错误边界策略

### 1.1 架构设计

```
推荐的错误边界层级：

┌─────────────────────────────────────────┐
│          Global ErrorBoundary           │ ← 兜底：显示"应用出错"页面
│  ┌───────────────────────────────────┐  │
│  │      Route ErrorBoundary         │  │ ← 路由级：页面级 fallback
│  │  ┌─────────────┐ ┌────────────┐  │  │
│  │  │Widget Error │ │Widget Error│  │  │ ← 组件级：模块级 fallback
│  │  │  Boundary   │ │  Boundary  │  │  │
│  │  │ ┌─────────┐ │ │ ┌────────┐ │  │  │
│  │  │ │Component│ │ │ │3rd Party│ │  │  │ ← 第三方隔离
│  │  │ └─────────┘ │ │ └────────┘ │  │  │
│  │  └─────────────┘ └────────────┘  │  │
│  └───────────────────────────────────┘  │
└─────────────────────────────────────────┘
```

### 1.2 各层职责

```jsx
// 第一层：全局兜底
function GlobalErrorBoundary({ children }) {
  return (
    <ErrorBoundary
      fallback={
        <div className="global-error">
          <h1>应用遇到了问题</h1>
          <p>我们的团队已经收到通知，正在修复中。</p>

          </button>

        </div>

        reportToSentry(error, "global");
      }}
    >
      {children}
    </ErrorBoundary>
  );
}

// 第二层：路由级
function RouteErrorBoundary({ children, routeName }) {
  return (
    <ErrorBoundary
      fallback={
        <div className="route-error">
          <h2>页面加载失败</h2>
          <p>{routeName} 页面出现了问题</p>
          <button onClick={() => window.location.reload()}>重试</button>
          <a href="/">返回首页</a>
        </div>
      }
      onError={(error) => {
        reportToSentry(error, `route:${routeName}`);
      }}
    >
      {children}
    </ErrorBoundary>
  );
}

// 第三层：组件/Widget 级
function WidgetErrorBoundary({ children, widgetName }) {
  return (
    <ErrorBoundary
      fallback={
        <div className="widget-error">
          <span>{widgetName} 加载失败</span>
          <button onClick={() => window.location.reload()}>重试</button>
        </div>
      }
    >
      {children}
    </ErrorBoundary>
  );
}
```

### 2.2 自建错误收集

```js
// 错误收集服务
class ErrorReporter {
  constructor(options = {}) {
    this.endpoint = options.endpoint || "/api/errors";
    this.appVersion = options.appVersion || "1.0.0";
    this.userId = options.userId;
    this.queue = [];
    this.flushInterval = options.flushInterval || 5000;

    // 定期批量发送
    setInterval(() => this.flush(), this.flushInterval);

    // 页面关闭前发送
    window.addEventListener("beforeunload", () => this.flush());
  }

  captureError(error, context = {}) {
    const errorData = {
      message: error.message,
      stack: error.stack,
      timestamp: Date.now(),
      url: window.location.href,
      userAgent: navigator.userAgent,
      appVersion: this.appVersion,
      userId: this.userId,
      ...context,
    };

    this.queue.push(errorData);

    // 队列过大时立即发送
    if (this.queue.length >= 10) {
      this.flush();
    }
  }

  async flush() {
    if (this.queue.length === 0) return;

    const errors = [...this.queue];
    this.queue = [];

    try {
      // 使用 sendBeacon 确保页面关闭时也能发送
      if (navigator.sendBeacon) {
        navigator.sendBeacon(
          this.endpoint,
          JSON.stringify({ errors })
        );
      } else {
        await fetch(this.endpoint, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ errors }),
          keepalive: true,
        });
      }
    } catch (err) {
      // 发送失败，重新加入队列
      this.queue.unshift(...errors);
    }
  }
}

// 初始化
const errorReporter = new ErrorReporter({
  endpoint: "/api/errors",
  appVersion: import.meta.env.VITE_APP_VERSION,
});

// 在 ErrorBoundary 中使用
class ErrorBoundary extends React.Component {
  componentDidCatch(error, errorInfo) {
    errorReporter.captureError(error, {
      componentStack: errorInfo.componentStack,
      boundaryName: this.props.name || "unknown",
    });
  }
  // ...
}
```

function useErrorTracking(componentName) {
  useEffect(() => {
    const startTime = performance.now();

    return () => {
      const renderTime = performance.now() - startTime;
      if (renderTime > 100) {

## 4. 重试机制

### 4.1 指数退避重试

```js
async function retryWithBackoff(fn, options = {}) {
  const {
    maxRetries = 3,
    baseDelay = 1000,
    maxDelay = 10000,
    shouldRetry = () => true,
  } = options;

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      if (attempt === maxRetries || !shouldRetry(error)) {
        throw error;
      }

      // 指数退避 + 随机抖动
      const delay = Math.min(
        baseDelay * Math.pow(2, attempt) + Math.random() * 1000,
        maxDelay
      );

      console.log(`重试 (${attempt + 1}/${maxRetries})，等待 ${Math.round(delay)}ms`);
      await new Promise((r) => setTimeout(r, delay));
    }
  }
}

// 使用
const data = await retryWithBackoff(
  () => fetch("/api/data").then((r) => r.json()),
  {
    maxRetries: 3,
    shouldRetry: (error) => {
      // 只重试网络错误和 5xx
      return (
        error.name === "TypeError" ||
        (error.status && error.status >= 500)
      );
    },
  }
);
```

### 4.2 React 中的重试组件

```jsx
function RetryableComponent({ fetchFn, children }) {
  const [state, setState] = useState({
    data: null,
    loading: true,
    error: null,
    retryCount: 0,
  });

  const maxRetries = 3;

  useEffect(() => {
    let cancelled = false;

    async function execute(attempt) {
      setState((prev) => ({ ...prev, loading: true, error: null }));

      try {
        const data = await fetchFn();
        if (!cancelled) {
          setState({ data, loading: false, error: null, retryCount: 0 });
        }
      } catch (error) {
        if (cancelled) return;

        if (attempt < maxRetries) {
          const delay = Math.min(1000 * Math.pow(2, attempt), 10000);
          setTimeout(() => execute(attempt + 1), delay);
        } else {
          setState((prev) => ({
            ...prev,
            loading: false,
            error,
            retryCount: attempt,
          }));
        }
      }
    }

    execute(0);
    return () => { cancelled = true; };
  }, [fetchFn]);

  const manualRetry = () => {
    setState((prev) => ({ ...prev, retryCount: 0 }));
  };

  return children({
    ...state,
    retry: manualRetry,
    canRetry: state.retryCount < maxRetries,
  });
}

// 使用
function UserProfile({ userId }) {
  return (
    <RetryableComponent fetchFn={() => fetchUser(userId)}>
      {({ data, loading, error, retry, canRetry }) => {
        if (loading) return <div>加载中...</div>;
        if (error) {
          return (
            <div>
              <p>加载失败: {error.message}</p>
              {canRetry && <button onClick={retry}>重试</button>}
            </div>
          );
        }
        return <UserCard user={data} />;
      }}
    </RetryableComponent>
  );
}
```

## 5. 网络错误处理

### 5.1 统一网络错误处理

```jsx
// 网络状态监听
function useNetworkStatus() {
  const [isOnline, setIsOnline] = useState(navigator.onLine);
  const [wasOffline, setWasOffline] = useState(false);

  useEffect(() => {
    const handleOnline = () => {
      setIsOnline(true);
      setWasOffline(true);
    };
    const handleOffline = () => {
      setIsOnline(false);
      setWasOffline(true);
    };

    window.addEventListener("online", handleOnline);
    window.addEventListener("offline", handleOffline);

    return () => {
      window.removeEventListener("online", handleOnline);
      window.removeEventListener("offline", handleOffline);
    };
  }, []);

  return { isOnline, wasOffline };
}

// 使用
function App() {
  const { isOnline, wasOffline } = useNetworkStatus();

  if (!isOnline) {
    return <OfflinePage />;
  }

  return (
    <div>
      {wasOffline && <ReconnectedBanner />}
      <MainApp />
    </div>
  );
}
```

### 5.2 请求失败统一处理

```jsx
function useApiWithRetry(url, options = {}) {
  const [state, setState] = useState({
    data: null,
    loading: true,
    error: null,
    retrying: false,
  });

  const fetchData = useCallback(async () => {
    setState((prev) => ({ ...prev, loading: true, error: null }));

    try {
      const response = await fetch(url, options);

      if (!response.ok) {
        const errorData = await response.json().catch(() => ({}));
        throw {
          status: response.status,
          message: errorData.message || `HTTP ${response.status}`,
          type: response.status >= 500 ? "SERVER_ERROR" : "CLIENT_ERROR",
        };
      }

      const data = await response.json();
      setState({ data, loading: false, error: null, retrying: false });
    } catch (error) {
      if (error.name === "AbortError") return;

      const normalizedError = {
        type: error.type || "NETWORK_ERROR",
        status: error.status,
        message: error.message || "网络请求失败",
      };

      setState({ data: null, loading: false, error: normalizedError, retrying: false });
    }
  }, [url]);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  return { ...state, retry: fetchData };
}
```

  errorReporter.captureError(
    event.reason instanceof Error
      ? event.reason
      : new Error(String(event.reason)),
    { type: "unhandledrejection" }

});

### 6.4 完整的全局错误处理

```js
// src/errorHandler.js
export function setupGlobalErrorHandlers() {
  // 1. 全局 JS 错误
  window.onerror = (message, source, lineno, colno, error) => {
    reportError({
      type: "js-error",
      message: String(message),
      source,
      lineno,
      colno,
      stack: error?.stack,
    });
    return true;
  };

  // 2. 未处理的 Promise rejection
  window.addEventListener("unhandledrejection", (event) => {
    reportError({
      type: "promise-rejection",
      reason: event.reason?.message || String(event.reason),
      stack: event.reason?.stack,
    });
    event.preventDefault();
  });

  // 3. 资源加载错误
  window.addEventListener("error", (event) => {
    if (event.target !== window) {
      reportError({
        type: "resource-error",
        tagName: event.target.tagName,
        source: event.target.src || event.target.href,
      });
    }
  }, true); // 使用捕获阶段

  // 4. React 错误（通过 ErrorBoundary）
  // 已在 ErrorBoundary 的 componentDidCatch 中处理
}

function reportError(errorData) {
  const payload = {
    ...errorData,
    url: window.location.href,
    userAgent: navigator.userAgent,
    timestamp: Date.now(),
  };

  if (navigator.sendBeacon) {
    navigator.sendBeacon("/api/errors", JSON.stringify(payload));
  } else {
    fetch("/api/errors", {
      method: "POST",
      body: JSON.stringify(payload),
      keepalive: true,
    }).catch(() => {});
  }
}
```

## 小结

- 分层错误边界：全局级、路由级、组件级，各有不同的 fallback 策略
- 错误日志应上报到 Sentry 等监控服务，包含上下文信息
- 用户友好的错误消息应该翻译技术术语，提供明确的操作指引
- 重试机制使用指数退避，只对可恢复的错误进行重试
- 全局错误处理捕获 window.onerror、unhandledrejection 等未被 React 捕获的错误
- ErrorBoundary 和 Suspense 配合使用，分别处理错误和加载状态
