# 高级类型模式

## 1. ComponentPropsWithoutRef 与 ComponentPropsWithRef

### 基本用法

```tsx
// 获取 <button> 元素的所有原生 props（不含 ref）
type ButtonHTMLProps = React.ComponentPropsWithoutRef<'button'>
// 包含 onClick, disabled, type, className 等所有 HTMLButtonElement 属性

// 获取 <a> 元素的所有原生 props
type AnchorHTMLProps = React.ComponentPropsWithoutRef<'a'>
// 包含 href, target, rel 等所有 HTMLAnchorElement 属性

// 获取组件的 props 类型
type MyComponentProps = React.ComponentPropsWithoutRef<typeof MyComponent>

// WithRef 版本 — 包含 ref 属性
type ButtonPropsWithRef = React.ComponentPropsWithRef<'button'>
// 包含 ref: React.Ref<HTMLButtonElement>
```

### 扩展原生 HTML 属性

```tsx
// 扩展 button 属性
interface ButtonProps extends React.ComponentPropsWithoutRef<'button'> {
  variant?: 'primary' | 'secondary'
  loading?: boolean
}

function Button({ variant = 'primary', loading, children, ...props }: ButtonProps) {
  return (
    <button
      className={`btn btn-${variant}`}
      disabled={loading || props.disabled}
      {...props}
    >
      {loading ? <Spinner /> : children}
    </button>
  )
}

// 使用 — 自动支持所有 button 原生属性
<Button
  variant="primary"
  loading={false}
  onClick={() => {}}           // ✅ 原生 button 属性
  type="submit"                // ✅
  disabled={false}             // ✅
  aria-label="Submit form"     // ✅
>
  提交
</Button>
```

### 获取组件 Props

```tsx
// 获取已存在组件的 props
function UserCard({ name, age }: { name: string; age: number }) {
  return <div>{name}, {age}</div>
}

// 获取 UserCard 的 props 类型
type UserCardProps = React.ComponentPropsWithoutRef<typeof UserCard>
// { name: string; age: number }

// 在其他地方复用这个类型
type EnhancedUserCardProps = UserCardProps & {
  avatar?: string
  onClick?: () => void
}
```

## 2. ElementType — 元素/组件类型

```tsx
// ElementType 表示可以作为 JSX 标签的任何东西
type MyElementType = React.ElementType
// 包括：
// - 字符串标签: 'div', 'span', 'button'
// - 函数组件: (props: Props) => ReactElement
// - 类组件: class MyComponent extends Component

// 实际应用
function renderTag(Tag: React.ElementType, props: Record<string, unknown>) {
  return <Tag {...props} />
}

// 限制为特定类型的元素
type TextElement = 'span' | 'p' | 'h1' | 'h2' | 'h3' | 'h4' | 'h5' | 'h6'

interface TypographyProps {
  as?: TextElement
  children: React.ReactNode
}

function Typography({ as: Tag = 'span', children }: TypographyProps) {
  return <Tag>{children}</Tag>
}

// 或者限制为 HTML 元素 + React 组件
function Box<C extends React.ElementType = 'div'>({
  as,
  children,
  ...props
}: {
  as?: C
  children: React.ReactNode
} & React.ComponentPropsWithoutRef<C>) {
  const Component = as || 'div'
  return <Component {...props}>{children}</Component>
}
```

## 3. PropsWithChildren 与实用类型

### React 内置工具类型

```tsx
// PropsWithChildren — 自动添加 children
type CardProps = React.PropsWithChildren<{
  title: string
  variant?: 'default' | 'outlined'
}>
// 等价于 { title: string; variant?: ...; children?: ReactNode }

// ComponentProps — 获取组件或元素类型的完整 props
type ButtonProps = React.ComponentProps<'button'>

// HTMLAttributes — HTML 元素的通用属性
type DivProps = React.HTMLAttributes<HTMLDivElement>

// CSSProperties — 内联样式类型
const style: React.CSSProperties = {
  display: 'flex',
  justifyContent: 'center',
  backgroundColor: '#fff'
}
```

### 常用工具类型速查

```tsx
// Partial<T> — 所有属性可选
type PartialProps = Partial<{ name: string; age: number }>
// { name?: string; age?: number }

// Required<T> — 所有属性必选
type RequiredProps = Required<{ name?: string; age?: number }>
// { name: string; age: number }

// Pick<T, K> — 选取部分属性
type PickedProps = Pick<{ name: string; age: number; email: string }, 'name' | 'email'>
// { name: string; email: string }

// Omit<T, K> — 排除部分属性
type OmittedProps = Omit<{ name: string; age: number; email: string }, 'email'>
// { name: string; age: number }

// Record<K, V> — 键值映射
type StringMap = Record<string, number>
// { [key: string]: number }

// Extract<T, U> — 从联合类型中提取
type Extracted = Extract<'a' | 'b' | 'c', 'a' | 'c'>
// 'a' | 'c'

// Exclude<T, U> — 从联合类型中排除
type Excluded = Exclude<'a' | 'b' | 'c', 'a' | 'c'>
// 'b'

// NonNullable<T> — 排除 null 和 undefined
type NonNull = NonNullable<string | null | undefined>
// string

// ReturnType<T> — 函数返回值类型
type FnReturn = ReturnType<() => string>
// string

// Parameters<T> — 函数参数类型元组
type FnParams = Parameters<(a: string, b: number) => void>
// [a: string, b: number]
```

## 4. 条件 Props

### 根据条件决定 props 是否必选

```tsx
// 根据 variant 决定是否需要 icon
type ButtonProps =
  | { variant: 'icon'; icon: React.ReactNode; label?: never }
  | { variant: 'text'; label: string; icon?: never }
  | { variant: 'both'; label: string; icon: React.ReactNode }

function Button(props: ButtonProps) {
  switch (props.variant) {
    case 'icon':
      return <button>{props.icon}</button>
    case 'text':
      return <button>{props.label}</button>
    case 'both':
      return <button>{props.icon} {props.label}</button>
  }
}

// 条件 props 模式
type ConditionalBaseProps = {
  label: string
}

type ConditionalProps = ConditionalBaseProps &
  (
    | { disabled: true; onClick?: never }
    | { disabled?: false; onClick: () => void }
  )

function ActionButton(props: ConditionalProps) {
  if (props.disabled) {
    return <button disabled>{props.label}</button>
  }
  return <button onClick={props.onClick}>{props.label}</button>
}
```

### 深层条件类型

```tsx
// 根据 mode 决定是受控还是非受控
type ControlledProps = {
  mode: 'controlled'
  value: string
  onChange: (value: string) => void
  defaultValue?: never
}

type UncontrolledProps = {
  mode: 'uncontrolled'
  defaultValue?: string
  value?: never
  onChange?: (value: string) => void
}

type InputProps = {
  label: string
  placeholder?: string
} & (ControlledProps | UncontrolledProps)

function Input(props: InputProps) {
  if (props.mode === 'controlled') {
    return (
      <div>
        <label>{props.label}</label>
        <input
          value={props.value}
          onChange={(e) => props.onChange(e.target.value)}
          placeholder={props.placeholder}
        />
      </div>
    )
  }

  return (
    <div>
      <label>{props.label}</label>
      <input
        defaultValue={props.defaultValue}
        onChange={(e) => props.onChange?.(e.target.value)}
        placeholder={props.placeholder}
      />
    </div>
  )
}
```

## 5. 模板字面量类型用于变体

### 生成所有变体组合

```tsx
// 生成 CSS 类名变体
type Size = 'sm' | 'md' | 'lg'
type Color = 'primary' | 'secondary' | 'danger'
type Variant = `${Size}-${Color}`
// 'sm-primary' | 'sm-secondary' | 'sm-danger'
// | 'md-primary' | 'md-secondary' | 'md-danger'
// | 'lg-primary' | 'lg-secondary' | 'lg-danger'

// 组合多个维度
type State = 'default' | 'hover' | 'active' | 'disabled'
type Theme = 'light' | 'dark'
type ClassName = `btn-${Variant}-${State}-${Theme}`

interface ButtonProps {
  className?: ClassName
  children: React.ReactNode
}

// CSS 变量命名
type CSSVarName = `--${string}`
function setVar(name: CSSVarName, value: string) {
  document.documentElement.style.setProperty(name, value)
}
setVar('--primary-color', '#007bff')  // ✅
setVar('primary-color', '#007bff')    // ❌ Error: 不符合模式
```

### 事件处理器映射

```tsx
// 自动生成事件属性名
type EventHandlerName = `on${Capitalize<string>}`
// 匹配 onClick, onChange, onSubmit 等

// 组件事件映射
type ComponentEvents = {
  [K in `on${Capitalize<string>}`]?: (...args: any[]) => void
}
```

## 6. 类型收窄（Type Narrowing）在 JSX 中

### 使用类型守卫

```tsx
// typeof 收窄
type InputValue = string | number | boolean

function Display({ value }: { value: InputValue }) {
  if (typeof value === 'string') {
    return <span>{value.toUpperCase()}</span>  // string
  }
  if (typeof value === 'number') {
    return <span>{value.toFixed(2)}</span>      // number
  }
  return <input type="checkbox" checked={value} />  // boolean
}

// instanceof 收窄
function handleError(error: Error | string) {
  if (error instanceof Error) {
    return <div className="error">{error.message}</div>
  }
  return <div className="error">{error}</div>
}

// in 操作符收窄
type LoadingState = { type: 'loading' }
type SuccessState = { type: 'success'; data: string }
type ErrorState = { type: 'error'; error: Error }
type State = LoadingState | SuccessState | ErrorState

function StatusDisplay({ state }: { state: State }) {
  if ('data' in state) {
    // TypeScript 知道 state 是 SuccessState
    return <div>{state.data}</div>
  }
  if ('error' in state) {
    return <div>Error: {state.error.message}</div>
  }
  return <Spinner />
}
```

### discriminated union 收窄

```tsx
type Shape =
  | { kind: 'circle'; radius: number }
  | { kind: 'rect'; width: number; height: number }
  | { kind: 'triangle'; base: number; height: number }

function ShapeInfo({ shape }: { shape: Shape }) {
  switch (shape.kind) {
    case 'circle':
      return <div>Circle: r={shape.radius}</div>
    case 'rect':
      return <div>Rect: {shape.width}x{shape.height}</div>
    case 'triangle':
      return <div>Triangle: base={shape.base} h={shape.height}</div>
  }
}
```

### assertion 函数

```tsx
// 断言函数收窄类型
function assertIsString(val: unknown): asserts val is string {
  if (typeof val !== 'string') {
    throw new Error(`Expected string, got ${typeof val}`)
  }
}

function Display({ value }: { value: unknown }) {
  assertIsString(value)
  // value 现在是 string
  return <div>{value.toUpperCase()}</div>
}

function assertDefined<T>(val: T | undefined | null): asserts val is T {
  if (val === undefined || val === null) {
    throw new Error('Value is not defined')
  }
}
```

## 7. Branded Types（品牌类型）

用于创建语义化更强的类型区分：

```tsx
// 使用 symbol 品牌
declare const brand: unique symbol

type Branded<T, B extends string> = T & { [brand]: B }

// 创建品牌类型
type UserId = Branded<string, 'UserId'>
type Email = Branded<string, 'Email'>
type PositiveNumber = Branded<number, 'PositiveNumber'>

// 构造函数（带验证）
function createUserId(id: string): UserId {
  if (!id.startsWith('user_')) {
    throw new Error('Invalid user ID format')
  }
  return id as UserId
}

function createEmail(email: string): Email {
  if (!email.includes('@')) {
    throw new Error('Invalid email format')
  }
  return email as Email
}

function createPositiveNumber(n: number): PositiveNumber {
  if (n <= 0) {
    throw new Error('Number must be positive')
  }
  return n as PositiveNumber
}

// 使用
function fetchUser(id: UserId) { ... }  // 只接受 UserId，不接受普通 string

const userId = createUserId('user_123')
const email = createEmail('test@example.com')
fetchUser(userId)       // ✅
fetchUser(email)        // ❌ Error: Email 不兼容 UserId
fetchUser('user_123')   // ❌ Error: string 不兼容 UserId
```

### 在 React 中使用

```tsx
// 表单字段类型安全
type FormFieldValue = Branded<string, 'FormFieldValue'>
type ValidatedField = Branded<string, 'ValidatedField'>

interface FormFieldProps {
  name: string
  value: FormFieldValue
  onChange: (value: FormFieldValue) => void
  validate?: (value: FormFieldValue) => ValidatedField | null
  error?: string
}

// API 响应类型
type UserId = Branded<string, 'UserId'>
type SessionToken = Branded<string, 'SessionToken'>

interface User {
  id: UserId
  name: string
  email: string
}

// 不会混淆不同 ID 类型
function getUser(id: UserId): Promise<User> { ... }
function getSession(token: SessionToken): Promise<Session> { ... }
```

## 8. 索引访问类型与映射类型

```tsx
// 索引访问类型 — 获取对象类型的某个属性类型
interface User {
  name: string
  address: {
    city: string
    street: string
  }
}

type Address = User['address']  // { city: string; street: string }
type Name = User['name']         // string

// 映射类型 — 批量转换属性
type ReadonlyUser = Readonly<User>       // 所有属性变为 readonly
type PartialUser = Partial<User>         // 所有属性变为可选

// 自定义映射类型
type Nullable<T> = {
  [K in keyof T]: T[K] | null
}
type NullableUser = Nullable<User>
// { name: string | null; address: { ... } | null }

// 只读深层映射
type DeepReadonly<T> = {
  readonly [K in keyof T]: T[K] extends object
    ? DeepReadonly<T[K]>
    : T[K]
}

// 可选深层映射
type DeepPartial<T> = {
  [K in keyof T]?: T[K] extends object
    ? DeepPartial<T[K]>
    : T[K]
}

// 使用
interface Config {
  api: {
    url: string
    timeout: number
  }
  theme: {
    primary: string
    dark: boolean
  }
}

type PartialConfig = DeepPartial<Config>
// api 可选，api.url 可选，theme 可选，theme.primary 可选...
```

## 9. 实战：类型安全的表单库

```tsx
// 定义表单 schema
interface FormSchema {
  username: string
  email: string
  age: number
  role: 'admin' | 'user' | 'guest'
  tags: string[]
}

// 字段级别的验证规则
type FieldRule<T> = {
  required?: boolean
  validate?: (value: T) => string | undefined
  transform?: (value: T) => T
}

type FormRules<T> = {
  [K in keyof T]: FieldRule<T[K]>
}

// 表单状态
type FormState<T> = {
  values: T
  errors: Partial<Record<keyof T, string>>
  touched: Partial<Record<keyof T, boolean>>
  isValid: boolean
}

// 使用
const userFormRules: FormRules<FormSchema> = {
  username: {
    required: true,
    validate: (v) => v.length >= 3 ? undefined : '至少3个字符'
  },
  email: {
    required: true,
    validate: (v) => v.includes('@') ? undefined : '邮箱格式不正确'
  },
  age: {
    required: true,
    validate: (v) => v > 0 ? undefined : '年龄必须大于0',
    transform: (v) => Math.floor(v)
  },
  role: { required: true },
  tags: { required: false }
}
```

## 总结

| 模式 | 用途 |
|------|------|
| `ComponentPropsWithoutRef<'el'>` | 获取 HTML 元素的原生属性类型 |
| `React.ElementType` | 泛型元素/组件类型 |
| 判别联合 | 根据某个字段值决定其他字段类型 |
| 模板字面量类型 | 生成组合变体类型 |
| 类型守卫 / 收窄 | 运行时类型检查后缩小类型范围 |
| Branded Types | 创建语义化强的区分类型 |
| 映射类型 | 批量转换属性修饰符 |
