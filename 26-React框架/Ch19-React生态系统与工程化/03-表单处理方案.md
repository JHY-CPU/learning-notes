# 表单处理方案

## 一、Controlled vs Uncontrolled 回顾

### 1.1 两种模式对比

```tsx
// Controlled（受控）- React 管理表单状态
function ControlledForm() {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    // 直接从 state 读取值
    console.log({ name, email });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        value={name}                    // 值由 React 控制
        onChange={(e) => setName(e.target.value)}  // 更新由 React 处理
      />
      <input
        value={email}
        onChange={(e) => setEmail(e.target.value)}
      />
      <button type="submit">提交</button>
    </form>
  );
}

// Uncontrolled（非受控）- DOM 管理表单状态
function UncontrolledForm() {
  const nameRef = useRef<HTMLInputElement>(null);
  const emailRef = useRef<HTMLInputElement>(null);

  const handleSubmit = (e) => {
    e.preventDefault();
    // 从 DOM 读取值
    console.log({
      name: nameRef.current?.value,
      email: emailRef.current?.value,
    });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input ref={nameRef} defaultValue="" />
      <input ref={emailRef} defaultValue="" />
      <button type="submit">提交</button>
    </form>
  );
}
```

### 1.2 选择建议

```
Controlled 适用：
✅ 需要实时验证
✅ 需要条件禁用提交按钮
✅ 需要格式化输入（如电话号码）
✅ 需要联动字段

Uncontrolled 适用：
✅ 简单表单
✅ 集成非 React 库
✅ 文件上传
✅ 大量字段（性能考虑）
```

```tsx
interface LoginForm {
```

### 2.4 setValue 和 watch

```tsx

const allValues = watch();  // 监听所有字段
```

```tsx
      </button>
```

## 三、Zod Schema 验证

### 3.1 基本用法

```bash
npm install zod @hookform/resolvers
```

```tsx
import { z } from 'zod';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';

// 定义 schema
const userSchema = z.object({
  name: z
    .string()
    .min(2, '姓名至少2个字符')
    .max(50, '姓名最多50个字符'),
  email: z
    .string()
    .email('邮箱格式不正确'),
  age: z
    .number()
    .min(18, '年龄必须大于18')
    .max(120, '年龄不合法')
    .optional(),
  password: z
    .string()
    .min(8, '密码至少8位')
    .regex(/[A-Z]/, '密码需包含大写字母')
    .regex(/[0-9]/, '密码需包含数字'),
  confirmPassword: z.string(),
  website: z
    .string()
    .url('URL 格式不正确')
    .optional()
    .or(z.literal('')),  // 允许空字符串
  role: z.enum(['admin', 'user', 'guest']),
  tags: z.array(z.string()).min(1, '至少选择一个标签'),
}).refine((data) => data.password === data.confirmPassword, {
  message: '两次密码不一致',
  path: ['confirmPassword'],  // 错误指向 confirmPassword 字段
});

// 从 schema 推导 TypeScript 类型
type UserFormData = z.infer<typeof userSchema>;

function UserForm() {
  const { register, handleSubmit, formState: { errors } } = useForm<UserFormData>({
    resolver: zodResolver(userSchema),
  });

  return (
    <form onSubmit={handleSubmit(console.log)}>
      <input {...register('name')} />
      {errors.name && <p>{errors.name.message}</p>}

      <input {...register('email')} />
      {errors.email && <p>{errors.email.message}</p>}

      <input type="password" {...register('password')} />
      {errors.password && <p>{errors.password.message}</p>}

      <input type="password" {...register('confirmPassword')} />
      {errors.confirmPassword && <p>{errors.confirmPassword.message}</p>}

      <select {...register('role')}>
        <option value="admin">管理员</option>
        <option value="user">用户</option>
        <option value="guest">访客</option>
      </select>

      <button type="submit">提交</button>
    </form>
  );
}
```

```tsx
const validationSchema = Yup.object({

            <ErrorMessage name="email" component="span" className="error" />
```

### 5.1 基本文件上传

```tsx
import { useForm } from 'react-hook-form';

interface UploadForm {
  title: string;
  file: FileList;
}

function UploadForm() {
  const { register, handleSubmit } = useForm<UploadForm>();

  const onSubmit = async (data: UploadForm) => {
    const formData = new FormData();
    formData.append('title', data.title);
    formData.append('file', data.file[0]);

    await fetch('/api/upload', {
      method: 'POST',

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input {...register('title', { required: true })} placeholder="标题" />
      <input
        type="file"
        accept="image/*,.pdf"
        {...register('file', { required: '请选择文件' })}
      />
      <button type="submit">上传</button>
    </form>
  );
}
```

### 5.2 拖拽上传

```tsx
'use client';

import { useState, useCallback } from 'react';
import { useDropzone } from 'react-dropzone';

function DragDropUpload() {
  const [files, setFiles] = useState<File[]>([]);
  const [previews, setPreviews] = useState<string[]>([]);

  const onDrop = useCallback((acceptedFiles: File[]) => {
    setFiles(prev => [...prev, ...acceptedFiles]);

    // 生成预览
    acceptedFiles.forEach(file => {
      const reader = new FileReader();
      reader.onload = () => {
        setPreviews(prev => [...prev, reader.result as string]);
      };
      reader.readAsDataURL(file);
    });
  }, []);

  const { getRootProps, getInputProps, isDragActive } = useDropzone({
    onDrop,
    accept: {
      'image/*': ['.png', '.jpg', '.jpeg', '.gif'],
    },
    maxSize: 5 * 1024 * 1024,  // 5MB
    maxFiles: 5,
  });

  return (
    <div>
      <div
        {...getRootProps()}
        className={`border-2 border-dashed p-8 text-center cursor-pointer
          ${isDragActive ? 'border-blue-500 bg-blue-50' : 'border-gray-300'}`}
      >
        <input {...getInputProps()} />
        {isDragActive ? (
          <p>放开以上传文件</p>
        ) : (
          <p>拖拽文件到这里，或点击选择</p>
        )}
      </div>

      <div className="grid grid-cols-3 gap-4 mt-4">
        {previews.map((src, index) => (
          <img key={index} src={src} alt={`预览 ${index}`} className="w-full h-32 object-cover" />
        ))}
      </div>

      <p className="mt-2 text-sm text-gray-500">
        {files.length} 个文件已选择
      </p>
    </div>
  );
}
```

## 六、多步骤表单

### 6.1 Wizard 模式

```tsx
import { useForm, FormProvider } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

// 各步骤的 schema
const step1Schema = z.object({
  name: z.string().min(2),
  email: z.string().email(),
});

const step2Schema = z.object({
  address: z.string().min(5),
  city: z.string().min(2),
  zipCode: z.string().regex(/^\d{6}$/, '邮编格式不正确'),
});

const step3Schema = z.object({
  cardNumber: z.string().regex(/^\d{16}$/),
  expiry: z.string(),
  cvv: z.string().regex(/^\d{3}$/),
});

// 合并 schema
const fullSchema = step1Schema.merge(step2Schema).merge(step3Schema);
type FormData = z.infer<typeof fullSchema>;

const steps = ['个人信息', '收货地址', '支付信息'];

function MultiStepForm() {
  const [currentStep, setCurrentStep] = useState(0);

  const methods = useForm<FormData>({
    resolver: zodResolver(fullSchema),
    mode: 'onChange',
  });

  const { trigger, handleSubmit } = methods;

  const nextStep = async () => {
    // 验证当前步骤的字段
    const fieldsToValidate = {
      0: ['name', 'email'] as const,
      1: ['address', 'city', 'zipCode'] as const,
      2: ['cardNumber', 'expiry', 'cvv'] as const,
    };

    const isValid = await trigger(fieldsToValidate[currentStep]);
    if (isValid) setCurrentStep(prev => prev + 1);
  };

  const prevStep = () => setCurrentStep(prev => prev - 1);

  const onSubmit = (data: FormData) => {
    console.log('完整数据:', data);
  };

  return (
    <FormProvider {...methods}>
      <form onSubmit={handleSubmit(onSubmit)}>
        {/* 步骤指示器 */}
        <div className="flex gap-4 mb-8">
          {steps.map((label, index) => (
            <div
              key={label}
              className={`flex-1 text-center py-2 rounded ${
                index <= currentStep ? 'bg-blue-500 text-white' : 'bg-gray-200'
              }`}
            >
              {label}
            </div>
          ))}
        </div>

        {/* 步骤内容 */}
        {currentStep === 0 && <Step1 />}
        {currentStep === 1 && <Step2 />}
        {currentStep === 2 && <Step3 />}

        {/* 导航按钮 */}
        <div className="flex justify-between mt-6">
          {currentStep > 0 && (
            <button type="button" onClick={prevStep}>上一步</button>
          )}
          {currentStep < steps.length - 1 ? (
            <button type="button" onClick={nextStep}>下一步</button>
          ) : (
            <button type="submit">提交订单</button>
          )}
        </div>
      </form>
    </FormProvider>
  );
}

// 各步骤组件使用 useFormContext
function Step1() {
  const { register, formState: { errors } } = useFormContext<FormData>();
  return (
    <div>
      <input {...register('name')} placeholder="姓名" />
      {errors.name && <p>{errors.name.message}</p>}
      <input {...register('email')} placeholder="邮箱" />
      {errors.email && <p>{errors.email.message}</p>}
    </div>
  );
}
```

## 七、表单性能优化

### 7.1 React Hook Form 性能优势

```tsx
// React Hook Form 使用 ref 而非 state
// 避免每次输入都重渲染

// ❌ 传统受控表单 - 每次输入都重渲染整个表单
function SlowForm() {
  const [form, setForm] = useState({ name: '', email: '', bio: '' });

  return (
    <form>
      {/* 每次输入，所有三个 input 都重渲染 */}
      <input value={form.name} onChange={e => setForm(p => ({...p, name: e.target.value}))} />
      <input value={form.email} onChange={e => setForm(p => ({...p, email: e.target.value}))} />
      <textarea value={form.bio} onChange={e => setForm(p => ({...p, bio: e.target.value}))} />
    </form>
  );
}

// ✅ React Hook Form - 只重渲染变化的字段
function FastForm() {
  const { register } = useForm();
  return (
    <form>
      {/* 只有变化的 input 会重渲染 */}
      <input {...register('name')} />
      <input {...register('email')} />
      <textarea {...register('bio')} />
    </form>
  );
}
```

### 7.2 减少不必要的重渲染

```tsx
import { useForm, useFormContext, useWatch } from 'react-hook-form';

// 使用 shouldUnregister 卸载时移除字段
const { register } = useForm({
  shouldUnregister: true,
});

// 使用 control 单独监听需要的字段
function PriceDisplay({ control }) {
  // 只有 price 变化时才重渲染
  const price = useWatch({ control, name: 'price' });
  const quantity = useWatch({ control, name: 'quantity' });

  return <div>总计: ¥{price * quantity}</div>;
}

// 拆分表单为小组件，减少重渲染范围
function BigForm() {
  const methods = useForm();
  return (
    <FormProvider {...methods}>
      <PersonalInfo />     {/* 独立组件，不会因其他字段变化重渲染 */}
      <AddressInfo />
      <PaymentInfo />
    </FormProvider>
  );
}
```
