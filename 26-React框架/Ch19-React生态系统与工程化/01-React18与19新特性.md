# React 18 与 19 新特性

## 一、React 18 新特性

### 1.1 自动批处理 (Automatic Batching)

```tsx
// React 17：只在 React 事件处理器中批处理
function handleClick() {
  setCount(c => c + 1);    // ✅ 批处理（React 事件中）
  setFlag(f => !f);         // ✅ 批处理
  // 只触发一次重渲染
}

setTimeout(() => {
  setCount(c => c + 1);    // ❌ 不批处理（setTimeout 中）
  setFlag(f => !f);         // ❌ 各自触发重渲染
  // 触发两次重渲染
}, 1000);

// React 18：所有更新自动批处理（无论来源）
import { createRoot } from 'react-dom/client';

function handleClick() {
  setCount(c => c + 1);
  setFlag(f => !f);
  // ✅ 一次重渲染
}

setTimeout(() => {
  setCount(c => c + 1);
  setFlag(f => !f);
  // ✅ 一次重渲染（React 18 新特性）
}, 1000);

fetch('/api/data').then(() => {
  setCount(c => c + 1);
  setData(newData);
  // ✅ 一次重渲染（Promise 回调中也批处理）
});

// 如果需要强制不批处理
import { flushSync } from 'react-dom';

flushSync(() => {
  setCount(c => c + 1);
});
// 此处 DOM 已更新
flushSync(() => {
  setFlag(f => !f);
});
// 两次重渲染，但在 flushSync 后 DOM 立即更新
```

### 1.2 Transitions (过渡)

```tsx
import { useTransition, useDeferredValue } from 'react';

// ===== useTransition =====
// 将更新标记为"非紧急"，不阻塞用户输入

function SearchResults() {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);
  const [isPending, startTransition] = useTransition();

  function handleChange(e) {
    const value = e.target.value;
    setQuery(value);  // 紧急更新：立即更新输入框

    startTransition(() => {
      // 非紧急更新：搜索结果可以延迟
      setResults(filterResults(value));
    });
  }

  return (
    <div>
      <input value={query} onChange={handleChange} />
      {isPending && <Spinner />}
      <Results items={results} />
    </div>
  );
}

// Transition 的特点：
// ✅ 可以被打断（用户输入会中断 transition）
// ✅ 保持 UI 响应（输入框不会卡顿）
// ✅ isPending 反映过渡状态
// ❌ 不能用于受控输入的值更新
// ❌ 只能包裹状态更新，不能包裹副作用
```

```tsx
// ===== useDeferredValue =====
// 创建一个值的延迟版本

function SearchPage() {
  const [query, setQuery] = useState('');
  const deferredQuery = useDeferredValue(query);

  // query：立即更新（用户输入实时反映）
  // deferredQuery：延迟更新（React 在空闲时更新）

  const isStale = query !== deferredQuery;  // 判断是否过时

  return (
    <div>
      <input value={query} onChange={e => setQuery(e.target.value)} />
      <div style={{ opacity: isStale ? 0.5 : 1 }}>
        <ExpensiveResults query={deferredQuery} />
      </div>
    </div>
  );
}

// useDeferredValue vs useTransition
// useTransition: 控制"何时"更新（包裹更新逻辑）
// useDeferredValue: 控制"哪个值"延迟（创建延迟副本）
```

### 1.3 Suspense 改进

```tsx
import { Suspense } from 'react';

// React 18 支持服务端 Suspense（流式 SSR）
function App() {
  return (
    <Layout>
      <Suspense fallback={<NavBarSkeleton />}>
        <NavBar />
      </Suspense>
      <Suspense fallback={<SidebarSkeleton />}>
        <Sidebar />
      </Suspense>
      <Suspense fallback={<ContentSkeleton />}>
        <Content />
      </Suspense>
    </Layout>
  );
}

// 嵌套 Suspense
function Page() {
  return (
    <Suspense fallback={<PageSkeleton />}>
      <Header />
      <Suspense fallback={<CommentsSkeleton />}>
        <Comments />
      </Suspense>
      <Suspense fallback={<RecommendationsSkeleton />}>
        <Recommendations />
      </Suspense>
    </Suspense>
  );
}
```

### 1.4 StrictMode 改进

```tsx
import { StrictMode } from 'react';

// React 18 StrictMode：
// 1. 组件挂载后，立即卸载再重新挂载
//    → 测试组件的 mount/unmount/cleanup 是否正确
// 2. 状态更新后，回退状态再恢复
//    → 测试 effect 的清理和重建

function App() {
  return (
    <StrictMode>
      {/* 开发模式下：组件可能被双调用以检测副作用问题 */}
      <MyComponent />
    </StrictMode>
  );
}

// 常见问题：effect 中创建了两次资源
// ✅ 正确：有清理函数
useEffect(() => {
  const controller = new AbortController();
  fetch('/api/data', { signal: controller.signal });
  return () => controller.abort();  // 清理
}, []);

// ❌ 错误：没有清理
useEffect(() => {
  const timer = setInterval(tick, 1000);
  // 没有 clearInterval → 双调用会创建两个 timer
}, []);
```

### 1.5 新的 Hooks

```tsx
// useId - 生成唯一 ID（SSR 安全）
function FormField({ label, helpText }) {
  const id = useId();

  return (
    <div>
      <label htmlFor={id}>{label}</label>
      <input id={id} aria-describedby={`${id}-help`} />
      <span id={`${id}-help`}>{helpText}</span>
    </div>
  );
}

// useSyncExternalStore - 订阅外部 store
import { useSyncExternalStore } from 'react';

function useOnlineStatus() {
  return useSyncExternalStore(
    subscribe,   // (callback) => unsubscribe
    () => navigator.onLine,  // getSnapshot
    () => true   // getServerSnapshot（服务端）
  );
}

// useInsertionEffect - 在 DOM 变更前插入样式
// 主要供 CSS-in-JS 库使用
function useCSS(rule) {
  useInsertionEffect(() => {
    const style = document.createElement('style');
    style.textContent = rule;
    document.head.appendChild(style);
    return () => document.head.removeChild(style);
  });
}
```

## 二、React 19 新特性

### 2.1 Server Components（正式发布）

```tsx
// Server Components - 在服务端执行，零客户端 JS
// 不需要 'use client' 标记就是 Server Component

// app/page.tsx
async function ProductPage({ id }: { id: string }) {
  // 直接访问数据库
  const product = await db.product.findUnique({ where: { id } });

  // 服务端计算，不发送到客户端
  const relatedProducts = await findRelated(product.category);

  return (
    <div>
      <ProductDetails product={product} />
      <Suspense fallback={<ReviewsSkeleton />}>
        <ProductReviews productId={id} />
      </Suspense>
    </div>
  );
}

// 关键特性：
// ✅ 零 bundle 大小（不发送到客户端）
// ✅ 可以直接访问后端资源
// ✅ 在构建时或请求时在服务端执行
// ✅ 可以导入 Client Components
// ❌ 不能使用 useState、useEffect
// ❌ 不能使用浏览器 API
```

### 2.2 Actions 与 useActionState

```tsx
// Actions - 直接在事件中调用异步函数
// 不再需要手动管理 pending/error 状态

// app/actions.ts
'use server';

async function createTodo(prevState: any, formData: FormData) {
  const title = formData.get('title') as string;

  if (!title) {
    return { error: '标题不能为空' };
  }

  await db.todo.create({ data: { title } });
  revalidatePath('/todos');
  return { success: true };
}

// app/TodoForm.tsx
'use client';

import { useActionState } from 'react';
import { createTodo } from './actions';

function TodoForm() {
  // useActionState 替代了 React 18 的 useFormState
  const [state, formAction, isPending] = useActionState(
    createTodo,
    null  // 初始状态
  );

  return (
    <form action={formAction}>
      <input name="title" placeholder="新待办" disabled={isPending} />
      <button type="submit" disabled={isPending}>
        {isPending ? '添加中...' : '添加'}
      </button>
      {state?.error && <p className="error">{state.error}</p>}
    </form>
  );
}

// useActionState 参数：
// 1. action: 异步函数 (prevState, payload) => newState
// 2. initialState: 初始状态
// 返回: [state, dispatch, isPending]
```

### 2.3 use() Hook

```tsx
// use() - 在渲染中读取 Promise 或 Context
// 这是一个新的原生 Hook，可以在条件和循环中使用

// 读取 Promise
function Comments({ commentsPromise }) {
  // use() 会在 Promise resolve 之前挂起组件
  const comments = use(commentsPromise);

  return (
    <ul>
      {comments.map(c => <li key={c.id}>{c.text}</li>)}
    </ul>
  );
}

// 配合 Suspense 使用
function Page() {
  const commentsPromise = fetchComments();

  return (
    <Suspense fallback={<CommentsSkeleton />}>
      <Comments commentsPromise={commentsPromise} />
    </Suspense>
  );
}

// 读取 Context（可以条件调用！）
function ThemeText() {
  if (Math.random() > 0.5) {
    // use(context) 可以在条件中使用
    // 传统的 useContext 不能
    const theme = use(ThemeContext);
    return <div>当前主题: {theme}</div>;
  }
  return <div>随机结果</div>;
}

// use() vs useContext:
// useContext: 不能在条件/循环中
// use(context): 可以在条件/循环中
```

### 2.4 ref 作为 prop

```tsx
// React 19：ref 作为普通 prop 传递
// 不再需要 forwardRef！

// ❌ React 18：需要 forwardRef
const Input = forwardRef(function Input(props, ref) {
  return <input ref={ref} {...props} />;
});

// ✅ React 19：ref 直接作为 prop
function Input({ ref, ...props }) {
  return <input ref={ref} {...props} />;
}

// 使用方式不变
function App() {
  const inputRef = useRef(null);
  return <Input ref={inputRef} placeholder="输入内容" />;
}

// 同样适用于 Server Components
// Server Component 也可以接收 ref prop
function ServerInput({ ref, ...props }) {
  // ref 在服务端无意义，但不会报错
  return <input {...props} />;
}
```

### 2.5 Context 作为 Provider

```tsx
// React 19：直接使用 Context 作为 Provider
// 不需要 .Provider

// ❌ React 18
import { ThemeContext } from './context';

function App() {
  return (
    <ThemeContext.Provider value="dark">
      <Child />
    </ThemeContext.Provider>
  );
}

// ✅ React 19
import { ThemeContext } from './context';

function App() {
  return (
    <ThemeContext value="dark">
      <Child />
    </ThemeContext>
  );
}

// 创建 Context 的方式不变
export const ThemeContext = createContext('light');
```

### 2.6 Document Metadata（原生支持）

```tsx
// React 19：原生支持 <title>, <meta>, <link> 标签
// 不需要 react-helmet 等第三方库

function BlogPost({ post }) {
  return (
    <article>
      {/* React 会自动将这些标签提升到 <head> */}
      <title>{post.title} | My Blog</title>
      <meta name="description" content={post.excerpt} />
      <meta property="og:title" content={post.title} />
      <link rel="canonical" href={`/blog/${post.slug}`} />

      <h1>{post.title}</h1>
      <p>{post.content}</p>
    </article>
  );
}

// 支持静态和动态 metadata
function Layout({ children, title }) {
  return (
    <>
      <title>{title}</title>
      <meta name="viewport" content="width=device-width, initial-scale=1" />
      <Navbar />
      <main>{children}</main>
      <Footer />
    </>
  );
}

// 在 Server Components 和 Client Components 中都可用
// SSR 时：直接输出到 HTML <head>
// CSR 时：动态更新 <head>
```

### 2.7 预加载 API

```tsx
// React 19 提供新的资源预加载 API

import {
  prefetchDNS,
  preconnect,
  preload,
  preinit,
} from 'react-dom';

function App() {
  // 预解析 DNS
  prefetchDNS('https://api.example.com');

  // 预连接（DNS + TCP + TLS）
  preconnect('https://cdn.example.com');

  // 预加载资源
  preload('/fonts/my-font.woff2', { as: 'font', type: 'font/woff2' });
  preload('/hero.jpg', { as: 'image' });

  // 预初始化（加载并执行）
  preinit('/scripts/analytics.js', { as: 'script' });

  return <div>App</div>;
}

// 可以在 Server Components 中使用
// 这些调用会生成对应的 <link> 标签
```

### 2.8 样式表优先级

```tsx
// React 19：支持 <link rel="stylesheet"> 内联
// SSR 时自动处理样式表优先级

function Component() {
  return (
    <div>
      {/* React 会确保样式表在使用前加载 */}
      <link rel="stylesheet" href="/styles/component.css" precedence="high" />
      <link rel="stylesheet" href="/styles/theme.css" precedence="default" />
      <div className="styled">内容</div>
    </div>
  );
}

// Server Components 中也可以使用
// 避免了 CSS-in-JS 在 SSR 时的闪烁问题
```

## 总结

| 版本 | 核心特性 |
|------|---------|
| **React 18** | 自动批处理、Transitions、Suspense 改进、StrictMode 双调用、useId、useSyncExternalStore |
| **React 19** | Server Components 正式化、Actions/useActionState、use() hook、ref as prop、Context as Provider、Document Metadata、预加载 API |
| **迁移重点** | 移除 forwardRef、.Provider 简化、useFormState → useActionState、清理废弃 API |
