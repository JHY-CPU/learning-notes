# 复杂表单模式

## 1. 动态表单字段

### 1.1 动态增减字段

```jsx
function DynamicForm() {
  const [fields, setFields] = useState([{ id: 1, name: '', email: '' }]);

  const addField = () => {
    setFields((prev) => [...prev, { id: Date.now(), name: '', email: '' }]);
  };

  const removeField = (id) => {
    if (fields.length <= 1) return; // 至少保留一个
    setFields((prev) => prev.filter((f) => f.id !== id));
  };

  const updateField = (id, key, value) => {
    setFields((prev) =>
      prev.map((f) => (f.id === id ? { ...f, [key]: value } : f))
    );
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('所有成员:', fields);
  };

  return (
    <form onSubmit={handleSubmit}>
      {fields.map((field, index) => (
        <div key={field.id} className="field-group">
          <span>成员 {index + 1}</span>
          <input
            placeholder="姓名"
            value={field.name}
            onChange={(e) => updateField(field.id, 'name', e.target.value)}
          />
          <input
            placeholder="邮箱"
            value={field.email}
            onChange={(e) => updateField(field.id, 'email', e.target.value)}
          />
          <button type="button" onClick={() => removeField(field.id)}>
            删除
          </button>
        </div>
      ))}
      <button type="button" onClick={addField}>+ 添加成员</button>
      <button type="submit">提交</button>
    </form>
  );
}
```

### 1.2 使用 useFieldArray（React Hook Form）

```jsx
import { useForm, useFieldArray } from 'react-hook-form';

function DynamicFormRHForm() {
  const { register, control, handleSubmit } = useForm({
    defaultValues: {
      members: [{ name: '', email: '' }],
    },
  });

  const { fields, append, remove, move } = useFieldArray({
    control,
    name: 'members',
  });

  return (
    <form onSubmit={handleSubmit((data) => console.log(data))}>
      {fields.map((field, index) => (
        <div key={field.id}>
          <input {...register(`members.${index}.name`)} placeholder="姓名" />
          <input {...register(`members.${index}.email`)} placeholder="邮箱" />
          <button type="button" onClick={() => remove(index)}>删除</button>
          {index > 0 && (
            <button type="button" onClick={() => move(index, index - 1)}>上移</button>
          )}
        </div>
      ))}
      <button type="button" onClick={() => append({ name: '', email: '' })}>
        添加
      </button>
      <button type="submit">提交</button>
    </form>
  );
}
```

## 2. 多步骤表单 (Wizard)

### 2.1 基本实现

```jsx
function MultiStepForm() {
  const [step, setStep] = useState(0);
  const [formData, setFormData] = useState({
    // Step 1: 基本信息
    name: '',
    email: '',
    phone: '',
    // Step 2: 地址信息
    province: '',
    city: '',
    address: '',
    // Step 3: 支付信息
    paymentMethod: '',
    cardNumber: '',
    // Step 4: 确认
  });

  const steps = ['基本信息', '地址信息', '支付方式', '确认提交'];

  const updateField = (field, value) => {
    setFormData((prev) => ({ ...prev, [field]: value }));
  };

  const canProceed = () => {
    switch (step) {
      case 0:
        return formData.name && formData.email && formData.phone;
      case 1:
        return formData.province && formData.city && formData.address;
      case 2:
        return formData.paymentMethod;
      default:
        return true;
    }
  };

  const next = () => setStep((s) => Math.min(s + 1, steps.length - 1));
  const prev = () => setStep((s) => Math.max(s - 1, 0));

  const handleSubmit = () => {
    console.log('提交:', formData);
  };

  return (
    <div className="wizard">
      {/* 进度指示器 */}
      <div className="steps-indicator">
        {steps.map((label, i) => (
          <div key={i} className={`step ${i <= step ? 'active' : ''}`}>
            {i + 1}. {label}
          </div>
        ))}
      </div>

      {/* 步骤内容 */}
      <div className="step-content">
        {step === 0 && (
          <div>
            <input placeholder="姓名" value={formData.name}
              onChange={(e) => updateField('name', e.target.value)} />
            <input placeholder="邮箱" value={formData.email}
              onChange={(e) => updateField('email', e.target.value)} />
            <input placeholder="电话" value={formData.phone}
              onChange={(e) => updateField('phone', e.target.value)} />
          </div>
        )}
        {step === 1 && (
          <div>
            <input placeholder="省份" value={formData.province}
              onChange={(e) => updateField('province', e.target.value)} />
            <input placeholder="城市" value={formData.city}
              onChange={(e) => updateField('city', e.target.value)} />
            <input placeholder="详细地址" value={formData.address}
              onChange={(e) => updateField('address', e.target.value)} />
          </div>
        )}
        {step === 2 && (
          <div>
            <select value={formData.paymentMethod}
              onChange={(e) => updateField('paymentMethod', e.target.value)}>
              <option value="">选择支付方式</option>
              <option value="alipay">支付宝</option>
              <option value="wechat">微信支付</option>
              <option value="card">银行卡</option>
            </select>
          </div>
        )}
        {step === 3 && (
          <div>
            <h3>确认信息</h3>
            <pre>{JSON.stringify(formData, null, 2)}</pre>
          </div>
        )}
      </div>

      {/* 导航按钮 */}
      <div className="actions">
        {step > 0 && <button onClick={prev}>上一步</button>}
        {step < steps.length - 1 ? (
          <button onClick={next} disabled={!canProceed()}>下一步</button>
        ) : (
          <button onClick={handleSubmit}>提交</button>
        )}
      </div>
    </div>
  );
}
```

### 2.2 分步验证

```jsx
function useMultiStepForm(steps) {
  const [currentStep, setCurrentStep] = useState(0);
  const [completedSteps, setCompletedSteps] = useState(new Set());

  const isFirstStep = currentStep === 0;
  const isLastStep = currentStep === steps.length - 1;

  const goTo = (step) => setCurrentStep(step);
  const next = () => {
    if (!isLastStep) {
      setCompletedSteps((prev) => new Set([...prev, currentStep]));
      setCurrentStep((s) => s + 1);
    }
  };
  const prev = () => {
    if (!isFirstStep) setCurrentStep((s) => s - 1);
  };

  const canAccessStep = (step) => {
    // 可以访问已完成的步骤或紧邻的下一步
    return step <= Math.max(...completedSteps, -1) + 1;
  };

  return { currentStep, isFirstStep, isLastStep, goTo, next, prev, canAccessStep, completedSteps };
}
```

## 3. 条件表单字段

```jsx
function ConditionalForm() {
  const [formData, setFormData] = useState({
    type: '',        // 'individual' | 'company'
    name: '',
    email: '',
    // 个人特有
    idCard: '',
    // 企业特有
    companyName: '',
    taxId: '',
    legalPerson: '',
  });

  const update = (field, value) => {
    setFormData((prev) => ({ ...prev, [field]: value }));
  };

  return (
    <form>
      <div>
        <label>
          <input type="radio" name="type" value="individual"
            checked={formData.type === 'individual'}
            onChange={(e) => update('type', e.target.value)} />
          个人
        </label>
        <label>
          <input type="radio" name="type" value="company"
            checked={formData.type === 'company'}
            onChange={(e) => update('type', e.target.value)} />
          企业
        </label>
      </div>

      {/* 公共字段 */}
      <input placeholder="联系人" value={formData.name}
        onChange={(e) => update('name', e.target.value)} />
      <input placeholder="邮箱" value={formData.email}
        onChange={(e) => update('email', e.target.value)} />

      {/* 条件字段 */}
      {formData.type === 'individual' && (
        <input placeholder="身份证号" value={formData.idCard}
          onChange={(e) => update('idCard', e.target.value)} />
      )}

      {formData.type === 'company' && (
        <>
          <input placeholder="公司名称" value={formData.companyName}
            onChange={(e) => update('companyName', e.target.value)} />
          <input placeholder="税号" value={formData.taxId}
            onChange={(e) => update('taxId', e.target.value)} />
          <input placeholder="法人代表" value={formData.legalPerson}
            onChange={(e) => update('legalPerson', e.target.value)} />
        </>
      )}
    </form>
  );
}
```

## 4. 联动表单

```jsx
function LinkedForm() {
  const [province, setProvince] = useState('');
  const [city, setCity] = useState('');

  const regions = {
    guangdong: ['广州', '深圳', '东莞', '佛山'],
    zhejiang: ['杭州', '宁波', '温州', '嘉兴'],
    jiangsu: ['南京', '苏州', '无锡', '常州'],
  };

  const cities = regions[province] || [];

  const handleProvinceChange = (e) => {
    setProvince(e.target.value);
    setCity(''); // 省份变化时重置城市
  };

  return (
    <div>
      <select value={province} onChange={handleProvinceChange}>
        <option value="">选择省份</option>
        {Object.keys(regions).map((key) => (
          <option key={key} value={key}>{key}</option>
        ))}
      </select>

      <select value={city} onChange={(e) => setCity(e.target.value)} disabled={!province}>
        <option value="">选择城市</option>
        {cities.map((c) => (
          <option key={c} value={c}>{c}</option>
        ))}
      </select>
    </div>
  );
}
```

## 5. 表单数据持久化

```jsx
function PersistentForm() {
  const DRAFT_KEY = 'form-draft';

  const [formData, setFormData] = useState(() => {
    // 页面加载时恢复草稿
    const saved = localStorage.getItem(DRAFT_KEY);
    return saved ? JSON.parse(saved) : { title: '', content: '' };
  });

  // 自动保存草稿（带防抖）
  useEffect(() => {
    const timer = setTimeout(() => {
      localStorage.setItem(DRAFT_KEY, JSON.stringify(formData));
    }, 1000);
    return () => clearTimeout(timer);
  }, [formData]);

  const handleSubmit = (e) => {
    e.preventDefault();
    // 提交后清除草稿
    localStorage.removeItem(DRAFT_KEY);
    console.log('提交:', formData);
  };

  const handleClearDraft = () => {
    localStorage.removeItem(DRAFT_KEY);
    setFormData({ title: '', content: '' });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input value={formData.title}
        onChange={(e) => setFormData((p) => ({ ...p, title: e.target.value }))} />
      <textarea value={formData.content}
        onChange={(e) => setFormData((p) => ({ ...p, content: e.target.value }))} />
      <button type="submit">发布</button>
      <button type="button" onClick={handleClearDraft}>清除草稿</button>
    </form>
  );
}
```

## 6. 文件上传

```jsx
function FileUpload() {
  const [files, setFiles] = useState([]);
  const [previews, setPreviews] = useState([]);
  const fileRef = useRef(null);

  const handleFileChange = (e) => {
    const selectedFiles = Array.from(e.target.files);

    // 验证文件
    const validFiles = selectedFiles.filter((file) => {
      const isValidType = ['image/jpeg', 'image/png', 'image/gif'].includes(file.type);
      const isValidSize = file.size <= 5 * 1024 * 1024; // 5MB
      return isValidType && isValidSize;
    });

    setFiles(validFiles);

    // 生成预览
    const newPreviews = validFiles.map((file) => ({
      name: file.name,
      url: URL.createObjectURL(file),
    }));
    setPreviews(newPreviews);
  };

  // 清理预览 URL
  useEffect(() => {
    return () => {
      previews.forEach((p) => URL.revokeObjectURL(p.url));
    };
  }, [previews]);

  const handleUpload = async () => {
    const formData = new FormData();
    files.forEach((file) => formData.append('files', file));

    try {
      const res = await fetch('/api/upload', { method: 'POST', body: formData });
      const data = await res.json();
      console.log('上传成功', data);
    } catch (err) {
      console.error('上传失败', err);
    }
  };

  // 拖拽上传
  const handleDragOver = (e) => e.preventDefault();
  const handleDrop = (e) => {
    e.preventDefault();
    const droppedFiles = Array.from(e.dataTransfer.files);
    // 同样走验证逻辑...
  };

  return (
    <div>
      <div onDragOver={handleDragOver} onDrop={handleDrop}>
        <input ref={fileRef} type="file" multiple accept="image/*"
          onChange={handleFileChange} style={{ display: 'none' }} />
        <button onClick={() => fileRef.current.click()}>选择文件</button>
        <p>或拖拽文件到此处</p>
      </div>

      <div className="previews">
        {previews.map((p) => (
          <div key={p.name}>
            <img src={p.url} alt={p.name} width={100} />
            <span>{p.name}</span>
          </div>
        ))}
      </div>

      {files.length > 0 && <button onClick={handleUpload}>上传</button>}
    </div>
  );
}
```

## 7. 表单状态管理对比

| 方案 | 适用场景 | 复杂度 | 性能 |
|------|---------|--------|------|
| useState + 手写 | 简单表单 | 低 | 一般 |
| useReducer | 中等复杂度 | 中 | 较好 |
| React Hook Form | 复杂表单、高性能 | 中 | 优秀 |
| Formik | 传统方案 | 中 | 一般 |
| useActionState (R19) | Server Action | 低 | 好 |

React Hook Form 基于**非受控组件**（ref），避免了每次输入都触发重渲染，性能优势明显。
