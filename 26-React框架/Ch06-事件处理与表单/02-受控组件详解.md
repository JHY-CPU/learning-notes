# 受控组件详解

## 1. 受控组件 vs 非受控组件

### 1.1 核心区别

| 特性 | 受控组件 | 非受控组件 |
|------|---------|-----------|
| 数据存储 | React state | DOM 本身 |
| 值来源 | `value` 属性 | `defaultValue` 属性 |
| 更新方式 | `onChange` + setState | DOM 自动管理 |
| 数据访问 | 直接读取 state | 需要 ref 访问 DOM |
| 适用场景 | 表单验证、即时反馈、联动 | 简单表单、第三方集成 |

### 1.2 受控组件工作流程

```
用户输入 → onChange 触发 → setState 更新 state → React 重新渲染 → value 更新显示
```

```jsx
// 受控组件：React 完全控制输入值
function ControlledInput() {
  const [value, setValue] = useState('');

  const handleChange = (e) => {
    setValue(e.target.value); // 1. 用户输入 → 2. 更新 state
  };

  return (
    // 3. React 渲染 → 4. input 显示新值
    <input value={value} onChange={handleChange} />
  );
}
```

## 2. 各类型表单元素的受控方式

### 2.1 文本输入 (input text)

```jsx
function TextInput() {
  const [name, setName] = useState('');

  return (
    <div>
      <input
        type="text"
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="请输入姓名"
      />
      <p>当前值: {name}</p>
    </div>
  );
}
```

### 2.2 数字输入 (input number)

```jsx
function NumberInput() {
  const [age, setAge] = useState(0);

  const handleChange = (e) => {
    // e.target.value 始终是字符串，需要转换
    const value = e.target.value;
    if (value === '') {
      setAge(0);
    } else {
      const num = parseInt(value, 10);
      if (!isNaN(num)) {
        setAge(num);
      }
    }
  };

  return (
    <input
      type="number"
      value={age}
      onChange={handleChange}
      min={0}
      max={120}
    />
  );
}
```

### 2.3 多行文本 (textarea)

```jsx
function TextArea() {
  const [bio, setBio] = useState('');

  return (
    <div>
      <textarea

```

### 2.4 单选按钮 (radio)

```jsx
function RadioGroup() {
  const [gender, setGender] = useState('');

  return (
    <div>
      <label>
        <input
          type="radio"
          name="gender"
          value="male"
          checked={gender === 'male'}
          onChange={(e) => setGender(e.target.value)}
        />
        男
      </label>
      <label>
        <input
          type="radio"
          name="gender"
          value="female"
          checked={gender === 'female'}
          onChange={(e) => setGender(e.target.value)}
        />
        女
      </label>
      <p>选择: {gender}</p>
    </div>
  );
}

// 通用 RadioGroup 组件
function RadioGroup({ name, options, value, onChange }) {
  return (
    <div>
      {options.map((option) => (
        <label key={option.value}>
          <input
            type="radio"
            name={name}
            value={option.value}
            checked={value === option.value}
            onChange={(e) => onChange(e.target.value)}
          />
          {option.label}
        </label>
      ))}
    </div>
  );
}

// 使用
function App() {
  const [plan, setPlan] = useState('free');
  return (
    <RadioGroup
      name="plan"
      value={plan}
      onChange={setPlan}
      options={[
        { value: 'free', label: '免费版' },
        { value: 'pro', label: '专业版' },
        { value: 'enterprise', label: '企业版' },
      ]}
    />
  );
}
```

### 2.5 复选框 (checkbox)

```jsx
function CheckboxExamples() {
  // 单个复选框
  const [agreed, setAgreed] = useState(false);

  // 多个复选框
  const [hobbies, setHobbies] = useState([]);

  const handleHobbyChange = (hobby) => {
    setHobbies((prev) =>
      prev.includes(hobby)
        ? prev.filter((h) => h !== hobby)
        : [...prev, hobby]
    );
  };

  return (
    <div>
      {/* 单个复选框 */}
      <label>
        <input
          type="checkbox"
          checked={agreed}
          onChange={(e) => setAgreed(e.target.checked)}
        />
        我同意用户协议
      </label>

      {/* 多个复选框 */}
      {['阅读', '运动', '音乐', '编程'].map((hobby) => (
        <label key={hobby}>
          <input
            type="checkbox"
            checked={hobbies.includes(hobby)}
            onChange={() => handleHobbyChange(hobby)}
          />
          {hobby}
        </label>
      ))}
      <p>已选: {hobbies.join(', ')}</p>
    </div>
  );
}
```

// 多选
function MultiSelect() {

## 3. 统一表单管理

### 3.1 多字段统一处理

```jsx
function RegistrationForm() {
  const [formData, setFormData] = useState({
    username: '',
    email: '',
    password: '',
    confirmPassword: '',
    gender: '',
    hobbies: [],
    bio: '',
    agreed: false,
  });

  // 统一的 onChange 处理
  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;

    setFormData((prev) => ({
      ...prev,
      [name]: type === 'checkbox' ? checked : value,
    }));
  };

  // 复选框数组处理
  const handleHobbyChange = (e) => {
    const { value, checked } = e.target;
    setFormData((prev) => ({
      ...prev,
      hobbies: checked
        ? [...prev.hobbies, value]
        : prev.hobbies.filter((h) => h !== value),
    }));
  };

  return (
    <form>
      <input name="username" value={formData.username} onChange={handleChange} />
      <input name="email" type="email" value={formData.email} onChange={handleChange} />
      <input name="password" type="password" value={formData.password} onChange={handleChange} />
      <textarea name="bio" value={formData.bio} onChange={handleChange} />
      <label>
        <input name="agreed" type="checkbox" checked={formData.agreed} onChange={handleChange} />
        同意协议
      </label>
    </form>
  );
}
```

### 3.2 自定义 useForm Hook

```jsx
function useForm(initialValues) {
  const [values, setValues] = useState(initialValues);
  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});

  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;
    setValues((prev) => ({
      ...prev,
      [name]: type === 'checkbox' ? checked : value,
    }));
  };

  const handleBlur = (e) => {
    const { name } = e.target;
    setTouched((prev) => ({ ...prev, [name]: true }));
  };

  const reset = () => {
    setValues(initialValues);
    setErrors({});
    setTouched({});
  };

  const getFieldProps = (name) => ({
    name,
    value: values[name],
    onChange: handleChange,
    onBlur: handleBlur,
  });

  return { values, errors, touched, handleChange, handleBlur, reset, setValues, setErrors, getFieldProps };
}

// 使用
function LoginForm() {
  const { values, errors, touched, getFieldProps, reset } = useForm({
    email: '',
    password: '',
  });

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('提交', values);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input {...getFieldProps('email')} type="email" placeholder="邮箱" />
      <input {...getFieldProps('password')} type="password" placeholder="密码" />
      <button type="submit">登录</button>
      <button type="button" onClick={reset}>重置</button>
    </form>
  );
}
```

## 4. 表单验证

### 4.1 实时验证

```jsx
function ValidatedForm() {
  const [email, setEmail] = useState('');
  const [error, setError] = useState('');

  const validateEmail = (value) => {
    if (!value) return '邮箱不能为空';
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) return '邮箱格式不正确';
    return '';
  };

  const handleChange = (e) => {
    const value = e.target.value;
    setEmail(value);
    // 实时验证
    setError(validateEmail(value));
  };

  const handleBlur = () => {
    // 失焦时验证（可选，有些验证适合失焦时触发）
    setError(validateEmail(email));
  };

  return (
    <div>
      <input
        value={email}
        onChange={handleChange}
        onBlur={handleBlur}
        className={error ? 'error' : ''}
      />
      {error && <span className="error-message">{error}</span>}
    </div>
  );
}
```

### 4.2 提交时验证

```jsx
function SubmitValidationForm() {
  const [formData, setFormData] = useState({ username: '', password: '' });
  const [errors, setErrors] = useState({});

  const validate = () => {
    const newErrors = {};
    if (!formData.username.trim()) {
      newErrors.username = '用户名不能为空';
    } else if (formData.username.length < 3) {
      newErrors.username = '用户名至少 3 个字符';
    }
    if (!formData.password) {
      newErrors.password = '密码不能为空';
    } else if (formData.password.length < 6) {
      newErrors.password = '密码至少 6 位';
    }
    return newErrors;
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const newErrors = validate();
    setErrors(newErrors);

    if (Object.keys(newErrors).length === 0) {
      // 验证通过
      console.log('提交成功', formData);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <input
          name="username"
          value={formData.username}
          onChange={(e) => setFormData((p) => ({ ...p, username: e.target.value }))}
        />
        {errors.username && <span>{errors.username}</span>}
      </div>
      <div>
        <input
          name="password"
          type="password"
          value={formData.password}
          onChange={(e) => setFormData((p) => ({ ...p, password: e.target.value }))}
        />
        {errors.password && <span>{errors.password}</span>}
      </div>
      <button type="submit">注册</button>
    </form>
  );
}
```

## 5. 受控组件的常见问题

### 5.1 输入法 (IME) 处理

中文、日文等输入法在组合输入过程中会触发多次 onChange，可能产生中间状态：

```jsx
function IMEInput() {
  const [value, setValue] = useState('');
  const [isComposing, setIsComposing] = useState(false);

  const handleChange = (e) => {
    setValue(e.target.value);
  };

  const handleCompositionStart = () => {
    setIsComposing(true); // 输入法开始组合
  };

  const handleCompositionEnd = (e) => {
    setIsComposing(false); // 输入法组合结束
    setValue(e.target.value); // 此时才是最终值
  };

  const handleKeyDown = (e) => {
    if (e.key === 'Enter' && !isComposing) {
      // 组合过程中按回车不提交
      console.log('提交', value);
    }
  };

  return (
    <input
      value={value}
      onChange={handleChange}
      onCompositionStart={handleCompositionStart}
      onCompositionEnd={handleCompositionEnd}
      onKeyDown={handleKeyDown}
    />
  );
}
```

### 5.2 光标位置丢失

某些受控操作（如输入格式化）会导致光标跳到末尾：

```jsx
function FormattedInput() {
  const [phone, setPhone] = useState('');

  const formatPhone = (value) => {
    // 只保留数字
    const digits = value.replace(/\D/g, '');
    // 格式化为 xxx-xxxx-xxxx
    if (digits.length <= 3) return digits;
    if (digits.length <= 7) return `${digits.slice(0, 3)}-${digits.slice(3)}`;
    return `${digits.slice(0, 3)}-${digits.slice(3, 7)}-${digits.slice(7, 11)}`;
  };

  const handleChange = (e) => {
    const formatted = formatPhone(e.target.value);
    setPhone(formatted);
  };

  return (
    <input
      value={phone}
      onChange={handleChange}
      placeholder="手机号"
      maxLength={13}
    />
  );
  // 注意：格式化后光标可能跳到末尾
  // 复杂场景需要手动通过 ref 恢复光标位置
}
```

### 5.3 defaultValue vs value

```jsx
// defaultValue：非受控，初始值设置后由 DOM 管理
<input defaultValue="初始值" /> // React 只在首次渲染设置

// value：受控，始终由 React state 驱动
const [val, setVal] = useState('初始值');
<input value={val} onChange={(e) => setVal(e.target.value)} />

// ⚠️ value={undefined} → 非受控
// ⚠️ value={''} 且无 onChange → 只读输入框
```

## 总结

| 场景 | 推荐方式 |
|------|---------|
| 需要即时验证/格式化 | 受控组件 |
| 需要联动（如省市选择） | 受控组件 |
| 简单表单、快速原型 | 非受控组件 |
| 文件上传 | 非受控组件 |
| 第三方库集成 | 视情况而定 |
