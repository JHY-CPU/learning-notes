# 表单验证与错误处理

## 1. 验证策略

### 1.1 验证时机对比

```jsx
function ValidationTiming() {
  const [email, setEmail] = useState('');
  const [error, setError] = useState('');
  const [touched, setTouched] = useState(false);

  const validate = (val) => {
    if (!val) return '邮箱不能为空';
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val)) return '格式不正确';
    return '';
  };

  // 策略一：onChange 实时验证
  const handleChange = (e) => {
    setEmail(e.target.value);
    if (touched) {
      setError(validate(e.target.value));
    }
  };

  // 策略二：onBlur 失焦验证
  const handleBlur = () => {
    setTouched(true);
    setError(validate(email));
  };

  // 策略三：onSubmit 提交验证
  const handleSubmit = (e) => {
    e.preventDefault();
    setTouched(true);
    const err = validate(email);
    setError(err);
    if (!err) {
      // 提交逻辑
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        value={email}
        onChange={handleChange}
        onBlur={handleBlur}
        className={touched && error ? 'input-error' : ''}
      />
      {touched && error && <span className="error">{error}</span>}
      <button type="submit">提交</button>
    </form>
  );
}
```

### 1.2 验证时机组合推荐

| 场景 | 推荐时机 | 理由 |
|------|---------|------|
| 邮箱/手机号 | onChange + onBlur | 实时格式检查 + 失焦确认 |
| 密码强度 | onChange | 实时反馈强度 |
| 确认密码 | onChange | 与密码实时比较 |
| 用户名重复 | onBlur (异步) | 避免频繁请求服务器 |
| 必填项 | onSubmit | 用户可能还在填写中 |

## 2. 多字段表单验证

### 2.1 统一验证逻辑

```jsx
function RegistrationForm() {
  const [values, setValues] = useState({
    username: '',
    email: '',
    password: '',
    confirmPassword: '',
  });

  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});

  // 验证规则定义
  const validators = {
    username: (val) => {
      if (!val.trim()) return '用户名不能为空';
      if (val.length < 3) return '至少 3 个字符';
      if (val.length > 20) return '不超过 20 个字符';
      if (!/^[a-zA-Z0-9_]+$/.test(val)) return '只能包含字母、数字和下划线';
      return '';
    },
    email: (val) => {
      if (!val.trim()) return '邮箱不能为空';
      if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val)) return '邮箱格式不正确';
      return '';
    },
    password: (val) => {
      if (!val) return '密码不能为空';
      if (val.length < 8) return '至少 8 个字符';
      if (!/[A-Z]/.test(val)) return '需包含大写字母';
      if (!/[a-z]/.test(val)) return '需包含小写字母';
      if (!/[0-9]/.test(val)) return '需包含数字';
      return '';
    },
    confirmPassword: (val, allValues) => {
      if (!val) return '请确认密码';
      if (val !== allValues.password) return '两次密码不一致';
      return '';
    },
  };

  const validateField = (name, value) => {
    const validator = validators[name];
    return validator ? validator(value, values) : '';
  };

  const validateAll = () => {
    const newErrors = {};
    for (const [name, value] of Object.entries(values)) {
      const err = validateField(name, value);
      if (err) newErrors[name] = err;
    }
    return newErrors;
  };

  const handleChange = (e) => {
    const { name, value } = e.target;
    setValues((prev) => ({ ...prev, [name]: value }));
    if (touched[name]) {
      setErrors((prev) => ({ ...prev, [name]: validateField(name, value) }));
    }
  };

  const handleBlur = (e) => {
    const { name } = e.target;
    setTouched((prev) => ({ ...prev, [name]: true }));
    setErrors((prev) => ({ ...prev, [name]: validateField(name, values[name]) }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const allTouched = {};
    for (const key of Object.keys(values)) allTouched[key] = true;
    setTouched(allTouched);

    const newErrors = validateAll();
    setErrors(newErrors);

    if (Object.keys(newErrors).length === 0) {
      console.log('注册成功', values);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      {['username', 'email', 'password', 'confirmPassword'].map((field) => (
        <div key={field}>
          <label>{field}</label>
          <input
            name={field}
            type={field.includes('password') ? 'password' : field === 'email' ? 'email' : 'text'}
            value={values[field]}
            onChange={handleChange}
            onBlur={handleBlur}
          />
          {touched[field] && errors[field] && (
            <span className="error">{errors[field]}</span>
          )}
        </div>
      ))}
      <button type="submit">注册</button>
    </form>
  );
}
```

## 3. 异步验证

### 3.1 用户名重复检查

```jsx
function UsernameInput() {
  const [username, setUsername] = useState('');
  const [status, setStatus] = useState('idle'); // idle | checking | available | taken
  const timeoutRef = useRef(null);

  const checkUsername = async (name) => {
    setStatus('checking');
    try {
      const res = await fetch(`/api/check-username?name=${encodeURIComponent(name)}`);
      const data = await res.json();
      setStatus(data.available ? 'available' : taken');
    } catch {
      setStatus('idle');
    }
  };

  const handleChange = (e) => {
    const value = e.target.value;
    setUsername(value);

    // 防抖：停止输入 500ms 后才发请求
    clearTimeout(timeoutRef.current);
    if (value.length >= 3) {
      timeoutRef.current = setTimeout(() => checkUsername(value), 500);
    } else {
      setStatus('idle');
    }
  };

  const statusMessages = {
    idle: '',
    checking: '检查中...',
    available: '✓ 用户名可用',
    taken: '✗ 用户名已被使用',
  };

  return (
    <div>
      <input value={username} onChange={handleChange} />
      <span className={`status-${status}`}>{statusMessages[status]}</span>
    </div>
  );
}
```

## 4. 错误信息展示

### 4.1 错误信息组件

```jsx
function FieldError({ error, touched }) {
  if (!touched || !error) return null;
  return <span className="field-error" role="alert">{error}</span>;
}

function Field({ label, name, type = 'text', value, onChange, onBlur, error, touched }) {
  return (
    <div className={`field ${touched && error ? 'field-error' : ''}`}>
      <label htmlFor={name}>{label}</label>
      <input
        id={name}
        name={name}
        type={type}
        value={value}
        onChange={onChange}
        onBlur={onBlur}
        aria-invalid={touched && !!error}
        aria-describedby={error ? `${name}-error` : undefined}
      />
      {touched && error && (
        <span id={`${name}-error`} className="error-message" role="alert">
          {error}
        </span>
      )}
    </div>
  );
}
```

### 4.2 错误汇总

```jsx
function ErrorSummary({ errors, touched }) {
  const visibleErrors = Object.entries(errors).filter(
    ([key]) => touched[key] && errors[key]
  );

  if (visibleErrors.length === 0) return null;

  return (
    <div className="error-summary" role="alert">
      <h3>请修正以下错误：</h3>
      <ul>
        {visibleErrors.map(([field, message]) => (
          <li key={field}>
            <a href={`#${field}`}>{message}</a>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

## 5. 使用 Zod 进行声明式验证

```jsx
import { z } from 'zod';

// 定义验证 schema
const registerSchema = z.object({
  username: z
    .string()
    .min(3, '至少 3 个字符')
    .max(20, '不超过 20 个字符')
    .regex(/^[a-zA-Z0-9_]+$/, '只能包含字母、数字和下划线'),
  email: z.string().email('邮箱格式不正确'),
  password: z
    .string()
    .min(8, '至少 8 个字符')
    .regex(/[A-Z]/, '需包含大写字母')
    .regex(/[a-z]/, '需包含小写字母')
    .regex(/[0-9]/, '需包含数字'),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  message: '两次密码不一致',
  path: ['confirmPassword'],
});

function ZodForm() {
  const [values, setValues] = useState({
    username: '',
    email: '',
    password: '',
    confirmPassword: '',
  });
  const [errors, setErrors] = useState({});

  const validateField = (name, value) => {
    const fieldSchema = registerSchema.shape[name];
    if (!fieldSchema) return '';

    const result = fieldSchema.safeParse(value);
    return result.success ? '' : result.error.errors[0].message;
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const result = registerSchema.safeParse(values);

    if (!result.success) {
      const fieldErrors = {};
      result.error.errors.forEach((err) => {
        fieldErrors[err.path[0]] = err.message;
      });
      setErrors(fieldErrors);
    } else {
      console.log('提交成功', result.data);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      {/* 表单字段... */}
    </form>
  );
}
```

## 6. 最佳实践

1. **不要过早验证**：用户刚开始输入就报错是糟糕的体验
2. **区分 touched/untouched**：只在字段被交互过之后才显示错误
3. **异步验证要防抖**：避免每个按键都发请求
4. **错误信息要具体**：不要只说"无效输入"，要说明具体要求
5. **无障碍**：使用 `aria-invalid`、`aria-describedby`、`role="alert"`
6. **优先使用声明式方案**：Zod / Yup 等库比手写验证更可靠
