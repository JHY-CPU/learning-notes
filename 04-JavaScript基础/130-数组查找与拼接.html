数组查找与拼接方法

查找元素位置、提取子数组、拼接数组。

indexOf() / lastIndexOf()

let arr = ['a', 'b', 'c', 'b', 'a']; arr.indexOf('b'); // 1（找到第一个） arr.indexOf('b', 2); // 3（从索引2开始找） arr.indexOf('x'); // -1（找不到） arr.lastIndexOf('b'); // 3（从末尾找）

includes()

let arr = [1, 2, 3, 4, 5]; arr.includes(3); // true arr.includes(6); // false arr.includes(1, 2); // false（从索引2开始找） // 对比 indexOf !== -1 arr.includes(3); // true（更语义化） arr.indexOf(3) !== -1; // true（旧方式）
💡 只检查是否存在用 includes()。

slice(start, end) — 提取子数组

let arr = ['a', 'b', 'c', 'd', 'e']; arr.slice(1, 3); // ['b', 'c']（从1到3，不包括3） arr.slice(2); // ['c', 'd', 'e']（从2到结尾） arr.slice(-2); // ['d', 'e']（最后2个） arr.slice(); // ['a','b','c','d','e']（浅拷贝数组） // ⚠️ slice 不修改原数组！返回新数组

concat() / 展开运算符 — 合并数组

let a = [1, 2]; let b = [3, 4]; let c = [5, 6]; // concat a.concat(b); // [1,2,3,4]（原数组不变） a.concat(b, c); // [1,2,3,4,5,6] a.concat(7, 8); // [1,2,3,4,7,8] // 展开运算符（推荐） [...a, ...b]; // [1,2,3,4] [...a, ...b, ...c]; // [1,2,3,4,5,6] [0, ...a, 3.5, ...b]; // [0,1,2,3.5,3,4]（灵活插入）

join() — 转字符串

let arr = ['apple', 'banana', 'cherry']; arr.join(); // "apple,banana,cherry"（默认逗号） arr.join(', '); // "apple, banana, cherry" arr.join(' - '); // "apple - banana - cherry" arr.join(''); // "applebananacherry" [1, 2, 3].join('-'); // "1-2-3" // 反向操作：split 把字符串变数组 '1-2-3'.split('-'); // ['1', '2', '3']
