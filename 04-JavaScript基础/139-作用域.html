作用域（Scope）

变量在程序中可访问的范围。

全局作用域

// 在函数或块外部声明的变量 let globalVar = '我是全局的'; function test() { console.log(globalVar); // ✅ 可以访问 } // 浏览器中的 window 对象就是全局对象 // var 声明的全局变量会成为 window 的属性 var globalVar2 = 'hello'; console.log(window.globalVar2); // 'hello'

函数作用域

function test() { let x = '函数内'; console.log(x); // ✅ 可以 } // console.log(x); // ❌ 报错！x is not defined // var 是函数作用域 function varTest() { if (true) { var y = 'var无视块'; } console.log(y); // ✅ 可以（var 函数作用域） }

块级作用域 (let / const)

// let 和 const 是块级作用域 if (true) { let a = '块内'; const b = '也是块内'; var c = 'var是函数作用域'; } // console.log(a); // ❌ 报错 // console.log(b); // ❌ 报错 console.log(c); // ✅ 'var是函数作用域' for (let i = 0; i < 3; i++) { // 每次循环 i 都是新的 } // console.log(i); // ❌ 报错

作用域链

// 内部作用域可以访问外部作用域 let outer = '外部'; function middle() { let mid = '中间'; function inner() { let innerVar = '内部'; console.log(outer); // ✅ 向上查找 console.log(mid); // ✅ console.log(innerVar); // ✅ } inner(); // console.log(innerVar); // ❌ 不能访问内部 } // 查找顺序：当前作用域 → 父作用域 → ... → 全局
💡 变量的查找沿作用域链向上，直到找到为止，找不到则报 ReferenceError。

词法作用域（静态作用域）

// JS 是词法作用域：作用域在写代码时就决定了 let value = 'global'; function outer() { let value = 'outer'; function inner() { console.log(value); // 取决于定义时的位置 } inner(); // 'outer' } outer(); // inner 定义在 outer 内，所以 value 是 outer 的 value
