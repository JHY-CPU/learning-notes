递归函数

函数调用自身，把大问题分解为相似的子问题。

递归的要素

// 1. 终止条件（base case）—— 不再递归 // 2. 递归条件 — 调用自身，每次让问题更小 function countdown(n) { // 终止条件 if (n <= 0) { console.log('Go!'); return; } // 递归 console.log(n); countdown(n - 1); } countdown(3); // 3, 2, 1, Go!

阶乘（递归经典）

// 5! = 5 × 4 × 3 × 2 × 1 = 120 // n! = n × (n-1)! function factorial(n) { if (n <= 1) return 1; // 终止条件 return n * factorial(n - 1); // 递归 } factorial(5); // 120 // 执行过程： // factorial(5) → 5 * factorial(4) // → 5 * 4 * factorial(3) // → 5 * 4 * 3 * factorial(2) // → 5 * 4 * 3 * 2 * factorial(1) // → 5 * 4 * 3 * 2 * 1 = 120

斐波那契数列

// 1, 1, 2, 3, 5, 8, 13, 21... // fib(n) = fib(n-1) + fib(n-2) function fib(n) { if (n <= 2) return 1; // 前两个是1 return fib(n - 1) + fib(n - 2); } fib(6); // 8 // ⚠️ 有大量重复计算，n=40 就很慢了
💡 斐波那契用递归效率低，因为有大量重复计算。可以用记忆化优化。

递归遍历树形数据

// 遍历嵌套菜单/评论等树形结构 const menu = { name: '根', children: [ { name: 'HTML', children: [ { name: '标签' }, { name: '属性' } ]}, { name: 'CSS', children: [ { name: '选择器' }, { name: '盒模型' } ]} ] }; function printTree(node, indent = 0) { console.log(' '.repeat(indent) + node.name); if (node.children) { node.children.forEach(child => printTree(child, indent + 1)); } } printTree(menu); // 根 // HTML // 标签 // 属性 // CSS // 选择器 // 盒模型

递归 vs 循环

// 凡是递归能解决的问题，循环也能解决 // 循环版本阶乘 function factorialLoop(n) { let result = 1; for (let i = 2; i <= n; i++) { result *= i; } return result; } // 递归优势： // - 代码更简洁（树形结构） // - 更接近数学定义 // 递归劣势： // - 性能略差（函数调用开销） // - 有调用栈限制（栈溢出）
