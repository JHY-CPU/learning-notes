class 基础

ES6 引入的 class 语法糖，本质还是基于原型的继承。

class 声明与基本语法

// class 声明 class Person { // 构造函数 constructor(name, age) { this.name = name; this.age = age; } // 实例方法（在 prototype 上） sayHi() { console.log('Hi, 我是' + this.name); } // getter get info() { return `${this.name}(${this.age}岁)`; } // setter set info(val) { [this.name, this.age] = val.split(' '); } } const p = new Person('Alice', 25); p.sayHi(); // 'Hi, 我是Alice' console.log(p.info); // 'Alice(25岁)' // class 本质是函数 typeof Person; // 'function' Person.prototype.sayHi; // 方法在 prototype 上 Person.prototype.constructor === Person; // true // ⚠️ class 不会提升（hoisting） // new Foo(); // ❌ ReferenceError // class Foo {} // ⚠️ class 中的代码自动是严格模式

class 与构造函数的对比

// 构造函数方式（ES5） function Person5(name) { this.name = name; } Person5.prototype.sayHi = function() { console.log('Hi, 我是' + this.name); }; // class 方式（ES6，语法糖） class Person6 { constructor(name) { this.name = name; } sayHi() { console.log('Hi, 我是' + this.name); } } // 本质上是一样的 const p5 = new Person5('Alice'); const p6 = new Person6('Bob'); console.log(typeof Person5); // 'function' console.log(typeof Person6); // 'function' console.log(p6 instanceof Person6); // true // class 特有的限制： // 1. 必须用 new 调用 // Person6(); // ❌ TypeError: Class constructor Person6 cannot be invoked without 'new' // 2. class 不可枚举 Object.keys(Person.prototype); // [] Object.keys(Person5.prototype); // ['sayHi']
💡 class 只是语法糖，让原型继承的写法更像传统面向对象语言。

class 表达式

// 匿名 class 表达式 const Person = class { constructor(name) { this.name = name; } }; // 具名 class 表达式 const Person2 = class MyClass { constructor(name) { this.name = name; } // MyClass 只在 class 内部可用 getClassName() { return MyClass.name; } }; const p = new Person2('Alice'); p.getClassName(); // 'MyClass' // MyClass.name; // ❌ ReferenceError（外部不可访问） // 立即执行的 class const singleton = new class { constructor() { this.name = '单例'; } }(); console.log(singleton.name); // '单例'

实例属性与访问器

// 实例属性可以写在 constructor 里 // 也可以直接写在类体中（ES2022+） class Person { // 公有实例属性（ES2022） name = '匿名'; age = 0; city = '未知'; constructor(name, age) { if (name) this.name = name; if (age) this.age = age; } // 访问器（getter/setter） get birthYear() { return new Date().getFullYear() - this.age; } // 计算属性名 ['say_' + 'hello']() { console.log('Hello!'); } } const p = new Person('Alice', 25); console.log(p.name); // 'Alice' console.log(p.birthYear); // 2001 // 访问器优点：可以加验证逻辑 class User { constructor(email) { this._email = email; } get email() { return this._email; } set email(val) { if (!val.includes('@')) throw '无效邮箱'; this._email = val; } }
