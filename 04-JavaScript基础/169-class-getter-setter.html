class getter/setter

用 get/set 定义访问器属性，在读写时执行逻辑。

getter 基本用法

class Person { constructor(firstName, lastName) { this.firstName = firstName; this.lastName = lastName; } // getter — 像属性一样读取 get fullName() { return this.firstName + ' ' + this.lastName; } get initials() { return this.firstName[0] + '.' + this.lastName[0] + '.'; } } const p = new Person('Alice', 'Wang'); console.log(p.fullName); // 'Alice Wang'（像属性一样，不是方法） console.log(p.initials); // 'A.W.' // ⚠️ 不加括号！ // p.fullName() // ❌ TypeError（getter 不是函数） // ⚠️ 不能手动赋值（如果没有 setter） // p.fullName = 'Bob Li'; // 静默失败（严格模式报错）

setter 基本用法

class Person { constructor(firstName, lastName) { this.firstName = firstName; this.lastName = lastName; } get fullName() { return this.firstName + ' ' + this.lastName; } set fullName(val) { const parts = val.split(' '); this.firstName = parts[0]; this.lastName = parts.slice(1).join(' '); } } const p = new Person('Alice', 'Wang'); console.log(p.fullName); // 'Alice Wang' p.fullName = 'Bob Li'; console.log(p.firstName); // 'Bob' console.log(p.lastName); // 'Li' // 读写看起来和普通属性一样 // 但背后执行了额外的逻辑

验证逻辑

class User { constructor(email) { this._email = email; } get email() { return this._email; } set email(val) { if (!val || !val.includes('@')) { throw new Error('邮箱必须包含 @'); } if (!val.includes('.')) { throw new Error('邮箱必须包含域名'); } this._email = val.toLowerCase(); } } const user = new User('alice@example.com'); console.log(user.email); // 'alice@example.com' // user.email = 'invalid'; // ❌ Error: 邮箱必须包含 @ // user.email = 'test@test'; // ❌ Error: 邮箱必须包含域名 user.email = 'BOB@EXAMPLE.COM'; console.log(user.email); // 'bob@example.com'（自动转小写） // 使用 _ 前缀表示"内部属性"，但这不是真正的私有 // ES2022 可以用 # 私有字段 + getter/setter 实现真正的私有

计算属性与只读属性

class Rectangle { constructor(width, height) { this.width = width; this.height = height; } // 只读 getter（没有 setter） get area() { return this.width * this.height; } get perimeter() { return 2 * (this.width + this.height); } // 计算属性名 get ['type']() { if (this.width === this.height) return '正方形'; return '长方形'; } // 依赖多个属性的计算 get aspectRatio() { return this.width / this.height; } get landscape() { return this.width >= this.height; } } const rect = new Rectangle(10, 5); console.log(rect.area); // 50 console.log(rect.perimeter); // 30 console.log(rect.type); // '长方形' console.log(rect.landscape); // true // 只读属性不能赋值 // rect.area = 100; // 静默失败（严格模式报错） // 计算属性 setter class Circle { constructor(radius) { this._radius = radius; } get radius() { return this._radius; } set radius(val) { this._radius = val; } get area() { return Math.PI * this._radius ** 2; } // set area(val) { // this._radius = Math.sqrt(val / Math.PI); // 反向计算 // } }

getter/setter 在对象字面量中使用

// 不只在 class 中能用，对象字面量也可以 const user = { _name: 'Alice', _age: 25, get name() { return this._name; }, set name(val) { if (!val) throw '名字不能为空'; this._name = val; }, get age() { return this._age; }, set age(val) { if (val < 0 || val > 150) throw '年龄不合法'; this._age = val; }, get isAdult() { return this._age >= 18; } }; console.log(user.name); // 'Alice' console.log(user.isAdult); // true user.age = 30; console.log(user.age); // 30 // defineProperty 方式 const obj = {}; Object.defineProperty(obj, 'data', { get() { return this._data; }, set(val) { this._data = val.toUpperCase(); } }); obj.data = 'hello'; console.log(obj.data); // 'HELLO'
