事件流与事件委托

事件捕获 → 目标 → 冒泡。利用冒泡实现事件委托。

事件流三阶段

// DOM 事件流分为三个阶段： // 1. 捕获阶段（Capture Phase） // 从 window → document → html → body → ... → 目标父元素 // 事件从外层向内层传播 // 2. 目标阶段（Target Phase） // 事件到达目标元素本身 // 3. 冒泡阶段（Bubble Phase） // 从目标元素 → 父元素 → ... → body → html → document → window // 事件从内层向外层传播 // ┌─── 捕获 ──→ window ←── 冒泡 ──┐ // │ document │ // │ html │ // │ body │ // │ parent │ // │ child │ // └─── target ─────────┘ // 默认情况下，事件在冒泡阶段触发 // addEventListener 的第三个参数指定阶段： // false（默认）→ 冒泡阶段触发 // true → 捕获阶段触发

stopPropagation — 阻止传播

// stopPropagation() — 阻止事件继续传播 // 在捕获阶段调用 → 阻止后续捕获和目标 // 在冒泡阶段调用 → 阻止后续冒泡 child.addEventListener('click', function(e) { e.stopPropagation(); // 父元素不会收到事件 console.log('子元素点击（不会冒泡）'); }); // stopImmediatePropagation() — 立即阻止所有传播 // 1. 阻止事件继续传播（同 stopPropagation） // 2. 阻止当前元素上后续事件处理函数执行 element.addEventListener('click', function(e) { e.stopImmediatePropagation(); console.log('第一个处理器'); }); element.addEventListener('click', function() { console.log('第二个处理器（不会执行）'); }); // 只输出 '第一个处理器'

事件委托（Event Delegation）

// 利用事件冒泡，在父元素上监听事件 // 而不是给每个子元素单独绑定 // ❌ 传统方式：给每个 li 绑定 document.querySelectorAll('li').forEach(li => { li.addEventListener('click', () => { console.log('点击了', li.textContent); }); }); // 问题：新增 li 不会自动绑定 // 问题：大量 li 时性能开销大 // ✅ 事件委托：在父元素上绑定一次 document.querySelector('ul').addEventListener('click', function(e) { const target = e.target.closest('li'); // 找到 li if (!target) return; // 没点到 li 就忽略 console.log('点击了', target.textContent); }); // 好处： // 1. 动态元素也能处理（新增 li 无需再绑定） // 2. 只需要一个监听器（省内存） // 3. 统一管理 // 常用委托模式： // table > td → 在 table 上监听 // ul > li → 在 ul 上监听 // div.items > div.item → 在 .items 上监听

事件委托示例

// 完整事件委托示例 const list = document.getElementById('task-list'); // 在父元素上绑定一次 list.addEventListener('click', function(e) { const item = e.target.closest('.task-item'); if (!item) return; // 根据不同子元素做不同操作 if (e.target.classList.contains('delete-btn')) { item.remove(); console.log('删除:', item.textContent); } else if (e.target.classList.contains('edit-btn')) { console.log('编辑:', item.textContent); } else { // 点击任务本身 item.classList.toggle('completed'); console.log('切换完成状态:', item.textContent); } }); // 动态添加的新项目自动生效 list.appendChild(createTaskItem('新任务')); // ✅ 不用再绑定 // 数据属性方案： // <li data-action="delete">...</li> list.addEventListener('click', function(e) { const action = e.target.closest('[data-action]')?.dataset.action; if (action === 'delete') { /* 删除 */ } if (action === 'edit') { /* 编辑 */ } });
