DOM 属性操作

getAttribute / setAttribute / removeAttribute / dataset。

getAttribute / setAttribute

// 获取属性值 const type = element.getAttribute('type'); const href = link.getAttribute('href'); const cls = element.getAttribute('class'); // 设置属性值 element.setAttribute('class', 'active'); element.setAttribute('data-id', '123'); element.setAttribute('disabled', ''); // 布尔属性 element.setAttribute('style', 'color:red'); // 同时设置多个属性 element.setAttribute('id', 'main'); element.setAttribute('title', '提示文字'); // ⚠️ 属性名不区分大小写（HTML 属性） // ⚠️ 自定义属性建议用 data- 前缀 // ⚠️ getAttribute('class') vs .className // getAttribute('class') → 返回属性值 // .className → 返回字符串 // ⚠️ getAttribute('href') vs .href // .href → 返回完整 URL // getAttribute('href') → 返回属性值原样

removeAttribute / hasAttribute

// 删除属性 element.removeAttribute('disabled'); element.removeAttribute('data-temp'); // 检查属性是否存在 if (element.hasAttribute('disabled')) { console.log('元素被禁用'); } if (element.hasAttribute('data-id')) { const id = element.getAttribute('data-id'); } // 获取所有属性 const attrs = element.attributes; // NamedNodeMap for (let attr of attrs) { console.log(attr.name, '=', attr.value); } // 布尔属性 element.setAttribute('checked', ''); // 设置（空值） element.setAttribute('checked', 'checked'); // 也可 element.removeAttribute('checked'); // 移除 // 常见布尔属性：disabled, checked, readonly, required, hidden

元素属性（Property）与 HTML 属性（Attribute）

// HTML 属性（Attribute）— HTML 中写的 // DOM 属性（Property）— 元素对象的属性 // 大多数情况下，两者自动同步 input.setAttribute('value', 'hello'); console.log(input.value); // 'hello'（同步到 property） input.value = 'world'; console.log(input.getAttribute('value')); // 'hello'（⚠️ 不同步回去！） // 例外：input 的 value // 用户输入 → property 变化 // setAttribute → property 变化 // property 变化 → attribute 不变！ // class vs className // HTML: class="foo" // JS: element.className = 'foo' // getAttribute('class') 和 .className 通常是同步的 // style（对象 vs 字符串） element.style.color = 'red'; // CSSStyleDeclaration 对象 element.setAttribute('style', 'color:red'); // 字符串 // 推荐： // 标准属性 → .property 方式 // 自定义属性 → getAttribute / setAttribute // data-* 属性 → dataset

dataset — data-* 属性

// HTML5 自定义数据属性 data-* // <div data-id="123" data-user-name="Alice"> // JS 中通过 dataset 访问 element.dataset.id; // '123' element.dataset.userName; // 'Alice'（驼峰命名） // 设置 element.dataset.id = '456'; element.dataset.userName = 'Bob'; element.dataset.newField = '新字段'; // data-new-field // 删除 delete element.dataset.temp; // ⚠️ dataset 是 DOMStringMap 对象 // ⚠️ data-abc-def-ghi → dataset.abcDefGhi // ⚠️ data- 后面的连字符转驼峰 // 用途：在 DOM 元素上存储数据 // 代替 <div data-id="123"> // 取数据时不用再解析 // 常见 data 属性： // data-id, data-index, data-type, data-value // data-toggle, data-target, data-action
