数组解构赋值

从数组中提取值并赋给变量。

基本用法

// 旧方式 let arr = [1, 2, 3]; let a = arr[0]; let b = arr[1]; let c = arr[2]; // 解构赋值 let [x, y, z] = [1, 2, 3]; console.log(x, y, z); // 1, 2, 3 // 只取部分 let [first, , third] = ['a', 'b', 'c']; console.log(first, third); // 'a', 'c' // 剩余模式 let [head, ...tail] = [1, 2, 3, 4]; console.log(head); // 1 console.log(tail); // [2, 3, 4]

默认值

// 解构的值是 undefined 时使用默认值 let [a = 10, b = 20] = [1]; console.log(a, b); // 1, 20 // 只有 undefined 才触发默认值（null 不触发） let [x = 'default'] = [undefined]; // 'default' let [y = 'default'] = [null]; // null let [z = 'default'] = []; // 'default'

交换变量

// 旧方式需要临时变量 let a = 1, b = 2; let tmp = a; a = b; b = tmp; // 解构交换一行搞定 [a, b] = [b, a]; console.log(a, b); // 2, 1

函数返回多个值

function getMinMax(arr) { return [Math.min(...arr), Math.max(...arr)]; } const [min, max] = getMinMax([3, 1, 4, 1, 5]); console.log(min, max); // 1, 5 // 忽略部分返回值 const [first] = getMinMax([3, 1, 4]); console.log(first); // 1

嵌套数组解构

const [a, [b, c]] = [1, [2, 3]]; console.log(a, b, c); // 1, 2, 3 // 复杂嵌套 const data = [1, [2, [3, 4]], 5]; const [x, [, [y, z]]] = data; console.log(x, y, z); // 1, 3, 4
