选择器 — querySelector / querySelectorAll

使用 CSS 选择器语法查找元素，更灵活强大。

querySelector

// 返回匹配 CSS 选择器的第一个元素 // 没找到返回 null const firstItem = document.querySelector('.item'); const mainTitle = document.querySelector('#title'); const firstDiv = document.querySelector('div'); const special = document.querySelector('[data-type="highlight"]'); // 复杂 CSS 选择器 const child = document.querySelector('ul > li:first-child'); const nth = document.querySelector('div:nth-of-type(2)'); const attr = document.querySelector('input[type="text"]'); const combo = document.querySelector('div.card > h2 + p'); // 在指定元素范围内查找 const container = document.querySelector('.container'); const innerItem = container.querySelector('.item'); // ⚠️ 只返回第一个匹配 // 如果要全部匹配，用 querySelectorAll

querySelectorAll

// 返回所有匹配 CSS 选择器的元素 // 返回 NodeList（静态集合） const allItems = document.querySelectorAll('.item'); const allButtons = document.querySelectorAll('button'); const allInputs = document.querySelectorAll('input, textarea, select'); const allSpecial = document.querySelectorAll('[data-type]'); // NodeList 有 forEach！ allItems.forEach(item => { console.log(item.textContent); }); // 也可以转数组 const arr = [...allItems]; const arr2 = Array.from(allItems); // 获取元素内的特定子元素 const container2 = document.getElementById('container'); const children = container2.querySelectorAll(':scope > .direct-child'); // :scope 表示当前元素（container2），不加的话规则是从全局匹配

CSS 选择器示例

// 基本选择器 querySelector('*') // 所有元素 querySelector('div') // 标签选择器 querySelector('.class') // 类选择器 querySelector('#id') // ID 选择器 querySelector('[data-key]') // 属性存在 querySelector('[type="text"]') // 属性等于 // 组合选择器 querySelector('div.item') // div 且 class=item querySelector('ul > li') // 直接子元素 querySelector('div p') // 后代元素 querySelector('h1 + p') // 紧邻兄弟 querySelector('h1 ~ p') // 所有后续兄弟 // 伪类 querySelector(':first-child') querySelector(':last-child') querySelector(':nth-child(2)') querySelector(':nth-of-type(odd)') querySelector(':not(.hidden)') querySelector(':empty') querySelector(':checked') querySelector(':disabled') // 伪元素（querySelector 可以，但不会返回元素） // querySelector('::before') // 返回 null

querySelector vs getElement 对比

// querySelector / querySelectorAll // ✅ 使用 CSS 选择器，灵活强大 // ✅ NodeList 有 forEach // ✅ #id, .class 不需要记不同方法 // ❌ NodeList 是静态的（不会自动更新） // ❌ 性能略慢（解析 CSS 选择器） // getElementById / getElementsByClassName / getElementsByTagName // ✅ 性能更快 // ✅ HTMLCollection 是动态的（自动更新） // ❌ 方法名多，需要记 // ❌ HTMLCollection 没有 forEach // ❌ 功能有限（不能复杂选择） // 现代开发推荐： // - querySelector(id) 替代 getElementById // - querySelectorAll 用于复杂选择 // - getElementsByTagName 用于需要动态集合的场景 // - 性能差异通常可忽略，优先用 querySelector
