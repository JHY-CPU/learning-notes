元素尺寸与位置

offset / client / scroll / getBoundingClientRect。

offset 系列 — 元素自身尺寸与偏移

// offsetWidth — 元素总宽度（content + padding + border + scrollbar） // offsetHeight — 元素总高度 const width = element.offsetWidth; const height = element.offsetHeight; // offsetLeft — 元素左上角相对于 offsetParent 的水平偏移 // offsetTop — 元素左上角相对于 offsetParent 的垂直偏移 const left = element.offsetLeft; const top = element.offsetTop; // offsetParent — 最近的已定位祖先元素（position ≠ static） // 如果无定位祖先，则为 body const parent = element.offsetParent; // 图示： // ┌────── offsetParent ──────┐ // │ ... │ // │ ┌──── element ─────┐ │ // │ │ offsetLeft → │ │ // │ │ offsetTop ↓ │ │ // │ │ ┌─content─┐ │ │ // │ │ │ │ │ │ // │ │ └─────────┘ │ │ // │ │←──offsetWidth──→│ │ // │ └─────────────────┘ │ // └──────────────────────────┘

client 系列 — 内容可见区域

// clientWidth — 可见内容宽度（content + padding，不含 border 和 scrollbar） // clientHeight — 可见内容高度 const visibleWidth = element.clientWidth; const visibleHeight = element.clientHeight; // clientLeft — 左边框宽度 // clientTop — 上边框宽度 const borderLeft = element.clientLeft; const borderTop = element.clientTop; // 对于 document.documentElement（<html>） // document.documentElement.clientWidth // 视口宽度（不含滚动条） // document.documentElement.clientHeight // 视口高度（不含滚动条） // ⚠️ clientWidth/Height 不包含滚动条 // ⚠️ offsetWidth/Height 包含滚动条

scroll 系列 — 滚动尺寸

// scrollWidth — 内容总宽度（包含溢出部分） // scrollHeight — 内容总高度 const totalWidth = element.scrollWidth; const totalHeight = element.scrollHeight; // scrollLeft — 水平滚动距离（可读写） // scrollTop — 垂直滚动距离（可读写） element.scrollLeft = 0; element.scrollTop = 100; // 判断是否滚动到底部 const isBottom = element.scrollHeight - element.scrollTop - element.clientHeight < 1; // 判断是否有滚动条 const hasScroll = element.scrollHeight > element.clientHeight; // 滚动百分比 const scrollPercent = element.scrollTop / (element.scrollHeight - element.clientHeight);

getBoundingClientRect — 元素视口坐标

// getBoundingClientRect() — 返回元素的视口坐标 // 返回 DOMRect 对象（相对于视口，不是文档） const rect = element.getBoundingClientRect(); // 返回对象： rect.x // 元素左边缘的 X 坐标 rect.y // 元素上边缘的 Y 坐标 rect.width // 元素宽度（含 border） rect.height // 元素高度（含 border） rect.top // 元素上边缘的 Y 坐标（同 y） rect.right // 元素右边缘的 X 坐标（x + width） rect.bottom // 元素下边缘的 Y 坐标（y + height） rect.left // 元素左边缘的 X 坐标（同 x） // 相对文档的坐标（+ 滚动距离） const docX = rect.left + window.scrollX; const docY = rect.top + window.scrollY; // 判断元素是否在视口内 function isInViewport(el) { const rect = el.getBoundingClientRect(); return ( rect.top < window.innerHeight && rect.bottom > 0 && rect.left < window.innerWidth && rect.right > 0 ); } // getClientRects() — 获取文本的每一行矩形 const textRects = element.getClientRects(); // 常用于获取文本行的精确位置
