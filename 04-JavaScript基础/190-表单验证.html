表单验证

HTML5 内置验证 + JavaScript 自定义验证。

HTML5 内置验证

// HTML5 为 input 提供了多种验证属性： // 必填 // <input required> // 类型 // <input type="email"> // <input type="url"> // <input type="number"> // <input type="tel"> // 长度限制 // <input minlength="3" maxlength="20"> // 数值范围 // <input type="number" min="0" max="100"> // 模式匹配 // <input pattern="[A-Za-z]+"> // ⚠️ HTML5 验证在 submit 时自动触发 // ⚠️ 不满足约束时会显示浏览器默认提示 // ⚠️ 可以通过 novalidate 属性禁用

验证 API

// checkValidity() — 检查元素是否通过验证 const isValid = input.checkValidity(); // 返回 true/false，不合法时触发 invalid 事件 // reportValidity() — 检查并显示错误提示 input.reportValidity(); // validationMessage — 获取错误消息 console.log(input.validationMessage); // validity 对象（ValidityState） input.validity.valueMissing // 必填但为空 input.validity.typeMismatch // 类型不匹配（email/url） input.validity.patternMismatch // 模式不匹配 input.validity.tooLong // 太长 input.validity.tooShort // 太短 input.validity.rangeUnderflow // 小于最小值 input.validity.rangeOverflow // 大于最大值 input.validity.stepMismatch // step 不匹配 input.validity.badInput // 输入格式错 input.validity.valid // 是否有效（true/false） // setCustomValidity() — 自定义错误消息 input.setCustomValidity('自定义错误消息'); // 设置非空字符串 → 验证不通过 // 设置空字符串 → 清除自定义错误 // invalid 事件 input.addEventListener('invalid', function() { console.log('验证失败:', this.validationMessage); });

实时验证模式

// 实时验证 — 用户输入时立即检查 input.addEventListener('input', function() { if (this.validity.valid) { this.classList.remove('error'); this.classList.add('valid'); errorMsg.textContent = ''; } else { this.classList.remove('valid'); this.classList.add('error'); if (this.validity.valueMissing) { errorMsg.textContent = '此项不能为空'; } else if (this.validity.patternMismatch) { errorMsg.textContent = '格式不正确'; } } }); // 失去焦点时验证 input.addEventListener('blur', function() { if (this.value && !this.validity.valid) { // 显示错误 } }); // 提交时全面验证 form.addEventListener('submit', function(e) { e.preventDefault(); if (!this.checkValidity()) { // 有错误，显示所有错误 } else { // 全部通过，提交 } }); // ⚠️ 注意：用 novalidate 禁用 HTML5 默认气泡 // 改用自定义错误显示
