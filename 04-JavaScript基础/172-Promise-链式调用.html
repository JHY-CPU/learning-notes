Promise 链式调用

每个 .then 返回新 Promise，可以无限链式调用。

链式调用基础

// .then() 返回一个新的 Promise // 所以可以一直 .then().then().then() Promise.resolve(1) .then(v => { console.log('第1步:', v); return v + 1; // 返回普通值 }) .then(v => { console.log('第2步:', v); return v + 1; }) .then(v => { console.log('第3步:', v); return v + 1; }); // 输出： // 第1步: 1 // 第2步: 2 // 第3步: 3 // .then() 的返回值决定下一个 .then() 接收的值

then 的返回值

// then 的回调可以返回3种类型： // 1. 返回普通值 → 下一个 then 接收该值 Promise.resolve(1) .then(v => v * 2) // 返回 2 .then(v => console.log(v)); // 2 // 2. 返回 Promise → 等待 Promise 完成，then 接收结果 Promise.resolve(1) .then(v => { return new Promise(resolve => { setTimeout(() => resolve(v * 10), 500); }); }) .then(v => console.log(v)); // 10（500ms后） // 3. 抛出错误 → 跳到 catch Promise.resolve(1) .then(v => { throw new Error('出错了'); }) .catch(e => console.log(e.message)); // '出错了' // 4. 返回 thenable 对象 → 自动展开 Promise.resolve(1) .then(v => ({ then(resolve) { resolve(v + 100); } })) .then(v => console.log(v)); // 101

错误传播

// 错误会跳过中间的 then，直达最近的 catch Promise.resolve(1) .then(v => { console.log('第1步:', v); return v + 1; }) .then(v => { console.log('第2步:', v); throw new Error('第2步出错'); }) .then(v => { // 这个不会执行 console.log('第3步:', v); }) .catch(e => { console.log('捕获错误:', e.message); // '第2步出错' }) .then(v => { // catch 之后可以继续链式调用 console.log('catch 之后的 then, v =', v); // undefined（catch 没有 return） }); // catch 也可以返回值，相当于恢复错误 Promise.resolve(1) .then(() => { throw new Error('失败了'); }) .catch(e => { console.log('错误:', e.message); return '恢复值'; // 错误恢复了 }) .then(v => console.log('恢复后:', v)); // '恢复后: 恢复值'

异步操作串行

// Promise 链的核心用法：串行异步操作 const delay = ms => new Promise(r => setTimeout(r, ms)); // 模拟：用户 → 订单 → 物流信息 function getUser(id) { return delay(300).then(() => ({ id, name: 'Alice' })); } function getOrders(userId) { return delay(300).then(() => ([ { id: 1, product: '手机' }, { id: 2, product: '电脑' } ])); } function getTracking(orderId) { return delay(300).then(() => ({ orderId, status: '已发货' })); } // 链式调用：每个异步操作依赖上一步的结果 getUser(1) .then(user => { console.log('用户:', user.name); return getOrders(user.id); }) .then(orders => { console.log('订单数:', orders.length); return getTracking(orders[0].id); }) .then(tracking => { console.log('物流状态:', tracking.status); }) .catch(err => console.error('错误:', err)); // 扁平化：避免嵌套回调（对比回调地狱）
💡 链式调用将嵌套的回调地狱（callback hell）变成了扁平的链式结构。

链式调用的常见误区

// ❌ 误区1：嵌套而不是链式 Promise.resolve(1).then(v1 => { Promise.resolve(v1 + 1).then(v2 => { console.log('嵌套:', v2); // 能工作，但嵌套了 }); }); // 应该用链式： Promise.resolve(1) .then(v1 => v1 + 1) .then(v2 => console.log('链式:', v2)); // ❌ 误区2：在 then 中创建 Promise 但不返回 Promise.resolve(1) .then(v => { new Promise(r => r(v + 1)); // 没 return！ }) .then(v => console.log('没返回:', v)); // undefined // ✅ 正确：返回 Promise Promise.resolve(1) .then(v => { return new Promise(r => r(v + 1)); }) .then(v => console.log('正确:', v)); // 2 // ❌ 误区3：忘记 catch 错误 // 链式调用末尾建议加 catch // ✅ 正确：finally 做清理 Promise.resolve(1) .then(v => v * 2) .then(v => console.log(v)) .catch(e => console.log(e)) .finally(() => console.log('清理'));
