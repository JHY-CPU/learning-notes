函数参数进阶

默认参数、剩余参数、arguments 对象。

默认参数（Default Parameters）

// ES6 之前：用 || 设置默认值 function greet1(name) { name = name || '游客'; console.log('你好, ' + name); } // ES6 默认参数 function greet2(name = '游客') { console.log(`你好, ${name}`); } greet2(); // "你好, 游客" greet2('Alice'); // "你好, Alice" // 多个默认参数 function createUser(name = '匿名', age = 18, role = 'user') { return { name, age, role }; }

默认参数注意事项

// 带默认值的参数在 undefined 时触发 function log(x = 'default') { console.log(x); } log(undefined); // 'default' ✓ log(null); // null（不是 undefined，不触发默认值） log(''); // ''（空字符串，不触发默认值） log(0); // 0 // 默认值可以是表达式，只在使用时求值 function getDefault() { console.log('只会在需要时执行'); return 42; } function f(x = getDefault()) { } // 默认参数可以引用前面的参数 function f2(a, b = a + 1) { return b; } f2(5); // 6

剩余参数（Rest Parameters）

// ...args 收集剩余参数为数组 function sum(...numbers) { return numbers.reduce((a, c) => a + c, 0); } sum(1, 2, 3, 4); // 10 // 剩余参数必须是最后一个 function logAll(first, ...rest) { console.log('第一个:', first); console.log('其余:', rest); } logAll('a', 'b', 'c', 'd'); // 第一个: a, 其余: ['b', 'c', 'd']

arguments 对象（旧方式）

// arguments 是类数组对象（不是真正的数组） function oldSum() { console.log(arguments); // [1, 2, 3] console.log(arguments.length); // 3 console.log(arguments[0]); // 1 // arguments 没有数组方法！ // arguments.map(...) // ❌ 报错 } oldSum(1, 2, 3); // 转成数组 function convert() { const args = Array.from(arguments); // 或 const args = [...arguments]; return args.map(x => x * 2); }
💡 推荐用 rest 参数 ...args 代替 arguments（箭头函数也没有 arguments）。
