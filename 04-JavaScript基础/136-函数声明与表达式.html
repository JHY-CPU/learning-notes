函数声明与函数表达式

定义函数的两种主要方式。

函数声明（Function Declaration）

function greet(name) { return `你好, ${name}!`; } // 特点：会提升（hoisting），可以在定义前调用 sayHi(); // ✅ 可以 function sayHi() { console.log('Hi!'); }

函数表达式（Function Expression）

// 匿名函数表达式 const greet = function(name) { return `你好, ${name}!`; }; // 命名函数表达式 const factorial = function calc(n) { return n <= 1 ? 1 : n * calc(n - 1); }; // 特点：不会提升，必须先赋值后调用 // hello(); // ❌ 报错！ const hello = function() { console.log('Hello'); };

函数声明 vs 表达式对比

// 函数声明：提升（hoisted） foo(); // ✅ 可以 function foo() { return 'foo'; } // 函数表达式：不提升 // bar(); // ❌ 报错！bar is not a function const bar = function() { return 'bar'; }; // var 函数表达式：变量提升，但值为 undefined // baz(); // ❌ 报错！baz is not a function（不是函数） var baz = function() { return 'baz'; }; // 相当于： // var baz; // baz(); // undefined() → TypeError

函数是对象

function add(a, b) { return a + b; } add.name; // "add"（函数名） add.length; // 2（形参个数） // 函数可以赋值给变量 const fn = add; // 函数可以作为参数 function exec(fn, a, b) { return fn(a, b); } exec(add, 1, 2); // 3 // 函数可以作为返回值 function makeAdder(x) { return function(y) { return x + y; }; }
