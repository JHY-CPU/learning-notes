字符串方法(二) — 截取与转换

截取子串、拆分拼接、替换内容。

slice(start, end) — 截取子串

let str = 'JavaScript'; str.slice(0, 4); // "Java"（从0到4，不包括4） str.slice(4); // "Script"（从4到结尾） str.slice(-6); // "Script"（负索引：从倒数第6到结尾） str.slice(0, -4); // "JavaSc"（从0到倒数第4） // 参数：start（必选）和 end（可选，不包含end） // 支持负数索引

substring(start, end) — 类似slice但不支持负数

let str = 'JavaScript'; str.substring(0, 4); // "Java" str.substring(4); // "Script" str.substring(4, 0); // "Java"（自动交换参数！） str.substring(-3); // "JavaScript"（负数转为0） // 对比 slice： // slice(4, 0) → ''（空字符串，不交换参数） // substring 会把负数当0处理

substr(start, length) — 已废弃

// ⚠️ 已废弃，不建议使用 let str = 'JavaScript'; str.substr(4, 3); // "Scr"（从索引4开始取3个字符） str.substr(-6, 3); // "Scr" // 建议用 slice() 代替

split(separator) — 拆分为数组

'apple,banana,cherry'.split(','); // ["apple", "banana", "cherry"] 'hello world'.split(' '); // ["hello", "world"] 'hello'.split(''); // ["h", "e", "l", "l", "o"] '1-2-3'.split('-', 2); // ["1", "2"]（限制数量）

replace(search, replacement) — 替换

let str = 'Hello World'; str.replace('World', 'JS'); // "Hello JS"（只替换第一个） str.replace('world', 'JS'); // "Hello World"（找不到，不替换） // 正则替换（全局） str.replace(/o/g, '0'); // "Hell0 W0rld" str.replace(/l/gi, 'L'); // "HeLLo WorLd"（全字不区分大小写） // replaceAll（ES2021） str.replaceAll('l', 'L'); // "HeLLo WorLd"（替换所有匹配）

concat() — 拼接（不常用）

// 不如直接用 + 'Hello'.concat(' ', 'World'); // "Hello World" 'Hello' + ' ' + 'World'; // "Hello World"（更简洁）
