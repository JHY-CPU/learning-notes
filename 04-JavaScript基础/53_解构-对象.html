对象解构赋值

从对象中提取属性并赋给变量。

基本用法

const user = { name: 'Alice', age: 25, email: 'alice@example.com' }; // 变量名必须和属性名一致 const { name, age } = user; console.log(name, age); // 'Alice', 25 // 只取需要的属性 const { email } = user; console.log(email); // 'alice@example.com'

重命名

const user = { name: 'Alice', age: 25 }; // 属性名: 新变量名 const { name: userName, age: userAge } = user; console.log(userName, userAge); // 'Alice', 25 // console.log(name); // ❌ 报错（name 不存在，被重命名了）

默认值

const user = { name: 'Alice' }; // 属性不存在时使用默认值 const { name, age = 18, role = 'user' } = user; console.log(name, age, role); // 'Alice', 18, 'user' // 重命名 + 默认值 const { name: userName = '匿名' } = {}; console.log(userName); // '匿名'

嵌套解构

const user = { name: 'Alice', address: { city: 'Beijing', district: 'Haidian' }, scores: { math: 90, english: 85 } }; // 解构嵌套对象 const { address: { city, district } } = user; console.log(city, district); // 'Beijing', 'Haidian' // 重命名嵌套属性 const { scores: { math: mathScore } } = user; console.log(mathScore); // 90

函数参数解构

// 十分常用！ function greet({ name, age }) { console.log(`我叫${name}，${age}岁`); } greet({ name: 'Alice', age: 25 }); // 默认值 + 解构 function createUser({ name = '匿名', age = 18, role = 'user' } = {}) { return { name, age, role }; } createUser({ name: 'Bob' }); // { name: 'Bob', age: 18, role: 'user' } createUser(); // { name: '匿名', age: 18, role: 'user' }
