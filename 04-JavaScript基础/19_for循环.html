for 循环

重复执行一段代码，直到条件不满足为止。

for 语法

for (初始化; 条件; 递增) { // 循环体 } // 执行顺序： // 1. 初始化（仅执行一次） // 2. 判断条件 → true则执行循环体 → false则退出 // 3. 执行递增 // 4. 回到第2步

基本 for 循环

// 从 0 打印到 4 for (let i = 0; i < 5; i++) { console.log(i); // 0, 1, 2, 3, 4 } // 解释： // let i = 0 → 初始化计数器 // i < 5 → 条件：当 i 小于5时继续 // i++ → 每次循环结束 i 加1

遍历数组

let fruits = ['apple', 'banana', 'cherry']; // 用 for 循环遍历数组 for (let i = 0; i < fruits.length; i++) { console.log(fruits[i]); } // 'apple', 'banana', 'cherry' // 反向遍历 for (let i = fruits.length - 1; i >= 0; i--) { console.log(fruits[i]); } // 'cherry', 'banana', 'apple'

嵌套 for 循环

// 外层循环执行一次，内层循环执行全部 for (let i = 1; i <= 3; i++) { for (let j = 1; j <= 3; j++) { console.log(`i=${i}, j=${j}`); } } // 输出 9 组 (3 × 3) // 应用：打印乘法表 for (let i = 1; i <= 9; i++) { let row = ''; for (let j = 1; j <= i; j++) { row += `${j}×${i}=${i*j} `; } console.log(row); }
嵌套循环的总执行次数 = 外循环次数 × 内循环次数

break 和 continue

// break - 立即退出循环 for (let i = 0; i < 10; i++) { if (i === 5) { break; // 到5就退出 } console.log(i); // 0, 1, 2, 3, 4 } // continue - 跳过当前次循环，继续下一次 for (let i = 0; i < 5; i++) { if (i === 2) { continue; // 跳过2 } console.log(i); // 0, 1, 3, 4（没有2） }

for 循环的变体

// 省略初始化 let i = 0; for (; i < 5; i++) { // ... } // 省略递增 for (let i = 0; i < 5;) { console.log(i); i++; // 在循环体内递增 } // 无限循环（避免！） // for (;;) { console.log('死循环'); } // 多个变量 for (let i = 0, j = 10; i < j; i++, j--) { console.log(i, j); }
⚠️ 避免无限循环！确保条件最终会变为 false。
