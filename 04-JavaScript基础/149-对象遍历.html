对象遍历

遍历对象的键、值、键值对。

for...in 遍历键

const user = { name: 'Alice', age: 25, role: 'admin' }; // 遍历对象的键 for (let key in user) { console.log(key); // 'name', 'age', 'role' console.log(user[key]); // 'Alice', 25, 'admin' } // ⚠️ for...in 会遍历继承的可枚举属性 // 可以用 hasOwnProperty 过滤 for (let key in user) { if (user.hasOwnProperty(key)) { console.log(key); // 只处理自身属性 } }

Object.keys() — 获取键数组

const user = { name: 'Alice', age: 25, role: 'admin' }; Object.keys(user); // ['name', 'age', 'role'] // 遍历键 Object.keys(user).forEach(key => { console.log(key, user[key]); });

Object.values() — 获取值数组

const user = { name: 'Alice', age: 25, role: 'admin' }; Object.values(user); // ['Alice', 25, 'admin'] // 常用于统计 const scores = { math: 90, english: 85, chinese: 92 }; const total = Object.values(scores).reduce((a, c) => a + c, 0); // 267

Object.entries() — 获取键值对数组

const user = { name: 'Alice', age: 25, role: 'admin' }; Object.entries(user); // [['name', 'Alice'], ['age', 25], ['role', 'admin']] // 遍历键值对 Object.entries(user).forEach(([key, value]) => { console.log(`${key}: ${value}`); }); // 数组转对象 const entries = [['a', 1], ['b', 2]]; Object.fromEntries(entries); // { a: 1, b: 2 }

遍历方式对比

const obj = { a: 1, b: 2, c: 3 }; // for...in — 最灵活，包含继承属性 for (let key in obj) { /* ... */ } // Object.keys + forEach — 仅自身属性 Object.keys(obj).forEach(key => { /* ... */ }); // Object.values — 只要值 Object.values(obj).forEach(val => { /* ... */ }); // Object.entries — 同时要键和值 Object.entries(obj).forEach(([k, v]) => { /* ... */ });
