删除与替换节点

从 DOM 中移除或替换元素。

removeChild

// removeChild — 从父元素中删除子节点 // 返回被删除的节点 const parent = document.getElementById('list'); const child = parent.children[0]; const removed = parent.removeChild(child); console.log(removed === child); // true // ⚠️ 如果 child 不是 parent 的子节点，会报错 // parent.removeChild(wrongElement); // ❌ NotFoundError // 删除所有子节点 while (parent.firstChild) { parent.removeChild(parent.firstChild); } // 删除所有元素子节点 while (parent.firstElementChild) { parent.removeChild(parent.firstElementChild); } // removeChild 返回的节点可以重新插入到其他地方 parent2.appendChild(removed); // 移动到另一个父元素

remove — 直接删除

// remove — 直接删除自身（不需要通过父元素） // ES2015+ 引入，更简洁 element.remove(); // 直接删除，不需要找父元素 // remove vs removeChild： // remove: 元素自行删除，不需要父元素引用 // removeChild: 父元素删除子元素，需要父元素引用 // ❌ 传统方式 // child.parentNode.removeChild(child); // ✅ 现代方式 // child.remove(); // ⚠️ remove 返回 undefined（不返回被删除的节点） // 如果需要继续使用，先保存引用或使用 removeChild

replaceChild

// replaceChild — 替换子节点 parent.replaceChild(newChild, oldChild); // newChild — 新节点 // oldChild — 被替换的旧节点（必须是 parent 的子节点） // 返回 oldChild const parent = document.getElementById('list'); const oldItem = parent.children[0]; const newItem = document.createElement('div'); newItem.textContent = '替换后的内容'; const replaced = parent.replaceChild(newItem, oldItem); console.log(replaced === oldItem); // true // ⚠️ oldChild 必须是 parent 的直接子节点 // ⚠️ 如果 oldChild 有事件监听，事件不会自动迁移到 newChild

replaceWith — 直接替换

// replaceWith — 用新节点替换当前节点 // 不需要父元素引用 element.replaceWith(newElement); element.replaceWith('文本', document.createElement('span'), '更多'); // 可以传多个参数（节点或字符串） // replaceChild vs replaceWith： // replaceChild: parent.replaceChild(new, old) // replaceWith: old.replaceWith(new) // replaceWith 更简洁

cloneNode — 克隆节点

// cloneNode — 复制元素 // cloneNode(true) — 深克隆（包括所有子节点） // cloneNode(false) — 浅克隆（只克隆自身，不包括子节点） const original = document.getElementById('original'); const deepClone = original.cloneNode(true); // 包括子元素 const shallowClone = original.cloneNode(false); // 只有自身 // ⚠️ 克隆的节点不在 DOM 中，需要手动插入 parent.appendChild(deepClone); // ⚠️ 克隆不会复制事件监听 // ⚠️ 不会复制 data 属性（自定义属性会被保留） // ⚠️ id 也被复制了（需要手动修改避免重复） // 常用：模板复制 const template = document.querySelector('.template-item'); const newItem = template.cloneNode(true); newItem.id = 'item-' + Date.now(); newItem.textContent = '新项目'; parent.appendChild(newItem);
