事件基础

addEventListener / removeEventListener / 事件对象。

addEventListener — 绑定事件

// 语法：element.addEventListener(事件类型, 处理函数, 配置) const btn = document.getElementById('btn'); // 基本用法 btn.addEventListener('click', function() { console.log('按钮被点击'); }); // 箭头函数 btn.addEventListener('click', () => { console.log('箭头函数处理'); }); // 命名函数（便于移除） function handleClick() { console.log('点击'); } btn.addEventListener('click', handleClick); // 第三个参数（配置对象） btn.addEventListener('click', handler, { capture: false, // 是否在捕获阶段触发（默认 false） once: false, // 是否只执行一次后自动移除 passive: false // 如果为 true，表示不调用 preventDefault }); // once — 只执行一次 btn.addEventListener('click', () => { console.log('只执行一次'); }, { once: true }); // ⚠️ 同一元素同一事件可以绑定多个处理函数（按顺序执行） // ⚠️ 匿名函数不能通过 removeEventListener 移除

removeEventListener — 移除事件

// 必须传入相同的函数引用 function handler() { console.log('click'); } element.addEventListener('click', handler); element.removeEventListener('click', handler); // ✅ 移除 // ❌ 匿名函数无法移除 element.addEventListener('click', function() {}); // element.removeEventListener('click', function() {}); // ❌ 不同引用 // ❌ 箭头函数也无法移除（除非保存引用） const arrowHandler = () => {}; element.addEventListener('click', arrowHandler); element.removeEventListener('click', arrowHandler); // ✅ 保存了引用 // 实际应用：条件性移除 // 只在特定条件下触发一次后移除 function onScroll() { if (window.scrollY > 500) { console.log('已滚动超过500px'); window.removeEventListener('scroll', onScroll); } } window.addEventListener('scroll', onScroll);

事件对象（event）

// 事件处理函数接收一个事件对象 element.addEventListener('click', function(event) { console.log(event); // MouseEvent 对象 }); // 常用属性和方法： event.type // 事件类型：'click', 'mouseover' 等 event.target // 触发事件的元素（真实来源） event.currentTarget // 绑定事件的元素（this 指向它） event.eventPhase // 事件阶段：1=捕获 2=目标 3=冒泡 // 鼠标事件属性 event.clientX / event.clientY // 鼠标在视口的坐标 event.pageX / event.pageY // 鼠标在页面的坐标 event.screenX / event.screenY // 鼠标在屏幕的坐标 event.button // 按下的鼠标键（0=左,1=中,2=右） // 键盘事件属性 event.key // 按下的键字符 event.code // 物理键代码 event.ctrlKey / event.shiftKey / event.altKey / event.metaKey // 方法 event.preventDefault() // 阻止默认行为 event.stopPropagation() // 停止传播（阻止冒泡/捕获） event.stopImmediatePropagation() // 立即停止所有传播

this 指向

// 在事件处理函数中： // 1. 普通函数 → this 指向绑定事件的元素（currentTarget） element.addEventListener('click', function() { console.log(this === element); // true console.log(this === event.currentTarget); // true }); // 2. 箭头函数 → this 继承自外层作用域（不绑定） element.addEventListener('click', () => { console.log(this); // 外层 this（可能是 window） }); // 建议：需要 this 指向元素就用普通函数 // 不需要或者需要外层 this 就用箭头函数
