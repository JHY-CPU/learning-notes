location 与 navigator

URL 信息与浏览器信息。

location — URL 信息

// location 包含当前 URL 的所有信息 // 完整 URL：https://user:pass@example.com:8080/path/page?q=hello#section location.href // 'https://user:pass@example.com:8080/path/page?q=hello#section' location.protocol // 'https:' location.host // 'example.com:8080'（域名 + 端口） location.hostname // 'example.com'（域名） location.port // '8080'（端口） location.pathname // '/path/page'（路径） location.search // '?q=hello'（查询参数） location.hash // '#section'（锚点） location.origin // 'https://example.com:8080'（协议 + 域名 + 端口） // location 方法： location.assign(url) // 跳转到新 URL（可后退） location.replace(url) // 替换当前 URL（不可后退） location.reload() // 重新加载当前页面 location.reload(true) // 强制刷新（从服务器，不是缓存） // ⚠️ location.href = url 等价于 location.assign(url) // ⚠️ location.replace 不会在历史记录中保留

解析 URL 参数

// 传统方式（手动解析） function getParams() { const search = location.search.substring(1); // 去掉 ? if (!search) return {}; return Object.fromEntries( search.split('&').map(pair => { const [key, val] = pair.split('='); return [decodeURIComponent(key), decodeURIComponent(val || '')]; }) ); } // 现代方式：URLSearchParams（推荐） const params = new URLSearchParams(location.search); params.get('key') // 获取参数值（单个） params.getAll('key') // 获取所有同名参数值 params.has('key') // 是否存在 params.keys() // 所有参数名 params.values() // 所有参数值 params.entries() // 所有 [key, value] params.append('key', 'value') // 追加 params.set('key', 'value') // 设置（覆盖） params.delete('key') // 删除 // 将 params 转回字符串 params.toString(); // 'key=value&key2=value2' // 操作 URL 而不跳转 const url = new URL('https://example.com?q=hello'); url.searchParams.set('q', 'world'); url.searchParams.append('page', '2'); console.log(url.href); // 'https://example.com?q=world&page=2' // hashchange 事件 window.addEventListener('hashchange', () => { console.log('hash 变为:', location.hash); });

navigator — 浏览器信息

// navigator 包含浏览器本身的信息 navigator.userAgent // 用户代理字符串（旧式浏览器检测） navigator.platform // 操作系统平台（'Win32', 'MacIntel' 等） navigator.language // 浏览器语言（'zh-CN', 'en-US'） navigator.languages // 用户偏好语言列表 navigator.onLine // 是否在线（true/false） navigator.cookieEnabled // 是否启用 Cookie // ⚠️ userAgent 不建议用于特性检测（可能被伪造） // 推荐用功能检测替代 // 现代 API（部分浏览器支持）： navigator.clipboard // 剪贴板 API navigator.geolocation // 地理位置 navigator.mediaDevices // 媒体设备（摄像头/麦克风） navigator.serviceWorker // Service Worker navigator.storage // 存储管理 navigator.connection // 网络信息 // 检测浏览器特性（推荐方式） function supportsWebP() { const elem = document.createElement('canvas'); return elem.toDataURL('image/webp').indexOf('data:image/webp') === 0; } function supportsTouch() { return 'ontouchstart' in window || navigator.maxTouchPoints > 0; }
