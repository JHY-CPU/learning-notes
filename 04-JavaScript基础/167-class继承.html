class 继承（extends）

通过 extends 关键字实现继承，super 调用父类。

extends 基本用法

// 父类 class Animal { constructor(name) { this.name = name; } eat() { console.log(this.name + ' 在吃东西'); } sound() { console.log(this.name + ' 发出了声音'); } } // 子类继承 class Dog extends Animal { constructor(name, breed) { super(name); // 必须调用 super() this.breed = breed; } // 重写方法 sound() { console.log(this.name + ' 汪汪叫'); } // 子类特有方法 fetch() { console.log(this.name + ' 在捡球'); } } const dog = new Dog('旺财', '金毛'); dog.eat(); // '旺财 在吃东西'（继承） dog.sound(); // '旺财 汪汪叫'（重写） dog.fetch(); // '旺财 在捡球'（特有） console.log(dog instanceof Dog); // true console.log(dog instanceof Animal); // true

super 关键字

// super 有两种用法： // 1. super() — 调用父类构造函数 // 2. super.method() — 调用父类方法 class Animal { constructor(name) { this.name = name; } info() { return '动物: ' + this.name; } } class Bird extends Animal { constructor(name, wingspan) { super(name); // 调用父类构造函数 this.wingspan = wingspan; } info() { return super.info() + ', 翼展: ' + this.wingspan + 'cm'; } } const bird = new Bird('老鹰', 200); console.log(bird.info()); // '动物: 老鹰, 翼展: 200cm' // ⚠️ 在 constructor 中必须先调用 super() 才能使用 this class Wrong extends Animal { constructor(name) { // this.name = name; // ❌ ReferenceError（super 之前不能访问 this） super(name); this.name = name; // ✅ } } // ⚠️ 如果子类没有 constructor，会自动生成 class Cat extends Animal {} const cat = new Cat('咪咪'); cat.eat(); // '咪咪 在吃东西'

方法重写与调用链

class Base { method() { console.log('Base.method'); } } class Middle extends Base { method() { console.log('Middle.method'); super.method(); // 调用父类方法 } } class Child extends Middle { method() { console.log('Child.method'); super.method(); // 调用 Middle.method } } const c = new Child(); c.method(); // 输出： // Child.method // Middle.method // Base.method // super 的指向是动态的，但在方法中始终指向父类的 prototype // super.method() 相当于 Base.prototype.method.call(this)

extends 内置类

// 继承 Array class MyArray extends Array { first() { return this[0]; } last() { return this[this.length - 1]; } sum() { return this.reduce((a, b) => a + b, 0); } } const arr = new MyArray(1, 2, 3, 4, 5); console.log(arr.first()); // 1 console.log(arr.last()); // 5 console.log(arr.sum()); // 15 console.log(arr.length); // 5（所有数组方法都可用） // 继承 Error class ValidationError extends Error { constructor(message, field) { super(message); this.name = 'ValidationError'; this.field = field; } } // 继承 RegExp class MyRegExp extends RegExp { constructor(pattern, flags, name) { super(pattern, flags); this.patternName = name; } } // ⚠️ 尽量别继承 Proxy（不能代理 this）

混入（Mixin）— 多重继承替代

// JavaScript 只支持单继承 // 通过 mixin 实现多重"继承" // mixin 是包含方法的对象 const FlyMixin = { fly() { console.log(this.name + ' 在飞'); }, land() { console.log(this.name + ' 降落了'); } }; const SwimMixin = { swim() { console.log(this.name + ' 在游泳'); } }; class Animal { constructor(name) { this.name = name; } } class Duck extends Animal { constructor(name) { super(name); } } // 混入 mixin（把方法复制到 prototype） Object.assign(Duck.prototype, FlyMixin, SwimMixin); const duck = new Duck('唐老鸭'); duck.fly(); // '唐老鸭 在飞' duck.swim(); // '唐老鸭 在游泳' // 更优雅的方式：创建 mixin 函数 function withFly(Base) { return class extends Base { fly() { console.log(this.name + ' 在飞'); } }; } function withSwim(Base) { return class extends Base { swim() { console.log(this.name + ' 在游泳'); } }; } class SuperDuck extends withSwim(withFly(Animal)) { constructor(name) { super(name); } }
