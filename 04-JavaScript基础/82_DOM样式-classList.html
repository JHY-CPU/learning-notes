DOM 样式 — classList

添加/删除/切换 CSS 类名，比 className 更灵活。

className vs classList

// className — 整个 class 属性字符串 const el = document.getElementById('box'); el.className; // 'box active' el.className = 'box'; // 替换所有类名 el.className += ' active'; // 追加（容易出错） // classList — DOMTokenList 对象（推荐） el.classList; // ['box', 'active'] // classList 方法一览： el.classList.add('foo'); // 添加类 el.classList.remove('active'); // 删除类 el.classList.toggle('active'); // 切换（有则删，无则加） el.classList.contains('active'); // 检查是否包含 el.classList.replace('foo', 'bar'); // 替换类名 el.classList.item(0); // 获取第0个类名 // 多个类名操作 el.classList.add('foo', 'bar', 'baz'); // 添加多个 el.classList.remove('foo', 'bar'); // 删除多个 el.classList.toggle('active', true); // 强制添加（第二个参数） el.classList.toggle('active', false); // 强制删除 // 推荐用 classList 而不是 className： // ✅ 不会影响其他类名 // ✅ 方法名清晰 // ✅ 支持多类名操作 // ✅ 更好的性能

classList.add / remove

// add — 添加一个或多个类 el.classList.add('active'); el.classList.add('highlight', 'large'); // 添加多个 // remove — 删除一个或多个类 el.classList.remove('active'); el.classList.remove('highlight', 'large'); // 删除不存在的类不会报错（静默忽略） // 对比 style： // style 适合动态计算的值（如宽度 = scrollWidth + 10） // classList 适合预定义的样式状态 // 最佳实践： // ✅ 用 classList 控制状态样式（active, disabled, error） // ✅ 用 style 控制动态计算值 // ❌ 避免混用太多行内样式

classList.toggle

// toggle — 切换类名 // 如果类存在 → 删除（返回 false） // 如果类不存在 → 添加（返回 true） const wasAdded = el.classList.toggle('active'); console.log(wasAdded); // true（刚添加）/ false（刚删除） // 带 force 参数 el.classList.toggle('active', true); // 强制添加 el.classList.toggle('active', false); // 强制删除 // 经典使用场景：开关/切换 // 按钮点击 → toggle active → CSS 控制样式变化 // 菜单展开/收起 // 明暗主题切换 // toggle 比手动判断更简洁： // ❌ if (el.classList.contains('active')) { // el.classList.remove('active'); // } else { // el.classList.add('active'); // } // ✅ el.classList.toggle('active');

contains / replace

// contains — 检查类名是否存在 if (el.classList.contains('active')) { console.log('元素处于激活状态'); } // replace — 替换类名（ES2018+） el.classList.replace('old-class', 'new-class'); // 如果 old-class 不存在，返回 false // 实际应用：状态切换 // loading → loaded // error → success // dark → light
