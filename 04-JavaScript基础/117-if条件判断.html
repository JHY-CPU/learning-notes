if 条件判断

根据条件执行不同的代码块。

if 语句

if (条件) { // 条件为 true 时执行 } let age = 18; if (age >= 18) { console.log('成年了'); }

条件可以是任何表达式，结果会被转成布尔值。

if...else 语句

if (条件) { // 条件为 true 时执行 } else { // 条件为 false 时执行 } let score = 60; if (score >= 60) { console.log('及格'); } else { console.log('不及格'); }

if...else if...else 多分支

if (条件1) { // 条件1为 true } else if (条件2) { // 条件1为 false，条件2为 true } else if (条件3) { // 条件1、2为 false，条件3为 true } else { // 以上条件都不满足 } let score = 85; if (score >= 90) { console.log('优秀'); } else if (score >= 80) { console.log('良好'); } else if (score >= 70) { console.log('中等'); } else if (score >= 60) { console.log('及格'); } else { console.log('不及格'); } // 输出: '良好'
注意条件的顺序！范围小的放前面，范围大的放后面。

嵌套 if

// if 内部可以再嵌套 if let age = 25; let hasID = true; if (age >= 18) { if (hasID) { console.log('可以进入'); } else { console.log('没带身份证'); } } else { console.log('未成年禁止进入'); }
嵌套层数太多不好读，可以用 && 合并条件。

条件中的 truthy / falsy

// if 条件会自动转布尔 if (name) { /* name 非空时执行 */ } if (count) { /* count 不为0时执行 */ } if (!user) { /* user 不存在时执行 */ } // 常见用法 let username = 'Alice'; if (username) { console.log('用户已登录:', username); } let cart = []; if (cart.length) { console.log('购物车有商品'); } else { console.log('购物车为空'); }
