this — 显式绑定（call/apply/bind）

手动指定函数的 this 指向。

call — 立即调用，参数逐个传入

function greet(greeting, punctuation) { console.log(greeting + ', ' + this.name + punctuation); } const user = { name: 'Alice' }; // call(thisArg, arg1, arg2, ...) greet.call(user, 'Hello', '!'); // 'Hello, Alice!' greet.call(user, 'Hi', '!!'); // 'Hi, Alice!!' // 借用数组方法 const obj = { 0: 'a', 1: 'b', length: 2 }; Array.prototype.push.call(obj, 'c'); console.log(obj); // { 0: 'a', 1: 'b', 2: 'c', length: 3 } // 借用 Math 方法 const numbers = [3, 1, 4, 1, 5]; Math.max.call(null, ...numbers); // 5（等同于 Math.max(...numbers)）

apply — 立即调用，参数用数组传入

function greet(greeting, punctuation) { console.log(greeting + ', ' + this.name + punctuation); } const user = { name: 'Bob' }; // apply(thisArg, [arg1, arg2, ...]) greet.apply(user, ['Hey', '.']); // 'Hey, Bob.' // apply 经典用法：展开数组 const nums = [3, 1, 4, 1, 5]; Math.max.apply(null, nums); // 5（不用 ... 展开语法） Math.min.apply(null, nums); // 1 // call vs apply // call: 参数逐个传，数量固定时用 // apply: 参数数组传，数量不固定时用 // 现代 JS 更多用 ... 展开运算符，所以 call/apply 用得少了

bind — 绑定 this，返回新函数（不立即调用）

function greet(greeting, punctuation) { console.log(greeting + ', ' + this.name + punctuation); } const user = { name: 'Charlie' }; // bind 返回新函数，this 被永久绑定 const bindGreet = greet.bind(user); bindGreet('Hello', '!'); // 'Hello, Charlie!' // 也可以预绑定部分参数（偏函数） const sayHi = greet.bind(user, 'Hi'); sayHi('!!'); // 'Hi, Charlie!!' // 经典用途：修复回调中的 this const btn = { text: 'Click me' }; btn.addEventListener = function(event, handler) { // 模拟事件绑定 handler(); }; const app = { name: 'MyApp', init() { // ❌ 不加 bind，this 会丢失 // btn.addEventListener('click', this.handleClick); // ✅ 用 bind 固定 this btn.addEventListener('click', this.handleClick.bind(this)); }, handleClick() { console.log(this.name + ' clicked'); } }; app.init(); // 'MyApp clicked' // bind 后的 this 不能被 call/apply 覆盖 const bound = greet.bind({ name: 'Fixed' }); bound.call({ name: 'TryChange' }, 'Hi', '!'); // 'Hi, Fixed!'（绑定生效）
💡 bind 最常用：修复回调函数中的 this 丢失。箭头函数出现后，很多人改用箭头函数替代 bind。

call / apply / bind 对比

function introduce(role, company) { console.log(`${this.name} 是${role}，在${company}工作`); } const person = { name: '张三' }; // call — 立即执行，参数列表 introduce.call(person, '工程师', '阿里'); // apply — 立即执行，参数数组 introduce.apply(person, ['设计师', '腾讯']); // bind — 返回新函数 const boundIntroduce = introduce.bind(person, '产品经理'); boundIntroduce('字节'); // | 方法 | 是否立即执行 | 参数形式 | 场景 | // |--------|-------------|-------------|--------------------| // | call | ✅ | 逐个传入 | 参数少且固定 | // | apply | ✅ | 数组传入 | 参数是数组 | // | bind | ❌ 返回函数 | 逐个传入 | 固定 this/偏函数 |
