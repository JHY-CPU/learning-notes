class 静态成员与私有字段

static 定义类级别的成员，# 定义私有字段（ES2022）。

static 静态成员

// 静态属性和方法属于类本身，而不是实例 class MathUtils { // 静态属性（ES2022+） static PI = 3.14159; static VERSION = '1.0.0'; // 静态方法 static add(a, b) { return a + b; } static multiply(a, b) { return a * b; } // 静态方法中不能访问实例属性 static createDefault() { return new MathUtils(); } } // 通过类名访问 console.log(MathUtils.PI); // 3.14159 console.log(MathUtils.add(2, 3)); // 5 // 实例不能访问静态成员 const m = new MathUtils(); // console.log(m.PI); // undefined // console.log(m.add); // undefined // 静态方法中的 this 指向类本身 class MyClass { static whoAmI() { console.log(this === MyClass); // true } } // 静态方法可以继承 class Child extends MathUtils {} console.log(Child.add(1, 2)); // 3（继承了静态方法）

静态方法实际应用

// 1. 工厂方法 class User { constructor(name, role) { this.name = name; this.role = role; } static createAdmin(name) { return new User(name, 'admin'); } static createGuest() { return new User('访客', 'guest'); } static fromJSON(json) { const data = JSON.parse(json); return new User(data.name, data.role); } } const admin = User.createAdmin('Alice'); const guest = User.createGuest(); const parsed = User.fromJSON('{"name":"Bob","role":"user"}'); // 2. 工具/辅助方法 class DateUtils { static formatDate(date) { return date.toISOString().split('T')[0]; } static isWeekend(date) { const day = date.getDay(); return day === 0 || day === 6; } static daysInMonth(year, month) { return new Date(year, month, 0).getDate(); } } // 3. 单例模式 class Database { static instance = null; static getInstance() { if (!Database.instance) { Database.instance = new Database(); } return Database.instance; } }

# 私有字段（ES2022）

class BankAccount { // 私有字段（以 # 开头） #balance = 0; #owner; constructor(owner, initialBalance) { this.#owner = owner; if (initialBalance > 0) this.#balance = initialBalance; } // 公有方法 deposit(amount) { if (amount <= 0) throw '金额必须大于0'; this.#balance += amount; return this.#balance; } withdraw(amount) { if (amount > this.#balance) throw '余额不足'; this.#balance -= amount; return this.#balance; } getBalance() { return this.#balance; } // 私有方法也可以！ #validateAmount(amount) { return amount > 0 && amount < 1000000; } } const account = new BankAccount('Alice', 1000); console.log(account.getBalance()); // 1000 account.deposit(500); console.log(account.getBalance()); // 1500 // ⚠️ 私有字段外部无法访问 // console.log(account.#balance); // ❌ SyntaxError // account.#validateAmount(100); // ❌ SyntaxError // 'balance' in account; // true（公有属性） // '#balance' in account; // false（私有字段的 in 检查不同）
💡 私有字段 # 是真正的硬限制（引擎级别），不同于 _ 命名约定（只是约定）。

私有字段的特性

// 1. 私有字段必须在类中声明 class Good { #name = 'declared'; // ✅ 先声明 } class Bad { constructor() { // this.#name = 'test'; // ❌ 未声明就使用（SyntaxError） } } // 2. 私有字段不能被子类访问 class Parent { #secret = 'parent secret'; getSecret() { return this.#secret; } } class Child extends Parent { reveal() { // return this.#secret; // ❌ SyntaxError return this.getSecret(); // ✅ 通过公有方法访问 } } // 3. 同名私有字段不冲突 class A { #val = 'A'; getVal() { return this.#val; } } class B { #val = 'B'; getVal() { return this.#val; } } // 4. 私有字段的类型检查 class Person { #age; constructor(age) { this.#age = age; } isSameAge(other) { // 同类的不同实例可以访问对方的私有字段 return this.#age === other.#age; } }
