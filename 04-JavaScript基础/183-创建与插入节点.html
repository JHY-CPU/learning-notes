创建与插入节点

动态创建 DOM 元素并插入到文档中。

createElement — 创建元素

// 创建元素节点 const div = document.createElement('div'); const p = document.createElement('p'); const span = document.createElement('span'); const img = document.createElement('img'); const input = document.createElement('input'); // 创建后需要设置属性和内容 div.textContent = 'Hello'; div.className = 'item'; div.id = 'item-1'; div.setAttribute('data-id', '123'); div.style.color = 'red'; // 创建文本节点（通常不直接使用） const text = document.createTextNode('纯文本'); // 创建注释节点 const comment = document.createComment('这是注释'); // 创建 DocumentFragment（用于批量操作） const fragment = document.createDocumentFragment();

appendChild / append

// appendChild — 在末尾追加子节点（只能传节点） parent.appendChild(child); parent.appendChild(document.createElement('div')); // 如果 child 已在文档中，会先移除再追加（移动位置） // append — 更灵活（可以传节点和字符串） parent.append(child); parent.append('文本节点', document.createElement('span'), '更多文本'); // append 可以传多个参数 // 字符串自动转为文本节点 // append vs appendChild： // appendChild: 只能传1个节点，返回追加的节点 // append: 可以传多个节点/字符串，返回 undefined // appendChild: 兼容性更好（老浏览器） // append: 更灵活（推荐使用）

prepend / insertBefore / after / before

// prepend — 在开头插入 parent.prepend('开头文本'); parent.prepend(document.createElement('div'), '文本'); // insertBefore — 在参照节点前插入 parent.insertBefore(newNode, referenceNode); parent.insertBefore(newNode, parent.firstChild); // 开头 parent.insertBefore(newNode, null); // 末尾（等价 appendChild） parent.insertBefore(newNode, parent.children[3]); // 第4个之前 // after — 在元素后面插入（兄弟节点） element.after(newNode); element.after('文本', document.createElement('div')); // before — 在元素前面插入（兄弟节点） element.before(newNode); element.before('文本', document.createElement('div')); // 对比总结： // parent.append(child) — 内部末尾 // parent.prepend(child) — 内部开头 // element.after(child) — 后面兄弟 // element.before(child) — 前面兄弟 // parent.insertBefore(c, r) — 内部参照节点前

DocumentFragment — 批量插入

// DocumentFragment 轻量文档片段 // 在内存中操作，不会触发重排 // 一次插入到 DOM 中（只触发一次重排） const fragment = document.createDocumentFragment(); for (let i = 0; i < 1000; i++) { const div = document.createElement('div'); div.textContent = '项目 ' + i; div.className = 'item'; fragment.appendChild(div); } // 一次性插入（只触发一次重排/重绘） parent.appendChild(fragment); // fragment 的子节点全部转入 parent // fragment 本身变为空 // ✅ 比循环内逐个 appendChild 性能好得多
