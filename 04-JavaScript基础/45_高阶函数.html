高阶函数

接收函数作为参数，或返回函数的函数。

函数作为参数

// 数组方法都是高阶函数 [1, 2, 3].map(x => x * 2); // 接收箭头函数 [1, 2, 3].filter(x => x > 1); // 接收回调 [1, 2, 3].forEach(console.log); // 自定义高阶函数 function operate(a, b, fn) { return fn(a, b); } operate(5, 3, (x, y) => x + y); // 8 operate(5, 3, (x, y) => x * y); // 15

函数作为返回值

function makeGreeting(lang) { if (lang === 'zh') return (name) => `你好, ${name}`; if (lang === 'en') return (name) => `Hello, ${name}`; if (lang === 'ja') return (name) => `こんにちは, ${name}`; return (name) => `Hi, ${name}`; } const greetZh = makeGreeting('zh'); const greetEn = makeGreeting('en'); greetZh('小明'); // "你好, 小明" greetEn('Alice'); // "Hello, Alice"

实用：函数组合（compose）

// 从右到左组合多个函数 const compose = (...fns) => (x) => fns.reduceRight((acc, fn) => fn(acc), x); const add1 = x => x + 1; const double = x => x * 2; const square = x => x * x; const composed = compose(square, double, add1); composed(2); // square(double(add1(2))) = square(double(3)) // = square(6) = 36

实用：函数管道（pipe）

// 从左到右组合（更常见） const pipe = (...fns) => (x) => fns.reduce((acc, fn) => fn(acc), x); const pipeline = pipe(add1, double, square); pipeline(2); // square(double(add1(2))) = 36（一样的结果） // 实用例子：处理用户输入 const trim = s => s.trim(); const lower = s => s.toLowerCase(); const capitalize = s => s[0].toUpperCase() + s.slice(1); const formatName = pipe(trim, lower, capitalize); formatName(' ALICE '); // 'Alice'
