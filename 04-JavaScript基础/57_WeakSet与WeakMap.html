WeakSet & WeakMap — 弱引用集合

键必须是对象，不阻止垃圾回收，没有 size/clear/遍历方法。

WeakMap

// 创建 const wm = new WeakMap(); // 键必须是对象！ const obj1 = { id: 1 }; const obj2 = { id: 2 }; // set / get / has / delete wm.set(obj1, '数据1'); wm.set(obj2, '数据2'); wm.get(obj1); // '数据1' wm.has(obj2); // true wm.delete(obj1); // true // ⚠️ 不能用的： // wm.size // undefined（没有 size） // wm.clear() // 没有 clear // for (let k of wm) {} // 不可遍历 // wm.set('str', 1) // ❌ 键必须是对象（TypeError） // 核心：弱引用 // 当 obj1 被其他代码释放（= null）， // WeakMap 中对应的条目自动被 GC 回收

WeakSet

// 创建 const ws = new WeakSet(); const obj1 = { id: 1 }; const obj2 = { id: 2 }; // add / has / delete ws.add(obj1); ws.add(obj2); ws.has(obj1); // true ws.delete(obj2); // true // ⚠️ 不能用的： // ws.size // undefined // ws.clear() // 没有 // for...of // 不可遍历 // ws.add(123) // ❌ 必须是对象（TypeError） // 核心用途：标记对象

实际应用场景

// 1. WeakMap — 私有数据存储 const privateData = new WeakMap(); class User { constructor(name, age) { privateData.set(this, { name, age }); } getName() { return privateData.get(this).name; } getAge() { return privateData.get(this).age; } } const u = new User('Alice', 25); u.getName(); // 'Alice' // 外部无法直接访问 privateData 里的内容 // 2. WeakMap — DOM 元素关联数据 // const wm = new WeakMap(); // const btn = document.getElementById('btn'); // wm.set(btn, { clicks: 0 }); // btn.addEventListener('click', () => { // const data = wm.get(btn); // data.clicks++; // }); // 当 btn 被移除时，相关数据自动 GC // 3. WeakSet — 标记/白名单 const visited = new WeakSet(); // visited.add(currentNode); // if (visited.has(node)) { /* 已访问 */ } // 4. 缓存计算结果 const cache = new WeakMap(); function processObj(obj) { if (cache.has(obj)) return cache.get(obj); const result = { /* 复杂计算 */ }; cache.set(obj, result); return result; }
💡 弱引用不一定立即回收，但保证了不会因 WeakMap/WeakSet 而阻止内存释放。

WeakMap vs Map / WeakSet vs Set

// Map — 强引用 let obj = { key: 'value' }; const map = new Map(); map.set(obj, 'data'); obj = null; // 对象仍存在（Map 持有强引用）无法 GC // WeakMap — 弱引用 let obj2 = { key: 'value' }; const wm = new WeakMap(); wm.set(obj2, 'data'); obj2 = null; // 对象被 GC 回收，WeakMap 条目自动移除 // Set vs WeakSet 同理 // | 特性 | Map/Set | WeakMap/WeakSet | // |--------------|---------------|-----------------| // | 键类型 | 任意 | 仅对象 | // | 弱引用 | ❌ | ✅ | // | 可遍历 | ✅ | ❌ | // | size | ✅ | ❌ | // | clear | ✅ | ❌ | // | GC 不阻塞 | ❌ | ✅ |
