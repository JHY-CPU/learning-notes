滚动操作

scrollTo / scrollBy / scrollIntoView / 滚动事件与优化。

窗口滚动方法

// window.scrollTo — 滚动到指定位置 window.scrollTo(x, y); // 绝对坐标 window.scrollTo({ // 选项对象 top: 0, left: 0, behavior: 'smooth' // 'auto' | 'smooth' }); // window.scrollBy — 相对当前位置滚动 window.scrollBy(0, 100); // 向下滚 100px window.scrollBy({ top: -50, // 向上滚 50px left: 0, behavior: 'smooth' }); // 滚动到页面顶部 window.scrollTo({ top: 0, behavior: 'smooth' }); // 滚动到页面底部 window.scrollTo({ top: document.documentElement.scrollHeight, behavior: 'smooth' }); // 当前滚动位置 window.scrollX // 水平滚动距离（别名 pageXOffset） window.scrollY // 垂直滚动距离（别名 pageYOffset）

元素滚动

// 元素滚动方法（同 window） element.scrollTo(x, y); element.scrollBy(x, y); element.scrollTop = 100; // 直接设置属性 element.scrollLeft = 0; // scrollIntoView — 将元素滚动到视口 element.scrollIntoView(); element.scrollIntoView(false); // 底部对齐 element.scrollIntoView({ behavior: 'smooth', // 'auto' | 'smooth' block: 'start', // 'start' | 'center' | 'end' | 'nearest' inline: 'nearest' // 'start' | 'center' | 'end' | 'nearest' }); // block 控制垂直对齐： // start — 元素顶部与视口顶部对齐（默认） // center — 元素居中 // end — 元素底部与视口底部对齐 // nearest — 就近对齐 // scroll 事件 element.addEventListener('scroll', function(e) { console.log('滚动位置:', this.scrollTop); }); // ⚠️ scroll 事件触发频率很高，需要防抖/节流

滚动优化（passive 与防抖/节流）

// 1. passive 监听 — 告诉浏览器不调用 preventDefault // 浏览器可以提前优化，滚动更流畅 window.addEventListener('scroll', handler, { passive: true }); // passive: true 表示不会调用 preventDefault // 如果调用了 preventDefault，会被忽略 // 2. 防抖（debounce）— 等用户停止滚动后再执行 function debounce(fn, delay = 200) { let timer; return function(...args) { clearTimeout(timer); timer = setTimeout(() => fn.apply(this, args), delay); }; } window.addEventListener('scroll', debounce(() => { console.log('用户停止滚动'); }, 150)); // 3. 节流（throttle）— 限制执行频率 function throttle(fn, interval = 100) { let last = 0; return function(...args) { const now = Date.now(); if (now - last >= interval) { last = now; fn.apply(this, args); } }; } window.addEventListener('scroll', throttle(() => { console.log('滚动位置:', window.scrollY); }, 200)); // 4. requestAnimationFrame + 节流（推荐） let ticking = false; window.addEventListener('scroll', () => { if (!ticking) { requestAnimationFrame(() => { console.log('滚动:', window.scrollY); ticking = false; }); ticking = true; } });
