DOM 内容操作

innerHTML / textContent / innerText — 修改元素内容。

innerHTML — 解析 HTML

// 获取元素内的 HTML 字符串 const content = element.innerHTML; // 设置 HTML（会解析标签） element.innerHTML = '<strong>加粗</strong> 和 <em>斜体</em>'; // 追加 HTML element.innerHTML += '<span>追加内容</span>'; // ⚠️ 安全风险：不要用 innerHTML 插入用户输入 // ❌ element.innerHTML = userInput; // XSS 漏洞！ // ✅ 如果需要插入用户输入，用 textContent // ✅ 或使用 setAttribute / createElement + textContent // ⚠️ 性能：innerHTML += 会重建所有 DOM // 频繁操作时用 createElement / insertAdjacentHTML 代替 // ⚠️ 会移除已有的事件监听 // 设置 innerHTML 会重新解析，之前绑定的事件丢失

textContent — 纯文本

// 获取所有文本内容（忽略 HTML 标签） const text = element.textContent; // 设置纯文本（自动转义 HTML） element.textContent = '这是 <b>文本</b>'; // 页面显示：这是 <b>文本</b>（标签被转义） // 比 innerText 更高效（不触发重排） // 包含隐藏元素的文本（display:none 的元素也能获取） // ⚠️ 会替换所有子元素 element.textContent = '新内容'; // 删掉所有子元素，替换为文本节点

innerText — 可见文本

// 获取"可见"文本内容 const visibleText = element.innerText; // 与 textContent 的区别： // 1. innerText 尊重 CSS（不返回隐藏元素的文本） // 2. innerText 会触发重排（因为要计算样式） // 3. innerText 会格式化换行（像用户看到的那样） // textContent vs innerText： // textContent：所有文本，包括隐藏元素，更高效 // innerText：只返回可见文本，会格式化输出 // 建议：优先用 textContent（除非需要可见文本）

insertAdjacentHTML — 高效插入

// 在指定位置插入 HTML（比 innerHTML += 更高效） // 不破坏已有元素的事件监听 element.insertAdjacentHTML('beforebegin', '<p>在前面</p>'); element.insertAdjacentHTML('afterbegin', '<p>内部开头</p>'); element.insertAdjacentHTML('beforeend', '<p>内部结尾</p>'); element.insertAdjacentHTML('afterend', '<p>在后面</p>'); // 位置说明： // beforebegin → <div>（元素自身之前） // afterbegin → 内容（元素内部第一个子元素之前） // beforeend → 内容（元素内部最后一个子元素之后） // afterend → </div>（元素自身之后） // 对应的 text 版本 element.insertAdjacentText('beforeend', '纯文本');

outerHTML — 包含自身

// innerHTML 是元素内部的内容 // outerHTML 包含元素自身 const div = document.querySelector('.demo-area'); console.log(div.innerHTML); // 内部 HTML console.log(div.outerHTML); // 包括 <div>...</div> // 设置 outerHTML 会替换整个元素 div.outerHTML = '<section>替换了原 div</section>'; // ⚠️ 原 div 的引用不再在 DOM 中（变量仍指向原元素）
