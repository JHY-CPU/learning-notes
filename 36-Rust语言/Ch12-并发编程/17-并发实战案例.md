## 二、具体用法

### 2.1 生产者消费者模式

```rust
use std::sync::{Arc, Mutex, mpsc};
use std::thread;

fn producer_consumer() {
    let (tx, rx) = mpsc::channel();
    let queue = Arc::new(Mutex::new(Vec::new()));

    // 生产者
    let producer = {
        let tx = tx.clone();
        thread::spawn(move || {
            for i in 0..100 {
                tx.send(i).unwrap();
            }
        })
    };

    // 消费者
    let consumers: Vec<_> = (0..4).map(|_| {
        let rx = rx.clone();
        thread::spawn(move || {
            for msg in rx {
                println!("处理: {}", msg);
            }
        })
    }).collect();

    drop(tx);
    producer.join().unwrap();
    for c in consumers { c.join().unwrap(); }
}
```

### 2.3 线程池实现

```rust
use std::sync::{mpsc, Arc, Mutex};
use std::thread;

struct ThreadPool {
    workers: Vec<Worker>,
    sender: mpsc::Sender<Job>,
}

type Job = Box<dyn FnOnce() + Send + 'static>;

impl ThreadPool {
    fn new(size: usize) -> Self {
        assert!(size > 0);

        let (sender, receiver) = mpsc::channel();
        let receiver = Arc::new(Mutex::new(receiver));

        let workers = (0..size)
            .map(|id| Worker::new(id, Arc::clone(&receiver)))
            .collect();

        ThreadPool { workers, sender }
    }

    fn execute<F>(&self, f: F)
    where
        F: FnOnce() + Send + 'static,
    {
        self.sender.send(Box::new(f)).unwrap();
    }
}

impl Drop for ThreadPool {
    fn drop(&mut self) {
        for worker in &self.workers {
            println!("关闭工作线程 {}", worker.id);
            if let Some(thread) = worker.thread.take() {
                thread.join().unwrap();
            }
        }
    }
}

struct Worker {
    id: usize,
    thread: Option<thread::JoinHandle<()>>,
}

impl Worker {
    fn new(id: usize, receiver: Arc<Mutex<mpsc::Receiver<Job>>>) -> Self {
        let thread = thread::spawn(move || {
            loop {
                let message = receiver.lock().unwrap().recv();
                match message {
                    Ok(job) => {
                        println!("工作线程 {} 执行任务", id);
                        job();
                    }
                    Err(_) => {
                        println!("工作线程 {} 关闭", id);
                        break;
                    }
                }
            }
        });

        Worker {
            id,
            thread: Some(thread),
        }
    }
}
```

### 2.4 Web 服务器请求处理

```rust
use std::sync::{Arc, Mutex};
use std::thread;
use std::sync::mpsc;

struct AppState {
    request_count: u64,
    cache: std::collections::HashMap<String, String>,
}

fn handle_request(state: Arc<Mutex<AppState>>, request: String) -> String {
    let mut state = state.lock().unwrap();
    state.request_count += 1;

    if let Some(cached) = state.cache.get(&request) {
        return cached.clone();
    }

    let response = format!("处理请求 #{}: {}", state.request_count, request);
    state.cache.insert(request, response.clone());
    response
}

fn web_server_example() {
    let state = Arc::new(Mutex::new(AppState {
        request_count: 0,
        cache: std::collections::HashMap::new(),
    }));

    let (tx, rx) = mpsc::channel();

    // 模拟请求处理
    for i in 0..10 {
        let state = Arc::clone(&state);
        let tx = tx.clone();
        thread::spawn(move || {
            let response = handle_request(state, format!("请求{}", i));
            tx.send(response).unwrap();
        });
    }

    drop(tx);
    for response in rx {
        println!("响应: {}", response);
    }
}
```

### 2.5 事件驱动架构

```rust
use std::sync::mpsc;
use std::collections::HashMap;

enum Event {
    UserLogin(String),
    UserLogout(String),
    Message(String, String), // from, content
}

struct EventBus {
    handlers: HashMap<String, Vec<Box<dyn Fn(&Event) + Send>>>,
}

impl EventBus {
    fn new() -> Self {
        EventBus { handlers: HashMap::new() }
    }

    fn subscribe(&mut self, event_type: &str, handler: impl Fn(&Event) + Send + 'static) {
        self.handlers
            .entry(event_type.to_string())
            .or_default()
            .push(Box::new(handler));
    }

    fn publish(&self, event: &Event) {
        let event_type = match event {
            Event::UserLogin(_) => "login",
            Event::UserLogout(_) => "logout",
            Event::Message(_, _) => "message",
        };

        if let Some(handlers) = self.handlers.get(event_type) {
            for handler in handlers {
                handler(event);
            }
        }
    }
}
```

## 四、架构模式对比

| 模式 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| 线程池 | 资源复用 | 固定并发度 | Web 服务器 |
| 生产者消费者 | 解耦 | 复杂度 | 数据处理管道 |
| Actor | 隔离性强 | 开销 | 分布式系统 |
| Pipeline | 并行度高 | 延迟 | 流处理 |
| Work Stealing | 负载均衡 | 实现复杂 | 通用并行 |

## 五、注意事项与常见陷阱

1. **设计先行**：先设计好并发架构再编码，画出数据流图
2. **错误处理**：并发场景的错误处理更复杂，需要统一的错误传播机制
3. **监控指标**：记录线程利用率和队列深度，及时发现瓶颈
4. **优雅关闭**：设计时考虑优雅关闭机制，避免数据丢失
5. **压力测试**：充分测试高并发场景，模拟真实负载
6. **容量规划**：根据业务量规划线程池大小和队列容量
7. **回退策略**：系统过载时的回退和降级策略
