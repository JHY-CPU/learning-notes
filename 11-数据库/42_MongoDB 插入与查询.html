📄 MongoDB 插入与查询

insertOne/insertMany、find 查询语法、查询运算符 ($eq/$gt/$in/$regex)、投影 (projection)、分页 (sort/skip/limit)、游标 (cursor) 与迭代。

插入文档

// ========== 插入操作 ========== // 插入单个文档 db.users.insertOne({ name: "Alice", age: 28, email: "alice@test.com", tags: ["tech", "music"], created_at: new Date() }) // 插入多个文档 db.users.insertMany([ { name: "Bob", age: 25, email: "bob@test.com" }, { name: "Carol", age: 30, email: "carol@test.com" }, { name: "Dave", age: 35, email: "dave@test.com" } ]) // 指定 _id (不指定则自动生成 ObjectId) db.users.insertOne({ _id: "user_001", name: "Eve", age: 27 }) // insertMany 选项 db.users.insertMany( [ { name: "Frank" }, { name: "Grace" } ], { ordered: false // 继续插入即使部分失败 (默认 true) } ) // ========== insertMany 返回值 ========== // { // "acknowledged": true, // "insertedIds": [ // ObjectId("5f8c..."), // ObjectId("5f8d...") // ] // } // ========== 其他插入方式 ========== // upsert (插入或更新) db.users.updateOne( { email: "alice@test.com" }, { $set: { name: "Alice Updated" } }, { upsert: true } ) // replace (替换整个文档) db.users.replaceOne( { name: "Bob" }, { name: "Bob", age: 26, city: "Shanghai" } )

查询文档

// ========== 基本查询 ========== // 查询所有文档 db.users.find() // 格式化输出 db.users.find().pretty() // 查询单条 db.users.findOne({ name: "Alice" }) // ========== 查询运算符 ========== // 比较运算符 db.users.find({ age: 28 }) // 等于 db.users.find({ age: { $eq: 28 } }) // 等于 (同上) db.users.find({ age: { $ne: 28 } }) // 不等于 db.users.find({ age: { $gt: 25 } }) // 大于 (>) db.users.find({ age: { $gte: 25 } }) // 大于等于 (>=) db.users.find({ age: { $lt: 30 } }) // 小于 (<) db.users.find({ age: { $lte: 30 } }) // 小于等于 (<=) // IN / NOT IN db.users.find({ age: { $in: [25, 28, 30] } }) // 在列表中 db.users.find({ age: { $nin: [25, 30] } }) // 不在列表中 // 逻辑运算符 db.users.find({ // AND (默认) age: { $gte: 25 }, name: "Alice" }) db.users.find({ // OR $or: [ { age: { $lt: 25 } }, { age: { $gt: 35 } } ] }) db.users.find({ // NOT age: { $not: { $gte: 30 } } }) db.users.find({ // NOR $nor: [ { age: 25 }, { name: "Alice" } ] }) // 字段存在 db.users.find({ email: { $exists: true } }) // 有 email 字段 db.users.find({ phone: { $exists: false } }) // 无 phone 字段 // 类型匹配 db.users.find({ age: { $type: "int" } }) // age 是整数

高级查询与投影

// ========== 正则与文本查询 ========== // 正则查询 (类似 SQL LIKE) db.users.find({ name: /^A/ }) // 以 A 开头 db.users.find({ name: /son$/i }) // 以 son 结尾 (忽略大小写) db.users.find({ name: /li/ }) // 包含 li db.users.find({ name: { $regex: "^A", $options: "i" } }) // 正则运算符 // ========== 数组查询 ========== db.users.find({ tags: "tech" }) // 包含 tech 标签 db.users.find({ tags: { $all: ["tech", "music"] } }) // 同时包含 db.users.find({ tags: { $size: 2 } }) // 数组长度为 2 db.users.find({ "tags.0": "tech" }) // 第一个元素是 tech // ========== 嵌套文档查询 ========== // 点号查询嵌套字段 db.users.find({ "address.city": "Beijing" }) db.users.find({ "scores.subject": "Math" }) // ========== 投影 (选择字段) ========== // 1 包含, 0 排除 db.users.find( { age: { $gte: 25 } }, { name: 1, email: 1, _id: 0 } // 只返回 name 和 email ) // 嵌套字段投影 db.users.find( {}, { "address.city": 1, name: 1 } ) // 数组切片 db.users.find( {}, { tags: { $slice: 1 } } // 只返回第一个 tag ) // ========== 分页与排序 ========== // 排序: 1 升序, -1 降序 db.users.find().sort({ age: 1, name: -1 }) // 分页 db.users.find() .sort({ created_at: -1 }) .skip(20) // 跳过前 20 条 .limit(10) // 返回 10 条 // 计数 db.users.countDocuments({ age: { $gte: 25 } }) db.users.find({ age: { $gte: 25 } }).count()

游标操作

// ========== 游标 (Cursor) ========== // find() 返回游标, 不是实际数据 // ========== 游标方法 ========== // 基本迭代 const cursor = db.users.find({ age: { $gte: 25 } }) cursor.forEach(user => { print(user.name + ": " + user.age) }) // 转换为数组 const users = db.users.find().toArray() // 限制返回条数 db.users.find().limit(5) // 跳过 db.users.find().skip(10) // 排序 db.users.find().sort({ age: -1 }) // 链式调用 db.users.find() .sort({ created_at: -1 }) .skip(20) .limit(10) // ========== 游标注意事项 ========== // 1. 游标有超时 (默认 10 分钟无操作自动关闭) // 2. 大数据量用游标迭代, 不要 toArray() // 3. .sort() 要在 .limit() 之前 (逻辑不是执行顺序) // 4. skip + limit 在大偏移量时性能差 (类似 SQL OFFSET) // 替代: 游标分页用 _id 或排序字段 // ========== 游标分页 (Keyset) ========== // 第 1 页 db.users.find().sort({ _id: 1 }).limit(20) // 第 2 页 (上一页最后 _id = xxx) db.users.find({ _id: { $gt: ObjectId("xxx") } }) .sort({ _id: 1 }).limit(20) // 第 N 页 db.users.find({ _id: { $gt: last_id } }) .sort({ _id: 1 }).limit(20)
💡 查询要点: 比较运算符 $gt/$lt/$gte/$lte; 逻辑 $or/$and/$not; 正则 /pattern/i; 投影 1包含/0排除; sort+skip+limit 做分页; 游标分页用 _id > last_id 替代 skip; 嵌套字段用点号 "address.city"。
