关系型数据库概念

// ========== 关系模型 ========== // 关系型数据库: 以表 (关系) 组织数据 // 1970 年 E.F. Codd 提出关系模型 // 核心概念: // 表 (Table) = 关系 (Relation) // 行 (Row) = 元组 (Tuple) // 列 (Column) = 属性 (Attribute) // 主键 (PK) = 唯一标识一行 // 外键 (FK) = 引用其他表的主键 // ========== 示例: 博客数据库 ========== // users 表: // ┌─────┬───────────┬──────────────────┐ // │ id │ username │ email │ ← 行 // ├─────┼───────────┼──────────────────┤ // │ 1 │ alice │ alice@test.com │ // │ 2 │ bob │ bob@test.com │ // │ 3 │ charlie │ charlie@test.com │ // └─────┴───────────┴──────────────────┘ // ↑ ↑ // 主键 列 // posts 表: // ┌────┬──────────┬──────────┬─────────┐ // │ id │ title │ user_id │ content │ // ├────┼──────────┼──────────┼─────────┤ // │ 1 │ post1 │ 1 │ ... │ ← FK→users.id // │ 2 │ post2 │ 1 │ ... │ // │ 3 │ post3 │ 2 │ ... │ // └────┴──────────┴──────────┴─────────┘ // ↑ // 外键 (引用 users.id)

主键与外键

// ========== 主键 (Primary Key) ========== // 主键: 唯一标识表中每一行 // 特点: // - 唯一性 (UNIQUE) // - 非空 (NOT NULL) // - 一个表只能有一个主键 // - 可单个列或多个列 (复合主键) // 自然主键: 业务含义 (如身份证号) // 代理主键: 无业务含义 (如自增 ID) ← 更推荐 // 常用主键类型: // 1. 自增整数 (INT AUTO_INCREMENT / SERIAL) // 2. UUID (全局唯一,适合分布式) // 3. 雪花算法 ID (分布式+有序) // ========== 外键 (Foreign Key) ========== // 外键: 引用其他表的主键 // 作用: 维护数据完整性 (参照完整性) // 外键约束选项: // ON DELETE CASCADE: 父表删除时,子表自动删除 // ON DELETE SET NULL: 父表删除时,子表设为 NULL // ON DELETE RESTRICT: 有子记录时禁止删除父记录 // ON DELETE NO ACTION: 同 RESTRICT // ON UPDATE CASCADE: 父表更新时,子表同步更新 // ========== 索引 (Index) ========== // 索引: 加速查询的数据结构 (B-Tree) // 主键自动建立索引 // 外键通常也建索引 // 经常查询的列适合加索引

数据类型

// ========== 常见 SQL 数据类型 ========== // ===== 数值类型 ===== // INT / INTEGER 整数 (-2^31 ~ 2^31-1) // BIGINT 大整数 (-2^63 ~ 2^63-1) // SMALLINT 小整数 // TINYINT 极小整数 (0~255) // DECIMAL(p,s) 精确小数 (如 DECIMAL(10,2)) // FLOAT / REAL 浮点数 (近似) // DOUBLE 双精度浮点 // ===== 字符串类型 ===== // CHAR(n) 定长 (如 CHAR(11),性能好) // VARCHAR(n) 变长 (如 VARCHAR(255)) // TEXT 长文本 (最大 64KB) // MEDIUMTEXT 中等文本 (最大 16MB) // LONGTEXT 长文本 (最大 4GB) // ===== 日期时间 ===== // DATE 日期 (2024-01-15) // TIME 时间 (14:30:00) // DATETIME 日期时间 (2024-01-15 14:30:00) // TIMESTAMP 时间戳 (时区敏感) // YEAR 年份 // ===== 布尔 ===== // BOOLEAN / BOOL TRUE / FALSE / NULL // (MySQL 中 TINYINT(1) 实现) // ===== 二进制 ===== // BLOB 二进制大对象 (图片/文件) // JSON JSON 数据 (5.7+) // ENUM 枚举 (ENUM('a','b','c'))

NULL 三值逻辑

// ========== NULL 处理 ========== // NULL: 表示"未知"或"不存在" // 不是 0,也不是空字符串! // 三值逻辑: // TRUE AND NULL = NULL (未知) // FALSE AND NULL = FALSE // NULL OR NULL = NULL // NULL = NULL = NULL (不是 TRUE!) // NULL IS NULL = TRUE ✅ (正确比较) // ========== 常见陷阱 ========== SELECT * FROM users WHERE age = NULL; -- ❌ 错误! 不会返回任何行 SELECT * FROM users WHERE age IS NULL; -- ✅ 正确 SELECT * FROM users WHERE age != 25; -- ❌ NULL 不会包含 SELECT * FROM users WHERE age != 25 OR age IS NULL; -- ✅ // ========== NULL 处理函数 ========== -- MySQL: SELECT IFNULL(age, 0) FROM users; -- NULL → 0 SELECT COALESCE(age, email, 'default') FROM users; -- 第一个非 NULL -- PostgreSQL: SELECT COALESCE(age, 0) FROM users; SELECT NULLIF(a, b); -- a == b 返回 NULL -- 通用: SELECT * FROM users WHERE age IS NOT NULL; // ========== 约束与 NULL ========== // NOT NULL: 禁止 NULL // UNIQUE: NULL 不受 UNIQUE 限制 (多个 NULL 可以) // 主键: 自动 NOT NULL + UNIQUE

DCL (数据控制语言) // GRANT / REVOKE // 权限管理 // 4.
