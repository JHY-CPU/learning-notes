MySQL 状态监控

// ========== SHOW STATUS ========== -- 查看数据库运行状态 -- 1. 连接相关 SHOW STATUS LIKE 'Threads_connected'; -- 当前连接数 SHOW STATUS LIKE 'Threads_running'; -- 正在运行的线程 SHOW STATUS LIKE 'Max_used_connections'; -- 最大连接数峰值 SHOW STATUS LIKE 'Aborted_connects'; -- 失败连接数 -- 2. 查询性能 SHOW STATUS LIKE 'Questions'; -- 总查询数 SHOW STATUS LIKE 'Queries'; -- 总请求数 SHOW STATUS LIKE 'Slow_queries'; -- 慢查询数 SHOW STATUS LIKE 'Select_full_join'; -- 全表 JOIN (需优化) SHOW STATUS LIKE 'Select_scan'; -- 全表扫描数 -- 3. 临时对象 SHOW STATUS LIKE 'Created_tmp_tables'; -- 创建临时表数 SHOW STATUS LIKE 'Created_tmp_disk_tables'; -- 磁盘临时表 (应尽量少) -- 4. 缓冲池 SHOW STATUS LIKE 'Innodb_buffer_pool_reads'; -- 从磁盘读取 SHOW STATUS LIKE 'Innodb_buffer_pool_read_requests'; -- 总读取请求 SHOW STATUS LIKE 'Innodb_buffer_pool_hit_rate'; -- 缓存命中率 -- 5. 写入相关 SHOW STATUS LIKE 'Innodb_os_log_written'; -- 日志写入量 SHOW STATUS LIKE 'Innodb_data_writes'; -- 数据写入次数 -- 6. 死锁 SHOW STATUS LIKE 'Innodb_deadlocks'; -- 死锁次数 -- ========== 关键指标 ========== -- 缓存命中率 > 95% (Innodb_buffer_pool_hit_rate) -- 线程连接数 < max_connections 的 80% -- 磁盘临时表 = 0 (Created_tmp_disk_tables) -- 慢查询 = 0 (Slow_queries) -- 全表 JOIN = 0 (Select_full_join)

INFORMATION_SCHEMA

// ========== INFORMATION_SCHEMA 查询 ========== -- 1. 查看所有表大小 SELECT table_schema AS database_name, table_name, ROUND((data_length + index_length) / 1024 / 1024, 2) AS total_size_mb, ROUND(data_length / 1024 / 1024, 2) AS data_size_mb, ROUND(index_length / 1024 / 1024, 2) AS index_size_mb, table_rows FROM information_schema.tables WHERE table_schema = 'mydb' ORDER BY total_size_mb DESC; -- 2. 查看最大表 SELECT table_schema, table_name, ROUND(((data_length + index_length) / 1024 / 1024), 2) AS size_mb FROM information_schema.tables ORDER BY size_mb DESC LIMIT 10; -- 3. 查看分区信息 SELECT * FROM information_schema.partitions WHERE table_schema = 'mydb' AND table_name = 'orders'; -- 4. 查看索引使用情况 (MySQL 5.6+) SELECT * FROM information_schema.statistics WHERE table_schema = 'mydb' AND table_name = 'orders'; -- ========== INNODB 引擎状态 ========== -- InnoDB 状态 (详细) SHOW ENGINE INNODB STATUS\G -- InnoDB 指标 SHOW STATUS LIKE 'Innodb_%'; -- 事务 SELECT * FROM information_schema.INNODB_TRX; -- 锁 SELECT * FROM performance_schema.data_locks; -- 锁等待 SELECT * FROM performance_schema.data_lock_waits;

进程与连接监控

// ========== 进程列表 ========== -- 查看当前所有连接 SHOW FULL PROCESSLIST; -- 输出: -- Id: 连接 ID -- User: 用户 -- Host: 客户端地址 -- db: 当前数据库 -- Command: Sleep/Query/Execute -- Time: 执行时间 (秒) -- State: 当前状态 (Locked/Sending data/...) -- Info: 正在执行的 SQL -- 常见的 State 问题: -- 'Sending data': 正在扫描大量数据 (需优化) -- 'Waiting for table lock': 表锁等待 -- 'Locked': 行锁等待 -- 'Creating sort index': 排序操作 -- 'Copying to tmp table': 临时表操作 -- 杀死长时间运行的查询 -- KILL 12345; -- 12345 是线程 ID -- 查询运行超过 60 秒的连接 SELECT id, user, host, db, command, time, state, info FROM information_schema.processlist WHERE time > 60; -- ========== 配置检查 ========== -- 查看关键配置 SHOW VARIABLES LIKE 'max_connections'; SHOW VARIABLES LIKE 'innodb_buffer_pool_size'; SHOW VARIABLES LIKE 'innodb_log_file_size'; SHOW VARIABLES LIKE 'tmp_table_size'; SHOW VARIABLES LIKE 'max_allowed_packet'; SHOW VARIABLES LIKE 'query_cache_type'; -- MySQL 8.0 已移除 -- 推荐配置: -- innodb_buffer_pool_size = 可用内存的 60-80% -- max_connections = 200-1000 (根据场景) -- innodb_log_file_size = 1-4GB

监控工具与方案

// ========== 监控工具 ========== -- 1. MySQL Enterprise Monitor (官方) -- - 付费 -- - 全面的监控告警 -- 2. Percona Monitoring and Management (PMM) -- - 免费开源 -- - 基于 Prometheus + Grafana -- - 包含 Query Analytics -- 3. Prometheus + mysqld_exporter -- 自建监控方案: -- - mysqld_exporter 采集 MySQL 指标 -- - Prometheus 存储 -- - Grafana 仪表盘 -- 4. 云厂商监控 -- - AWS RDS: CloudWatch -- - 阿里云 RDS: 云监控 -- ========== 关键告警规则 ========== -- CPU > 80% -- 连接数 > max_connections 的 80% -- 复制延迟 > 10 秒 -- 慢查询 > 100/分钟 -- 磁盘使用 > 85% -- 缓存命中率 < 95% -- 死锁 > 0 -- ========== 日常检查清单 ========== -- 每天: -- 1. 检查慢查询日志 -- 2. 查看磁盘使用率 -- 3. 检查主从延迟 -- 每周: -- 1. 分析索引使用情况 -- 2. 查看大表增长趋势 -- 3. 检查未使用索引 -- 每月: -- 1. 优化表 (OPTIMIZE TABLE) -- 2. 审查查询性能 -- 3. 检查安全配置
💡 监控要点: SHOW STATUS 看关键指标; 缓存命中率 > 95%; 连接数不超过上限 80%; 避免磁盘临时表; 设置慢查询告警; 定期检查大表增长; 使用 Prometheus + Grafana 搭建监控体系。
