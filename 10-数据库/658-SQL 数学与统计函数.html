基本数学函数

// ========== MySQL 数学函数 ========== -- 1. 四舍五入与截断 SELECT ROUND(123.4567); -- 123 (默认 0 位小数) SELECT ROUND(123.4567, 2); -- 123.46 (保留 2 位) SELECT ROUND(123.4567, -1); -- 120 (十位四舍五入) SELECT TRUNCATE(123.4567, 2); -- 123.45 (截断, 不四舍五入) SELECT CEIL(123.1); -- 124 (向上取整) SELECT FLOOR(123.9); -- 123 (向下取整) -- 2. 绝对值与符号 SELECT ABS(-10); -- 10 SELECT SIGN(-5); -- -1 SELECT SIGN(0); -- 0 SELECT SIGN(8); -- 1 -- 3. 幂与根 SELECT POWER(2, 10); -- 1024 (2^10) SELECT SQRT(144); -- 12 SELECT POW(3, 3); -- 27 SELECT EXP(1); -- 2.71828... (e^1) -- 4. 对数 SELECT LOG(100); -- 4.605... (自然对数) SELECT LOG(10, 100); -- 2 (以 10 为底) SELECT LOG2(1024); -- 10 (以 2 为底) SELECT LOG10(1000); -- 3 (以 10 为底) -- 5. 三角函数 SELECT PI(); -- 3.141593 SELECT SIN(RADIANS(30)); -- 0.5 (SIN 30°) SELECT COS(RADIANS(60)); -- 0.5 SELECT DEGREES(PI()); -- 180 (弧度转角度) -- 6. 取余 SELECT MOD(10, 3); -- 1 SELECT 10 % 3; -- 1 (同余) -- 7. 随机数 SELECT RAND(); -- 0~1 随机浮点数 SELECT RAND(42); -- 固定种子 (可重复) SELECT FLOOR(RAND() * 100) + 1; -- 1~100 随机整数

统计聚合函数

// ========== 统计函数 ========== -- 创建示例表 CREATE TABLE exam_scores ( id INT PRIMARY KEY, student_id INT, subject VARCHAR(20), score DECIMAL(5,2), class VARCHAR(10) ); INSERT INTO exam_scores VALUES (1, 1, 'Math', 85, 'A'), (2, 2, 'Math', 92, 'A'), (3, 3, 'Math', 78, 'A'), (4, 4, 'Math', 95, 'A'), (5, 5, 'Math', 88, 'A'), (6, 6, 'Math', 76, 'B'), (7, 7, 'Math', 91, 'B'), (8, 8, 'Math', 69, 'B'), (9, 9, 'Math', 84, 'B'), (10, 10, 'Math', 97, 'B'); -- ========== 基本统计 ========== -- 描述性统计 SELECT COUNT(*) AS n, AVG(score) AS mean, STDDEV(score) AS stddev, -- 标准差 (样本) STDDEV_POP(score) AS stddev_pop, -- 总体标准差 VARIANCE(score) AS variance, -- 方差 (样本) VAR_POP(score) AS var_pop, -- 总体方差 MIN(score) AS min_score, MAX(score) AS max_score, ROUND(AVG(score), 2) AS avg_score FROM exam_scores; -- ========== 分组合计统计 ========== -- 按班级统计 SELECT class, COUNT(*) AS students, AVG(score) AS avg_score, STDDEV(score) AS stddev, MIN(score) AS min_score, MAX(score) AS max_score, MAX(score) - MIN(score) AS range_score FROM exam_scores GROUP BY class; -- ========== 中位数 (MySQL 8.0+) ========== -- 使用窗口函数 PERCENT_RANK WITH ranked AS ( SELECT score, ROW_NUMBER() OVER (ORDER BY score) AS rn, COUNT(*) OVER () AS total FROM exam_scores ) SELECT AVG(score) AS median FROM ranked WHERE rn IN (FLOOR((total+1)/2), CEIL((total+1)/2)); -- 或者用 PERCENTILE_CONT (PostgreSQL) -- SELECT PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY score) AS median -- FROM exam_scores; -- ========== 四分位数 ========== -- PostgreSQL: -- SELECT -- PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY score) AS q1, -- PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY score) AS q2, -- PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY score) AS q3 -- FROM exam_scores;

窗口统计函数

// ========== 窗口统计 ========== -- 1. 移动平均 (3 期) SELECT date, revenue, AVG(revenue) OVER (ORDER BY date ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS ma_3, AVG(revenue) OVER (ORDER BY date ROWS BETWEEN 4 PRECEDING AND CURRENT ROW) AS ma_5 FROM daily_revenue; -- 2. 累计统计 SELECT date, revenue, SUM(revenue) OVER (ORDER BY date) AS cumulative_revenue, AVG(revenue) OVER (ORDER BY date) AS running_avg, MAX(revenue) OVER (ORDER BY date ROWS UNBOUNDED PRECEDING) AS historical_max FROM daily_revenue; -- 3. 分组内统计 SELECT class, student_id, score, AVG(score) OVER (PARTITION BY class) AS class_avg, STDDEV(score) OVER (PARTITION BY class) AS class_stddev, (score - AVG(score) OVER (PARTITION BY class)) / NULLIF(STDDEV(score) OVER (PARTITION BY class), 0) AS z_score FROM exam_scores; -- 4. 同比/环比 SELECT date, revenue, LAG(revenue) OVER (ORDER BY date) AS prev_day, (revenue - LAG(revenue) OVER (ORDER BY date)) / LAG(revenue) OVER (ORDER BY date) * 100 AS daily_change_pct, revenue - AVG(revenue) OVER (ORDER BY date ROWS 6 PRECEDING) AS vs_week_avg FROM daily_revenue; -- ========== 相关性 ========== -- Pearson 相关系数 (MySQL) SELECT (COUNT(*) * SUM(x * y) - SUM(x) * SUM(y)) / SQRT( (COUNT(*) * SUM(x * x) - SUM(x) * SUM(x)) * (COUNT(*) * SUM(y * y) - SUM(y) * SUM(y)) ) AS pearson_corr FROM data_pairs; -- PostgreSQL CORR 函数 -- SELECT CORR(price, sales_volume) AS price_sales_corr -- FROM product_daily;

金融与业务计算

// ========== 金融函数 ========== -- 1. 百分比计算 SELECT product_name, sales_amount, SUM(sales_amount) OVER () AS total_sales, ROUND(sales_amount / SUM(sales_amount) OVER () * 100, 2) AS pct_of_total FROM product_sales; -- 2. 累计百分比 (帕累托) WITH sales_data AS ( SELECT product_name, sales_amount, SUM(sales_amount) OVER () AS total, SUM(sales_amount) OVER (ORDER BY sales_amount DESC) AS running_total FROM product_sales ) SELECT product_name, sales_amount, ROUND(sales_amount / total * 100, 2) AS pct, ROUND(running_total / total * 100, 2) AS running_pct FROM sales_data; -- 3. 加权平均 SELECT SUM(price * quantity) / SUM(quantity) AS weighted_avg_price FROM order_items; -- 4. CAGR (年均复合增长率) -- ((end_value / start_value)^(1/years) - 1) SELECT ROUND( (POWER(1.0 * (SELECT revenue FROM sales WHERE year = 2024) / (SELECT revenue FROM sales WHERE year = 2020), 1.0 / 4) - 1) * 100, 2) AS cagr_pct; -- 5. 分箱 / 直方图 SELECT CASE WHEN score >= 90 THEN '90-100' WHEN score >= 80 THEN '80-89' WHEN score >= 70 THEN '70-79' WHEN score >= 60 THEN '60-69' ELSE '<60' END AS score_range, COUNT(*) AS count, RPAD('', COUNT(*), '█') AS bar FROM exam_scores GROUP BY score_range ORDER BY score_range DESC; -- 6. 异常值检测 (3σ 法则) WITH stats AS ( SELECT AVG(score) AS mean, STDDEV(score) AS stddev FROM exam_scores ) SELECT id, student_id, score, ROUND((score - mean) / stddev, 2) AS z_score, CASE WHEN ABS(score - mean) > 3 * stddev THEN '异常' WHEN ABS(score - mean) > 2 * stddev THEN '注意' ELSE '正常' END AS flag FROM exam_scores, stats;

PostgreSQL 统计扩展

// ========== PostgreSQL 统计函数 ========== -- PostgreSQL 有更丰富的统计函数 -- 1. 基本统计 -- AVG(), COUNT(), SUM(), MIN(), MAX() -- STDDEV_SAMP(), STDDEV_POP() -- VAR_SAMP(), VAR_POP() -- CORR() — 相关系数 -- COVAR_SAMP(), COVAR_POP() — 协方差 -- 2. 排序统计 -- PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY col) — 中位数 -- PERCENTILE_DISC(0.5) WITHIN GROUP (ORDER BY col) — 离散百分位数 -- MODE() WITHIN GROUP (ORDER BY col) — 众数 -- 3. 线性回归 -- REGR_SLOPE(y, x) — 斜率 -- REGR_INTERCEPT(y, x) — 截距 -- REGR_R2(y, x) — R² 决定系数 -- 4. 实际示例 (PostgreSQL) -- SELECT -- CORR(price, sales_volume) AS price_sales_corr, -- REGR_SLOPE(sales_volume, price) AS sales_per_price, -- REGR_INTERCEPT(sales_volume, price) AS base_sales, -- REGR_R2(sales_volume, price) AS r_squared -- FROM product_daily -- WHERE product_id = 42; -- ========== 性能注意事项 ========== -- 1. STDDEV/VARIANCE 在大量数据上计算成本高 -- 2. 窗口统计需要内存排序, 注意内存限制 -- 3. 大量数据统计建议用物化视图预计算 -- 4. PostgreSQL 统计函数扩展需要启用 -- 5. 浮点精度: 使用 DECIMAL 类型避免精度丢失
💡 要点: ROUND/CEIL/FLOOR 控制精度; STDDEV/VARIANCE 描述分布离散度; 窗口移动平均用于平滑趋势; CORR 衡量相关性; 加权平均比简单平均更准确; 3σ 法则检测异常值; 直方图分箱观察分布; PostgreSQL 统计函数更丰富。
