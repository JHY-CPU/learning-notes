# 聊天与社交系统

## 核心概念

聊天与社交系统是网游的重要组成部分，负责玩家之间的信息交流和社交关系管理。包括文字聊天、语音聊天、好友系统、公会系统和离线消息等功能。系统设计需要考虑消息分发效率、内容安全、数据库扩展性和实时性。

## 数据库设计

### 核心表结构

```sql
-- 好友关系表
CREATE TABLE friendships (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    player_id BIGINT NOT NULL,
    friend_id BIGINT NOT NULL,
    status ENUM('pending', 'accepted', 'blocked') DEFAULT 'pending',
    intimacy INT DEFAULT 0,           -- 亲密度
    nickname VARCHAR(50),             -- 备注名
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_friendship (player_id, friend_id),
    INDEX idx_player (player_id),
    INDEX idx_friend (friend_id)
);

-- 聊天消息表（离线消息存储）
CREATE TABLE chat_messages (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    channel_type ENUM('whisper', 'guild', 'team', 'system') NOT NULL,
    sender_id BIGINT NOT NULL,
    receiver_id BIGINT,               -- 私聊目标，频道消息为 NULL
    guild_id BIGINT,                  -- 公会频道
    content TEXT NOT NULL,
    content_type ENUM('text', 'image', 'item_link', 'voice') DEFAULT 'text',
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_receiver (receiver_id, is_read, created_at),
    INDEX idx_channel (channel_type, guild_id, created_at)
);

-- 离线消息表（玩家不在线时暂存）
CREATE TABLE offline_messages (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    target_id BIGINT NOT NULL,
    message_data BLOB NOT NULL,       -- 序列化的消息
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_target (target_id, created_at)
);

-- 公会表
CREATE TABLE guilds (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL UNIQUE,
    leader_id BIGINT NOT NULL,
    level INT DEFAULT 1,
    experience BIGINT DEFAULT 0,
    member_count INT DEFAULT 1,
    max_members INT DEFAULT 50,
    announcement TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

-- 公会成员表
CREATE TABLE guild_members (
    guild_id BIGINT NOT NULL,
    player_id BIGINT NOT NULL,
    role ENUM('leader', 'officer', 'member') DEFAULT 'member',

    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
```

## 完整 C++ 聊天系统

```cpp
#include <vector>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <queue>
#include <functional>
#include <cstdint>
#include <ctime>

using PlayerId = uint32_t;
using GuildId = uint32_t;
using MessageId = uint64_t;

// ===== 频道类型 =====
enum class ChannelType {
    WORLD,        // 世界频道
    LOCAL,        // 区域频道
    WHISPER,      // 私聊
    GUILD,        // 公会
    TEAM,         // 队伍
    SYSTEM        // 系统公告
};

// ===== 聊天消息 =====
struct ChatMessage {
    MessageId id;
    PlayerId sender;
    PlayerId target;          // 私聊目标（非私聊为 0）
    ChannelType channel;
    GuildId guildId = 0;
    std::string content;
    std::string senderName;
    uint64_t timestamp;
    bool isFiltered = false;  // 是否被过滤

    enum ContentType { TEXT, IMAGE, ITEM_LINK, VOICE };
    ContentType contentType = TEXT;
};

// ===== 内容过滤系统 =====
class ContentFilter {
public:
    std::unordered_set<std::string> sensitiveWords;

    void LoadDictionary(const std::string& filepath) {
        // 从文件加载敏感词库
    }

    // 检查并过滤内容
    bool Filter(std::string& content) const {
        // 使用 Aho-Corasick 自动机进行多模式匹配
        for (const auto& word : sensitiveWords) {
            size_t pos = 0;
            while ((pos = content.find(word, pos)) != std::string::npos) {
                content.replace(pos, word.length(), std::string(word.length(), '*'));
                pos += word.length();
            }
        }
        return content.find("***") == std::string::npos;  // 返回是否包含敏感词
    }

    bool HasSensitiveContent(const std::string& content) const {
        std::string copy = content;
        return !Filter(copy);
    }
};

// ===== 频率限制器 =====
class ChatRateLimiter {
public:
    struct PlayerRate {
        std::queue<float> timestamps;
        int warningCount = 0;
    };

    float windowSeconds = 5.0f;   // 5秒窗口
    int maxMessages = 5;          // 5秒内最多5条
    std::unordered_map<PlayerId, PlayerRate> limits;

    bool Allow(PlayerId playerId, float currentTime) {
        auto& rate = limits[playerId];

        // 清理过期时间戳
        while (!rate.timestamps.empty() &&
               rate.timestamps.front() < currentTime - windowSeconds) {
            rate.timestamps.pop();
        }

        if ((int)rate.timestamps.size() >= maxMessages) {
            rate.warningCount++;
            return false;
        }

        rate.timestamps.push(currentTime);
        return true;
    }

    // 不同频道的频率限制
    bool AllowChannel(PlayerId playerId, ChannelType channel, float currentTime) {
        // 世界频道限制更严格
        if (channel == ChannelType::WORLD) {
            windowSeconds = 10.0f;
            maxMessages = 3;
        } else if (channel == ChannelType::WHISPER) {
            windowSeconds = 3.0f;
            maxMessages = 10;
        }
        return Allow(playerId, currentTime);
    }
};

// ===== 消息路由器 =====
class ChatRouter {
public:
    ContentFilter filter;
    ChatRateLimiter rateLimiter;

    // 消息回调
    using MessageCallback = std::function<void(const ChatMessage&)>;
    std::unordered_map<PlayerId, MessageCallback> playerCallbacks;

    // 发送消息
    bool RouteMessage(ChatMessage& msg, float currentTime) {
        // 1. 内容过滤
        if (filter.HasSensitiveContent(msg.content)) {
            if (!filter.Filter(msg.content)) {
                SendError(msg.sender, "消息包含敏感内容，已被过滤");
                return false;
            }
            msg.isFiltered = true;
        }

        // 2. 频率检查
        if (!rateLimiter.AllowChannel(msg.sender, msg.channel, currentTime)) {
            SendError(msg.sender, "发送过于频繁，请稍后再试");
            return false;
        }

        // 3. 按频道分发
        switch (msg.channel) {
            case ChannelType::WORLD:
                BroadcastAll(msg);
                break;
            case ChannelType::LOCAL:
                BroadcastAOI(msg.sender, msg);
                break;
            case ChannelType::WHISPER:
                DeliverWhisper(msg);
                break;
            case ChannelType::GUILD:
                BroadcastGuild(msg.guildId, msg);
                break;
            case ChannelType::TEAM:
                BroadcastTeam(msg.sender, msg);
                break;
            case ChannelType::SYSTEM:
                BroadcastAll(msg);
                break;
        }

        return true;
    }

    // 注册玩家的消息接收回调
    void RegisterPlayer(PlayerId playerId, MessageCallback callback) {
        playerCallbacks[playerId] = callback;
    }

    void UnregisterPlayer(PlayerId playerId) {
        playerCallbacks.erase(playerId);
    }

private:
    void BroadcastAll(const ChatMessage& msg) {
        for (const auto& [pid, cb] : playerCallbacks) {
            cb(msg);
        }
    }

    void BroadcastAOI(PlayerId sender, const ChatMessage& msg) {
        // 获取发送者附近的玩家
        auto nearbyPlayers = GetNearbyPlayers(sender, 100.0f);
        for (PlayerId pid : nearbyPlayers) {
            auto it = playerCallbacks.find(pid);
            if (it != playerCallbacks.end()) {
                it->second(msg);
            }
        }
    }

    void DeliverWhisper(const ChatMessage& msg) {
        auto it = playerCallbacks.find(msg.target);
        if (it != playerCallbacks.end()) {
            it->second(msg);
        } else {
            // 目标不在线，存储为离线消息
            StoreOfflineMessage(msg.target, msg);
        }
    }

    void BroadcastGuild(GuildId guildId, const ChatMessage& msg) {
        auto members = GetGuildOnlineMembers(guildId);
        for (PlayerId pid : members) {
            auto it = playerCallbacks.find(pid);
            if (it != playerCallbacks.end()) {
                it->second(msg);
            }
        }
    }

    void BroadcastTeam(PlayerId sender, const ChatMessage& msg) {
        auto teammates = GetTeamMembers(sender);
        for (PlayerId pid : teammates) {
            auto it = playerCallbacks.find(pid);
            if (it != playerCallbacks.end()) {
                it->second(msg);
            }
        }
    }

    void SendError(PlayerId playerId, const std::string& errorMsg) {
        ChatMessage sysMsg;
        sysMsg.channel = ChannelType::SYSTEM;
        sysMsg.content = errorMsg;
        auto it = playerCallbacks.find(playerId);
        if (it != playerCallbacks.end()) {
            it->second(sysMsg);
        }
    }

    void StoreOfflineMessage(PlayerId target, const ChatMessage& msg) {
        // 存入数据库
    }

    std::vector<PlayerId> GetNearbyPlayers(PlayerId player, float radius) { return {}; }
    std::vector<PlayerId> GetGuildOnlineMembers(GuildId guildId) { return {}; }
    std::vector<PlayerId> GetTeamMembers(PlayerId player) { return {}; }
};
```

## 常见陷阱与解决方案

1. **敏感词过滤过于严格**：正常交流被误判。解决方案：使用模糊匹配 + 白名单 + 上下文分析。

2. **频率限制不当**：正常聊天被限制。解决方案：不同频道使用不同频率限制，组队聊天限制宽松。

3. **离线消息未限制数量**：存储爆炸。解决方案：限制每玩家最大离线消息数（如 50 条），过期自动清理。

4. **好友状态广播压力大**：大量好友上线/下线广播消耗服务端。解决方案：批量通知 + 延迟广播 + 客户端缓存。
