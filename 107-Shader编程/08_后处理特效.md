# 后处理特效

## 1. 核心理论

后处理（Post-Processing）是在场景渲染完成后对整个屏幕图像进行的图像处理操作。通过全屏Pass在片元着色器中操作屏幕像素，实现各种视觉特效。后处理的本质是**图像处理在实时渲染中的应用**。

### 1.1 全屏Pass原理

全屏Pass不需要实际的几何体，而是通过一个覆盖整个视口的三角形（或四边形）来触发每个屏幕像素的片元着色器执行。

```hlsl
// 全屏三角形顶点着色器（无需顶点缓冲区，仅3个顶点）
// 比四边形少2个顶点，且避免接缝问题
Varyings FullscreenVert(uint vertexID : SV_VertexID)
{
    Varyings output;
    // 生成覆盖[-1,1] NDC空间的超大三角形
    output.positionCS = float4(
        (vertexID == 2) ? 3.0 : -1.0,
        (vertexID == 0) ? -3.0 : 1.0,
        0.0, 1.0
    );
    // UV坐标对应计算
    output.uv = float2(
        (vertexID == 2) ? 2.0 : 0.0,
        (vertexID == 0) ? 2.0 : 0.0
    );
    return output;
}
```

```
全屏三角形的空间覆盖：

        (2,1)
        /\
       /  \
      /    \
(-1,-1)----(3,-1)

实际覆盖远超[-1,1]视口，但光栅化会自动裁剪到视口
```

## 2. Bloom泛光

Bloom模拟明亮区域的光溢出效果，使光源和高亮区域产生辉光。

### 2.1 Bloom流水线

```
Bloom完整流程：
原始场景 (HDR)
    ↓
1. 阈值提取（Brightness Threshold）
   保留亮度 > threshold 的像素
    ↓
2. 下采样（Downsample）
   逐步降低分辨率（如 1/2, 1/4, 1/8, 1/16...）
   每级使用13-tap滤波器平滑
    ↓
3. 高斯模糊（Gaussian Blur）
   水平模糊 + 垂直模糊（分离式，各一次Pass）
   或使用Kawase/Mipmap模糊
    ↓
4. 上采样（Upsample）
   逐步恢复分辨率，与上级结果混合
    ↓
5. 合成（Composite）
   原始图像 + Bloom × 强度 = 最终图像
```

### 2.2 GLSL实现

```glsl
// Pass 1：亮度阈值提取
uniform sampler2D u_SceneColor;
uniform float u_Threshold;

in vec2 v_UV;
out vec4 fragColor;

void main()
{
    vec3 color = texture(u_SceneColor, v_UV).rgb;
    float brightness = dot(color, vec3(0.2126, 0.7152, 0.0722)); // 相对亮度

    // 软阈值提取（避免硬切边）
    float contribution = max(brightness - u_Threshold, 0.0);
    contribution /= max(brightness, 0.0001); // 归一化保留颜色比例
    vec3 bloom = color * contribution;

    fragColor = vec4(bloom, 1.0);
}

// Pass 2：高斯模糊（水平方向）
uniform sampler2D u_InputTex;
uniform vec2 u_TexelSize; // 1.0 / textureSize
uniform int u_BlurRadius;

void BlurHorizontal()
{
    // 高斯核权重（5-tap归一化）
    float weights[5] = float[](0.227027, 0.1945946, 0.1216216, 0.054054, 0.016216);

    vec3 result = texture(u_InputTex, v_UV).rgb * weights[0];

    for (int i = 1; i < 5; i++)
    {
        float offset = float(i) * u_TexelSize.x;
        result += texture(u_InputTex, v_UV + vec2(offset, 0.0)).rgb * weights[i];
        result += texture(u_InputTex, v_UV - vec2(offset, 0.0)).rgb * weights[i];
    }

    fragColor = vec4(result, 1.0);
}

// Pass 3：合成
uniform sampler2D u_SceneColor;
uniform sampler2D u_BloomTex;
uniform float u_BloomIntensity;

void Composite()
{
    vec3 scene = texture(u_SceneColor, v_UV).rgb;
    vec3 bloom = texture(u_BloomTex, v_UV).rgb;
    fragColor = vec4(scene + bloom * u_BloomIntensity, 1.0);
}
```

### 2.3 HLSL实现（Unity URP）

```hlsl
// 13-tap下采样滤波器（Unity URP使用的高质量下采样）
half4 Downsample13Tap(TEXTURE2D(tex), float2 uv, float2 texelSize)
{
    // 中心4个纹素的加权平均
    half4 A = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2(-1, -1));
    half4 B = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2( 0, -1));
    half4 C = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2( 1, -1));
    half4 D = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2(-1,  0));
    half4 E = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2( 0,  0));
    half4 F = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2( 1,  0));
    half4 G = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2(-1,  1));
    half4 H = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2( 0,  1));
    half4 I = SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + texelSize * float2( 1,  1));

    // 加权平均
    half4 result = (B + D + F + H) * 0.0625 + (A + C + G + I) * 0.03125 + E * 0.125;
    return result;
}

// Kawase模糊（近似高斯，仅4次采样）
half4 KawaseBlur(TEXTURE2D(tex), float2 uv, float2 texelSize, float pixelOffset)
{
    half4 o = 0;
    o += SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + float2( pixelOffset,  pixelOffset) * texelSize);
    o += SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + float2( pixelOffset, -pixelOffset) * texelSize);
    o += SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + float2(-pixelOffset,  pixelOffset) * texelSize);
    o += SAMPLE_TEXTURE2D(tex, sampler_LinearClamp, uv + float2(-pixelOffset, -pixelOffset) * texelSize);
    return o * 0.25;
}
```

## 3. 色调映射（Tone Mapping）

将HDR颜色值映射到LDR显示设备可表示的[0,1]范围。

### 3.1 Reinhard色调映射

```
Reinhard公式：
L_out = L_in / (L_in + 1)

全局Reinhard（保持亮度比例）：
L_white = 场景中最亮值
L_out = L_in × (1 + L_in / L_white²) / (1 + L_in)
```

```glsl
vec3 Reinhard(vec3 hdrColor)
{
    return hdrColor / (hdrColor + vec3(1.0));
}
```

### 3.2 ACES色调映射（电影级）

```
ACES Filmic公式：
f(x) = (x × (a × x + b)) / (x × (c × x + d) + e)
其中 a=2.51, b=0.03, c=2.43, d=0.59, e=0.14
```

```glsl
vec3 ACESFilm(vec3 x)
{
    float a = 2.51;
    float b = 0.03;
    float c = 2.43;
    float d = 0.59;
    float e = 0.14;
    return clamp((x * (a * x + b)) / (x * (c * x + d) + e), 0.0, 1.0);
}

// Unity HDRP使用的ACES近似
vec3 ACESTonemap(vec3 color)
{
    float3 a = color * (color + 0.0245786) - 0.000090537;
    float3 b = color * (0.983729 * color + 0.4329510) + 0.238081;
    return a / b;
}
```

### 3.3 Uncharted 2色调映射

```glsl
// 著名的Filmic Tone Mapping（Uncharted 2 / John Hable）
vec3 Uncharted2Partial(vec3 x)
{
    float A = 0.15; // 肩部强度
    float B = 0.50; // 线性段
    float C = 0.10; // 肩部偏移
    float D = 0.20; // 黑位偏移
    float E = 0.02; // 线性段偏移
    float F = 0.30; // 白位偏移
    return ((x*(A*x+C*B)+D*E)/(x*(A*x+B)+D*F))-E/F;
}

vec3 Uncharted2Tonemap(vec3 color)
{
    float exposureBias = 2.0;
    vec3 curr = Uncharted2Partial(color * exposureBias);
    vec3 W = vec3(11.2);
    vec3 whiteScale = vec3(1.0) / Uncharted2Partial(W);
    return curr * whiteScale;
}
```

## 4. 运动模糊（Motion Blur）

### 4.1 基于速度缓冲的运动模糊

```glsl
// 速度缓冲（Velocity Buffer）：存储每个像素的屏幕空间运动速度
uniform sampler2D u_ColorTex;
uniform sampler2D u_VelocityTex;
uniform int u_SampleCount; // 采样数，通常8-16
uniform float u_Intensity;

vec3 MotionBlur()
{
    vec2 velocity = texture(u_VelocityTex, v_UV).xy * u_Intensity;
    vec3 color = vec3(0.0);
    float totalWeight = 0.0;

    // 沿速度方向采样多次
    for (int i = 0; i < u_SampleCount; i++)
    {
        float t = float(i) / float(u_SampleCount - 1) - 0.5; // [-0.5, 0.5]
        vec2 sampleUV = v_UV + velocity * t;

        // 边界检查
        if (sampleUV.x >= 0.0 && sampleUV.x <= 1.0 &&
            sampleUV.y >= 0.0 && sampleUV.y <= 1.0)
        {
            color += texture(u_ColorTex, sampleUV).rgb;
            totalWeight += 1.0;
        }
    }

    return color / max(totalWeight, 1.0);
}
```

```hlsl
// HLSL — 速度缓冲生成（通过前一帧和当前帧的MVP矩阵）
struct Varyings
{
    float4 positionCS : SV_POSITION;
    float4 curPositionCS : TEXCOORD0;
    float4 prevPositionCS : TEXCOORD1;
};

half4 VelocityFrag(Varyings input) : SV_Target
{
    // 当前帧和前一帧的NDC坐标
    float2 curNDC = input.curPositionCS.xy / input.curPositionCS.w;
    float2 prevNDC = input.prevPositionCS.xy / input.prevPositionCS.w;

    // 屏幕空间速度
    float2 velocity = (curNDC - prevNDC) * 0.5;
    return half4(velocity, 0.0, 1.0);
}
```

## 5. 色彩分级（Color Grading）

### 5.1 LUT色彩分级

```hlsl
// 使用3D LUT（Look-Up Table）进行色彩分级
// LUT是一个 n×n 的2D纹理，编码 n³ 的3D颜色查找表
// 通常 n = 32 或 64

TEXTURE2D(_ColorLUT); // 32x32的2D纹理，排列为 32列×32行

half3 ApplyColorLUT(half3 color)
{
    // 将[0,1]颜色映射到LUT坐标
    half3 lutCoord = saturate(color) * (31.0 / 32.0) + (0.5 / 32.0);

    // 计算2D UV（将3D坐标展平到2D纹理）
    half2 uv;
    uv.x = (lutCoord.x + floor(lutCoord.z * 31.0)) / 32.0;
    uv.y = lutCoord.y;

    return SAMPLE_TEXTURE2D(_ColorLUT, sampler_LinearClamp, uv).rgb;
}
```

### 5.2 屏幕空间色彩调整

```glsl
// 对比度
vec3 AdjustContrast(vec3 color, float contrast)
{
    return (color - 0.5) * contrast + 0.5;
}

// 饱和度
vec3 AdjustSaturation(vec3 color, float saturation)
{
    float luminance = dot(color, vec3(0.2126, 0.7152, 0.0722));
    return mix(vec3(luminance), color, saturation);
}

// 色温
vec3 ColorTemperature(vec3 color, float temp)
{
    // 简化的色温调整
    float3 warm = float3(1.0, 0.9, 0.7);
    float3 cool = float3(0.7, 0.8, 1.0);
    return color * mix(cool, warm, temp);
}
```

## 6. 屏幕扭曲（Screen Distortion）

```glsl
// 正弦波扭曲（受伤、中毒效果）
vec2 ScreenDistort(vec2 uv, float time, float strength)
{
    uv.x += sin(uv.y * 10.0 + time) * strength;
    uv.y += cos(uv.x * 10.0 + time * 1.3) * strength;
    return uv;
}

// 径向扭曲（冲击波效果）
vec2 RadialDistort(vec2 uv, vec2 center, float time, float strength)
{
    vec2 dir = uv - center;
    float dist = length(dir);
    float wave = sin(dist * 20.0 - time * 5.0) * exp(-dist * 3.0) * strength;
    return uv + normalize(dir) * wave;
}

// 鱼眼镜头
vec2 Fisheye(vec2 uv, float power)
{
    vec2 center = uv - 0.5;
    float dist = length(center);
    float newDist = pow(dist, power);
    return center * (newDist / dist) + 0.5;
}
```

## 7. 完整后处理链示例（Unity URP）

```hlsl
// Unity URP自定义后处理
Shader "Custom/PostProcess"
{
    HLSLPROGRAM
    #pragma vertex FullscreenVert
    #pragma fragment frag

    #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"

    TEXTURE2D(_MainTex);
    SAMPLER(sampler_MainTex);
    float4 _MainTex_TexelSize;

    // 参数
    half _Brightness;
    half _Contrast;
    half _Saturation;
    half _VignetteIntensity;
    half _VignetteSmoothness;

    struct Varyings {
        float4 positionCS : SV_POSITION;
        float2 uv : TEXCOORD0;
    };

    Varyings FullscreenVert(uint vertexID : SV_VertexID) {
        Varyings o;
        o.positionCS = float4(
            vertexID == 2 ? 3.0 : -1.0,
            vertexID == 0 ? -3.0 : 1.0,
            0.0, 1.0);
        o.uv = float2(
            vertexID == 2 ? 2.0 : 0.0,
            vertexID == 0 ? 2.0 : 0.0);
        return o;
    }

    half4 frag(Varyings i) : SV_Target {
        half4 color = SAMPLE_TEXTURE2D(_MainTex, sampler_MainTex, i.uv);

        // 亮度
        color.rgb *= _Brightness;

        // 对比度
        color.rgb = (color.rgb - 0.5h) * _Contrast + 0.5h;

        // 饱和度
        half lum = Luminance(color.rgb);
        color.rgb = lerp(lum.xxx, color.rgb, _Saturation);

        // 暗角（Vignette）
        half2 vignetteUV = i.uv - 0.5h;
        half vignette = 1.0h - dot(vignetteUV, vignetteUV) * _VignetteIntensity;
        vignette = smoothstep(0.0h, _VignetteSmoothness, vignette);
        color.rgb *= vignette;

        // Gamma校正
        color.rgb = saturate(color.rgb);

        return color;
    }
    ENDHLSL
}
```

// 色差（Chromatic Aberration）
vec3 ChromaticAberration(sampler2D tex, vec2 uv, float amount)
{
    vec2 dir = uv - 0.5;
    float dist = length(dir);
    vec2 offset = dir * dist * amount;

    float r = texture(tex, uv + offset).r;
    float g = texture(tex, uv).g;
    float b = texture(tex, uv - offset).b;
    return vec3(r, g, b);
}

// 胶片颗粒（Film Grain）
float FilmGrain(vec2 uv, float time, float intensity)
{
    float noise = fract(sin(dot(uv + time, vec2(12.9898, 78.233))) * 43758.5453);
    return (noise - 0.5) * intensity;

## 9. 性能提示

- 每个后处理效果至少一次全屏Pass + 一次纹理读取，带宽是主要瓶颈
- 使用半分辨率（1/2宽高）处理模糊类效果，像素数减少75%
- 多个效果尽量合并到一个Pass中（如亮度+对比度+饱和度一起算）
- Bloom的下采样链使用mipmap自动降低分辨率，比手动Blit更高效
- 使用R11G11B10浮点格式代替RGBA32F可节省50%带宽
- 避免不必要的中间RT分配，复用RT池

## 10. 常见问题与调试

**问题1：Bloom光晕有明显接缝**
- 检查全屏三角形是否正确（不要用四边形，会有接缝）
- 检查UV计算是否正确处理了纹理边界

**问题2：色调映射后颜色偏暗**
- 确认输入是HDR值（使用R11G11B10F或RGBA16F格式RT）
- 检查ACES曲线的参数是否适合你的场景亮度范围

**问题3：运动模糊导致UI也被模糊**
- 将UI渲染放在后处理之后
- 或在后处理中排除UI层

**问题4：后处理链性能差**
- 使用RenderDoc/PIX查看每个Pass的带宽消耗
- 合并Pass，降低分辨率，减少采样次数

## 11. 实际使用案例

- Unity Post-Processing Stack v2和URP Volume系统集成Bloom、色调映射、景深等
- Unreal Engine的后处理材质系统允许自定义全屏效果
- 《赛博朋克2077》的后处理链包含胶片颗粒、色差、晕影等十余种效果
- 移动端游戏通常精简后处理链，只保留Bloom和色调映射
- 《荒野大镖客2》使用ACESTonemap + 高质量Bloom + 色差实现电影级画面
- Shadertoy上的许多着色器都是后处理效果的创意实验
