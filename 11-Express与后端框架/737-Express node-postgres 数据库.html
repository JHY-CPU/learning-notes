// ========== node-postgres ========== // Node.js PostgreSQL 客户端 // 安装: // npm install pg const { Pool, Client } = require('pg'); // ========== 连接池配置 ========== const pool = new Pool({ host: process.env.DB_HOST || 'localhost', port: parseInt(process.env.DB_PORT, 10) || 5432, database: process.env.DB_NAME || 'myapp', user: process.env.DB_USER || 'postgres', password: process.env.DB_PASS || 'postgres', // 连接池设置 max: 20, // 最大连接数 min: 2, // 最小空闲连接 idleTimeoutMillis: 30000, // 空闲连接超时 connectionTimeoutMillis: 5000, // 连接超时 // 自动重连 maxRetries: 3, retryDelay: 3000, }); // ========== 查询 ========== // 简单查询 const result = await pool.query('SELECT NOW()'); console.log(result.rows[0].now); // 参数化查询 (防 SQL 注入) const userResult = await pool.query( 'SELECT * FROM users WHERE email = $1 AND is_active = $2', [email, true] ); // ========== Express 集成 ========== // db.js: const { Pool } = require('pg'); const pool = new Pool({ connectionString: process.env.DATABASE_URL, ssl: process.env.NODE_ENV === 'production' ? { rejectUnauthorized: false } : false, }); // 导出 query 方法 module.exports = { query: (text, params) => pool.query(text, params), // 获取客户端 (用于事务) getClient: () => pool.connect(), // 健康检查 health: async () => { try { await pool.query('SELECT 1'); return { status: 'ok', poolSize: pool.totalCount, idle: pool.idleCount }; } catch (err) { return { status: 'error', message: err.message }; } }, }; // app.js: const db = require('./db'); app.get('/users', async (req, res) => { const result = await db.query( 'SELECT id, name, email FROM users ORDER BY created_at DESC LIMIT $1 OFFSET $2', [limit, offset] ); res.success(result.rows); }); // ========== 单个 Client (少用) ========== // 用于需要长连接或 LISTEN/NOTIFY const client = new Client({ connectionString: process.env.DATABASE_URL, }); await client.connect(); client.on('notification', (msg) => { console.log('Channel:', msg.channel, 'Payload:', msg.payload); }); await client.query('LISTEN new_user');

事务管理

// ========== PostgreSQL 事务 ========== // 多条语句原子执行 const db = require('../db'); // ========== 手动事务 ========== async function transferMoney(fromId, toId, amount) { const client = await db.getClient(); try { await client.query('BEGIN'); // 扣款 const debit = await client.query( 'UPDATE accounts SET balance = balance - $1 WHERE id = $2 AND balance >= $1 RETURNING balance', [amount, fromId] ); if (debit.rows.length === 0) { throw new Error('Insufficient funds'); } // 加款 await client.query( 'UPDATE accounts SET balance = balance + $1 WHERE id = $2', [amount, toId] ); // 记录交易 await client.query( 'INSERT INTO transactions (from_id, to_id, amount) VALUES ($1, $2, $3)', [fromId, toId, amount] ); await client.query('COMMIT'); return { success: true }; } catch (err) { await client.query('ROLLBACK'); throw err; } finally { client.release(); // 释放回连接池 } } // ========== 事务辅助函数 ========== async function transaction(callback) { const client = await db.getClient(); try { await client.query('BEGIN'); const result = await callback(client); await client.query('COMMIT'); return result; } catch (err) { await client.query('ROLLBACK'); throw err; } finally { client.release(); } } // 使用: app.post('/transfer', async (req, res) => { const result = await transaction(async (client) => { const { fromId, toId, amount } = req.body; await client.query( 'UPDATE accounts SET balance = balance - $1 WHERE id = $2', [amount, fromId] ); await client.query( 'UPDATE accounts SET balance = balance + $1 WHERE id = $2', [amount, toId] ); return { success: true }; }); res.success(result); }); // ========== SAVEPOINT ========== await client.query('SAVEPOINT sp1'); await client.query('ROLLBACK TO SAVEPOINT sp1'); // 部分回滚 // ========== 批量插入 ========== async function bulkInsertUsers(users) { const values = []; const params = []; let i = 1; for (const user of users) { values.push(`($${i}, $${i + 1}, $${i + 2})`); params.push(user.name, user.email, user.password); i += 3; } const result = await db.query( `INSERT INTO users (name, email, password_hash) VALUES ${values.join(', ')} ON CONFLICT (email) DO NOTHING RETURNING id`, params ); return result.rows; }

// ========== PostgreSQL 高级特性 ========== // ========== 迁移 (手动) ========== // migrations/001_create_users.js: async function up(client) { await client.query(` CREATE TABLE IF NOT EXISTS users ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, email VARCHAR(255) UNIQUE NOT NULL, password_hash VARCHAR(255) NOT NULL, role VARCHAR(50) DEFAULT 'user', is_active BOOLEAN DEFAULT true, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX idx_users_email ON users(email); CREATE INDEX idx_users_created_at ON users(created_at); `); } async function down(client) { await client.query('DROP TABLE IF EXISTS users'); } // 运行迁移: // const client = await pool.connect(); // await up(client); // client.release(); // ========== LISTEN/NOTIFY (实时通知) ========== // 数据库事件通知 // 监听 (在初始化时): app.listen(3000, async () => { const client = await db.getClient(); await client.query('LISTEN new_order'); await client.query('LISTEN user_updated'); client.on('notification', (msg) => { const payload = JSON.parse(msg.payload); console.log(`Channel ${msg.channel}:`, payload); if (msg.channel === 'new_order') { // 通知相关用户 io.to(`user:${payload.userId}`).emit('order:new', payload); } }); }); // 触发器发送通知: // CREATE OR REPLACE FUNCTION notify_new_order() // RETURNS TRIGGER AS $$ // BEGIN // PERFORM pg_notify('new_order', row_to_json(NEW)::text); // RETURN NEW; // END; // $$ LANGUAGE plpgsql; // // CREATE TRIGGER order_after_insert // AFTER INSERT ON orders // FOR EACH ROW EXECUTE FUNCTION notify_new_order(); // ========== 连接池监控 ========== setInterval(async () => { const { totalCount, idleCount, waitingCount } = pool; logger.info('Pool stats', { totalCount, idleCount, waitingCount }); }, 60000); // ========== 错误处理 ========== pool.on('error', (err) => { logger.error('Unexpected pool error', { err }); }); // ========== 关闭 ========== process.on('SIGTERM', async () => { await pool.end(); });
