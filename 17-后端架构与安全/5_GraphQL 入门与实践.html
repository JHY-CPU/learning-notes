GraphQL 核心概念

// ========== GraphQL vs REST ========== // REST: 多个端点, 固定返回结构 // GET /users → {id, name, email, ...} // GET /users/1/orders → [...] // GraphQL: 单个端点, 客户端指定返回字段 // POST /graphql // query { user(id: 1) { name email } } // → { "user": { "name": "John", "email": "john@test.com" } } // ========== Schema 定义 ========== // type Query { // 查询入口 // user(id: ID!): User // users(page: Int, limit: Int): [User!]! // } // // type Mutation { // 写操作入口 // createUser(input: CreateUserInput!): User! // updateUser(id: ID!, input: UpdateUserInput!): User! // deleteUser(id: ID!): Boolean! // } // // type Subscription { // 订阅 (实时) // userCreated: User! // userUpdated: User! // } // // type User { // id: ID! // name: String! // email: String! // posts: [Post!]! // 关联 // createdAt: DateTime! // } // // type Post { // id: ID! // title: String! // content: String! // author: User! // } // // input CreateUserInput { // name: String! // email: String! @constraint(format: "email") // password: String! @constraint(minLength: 8) // } // ========== 查询示例 ========== // 查询特定字段: // query { // user(id: 1) { // name // email // posts { // title // } // } // } // 查询 + 变量: // query GetUser($id: ID!) { // user(id: $id) { // name // email // } // } // 变量: { "id": 1 } // 多个查询: // query { // users(page: 1, limit: 10) { id name } // user(id: 1) { email } // }

N+1 问题与 DataLoader

// ========== N+1 问题 ========== // 查询: { users { posts { title } } } // 1 次查用户列表 → N 次查每个用户的文章 // 共 1+N 次查询! // ========== DataLoader 解决 N+1 ========== // DataLoader: 批处理 + 缓存 import DataLoader from 'dataloader'; function createLoaders(db) { return { userById: new DataLoader(async (ids) => { const users = await db.users.findByIds(ids); return ids.map(id => users.find(u => u.id === id)); }), postsByAuthorId: new DataLoader(async (authorIds) => { const posts = await db.posts.findByAuthorIds(authorIds); return authorIds.map(id => posts.filter(p => p.authorId === id) ); }), }; } // Resolver 中使用: const resolvers = { User: { posts: async (parent, _, { dataLoaders }) => { return dataLoaders.postsByAuthorId.load(parent.id); }, }, Query: { users: async (_, args, { dataLoaders }) => { const users = await db.users.findAll(); // 只 load 不会立即执行 users.forEach(u => dataLoaders.postsByAuthorId.load(u.id)); return users; }, }, }; // 现在: 1 次用户 + 1 次文章 = 2 次查询 // DataLoader 自动合并同一 tick 内的 load 请求 // ========== GraphQL 安全 ========== // 1. 查询深度限制 // import depthLimit from 'graphql-depth-limit'; // server.start({ validationRules: [depthLimit(7)] }) // 2. 查询复杂度限制 // 根据字段权重计算复杂度 // query { users { posts { comments { likes } } } } → 复杂度 10 // 3. 超时 // requestTimeout: 10000 // 4. 持久化查询 (Persisted Queries) // 只允许预定义的查询 // 5. 认证 + 授权 // context 中验证用户 // resolver 中检查权限
