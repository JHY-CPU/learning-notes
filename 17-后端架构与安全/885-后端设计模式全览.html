行为型模式

// ========== 策略模式 (Strategy) ========== // 定义算法族, 运行时切换 // 适用: 多种验证/认证/支付/压缩 // type Validator interface { // Validate(value string) error // } // // type EmailValidator struct{} // func (v *EmailValidator) Validate(email string) error { // if !strings.Contains(email, "@") { // return errors.New("invalid email") // } // return nil // } // // type LengthValidator struct { // Min, Max int // } // func (v *LengthValidator) Validate(s string) error { // if len(s) < v.Min || len(s) > v.Max { // return fmt.Errorf("length must be %d-%d", v.Min, v.Max) // } // return nil // } // // // 使用: // validators := []Validator{EmailValidator{}, &LengthValidator{Min: 6, Max: 100}} // for _, v := range validators { // if err := v.Validate(input); err != nil { ... } // } // ========== 观察者模式 (Observer) ========== // 一对多依赖, 状态变化通知所有观察者 // 适用: 事件系统, 消息通知 // type EventBus struct { // listeners map[string][]func(Event) // mu sync.RWMutex // } // // func (b *EventBus) Subscribe(event string, handler func(Event)) { // b.mu.Lock() // defer b.mu.Unlock() // b.listeners[event] = append(b.listeners[event], handler) // } // // func (b *EventBus) Publish(event Event) { // b.mu.RLock() // handlers := b.listeners[event.Type] // b.mu.RUnlock() // for _, h := range handlers { // h(event) // 可改为 goroutine // } // } // // // 使用: // bus := &EventBus{listeners: make(map[string][]func(Event))} // bus.Subscribe("order.created", func(e Event) { // sendEmail(e.Data) // }) // bus.Publish(Event{Type: "order.created", Data: order}) // ========== 命令模式 (Command) ========== // 将请求封装为对象 // 适用: 队列/撤销/事务 // type Command interface { // Execute() error // Undo() error // } // // type CreateOrderCommand struct { // orderID string // repo OrderRepository // items []Item // } // // func (c *CreateOrderCommand) Execute() error { // return c.repo.Save(c.orderID, c.items) // } // // func (c *CreateOrderCommand) Undo() error { // return c.repo.Delete(c.orderID) // } // ========== 模板方法 (Template Method) ========== // 定义算法骨架, 子类实现具体步骤 // 适用: 框架流程, 数据处理 // type DataExporter interface { // FetchData() ([]Record, error) // Format(records []Record) ([]byte, error) // Write(data []byte) error // } // // func Export(e DataExporter) error { // data, err := e.FetchData() // if err != nil { return err } // // formatted, err := e.Format(data) // if err != nil { return err } // // return e.Write(formatted) // } // // CSV 导出: FetchData → Format CSV → Write file // JSON 导出: FetchData → Format JSON → Write HTTP

架构型与弹性模式

// ========== 中间件链 (Middleware Chain) ========== // 请求处理管道, 每个中间件决定是否传至下一个 // 适用: Web 框架, 认证/日志/限流 // type Middleware func(Handler) Handler // type Handler func(ctx *Context) // // func Chain(middlewares ...Middleware) Middleware { // return func(final Handler) Handler { // for i := len(middlewares) - 1; i >= 0; i-- { // final = middlewares[i](final) // } // return final // } // } // // func Logger(next Handler) Handler { // return func(ctx *Context) { // log.Printf("request: %s", ctx.Path) // next(ctx) // } // } // // func Auth(next Handler) Handler { // return func(ctx *Context) { // if !isValid(ctx.Token) { // ctx.Status = 401 // return // 终止链 // } // next(ctx) // } // } // ========== 依赖注入 (DI) ========== // 外部提供依赖, 而非内部创建 // 适用: 解耦, 测试 // type UserService struct { // repo UserRepository // 依赖接口, 非具体实现 // cache CacheClient // } // // func NewUserService(repo UserRepository, cache CacheClient) *UserService { // return &UserService{repo: repo, cache: cache} // } // // // 测试: 注入 mock // func TestUserService(t *testing.T) { // mockRepo := new(MockUserRepository) // svc := NewUserService(mockRepo, nil) // } // ========== 管道模式 (Pipeline) ========== // 数据经过多个处理步骤 // 适用: 数据处理, ETL // type PipeFunc func(input interface{}) (output interface{}, err error) // // type Pipeline struct { // stages []PipeFunc // } // // func (p *Pipeline) Add(fn PipeFunc) *Pipeline { // p.stages = append(p.stages, fn) // return p // } // // func (p *Pipeline) Execute(input interface{}) (interface{}, error) { // data := input // for _, stage := range p.stages { // var err error // data, err = stage(data) // if err != nil { return nil, err } // } // return data, nil // } // ========== 断路器 (Circuit Breaker) ========== // 保护下游故障服务不被持续调用 // type CircuitBreaker struct { // state string // closed / open / half-open // failureCount int // threshold int // timeout time.Duration // lastFailure time.Time // } // // func (cb *CircuitBreaker) Call(fn func() error) error { // switch cb.state { // case "open": // if time.Since(cb.lastFailure) > cb.timeout { // cb.state = "half-open" // 尝试恢复 // } else { // return ErrCircuitOpen // 快速失败 // } // case "half-open": // // 允许一个请求通过 // } // // if err := fn(); err != nil { // cb.failureCount++ // cb.lastFailure = time.Now() // if cb.failureCount >= cb.threshold { // cb.state = "open" // } // return err // } // // // 成功 → 重置 // cb.failureCount = 0 // cb.state = "closed" // return nil // } // // 状态: closed (正常) → open (断开) → half-open (尝试) → closed (恢复)
