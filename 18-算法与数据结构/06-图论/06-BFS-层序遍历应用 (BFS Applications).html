分层 BFS

function levelOrder(graph, start) { const queue = [start], visited = new Set([start]); const levels = []; while (queue.length) { const levelSize = queue.length; const currentLevel = []; for (let i = 0; i < levelSize; i++) { const v = queue.shift(); currentLevel.push(v); for (const w of graph[v]) if (!visited.has(w)) { visited.add(w); queue.push(w); } } levels.push(currentLevel); } return levels; }
