// React Fiber 架构中的链表
class FiberNode {
  constructor(tag, key) {
    this.tag = tag;      // 组件类型
    this.key = key;      // 唯一标识
    this.child = null;   // 第一个子节点
    this.sibling = null; // 下一个兄弟节点
    this.return = null;  // 父节点
    this.stateNode = null;
    this.pendingProps = null;
    this.memoizedProps = null;
    this.memoizedState = null;
    this.effectTag = null;
  }
}
