# 装饰器工厂

## 一、概念说明

装饰器工厂是返回装饰器的函数，允许向装饰器传递参数。通过工厂模式，可以创建可配置的装饰器，根据参数调整装饰行为。装饰器工厂让装饰器更加灵活和可复用。

## 二、具体用法

### 2.1 带参数的装饰器工厂

```typescript
// 装饰器工厂：返回装饰器函数
function log(prefix: string) {
  return function (
    target: any,
    context: ClassMethodDecoratorContext
  ) {
    return function (this: any, ...args: any[]) {
      console.log(`${prefix} 调用 ${String(context.name)}`);
      return (target as Function).apply(this, args);
    };
  };
}

class Service {
  @log("[INFO]")
  fetch(): string {
    return "数据";
  }

  @log("[DEBUG]")
  process(): void {
    // 处理逻辑
  }
}

const service = new Service();
service.fetch();
// 输出: [INFO] 调用 fetch
```

### 2.2 属性验证装饰器

```typescript
function min(minValue: number) {
  return function (value: any, context: ClassFieldDecoratorContext) {
    return function (initialValue: number) {
      if (initialValue < minValue) {
        throw new Error(`${String(context.name)} 必须 >= ${minValue}`);
      }
      return initialValue;
    };
  };
}

class Product {
  @min(0)
  price = 99.9;

  @min(1)
  quantity = 10;
}

const product = new Product();
console.log(product.price, product.quantity);
// 输出: 99.9 10
```

### 2.3 性能计时装饰器

```typescript
function timing(label: string) {
  return function (
    target: any,
    context: ClassMethodDecoratorContext
  ) {
    return function (this: any, ...args: any[]) {
      const start = performance.now();
      const result = target.apply(this, args);
      const end = performance.now();
      console.log(`${label}: ${(end - start).toFixed(2)}ms`);
      return result;
    };
  };
}

class MathOps {
  @timing("阶乘计算")
  factorial(n: number): number {
    let result = 1;
    for (let i = 2; i <= n; i++) result *= i;
    return result;
  }
}

const ops = new MathOps();
ops.factorial(20);
// 输出: 阶乘计算: 0.01ms
```

### 2.4 权限检查装饰器

```typescript
function requireAuth(role: "admin" | "user") {
  return function (
    target: any,
    context: ClassMethodDecoratorContext
  ) {
    return function (this: any, ...args: any[]) {
      // 模拟权限检查
      const currentUserRole = "admin"; // 实际从 context 获取
      if (currentUserRole !== role) {
        throw new Error(`需要 ${role} 权限`);
      }
      return target.apply(this, args);
    };
  };
}

class AdminPanel {
  @requireAuth("admin")
  deleteUser(id: number): void {
    console.log(`删除用户 ${id}`);
  }
}

const panel = new AdminPanel();
panel.deleteUser(123);
// 输出: 删除用户 123
```

### 2.5 链式装饰器

```typescript
// 多个装饰器可以链式使用
function readonly() {
  return function (target: any, context: ClassMethodDecoratorContext) {
    return function (this: any, ...args: any[]) {
      console.log(`${String(context.name)} 是只读的`);
      return target.apply(this, args);
    };
  };
}

function log2() {
  return function (target: any, context: ClassMethodDecoratorContext) {
    return function (this: any, ...args: any[]) {
      console.log(`调用 ${String(context.name)}`);
      return target.apply(this, args);
    };
  };
}

class Config {
  @readonly()
  @log2()
  getValue() {
    return 42;
  }
}
// 执行顺序: log2 → readonly → 原方法
```

## 三、注意事项与常见陷阱

1. **三层函数嵌套**：工厂返回装饰器，装饰器返回替换函数
2. **参数在工厂调用时确定**：`@log("[INFO]")` 中的参数在装饰时就固定了
3. **装饰器工厂可链式调用**：`@a() @b() class C {}`，执行顺序从下到上
4. **元数据传递**：可以配合 `Symbol.metadata` 传递元信息
5. **类型安全**：工厂参数应有明确类型，避免使用 `any`
6. **NestJS 广泛使用**：`@Controller()`、`@Get()`、`@Post()` 等都是装饰器工厂
