# 组合式函数类型

## 一、概念说明

组合式函数（Composables）是 Vue 3 中复用逻辑的核心模式，类似于 React 的自定义 Hook。TypeScript 可以精确地推断组合式函数的返回值类型，让使用者获得完整的类型提示。

## 二、具体用法

### 2.1 基本组合式函数

```typescript
// composables/useCounter.ts
import { ref, computed } from 'vue';

export function useCounter(initialValue = 0) {
  const count = ref(initialValue);

  const doubled = computed(() => count.value * 2);
  const isPositive = computed(() => count.value > 0);

  const increment = () => count.value++;
  const decrement = () => count.value--;
  const reset = () => (count.value = initialValue);

  // 返回类型自动推断
  return {
    count,        // Ref<number>
    doubled,      // ComputedRef<number>
    isPositive,   // ComputedRef<boolean>
    increment,    // () => void
    decrement,
    reset,
  };
}
```

### 2.2 泛型组合式函数

```typescript
// composables/useFetch.ts
import { ref, shallowRef } from 'vue';

interface UseFetchReturn<T> {
  data: Readonly<import('vue').Ref<T | null>>;
  loading: Readonly<import('vue').Ref<boolean>>;
  error: Readonly<import('vue').Ref<Error | null>>;
  execute: () => Promise<void>;
}

export function useFetch<T>(url: string): UseFetchReturn<T> {
  const data = shallowRef<T | null>(null);
  const loading = ref(false);
  const error = ref<Error | null>(null);

  const execute = async () => {
    loading.value = true;
    error.value = null;

    try {
      const response = await fetch(url);
      if (!response.ok) throw new Error(`HTTP ${response.status}`);
      data.value = await response.json();
    } catch (err) {
      error.value = err instanceof Error ? err : new Error(String(err));
    } finally {
      loading.value = false;
    }
  };

  return { data, loading, error, execute };
}

// 使用 — 类型自动推断
// const { data, loading } = useFetch<User[]>('/api/users');
// data.value 类型是 User[] | null
```

### 2.3 带状态管理的组合式函数

```typescript
// composables/useToggle.ts
import { ref } from 'vue';

export function useToggle(initialValue = false) {
  const value = ref(initialValue);

  const toggle = () => (value.value = !value.value);
  const setTrue = () => (value.value = true);
  const setFalse = () => (value.value = false);

  return {
    value,    // Ref<boolean>
    toggle,   // () => void
    setTrue,
    setFalse,
  } as const; // as const 确保精确类型
}
```

### 2.4 依赖注入的组合式函数

```typescript
// composables/useAuth.ts
import { inject, provide, type InjectionKey } from 'vue';

interface AuthContext {
  user: Ref<{ name: string } | null>;
  login: (username: string, password: string) => Promise<void>;
  logout: () => void;
}

const AUTH_KEY: InjectionKey<AuthContext> = Symbol('auth');

export function provideAuth() {
  const user = ref<{ name: string } | null>(null);

  const login = async (username: string, password: string) => {
    // 登录逻辑
    user.value = { name: username };
  };

  const logout = () => {
    user.value = null;
  };

  const context: AuthContext = { user, login, logout };
  provide(AUTH_KEY, context);
  return context;
}

export function useAuth(): AuthContext {
  const auth = inject(AUTH_KEY);
  if (!auth) {
    throw new Error('useAuth 必须在 provideAuth 内使用');
  }
  return auth;
}
```

### 2.5 组合多个组合式函数

```typescript
// composables/useUser.ts
import { useFetch } from './useFetch';
import { useToggle } from './useToggle';

export function useUser(userId: number) {
  const { data: user, loading, execute } = useFetch<User>(`/api/users/${userId}`);
  const { value: isEditing, toggle: toggleEdit } = useToggle();

  // 组合逻辑
  const save = async () => {
    // 保存逻辑
    toggleEdit();
  };

  return { user, loading, isEditing, toggleEdit, save, refresh: execute };
}
```

## 三、注意事项与常见陷阱

1. **返回对象的类型自动推断**：通常不需要显式标注返回类型
2. **`ref` 和 `computed` 的类型自动推断**：不要手动标注
3. **使用 `as const` 精确化返回类型**：确保字面量类型不被拓宽
4. **`shallowRef` 用于大数据**：避免深层响应式代理的性能开销
5. **`InjectionKey` 类型化依赖注入**：确保 inject 的类型安全
