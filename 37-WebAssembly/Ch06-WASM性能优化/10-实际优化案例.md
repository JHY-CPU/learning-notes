### 2.1 图像处理优化

```rust
// 优化前：逐像素处理
pub fn blur_image_naive(input: &[u8], width: usize, height: usize) -> Vec<u8> {
    let mut output = vec![0u8; input.len()];
    for y in 1..height-1 {
        for x in 1..width-1 {
            for c in 0..4 { // RGBA
                let mut sum = 0u32;
                for dy in -1..=1 {
                    for dx in -1..=1 {
                        let idx = ((y as isize + dy) * width as isize + x as isize + dx) as usize * 4 + c;
                        sum += input[idx] as u32;
                    }
                }
                output[(y * width + x) * 4 + c] = (sum / 9) as u8;
            }
        }
    }
    output
}

// 优化后：SIMD + 缓存友好
pub fn blur_image_optimized(input: &[u8], width: usize, height: usize) -> Vec<u8> {
    let mut output = vec![0u8; input.len()];
    // 分离通道处理，提高缓存命中率
    for c in 0..4 {
        for y in 1..height-1 {
            for x in (1..width-1).step_by(4) {
                // 使用 SIMD 同时处理 4 个像素
                let mut sum = [0u32; 4];
                for dy in -1..=1 {
                    for dx in -1..=1 {
                        for i in 0..4 {
                            let px = x + i;
                            let idx = ((y as isize + dy) * width as isize + px as isize) as usize * 4 + c;
                            sum[i] += input[idx] as u32;
                        }
                    }
                }
                for i in 0..4 {
                    let px = x + i;
                    output[(y * width + px) * 4 + c] = (sum[i] / 9) as u8;
                }
            }
        }
    }
    output
}
```

### 2.3 音频处理优化

```rust
// 使用环形缓冲区避免频繁分配
pub struct AudioBuffer {
    data: Vec<f32>,
    read_pos: usize,
    write_pos: usize,
    capacity: usize,
}

impl AudioBuffer {
    pub fn new(capacity: usize) -> Self {
        AudioBuffer {
            data: vec![0.0; capacity],
            read_pos: 0,
            write_pos: 0,
            capacity,
        }
    }

    pub fn write(&mut self, samples: &[f32]) {
        for &sample in samples {
            self.data[self.write_pos] = sample;
            self.write_pos = (self.write_pos + 1) % self.capacity;
        }
    }

    pub fn read(&mut self, output: &mut [f32]) {
        for sample in output.iter_mut() {
            *sample = self.data[self.read_pos];
            self.read_pos = (self.read_pos + 1) % self.capacity;
        }
    }
}
```

#[wasm_bindgen]
pub fn benchmark(label: &str, iterations: usize, f: &js_sys::Function) -> f64 {
    let start = web_sys::window().unwrap().performance().unwrap().now();
    for _ in 0..iterations {
        f.call0(&JsValue::null()).unwrap();
    }
    let elapsed = web_sys::window().unwrap().performance().unwrap().now() - start;
    web_sys::console::log_1(
        &format!("{}: {:.2}ms ({} 次迭代, {:.4}ms/次)", label, elapsed, iterations, elapsed / iterations as f64).into()
    );
    elapsed
}

## 三、注意事项与常见陷阱

1. **测量先行**：先用 Profile 工具确认瓶颈，不要盲目优化
2. **优化优先级**：算法 > 数据结构 > 微优化
3. **真实数据**：用真实数据测试，不要用合成数据
4. **多平台测试**：不同硬件/浏览器性能差异大
5. **持续监控**：建立性能回归测试，防止优化被破坏
