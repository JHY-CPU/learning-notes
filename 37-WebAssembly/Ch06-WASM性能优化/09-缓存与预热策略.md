# 缓存与预热策略

## 一、概念说明

WASM 模块编译和实例化有开销。通过缓存编译结果和预热策略可减少用户感知的延迟。

```javascript
// 使用 IndexedDB 缓存编译后的模块
async function cacheModule(name, bytes) {
    const db = await openDB('wasm-cache', 1, {
        upgrade(db) { db.createObjectStore('modules'); }
    });
    const module = await WebAssembly.compile(bytes);
    await db.put('modules', module, name);
    return module;
}
```

## 二、具体用法

### 2.1 Service Worker 缓存

```javascript
// sw.js - 缓存 WASM 文件
self.addEventListener('fetch', (event) => {
    if (event.request.url.endsWith('.wasm')) {
        event.respondWith(
            caches.open('wasm-v1').then(cache =>
                cache.match(event.request).then(cached =>
                    cached || fetch(event.request).then(resp => {
                        cache.put(event.request, resp.clone());
                        return resp;
                    })
                )
            )
        );
    }
});
```

### 2.3 分层加载

```javascript
class WasmLoader {
    constructor() {
        this.modules = new Map();
        this.loading = new Map();
    }

    // 关键模块：立即加载
    async loadCritical() {
        const [core, renderer] = await Promise.all([
            this._load('core'),
            this._load('renderer'),
        ]);
        return { core, renderer };
    }

    // 非关键模块：空闲时加载
    loadLazy(name) {
        if ('requestIdleCallback' in window) {
            requestIdleCallback(() => this._load(name));
        } else {
            setTimeout(() => this._load(name), 100);
        }
    }

    // 按需加载
    async loadOnDemand(name) {
        if (this.modules.has(name)) return this.modules.get(name);
        if (this.loading.has(name)) return this.loading.get(name);

        const promise = this._doLoad(name);
        this.loading.set(name, promise);
        const mod = await promise;
        this.loading.delete(name);
        this.modules.set(name, mod);
        return mod;
    }
}
```

### 2.4 编译结果缓存

```javascript
const CACHE_KEY = 'wasm-modules-v1';

async function getCachedModule(name, url) {
    // 检查 Cache API
    const cache = await caches.open(CACHE_KEY);
    const cached = await cache.match(name);

    if (cached) {
        const bytes = await cached.arrayBuffer();
        return await WebAssembly.compile(bytes);
    }

    // 首次加载，下载并缓存
    const resp = await fetch(url);
    await cache.put(name, resp.clone());
    const bytes = await resp.arrayBuffer();
    return await WebAssembly.compile(bytes);
}
```

### 2.5 实例池

```javascript
class InstancePool {
    constructor(module, imports, poolSize = 4) {
        this.instances = [];
        this.available = [];

        for (let i = 0; i < poolSize; i++) {
            const inst = WebAssembly.instantiate(module, imports);
            this.instances.push(inst);
            this.available.push(inst);
        }
    }

    async acquire() {
        if (this.available.length === 0) {
            return await WebAssembly.instantiate(this.module, this.imports);
        }
        return await this.available.pop();
    }

    release(instance) {
        this.available.push(instance);
    }
}
```

## 三、注意事项与常见陷阱

1. **缓存失效**：WASM 文件更新后需清除旧缓存
2. **存储空间**：IndexedDB 有存储限制，需管理缓存大小
3. **HTTPS 要求**：Service Worker 需要 HTTPS
4. **版本管理**：缓存 key 需包含版本号，避免使用旧版本
5. **预热数据**：选择有代表性的输入数据进行预热
