## 二、具体用法

### 2.1 项目结构

```
my-wasm-project/
├── src/
│   ├── lib.rs (或 main.c)
│   └── utils.rs
├── tests/
│   ├── unit/
│   └── integration/
├── dist/
│   ├── module.wasm
│   └── module.js
├── scripts/
│   ├── build.sh
│   └── test.sh
├── Cargo.toml (或 Makefile)
├── package.json
└── README.md
```

### 2.2 构建脚本

```bash
#!/bin/bash
# build.sh

# 清理
rm -rf dist/

# 构建
if [ "$1" = "release" ]; then
  wasm-pack build --release
else
  wasm-pack build --dev
fi

# 优化
wasm-opt -O3 dist/module.wasm -o dist/module.opt.wasm

# 测试
npm test

# 打包大小检查
wc -c dist/module.wasm
```

### 2.3 CI/CD 集成

```yaml
# GitHub Actions
name: WASM CI
on: [push, pull_request]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Install Rust
        uses: actions-rs/toolchain@v1
        with:
          toolchain: stable
      - name: Build WASM
        run: wasm-pack build --release
      - name: Test
        run: npm test
      - name: Upload artifact
        uses: actions/upload-artifact@v3
        with:
          name: wasm-module
          path: dist/module.wasm
```

### 2.4 版本管理


## 五、性能监控

```bash
# 在构建时记录包大小
#!/bin/bash
SIZE=$(wc -c < dist/module.wasm)
echo "WASM size: ${SIZE} bytes" >> build.log

# 设置大小阈值
MAX_SIZE=500000  # 500KB
if [ "$SIZE" -gt "$MAX_SIZE" ]; then
  echo "ERROR: WASM exceeds ${MAX_SIZE} bytes!"
  exit 1
fi

# 使用 twiggy 分析代码大小
twiggy top dist/module.wasm -n 20
```

## 六、本地开发环境

```json
// package.json 开发脚本
{
  "scripts": {
    "build": "wasm-pack build --target bundler",
    "build:release": "wasm-pack build --release --target web",
    "build:optimize": "wasm-pack build --release --target web && wasm-opt -Oz -o pkg/optimized.wasm pkg/my_lib_bg.wasm",
    "test": "wasm-pack test --headless --chrome",
    "test:browser": "wasm-pack test --chrome",
    "dev": "webpack serve --mode development",
    "lint": "cargo clippy -- -D warnings",
    "fmt": "cargo fmt",
    "size": "twiggy top pkg/my_lib_bg.wasm"
  }
}
```

## 七、环境配置

```bash
# .env 文件（不提交到版本控制）
WASM_OPT_LEVEL=3
WASM_PACK_TARGET=web
CARGO_PROFILE=release

# 使用 direnv 自动加载环境
# .envrc
export RUSTUP_HOME=$HOME/.rustup
export CARGO_HOME=$HOME/.cargo
export PATH=$CARGO_HOME/bin:$PATH

# Docker 环境一致性
# Dockerfile
FROM rust:latest
RUN cargo install wasm-pack
RUN apt-get update && apt-get install -y binaryen
WORKDIR /app
COPY . .
RUN wasm-pack build --release
```
