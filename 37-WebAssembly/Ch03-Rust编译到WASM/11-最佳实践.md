### 2.3 错误处理

```rust
use wasm_bindgen::prelude::*;

pub type Result<T> = std::result::Result<T, JsValue>;

#[wasm_bindgen]
pub fn safe_operation(input: &str) -> Result<String> {
    if input.is_empty() {
        return Err(JsValue::from_str("输入不能为空"));
    }
    Ok(input.to_uppercase())
}
```

```toml

edition = "2021"

license = "MIT OR Apache-2.0"

[lib]

[dependencies.serde-wasm-bindgen]

[features]
```

## 七、条件编译

```rust
use wasm_bindgen::prelude::*;

// WASM 目标专用代码
#[cfg(target_arch = "wasm32")]
use web_sys::console;

#[cfg(target_arch = "wasm32")]
#[wasm_bindgen]
pub fn wasm_greet(name: &str) {
    console::log_1(&format!("Hello, {}!", name).into());
}

// 非 WASM 目标（测试、native）
#[cfg(not(target_arch = "wasm32"))]
pub fn wasm_greet(name: &str) {
    println!("Hello, {}!", name);
}

// 功能特性开关
#[cfg(feature = "with-serde")]
use serde::{Deserialize, Serialize};

#[cfg(feature = "with-serde")]
#[derive(Serialize, Deserialize)]
#[wasm_bindgen]
pub struct Config {
    pub width: u32,
    pub height: u32,
}
```

## 八、性能优化清单

```rust
// 1. 使用 #[inline] 优化小函数
#[inline]
fn fast_add(a: i32, b: i32) -> i32 {
    a + b
}

// 2. 使用 unchecked 索引（跳过边界检查）
#[inline(never)]
fn fast_sum(data: &[i32]) -> i32 {
    let mut sum = 0;
    for i in 0..data.len() {
        sum += unsafe { *data.get_unchecked(i) };
    }
    sum
}

// 3. 预分配 Vec 容量
fn build_vec(n: usize) -> Vec<i32> {
    let mut v = Vec::with_capacity(n);
    for i in 0..n {
        v.push(i as i32);
    }
    v
}

// 4. 避免不必要的 String 分配
fn process(data: &str) -> usize {
    data.len()  // 不需要 to_string()
}

// 5. 使用 Iterator 而非循环
fn sum_squares(data: &[i32]) -> i32 {
    data.iter().map(|x| x * x).sum()
}

// 6. 使用 wee_alloc 减小分配器开销
#[cfg(target_arch = "wasm32")]
#[global_allocator]
static ALLOC: wee_alloc::WeeAlloc = wee_alloc::WeeAlloc::INIT;
```

## 九、发布流程

```bash
#!/bin/bash
# release.sh

set -e

VERSION=$1
if [ -z "$VERSION" ]; then
  echo "Usage: ./release.sh <version>"
  exit 1
fi

echo "=== 发布版本 $VERSION ==="

# 1. 更新版本号
sed -i "s/version = \".*\"/version = \"$VERSION\"/" Cargo.toml
sed -i "s/\"version\": \".*\"/\"version\": \"$VERSION\"/" package.json

# 2. 运行测试
wasm-pack test --headless --chrome

# 3. 构建
wasm-pack build --release --target web
wasm-opt -Oz -o pkg/optimized.wasm pkg/my_lib_bg.wasm
mv pkg/optimized.wasm pkg/my_lib_bg.wasm

# 4. 检查大小
SIZE=$(wc -c < pkg/my_lib_bg.wasm)
echo "WASM size: ${SIZE} bytes"

# 5. Git 提交和标签
git add -A
git commit -m "release: v$VERSION"
git tag "v$VERSION"

# 6. 推送
git push && git push --tags

# 7. 发布到 npm
cd pkg && npm publish

echo "=== 发布完成 ==="
```
