### 2.3 错误处理

```rust
use wasm_bindgen::prelude::*;

pub type Result<T> = std::result::Result<T, JsValue>;

#[wasm_bindgen]
pub fn safe_operation(input: &str) -> Result<String> {
    if input.is_empty() {
        return Err(JsValue::from_str("输入不能为空"));
    }
    Ok(input.to_uppercase())
}
```

### 2.4 测试策略

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use wasm_bindgen_test::*;

    #[wasm_bindgen_test]
    fn test_safe_operation() {
        assert!(safe_operation("hello").is_ok());
        assert!(safe_operation("").is_err());
    }
}
```
