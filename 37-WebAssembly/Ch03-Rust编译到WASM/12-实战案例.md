# 实战案例

## 一、概念说明

通过实际案例学习 Rust WASM 开发。

```rust
// 图像处理案例
use wasm_bindgen::prelude::*;
use web_sys::{ImageData, CanvasRenderingContext2d};

#[wasm_bindgen]
pub fn grayscale(image_data: &ImageData) -> ImageData {
    let data = image_data.data();
    let mut result = data.clone();

    for i in (0..data.length()).step_by(4) {
        let r = data.get(i) as f64;
        let g = data.get(i + 1) as f64;
        let b = data.get(i + 2) as f64;

        let gray = (0.299 * r + 0.587 * g + 0.114 * b) as u8;

        result.set_index(i, gray);
        result.set_index(i + 1, gray);
        result.set_index(i + 2, gray);
    }

    ImageData::new_with_u8_clamped_array_and_sh(&result, image_data.width(), image_data.height())
        .unwrap()
}
```

## 二、具体用法

### 2.1 案例：JSON 处理

```rust
use wasm_bindgen::prelude::*;
use serde::{Serialize, Deserialize};

#[derive(Serialize, Deserialize)]
pub struct User {
    name: String,
    age: u32,
}

#[wasm_bindgen]
pub fn process_json(json: &str) -> Result<String, JsValue> {
    let user: User = serde_json::from_str(json)
        .map_err(|e| JsValue::from_str(&e.to_string()))?;

    let result = serde_json::to_string(&user)
        .map_err(|e| JsValue::from_str(&e.to_string()))?;

    Ok(result)
}
```

### 2.2 案例：加密

```rust
use wasm_bindgen::prelude::*;
use sha2::{Sha256, Digest};

#[wasm_bindgen]
pub fn sha256(data: &[u8]) -> Vec<u8> {
    let mut hasher = Sha256::new();
    hasher.update(data);
    hasher.finalize().to_vec()
}
```

### 2.3 案例：Canvas 绘图

```rust
use wasm_bindgen::prelude::*;
use web_sys::{CanvasRenderingContext2d, HtmlCanvasElement};

#[wasm_bindgen]
pub fn draw_pattern(canvas: &HtmlCanvasElement, width: u32, height: u32) {
    let context = canvas
        .get_context("2d")
        .unwrap()
        .unwrap()
        .dyn_into::<CanvasRenderingContext2d>()
        .unwrap();

    context.set_fill_style(&"#000000".into());
    context.fill_rect(0.0, 0.0, width as f64, height as f64);

    for i in 0..width {
        for j in 0..height {
            let r = (i as f64 / width as f64 * 255.0) as u8;
            let g = (j as f64 / height as f64 * 255.0) as u8;
            let color = format!("rgb({}, {}, 128)", r, g);
            context.set_fill_style(&color.into());
            context.fill_rect(i as f64, j as f64, 1.0, 1.0);
        }
    }
}
```
