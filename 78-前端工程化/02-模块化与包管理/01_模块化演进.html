模块化演进

核心概念：JavaScript模块化经历了从全局变量 → IIFE → CommonJS → AMD → UMD → ES Module的演进。ES Module（ESM）是ECMAScript官方标准，也是当前和未来的主流模块化方案。理解各模块规范的差异、循环依赖的处理方式以及Tree Shaking的前提条件，是掌握现代前端工程化的基础。

一、模块化演进历史

1.1 全局变量时代（远古）

// ❌ 所有代码共享全局作用域，命名冲突严重
var name = 'jQuery';
var version = '1.0';

// 另一个文件
var name = 'React';  // 覆盖了上面的name！
// 全局命名空间污染，无法管理依赖关系

1.2 IIFE（立即执行函数表达式）

// IIFE - 创建私有作用域
var MyModule = (function() {
    var privateVar = '私有变量';  // 外部无法访问

    function privateFn() {
        console.log('私有函数');
    }

    // 暴露公共接口
    return {
        publicVar: '公共变量',
        publicFn: function() {
            privateFn();
            return privateVar;
        }
    };
})();

MyModule.publicFn();  // ✅ 可以调用
MyModule.privateVar;  // undefined - 无法访问私有变量
  • 优点：避免全局污染，实现私有作用域
  • 缺点：没有统一的模块加载机制，依赖管理困难，需要手动维护script标签顺序

二、CommonJS（CJS）

2.1 基本语法

// math.js - 导出模块
const PI = 3.14159;

function add(a, b) {
    return a + b;
}

function subtract(a, b) {
    return a - b;
}

// 方式1：逐个导出
module.exports.add = add;
module.exports.subtract = subtract;

// 方式2：整体导出（覆盖整个module.exports）
module.exports = {
    add,
    subtract,
    PI
};

// app.js - 导入模块
const math = require('./math');
console.log(math.add(1, 2));

// 或解构导入
const { add, subtract } = require('./math');

2.2 核心特性

  • 运行时加载：require在运行时执行，可以动态加载模块
  • 值的拷贝：导出的是值的拷贝（不是引用），内部变化不影响外部
  • 同步加载：适合服务器端（文件在本地磁盘），不适合浏览器（网络加载慢）
  • 缓存机制：模块首次加载后缓存，后续require返回缓存结果
  • 循环依赖：返回已执行部分的结果（可能不完整）
// 循环依赖示例
// a.js
console.log('a开始');
exports.done = false;
const b = require('./b');
console.log('b在a中:', b.done);
exports.done = true;
console.log('a结束');

// b.js
console.log('b开始');
const a = require('./a');
console.log('a在b中:', a.done);  // false！因为a还没执行完
exports.done = true;
console.log('b结束');

// 执行a.js输出：
// a开始 → b开始 → a在b中: false → b结束 → b在a中: true → a结束

三、AMD（Asynchronous Module Definition）

3.1 基本语法

// AMD使用define函数定义模块
// 需要配合RequireJS等加载器使用

// math.js
define('math', [], function() {
    return {
        add: function(a, b) { return a + b; },
        PI: 3.14159
    };
});

// app.js - 依赖声明
define('app', ['math', 'jquery'], function(math, $) {
    // math和jquery已经加载并执行
    console.log(math.add(1, 2));
    $('body').addClass('loaded');

    return {
        init: function() { /* ... */ }
    };
});

// 动态加载
require(['math'], function(math) {
    console.log(math.PI);
});

3.2 特性

  • 异步加载：适合浏览器环境，依赖并行下载
  • 前置声明依赖：依赖在回调函数的参数中声明
  • 依赖前置：所有依赖必须在回调前声明（ESM的import也是如此）
  • 已过时：现代项目不再使用AMD，被ESM和打包工具取代

四、UMD（Universal Module Definition）

4.1 基本语法

// UMD - 同时兼容CommonJS、AMD和全局变量
(function(root, factory) {
    if (typeof define === 'function' && define.amd) {
        // AMD环境（RequireJS）
        define(['jquery'], factory);
    } else if (typeof module === 'object' && module.exports) {
        // CommonJS环境（Node.js）
        module.exports = factory(require('jquery'));
    } else {
        // 浏览器全局变量
        root.MyLib = factory(root.jQuery);
    }
}(typeof self !== 'undefined' ? self : this, function($) {
    // 模块的实际代码
    return {
        version: '1.0.0',
        init: function() {
            $('body').addClass('initialized');
        }
    };
}));

4.2 使用场景

  • 需要同时支持Node.js和浏览器的库
  • 需要兼容旧环境的第三方库
  • 大多数NPM包在ESM普及前都使用UMD格式
  • 现代项目通过构建工具自动处理，很少手写UMD

五、ES Module（ESM）

5.1 基本语法

// ===== 导出（Export）=====

// math.js - 命名导出
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export class Calculator {
    multiply(a, b) { return a * b; }
}

// utils.js - 默认导出
export default function formatDate(date) {
    return date.toISOString().split('T')[0];
}

// 也可以混合使用
export const helper = () => {};
export default function main() {}

// ===== 导入（Import）=====

// 命名导入
import { add, PI } from './math.js';

// 默认导入
import formatDate from './utils.js';

// 混合导入
import main, { helper } from './module.js';

// 整体导入（所有导出作为命名空间）
import * as math from './math.js';
math.add(1, 2);

// 动态导入（返回Promise）
const module = await import('./math.js');
module.add(1, 2);

// 仅执行副作用（如polyfill）
import './polyfills.js';

// 重命名
import { add as sum } from './math.js';
export { add as addition } from './math.js';

5.2 ESM核心特性

特性ESMCommonJS

5.3 ESM的Live Binding（实时绑定）

// counter.js
export let count = 0;

export function increment() {
    count++;  // ESM中导出的是引用，内部变化外部可见
}

// app.js
import { count, increment } from './counter.js';

console.log(count);  // 0
increment();
console.log(count);  // 1  ← Live Binding！

// 对比CommonJS：
// let count = 0;
// module.exports = { count, increment };
// 外部拿到的count是值的拷贝，increment后外部count仍是0

5.4 顶层await（ES2022）

// 顶层await - 模块加载时的异步操作
const response = await fetch('/api/config');
export const config = await response.json();

// 动态导入
const lodash = await import('https://cdn.example.com/lodash.js');

// 资源初始化
const db = await connectDatabase();
export default db;

// 注意：使用顶层await的模块会变成异步模块
// 导入它的模块也会等待它完成
// import './module-with-top-level-await.js';  // 这里会等待

六、循环依赖详解

6.1 CommonJS循环依赖

// a.js
exports.done = false;
const b = require('./b');  // 执行b.js
console.log('a中b.done:', b.done);  // true（b已执行完）
exports.done = true;

// b.js
const a = require('./a');  // 返回a的部分导出（只返回已执行的部分）
console.log('b中a.done:', a.done);  // false！a还没执行到exports.done = true
exports.done = true;

// CommonJS循环依赖返回部分结果

6.2 ESM循环依赖

// a.mjs
import { b } from './b.mjs';
export const a = 'a';
console.log('a中b:', b);  // 如果b.mjs还没完成解析，可能是undefined

// b.mjs
import { a } from './a.mjs';
export const b = 'b';
console.log('b中a:', a);  // 取决于解析顺序

// ESM循环依赖的值是引用（live binding）
// 解析阶段建立引用关系，执行阶段才赋值
// 如果模块还没执行到导出语句，导入方获取到的是undefined

6.3 如何避免循环依赖

  • 重构模块结构：提取公共依赖到第三个模块
  • 延迟导入：在函数内部使用 require() 或 import()
  • 依赖注入：通过参数传递依赖而非直接导入
  • 使用依赖检查工具：madge、dependency-cruiser 检测循环依赖
// 解决方案1：提取公共模块
// common.js
export const SHARED_CONSTANT = 'shared';

// a.js
import { SHARED_CONSTANT } from './common.js';
export const a = 'a';

// b.js
import { SHARED_CONSTANT } from './common.js';
export const b = 'b';

// 解决方案2：延迟导入
// a.js
export function getB() {
    const { b } = require('./b');  // 运行时加载
    return b;
}

// 解决方案3：动态import
// a.js
export const a = 'a';
export async function getB() {
    const { b } = await import('./b.js');
    return b;
}

七、Tree Shaking前提条件

7.1 为什么只有ESM支持Tree Shaking

ESM（编译时）： CommonJS（运行时）： import { add } from './math' const math = require('./math') ↑ ↑ 编译时确定依赖关系 运行时才能确定依赖关系 → 静态分析可能 → 无法静态分析 → Tree Shaking ✅ → Tree Shaking ❌ 关键区别： ESM: import的路径必须是字符串字面量 import { x } from './module.js' // ✅ 静态 CJS: require的路径可以是动态的 const name = getModuleName(); const mod = require(name); // ❌ 动态，无法分析

7.2 Tree Shaking的条件

  • 使用ESM语法：import/export 而非 require/module.exports
  • 构建工具支持：Webpack、Rollup、Vite等都支持Tree Shaking
  • 没有副作用：模块不产生副作用（修改全局状态等），或在 package.json 中声明 sideEffects
  • 不使用动态导入：避免 import(`./${name}.js`) 这种动态路径
  • 使用具名导入：import { Button } from 'antd' 比 import _ from 'lodash' 更容易被shaking

7.3 sideEffects字段

// package.json
{
    "name": "my-lib",
    "sideEffects": false  // 声明所有模块都没有副作用
}

// 或指定有副作用的文件
{
    "sideEffects": [
        "*.css",
        "*.scss",
        "./src/polyfills.js",
        "./src/register-global.js"
    ]
}

// 副作用是指：模块加载时执行了不影响导出的操作
// 例如：
// - 注册全局变量
// - 修改原型链
// - 执行console.log
// - 导入CSS（CSS是典型副作用）

// 没有副作用的模块可以安全地被Tree Shaking

八、现代模块化最佳实践

8.1 推荐做法

  • 始终使用ESM：新项目统一使用 import/export
  • 具名导出优先：默认导出在重构时容易出错（导入名随意）
  • 一个模块一个职责：保持模块单一职责，便于Tree Shaking
  • index.js统一导出：目录结构中用index.js统一导出子模块
  • 避免循环依赖：使用依赖检查工具
  • 正确声明sideEffects：让构建工具可以安全Tree Shaking
// 推荐：目录结构
components/
├── index.js          // 统一导出
├── Button/
│   ├── index.js
│   ├── Button.tsx
│   └── Button.module.css
└── Modal/
    ├── index.js
    ├── Modal.tsx
    └── Modal.module.css

// components/index.js
export { Button } from './Button';
export { Modal } from './Modal';

// 使用方
import { Button, Modal } from '@/components';
总结：JavaScript模块化从全局变量发展到ES Module，经历了IIFE（私有作用域）→ CommonJS（Node.js同步）→ AMD（浏览器异步）→ UMD（通用兼容）→ ESM（官方标准）的演进。ESM是当前和未来的主流方案，支持编译时静态分析、Tree Shaking、顶层await等现代特性。理解循环依赖的本质和Tree Shaking的前提条件是模块化编程的基础。
